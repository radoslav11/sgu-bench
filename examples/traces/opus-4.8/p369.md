Looking at this problem:

We have black cells. The rectangle rule means: if rows r1,r2 both have columns c1,c2 (3 of the 4 corners present), we can fill the 4th. 

This is the classic "closure" problem. Think of it as a bipartite graph between rows and columns, where a black cell (r,c) is an edge. The operation: if we have edges (r1,c1),(r1,c2),(r2,c1) we can add (r2,c2). 

The closure of this operation on a connected component of the bipartite graph results in a complete bipartite graph between all rows and all columns in that component. So for each connected component with `R` distinct rows and `C` distinct columns, the final count is `R*C`. Sum over all components.

Let me verify with examples:
- Example 1: cells (1,1),(1,2),(2,2). Rows {1,2}, cols {1,2}, all connected → 2*2=4. ✓
- Example 2: (0,0),(1,0),(0,1),(1,2),(2,1). Rows {0,1,2}, cols {0,1,2}. Connected component: row0-col0,row1-col0,row0-col1,row1-col2,row2-col1. All connected (one component): 3 rows, 3 cols → 9. ✓

Use DSU over rows and columns (coordinate compress, treat rows and columns as separate node sets).

```cpp
#include <bits/stdc++.h>
using namespace std;

int par[400005];
int find(int x){ while(par[x]!=x){ par[x]=par[par[x]]; x=par[x];} return x;}
void uni(int a,int b){ a=find(a); b=find(b); if(a!=b) par[a]=b;}

int main(){
    int k;
    scanf("%d",&k);
    vector<int> xs(k), ys(k);
    vector<int> rowvals, colvals;
    for(int i=0;i<k;i++){
        scanf("%d %d",&xs[i],&ys[i]);
        rowvals.push_back(xs[i]);
        colvals.push_back(ys[i]);
    }
    sort(rowvals.begin(), rowvals.end());
    rowvals.erase(unique(rowvals.begin(),rowvals.end()),rowvals.end());
    sort(colvals.begin(), colvals.end());
    colvals.erase(unique(colvals.begin(),colvals.end()),colvals.end());
    int R=rowvals.size(), C=colvals.size();
    int total=R+C;
    for(int i=0;i<total;i++) par[i]=i;
    for(int i=0;i<k;i++){
        int r=lower_bound(rowvals.begin(),rowvals.end(),xs[i])-rowvals.begin();
        int c=lower_bound(colvals.begin(),colvals.end(),ys[i])-colvals.begin();
        uni(r, R+c);
    }
    // for each component count rows and cols
    vector<long long> rcnt(total,0), ccnt(total,0);
    for(int i=0;i<R;i++) rcnt[find(i)]++;
    for(int i=0;i<C;i++) ccnt[find(R+i)]++;
    long long ans=0;
    for(int i=0;i<total;i++){
        ans += rcnt[i]*ccnt[i];
    }
    printf("%lld\n", ans);
    return 0;
}
```
