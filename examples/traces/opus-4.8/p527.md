This is a minimum vertex cover problem on a bipartite graph (König's theorem). Each rock at (i,j) is an edge between row i and column j. We need minimum vertex cover = maximum bipartite matching.

```cpp
#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<int> adj[30];
int matchR[30];
bool used[30];

bool tryK(int u){
    for(int v: adj[u]){
        if(!used[v]){
            used[v]=true;
            if(matchR[v]==-1 || tryK(matchR[v])){
                matchR[v]=u;
                return true;
            }
        }
    }
    return false;
}

int main(){
    cin>>n>>m;
    for(int i=0;i<n;i++){
        string s; cin>>s;
        for(int j=0;j<m;j++){
            if(s[j]=='*') adj[i].push_back(j);
        }
    }
    memset(matchR,-1,sizeof(matchR));
    int res=0;
    for(int i=0;i<n;i++){
        memset(used,0,sizeof(used));
        if(tryK(i)) res++;
    }
    cout<<res<<endl;
    return 0;
}
```
