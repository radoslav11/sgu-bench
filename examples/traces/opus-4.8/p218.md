Bottleneck assignment problem. Binary search on the threshold value, and check if a perfect matching exists using only edges with cost ≤ threshold.

```cpp
#include <bits/stdc++.h>
using namespace std;

int n;
int cost[500][500];
int matchL[500], matchR[500];
bool used[500];
int thr;

bool tryK(int u){
    for(int v=0;v<n;v++){
        if(cost[u][v]<=thr && !used[v]){
            used[v]=true;
            if(matchR[v]==-1 || tryK(matchR[v])){
                matchL[u]=v;
                matchR[v]=u;
                return true;
            }
        }
    }
    return false;
}

bool feasible(int t){
    thr=t;
    for(int i=0;i<n;i++){matchL[i]=-1;matchR[i]=-1;}
    int cnt=0;
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++) used[j]=false;
        if(tryK(i)) cnt++;
        else return false;
    }
    return cnt==n;
}

int main(){
    scanf("%d",&n);
    vector<int> vals;
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++){
            scanf("%d",&cost[i][j]);
            vals.push_back(cost[i][j]);
        }
    sort(vals.begin(),vals.end());
    vals.erase(unique(vals.begin(),vals.end()),vals.end());
    
    int lo=0, hi=(int)vals.size()-1, ans=hi;
    while(lo<=hi){
        int mid=(lo+hi)/2;
        if(feasible(vals[mid])){
            ans=mid;
            hi=mid-1;
        } else lo=mid+1;
    }
    // recompute matching for answer
    feasible(vals[ans]);
    printf("%d\n",vals[ans]);
    for(int i=0;i<n;i++){
        printf("%d %d\n",i+1,matchL[i]+1);
    }
    return 0;
}
```
