Looking at this problem:

- Read scores, for each faculty compute total grade based on K exams.
- For each specialty, if total grade >= required, accept.

Let me compute the faculty's total grade: for each of K exams, get subject score, convert to grade 2-5, sum.

Need careful reading with getline for names.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int N, M;
    // read N M
    {
        string line;
        getline(cin, line);
        stringstream ss(line);
        ss >> N >> M;
    }
    vector<int> B(M);
    {
        string line;
        getline(cin, line);
        stringstream ss(line);
        for(int i=0;i<M;i++) ss>>B[i];
    }
    
    vector<string> results;
    
    for(int f=0; f<N; f++){
        string facName;
        getline(cin, facName);
        // K and subjects
        int K;
        vector<int> subj;
        {
            string line;
            getline(cin, line);
            stringstream ss(line);
            ss >> K;
            subj.resize(K);
            for(int i=0;i<K;i++) ss>>subj[i];
        }
        // K triplets
        int totalGrade = 0;
        {
            string line;
            getline(cin, line);
            stringstream ss(line);
            for(int i=0;i<K;i++){
                int X,Y,Z;
                ss>>X>>Y>>Z;
                int T = B[subj[i]-1];
                int g;
                if(T < X) g=2;
                else if(T < Y) g=3;
                else if(T < Z) g=4;
                else g=5;
                totalGrade += g;
            }
        }
        int S;
        {
            string line;
            getline(cin, line);
            stringstream ss(line);
            ss >> S;
        }
        for(int s=0;s<S;s++){
            string specName;
            getline(cin, specName);
            int req;
            {
                string line;
                getline(cin, line);
                stringstream ss(line);
                ss >> req;
            }
            if(totalGrade >= req){
                results.push_back(facName + " " + specName);
            }
        }
    }
    
    if(results.empty()){
        cout << "Army" << "\n";
    } else {
        for(auto &r : results) cout << r << "\n";
    }
    return 0;
}
```
