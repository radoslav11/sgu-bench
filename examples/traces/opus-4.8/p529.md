Looking at this problem, it's a **dynamic MST** with single-edge weight updates, solvable offline via divide-and-conquer on time, using contraction (edges always in MST for this segment) and reduction (edges never in MST for this segment) to shrink the graph to O(#updates) per node.

Key ideas:
- Edges modified in `[l,r]` are "special". With specials at −∞, non-special MST edges are *forced* (contract them). With specials at +∞, non-special edges not in MST can be *removed*.
- This bounds remaining edges to O(#special), giving O((m + q log q) log) overall.

#include <bits/stdc++.h>
using namespace std;

int n,m,t;
vector<int> curW, qe, qc, markT;
int stampCnt=0;
vector<long long> ans;
struct E{int u,v,id;};

struct DSU{
    vector<int> p;
    void init(int s){p.resize(s); for(int i=0;i<s;i++)p[i]=i;}
    int f(int x){while(p[x]!=x){p[x]=p[p[x]];x=p[x];}return x;}
    bool u(int a,int b){a=f(a);b=f(b);if(a==b)return false;p[a]=b;return true;}
};

long long mstWeight(const vector<E>& ne,int V){
    if(ne.empty())return 0;
    vector<int> ord(ne.size());
    for(int i=0;i<(int)ne.size();i++)ord[i]=i;
    sort(ord.begin(),ord.end(),[&](int a,int b){return curW[ne[a].id]<curW[ne[b].id];});
    DSU d; d.init(V);
    long long s=0;
    for(int i:ord) if(d.u(ne[i].u,ne[i].v)) s+=curW[ne[i].id];
    return s;
}

void solve(int l,int r,const vector<E>& edges,int V,long long base){
    int Ec=edges.size();
    stampCnt++;
    for(int i=l;i<=r;i++) markT[qe[i]]=stampCnt;
    vector<char> isSp(Ec);
    for(int i=0;i<Ec;i++) isSp[i]=(markT[edges[i].id]==stampCnt);
    vector<int> order(Ec);
    for(int i=0;i<Ec;i++)order[i]=i;
    sort(order.begin(),order.end(),[&](int a,int b){
        if(isSp[a]!=isSp[b]) return isSp[a]>isSp[b];
        return curW[edges[a].id]<curW[edges[b].id];
    });
    // contraction detection
    DSU kr; kr.init(V);
    vector<char> forced(Ec,0);
    for(int idx:order){
        int u=edges[idx].u,v=edges[idx].v;
        if(isSp[idx]) kr.u(u,v);
        else { if(kr.u(u,v)) forced[idx]=1; }
    }
    DSU cont; cont.init(V);
    for(int idx=0;idx<Ec;idx++) if(forced[idx]){ cont.u(edges[idx].u,edges[idx].v); base+=curW[edges[idx].id]; }
    // reduction
    DSU red; red.init(V);
    for(int idx=0;idx<Ec;idx++) if(forced[idx]) red.u(edges[idx].u,edges[idx].v);
    vector<char> keep(Ec,0);
    for(int idx:order){
        if(isSp[idx]||forced[idx]) continue;
        if(red.u(edges[idx].u,edges[idx].v)) keep[idx]=1;
    }
    vector<int> comp(V,-1); int cnt=0;
    vector<E> ne; ne.reserve(Ec);
    for(int idx=0;idx<Ec;idx++){
        if(forced[idx]) continue;
        if(!(isSp[idx]||keep[idx])) continue;
        int su=cont.f(edges[idx].u), sv=cont.f(edges[idx].v);
        if(su==sv) continue;
        if(comp[su]<0) comp[su]=cnt++;
        if(comp[sv]<0) comp[sv]=cnt++;
        ne.push_back({comp[su],comp[sv],edges[idx].id});
    }
    int Vp=cnt;
    if(l==r){
        curW[qe[l]]=qc[l];
        ans[l]=base+mstWeight(ne,Vp);
        return;
    }
    int mid=(l+r)/2;
    solve(l,mid,ne,Vp,base);
    solve(mid+1,r,ne,Vp,base);
}
