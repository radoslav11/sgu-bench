Topological sort assigning coin values.

```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,m;
    scanf("%d %d",&n,&m);
    vector<vector<int>> adj(n+1);
    vector<int> indeg(n+1,0);
    set<pair<int,int>> seen;
    for(int i=0;i<m;i++){
        int p,q;
        scanf("%d %d",&p,&q);
        // p lighter than q: p < q in value, edge p->q
        if(seen.count({p,q})) continue;
        seen.insert({p,q});
        adj[p].push_back(q);
        indeg[q]++;
    }
    // topological sort, assign smallest values to nodes with indeg 0 first
    priority_queue<int,vector<int>,greater<int>> pq;
    for(int i=1;i<=n;i++) if(indeg[i]==0) pq.push(i);
    vector<int> val(n+1,0);
    int cur=1;
    while(!pq.empty()){
        int u=pq.top();pq.pop();
        val[u]=cur++;
        for(int v:adj[u]){
            if(--indeg[v]==0) pq.push(v);
        }
    }
    if(cur!=n+1){
        printf("No solution\n");
        return 0;
    }
    for(int i=1;i<=n;i++){
        printf("%d%c",val[i],i==n?'\n':' ');
    }
    return 0;
}
```
