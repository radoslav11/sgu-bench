```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    string n; int k;
    cin>>n>>k;
    int keep = n.size()-k;
    string res;
    int toRemove = k;
    for(char c: n){
        while(!res.empty() && toRemove>0 && res.back()<c){
            res.pop_back();
            toRemove--;
        }
        res.push_back(c);
    }
    res.resize(keep);
    cout<<res<<endl;
}
```
