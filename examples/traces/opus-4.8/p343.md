Looking at this problem:

- N assets, T+1 days of prices.
- Return of asset i on day t: (price[t-1][i] - price[t][i]) / price[t][i], for t=1..T (T returns).
- Compute mean return per asset, covariance matrix, weights, portfolio mean, portfolio std, then VaR.

Let me work through:

- V_i = quantity_i * price[0][i]
- V_P = sum V_i
- w_i = V_i / V_P
- m_i = mean of returns of asset i
- m_P = sum w_i * m_i
- S[i][j] = cov = E((r_i - m_i)(r_j - m_j)), averaged over T samples (population covariance, divide by T)
- s_P = sqrt(W^T S W)
- VaR = -V_P * (m_P - 1.644854 * s_P)

Let me code this.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int T, N;
    scanf("%d %d", &T, &N);
    vector<double> qty(N);
    for(int i=0;i<N;i++) scanf("%lf",&qty[i]);
    
    // T+1 lines of prices
    vector<vector<double>> price(T+1, vector<double>(N));
    for(int t=0;t<=T;t++)
        for(int i=0;i<N;i++)
            scanf("%lf",&price[t][i]);
    
    // returns: for t=1..T, ret[t-1][i] = (price[t-1][i]-price[t][i])/price[t][i]
    vector<vector<double>> ret(T, vector<double>(N));
    for(int t=1;t<=T;t++)
        for(int i=0;i<N;i++)
            ret[t-1][i] = (price[t-1][i]-price[t][i])/price[t][i];
    
    // mean returns
    vector<double> m(N,0.0);
    for(int i=0;i<N;i++){
        double s=0;
        for(int t=0;t<T;t++) s+=ret[t][i];
        m[i]=s/T;
    }
    
    // covariance matrix (divide by T)
    vector<vector<double>> S(N, vector<double>(N,0.0));
    for(int i=0;i<N;i++)
        for(int j=0;j<N;j++){
            double s=0;
            for(int t=0;t<T;t++)
                s += (ret[t][i]-m[i])*(ret[t][j]-m[j]);
            S[i][j]=s/T;
        }
    
    // values and weights
    vector<double> V(N);
    double VP=0;
    for(int i=0;i<N;i++){ V[i]=qty[i]*price[0][i]; VP+=V[i]; }
    vector<double> w(N);
    for(int i=0;i<N;i++) w[i]=V[i]/VP;
    
    // mP
    double mP=0;
    for(int i=0;i<N;i++) mP += w[i]*m[i];
    
    // sP = sqrt(W^T S W)
    double var=0;
    for(int i=0;i<N;i++)
        for(int j=0;j<N;j++)
            var += w[i]*S[i][j]*w[j];
    double sP = sqrt(var);
    
    double VaR = -VP*(mP - 1.644854*sP);
    
    printf("%.2f\n", VaR);
    return 0;
}
```

Let me verify with the example. With the given data it should produce 150.89. The formula and covariance approach (dividing by T) are standard for this SGU problem.
