Actually the rule: the winner of game i plays game i+1. So there's a chain — one player carries over. We need to assign remaining game counts.

#include <bits/stdc++.h>
using namespace std;
int main(){
    int n; cin>>n;
    vector<int> rem(n);
    int total=0;
    for(int i=0;i<n;i++){cin>>rem[i]; total+=rem[i];}
    int G=total/2;
    cout<<G<<"\n";
    auto pickOther=[&](int cur)->int{
        int best=-1;
        for(int i=0;i<n;i++) if(i!=cur && rem[i]>0){
            if(best==-1||rem[i]>rem[best]) best=i;
        }
        return best;
    };
    // first game: pick two largest

    for(int i=0;i<n;i++) if(rem[i]>0){
        if(a==-1||rem[i]>rem[a]){b=a;a=i;}

    int cur;
