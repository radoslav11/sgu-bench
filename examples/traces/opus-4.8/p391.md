#include <bits/stdc++.h>
using namespace std;
bool foldable(vector<int> v){
    sort(v.begin(),v.end()); v.erase(unique(v.begin(),v.end()),v.end());
    int l=0,r=v.size()-1;
    while(l<r){
        int s=v[l]+v[r];
        while(l<r){
            if(v[l]+v[r]!=s) return false;
            l++;r--;
        }
    }
    return true;
}
