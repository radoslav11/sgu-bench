For each number, check if number^M is divisible by K.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int N;
    long long M, K;
    cin >> N >> M >> K;
    int count = 0;
    for(int i=0;i<N;i++){
        long long a;
        cin >> a;
        // compute a^M mod K
        long long base = a % K;
        long long result = 1 % K;
        long long e = M;
        while(e>0){
            if(e&1) result = (result*base)%K;
            base = (base*base)%K;
            e>>=1;
        }
        if(result==0) count++;
    }
    cout << count << endl;
    return 0;
}
```
