#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,m;cin>>n>>m;
    vector<int> hp(n+1,100), sc(n+1,0);
    for(int i=0;i<m;i++){
        int a,b;cin>>a>>b;
        if(hp[b]>0) sc[a]+=3;
        hp[b]-=8;
    }
    for(int i=1;i<=n;i++){
        if(hp[i]>0) sc[i]+=hp[i]/2;
        cout<<hp[i]<<" "<<sc[i]<<"\n";
    }
}
