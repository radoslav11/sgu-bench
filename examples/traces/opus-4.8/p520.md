using namespace std;

int n, m;
vector<int> adj[1005];
int dist_[1005];
// memo[u][d], d from 1..n+1
vector<vector<int>> memo;

int win(int u, int d) {
    if (d > n + 1) return 0; // no safe move possible
    int &r = memo[u][d];
    if (r != -1) return r;
    r = 0; // assume lose
    for (int w : adj[u]) {
        if (dist_[w] >= d) {
            // safe move
            if (win(w, d + 1) == 0) {
                r = 1;
                break;
            }
        }
    }
    return r;
}
