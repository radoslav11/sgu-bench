Looking at this problem: maximize cost/time ratio over a cycle. This is the maximum mean cycle (ratio) problem.

Binary search on ratio λ. For a given λ, weight of edge = C - λ*T. We want to check if there's a cycle with positive total weight (since for the cycle (sumC)/(sumT) > λ ⟺ sumC - λ*sumT > 0). Find positive cycle using Bellman-Ford style detection, then recover it.

Let me binary search and at the end recover the cycle.

```cpp
#include <bits/stdc++.h>
using namespace std;

int N, M;
struct Edge { int a, b, c, t; };
vector<Edge> edges;

// returns a cycle (list of nodes) if there's a cycle with positive total weight (C - lambda*T)
// otherwise empty
vector<int> findPosCycle(double lambda) {
    vector<double> dist(N+1, 0.0);
    vector<int> par(N+1, -1);
    vector<int> parEdge(N+1, -1);
    int x = -1;
    for (int iter = 0; iter < N; iter++) {
        x = -1;
        for (int e = 0; e < M; e++) {
            int a = edges[e].a, b = edges[e].b;
            double w = edges[e].c - lambda * edges[e].t;
            if (dist[a] + w > dist[b] + 1e-9) {
                dist[b] = dist[a] + w;
                par[b] = a;
                parEdge[b] = e;
                x = b;
            }
        }
        if (x == -1) break;
    }
    if (x == -1) return {};
    // x is on or reachable from a positive cycle; go back N steps to ensure inside cycle
    for (int i = 0; i < N; i++) x = par[x];
    // recover cycle
    vector<int> cyc;
    int cur = x;
    while (true) {
        cyc.push_back(cur);
        cur = par[cur];
        if (cur == x && cyc.size() > 1) break;
        if ((int)cyc.size() > N + 2) break;
    }
    reverse(cyc.begin(), cyc.end());
    return cyc;
}

int main(){
    scanf("%d %d", &N, &M);
    for (int i = 0; i < M; i++) {
        int a,b,c,t;
        scanf("%d %d %d %d", &a,&b,&c,&t);
        edges.push_back({a,b,c,t});
    }
    
    // binary search on lambda
    double lo = 0.0, hi = 101.0; // max ratio < 100
    // first check if any positive cycle exists at lambda=0 (always, since costs positive and cycle exists)
    // Actually need to check feasibility: is there any cycle at all?
    if (findPosCycle(0.0).empty()) {
        // no cycle
        printf("0\n");
        return 0;
    }
    
    for (int it = 0; it < 100; it++) {
        double mid = (lo + hi) / 2;
        if (!findPosCycle(mid).empty()) lo = mid;
        else hi = mid;
    }
    
    vector<int> cyc = findPosCycle(lo - 1e-6);
    if (cyc.empty()) {
        // try a bit lower
        for (double d = 1e-4; d < 1.0 && cyc.empty(); d *= 2) {
            cyc = findPosCycle(lo - d);
        }
    }
    
    if (cyc.empty()) {
        printf("0\n");
        return 0;
    }
    
    printf("%d\n", (int)cyc.size());
    for (int i = 0; i < (int)cyc.size(); i++) {
        printf("%d%c", cyc[i], i+1<(int)cyc.size()?' ':'\n');
    }
    return 0;
}
```
