Looking at this problem:

- Maintain ice-cream stock sorted by price (a map from price -> count).
- For BUY n t: take the n cheapest, compute total cost. If cost <= t, sell them (remove from stock), output HAPPY, else UNHAPPY.

Use a std::map<price, count>. For buying, iterate from cheapest. Since total queries ≤ 1e5 and we remove items as we sell, the amortized cost is fine — each price entry is created once and removed once.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    map<long long,long long> stock; // price -> count
    
    string cmd;
    string output;
    
    while(cin >> cmd){
        if(cmd == "ARRIVE"){
            long long n,c;
            cin >> n >> c;
            stock[c] += n;
        } else { // BUY
            long long n,t;
            cin >> n >> t;
            // determine cheapest n pieces cost
            long long need = n;
            long long cost = 0;
            bool feasible = true;
            // iterate
            auto it = stock.begin();
            // First check feasibility without modifying
            // We'll accumulate; if cost exceeds t or not enough pieces, fail
            long long remaining = n;
            for(; it != stock.end() && remaining > 0; ++it){
                long long take = min(remaining, it->second);
                cost += take * it->first;
                remaining -= take;
                if(cost > t){ feasible = false; break; }
            }
            if(remaining > 0) feasible = false;
            
            if(feasible){
                output += "HAPPY\n";
                // remove pieces
                long long rem = n;
                while(rem > 0 && !stock.empty()){
                    auto b = stock.begin();
                    if(b->second <= rem){
                        rem -= b->second;
                        stock.erase(b);
                    } else {
                        b->second -= rem;
                        rem = 0;
                    }
                }
            } else {
                output += "UNHAPPY\n";
            }
        }
    }
    
    cout << output;
    return 0;
}
```
