#include <bits/stdc++.h>
using namespace std;

bool isPrime(long long n){
    if(n<2) return false;
    for(long long i=2;i*i<=n;i++) if(n%i==0) return false;
    return true;
}

int main(){
    int n; cin>>n;
    while(n--){
        long long x; cin>>x;
        bool found=false;
        for(long long i=2;i*i<=x;i++){
            if(x%i==0){
                if(isPrime(i)&&isPrime(x/i)) found=true;
                break;
            }
        }
        cout<<(found?"Yes":"No")<<"\n";
    }
}
