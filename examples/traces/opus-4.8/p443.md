#include <bits/stdc++.h>
using namespace std;

int key(int n){
    vector<int> p;
    for(int i=2;(long long)i*i<=n;i++){
        if(n%i==0){
            p.push_back(i);
            while(n%i==0)n/=i;
        }
    }
    if(n>1)p.push_back(n);
    int mx=*max_element(p.begin(),p.end());
    int sum=0;
    for(int x:p)sum+=x;
    return mx-(sum-mx);
}

int main(){
    int a,b;cin>>a>>b;
    cout<<(key(a)>key(b)?"a":"b")<<endl;
}
