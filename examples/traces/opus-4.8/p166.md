            string &L = lines[row];
            while((int)L.size() < col) L.push_back(' ');
            L.insert(L.begin()+col, c);

        } else if(c=='L'){
            if(col>0) col--;
        } else if(c=='R'){
            col++;
            // if next command is letter/space, additional spaces added when inserted
            // handled by insert padding
        } else if(c=='U'){
            if(row>0){
                row--;
            }
        } else if(c=='D'){
            if(row < (int)lines.size()-1){
                row++;
            }
        } else if(c=='N'){
            string &L = lines[row];
            string tail;
            if(col < (int)L.size()){
                tail = L.substr(col);
                L = L.substr(0,col);

            col = 0;

                if(row < (int)lines.size()-1){
                    // pad current line with spaces to col, then attach next line
                    while((int)L.size() < col) L.push_back(' ');
                    L += lines[row+1];
                    lines.erase(lines.begin()+row+1);
                }
                // else after end of last line: nothing

            string &L = lines[row];
            if(col > (int)L.size()){

                // col==0, first symbol of line

                    lines[row-1] += L;
                    lines.erase(lines.begin()+row);
                    row--;
                    col = prevLen;
                }
                // else first symbol of first line: nothing
            }
        } else if(c=='Q'){
            break;
