Each edge contributes: (A^T A) has entries summing. Sum = sum over edges of degree contributions. Actually (A^T A)_{jk} = sum_i a_ij a_ik = number of shared vertices between edge j and k. Sum over all j,k = sum_i (sum_j a_ij)^2 = sum_i deg(i)^2.

```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,m;
    scanf("%d %d",&n,&m);
    vector<long long> deg(n+1,0);
    for(int i=0;i<m;i++){
        int a,b;scanf("%d %d",&a,&b);
        deg[a]++;deg[b]++;
    }
    long long s=0;
    for(int i=1;i<=n;i++)s+=deg[i]*deg[i];
    printf("%lld\n",s);
}
```
