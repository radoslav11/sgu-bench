So **height = min(up_run, down_run)** in steps, where up_run is strictly increasing, down_run strictly decreasing.

```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    int T; scanf("%d",&T);
    while(T--){
        int n; scanf("%d",&n);
        vector<long long> a(n);
        for(auto&x:a) scanf("%lld",&x);
        // up[i]: length of strictly increasing run ending at i (in steps)
        // down[i]: strictly increasing run starting... 
        vector<int> upL(n,0), downR(n,0);
        for(int i=1;i<n;i++) if(a[i]>a[i-1]) upL[i]=upL[i-1]+1;
        for(int i=n-2;i>=0;i--) if(a[i]>a[i+1]) downR[i]=downR[i+1]+1;
        int hill=0, dale=0;
        // hill: peak at i, up steps from left = upL[i], down steps to right = downR[i]
        for(int i=0;i<n;i++){
            int up=upL[i], down=downR[i];
            if(up>0&&down>0) hill=max(hill,min(up,down));
        }
        // dale: valley at i, down on left, up on right
        vector<int> downL(n,0), upR(n,0);
        for(int i=1;i<n;i++) if(a[i]<a[i-1]) downL[i]=downL[i-1]+1;
        for(int i=n-2;i>=0;i--) if(a[i]<a[i+1]) upR[i]=upR[i+1]+1;
        for(int i=0;i<n;i++){
            int d=downL[i], u=upR[i];
            if(d>0&&u>0) dale=max(dale,min(d,u));
        }
        printf("%d %d\n",hill,dale);
    }
}
```
