Color = number of prime factors (with multiplicity) + 1.

```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    int n; cin>>n;
    vector<int> col(n+1,0);
    int M=0;
    for(int i=1;i<=n;i++){
        int x=i,cnt=0;
        for(int p=2;(long long)p*p<=x;p++){
            while(x%p==0){x/=p;cnt++;}
        }
        if(x>1)cnt++;
        col[i]=cnt+1;
        M=max(M,col[i]);
    }
    cout<<M<<"\n";
    for(int i=1;i<=n;i++)cout<<col[i]<<(i<n?' ':'\n');
    return 0;
}
```
