Looking at this problem:

- The polygon is convex (segment between any two interior points stays inside).
- Vertices given in random order, so I need to sort them to form a convex polygon (convex hull).
- For each segment, find the portion inside the polygon (excluding boundary), compute its length.

Let me clip each segment against the convex polygon (intersection of half-planes), then compute the clipped length.

Plan: Build convex hull from points. For each segment, clip parametrically against all edges (treating polygon as intersection of half-planes). Since boundary isn't included but length on boundary is zero anyway, we don't need special handling.

For clipping a segment against convex polygon: parametrize P = A + t*(B-A), t in [0,1]. For each edge with inward normal, keep t where point is on inside. Intersect all constraints.

Let me set up the polygon CCW so interior is to the left of each edge.

For edge from V[i] to V[i+1], inward side: cross(V[i+1]-V[i], P-V[i]) >= 0 (for CCW).

For each constraint: f(t) = cross(edge, A-V[i]) + t*cross(edge, B-A) >= 0. Linear in t. Update [tlo, thi].

```cpp
#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef double ld;

struct P { ld x, y; };

int main(){
    int n;
    scanf("%d",&n);
    vector<P> pts(n);
    for(int i=0;i<n;i++) scanf("%lf %lf",&pts[i].x,&pts[i].y);
    
    // convex hull
    sort(pts.begin(),pts.end(),[](const P&a,const P&b){
        return a.x<b.x || (a.x==b.x && a.y<b.y);
    });
    auto cross=[](const P&O,const P&A,const P&B){
        return (A.x-O.x)*(B.y-O.y)-(A.y-O.y)*(B.x-O.x);
    };
    int N=n;
    vector<P> hull(2*N);
    int k=0;
    for(int i=0;i<N;i++){
        while(k>=2 && cross(hull[k-2],hull[k-1],pts[i])<=0) k--;
        hull[k++]=pts[i];
    }
    int lower=k+1;
    for(int i=N-2;i>=0;i--){
        while(k>=lower && cross(hull[k-2],hull[k-1],pts[i])<=0) k--;
        hull[k++]=pts[i];
    }
    hull.resize(k-1);
    // hull is CCW
    int H=hull.size();
    
    int m;
    scanf("%d",&m);
    while(m--){
        ld ax,ay,bx,by;
        scanf("%lf %lf %lf %lf",&ax,&ay,&bx,&by);
        ld dx=bx-ax, dy=by-ay;
        ld tlo=0, thi=1;
        bool valid=true;
        for(int i=0;i<H && valid;i++){
            P V=hull[i], W=hull[(i+1)%H];
            ld ex=W.x-V.x, ey=W.y-V.y;
            // inward: cross(edge, P-V) >= 0
            // cross = ex*(Py-Vy) - ey*(Px-Vx)
            // = ex*(ay+t*dy - Vy) - ey*(ax+t*dx - Vx)
            ld c0 = ex*(ay-V.y) - ey*(ax-V.x);
            ld c1 = ex*dy - ey*dx;
            // constraint: c0 + t*c1 >= 0
            if(fabs(c1)<1e-12){
                if(c0 < 0) valid=false;
            } else {
                ld tb = -c0/c1;
                if(c1>0){
                    // t >= tb
                    if(tb>tlo) tlo=tb;
                } else {
                    // t <= tb
                    if(tb<thi) thi=tb;
                }
            }
        }
        ld len=0;
        if(valid && thi>tlo){
            ld seglen=sqrt(dx*dx+dy*dy)*(thi-tlo);
            len=seglen;
        }
        printf("%.2f\n",len);
    }
    return 0;
}
```
