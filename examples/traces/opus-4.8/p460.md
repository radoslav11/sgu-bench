#include <bits/stdc++.h>
using namespace std;
int main(){
    int n; cin>>n;
    while(n--){
        string s; cin>>s;
        int len=s.size();
        string last2 = len>=2 ? s.substr(len-2) : "";
        char last = s.back();
        if(last2=="ch"||last=='x'||last=='s'||last=='o'){
            s+="es";
        } else if(last=='f'){
            s=s.substr(0,len-1)+"ves";
        } else if(last2=="fe"){
            s=s.substr(0,len-2)+"ves";
        } else if(last=='y'){
            s=s.substr(0,len-1)+"ies";
        } else {
            s+="s";
        }
        cout<<s<<"\n";
    }
}
