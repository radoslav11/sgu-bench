#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,m,is,js;
    cin>>n>>m>>is>>js;
    vector<vector<bool>> dead(n+2, vector<bool>(m+2,false));
    int i=is, j=js;
    string res="";
    // directions: D(i+1,j), L(i,j-1), U(i-1,j), R(i,j+1)
    int di[4]={1,0,-1,0};
    int dj[4]={0,-1,0,1};
    char ch[4]={'D','L','U','R'};
    while(true){
        dead[i][j]=true;
        int found=-1;
        for(int k=0;k<4;k++){
            int ni=i+di[k], nj=j+dj[k];
            if(ni>=1&&ni<=n&&nj>=1&&nj<=m&&!dead[ni][nj]){
                found=k;break;
            }
        }
        if(found==-1) break;
        res+=ch[found];
        i+=di[found]; j+=dj[found];
    }
    cout<<res<<endl;
}
