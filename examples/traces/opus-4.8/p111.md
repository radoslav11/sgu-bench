#include <bits/stdc++.h>
using namespace std;

// Big integer as vector of digits (base 10), least significant first
typedef vector<int> Big;

Big fromString(const string& s){
    Big b;
    for(int i=s.size()-1;i>=0;i--) b.push_back(s[i]-'0');
    return b;
}

void trim(Big& a){
    while(a.size()>1 && a.back()==0) a.pop_back();
}

// compare a and b, return -1,0,1
int cmp(const Big& a, const Big& b){
    if(a.size()!=b.size()) return a.size()<b.size()?-1:1;
    for(int i=a.size()-1;i>=0;i--){
        if(a[i]!=b[i]) return a[i]<b[i]?-1:1;
    }
    return 0;
}

Big add(const Big& a, const Big& b){
    Big c;
    int carry=0;
    for(size_t i=0;i<max(a.size(),b.size())||carry;i++){
        int s=carry;
        if(i<a.size()) s+=a[i];
        if(i<b.size()) s+=b[i];
        c.push_back(s%10);
        carry=s/10;
    }
    return c;
}

Big sub(const Big& a, const Big& b){ // a>=b
    Big c;
    int borrow=0;
    for(size_t i=0;i<a.size();i++){
        int s=a[i]-borrow-(i<b.size()?b[i]:0);
        if(s<0){s+=10;borrow=1;} else borrow=0;
        c.push_back(s);
    }
    trim(c);
    return c;
}

Big mul(const Big& a, const Big& b){
    vector<int> res(a.size()+b.size(),0);
    for(size_t i=0;i<a.size();i++){
        int carry=0;
        for(size_t j=0;j<b.size()||carry;j++){
            long long cur=res[i+j]+carry+(long long)a[i]*(j<b.size()?b[j]:0);
            res[i+j]=cur%10;
            carry=cur/10;
        }
    }
    Big c(res.begin(),res.end());
    trim(c);
    return c;
}

// divide by 2
Big div2(const Big& a){
    Big c(a.size());
    int rem=0;
    for(int i=a.size()-1;i>=0;i--){
        int cur=rem*10+a[i];
        c[i]=cur/2;
        rem=cur%2;
    }
    trim(c);
    return c;
}

string toString(const Big& a){
    string s;
    for(int i=a.size()-1;i>=0;i--) s+=('0'+a[i]);
    return s;
}

int main(){
    string s;
    cin>>s;
    Big X=fromString(s);
    Big lo=fromString("0");
    // hi = 10^(ceil(len/2))
    Big hi=fromString("1");
    int len=s.size();
    int digits=(len+1)/2+1;
    for(int i=0;i<digits;i++) hi=mul(hi,fromString("10"));
    
    Big one=fromString("1");
    Big two=fromString("2");
    // binary search largest m with m*m<=X
    while(cmp(lo,hi)<0){
        // mid = (lo+hi+1)/2
        Big mid=div2(add(add(lo,hi),one));
        Big sq=mul(mid,mid);
        if(cmp(sq,X)<=0){
            lo=mid;
        } else {
            hi=sub(mid,one);
        }
    }
    cout<<toString(lo)<<endl;
    return 0;
}
