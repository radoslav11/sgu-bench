Looking at this problem:

We need to count integer solutions (x,y) to ax+by+c=0 with x1≤x≤x2, y1≤y≤y2.

Cases:
- a=0, b=0: if c=0, all pairs in range count; else 0.
- a=0, b≠0: need by+c=0 → y=-c/b, must be integer and in range; then x ranges freely → count = (x2-x1+1) if valid.
- a≠0, b=0: symmetric.
- a≠0, b≠0: use extended gcd to solve ax+by=-c.

Let me use extended Euclidean. ax+by = -c. Let g=gcd(a,b). Need g | c. General solution: x = x0 + (b/g)*t, y = y0 - (a/g)*t. Find range of t satisfying both x and y constraints, count integers.

Need careful interval intersection with possibly negative steps.

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll extgcd(ll a, ll b, ll &x, ll &y){
    if(b==0){ x=1; y=0; return a; }
    ll x1,y1; ll g=extgcd(b,a%b,x1,y1);
    x=y1; y=x1-(a/b)*y1; return g;
}

// floor and ceil division for ll
ll fdiv(ll a, ll b){ // floor a/b, b>0
    ll q=a/b, r=a%b;
    if(r!=0 && (r<0)) q--;
    return q;
}
ll cdiv(ll a, ll b){ // ceil a/b, b>0
    ll q=a/b, r=a%b;
    if(r!=0 && (r>0)) q++;
    return q;
}

int main(){
    ll a,b,c,x1,x2,y1,y2;
    cin>>a>>b>>c>>x1>>x2>>y1>>y2;
    if(x1>x2 || y1>y2){ cout<<0<<endl; return 0; }
    
    if(a==0 && b==0){
        if(c==0){
            // all pairs
            ll cnt=(x2-x1+1)*(y2-y1+1);
            cout<<cnt<<endl;
        } else cout<<0<<endl;
        return 0;
    }
    if(a==0){
        // by+c=0 -> y=-c/b
        if((-c)%b!=0){ cout<<0<<endl; return 0; }
        ll y=(-c)/b;
        if(y>=y1 && y<=y2) cout<<(x2-x1+1)<<endl;
        else cout<<0<<endl;
        return 0;
    }
    if(b==0){
        if((-c)%a!=0){ cout<<0<<endl; return 0; }
        ll x=(-c)/a;
        if(x>=x1 && x<=x2) cout<<(y2-y1+1)<<endl;
        else cout<<0<<endl;
        return 0;
    }
    
    // ax+by = -c
    ll rhs=-c;
    ll x0,y0;
    ll g=extgcd(a,b,x0,y0);
    if(rhs%g!=0){ cout<<0<<endl; return 0; }
    // particular solution
    ll mul=rhs/g;
    // x0*mul, y0*mul might overflow? a,b,c <=1e8. x0,y0 from extgcd bounded by b,a roughly ~1e8. mul up to ~1e8/1 = could be up to 2e8. product ~ 2e16 fits in ll.
    ll px=x0*mul;
    ll py=y0*mul;
    ll dx=b/g; // x = px + dx*t
    ll dy=-a/g; // y = py + dy*t
    
    // find t range for x1<=px+dx*t<=x2
    // and y1<=py+dy*t<=y2
    // intersect
    ll tlo=LLONG_MIN, thi=LLONG_MAX;
    
    // for dx
    {
        // x1<=px+dx*t<=x2
        if(dx>0){
            // t >= (x1-px)/dx  ceil ; t <= (x2-px)/dx floor
            ll lo=cdiv(x1-px,dx);
            ll hi=fdiv(x2-px,dx);
            tlo=max(tlo,lo); thi=min(thi,hi);
        } else {
            ll d=-dx; // positive
            // x1<=px - d*t <=x2  -> px-x2 <= d*t <= px-x1
            ll lo=cdiv(px-x2,d);
            ll hi=fdiv(px-x1,d);
            tlo=max(tlo,lo); thi=min(thi,hi);
        }
    }
    {
        if(dy>0){
            ll lo=cdiv(y1-py,dy);
            ll hi=fdiv(y2-py,dy);
            tlo=max(tlo,lo); thi=min(thi,hi);
        } else {
            ll d=-dy;
            ll lo=cdiv(py-y2,d);
            ll hi=fdiv(py-y1,d);
            tlo=max(tlo,lo); thi=min(thi,hi);
        }
    }
    
    if(thi<tlo) cout<<0<<endl;
    else cout<<(thi-tlo+1)<<endl;
    return 0;
}
```
