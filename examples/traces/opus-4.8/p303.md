Looking at this problem:

We have segments forming province borders. The wall must be a simple polygon along borders, separating the two headquarters. We want minimum total cost.

This is a classic planar dual / min-cut along edges problem. The wall is a cycle of segments separating two points. 

Approach: Build a planar graph from the segments (vertices = endpoints, edges = segments). The segments partition the plane into faces. The two headquarters are in two faces. We need a cycle separating them — which corresponds to a minimum cut in the dual graph between the two faces.

Actually the standard approach: The wall is a simple closed cycle of segments that separates point A from point B. Finding minimum weight cycle separating two points = min cut in the planar dual graph.

Let me think with the dual graph. Build the planar subdivision (faces). Each segment borders two faces. In the dual, each face is a node; each segment is a dual edge connecting the two faces it separates, with weight = passability. A minimum cut between face containing A and face containing B corresponds to a set of segments... but we need a simple cycle forming the wall.

For a planar graph, min cut between two faces equals shortest path in dual between... Actually min cut separating two faces = the set of edges crossed by a cycle. The minimum cycle separating two points equals min cut between the two faces in the dual graph (a min cut, not necessarily a simple path, but for separating two faces the min cut edges form the wall).

Let me build faces using the half-edge structure: for each directed edge, find next edge by rotating. Then find faces. Determine which face contains A and B. Build dual graph and find min cut between the two faces = max flow.

Then the cut edges are the wall segments. Sum of weights = answer.

Given complexity and N≤300, this is involved but doable. Let me implement:

1. Build vertices (unique endpoints).
2. For each vertex, sort incident half-edges by angle.
3. Trace faces.
4. Point-in-face test for A and B.
5. Min cut between the two faces using max-flow in dual.

The min cut edges = segments where one side face is reachable from source and other not. Output those.

#include <bits/stdc++.h>
using namespace std;

struct Edge{int to,seg;double w;int rev;};
int N;
double X[700],Y[700];
int segx1[400],segy1[400],segx2[400],segy2[400],segv[400];

// max flow
struct MF{
    struct E{int to;double cap;int rev;int id;};
    vector<vector<E>> g;
    vector<int> level,iter;
    int n;
    void init(int n_){n=n_;g.assign(n,{});}
    void add(int a,int b,double cap,int id){
        g[a].push_back({b,cap,(int)g[b].size(),id});
        g[b].push_back({a,cap,(int)g[a].size()-1,id});
    }
    bool bfs(int s,int t){
        level.assign(n,-1);queue<int>q;level[s]=0;q.push(s);
        while(!q.empty()){int v=q.front();q.pop();
            for(auto&e:g[v])if(e.cap>1e-9&&level[e.to]<0){level[e.to]=level[v]+1;q.push(e.to);}}
        return level[t]>=0;
    }
    double dfs(int v,int t,double f){
        if(v==t)return f;
        for(int&i=iter[v];i<(int)g[v].size();i++){
            E&e=g[v][i];
            if(e.cap>1e-9&&level[v]<level[e.to]){
                double d=dfs(e.to,t,min(f,e.cap));
                if(d>1e-9){e.cap-=d;g[e.to][e.rev].cap+=d;return d;}
            }
        }
        return 0;
    }
    double maxflow(int s,int t){
        double fl=0;
        while(bfs(s,t)){iter.assign(n,0);double f;while((f=dfs(s,t,1e18))>1e-9)fl+=f;}
        return fl;
    }
};

int main(){
    scanf("%d",&N);
    int nv=0;
    vector<pair<double,double>> pts;
    auto getv=[&](double x,double y)->int{
        for(int i=0;i<nv;i++)if(fabs(pts[i].first-x)<1e-9&&fabs(pts[i].second-y)<1e-9)return i;
        pts.push_back({x,y});return nv++;
    };
    vector<int> ea,eb;
    for(int i=0;i<N;i++){
        scanf("%d %d %d %d %d",&segx1[i],&segy1[i],&segx2[i],&segy2[i],&segv[i]);
        int a=getv(segx1[i],segy1[i]);
        int b=getv(segx2[i],segy2[i]);
        ea.push_back(a);eb.push_back(b);
    }
    double ax,ay,bx,by;
    scanf("%lf %lf %lf %lf",&ax,&ay,&bx,&by);
    
    // build half-edges: 2*N directed edges. he 2*i from a to b (seg i), 2*i+1 from b to a
    int H=2*N;
    vector<int> hfrom(H),hto(H),hseg(H);
    for(int i=0;i<N;i++){
        hfrom[2*i]=ea[i];hto[2*i]=eb[i];hseg[2*i]=i;
        hfrom[2*i+1]=eb[i];hto[2*i+1]=ea[i];hseg[2*i+1]=i;
    }
    // for each vertex, list of outgoing half-edges sorted by angle
    vector<vector<int>> out(nv);
    for(int h=0;h<H;h++) out[hfrom[h]].push_back(h);
    for(int v=0;v<nv;v++){
        sort(out[v].begin(),out[v].end(),[&](int p,int q){
            double a1=atan2(pts[hto[p]].second-pts[v].second,pts[hto[p]].first-pts[v].first);
            double a2=atan2(pts[hto[q]].second-pts[v].second,pts[hto[q]].first-pts[v].first);
            return a1<a2;
        });
    }
    // map: for half-edge h, find its index in out[hfrom[h]]
    // next half-edge in face: from incoming half-edge h (arrives at hto[h]),
    // the next is the outgoing edge that is the reverse rotated.
    // standard: next(h) = the outgoing edge at vertex hto[h] that comes clockwise-next after reverse of h
    // reverse of h: rev(2i)=2i+1, rev(2i+1)=2i
    auto rev=[&](int h){return h^1;};
    // pos of half-edge among out[its from]
    vector<int> pos(H);
    for(int v=0;v<nv;v++)for(int k=0;k<(int)out[v].size();k++)pos[out[v][k]]=k;
    
    auto nexte=[&](int h)->int{
        // arrive at vertex u=hto[h]. reverse edge r leaves u. 
        int r=rev(h);
        int u=hfrom[r]; // = hto[h]
        int k=pos[r];
        int sz=out[u].size();
        int nk=(k-1+sz)%sz; // previous in CCW order = clockwise next
        return out[u][nk];
    };
    
    // trace faces
    vector<int> faceOf(H,-1);
    vector<vector<int>> faces; // list of half-edges
    for(int h=0;h<H;h++){
        if(faceOf[h]!=-1)continue;
        int fid=faces.size();
        vector<int> cyc;
        int cur=h;
        do{
            faceOf[cur]=fid;
            cyc.push_back(cur);
            cur=nexte(cur);
        }while(cur!=h);
        faces.push_back(cyc);
    }
    int F=faces.size();
    // compute signed area of each face; locate A,B in faces
    // point in polygon for bounded faces. The outer face has negative area (CCW traversal gives... )
    // We need face containing a point: a point is in face f if it's inside the polygon of f.
    auto faceArea=[&](int f)->double{
        double s=0;auto&cyc=faces[f];
        for(int h:cyc){int u=hfrom[h],w=hto[h];s+=pts[u].first*pts[w].second-pts[w].first*pts[u].second;}
        return s/2;
    };
    auto pip=[&](int f,double px,double py)->bool{
        // ray cast using face polygon vertices
        auto&cyc=faces[f];bool in=false;
        for(int h:cyc){
            double x1=pts[hfrom[h]].first,y1=pts[hfrom[h]].second;
            double x2=pts[hto[h]].first,y2=pts[hto[h]].second;
            if((y1>py)!=(y2>py)){
                double xint=x1+(py-y1)/(y2-y1)*(x2-x1);
                if(px<xint)in=!in;
            }
        }
        return in;
    };
    int fa=-1,fb=-1;
    for(int f=0;f<F;f++){
        if(faceArea(f)<=0)continue; // only bounded CCW faces
        if(pip(f,ax,ay))fa=f;
        if(pip(f,bx,by))fb=f;
    }
    // pick smallest containing face (most nested) - choose minimal area
    // refine: among faces containing A pick min area
    auto bestFace=[&](double px,double py)->int{
        int best=-1;double ba=1e18;
        for(int f=0;f<F;f++){
            if(faceArea(f)<=0)continue;
            if(pip(f,px,py)){double ar=faceArea(f);if(ar<ba){ba=ar;best=f;}}
        }
        return best;
    };
    fa=bestFace(ax,ay);fb=bestFace(by==by?bx:bx,by);
    
    // dual graph: each segment i borders faceOf[2i] and faceOf[2i+1]
    MF mf;mf.init(F);
    for(int i=0;i<N;i++){
        int f1=faceOf[2*i],f2=faceOf[2*i+1];
        if(f1!=f2) mf.add(f1,f2,segv[i],i);
    }
    mf.maxflow(fa,fb);
    // find reachable from fa
    vector<int> reach(F,0);
    {queue<int>q;q.push(fa);reach[fa]=1;
        while(!q.empty()){int v=q.front();q.pop();
            for(auto&e:mf.g[v])if(e.cap>1e-9&&!reach[e.to]){reach[e.to]=1;q.push(e.to);}}}
    double total=0;vector<int> res;
    for(int i=0;i<N;i++){
        int f1=faceOf[2*i],f2=faceOf[2*i+1];
        if(f1!=f2 && reach[f1]!=reach[f2]){total+=segv[i];res.push_back(i+1);}
    }
    printf("%d\n",(int)llround(total));
    printf("%d\n",(int)res.size());
    for(int i=0;i<(int)res.size();i++)printf("%d%c",res[i],i+1<(int)res.size()?' ':'\n');
    return 0;
}
