Bishops on black and white diagonals are independent. Rotate the board 45 degrees to convert diagonals into rows/columns problem.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n,k;
    cin>>n>>k;
    
    // diagonals: classify cells by (i+j) parity -> two independent sets
    // After 45-degree rotation, bishops on one color become rooks on rows.
    // The diagonals of one color have lengths. We count ways to place
    // m non-attacking bishops on a color set.
    
    // For diagonals (i-j) constant within same color (i+j) parity grouping.
    // Standard approach: separate by color of (i+j).
    // For each color, diagonals along (i+j) directions form rows whose sizes
    // we compute, then DP placing bishops so no two share a "column" (i-j).
    
    // Let's build the two groups. Group by parity of (i+j).
    // Within a group, the relevant lines are the anti-diagonals; we sort
    // their sizes and DP.
    
    // dp over diagonals: each diagonal of size s, placing bishops choosing
    // distinct other-diagonal indices.
    
    auto solve=[&](vector<int> sizes)->vector<long long>{
        // sizes: number of cells available on each "row" (one diagonal direction)
        // but bishops attack along both diagonals; after rotation one direction = rows, other = columns
        // We DP: process rows sorted ascending, dp[j]=ways to place j bishops
        sort(sizes.begin(), sizes.end());
        int m=sizes.size();
        vector<long long> dp(m+1,0);
        dp[0]=1;
        int placed=0;
        for(int idx=0;idx<m;idx++){
            int s=sizes[idx];
            // available columns in this row = s - placed (already used columns)
            for(int j=min(idx+1,m); j>=1; j--){
                long long add = (long long)(s-(j-1));
                if(add>0) dp[j]+=dp[j-1]*add;
            }
        }
        return dp;
    };
    
    // Build groups. Rotate coordinates: u=i+j, v=i-j.
    // For one color (parity of i+j fixed), the cells form a grid in (u,v).
    // Each diagonal in u-direction is a "row", v-direction is a "column".
    // The number of cells on each row (fixed u) is what we need.
    
    // Collect for each parity, map from u to count of cells.
    map<int,int> cntEven, cntOdd;
    for(int i=0;i<n;i++)for(int j=0;j<n;j++){
        int u=i+j;
        if(u%2==0) cntEven[u]++;
        else cntOdd[u]++;
    }
    
    vector<int> evenSizes, oddSizes;
    for(auto&p:cntEven) evenSizes.push_back(p.second);
    for(auto&p:cntOdd) oddSizes.push_back(p.second);
    
    vector<long long> de=solve(evenSizes);
    vector<long long> doo=solve(oddSizes);
    
    long long ans=0;
    int me=de.size()-1, mo=doo.size()-1;
    for(int a=0;a<=me;a++){
        int b=k-a;
        if(b>=0 && b<=mo){
            ans += de[a]*doo[b];
        }
    }
    cout<<ans<<endl;
    return 0;
}
```
