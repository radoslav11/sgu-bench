using namespace std;
int n,m;
bool adj[7][7];
double X[7],Y[7];
mt19937 rng(12345);
double rnd(double a,double b){ return a + (b-a)*(rng()/(double)UINT32_MAX); }

const double EDGE_TOL=1e-7;
const double MARGIN=0.03;

// check candidate (cx,cy) for vertex v against processed set (order positions in vertex 'order' up to idx)
bool checkPair(int v, double cx, double cy, vector<int>&order, int idx){
  for(int j=0;j<idx;j++){
    int u=order[j];
    double dx=cx-X[u], dy=cy-Y[u];
    double d=sqrt(dx*dx+dy*dy);
    if(adj[v][u]){
      if(fabs(d-1)>EDGE_TOL) return false;
    } else {
      if(d<MARGIN) return false;
      if(fabs(d-1)<MARGIN) return false;
    }
  }
  return true;
}
