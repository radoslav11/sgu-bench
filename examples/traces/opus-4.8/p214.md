Looking at this problem:

We need to align λ and μ. We can insert characters into either string. When we align character λ[i] with μ[j], cost is d(λ[i], μ[j]). 

If we have an unmatched character in λ (say λ[i]), it means α has λ[i] at this position, and β must have some character there too (since strings equal length). We can choose the cheapest character to pair with λ[i] — cost = min over c of d(λ[i], c). Similarly for unmatched μ[j], cost = min over c of d(c, μ[j]).

This is like edit distance / alignment. Let:
- `gapL[i]` = min cost to keep λ[i] and pair with best char = min_c d(λ[i], c)
- `gapM[j]` = min cost to keep μ[j] and pair with best char = min_c d(c, μ[j])
- match cost = d(λ[i], μ[j])

DP: dp[i][j] = min cost aligning λ[0..i), μ[0..j).
- dp[i][j] = dp[i-1][j-1] + d(λ[i-1],μ[j-1])  (match positions)
- dp[i][j] = dp[i-1][j] + gapL[i-1]  (λ char unmatched: insert best char into μ... actually α gets λ char, β gets a chosen char)
- dp[i][j] = dp[i][j-1] + gapM[j-1]

We must reconstruct α and β.

When matching: α gets λ[i-1], β gets μ[j-1].
When gapL: α gets λ[i-1], β gets the best char c that minimizes d(λ[i-1],c).
When gapM: β gets μ[j-1], α gets best char c minimizing d(c,μ[j-1]).

Let me precompute best matching chars.

Length: each step adds 1 char to each, total ≤ |λ|+|μ| ≤ 4000. Good.

#include <bits/stdc++.h>
using namespace std;
int main(){
    string sigma; getline(cin, sigma);
    string lam, mu; getline(cin,lam); getline(cin,mu);
    int S=sigma.size();
    vector<vector<int>> d(S, vector<int>(S));
    for(int i=0;i<S;i++)for(int j=0;j<S;j++) cin>>d[i][j];
    map<char,int> idx;
    for(int i=0;i<S;i++) idx[sigma[i]]=i;
    // bestRow[i]: char c minimizing d[i][c]; bestCol[j]: char minimizing d[c][j]
    vector<int> bestRowCost(S), bestRowCh(S), bestColCost(S), bestColCh(S);
    for(int i=0;i<S;i++){
        int bc=INT_MAX,bch=0;
        for(int c=0;c<S;c++) if(d[i][c]<bc){bc=d[i][c];bch=c;}
        bestRowCost[i]=bc; bestRowCh[i]=bch;
    }
    for(int j=0;j<S;j++){
        int bc=INT_MAX,bch=0;
        for(int c=0;c<S;c++) if(d[c][j]<bc){bc=d[c][j];bch=c;}
        bestColCost[j]=bc; bestColCh[j]=bch;
    }
    int n=lam.size(), m=mu.size();
    vector<int> li(n), mi(m);
    for(int i=0;i<n;i++) li[i]=idx[lam[i]];
    for(int j=0;j<m;j++) mi[j]=idx[mu[j]];
    
    // dp
    vector<vector<long long>> dp(n+1, vector<long long>(m+1));
    for(int i=0;i<=n;i++) for(int j=0;j<=m;j++) dp[i][j]=LLONG_MAX/2;
    dp[0][0]=0;
    for(int i=0;i<=n;i++){
        for(int j=0;j<=m;j++){
            if(dp[i][j]==LLONG_MAX/2) continue;
            long long cur=dp[i][j];
            if(i<n) dp[i+1][j]=min(dp[i+1][j], cur+bestRowCost[li[i]]);
            if(j<m) dp[i][j+1]=min(dp[i][j+1], cur+bestColCost[mi[j]]);
            if(i<n&&j<m) dp[i+1][j+1]=min(dp[i+1][j+1], cur+(long long)d[li[i]][mi[j]]);
        }
    }
    // reconstruct
    string A,B;
    int i=n,j=m;
    while(i>0||j>0){
        long long cur=dp[i][j];
        if(i>0&&j>0 && cur==dp[i-1][j-1]+(long long)d[li[i-1]][mi[j-1]]){
            A.push_back(lam[i-1]); B.push_back(mu[j-1]); i--; j--;
        } else if(i>0 && cur==dp[i-1][j]+bestRowCost[li[i-1]]){
            A.push_back(lam[i-1]); B.push_back(sigma[bestRowCh[li[i-1]]]); i--;
        } else {
            B.push_back(mu[j-1]); A.push_back(sigma[bestColCh[mi[j-1]]]); j--;
        }
    }
    reverse(A.begin(),A.end()); reverse(B.begin(),B.end());
    cout<<dp[n][m]<<"\n"<<A<<"\n"<<B<<"\n";
}
