This is an Eulerian path problem on a multigraph with vertices 0-6.

Each domino is an edge. We need to find an Euler path/circuit.

```cpp
#include <bits/stdc++.h>
using namespace std;

int N;
int u[105], v[105];
int deg[7];
bool used[105];
// adjacency: for each vertex, list of (other vertex, edge index)
vector<pair<int,int>> adj[7];
vector<int> circuit; // edge indices in order traversed
// We need to track direction; store vertex sequence

// Iterative Hierholzer
int main(){
    scanf("%d",&N);
    for(int i=0;i<N;i++){
        scanf("%d %d",&u[i],&v[i]);
        deg[u[i]]++;
        deg[v[i]]++;
        adj[u[i]].push_back({v[i],i});
        adj[v[i]].push_back({u[i],i});
    }
    
    // Check connectivity of vertices that have edges
    // Count odd degree vertices
    vector<int> odd;
    for(int i=0;i<7;i++) if(deg[i]%2) odd.push_back(i);
    
    if(odd.size()!=0 && odd.size()!=2){
        printf("No solution\n");
        return 0;
    }
    
    // Check connectivity: all edges must be in one connected component
    // Find a starting vertex (one with edges)
    int start=-1;
    if(odd.size()==2) start=odd[0];
    else {
        for(int i=0;i<7;i++) if(deg[i]>0){ start=i; break;}
    }
    if(start==-1){
        // no edges, N>=1 means there's at least one piece... actually N>=1
        // if N>=1 there must be edges
        printf("No solution\n");
        return 0;
    }
    
    // connectivity check via DFS over vertices using edges
    {
        vector<bool> vis(7,false);
        stack<int> st;
        st.push(start);
        vis[start]=true;
        while(!st.empty()){
            int x=st.top(); st.pop();
            for(auto&e:adj[x]){
                if(!vis[e.first]){
                    vis[e.first]=true;
                    st.push(e.first);
                }
            }
        }
        for(int i=0;i<7;i++){
            if(deg[i]>0 && !vis[i]){
                printf("No solution\n");
                return 0;
            }
        }
    }
    
    // Hierholzer
    // pointer for each vertex
    vector<int> ptr(7,0);
    // stack holds (vertex, edge used to get here)
    stack<pair<int,int>> st; // vertex, edgeIndex (edge that led to this vertex)
    st.push({start,-1});
    vector<int> edgePath; // edges in order (reverse)
    vector<int> vertPath;
    
    while(!st.empty()){
        int x=st.top().first;
        // find unused edge
        while(ptr[x]<(int)adj[x].size() && used[adj[x][ptr[x]].second]) ptr[x]++;
        if(ptr[x]==(int)adj[x].size()){
            // dead end
            edgePath.push_back(st.top().second);
            vertPath.push_back(x);
            st.pop();
        } else {
            auto e=adj[x][ptr[x]];
            used[e.second]=true;
            st.push({e.first,e.second});
        }
    }
    
    // edgePath and vertPath are in reverse order
    reverse(edgePath.begin(),edgePath.end());
    reverse(vertPath.begin(),vertPath.end());
    // edgePath[0] is -1 (start), then edges. vertPath gives vertex sequence
    // Check all edges used
    if((int)edgePath.size()-1 != N){
        printf("No solution\n");
        return 0;
    }
    
    // vertPath: sequence of vertices, length N+1
    // for each step i (1..N), edge edgePath[i] connects vertPath[i-1] to vertPath[i]
    // determine orientation: if u[edge]==vertPath[i-1] then '+', else '-'
    for(int i=1;i<(int)edgePath.size();i++){
        int eidx=edgePath[i];
        int prev=vertPath[i-1];
        // edge goes from prev to vertPath[i]
        if(u[eidx]==prev){
            printf("%d +\n",eidx+1);
        } else {
            printf("%d -\n",eidx+1);
        }
    }
    
    return 0;
}
```
