```cpp
#include <bits/stdc++.h>
using namespace std;
int M,N;
long long dp[10][1<<9];
void rec(int row,int cur,int nxt,long long val,long long* ndp){
    if(row==M){ ndp[nxt]+=val; return;}
    if(cur&(1<<row)){ rec(row+1,cur,nxt,val,ndp); return;}
    // cell (row) empty, must fill
    // 1) vertical domino: fill row,row+1 in current column
    if(row+1<M && !(cur&(1<<(row+1))))
        rec(row+2,cur,nxt,val,ndp);
    // 2) horizontal domino into next column
    if(!(nxt&(1<<row)))
        rec(row+1,cur,nxt|(1<<row),val,ndp);
    // corners (2x2 minus one cell), covering current cell
    // The 2x2 occupies current col rows row,row+1 and next col rows row,row+1
    if(row+1<M && !(cur&(1<<(row+1)))){
        // current both filled, next: one of the two filled (missing corner)
        // missing top-right: fill next row+1 only
        if(!(nxt&(1<<(row+1))))
            rec(row+2,cur,nxt|(1<<(row+1)),val,ndp);
        // missing bottom-right: fill next row only
        if(!(nxt&(1<<row)))
            rec(row+2,cur,nxt|(1<<row),val,ndp);
    }
    // corners where current column has one cell missing (this empty cell + extends)
    // covering cur row, next row, next row+1 (current row+1 NOT filled here->missing)
    if(row+1<M){
        // fill cur row, next row, next row+1 (missing current bottom)
        if(!(nxt&(1<<row)) && !(nxt&(1<<(row+1))))
            rec(row+1,cur,nxt|(1<<row)|(1<<(row+1)),val,ndp);
    }
    // missing current top: covers cur row+1, next row, next row+1; but cur row is current empty cell - can't leave it
    // Since we must cover current cell (row), the missing-current-top corner doesn't cover row, skip
}
int main(){
    cin>>M>>N;
    static long long cur[1<<9], nxt[1<<9];
    for(int i=0;i<(1<<M);i++) cur[i]=0;
    cur[0]=1;
    for(int c=0;c<N;c++){
        for(int i=0;i<(1<<M);i++) nxt[i]=0;
        for(int m=0;m<(1<<M);m++) if(cur[m]) rec(0,m,0,cur[m],nxt);
        for(int i=0;i<(1<<M);i++) cur[i]=nxt[i];
    }
    cout<<cur[0]<<endl;
}
```
