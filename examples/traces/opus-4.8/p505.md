#include <bits/stdc++.h>
using namespace std;

struct Trie {
    vector<array<int,26>> ch;
    vector<int> in, out;
    Trie(){ newNode(); }
    int newNode(){
        ch.push_back(array<int,26>());
        ch.back().fill(-1);
        return (int)ch.size()-1;
    }
    // insert string, return end node
    int insertStr(const string &s, bool rev){
        int cur=0;
        if(!rev){
            for(char c: s){
                int x=c-'a';
                if(ch[cur][x]==-1) ch[cur][x]=newNode();
                cur=ch[cur][x];
            }
        } else {
            for(int i=(int)s.size()-1;i>=0;i--){
                int x=s[i]-'a';
                if(ch[cur][x]==-1) ch[cur][x]=newNode();
                cur=ch[cur][x];
            }
        }
        return cur;
    }
    // find node for string, -1 if not present
    int findStr(const string &s, bool rev){
        int cur=0;
        if(!rev){
            for(char c: s){
                int x=c-'a';
                if(ch[cur][x]==-1) return -1;
                cur=ch[cur][x];
            }
        } else {
            for(int i=(int)s.size()-1;i>=0;i--){
                int x=s[i]-'a';
                if(ch[cur][x]==-1) return -1;
                cur=ch[cur][x];
            }
        }
        return cur;
    }
    void buildEuler(){
        int N=ch.size();
        in.assign(N,0); out.assign(N,0);
        int timer=0;
        // iterative dfs
        vector<pair<int,int>> st;
        st.push_back({0,0});
        in[0]=++timer;
        while(!st.empty()){
            auto &top=st.back();
            int node=top.first;
            bool found=false;
            while(top.second<26){
                int c=ch[node][top.second];
                top.second++;
                if(c!=-1){
                    in[c]=++timer;
                    st.push_back({c,0});
                    found=true;
                    break;
                }
            }
            if(!found){
                out[node]=timer;
                st.pop_back();
            }
        }
    }
};

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin>>n;
    vector<string> g(n);
    for(int i=0;i<n;i++) cin>>g[i];
    
    Trie pref, suf;
    vector<int> endPref(n), endSuf(n);
    for(int i=0;i<n;i++){
        endPref[i]=pref.insertStr(g[i], false);
        endSuf[i]=suf.insertStr(g[i], true);
    }
    pref.buildEuler();
    suf.buildEuler();
    
    // points: (x = in-time of endPref node, y = in-time of endSuf node)
    int m;
    cin>>m;
    vector<long long> ans(m,0);
    
    // build points
    vector<int> px(n), py(n);
    for(int i=0;i<n;i++){
        px[i]=pref.in[endPref[i]];
        py[i]=suf.in[endSuf[i]];
    }
    // sort points by x
    vector<int> order(n);
    iota(order.begin(),order.end(),0);
    sort(order.begin(),order.end(),[&](int a,int b){return px[a]<px[b];});
    
    // BIT over y, size = number of suf nodes (in-times range 1..M)
    int Msz = suf.in.size(); // max in-time is suf.ch.size()-? actually timer max = number of nodes
    int bitN = (int)suf.ch.size()+1;
    vector<int> bit(bitN+2,0);
    auto bitUpdate=[&](int i,int v){
        for(;i<=bitN;i+= i&(-i)) bit[i]+=v;
    };
    auto bitQuery=[&](int i){
        int s=0;
        for(;i>0;i-= i&(-i)) s+=bit[i];
        return s;
    };
    
    // events: for each query, rectangle [x1,x2]x[y1,y2]
    // event at threshold T, sign, y1,y2, queryIdx
    struct Event{int thr; int sign; int y1; int y2; int qi;};
    vector<Event> events;
    events.reserve(2*m);
    
    string p,s;
    for(int q=0;q<m;q++){
        cin>>p>>s;
        int np=pref.findStr(p,false);
        int ns=suf.findStr(s,true);
        if(np==-1 || ns==-1){
            // answer 0, no events
            continue;
        }
        int x1=pref.in[np], x2=pref.out[np];
        int y1=suf.in[ns], y2=suf.out[ns];
        // count points with x in [x1,x2], y in [y1,y2]
        // = F(x2) - F(x1-1)
        events.push_back({x2, +1, y1, y2, q});
        events.push_back({x1-1, -1, y1, y2, q});
    }
    
    // sort events by thr ascending
    sort(events.begin(),events.end(),[](const Event&a,const Event&b){return a.thr<b.thr;});
    
    int pi=0; // pointer into order (points sorted by x)
    for(auto &e: events){
        // add all points with px <= e.thr
        while(pi<n && px[order[pi]]<=e.thr){
            int idx=order[pi];
            bitUpdate(py[idx], 1);
            pi++;
        }
        long long cnt = bitQuery(e.y2) - bitQuery(e.y1-1);
        ans[e.qi]+= (long long)e.sign * cnt;
    }
    
    // output
    string outbuf;
    outbuf.reserve(m*3);
    for(int q=0;q<m;q++){
        outbuf += to_string(ans[q]);
        outbuf += '\n';
    }
    cout<<outbuf;
    return 0;
}
