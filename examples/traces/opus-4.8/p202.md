Looking at this problem - the Frame-Stewart algorithm for multi-peg Tower of Hanoi.

The Frame-Stewart approach: to move n disks using m pegs:
- Choose k, move top k disks to an intermediate peg using all m pegs
- Move remaining n-k disks using m-1 pegs (since one peg holds the k disks)
- Move k disks to destination using all m pegs

We need to find optimal k for minimum moves, then generate the moves.

Let me compute the minimum number of moves with DP, storing the optimal split, then recursively generate moves. I need to track which disk is on top of each peg for output.

Use `unsigned long long` / `__int128` for safety since N can be 64.

For tracking tops, I'll simulate actual peg states (stacks).

Let me code this carefully. The DP: `f[n][m]` = min moves for n disks, m pegs. With m>=3: f[0]=0, f[n][3]=2^n-1. For general: f[n][m] = min over k of 2*f[k][m] + f[n-k][m-1]. Base f[1][m]=1.

```cpp
#include <bits/stdc++.h>
using namespace std;

typedef unsigned long long ull;
const ull INF = ~0ULL;

int N, M;
ull f[65][66];
int bestk[65][66];

ull solve(int n, int m){
    if(n==0) return 0;
    if(n==1) return 1;
    if(f[n][m]!=INF) return f[n][m];
    ull res=INF; int bk=1;
    if(m>=3){
        // try k from 1 to n-1
        for(int k=1;k<n;k++){
            ull a=solve(k,m);
            ull b=solve(n-k,m-1);
            // 2*a + b, careful overflow but values fit
            ull cand = 2*a + b;
            if(cand<res){res=cand;bk=k;}
        }
    }
    f[n][m]=res; bestk[n][m]=bk;
    return res;
}

// stacks: peg state, top is back of vector. We track disk radii.
vector<int> peg[66];

void output_move(int disk, int from, int to){
    // move disk from 'from' to 'to'
    // disk should be top of from
    peg[from].pop_back();
    if(peg[to].empty()){
        printf("move %d from %d to %d\n", disk, from, to);
    } else {
        printf("move %d from %d to %d atop %d\n", disk, from, to, peg[to].back());
    }
    peg[to].push_back(disk);
}

// move top n disks from 'from' to 'to' using m available pegs.
// 'avail' is list of pegs that can be used (size m), includes from and to.
void move(int n, int from, int to, vector<int> avail){
    if(n==0) return;
    if(n==1){
        // move the single disk
        output_move(peg[from].back(), from, to);
        return;
    }
    int m = avail.size();
    // pick intermediate peg (not from, not to)
    int inter=-1;
    for(int p: avail) if(p!=from && p!=to){inter=p;break;}
    int k = bestk[n][m];
    // move top k disks from->inter using all m pegs
    move(k, from, inter, avail);
    // move remaining n-k from->to using m-1 pegs (exclude inter)
    vector<int> avail2;
    for(int p: avail) if(p!=inter) avail2.push_back(p);
    move(n-k, from, to, avail2);
    // move k disks from inter->to using all m pegs
    move(k, inter, to, avail);
}

int main(){
    scanf("%d %d",&N,&M);
    for(int i=0;i<=64;i++)for(int j=0;j<=65;j++)f[i][j]=INF;
    ull ans = solve(N, M);
    printf("%llu\n", ans);
    // init pegs
    for(int d=N; d>=1; d--) peg[1].push_back(d);
    vector<int> avail;
    for(int p=1;p<=M;p++) avail.push_back(p);
    move(N, 1, M, avail);
    return 0;
}
```

Let me verify the recursion handles base cases. For n=1, bestk not set but handled. For larger n with m pegs, bestk[n][m] is set in solve.

One concern: when m=3 in recursion (only 3 pegs available because higher pegs got "used up"), the formula still works since we try all k including k giving standard 2^n-1.

For n=1 with any m, returns 1. Good.
