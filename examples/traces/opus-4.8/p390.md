using namespace std;
typedef unsigned long long ull;
typedef long long ll;

// sum of digit sums from 1 to n
ull sumDigits(ull n){
    if(n==0) return 0;
    ull total=0;
    ull p=1;
    while(p<=n){
        ull higher = n/(p*10);
        ull cur = (n/p)%10;
        ull lower = n%p;
        // sum over each digit position
        total += higher*45*p;
        total += (cur*(cur-1)/2)*p;
        total += cur*(lower+1);
        if(p> n/10) break;
        p*=10;
    }
    return total;
}
