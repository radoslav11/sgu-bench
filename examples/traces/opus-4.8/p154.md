The number of trailing zeros in N! is determined by the count of factor 5s. Binary search for N.

```cpp
#include <bits/stdc++.h>
using namespace std;

long long zeros(long long n){
    long long c=0;
    for(long long p=5;p<=n;p*=5) c+=n/p;
    return c;
}

int main(){
    long long Q;
    cin>>Q;
    long long lo=0, hi=500000000LL, ans=-1;
    while(lo<=hi){
        long long mid=(lo+hi)/2;
        long long z=zeros(mid);
        if(z>=Q){
            if(z==Q) ans=mid;
            hi=mid-1;
        } else {
            lo=mid+1;
        }
    }
    if(ans==-1) cout<<"No solution\n";
    else cout<<ans<<"\n";
    return 0;
}
```
