Each robber gets floor(Xi*M/Y) initially, then we distribute remaining coins to those with largest fractional parts.

The unfairness is |Xi/Y - Ki/M| = |Xi*M - Ki*Y|/(Y*M). Minimizing sum means minimizing sum of |Xi*M - Ki*Y|.

Give each Ki = floor(Xi*M/Y), remainder r = Xi*M mod Y. Total distributed = sum of floors, need M total. The number of extra coins to give = M - sum(floors). Give them to those with largest remainders (giving +1 changes |...| from r to Y-r, beneficial when r > Y/2 roughly, but we must give exactly the needed count, so pick largest remainders).

```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    int n; long long M, Y;
    scanf("%d %lld %lld", &n, &M, &Y);
    vector<long long> X(n), K(n), rem(n);
    long long sum=0;
    for(int i=0;i<n;i++){
        scanf("%lld",&X[i]);
        long long prod = X[i]*M;
        K[i] = prod / Y;
        rem[i] = prod % Y;
        sum += K[i];
    }
    long long extra = M - sum;
    vector<int> idx(n);
    iota(idx.begin(), idx.end(), 0);
    sort(idx.begin(), idx.end(), [&](int a,int b){
        return rem[a] > rem[b];
    });
    for(int i=0;i<extra;i++) K[idx[i]]++;
    for(int i=0;i<n;i++) printf("%lld ", K[i]);
    printf("\n");
    return 0;
}
```
