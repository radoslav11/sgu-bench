struct Mat {

Mat mul(const Mat&A,const Mat&B){
    Mat C;
    for(int i=0;i<K;i++){
        bitset<50> row;
        for(int k=0;k<K;k++) if(A.r[i][k]) row ^= B.r[k];

}
