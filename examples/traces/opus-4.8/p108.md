Looking at this problem:

- Sieve self-numbers up to N.
- For each number i, mark d(i) as not-self.
- Count self-numbers, and answer queries by index.

Memory limit is 4096 KB with N up to 10^7. A boolean array of 10^7 bytes = 10 MB exceeds limit. So use a bitset: 10^7 bits = 1.25 MB. That fits.

For queries by index, we can compute prefix counts but storing all self-numbers' values could be many. Instead, we need a[s_i]. With K up to 5000 queries, and indices up to ~9 million potential self-numbers, we need to find the s-th self-number. We can do a single pass collecting needed indices: sort queries, walk through marking, find the ones we need.

Let me use a bitset and iterate.

```cpp
#include <bits/stdc++.h>
using namespace std;

static bitset<10000001> notSelf;

int main(){
    int N, K;
    scanf("%d %d", &N, &K);
    vector<int> s(K);
    for(int i=0;i<K;i++) scanf("%d", &s[i]);
    
    // mark d(i) for all i
    for(int i=1;i<=N;i++){
        int sum=i, x=i;
        while(x){ sum+=x%10; x/=10; }
        if(sum<=N) notSelf[sum]=1;
    }
    
    // We need: count of self-numbers in [1..N], and a[s_j] for each query.
    // Sort queries to find by index in one pass.
    vector<pair<int,int>> q(K); // (index, original position)
    for(int i=0;i<K;i++) q[i]={s[i], i};
    sort(q.begin(), q.end());
    
    vector<int> ans(K);
    long long count=0;
    int qi=0;
    for(int i=1;i<=N;i++){
        if(!notSelf[i]){
            count++;
            while(qi<K && q[qi].first==count){
                ans[q[qi].second]=i;
                qi++;
            }
        }
    }
    
    printf("%lld\n", count);
    for(int i=0;i<K;i++){
        printf("%d", ans[i]);
        if(i+1<K) printf(" ");
    }
    printf("\n");
    return 0;
}
```

This should work within constraints. The bitset of 10^7+1 bits is about 1.25 MB. The marking loop is O(N). Query resolution in one pass.
