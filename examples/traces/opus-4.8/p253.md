Looking at this problem:

- Convex polygon with N vertices (CCW order)
- M points, count how many are inside (or on boundary) the polygon
- If count >= K, output YES.

Standard point-in-convex-polygon in O(log N) per query.

Approach: pick vertex 0 as fan origin. Use binary search on angle.

```cpp
#include <bits/stdc++.h>
using namespace std;

int N, M, K;
vector<long long> X, Y;

long long cross(long long ox, long long oy, long long ax, long long ay, long long bx, long long by){
    return (ax-ox)*(by-oy) - (ay-oy)*(bx-ox);
}

// point in convex polygon (CCW), including boundary
bool inside(long long px, long long py){
    // relative to vertex 0
    // check point is within the angular range [vertex1, vertex N-1]
    long long c1 = cross(X[0],Y[0],X[1],Y[1],px,py);
    long long cn = cross(X[0],Y[0],X[N-1],Y[N-1],px,py);
    if(c1 < 0 || cn > 0) return false;
    // binary search for the triangle
    int lo = 1, hi = N-1;
    while(hi - lo > 1){
        int mid = (lo+hi)/2;
        if(cross(X[0],Y[0],X[mid],Y[mid],px,py) >= 0) lo = mid;
        else hi = mid;
    }
    // now point in fan between lo and lo+1; check the far edge
    long long c = cross(X[lo],Y[lo],X[lo+1],Y[lo+1],px,py);
    return c >= 0;
}

int main(){
    scanf("%d %d %d", &N, &M, &K);
    X.resize(N); Y.resize(N);
    for(int i=0;i<N;i++){
        scanf("%lld %lld", &X[i], &Y[i]);
    }
    int cnt = 0;
    for(int i=0;i<M;i++){
        long long px, py;
        scanf("%lld %lld", &px, &py);
        if(inside(px,py)) cnt++;
    }
    printf(cnt>=K ? "YES\n" : "NO\n");
    return 0;
}
```

Wait, I need to be careful with the binary search edge cases. Let me reconsider. The standard approach:

- `c1 = cross(v0, v1, p)` must be >= 0 (left side of edge v0->v1)
- `cn = cross(v0, v_{n-1}, p)` must be <= 0 (right side of edge v0->v_{n-1})

These ensure the point is in the angular wedge. Then binary search finds the sector, and we check the outer edge. This handles boundary correctly with `>=` / `<=`.
