Looking at this problem:

We need to find the next correct bracket sequence in lexicographic order.

Since '(' < ')', to get the next sequence, we want to find the rightmost position where we can change a '(' to ')' and still complete the rest validly with the smallest sequence.

Approach: Scan from right to left. Find a '(' that we can flip to ')'. When we flip position i from '(' to ')', the remaining suffix (positions i+1..n-1) must be filled with the lexicographically smallest valid completion given the balance constraints.

Let me think in terms of balance. Let `open` = number of '(' used so far up to position i, etc. Actually, let me compute prefix balance.

For each position i from right to left, if s[i]=='(', try making it ')'. After setting prefix balance, the remaining positions must form a valid completion. Let balance after position i (with the flip) be `bal`. We have `rem = n-i-1` positions left. We need to fill them: the smallest completion is to put as many '(' as possible then ')'. We need: number of remaining '(' = a, ')' = b, a+b=rem, total balance ends at 0: bal + a - b = 0. Also bal >=0 always with minimal prefix. The smallest fill = '(' repeated then ')' — but to be lexicographically smallest we put '(' first. a = (rem - bal)/2, b = rem - a. Need rem-bal >=0 and even, and a>=0, b>=0. Also bal must stay >=0 throughout — putting '(' first keeps it non-negative.

We want the rightmost i to flip to get the next (lexicographically next means change as far right as possible).

```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    string s; getline(cin,s);
    // strip possible whitespace
    while(!s.empty() && (s.back()=='\n'||s.back()=='\r'||s.back()==' ')) s.pop_back();
    int n=s.size();
    // prefix balance before position i
    // compute balance up to each index
    for(int i=n-1;i>=0;i--){
        if(s[i]=='('){
            // balance before i
            int bal=0;
            for(int j=0;j<i;j++) bal += (s[j]=='(')?1:-1;
            bal += 1; // we now flip s[i] to ')'
            // actually after flipping s[i]=')', balance = bal_before -1... wait
            // recompute
            int balBefore=0;
            for(int j=0;j<i;j++) balBefore += (s[j]=='(')?1:-1;
            int newBal = balBefore - 1; // s[i]=')'
            if(newBal<0) continue;
            int rem = n-i-1;
            if((rem-newBal)<0) continue;
            if(((rem-newBal)&1)) continue;
            int a=(rem-newBal)/2;
            int b=rem-a;
            string res = s.substr(0,i);
            res += ')';
            res += string(a,'(');
            res += string(b,')');
            cout<<res<<"\n";
            return 0;
        }
    }
    cout<<"No solution\n";
    return 0;
}
```

This is O(n^2) which for n=10000 is 10^8, acceptable within 0.25s? Might be tight. Let me optimize by precomputing prefix balances.

```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    string s; getline(cin,s);
    while(!s.empty() && (s.back()=='\n'||s.back()=='\r'||s.back()==' ')) s.pop_back();
    int n=s.size();
    vector<int> pre(n+1,0);
    for(int i=0;i<n;i++) pre[i+1]=pre[i]+(s[i]=='('?1:-1);
    for(int i=n-1;i>=0;i--){
        if(s[i]=='('){
            int newBal=pre[i]-1;
            if(newBal<0) continue;
            int rem=n-i-1;
            if((rem-newBal)<0 || ((rem-newBal)&1)) continue;

            res+=')'; res+=string(a,'('); res+=string(b,')');
            cout<<res<<"\n";
            return 0;
        }
    }
    cout<<"No solution\n";
}
```
