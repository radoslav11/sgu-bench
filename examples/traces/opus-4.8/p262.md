Looking at this: we have up to N*M=100 pixels, K up to 6 symbols. We need minimal subset of pixels such that all K symbols have distinct "signatures" on that subset.

Each pair of symbols must differ in at least one chosen pixel. Pixels that differ between a pair form a set. We need to pick pixels covering all C(K,2) pairs, minimizing count. This is a set cover / minimum hitting set.

With up to 15 pairs, and we want minimum pixels. Each pixel covers some subset of pairs (the pairs where it differs). We need to cover all pairs with minimum number of pixels.

We can do this by trying sizes from 0 upward, or use the fact that pairs ≤ 15, so do a min set cover via BFS over pair-masks or iterative.

Let me just do iterative: try subsets by size. Number of distinct pixel "coverage masks" ≤ 2^15 but only up to 100 pixels. Min set cover over universe of 15 elements - we can do DP over masks: dp[mask] = min pixels to cover mask. Use the pixels' masks.

dp size 2^15 = 32768. For each, try each pixel. Fine.

```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    int N,M,K;
    cin>>N>>M>>K;
    vector<vector<string>> sym(K, vector<string>(N));
    for(int k=0;k<K;k++)for(int i=0;i<N;i++)cin>>sym[k][i];
    
    int P=N*M;
    int numPairs=K*(K-1)/2;
    int full=(1<<numPairs)-1;
    // for each pixel, mask of pairs it distinguishes
    vector<int> pmask(P,0);
    for(int p=0;p<P;p++){
        int i=p/M, j=p%M;
        int pidx=0;
        for(int a=0;a<K;a++)for(int b=a+1;b<K;b++){
            if(sym[a][i][j]!=sym[b][i][j]) pmask[p]|=(1<<pidx);
            pidx++;
        }
    }
    vector<int> dp(1<<numPairs, INT_MAX);
    vector<int> choice(1<<numPairs,-1);
    dp[0]=0;
    for(int mask=0;mask<(1<<numPairs);mask++){
        if(dp[mask]==INT_MAX)continue;
        for(int p=0;p<P;p++){
            int nm=mask|pmask[p];
            if(dp[mask]+1<dp[nm]){
                dp[nm]=dp[mask]+1;
                choice[nm]=p;
            }
        }
    }
    // reconstruct
    vector<int> used(P,0);
    int cur=full;
    while(cur!=0){
        int p=choice[cur];
        used[p]=1;
        cur^=(cur & pmask[p]); // remove pairs covered... careful
        // actually cur = previous mask, dp came from mask where mask|pmask[p]=cur
        // find previous: cur without pmask[p] contribution isn't unique; just use cur & ~pmask[p]
        // redo properly below
        break;
    }
    // proper reconstruction
    fill(used.begin(),used.end(),0);
    cur=full;
    while(cur!=0){
        int p=choice[cur];
        used[p]=1;
        // previous mask must satisfy prev|pmask[p]==cur and dp[prev]=dp[cur]-1
        // prev = cur & ~something; we set nm=mask|pmask[p]. So mask could be cur&~pmask[p] but need dp match
        int prev=cur & ~pmask[p];
        if(dp[prev]==dp[cur]-1){ cur=prev; }
        else {
            // search
            int best=-1;
            for(int sub=cur; ; sub=(sub-1)&cur){
                if((sub|pmask[p])==cur && dp[sub]==dp[cur]-1){best=sub;break;}
                if(sub==0)break;
            }
            cur=best;
        }
    }
    cout<<dp[full]<<"\n";
    for(int i=0;i<N;i++){
        for(int j=0;j<M;j++) cout<<used[i*M+j];
        cout<<"\n";
    }
}
```

The reconstruction simplification: since `prev = cur & ~pmask[p]` gives a valid predecessor and dp[prev] ≤ dp[cur]-1 always (it's covered), and dp[cur] was set from some mask. Let me just trust `prev = cur & ~pmask[p]`; dp[prev] is ≤ dp[cur]-1 but the choice stored ensures consistency. Actually it should be exactly dp[cur]-1 since adding p to prev gives cur with one more. Let me keep the fallback search for safety.
