#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<string> names;
vector<int> Len;

string enc(const vector<int>& v){
    string s;
    s.resize(v.size()*sizeof(int));
    memcpy(&s[0], v.data(), v.size()*sizeof(int));
    return s;
}

bool isGoal(const vector<int>& masks){
    int total = n + m;
    for(int i=0;i<total;i++){
        int len = Len[i];
        bool hasEnd = (masks[i] >> len) & 1;
        if(i < n){
            if(!hasEnd) return false; // positive must match
        } else {
            if(hasEnd) return false; // negative must not match
        }
    }
    return true;
}

// apply token, return false if invalid (some positive becomes empty)
bool applyToken(const vector<int>& cur, char tok, vector<int>& out){
    int total = n + m;
    out.assign(total, 0);
    for(int i=0;i<total;i++){
        int mask = cur[i];
        int len = Len[i];
        int nm = 0;
        if(tok=='*'){
            if(mask!=0){
                // find min set bit
                int mn = __builtin_ctz(mask);
                for(int p=mn;p<=len;p++) nm |= (1<<p);
            }
        } else if(tok=='?'){
            for(int p=0;p<len;p++){
                if((mask>>p)&1) nm |= (1<<(p+1));
            }
        } else {
            // literal
            for(int p=0;p<len;p++){
                if((mask>>p)&1){
                    if(names[i][p]==tok) nm |= (1<<(p+1));
                }
            }
        }
        out[i]=nm;
    }
    // prune: positives must stay non-empty
    for(int i=0;i<n;i++) if(out[i]==0) return false;
    return true;
}
