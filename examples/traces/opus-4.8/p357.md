Looking at this problem:

- We have buttons for digits 0-9, up, down, and "--".
- We start at channel X, want to reach Y.
- Up/down change channel by ±1 (with wraparound), each one click.
- Direct entry: single digit (0-9) -> one click if that digit works (gives channel 0-9). Two digit: "--" then two digits -> 3 clicks, requires "--" and both digit buttons work.

Let me compute the minimum cost to reach each channel directly (without up/down), then BFS/Dijkstra with up/down moves of cost 1.

Direct cost to a channel:
- channel 0-9: cost 1 if digit button works.
- channel 10-99: cost 3 if "--" works and both digit buttons work.

Starting point X has cost 0.

Then from any reachable channel, we can press up (cost1) if up works, down (cost1) if down works.

This is a shortest path. Let me do Dijkstra (or 0-1... but weights are uniform 1 except initial direct entries). Actually all edge costs are 1 for up/down, and direct entries are initial costs. Use Dijkstra with these initial distances.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int d[10]; // digit buttons 0-9
    int up, down, minus;
    // line1: 1 2 3 up
    cin >> d[1] >> d[2] >> d[3] >> up;
    // line2: 4 5 6 down
    cin >> d[4] >> d[5] >> d[6] >> down;
    // line3: 7 8 9
    cin >> d[7] >> d[8] >> d[9];
    // line4: -- 0
    cin >> minus >> d[0];
    int X, Y;
    cin >> X >> Y;
    
    const int INF = INT_MAX;
    vector<int> dist(100, INF);
    
    // initial costs
    dist[X] = 0;
    for(int ch=0; ch<100; ch++){
        int c = INF;
        if(ch <= 9){
            if(d[ch]) c = 1;
        } else {
            int hi = ch/10, lo = ch%10;
            if(minus && d[hi] && d[lo]) c = 3;
        }
        if(c < dist[ch]) dist[ch] = c;
    }
    
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
    for(int ch=0; ch<100; ch++){
        if(dist[ch] < INF) pq.push({dist[ch], ch});
    }
    
    while(!pq.empty()){
        auto [du, u] = pq.top(); pq.pop();
        if(du > dist[u]) continue;
        // up
        if(up){
            int v = (u+1)%100;
            if(du+1 < dist[v]){ dist[v] = du+1; pq.push({dist[v], v}); }
        }
        if(down){
            int v = (u+99)%100;
            if(du+1 < dist[v]){ dist[v] = du+1; pq.push({dist[v], v}); }
        }
    }
    
    if(dist[Y] == INF) cout << -1 << endl;
    else cout << dist[Y] << endl;
    
    return 0;
}
```
