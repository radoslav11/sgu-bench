The probability of exactly K correct matches = C(N,K) * D(N-K) / N!, where D is derangement count.

C(N,K) * D(N-K) / N! = D(N-K) / (K! * (N-K)!).

So answer = D(N-K) / (K! * (N-K)!) where D(m) = number of derangements of m elements.

D(0)=1, D(1)=0. Note when N-K=1, answer is 0.

I need big integers. Let me implement bignum.

```cpp
#include <bits/stdc++.h>
using namespace std;

struct Big {
    vector<int> d; // base 10^9, little endian
    static const int BASE = 1000000000;
    Big(long long v=0){ if(v==0) d.push_back(0); else while(v){d.push_back(v%BASE); v/=BASE;} if(d.empty())d.push_back(0);}
};

Big add(const Big&a,const Big&b){
    Big r; r.d.clear();
    long long carry=0;
    for(size_t i=0;i<a.d.size()||i<b.d.size()||carry;i++){
        long long s=carry;
        if(i<a.d.size())s+=a.d[i];
        if(i<b.d.size())s+=b.d[i];
        r.d.push_back(s%Big::BASE);
        carry=s/Big::BASE;
    }
    if(r.d.empty())r.d.push_back(0);
    return r;
}

Big mulSmall(const Big&a,long long m){
    Big r; r.d.clear();
    long long carry=0;
    for(size_t i=0;i<a.d.size()||carry;i++){
        long long s=carry;
        if(i<a.d.size())s+=(long long)a.d[i]*m;
        r.d.push_back(s%Big::BASE);
        carry=s/Big::BASE;
    }
    if(r.d.empty())r.d.push_back(0);
    return r;
}

Big mul(const Big&a,const Big&b){
    vector<long long> res(a.d.size()+b.d.size(),0);
    for(size_t i=0;i<a.d.size();i++){
        long long carry=0;
        for(size_t j=0;j<b.d.size()||carry;j++){
            long long cur=res[i+j]+carry+(j<b.d.size()?(long long)a.d[i]*b.d[j]:0);
            res[i+j]=cur%Big::BASE;
            carry=cur/Big::BASE;
        }
    }
    Big r; r.d.clear();
    for(auto x:res)r.d.push_back((int)x);
    while(r.d.size()>1&&r.d.back()==0)r.d.pop_back();
    if(r.d.empty())r.d.push_back(0);
    return r;
}

bool isZero(const Big&a){ return a.d.size()==1&&a.d[0]==0; }

string toStr(const Big&a){
    string s=to_string(a.d.back());
    for(int i=(int)a.d.size()-2;i>=0;i--){
        string t=to_string(a.d[i]);
        s+=string(9-t.size(),'0')+t;
    }
    return s;
}

// gcd for bignums - use string-based or repeated. Implement division.
Big divmodSmall(const Big&a,long long m,long long&rem){
    Big r; r.d.assign(a.d.size(),0);
    long long cur=0;
    for(int i=(int)a.d.size()-1;i>=0;i--){
        cur=cur*Big::BASE+a.d[i];
        r.d[i]=cur/m;
        cur%=m;
    }
    while(r.d.size()>1&&r.d.back()==0)r.d.pop_back();
    rem=cur;
    return r;
}

// compare
int cmp(const Big&a,const Big&b){
    if(a.d.size()!=b.d.size())return a.d.size()<b.d.size()?-1:1;
    for(int i=(int)a.d.size()-1;i>=0;i--)if(a.d[i]!=b.d[i])return a.d[i]<b.d[i]?-1:1;
    return 0;
}

Big sub(const Big&a,const Big&b){ // a>=b
    Big r; r.d.clear();
    long long borrow=0;
    for(size_t i=0;i<a.d.size();i++){
        long long s=a.d[i]-borrow-(i<b.d.size()?b.d[i]:0);
        if(s<0){s+=Big::BASE;borrow=1;}else borrow=0;
        r.d.push_back(s);
    }
    while(r.d.size()>1&&r.d.back()==0)r.d.pop_back();
    return r;
}

// full division a/b -> quotient, remainder via long division (b can be big)
pair<Big,Big> divmod(const Big&a,const Big&b){
    if(cmp(a,b)<0) return {Big(0),a};
    Big q; q.d.assign(a.d.size(),0);
    Big rem(0);
    for(int i=(int)a.d.size()-1;i>=0;i--){
        // rem = rem*BASE + a.d[i]
        rem=mulSmall(rem,Big::BASE);
        rem=add(rem,Big(a.d[i]));
        // find max x s.t. b*x <= rem
        long long lo=0,hi=Big::BASE-1,ans=0;
        while(lo<=hi){
            long long mid=(lo+hi)/2;
            Big t=mulSmall(b,mid);
            if(cmp(t,rem)<=0){ans=mid;lo=mid+1;}else hi=mid-1;
        }
        q.d[i]=ans;
        rem=sub(rem,mulSmall(b,ans));
    }
    while(q.d.size()>1&&q.d.back()==0)q.d.pop_back();
    return {q,rem};
}

Big gcd(Big a,Big b){
    while(!isZero(b)){
        Big r=divmod(a,b).second;
        a=b; b=r;
    }
    return a;
}

int main(){
    int K,N;
    cin>>K>>N;
    int m=N-K;
    // derangement D(m)
    Big D;
    if(m==0) D=Big(1);
    else if(m==1) D=Big(0);
    else{
        Big d0(1),d1(0);
        for(int i=2;i<=m;i++){
            // D(i) = (i-1)*(D(i-1)+D(i-2))
            Big t=add(d1,d0);
            Big di=mulSmall(t,i-1);
            d0=d1; d1=di;
        }
        D=d1;
    }
    if(isZero(D)){ cout<<"0\n"; return 0;}
    // denominator = K! * (N-K)!
    Big den(1);
    for(int i=2;i<=K;i++) den=mulSmall(den,i);
    for(int i=2;i<=m;i++) den=mulSmall(den,i);
    // reduce D/den
    Big g=gcd(D,den);
    Big A=divmod(D,g).first;
    Big B=divmod(den,g).first;
    if(cmp(B,Big(1))==0) cout<<toStr(A)<<"\n";
    else cout<<toStr(A)<<"/"<<toStr(B)<<"\n";
    return 0;
}
```
