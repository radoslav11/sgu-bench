Looking at this: classic broken profile / row-by-row DP with bitmask of king placements per row.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n,k;
    cin>>n>>k;
    
    // valid masks for a row (no two adjacent in same row)
    vector<int> masks;
    vector<int> cnt;
    for(int m=0;m<(1<<n);m++){
        if(m & (m<<1)) continue;
        masks.push_back(m);
        cnt.push_back(__builtin_popcount(m));
    }
    int M=masks.size();
    
    // dp[row][kings][maskIndex]
    // dp over rows
    // dp[kingsUsed][maskIndex]
    vector<vector<long long>> dp(k+1, vector<long long>(M,0));
    
    // first row
    for(int i=0;i<M;i++){
        if(cnt[i]<=k) dp[cnt[i]][i]+=1;
    }
    
    for(int row=1;row<n;row++){
        vector<vector<long long>> ndp(k+1, vector<long long>(M,0));
        for(int kk=0;kk<=k;kk++){
            for(int i=0;i<M;i++){
                if(dp[kk][i]==0) continue;
                long long val=dp[kk][i];
                int prev=masks[i];
                for(int j=0;j<M;j++){
                    int cur=masks[j];
                    // check no conflict between prev and cur
                    // adjacency: same column, or diagonal
                    if(cur & prev) continue;
                    if(cur & (prev<<1)) continue;
                    if(cur & (prev>>1)) continue;
                    int nk=kk+cnt[j];
                    if(nk>k) continue;
                    ndp[nk][j]+=val;
                }
            }
        }
        dp=move(ndp);
    }
    
    long long ans=0;
    for(int i=0;i<M;i++) ans+=dp[k][i];
    
    cout<<ans<<endl;
    return 0;
}
```
