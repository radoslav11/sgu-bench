Looking at this:

- Pages 1-2 are reserved (title/instructions).
- Records start on page 3.
- Each group of numbers with the same first digit starts on a new page.
- For each group, count how many pages needed = ceil(count/K).
- Total pages = 2 + sum of pages for each group.

Let me count distinct first digits and their counts, sum up ceil(count/K), add 2.

```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    int K, N;
    cin >> K >> N;
    int cnt[10] = {0};
    for(int i=0;i<N;i++){
        string s; cin >> s;
        cnt[s[0]-'0']++;
    }
    int pages = 2;
    for(int d=0;d<10;d++){
        if(cnt[d]>0){
            pages += (cnt[d]+K-1)/K;
        }
    }
    cout << pages << endl;
    return 0;
}
```
