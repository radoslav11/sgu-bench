#include <bits/stdc++.h>
using namespace std;
int M,N,A[10],B[10];
int main(){
    cin>>M>>N;
    for(int i=0;i<N;i++)cin>>A[i]>>B[i];
    // dp over minutes: state = last used minute per volunteer (cap). 
    // available array: nextfree[i]
    for(int T=0;;T++){
        // maximize balloons in T minutes via DP
        // state: vector of remaining rest (0..4) for each volunteer
        map<vector<int>,int> cur;
        cur[vector<int>(N,0)]=0;
        for(int t=0;t<T;t++){
            map<vector<int>,int> nxt;
            for(auto&pr:cur){
                vector<int> s=pr.first; int v=pr.second;
                // option: nobody
                vector<int> ns=s; for(int&x:ns)if(x>0)x--;
                auto it=nxt.find(ns); if(it==nxt.end()||it->second<v)nxt[ns]=v;
