Looking at this problem:

We have n points (oases). A caravan goes from oasis s to oasis t. It wants to minimize the maximal segment length, but the path is a polyline whose vertices are at oases (since they need to be at oases to be safe). So the path goes through a sequence of oases, and we minimize the maximum edge length along the path.

This is a minimax path / bottleneck shortest path problem. The answer for each query (s,t) is the minimum over all paths from s to t of the maximum edge length on that path. The actual answer output is that bottleneck value (the length of the maximal segment of the optimal path).

Let me verify with the example:
- 3 points: (0,0),(50,10),(150,0)
- Query 1→2: direct distance = sqrt(2500+100)=sqrt(2600)=50.990... matches.
- Query 1→3: direct distance = 150. But answer is 100.498. Path 1→2→3: edge 1-2=50.99, edge 2-3 = sqrt(100²+10²)=sqrt(10100)=100.498. Max = 100.498 < 150. So bottleneck path gives 100.498. Matches!
- Query 2→3: 100.498, matches.

So this is the minimax (bottleneck) path problem. The complete graph has C(n,2) edges. With n possibly up to ~some thousands, we need an efficient approach. The bottleneck path equals the maximum edge on the path in the minimum spanning tree between s and t.

So: build MST (Euclidean), then for each query find max edge weight on path between s,t in the tree.

With many queries, use binary lifting or just answer with tree path queries. Let me use binary lifting for LCA with max edge.

Since n unknown bounds but complete graph - building MST on complete Euclidean graph. If n is small (constraints cut off), O(n²) MST (Prim) is fine. Let me use Prim's O(n²) which handles dense complete graph.

Then build tree, do binary lifting for max-edge queries.

Let me write the code.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n;
    if(scanf("%d",&n)!=1) return 0;
    vector<long long> x(n), y(n);
    for(int i=0;i<n;i++) scanf("%lld %lld",&x[i],&y[i]);
    
    // Prim's MST O(n^2) on complete Euclidean graph (squared distances for comparison)
    vector<double> minDist(n, 1e30);
    vector<int> parent(n, -1);
    vector<char> inMST(n,0);
    minDist[0]=0;
    // store tree edges
    vector<vector<pair<int,double>>> adj(n);
    for(int it=0; it<n; it++){
        int u=-1; double best=1e30;
        for(int v=0;v<n;v++) if(!inMST[v] && minDist[v]<best){best=minDist[v];u=v;}
        if(u==-1) break;
        inMST[u]=1;
        if(parent[u]!=-1){
            double w = sqrt((double)((x[u]-x[parent[u]])*(x[u]-x[parent[u]]) + (y[u]-y[parent[u]])*(y[u]-y[parent[u]])));
            adj[u].push_back({parent[u], w});
            adj[parent[u]].push_back({u, w});
        }
        for(int v=0;v<n;v++) if(!inMST[v]){
            double d = (double)((x[u]-x[v])*(x[u]-x[v]) + (y[u]-y[v])*(y[u]-y[v]));
            if(d < minDist[v]){ minDist[v]=d; parent[v]=u; }
        }
    }
    
    // Binary lifting for max edge on path
    int LOG=1;
    while((1<<LOG) < n) LOG++;
    LOG++;
    vector<vector<int>> up(LOG, vector<int>(n, -1));
    vector<vector<double>> mx(LOG, vector<double>(n, 0.0));
    vector<int> depth(n, 0);
    
    // BFS/DFS from node 0
    vector<int> order;
    vector<char> vis(n,0);
    // iterative DFS
    {
        stack<int> st;
        st.push(0); vis[0]=1; up[0][0]=-1; depth[0]=0;
        order.push_back(0);
        while(!st.empty()){
            int u=st.top(); st.pop();
            for(auto &e: adj[u]){
                int v=e.first;
                if(!vis[v]){
                    vis[v]=1;
                    up[0][v]=u;
                    mx[0][v]=e.second;
                    depth[v]=depth[u]+1;
                    st.push(v);
                }
            }
        }
    }
    
    for(int k=1;k<LOG;k++){
        for(int v=0;v<n;v++){
            int mid=up[k-1][v];
            if(mid==-1){ up[k][v]=-1; mx[k][v]=mx[k-1][v]; }
            else {
                up[k][v]=up[k-1][mid];
                mx[k][v]=max(mx[k-1][v], mx[k-1][mid]);
            }
        }
    }
    
    int q;
    scanf("%d",&q);
    while(q--){
        int s,t;
        scanf("%d %d",&s,&t);
        s--; t--;
        if(s==t){ printf("%.10f\n", 0.0); continue; }
        double ans=0;
        int a=s,b=t;
        if(depth[a]<depth[b]) swap(a,b);
        int diff=depth[a]-depth[b];
        for(int k=0;k<LOG;k++){
            if(diff & (1<<k)){
                ans=max(ans, mx[k][a]);
                a=up[k][a];
            }
        }
        if(a!=b){
            for(int k=LOG-1;k>=0;k--){
                if(up[k][a]!=up[k][b]){
                    ans=max(ans, mx[k][a]);
                    ans=max(ans, mx[k][b]);
                    a=up[k][a];
                    b=up[k][b];
                }
            }
            ans=max(ans, mx[0][a]);
            ans=max(ans, mx[0][b]);
        }
        printf("%.10f\n", ans);
    }
    
    return 0;
}
```
