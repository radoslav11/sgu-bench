The largest circle inside a convex polygon: maximize the minimum distance from a point to all edges. Binary search on radius, shrink each edge inward by radius, check if intersection is non-empty.

```cpp
#include <bits/stdc++.h>
using namespace std;

int N;
vector<double> X, Y;

// half-plane intersection feasibility: each edge shifted inward by r
// edge from i to i+1, ccw, interior is to the left
// check if intersection of half-planes is non-empty

struct Line { double a,b,c; }; // a*x+b*y<=c (inside)

bool feasible(double r){
    // for each edge, normal pointing inward (left side)
    // edge direction d=(dx,dy), inward normal = (-dy,dx) normalized
    // points inside: dot(normal, p - vertex) >= 0
    // shifted by r: dot(normal,p-vertex) >= r
    // i.e. -normal . p <= -normal.vertex - r
    vector<array<double,3>> H; // a,b,c : a*x+b*y<=c
    for(int i=0;i<N;i++){
        int j=(i+1)%N;
        double dx=X[j]-X[i], dy=Y[j]-Y[i];
        double len=sqrt(dx*dx+dy*dy);
        double nx=-dy/len, ny=dx/len; // inward normal
        // nx*(x-X[i])+ny*(y-Y[i]) >= r
        // -nx*x - ny*y <= -nx*X[i]-ny*Y[i] - r
        H.push_back({-nx,-ny, -nx*X[i]-ny*Y[i]-r});
    }
    // half-plane intersection non-empty? Use LP feasibility via incremental.
    // Simple approach: clip a large box polygon by each half-plane.
    vector<pair<double,double>> poly;
    double B=2e7;
    poly={{-B,-B},{B,-B},{B,B},{-B,B}};
    for(auto&h:H){
        double a=h[0],b=h[1],c=h[2];
        vector<pair<double,double>> np;
        int m=poly.size();
        for(int i=0;i<m;i++){
            auto&P=poly[i]; auto&Q=poly[(i+1)%m];
            double dp=a*P.first+b*P.second-c;
            double dq=a*Q.first+b*Q.second-c;
            if(dp<=1e-12) np.push_back(P);
            if((dp<-1e-12&&dq>1e-12)||(dp>1e-12&&dq<-1e-12)){
                double t=dp/(dp-dq);
                np.push_back({P.first+t*(Q.first-P.first),P.second+t*(Q.second-P.second)});
            }
        }
        poly=np;
        if(poly.empty()) return false;
    }
    return !poly.empty();
}

int main(){
    scanf("%d",&N);
    X.resize(N);Y.resize(N);
    for(int i=0;i<N;i++) scanf("%lf %lf",&X[i],&Y[i]);
    double lo=0, hi=2e7;
    for(int it=0;it<100;it++){
        double mid=(lo+hi)/2;
        if(feasible(mid)) lo=mid; else hi=mid;
    }
    printf("%.5f\n",lo);
    return 0;
}
```
