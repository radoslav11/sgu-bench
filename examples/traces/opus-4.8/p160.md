DP over residues mod M.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int N, M;
    scanf("%d %d", &N, &M);
    vector<int> a(N);
    for(int i=0;i<N;i++) scanf("%d",&a[i]);
    
    // dp[r] = best (achievable) flag; we track for each residue the way to reach it
    // process levers one by one. For each residue store predecessor info.
    
    const int INF = -1;
    vector<int> reach(M, INF); // step at which reached, store nothing
    // We'll store for each residue: which lever used and previous residue
    vector<int> prevRes(M, -1), leverUsed(M, -1), stepReached(M, -1);
    
    reach[1] = 0;
    
    for(int i=0;i<N;i++){
        int x = a[i] % M;
        // we must iterate over current reachable residues, but to avoid using same lever twice,
        // we only allow transitions from residues reached before this lever.
        // collect current snapshot
        vector<pair<int,int>> updates; // (newRes, fromRes)
        for(int r=0;r<M;r++){
            if(reach[r]==INF) continue;
            int nr = (r * x) % M;
            if(reach[nr]==INF){
                updates.push_back({nr, r});
            }
        }
        for(auto &u : updates){
            int nr=u.first, r=u.second;
            if(reach[nr]==INF){
                reach[nr]=0;
                prevRes[nr]=r;
                leverUsed[nr]=i;
                stepReached[nr]=i;
            }
        }
    }
    
    // find max reachable residue
    int best=-1;
    for(int r=M-1;r>=0;r--){
        if(reach[r]!=INF){ best=r; break; }
    }
    
    // reconstruct
    vector<int> levers;
    int cur=best;
    while(cur!=1 || leverUsed[cur]!=-1){
        if(leverUsed[cur]==-1) break;
        levers.push_back(leverUsed[cur]+1);
        cur=prevRes[cur];
    }
    sort(levers.begin(), levers.end());
    
    printf("%d\n", best);
    for(int i=0;i<(int)levers.size();i++){
        printf("%d%c", levers[i], i+1<(int)levers.size()?' ':'\n');
    }
    if(levers.empty()) printf("\n");
    return 0;
}
```
