Sort by A, then count how many have a B less than the maximum B seen so far.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n;
    scanf("%d",&n);
    vector<pair<long long,long long>> v(n);
    for(int i=0;i<n;i++) scanf("%lld %lld",&v[i].first,&v[i].second);
    sort(v.begin(),v.end());
    long long maxB=-1;
    int count=0;
    for(int i=0;i<n;i++){
        if(v[i].second<maxB) count++;
        else maxB=v[i].second;
    }
    printf("%d\n",count);
    return 0;
}
```
