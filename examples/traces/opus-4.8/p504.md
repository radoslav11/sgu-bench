Looking at this problem:

We need to find the largest square where every row and every column (within the square) is a palindrome.

Let me think. For a square from (r1,c1) to (r2,c2) of size n, every row substring must be a palindrome and every column substring must be a palindrome.

A row palindrome of the square at row i, columns [c1,c2]: needs grid[i][c1+k]==grid[i][c2-k]. A column palindrome at column j, rows [r1,r2]: grid[r1+k][j]==grid[r2-k][j].

Key approach: For a fixed size n, define horizontal palindrome - position (i,j) is "good horizontal start" if the row substring starting at column j of length n is palindrome. Similarly vertical.

For the square to work: all rows i in [r1,r2] must have row-palindrome at column c1, and all columns j in [c1,c2] must have column-palindrome at row r1.

So for size n, we need: a column c1 such that for n consecutive rows, the horizontal palindrome of length n at that column holds, AND a row r1 such that for n consecutive columns starting at c1, vertical palindrome of length n holds.

Let H[i][j] = 1 if row i, columns [j, j+n-1] is palindrome. We need a vertical run of n ones in column j of H, i.e. H[i..i+n-1][j] all 1. Let V[i][j]=1 if column j rows [i,i+n-1] palindrome; need horizontal run of n ones in row i: V[i][j..j+n-1] all 1.

The square at (r1,c1): rows r1..r1+n-1 all H-good at c1, AND columns c1..c1+n-1 all V-good at r1. These share same corner. So condition: there's (i,j) with vertical-run-of-n in H column j ending properly (rows i..i+n-1) and horizontal-run-of-n in V row i (cols j..j+n-1).

To compute H for a given n efficiently for all positions: use precomputed palindrome info. We can compute for each (row,j) the maximum palindrome length... Actually use hashing or Manacher per row to know if substring [j,j+n-1] is palindrome — center based. For odd/even, substring of length n centered appropriately. Use Manacher per row to get radius, then check.

We binary search on n? Largest n — but validity isn't monotonic necessarily... Actually if size n square works, does n-2 work? Inner square of palindromic rows/cols is also palindromic. Yes! Removing outer layer keeps palindromes. So parity-monotonic. Binary search separately for odd and even n.

Let me precompute Manacher per row and per column to answer "is row i cols [j,j+n-1] palindrome" in O(1). Then for given n build H and V arrays (700x700) and check runs - O(h*w) per check. Binary search ~log(700) for each parity → fine.

struct Man{ // for a string, answer pal(l,r)
  vector<int> d1,d2; int n;
  void build(const string&s){ n=s.size(); d1.assign(n,0); d2.assign(n,0);

  }

using namespace std;
int h,w; vector<string> g;
struct Man{ vector<int> d1,d2; int n;
  void build(const string&s){ n=s.size(); d1.assign(n,0); d2.assign(n,0);
    for(int i=0,l=0,r=-1;i<n;i++){int k=(i>r)?1:min(d1[l+r-i],r-i+1);
      while(i-k>=0&&i+k<n&&s[i-k]==s[i+k])k++; d1[i]=k; if(i+k-1>r){l=i-k+1;r=i+k-1;}}
    for(int i=0,l=0,r=-1;i<n;i++){int k=(i>r)?0:min(d2[l+r-i+1],r-i+1);
      while(i-k-1>=0&&i+k<n&&s[i-k-1]==s[i+k])k++; d2[i]=k; if(i+k-1>r){l=i-k;r=i+k-1;}}
  }
  bool pal(int l,int r){ int len=r-l+1; int c=(l+r)/2;
    if(len&1) return d1[c]>=len/2+1; else return d2[c+1]>=len/2; }
};
vector<Man> rowM, colM;
int ar1,ac1,an;
bool check(int n){
  // H[i][j]: row i cols [j,j+n-1] pal ; need col run; V[i][j]: col j rows [i,i+n-1]; row run
  static vector<vector<char>> H,V; H.assign(h,vector<char>(w,0)); V.assign(h,vector<char>(w,0));
  for(int i=0;i<h;i++)for(int j=0;j+n<=w;j++) H[i][j]=rowM[i].pal(j,j+n-1);
  for(int j=0;j<w;j++)for(int i=0;i+n<=h;i++) V[i][j]=colM[j].pal(i,i+n-1);
  // vertical run in H column: vrun[i][j] = run of 1s downward
  static vector<vector<int>> vr,hr; vr.assign(h+1,vector<int>(w,0)); hr.assign(h,vector<int>(w+1,0));
  for(int j=0;j<w;j++)for(int i=h-1;i>=0;i--) vr[i][j]=H[i][j]?vr[i+1][j]+1:0;
  for(int i=0;i<h;i++)for(int j=w-1;j>=0;j--) hr[i][j]=V[i][j]?hr[i][j+1]+1:0;
  for(int i=0;i+n<=h;i++)for(int j=0;j+n<=w;j++) if(vr[i][j]>=n&&hr[i][j]>=n){ar1=i;ac1=j;an=n;return true;}
  return false;
}

  rowM.resize(h); for(int i=0;i<h;i++)rowM[i].build(g[i]);
  colM.resize(w); for(int j=0;j<w;j++){string c(h,' ');for(int i=0;i<h;i++)c[i]=g[i][j];colM[j].build(c);}
  int best=1,br=0,bc=0; // n=1 always works
  // odd
  {int lo=1,hi=min(h,w),res=1,rr=0,rc=0; while(lo<=hi){int mid=(lo+hi)/2; if(mid%2==0)mid--; if(mid<lo){lo=mid+2;continue;} if(check(mid)){res=mid;rr=ar1;rc=ac1;lo=mid+2;}else hi=mid-1;} if(res>best){best=res;br=rr;bc=rc;}}
  {int lo=2,hi=min(h,w),res=0,rr=0,rc=0; while(lo<=hi){int mid=(lo+hi)/2; if(mid%2==1)mid--; if(mid<lo){lo=mid+2;continue;} if(check(mid)){res=mid;rr=ar1;rc=ac1;lo=mid+2;}else hi=mid-1;} if(res>best){best=res;br=rr;bc=rc;}}

}
