Sort by key, then build treap using a stack based on auxiliary key (min-heap).

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int n;
    scanf("%d",&n);
    vector<int> k(n+1),a(n+1),idx(n+1);
    for(int i=1;i<=n;i++){
        scanf("%d %d",&k[i],&a[i]);
        idx[i]=i;
    }
    sort(idx.begin()+1,idx.end(),[&](int x,int y){return k[x]<k[y];});
    vector<int> par(n+1,0),lc(n+1,0),rc(n+1,0);
    // build cartesian tree on sorted by key, min-heap on a
    vector<int> stk;
    for(int i=1;i<=n;i++){
        int cur=idx[i];
        int last=0;
        while(!stk.empty() && a[stk.back()]>a[cur]){
            last=stk.back();
            stk.pop_back();
        }
        // last is right child of cur
        if(last){
            lc[cur]=last;
            par[last]=cur;
        }
        if(!stk.empty()){
            // cur is right child of stk.back()
            rc[stk.back()]=cur;
            par[cur]=stk.back();
        }
        stk.push_back(cur);
    }
    printf("YES\n");
    for(int i=1;i<=n;i++){
        printf("%d %d %d\n",par[i],lc[i],rc[i]);
    }
    return 0;
}
```
