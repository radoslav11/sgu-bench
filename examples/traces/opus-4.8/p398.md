```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,x;
    cin>>n>>x;
    vector<set<int>> adj(n+1);
    for(int i=1;i<=n;i++){
        int d;cin>>d;
        for(int j=0;j<d;j++){int f;cin>>f;adj[i].insert(f);}
    }
    set<int> fof;
    for(int b:adj[x]){
        for(int c:adj[b]){
            if(c!=x && !adj[x].count(c)) fof.insert(c);
        }
    }
    cout<<fof.size()<<"\n";
    bool first=true;
    for(int c:fof){
        if(!first)cout<<" ";
        cout<<c;first=false;
    }
    cout<<"\n";
    return 0;
}
```
