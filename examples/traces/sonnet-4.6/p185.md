Looking at this problem, I need to find two shortest paths from node 1 to node N that don't share any edge (but can share vertices).

**Approach:**
1. Find the shortest path distance from node 1 to all nodes (dist_from_1) and from node N to all nodes (dist_from_N) using Dijkstra.
2. An edge (u,v,w) is on *some* shortest path if dist_from_1[u] + w + dist_from_N[v] == shortest_total OR dist_from_1[v] + w + dist_from_N[u] == shortest_total.
3. Build a directed graph of edges that lie on shortest paths.
4. Find two edge-disjoint paths in this DAG using max flow (capacity 1 on each edge).
5. Reconstruct both paths.

The flow network: source=1, sink=N, each edge on a shortest path gets capacity 1 (directed appropriately). We need flow of 2.

For reconstruction, after running flow, trace paths by following edges with used flow.

Let me implement this:

- Build directed edges on shortest paths with capacity 1
- Run max flow, check if we get flow = 2
- Reconstruct two paths

For path reconstruction after flow: follow edges from node 1 where flow was used (residual capacity decreased), greedily find two paths.

#include<bits/stdc++.h>
using namespace std;
const long long INF=1e18;
// Dinic's max flow
struct Edge{int to,rev;int cap;};
vector<Edge>graph[805];
int level[805],iter[805];
void add_edge(int from,int to,int cap){
    graph[from].push_back({to,(int)graph[to].size(),cap});
    graph[to].push_back({from,(int)graph[from].size()-1,0});
}
bool bfs(int s,int t,int n){
    fill(level,level+n,-1);level[s]=0;
    queue<int>q;q.push(s);
    while(!q.empty()){int v=q.front();q.pop();for(auto&e:graph[v])if(e.cap>0&&level[e.to]<0){level[e.to]=level[v]+1;q.push(e.to);}}
    return level[t]>=0;
}
int dfs(int v,int t,int f){
    if(v==t)return f;
    for(int&i=iter[v];i<graph[v].size();i++){
        Edge&e=graph[v][i];
        if(e.cap>0&&level[v]<level[e.to]){int d=dfs(e.to,t,min(f,e.cap));if(d>0){e.cap-=d;graph[e.to][e.rev].cap+=d;return d;}}
    }return 0;
}
int max_flow(int s,int t,int n){
    int flow=0;
    while(bfs(s,t,n)){fill(iter,iter+n,0);int d;while((d=dfs(s,t,1e9))>0)flow+=d;}
    return flow;
}
int main(){
    int N,M;scanf("%d%d",&N,&M);
    vector<tuple<int,int,int>>edges(M);
    vector<vector<pair<int,int>>>adj(N+1);
    for(int i=0;i<M;i++){int x,y,l;scanf("%d%d%d",&x,&y,&l);edges[i]={x,y,l};adj[x].push_back({y,l});adj[y].push_back({x,l});}
    auto dijk=[&](int src){
        vector<long long>d(N+1,INF);d[src]=0;
        priority_queue<pair<long long,int>,vector<pair<long long,int>>,greater<>>pq;pq.push({0,src});
        while(!pq.empty()){auto[dd,u]=pq.top();pq.pop();if(dd>d[u])continue;for(auto[v,w]:adj[u])if(d[u]+w<d[v]){d[v]=d[u]+w;pq.push({d[v],v});}}
        return d;
    };
    auto d1=dijk(1),dN=dijk(N);
    if(dN[1]==INF){puts("No solution");return 0;}
    long long best=dN[1];
    // add edges on shortest paths
    for(auto[x,y,l]:edges){
        if(d1[x]+l+dN[y]==best)add_edge(x,y,1);
        if(d1[y]+l+dN[x]==best)add_edge(y,x,1);
    }
    int fl=max_flow(1,N,N+1);
    if(fl<2){puts("No solution");return 0;}
    // reconstruct two paths
    for(int p=0;p<2;p++){
        vector<int>path;int cur=1;path.push_back(cur);
        while(cur!=N){for(auto&e:graph[cur]){if(e.to>=1&&e.to<=N&&graph[e.to][e.rev].cap>0&&e.cap==0){// used edge
            graph[e.to][e.rev].cap--;cur=e.to;path.push_back(cur);break;}}}
        for(int i=0;i<path.size();i++)printf("%d%c",path[i]," \n"[i+1==path.size()]);
    }
}
