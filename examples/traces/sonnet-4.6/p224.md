Looking at this problem, I need to count the number of ways to place k queens on an n×n board such that no two queens attack each other.

With n ≤ 10 and k ≤ n², I can use backtracking.

I'll place queens one by one, iterating through cells in row-major order to avoid duplicates. For each queen placement, I track which rows, columns, and diagonals are occupied.

Let me implement this with backtracking:
- Place queens cell by cell (in order to avoid counting same configurations multiple times)
- Track which rows, columns, main diagonals (row-col), and anti-diagonals (row+col) are used

```cpp
#include <bits/stdc++.h>
using namespace std;

int n, k;
long long ans = 0;

// rows, cols, diag1 (r-c), diag2 (r+c)
bool row_used[10], col_used[10], diag1_used[20], diag2_used[20];

void solve(int placed, int start_pos) {
    if (placed == k) {
        ans++;
        return;
    }
    // cells remaining from start_pos to n*n-1
    int remaining = n * n - start_pos;
    int need = k - placed;
    if (remaining < need) return;
    
    for (int pos = start_pos; pos < n * n; pos++) {
        int r = pos / n;
        int c = pos % n;
        if (!row_used[r] && !col_used[c] && !diag1_used[r - c + n - 1] && !diag2_used[r + c]) {
            row_used[r] = true;
            col_used[c] = true;
            diag1_used[r - c + n - 1] = true;
            diag2_used[r + c] = true;
            solve(placed + 1, pos + 1);
            row_used[r] = false;
            col_used[c] = false;
            diag1_used[r - c + n - 1] = false;
            diag2_used[r + c] = false;
        }
    }
}
```
