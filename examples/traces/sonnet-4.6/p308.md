```cpp
#include <bits/stdc++.h>
using namespace std;
typedef array<double,4> St;

// state: [t, phi, dt/ds, dphi/ds]
// Geodesic eqs from Christoffel symbols with E=cosh(2t), F=0, G=cosh^2(t)
St deriv(St s){
    double t=s[0],p=s[1],dt=s[2],dp=s[3];
    double sh=sinh(t),ch=cosh(t);
    // Christoffel: Gamma^t_tt = sh*ch/cosh(2t)... careful
    // E=cosh(2t), G=ch^2
    double E=cosh(2*t), G=ch*ch;
    double dE=2*sinh(2*t), dG=2*sh*ch;
    // d2t/ds2 = -Gtt_t*dt^2 - Gtt_phi... 
    // Gamma^t_tt = dE/(2E)*0.5 = E_t/(2E)
    // Gamma^t_tt = dE/(2E), Gamma^t_pp = -dG/(2E), Gamma^p_tp = dG/(2G)
    double Gam_t_tt = dE/(2*E);
    double Gam_t_pp = -dG/(2*E);
    double Gam_p_tp = dG/(2*G);
    double d2t = -Gam_t_tt*dt*dt - Gam_t_pp*dp*dp;
    double d2p = -2*Gam_p_tp*dt*dp;
    return {dt,dp,d2t,d2p};
}

St rk4(St s, double h){
    auto k1=deriv(s);
    St s2; for(int i=0;i<4;i++) s2[i]=s[i]+h/2*k1[i];
    auto k2=deriv(s2);
    St s3; for(int i=0;i<4;i++) s3[i]=s[i]+h/2*k2[i];
    auto k3=deriv(s3);
    St s4; for(int i=0;i<4;i++) s4[i]=s[i]+h*k3[i];
    auto k4=deriv(s4);
    St r; for(int i=0;i<4;i++) r[i]=s[i]+h/6*(k1[i]+2*k2[i]+2*k3[i]+k4[i]);
    return r;
}

int main(){
    double xa,ya,za,xb,yb,zb;
    scanf("%lf%lf%lf%lf%lf%lf",&xa,&ya,&za,&xb,&yb,&zb);
    double ta=asinh(za), pa=atan2(ya,xa);
    double tb=asinh(zb), pb=atan2(yb,xb);
    double dphi=pb-pa;
    while(dphi>M_PI) dphi-=2*M_PI;
    while(dphi<-M_PI) dphi+=2*M_PI;
    // Shoot from (ta,0) with angle alpha in tangent space
    // dt/ds = cos(alpha)/sqrt(E), dphi/ds = sin(alpha)/sqrt(G)
    double best=1e18;
    int steps=5000;
    double maxs=20.0, h=maxs/steps;
    for(int ai=0;ai<=1000;ai++){
        double alpha=M_PI*ai/1000.0;
        for(int sg : {1,-1}){
            double E0=cosh(2*ta), G0=cosh(ta)*cosh(ta);
            double dts=cos(alpha)/sqrt(E0), dps=sg*sin(alpha)/sqrt(G0);
            St s={ta,0.0,dts,dps};
            for(int i=0;i<steps;i++){
                St ns=rk4(s,h);
                double ds=(i+1)*h;
                double pp=ns[1];
                while(pp>M_PI) pp-=2*M_PI;
                while(pp<-M_PI) pp+=2*M_PI;
                double dist2=(ns[0]-tb)*(ns[0]-tb)+(pp-dphi)*(pp-dphi);
                if(dist2<0.01) best=min(best,ds);
                s=ns;
            }
        }
    }
    printf("%.10f\n",best);
}
```
