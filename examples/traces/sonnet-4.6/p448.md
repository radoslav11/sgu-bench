```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int N, M;
int R[16][16];
// wins[mask] = array of size N, wins[mask][p] = number of arrangements where p wins
// But 2^16 * 16 might be large... 65536 * 16 = ~1M, fine
ll wins[1<<16][16];
bool done[1<<16];
int sz[1<<16];

int ht(int n) {
    if(n <= 1) return 0;
    int h = 0;
    while((1<<h) < n) h++;
    return h;
}

void solve(int mask) {
    if(done[mask]) return;
    done[mask] = true;
    int n = sz[mask];
    if(n == 1) {
        // find the single player
        for(int i = 0; i < N; i++)
            if((mask>>i)&1) { wins[mask][i] = 1; return; }
    }
    int h = ht(n);
    int mx = 1<<(h-1);
    int mn = n - mx;
    // enumerate all subsets of mask as potential left half
    for(int sub = (mask-1)&mask; sub > 0; sub = (sub-1)&mask) {
        int co = mask ^ sub;
        int ls = sz[sub], rs = sz[co];
        if(ls < mn || ls > mx) continue;
        if(rs < mn || rs > mx) continue;
        solve(sub);
        solve(co);
        // sub is left, co is right
        for(int p = 0; p < N; p++) {
            if(!((mask>>p)&1)) continue;
            if((sub>>p)&1) {
                // p wins left, p must beat winner of right
                ll right_sum = 0;
                for(int q = 0; q < N; q++) {
                    if((co>>q)&1 && R[p][q]) {
                        right_sum += wins[co][q];
                    }
                }
                wins[mask][p] += wins[sub][p] * right_sum;
            } else {
                // p in co (right), p wins right, must beat winner of left
                ll left_sum = 0;
                for(int q = 0; q < N; q++) {
                    if((sub>>q)&1 && R[p][q]) {
                        left_sum += wins[sub][q];
                    }
                }
                wins[mask][p] += wins[co][p] * left_sum;
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cin >> N >> M;
    M--; // 0-indexed
    for(int i = 0; i < N; i++)
        for(int j = 0; j < N; j++)
            cin >> R[i][j];
    for(int i = 0; i < (1<<N); i++)
        sz[i] = __builtin_popcount(i);
    int full = (1<<N)-1;
    solve(full);
    cout << wins[full][M] << endl;
}
```
