For each operation (P, Q) applied in reverse: if the current position j is within [P, Q], it gets mapped to P + Q - j.

#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N, M;
    // Handle multiple test cases in input
    // Read until EOF
    while(cin >> N >> M){
        vector<int> P(M), Q(M);
        for(int i = 0; i < M; i++){
            cin >> P[i] >> Q[i];
        }
        
        // For each final position j (1-indexed), trace back through operations
        // to find original position
        for(int j = 1; j <= N; j++){
            int pos = j;
            for(int op = M-1; op >= 0; op--){
                if(pos >= P[op] && pos <= Q[op]){
                    pos = P[op] + Q[op] - pos;
                }
            }
            if(j > 1) cout << ' ';
            cout << pos;
        }
        cout << '\n';
    }
    return 0;
}
