#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<int> tree[105];
int sz[105], color[105];
int country_cnt;

void calc_size(int v, int par) {
    sz[v] = 1;
    for (int u : tree[v]) {
        if (u != par) {
            calc_size(u, v);
            sz[v] += sz[u];
        }
    }
}

void assign_color(int v, int par, int col) {
    color[v] = col;
    for (int u : tree[v]) {
        if (u != par) {
            if (sz[u] % 2 == 0) {
                country_cnt++;
                assign_color(u, v, country_cnt);
            } else {
                assign_color(u, v, col);
            }
        }
    }
}
