bool Q[20][20];

// Rotate (r,c) by rot (0=0,1=90cw,2=180,3=270cw) in NxN grid
pair<int,int> rotate_cell(int r, int c, int rot) {
    for(int i=0;i<rot;i++) {
        int nr = c, nc = N-1-r;
        r=nr; c=nc;
    }
    return {r,c};
}
