using namespace std;

int n, m;
char board[15][15];
int kr, kc; // initial king position
vector<pair<int,int>> pieces; // black pieces positions
int np; // number of black pieces

bool inBounds(int r, int c) { return r>=0&&r<n&&c>=0&&c<m; }

// Compute attacked squares given king at (kr,kc) and black pieces present = mask
// Returns set of attacked squares as bool array
void computeAttacked(int kingr, int kingc, int mask, vector<vector<bool>>& attacked) {
    attacked.assign(n, vector<bool>(m, false));
    // For each piece in mask
    for (int i = 0; i < np; i++) {
        if (!(mask & (1<<i))) continue;
        int pr = pieces[i].first, pc = pieces[i].second;
        char type = board[pr][pc];
        if (type == 'K') { // knight
            int dr[]={-2,-2,-1,-1,1,1,2,2}, dc[]={-1,1,-2,2,-2,2,-1,1};

            for (int d=0;d<4;d++) {

                    attacked[r][c]=true;

                    if (blocked) break;
                    r+=dirs[d][0]; c+=dirs[d][1];

    }
