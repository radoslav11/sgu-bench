using namespace std;
typedef long long ll;
typedef pair<ll,int> pli;

int N, M, H, S, E;
struct Edge { int to, T; ll R, P; };
vector<Edge> adj[105];
int exits[105];
bool isExit[105];

// Returns min time to reach each node using only edges with temp <= threshold
// Also fills parent for path reconstruction
vector<ll> dijkstra(ll threshold, vector<int>& parent) {
    vector<ll> dist(N+1, LLONG_MAX);
    parent.assign(N+1, -1);
    priority_queue<pli, vector<pli>, greater<pli>> pq;
    dist[S] = 0;
    pq.push({0, S});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        for (auto& e : adj[u]) {
            ll temp = e.R + e.P * (d + e.T);
            if (temp <= threshold) {
                ll nd = d + e.T;
                if (nd < dist[e.to]) {
                    dist[e.to] = nd;
                    parent[e.to] = u;
                    pq.push({nd, e.to});
                }
            }
        }
    }
    return dist;
}
