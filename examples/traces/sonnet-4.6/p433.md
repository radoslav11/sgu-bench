Looking at this problem, I need to tile an N×M rectangle with bricks of sizes 1×L1, 1×L2, ..., 1×LK (bricks can be placed horizontally or vertically).

The output requires adjacent cells of same brick have same letter, adjacent cells of different bricks have different letters. This is essentially a proper coloring constraint on bricks where adjacent bricks get different letters.

**Approach:**
1. Use backtracking: fill cells left-to-right, top-to-bottom
2. At each unfilled cell, try placing each brick type in horizontal or vertical orientation
3. After filling, assign letters to bricks such that adjacent bricks have different letters (graph coloring with 26 colors - should be easy)

**Implementation:**
- `grid[i][j]` = brick ID assigned to cell (i,j), -1 if unfilled
- Fill the first empty cell by trying all brick placements
- After complete fill, build adjacency graph of bricks and color it with 26 letters

Let me think about the constraints: M+N ≤ 20, so max grid is about 19×1 or 10×10 = 100 cells. Backtracking should work.

For coloring: each brick gets a letter. Adjacent bricks (sharing an edge) must have different letters. With 26 colors and likely small number of bricks, greedy coloring works.

```cpp
#include <bits/stdc++.h>
using namespace std;

int N, M, K;
vector<int> L;
vector<vector<int>> grid;
int brick_count = 0;

bool solve() {
    // Find first empty cell
    int fi = -1, fj = -1;
    for (int i = 0; i < N && fi == -1; i++)
        for (int j = 0; j < M && fi == -1; j++)
            if (grid[i][j] == -1) { fi = i; fj = j; }
    
    if (fi == -1) return true; // All filled
    
    int id = brick_count++;
    
    for (int k = 0; k < K; k++) {
        int len = L[k];
        // Try horizontal
        if (fj + len <= M) {
            bool ok = true;
            for (int j = fj; j < fj + len; j++) if (grid[fi][j] != -1) { ok = false; break; }
            if (ok) {
                for (int j = fj; j < fj + len; j++) grid[fi][j] = id;
                if (solve()) return true;
                for (int j = fj; j < fj + len; j++) grid[fi][j] = -1;
            }
        }
        // Try vertical
        if (len > 1 && fi + len <= N) {
            bool ok = true;
            for (int i = fi; i < fi + len; i++) if (grid[i][fj] != -1) { ok = false; break; }
            if (ok) {
                for (int i = fi; i < fi + len; i++) grid[i][fj] = id;
                if (solve()) return true;
                for (int i = fi; i < fi + len; i++) grid[i][fj] = -1;
            }
        }
    }
    
    brick_count--;
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    cin >> N >> M >> K;
    L.resize(K);
    for (int i = 0; i < K; i++) cin >> L[i];
    
    grid.assign(N, vector<int>(M, -1));
    
    if (!solve()) { cout << "NO\n"; return 0; }
    
    // Color bricks
    int total = brick_count;
    vector<set<int>> adj(total);
    int dx[] = {0,0,1,-1}, dy[] = {1,-1,0,0};
    for (int i = 0; i < N; i++)
        for (int j = 0; j < M; j++)
            for (int d = 0; d < 4; d++) {
                int ni = i+dx[d], nj = j+dy[d];
                if (ni>=0&&ni<N&&nj>=0&&nj<M&&grid[ni][nj]!=grid[i][j]) {
                    adj[grid[i][j]].insert(grid[ni][nj]);
                }
            }
    
    vector<int> color(total, -1);
    for (int b = 0; b < total; b++) {
        set<int> used;
        for (int nb : adj[b]) if (color[nb] != -1) used.insert(color[nb]);
        for (int c = 0; c < 26; c++) if (!used.count(c)) { color[b] = c; break; }
    }
    
    cout << "YES\n";
    for (int i = 0; i < N; i++) {
        for (int j = 0; j < M; j++) cout << (char)('a' + color[grid[i][j]]);
        cout << "\n";
    }
    return 0;
}
```
