#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    
    vector<int> scores(n, 0);
    
    for (int j = 0; j < m; j++) {
        int a, b;
        cin >> a >> b;
        
        for (int i = 0; i < n; i++) {
            int pa, pb;
            cin >> pa >> pb;
            
            // Check winner/tie
            int winner = (a > b) ? 1 : (a < b) ? -1 : 0;
            int pwinner = (pa > pb) ? 1 : (pa < pb) ? -1 : 0;
            if (winner == pwinner) scores[i] += 2;
            
            // Check difference
            if ((a - b) == (pa - pb)) scores[i] += 3;
            
            // Check first team goals
            if (a == pa) scores[i] += 1;
            
            // Check second team goals
            if (b == pb) scores[i] += 1;
        }
    }
    
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << scores[i];
    }
    cout << endl;
    
    return 0;
}
