```cpp
#include <iostream>
#include <string>
using namespace std;

int main() {
    string secret, guess;
    cin >> secret >> guess;
    
    int bulls = 0, cows = 0;
    
    for (int i = 0; i < 4; i++) {
        if (secret[i] == guess[i]) bulls++;
    }
    
    for (int i = 0; i < 4; i++) {
        for (int j = 0; j < 4; j++) {
            if (i != j && secret[i] == guess[j]) cows++;
        }
    }
    
    cout << bulls << " " << cows << endl;
    return 0;
}
```
