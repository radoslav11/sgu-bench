The number of abelian groups of order n is the product over all prime powers in the factorization of n of the number of partitions of the exponent.

For example, if n = p1^a1 * p2^a2 * ..., then the answer is p(a1) * p(a2) * ... where p(k) is the number of integer partitions of k.

Since n ≤ 10^12, the maximum exponent of any prime is at most 39 (since 2^40 > 10^12). So we need partitions up to p(39).

Steps:
1. Factorize n
2. For each prime factor with exponent e, compute the number of partitions of e
3. Multiply all partition numbers together

Let me write it properly:

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    // Compute partition numbers p(0..40)
    // p(n) = number of ways to write n as sum of positive integers
    // DP: partition[i][j] = number of partitions of i using parts <= j
    vector<long long> part(41, 0);
    // Use standard partition DP
    vector<vector<long long>> dp(41, vector<long long>(41, 0));
    for (int j = 0; j <= 40; j++) dp[0][j] = 1;
    for (int i = 1; i <= 40; i++) {
        dp[i][0] = 0;
        for (int j = 1; j <= 40; j++) {
            dp[i][j] = dp[i][j-1] + (i >= j ? dp[i-j][j] : 0);
        }
    }
    for (int i = 0; i <= 40; i++) part[i] = dp[i][i];
    
    long long n;
    cin >> n;
    
    long long ans = 1;
    // Factorize n
    for (long long p = 2; p * p <= n; p++) {
        if (n % p == 0) {
            int exp = 0;
            while (n % p == 0) {
                exp++;
                n /= p;
            }
            ans *= part[exp];
        }
    }
    if (n > 1) {
        ans *= part[1]; // exponent is 1, p(1) = 1
    }
    
    cout << ans << endl;
    return 0;
}
```
