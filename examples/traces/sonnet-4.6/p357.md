Looking at this problem, I need to find the minimum number of button presses to switch from channel X to channel Y.

The operations available:
1. Press ↑: channel = (channel + 1) % 100 (1 click)
2. Press ↓: channel = (channel - 1 + 100) % 100 (1 click)
3. Press a single digit 0-9: switches to that channel (1 click) - for channels 0-9
4. Press "--" then two digits: switches to channel 10-99 (3 clicks total)

Button availability:
- Digits 1-9: from the input
- ↑ and ↓: from the input
- "--": from the input
- digit 0: from the input

Let me parse the input:
- Line 1: buttons 1, 2, 3, ↑
- Line 2: buttons 4, 5, 6, ↓
- Line 3: buttons 7, 8, 9
- Line 4: buttons --, 0

For single digit channels (0-9): can be accessed with 1 click if the digit button works.
For two digit channels (10-99): can be accessed with 3 clicks if "--" works and both digit buttons work.

I'll use BFS on channel states (0-99).

For each channel, the possible transitions:
1. If ↑ works: go to (ch+1)%100 with cost 1
2. If ↓ works: go to (ch-1+100)%100 with cost 1
3. For any channel c from 0 to 9: if digit[c] works, go to channel c with cost 1
4. For any channel c from 10 to 99: if "--" works and digit[c/10] works and digit[c%10] works, go to channel c with cost 3

This is a shortest path problem. Since costs are either 1 or 3, I can use Dijkstra or BFS with a deque (0-1 BFS won't work directly since cost is 1 or 3, so use Dijkstra).

Actually with only 100 nodes and small costs, Dijkstra is fine.

Let me code this up:

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    // Parse buttons
    // digit[0..9], up, down, dash
    int digit[10];
    int up_btn, down_btn, dash_btn;
    
    int a, b, c, d;
    // Line 1: 1, 2, 3, up
    cin >> a >> b >> c >> d;
    digit[1]=a; digit[2]=b; digit[3]=c; up_btn=d;
    // Line 2: 4, 5, 6, down
    cin >> a >> b >> c >> d;
    digit[4]=a; digit[5]=b; digit[6]=c; down_btn=d;
    // Line 3: 7, 8, 9
    cin >> a >> b >> c;
    digit[7]=a; digit[8]=b; digit[9]=c;
    // Line 4: --, 0
    cin >> a >> b;
    dash_btn=a; digit[0]=b;
    
    int X, Y;
    cin >> X >> Y;
    
    if (X == Y) { cout << 0; return 0; }
    
    // Dijkstra
    vector<int> dist(100, INT_MAX);
    priority_queue<pair<int,int>, vector<pair<int,int>>, greater<>> pq;
    dist[X] = 0;
    pq.push({0, X});
    
    while (!pq.empty()) {
        auto [cost, u] = pq.top(); pq.pop();
        if (cost > dist[u]) continue;
        
        // up
        if (up_btn) {
            int v = (u+1)%100;
            if (dist[u]+1 < dist[v]) { dist[v]=dist[u]+1; pq.push({dist[v],v}); }
        }
        // down
        if (down_btn) {
            int v = (u-1+100)%100;
            if (dist[u]+1 < dist[v]) { dist[v]=dist[u]+1; pq.push({dist[v],v}); }
        }
        // single digit
        for (int c2=0; c2<=9; c2++) {
            if (digit[c2]) {
                int v = c2;
                if (dist[u]+1 < dist[v]) { dist[v]=dist[u]+1; pq.push({dist[v],v}); }
            }
        }
        // two digit with --
        if (dash_btn) {
            for (int c2=10; c2<=99; c2++) {
                int d1=c2/10, d2=c2%10;
                if (digit[d1] && digit[d2]) {
                    int v=c2;
                    if (dist[u]+3 < dist[v]) { dist[v]=dist[u]+3; pq.push({dist[v],v}); }
                }
            }
        }
    }
    
    cout << (dist[Y]==INT_MAX ? -1 : dist[Y]);
    return 0;
}
```
