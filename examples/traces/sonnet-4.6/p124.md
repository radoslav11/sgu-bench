The classic point-in-polygon problem with axis-aligned segments.

I'll use the ray casting method, shooting a ray upward (in the +Y direction) from the point and counting crossings.

Key considerations:
1. If the point lies on any segment → BORDER
2. For ray casting with horizontal/vertical segments, need careful handling
3. Since all segments are axis-aligned, I need to handle horizontal and vertical segments differently

For the upward ray (x = X0, y > Y0):
- Vertical segments: if the segment's x == X0 and the segment overlaps y > Y0, this needs careful handling (border check first)
- Horizontal segments: skip (parallel to... wait, horizontal segments are perpendicular to our upward ray)

Actually, let me use a rightward ray (+X direction) from (X0, Y0):

For each segment:
- If horizontal: check if point is on it (BORDER), otherwise skip for ray casting
- If vertical: check if point is on it (BORDER), otherwise check ray crossing

For vertical segment at x=c from y=a to y=b (a<b):
- Ray crosses if c > X0 and Y0 is in range [a, b)
  (using half-open interval to avoid double-counting at corners)

Let me implement this carefully:

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int K;
    cin >> K;
    
    vector<int> x1v(K), y1v(K), x2v(K), y2v(K);
    for(int i = 0; i < K; i++){
        cin >> x1v[i] >> y1v[i] >> x2v[i] >> y2v[i];
    }
    
    int X0, Y0;
    cin >> X0 >> Y0;
    
    // Check BORDER first
    for(int i = 0; i < K; i++){
        int ax = x1v[i], ay = y1v[i], bx = x2v[i], by = y2v[i];
        if(ay == by){
            // horizontal segment
            if(Y0 == ay){
                int minx = min(ax, bx), maxx = max(ax, bx);
                if(X0 >= minx && X0 <= maxx){
                    cout << "BORDER" << endl;
                    return 0;
                }
            }
        } else {
            // vertical segment
            if(X0 == ax){
                int miny = min(ay, by), maxy = max(ay, by);
                if(Y0 >= miny && Y0 <= maxy){
                    cout << "BORDER" << endl;
                    return 0;
                }
            }
        }
    }
    
    // Ray casting: shoot ray in +X direction from (X0, Y0)
    // Count intersections with vertical segments
    // Use half-open interval: [miny, maxy) to handle corners
    int count = 0;
    for(int i = 0; i < K; i++){
        int ax = x1v[i], ay = y1v[i], bx = x2v[i], by = y2v[i];
        if(ay == by){
            // horizontal, skip
            continue;
        }
        // vertical segment at x = ax
        if(ax <= X0) continue; // must be strictly to the right
        int miny = min(ay, by), maxy = max(ay, by);
        // Use half-open [miny, maxy)
        if(Y0 >= miny && Y0 < maxy){
            count++;
        }
    }
    
    if(count % 2 == 1){
        cout << "INSIDE" << endl;
    } else {
        cout << "OUTSIDE" << endl;
    }
    
    return 0;
}
```
