```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    int N,M;
    cin>>N>>M;
    int K=M-N+1;
    if(K<1){cout<<"No solution";return 0;}
    // Need sum(si) in [3K, min(10K,M)], R=M-sum(si)>=0
    // Also stations in circles = sum(si)-(K-1) <= N
    // sum(si) <= N+K-1
    int lo=3*K, hi=min(10*K, min(M, N+K-1));
    if(lo>hi){cout<<"No solution";return 0;}
    int S=lo; int R=M-S;
    // distribute S among K circles, each 3-10
    vector<int> sz(K,3);
    // already S=3K assigned
    // assign stations
    vector<vector<int>> circles(K);
    int stid=1;
    // shared[i] = station shared between circle i and i+1
    vector<int> shared(K-1);
    for(int i=0;i<K;i++){circles[i].resize(sz[i]);}
    // build station numbers
    // circle 0: stations 1..sz[0]
    // circle i shares one station with circle i-1
    int cur=1;
    for(int i=0;i<K;i++){
        for(int j=0;j<sz[i];j++){
            if(i>0&&j==0){circles[i][0]=shared[i-1];}
            else{circles[i][j]=cur++;}
        }
        if(i<K-1)shared[i]=circles[i][0];// share first station of next... let me share last of current
    }
    // Fix: share last station of circle i with first of circle i+1
    cur=1;
    for(int i=0;i<K;i++){
        if(i==0){for(int j=0;j<sz[i];j++)circles[i][j]=cur++;}
        else{circles[i][0]=circles[i-1][sz[i-1]-1];for(int j=1;j<sz[i];j++)circles[i][j]=cur++;}
    }
    // radial lines
    vector<pair<int,int>> radials;
    // find available transfer stations (on exactly one circle or shared but already used)
    // use non-shared stations of circles for radials
    set<int> sharedSt(shared.begin(),shared.end());
    // available: all circle stations not shared, not yet used for radial
    vector<int> avail;
    for(int i=0;i<K;i++)for(int s:circles[i])if(!sharedSt.count(s))avail.push_back(s);
    if((int)avail.size()<R){cout<<"No solution";return 0;}
    for(int i=0;i<R;i++)radials.push_back({avail[i],cur++});
    if(cur-1!=N){cout<<"No solution";return 0;}
    cout<<K<<"\n";
    for(int i=0;i<K;i++){cout<<sz[i];for(int s:circles[i])cout<<" "<<s;cout<<"\n";}
    cout<<R<<"\n";
    for(auto&p:radials)cout<<p.first<<" "<<p.second<<"\n";
}
```
