The largest circle inscribed in a convex polygon has radius equal to the minimum distance from the center to any edge (the inradius). The optimal center is the point that maximizes the minimum distance to all edges - this is a linear programming problem (maximizing the minimum of linear functions).

For a convex polygon, the largest inscribed circle's center is the Chebyshev center, which can be found via ternary search or linear programming.

Since the polygon is convex, the function "minimum distance to all edges" is concave, so we can use ternary search in 2D. We can do nested ternary search on x and y coordinates.

Wait, but nested ternary search works only if for each fixed x, the optimal y gives a unimodal function in x. Since the minimum of linear functions (which are concave) - yes, this works.

Let me implement nested ternary search:

For each fixed x, find the y that maximizes min_distance(x, y) using ternary search. Then ternary search over x.

The distance from point (px, py) to line ax+by+c=0 (with a²+b²=1) is |a*px + b*py + c|. For a convex polygon with CCW orientation, the interior is on the left side of each edge, so the signed distance is positive inside.

Let me compute for each edge the line equation such that the signed distance is positive inside.

For edge from (x1,y1) to (x2,y2) in CCW order:
- Direction: (dx, dy) = (x2-x1, y2-y1)
- Inward normal: (-dy, dx) normalized... wait, for CCW, the inward normal points left, which is (-dy, dx).
- Line: -dy*(x - x1) + dx*(y - y1) = 0 → -dy*x + dx*y + (dy*x1 - dx*y1) = 0
- Distance = (-dy*x + dx*y + dy*x1 - dx*y1) / sqrt(dy²+dx²)

This should be positive for interior points.

```cpp
#include <bits/stdc++.h>
using namespace std;

int N;
vector<double> A, B, C; // line: A*x + B*y + C >= 0 for interior

double minDist(double x, double y) {
    double mn = 1e18;
    for (int i = 0; i < N; i++) {
        double d = A[i]*x + B[i]*y + C[i];
        if (d < mn) mn = d;
    }
    return mn;
}

double ternaryY(double x, double ylo, double yhi) {
    for (int i = 0; i < 200; i++) {
        double m1 = ylo + (yhi - ylo) / 3;
        double m2 = yhi - (yhi - ylo) / 3;
        if (minDist(x, m1) < minDist(x, m2))
            ylo = m1;
        else
            yhi = m2;
    }
    return minDist(x, (ylo+yhi)/2);
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    
    cin >> N;
    vector<double> px(N), py(N);
    for(int i = 0; i < N; i++) cin >> px[i] >> py[i];
    
    for(int i = 0; i < N; i++){
        double x1 = px[i], y1 = py[i];
        double x2 = px[(i+1)%N], y2 = py[(i+1)%N];
        double dx = x2-x1, dy = y2-y1;
        double len = sqrt(dx*dx+dy*dy);
        // inward normal for CCW: (-dy, dx)
        double a = -dy/len, b = dx/len;
        double c = -(a*x1 + b*y1);
        A.push_back(a);
        B.push_back(b);
        C.push_back(c);
    }
    
    // bounding box
    double xlo = *min_element(px.begin(),px.end());
    double xhi = *max_element(px.begin(),px.end());
    double ylo = *min_element(py.begin(),py.end());
    double yhi = *max_element(py.begin(),py.end());
    
    // ternary search over x
    double xl = xlo, xr = xhi;
    for(int i = 0; i < 200; i++){
        double m1 = xl + (xr-xl)/3;
        double m2 = xr - (xr-xl)/3;
        if(ternaryY(m1, ylo, yhi) < ternaryY(m2, ylo, yhi))
            xl = m1;
        else
            xr = m2;
    }
    
    double ans = ternaryY((xl+xr)/2, ylo, yhi);
    cout << fixed << setprecision(6) << ans << endl;
    return 0;
}
```
