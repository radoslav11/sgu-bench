#include<bits/stdc++.h>
using namespace std;

int main(){
    // We need to compute a^b - b^a for potentially large numbers
    // a,b up to 100, so 100^100 is a huge number
    // We need big integer arithmetic
    
    int a, b;
    cin >> a >> b;
    
    // Use Python-style big integers via strings or just use Java BigInteger... 
    // We need to implement big integer in C++
    
    // Let's implement big integer multiplication and subtraction
    
    // Represent numbers as vectors of digits (base 10), least significant first
    
    auto multiply = [](vector<int> num, int x) -> vector<int> {
        int carry = 0;
        for(int i = 0; i < (int)num.size(); i++){
            long long cur = (long long)num[i] * x + carry;
            num[i] = cur % 10;
            carry = cur / 10;
        }
        while(carry > 0){
            num.push_back(carry % 10);
            carry /= 10;
        }
        return num;
    };
    
    auto power = [&](int base, int exp) -> vector<int> {
        vector<int> result = {1};
        for(int i = 0; i < exp; i++){
            result = multiply(result, base);
        }
        return result;
    };
    
    vector<int> ab = power(a, b);
    vector<int> ba = power(b, a);
    
    // Make same length
    while(ab.size() < ba.size()) ab.push_back(0);
    while(ba.size() < ab.size()) ba.push_back(0);
    
    int n = ab.size();
    
    // Compare ab and ba
    // ab is stored least significant first, so compare from most significant
    bool ab_greater = false, equal = true;
    for(int i = n-1; i >= 0; i--){
        if(ab[i] > ba[i]){ ab_greater = true; equal = false; break; }
        if(ab[i] < ba[i]){ ab_greater = false; equal = false; break; }
    }
    
    if(equal){
        cout << 0 << endl;
        return 0;
    }
    
    // Compute |ab - ba|
    vector<int> result(n, 0);
    if(ab_greater){
        // ab - ba
        int borrow = 0;
        for(int i = 0; i < n; i++){
            int diff = ab[i] - ba[i] - borrow;
            if(diff < 0){ diff += 10; borrow = 1; }
            else borrow = 0;
            result[i] = diff;
        }
    } else {
        // ba - ab
        int borrow = 0;
        for(int i = 0; i < n; i++){
            int diff = ba[i] - ab[i] - borrow;
            if(diff < 0){ diff += 10; borrow = 1; }
            else borrow = 0;
            result[i] = diff;
        }
    }
    
    // Remove leading zeros
    while(result.size() > 1 && result.back() == 0) result.pop_back();
    
    if(!ab_greater) cout << "-";
    for(int i = result.size()-1; i >= 0; i--) cout << result[i];
    cout << endl;
    
    return 0;
}
