Looking at this problem, I need to count N×M binary grids where no 2×2 subgrid is all the same color, modulo P.

**Approach:**
- Since M ≤ 5, I can use column-by-column DP with matrix exponentiation (since N can be huge).
- State: the current column's coloring (2^M possibilities).
- Transition: from column i to column i+1, a transition is valid if no 2×2 block is monochromatic. A 2×2 block involves rows j and j+1 in adjacent columns. So for each pair of adjacent rows j, j+1: NOT (col[i][j] == col[i][j+1] == col[i+1][j] == col[i+1][j+1]).

**Matrix exponentiation with big N:**
- N can be up to 10^100, so I need to represent it as a big integer and do matrix exponentiation.
- Matrix size is 2^M × 2^M, max 32×32.
- Operations mod P.

**Building the transition matrix:**
- States are bitmasks of size M (each bit = color of row in that column).
- Transition from state `a` to state `b` is valid if for all j in [0, M-2]: NOT (a[j]==a[j+1] AND b[j]==b[j+1] AND a[j]==b[j]).
  
  Specifically, for each j: if bits j and j+1 of `a` are equal, and bits j and j+1 of `b` are equal, and bit j of `a` equals bit j of `b`, then it's invalid (2×2 same color).

**Initial state:** All column patterns are valid (any of 2^M states for first column).

**Answer:** Sum of all entries in (T^(N-1)) * initial_vector, where initial_vector = all 1s.

#include <bits/stdc++.h>
using namespace std;
typedef vector<vector<long long>> Mat;
int MOD, SZ;
Mat mul(const Mat& A, const Mat& B){
    Mat C(SZ,vector<long long>(SZ,0));
    for(int i=0;i<SZ;i++) for(int k=0;k<SZ;k++) if(A[i][k]) for(int j=0;j<SZ;j++)
        C[i][j]=(C[i][j]+A[i][k]*B[k][j])%MOD;
    return C;
}
vector<int> toBits(string n){
    vector<int> bits;
    while(n!="0"&&n!=""){
        int r=0; string q="";
        for(char c:n){int d=r*10+(c-'0');q+=(char)('0'+d/2);r=d%2;}
        bits.push_back(r);
        // remove leading zeros
        int s=0; while(s<(int)q.size()-1&&q[s]=='0')s++;
        n=q.substr(s);
    }
    reverse(bits.begin(),bits.end());
    return bits;
}
Mat matpow(Mat M, string exp){
    // subtract 1 from exp
    // actually compute M^exp
    vector<int> bits=toBits(exp);
    Mat R(SZ,vector<long long>(SZ,0));
    for(int i=0;i<SZ;i++) R[i][i]=1;
    for(int b:bits){R=mul(R,R);if(b)R=mul(R,M);}
    return R;
}
string subOne(string n){
    int i=n.size()-1;
    while(n[i]=='0'){n[i]='9';i--;}
    n[i]--;
    int s=0;while(s<(int)n.size()-1&&n[s]=='0')s++;
    return n.substr(s);
}
int main(){
    string N; int M; cin>>N>>M>>MOD;
    SZ=1<<M;
    Mat T(SZ,vector<long long>(SZ,0));
    for(int a=0;a<SZ;a++) for(int b=0;b<SZ;b++){
        bool ok=true;
        for(int j=0;j<M-1;j++){
            int aj=(a>>j)&1,aj1=(a>>(j+1))&1,bj=(b>>j)&1,bj1=(b>>(j+1))&1;
            if(aj==aj1&&bj==bj1&&aj==bj){ok=false;break;}
        }
        if(ok) T[a][b]=1;
    }
    // answer = sum of T^(N-1) applied to all-ones vector
    if(N=="1"){
        long long ans=SZ%MOD; cout<<ans; return 0;
    }
    string exp=subOne(N);
    Mat R=matpow(T,exp);
    long long ans=0;
    for(int i=0;i<SZ;i++) for(int j=0;j<SZ;j++) ans=(ans+R[i][j])%MOD;
    cout<<ans;
}
