For N=3, I can enumerate row by row. Each row has 3 cells, each 0-9, so 1000 possibilities per row. With 3 rows that's 10^9 - still too much.

But with constraint checking: When I fix row 1 and row 2, I can check B values for row 1 (since all neighbors of row 1 cells are known when row 2 is fixed). Then enumerate row 3 and check row 2 and row 3.

Strategy:
- Enumerate all possible values for row 0 (1000 options)
- Enumerate all possible values for row 1 (1000 options)
  - Check if row 0's B values are satisfied (row 0 neighbors are within row 0 and row 1)
  - Check partial constraints for row 1 (left/right/up neighbors known, down not yet)
- Enumerate all possible values for row 2 (1000 options)
  - Check row 1's B values fully
  - Check row 2's B values fully

This gives 1000 * 1000 * 1000 = 10^9 in worst case... but with early termination it should be much faster since we prune after row 0+1 check.

Actually after fixing rows 0 and 1, we check row 0 fully (all its neighbors are in rows 0 and 1). That prunes heavily. Then for each valid (row0, row1) pair, we try 1000 row2 options and verify.

#include <bits/stdc++.h>
using namespace std;
int N, B[3][3], A[3][3];
int dx[]={0,0,1,-1}, dy[]={1,-1,0,0};

bool checkCell(int i, int j){
    int cnt=0;
    for(int d=0;d<4;d++){int ni=i+dx[d],nj=j+dy[d];if(ni>=0&&ni<N&&nj>=0&&nj<N&&A[ni][nj]>A[i][j])cnt++;}
    return cnt==B[i][j];
}

// Check if cell (i,j) can still be consistent given assigned cells (pos = next to assign)
// All cells with index < pos are assigned
bool partialCheck(int i, int j, int pos){
    int cnt=0, unknown=0;
    for(int d=0;d<4;d++){
        int ni=i+dx[d],nj=j+dy[d];
        if(ni<0||ni>=N||nj<0||nj>=N) continue;
        int npos=ni*N+nj;
        if(npos<pos){if(A[ni][nj]>A[i][j])cnt++;}
        else unknown++;
    }
    if(cnt>B[i][j]) return false;
    if(cnt+unknown<B[i][j]) return false;
    return true;
}

bool solve(int pos){
    if(pos==N*N){
        for(int i=0;i<N;i++)for(int j=0;j<N;j++)if(!checkCell(i,j))return false;
        return true;
    }
    int i=pos/N,j=pos%N;
    for(int v=0;v<=9;v++){
        A[i][j]=v;
        bool ok=true;
        // Check current cell partially
        if(!partialCheck(i,j,pos+1)){ok=false;}
        // Check all previous cells that have (i,j) as neighbor
        if(ok){
            for(int d=0;d<4&&ok;d++){
                int ni=i+dx[d],nj=j+dy[d];

                    if(!partialCheck(ni,nj,pos+1)) ok=false;
                }
            }
        }
        if(ok&&solve(pos+1)) return true;
    }
    return false;
}
