Two kings in adjacent rows attack if they're in the same column or adjacent columns.

**Compatibility between masks:**
- `mask1` (previous row) and `mask2` (current row) are compatible if:
  - `mask1 & mask2 == 0` (no same column)

  - `(mask1 << 1) & mask2 == 0` (no king in mask1 one to the right of king in mask2)

**Self-compatible mask:** no two adjacent bits set, i.e., `mask & (mask >> 1) == 0`.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;
    
    vector<int> valid_masks;
    for (int m = 0; m < (1 << n); m++) {
        if ((m & (m >> 1)) == 0) valid_masks.push_back(m);
    }
    
    auto popcount = [](int x) { return __builtin_popcount(x); };
    
    // dp[mask][cnt] = number of ways
    int maxk = k;
    vector<vector<long long>> dp(1 << n, vector<long long>(maxk + 1, 0));
    
    // Initialize first row
    for (int m : valid_masks) {
        int cnt = popcount(m);
        if (cnt <= maxk) dp[m][cnt]++;
    }
    
    for (int row = 1; row < n; row++) {
        vector<vector<long long>> ndp(1 << n, vector<long long>(maxk + 1, 0));
        for (int pm : valid_masks) {
            for (int nm : valid_masks) {
                // check compatibility
                if (pm & nm) continue;
                if ((pm >> 1) & nm) continue;
                if ((pm << 1) & nm) continue;
                int add = popcount(nm);
                for (int cnt = 0; cnt + add <= maxk; cnt++) {
                    if (dp[pm][cnt] == 0) continue;
                    ndp[nm][cnt + add] += dp[pm][cnt];
                }
            }
        }
        dp = ndp;
    }
    
    long long ans = 0;
    for (int m = 0; m < (1 << n); m++) {
        ans += dp[m][k];
    }
    cout << ans << endl;
    return 0;
}
```
