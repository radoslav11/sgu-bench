Looking at this problem, I need to simulate the spread of weed until no more squares can be filled.

Rule: A free square becomes occupied if at least 2 of its adjacent squares are occupied.

I need to repeatedly apply this rule until no changes occur, then count the occupied squares.

Let me think about this with BFS/queue approach:
- Start with initial weed squares
- For each free square, count how many adjacent squares are weedy
- If count >= 2, it becomes weed, add to queue, update neighbors' counts

Let me implement this:

1. Parse grid
2. For each free cell, compute count of weed neighbors
3. Put all free cells with count >= 2 into a queue
4. Process queue: mark cell as weed, update neighbor counts, if any neighbor (free cell) now has count >= 2, add to queue
5. Count total weed cells

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N, M;
    cin >> N >> M;
    
    vector<string> grid(N);
    for(int i = 0; i < N; i++) cin >> grid[i];
    
    vector<vector<int>> cnt(N, vector<int>(M, 0));
    vector<vector<bool>> weed(N, vector<bool>(M, false));
    
    int dx[] = {0,0,1,-1};
    int dy[] = {1,-1,0,0};
    
    // Initialize weed
    for(int i = 0; i < N; i++)
        for(int j = 0; j < M; j++)
            if(grid[i][j] == 'X') weed[i][j] = true;
    
    // Compute initial counts for free cells
    for(int i = 0; i < N; i++)
        for(int j = 0; j < M; j++)
            if(!weed[i][j])
                for(int d = 0; d < 4; d++){
                    int ni = i+dx[d], nj = j+dy[d];
                    if(ni>=0 && ni<N && nj>=0 && nj<M && weed[ni][nj])
                        cnt[i][j]++;
                }
    
    queue<pair<int,int>> q;
    for(int i = 0; i < N; i++)
        for(int j = 0; j < M; j++)
            if(!weed[i][j] && cnt[i][j] >= 2)
                q.push({i,j});
    
    while(!q.empty()){
        auto [x,y] = q.front(); q.pop();
        if(weed[x][y]) continue;
        weed[x][y] = true;
        // Update neighbors
        for(int d = 0; d < 4; d++){
            int nx = x+dx[d], ny = y+dy[d];
            if(nx>=0 && nx<N && ny>=0 && ny<M && !weed[nx][ny]){
                cnt[nx][ny]++;
                if(cnt[nx][ny] >= 2)
                    q.push({nx,ny});
            }
        }
    }
    
    int ans = 0;
    for(int i = 0; i < N; i++)
        for(int j = 0; j < M; j++)
            if(weed[i][j]) ans++;
    
    cout << ans << endl;
    
    return 0;
}
```
