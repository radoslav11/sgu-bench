Looking at this problem, I need to:

1. Parse a B++ program (functions with 0-2 arguments, operators L/R/C/function calls/argument references)
2. Simulate robot movement on a grid
3. Track which '*' cells are visited

Let me break down the parsing and execution:

**Parsing:**
- Functions have format: `name(args){body}`
- Body contains: L, R, C (commands), argument names (execute as L/R/C), function calls `name(args)`

**Robot state:**
- Position (row, col), direction (up=0, right=1, down=2, left=3 initially facing up)
- Turn left: direction = (dir+3)%4, Turn right: direction = (dir+1)%4

**Execution:**
- Start with function `m()`
- Each function call passes actual argument values (which are L/R/C characters)

#include <bits/stdc++.h>
using namespace std;

// Parse functions
struct Function {
    string name;
    vector<string> params; // parameter names (capital letters)
    string body; // the body content (operators as a sequence to parse)
};

map<string, Function> funcs;
int N, M;
char grid[55][55];
int robotR, robotC, robotDir; // dir: 0=up,1=right,2=down,3=left
vector<pair<int,int>> visited;
set<pair<int,int>> visitedSet;

void executeCmd(char cmd) {
    if (cmd == 'L') robotDir = (robotDir + 3) % 4;
    else if (cmd == 'R') robotDir = (robotDir + 1) % 4;
    else if (cmd == 'C') {
        int dr[] = {-1, 0, 1, 0};
        int dc[] = {0, 1, 0, -1};
        int nr = robotR + dr[robotDir];
        int nc = robotC + dc[robotDir];
        if (nr >= 0 && nr < N && nc >= 0 && nc < M) {
            robotR = nr; robotC = nc;
        }
        pair<int,int> pos = {robotR, robotC};
        if (grid[robotR][robotC] == '*' && visitedSet.find(pos) == visitedSet.end()) {
            visited.push_back(pos);
            visitedSet.insert(pos);
        }
    }
}

void executeFunction(const string& fname, map<string,char>& env);

void executeBody(const string& body, map<string,char>& env) {
    int i = 0, n = body.size();
    while (i < n) {
        char c = body[i];
        if (c == 'L' || c == 'R' || c == 'C') { executeCmd(c); i++; }
        else if (isupper(c)) { // argument
            executeCmd(env[string(1,c)]); i++;
        } else if (islower(c)) { // function call
            string fname(1, c); i++;
            // expect '('
            i++; // skip '('
            vector<char> args;
            while (body[i] != ')') {
                char ac = body[i];
                if (ac == 'L' || ac == 'R' || ac == 'C') args.push_back(ac);
                else if (isupper(ac)) args.push_back(env[string(1,ac)]);
                i++;
                if (i < n && body[i] == ',') i++;
            }
            i++; // skip ')'
            map<string,char> newEnv;
            Function& f = funcs[fname];
            for (int j = 0; j < (int)f.params.size(); j++) newEnv[f.params[j]] = args[j];
            executeFunction(fname, newEnv);
        } else i++;
    }
}

void executeFunction(const string& fname, map<string,char>& env) {
    executeBody(funcs[fname].body, env);
}

int main() {
    cin >> N >> M;
    for (int i = 0; i < N; i++) { cin >> grid[i]; for (int j = 0; j < M; j++) if (grid[i][j] == 'R') { robotR=i; robotC=j; } }
    robotDir = 0;
    string all, tok; while (cin >> tok) all += tok;
    int i = 0;
    while (i < (int)all.size()) {
        if (!islower(all[i])) { i++; continue; }
        Function f; f.name = string(1, all[i]); i++;
        i++; // '('
        while (all[i] != ')') { if (isupper(all[i])) f.params.push_back(string(1,all[i])); if (all[i]==',') {}; i++; }
        i++; i++; // '){'
        int depth=1; string body;
        while (depth > 0) { if(all[i]=='{') depth++; else if(all[i]=='}') depth--; if(depth>0) body+=all[i]; i++; }
        f.body = body; funcs[f.name] = f;
    }
    map<string,char> env;
    executeFunction("m", env);
    cout << visited.size() << "\n";
    for (auto& p : visited) cout << p.first+1 << " " << p.second+1 << "\n";
}
