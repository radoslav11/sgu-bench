```cpp
#include <bits/stdc++.h>
using namespace std;

// Count numbers in [1, X] with no "13" substring
long long countValid(long long X) {
    if (X <= 0) return 0;
    string s = to_string(X);
    int n = s.size();
    // dp[tight][last][started]
    // iterate digit by digit
    // state: tight, prev_digit (0-9, or -1 if not started)
    
    // Use map or arrays
    // tight: bool, prev: 0..9 or 10 for "not started"
    long long dp[2][11]; // [tight][prev_digit: 0-10 where 10=not started]
    memset(dp, 0, sizeof(dp));
    dp[1][10] = 1; // initially: tight, not started
    
    for (int i = 0; i < n; i++) {
        long long ndp[2][11];
        memset(ndp, 0, sizeof(ndp));
        int lim = s[i] - '0';
        for (int tight = 0; tight <= 1; tight++) {
            for (int prev = 0; prev <= 10; prev++) {
                if (dp[tight][prev] == 0) continue;
                int maxd = tight ? lim : 9;
                for (int d = 0; d <= maxd; d++) {
                    if (prev == 1 && d == 3) continue; // forbidden "13"
                    int ntight = tight && (d == maxd);
                    int nprev;
                    if (prev == 10 && d == 0) nprev = 10; // still not started
                    else nprev = d;
                    ndp[ntight][nprev] += dp[tight][prev];
                }
            }
        }
        memcpy(dp, ndp, sizeof(dp));
    }
    
    long long res = 0;
    for (int tight = 0; tight <= 1; tight++)
        for (int prev = 0; prev <= 9; prev++) // prev != 10 means number has started
            res += dp[tight][prev];
    return res; // includes all valid numbers in [1, X]
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int T;
    cin >> T;
    while (T--) {
        long long N;
        cin >> N;
        // binary search for answer
        long long lo = 1, hi = 2e18;
        while (lo < hi) {
            long long mid = lo + (hi - lo) / 2;
            if (countValid(mid) >= N) hi = mid;
            else lo = mid + 1;
        }
        cout << lo << "\n";
    }
    return 0;
}
```
