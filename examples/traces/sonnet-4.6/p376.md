Looking at this problem, I need to simulate a multi-stage competition.

**Understanding the problem:**
- N participants, K riding stages
- Swimming stage first: time = 2·10^7 - Wi
- K riding stages, each with N horses
- At each riding stage start, participants arriving take horses in order of arrival (ties broken by strength - higher strength gets better horse)
- Horse ridingness for stage i, horse j: f(i,j) = 3Ai² + 5Ai·Bj + 2Bj²
- Riding stage time for participant i with horse j: 2·10^7 - Ri - Sj
- Final order by finish time (ties broken by strength - higher strength = better place)

**Key observations:**
- f(i,j) = 3Ai² + 5Ai·Bj + 2Bj² = (3Ai + 2Bj)(Ai + Bj)... let me factor: 3A² + 5AB + 2B² = (3A+2B)(A+B). So ridingness depends on both stage index and horse index.
- For a given stage i, horses have ridingness values f(i,j) for j=1..N. The best horse (highest ridingness) goes to the first participant.
- Need to sort horses at each stage by ridingness (descending), assign them in order of participant arrival.

**Simulation:**
1. Compute swim finish time for each participant: `swim_time[i] = 2e7 - W[i]`
2. For each of K stages:
   - Sort participants by current arrival time (ties: higher strength arrives first → gets better horse)
   - Compute horse ridingness for this stage: f(i,j) for j=1..N, sort descending
   - Assign horses to participants in order
   - Update each participant's arrival time: add (2e7 - R[i] - S[assigned_horse])

**Final output:** Sort participants by their final finish time (ascending), ties broken by strength (descending = better place). Output their original indices (1-based) in order from 1st place to last.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N, K;
    cin >> N >> K;
    
    vector<long long> W(N), R(N), P(N);
    for(int i=0;i<N;i++) cin >> W[i] >> R[i] >> P[i];
    
    vector<long long> A(K), B(N);
    for(int i=0;i<K;i++) cin >> A[i];
    for(int j=0;j<N;j++) cin >> B[j];
    
    // Current time for each participant
    vector<long long> cur_time(N);
    for(int i=0;i<N;i++) cur_time[i] = 20000000LL - W[i];
    
    // Simulate K stages
    for(int stage=0;stage<K;stage++){
        // Compute horse ridingness for this stage
        vector<long long> horse_rid(N);
        for(int j=0;j<N;j++){
            horse_rid[j] = 3LL*A[stage]*A[stage] + 5LL*A[stage]*B[j] + 2LL*B[j]*B[j];
        }
        // Sort horses descending
        sort(horse_rid.begin(), horse_rid.end(), greater<long long>());
        
        // Sort participants by arrival time (ascending), ties: higher strength first
        vector<int> order(N);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b){
            if(cur_time[a] != cur_time[b]) return cur_time[a] < cur_time[b];
            return P[a] > P[b];
        });
        
        // Assign horses
        vector<long long> new_time(N);
        for(int rank=0;rank<N;rank++){
            int idx = order[rank];
            long long S = horse_rid[rank]; // best available horse
            new_time[idx] = cur_time[idx] + (20000000LL - R[idx] - S);
        }
        cur_time = new_time;
    }
    
    // Final ranking
    vector<int> final_order(N);
    iota(final_order.begin(), final_order.end(), 0);
    sort(final_order.begin(), final_order.end(), [&](int a, int b){
        if(cur_time[a] != cur_time[b]) return cur_time[a] < cur_time[b];
        return P[a] > P[b];
    });
    
    for(int i=0;i<N;i++){
        if(i) cout << ' ';
        cout << final_order[i]+1;
    }
    cout << endl;
    
    return 0;
}
```
