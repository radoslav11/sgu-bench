    // DP: dp[j] = min tables to pack first i items using j space in current table
    // Since all items <= r (p<=r since rem<r, and if rem==1, p=2<=r since r>=3)
    // Greedy: sort descending, first fit decreasing
    sort(parts.begin(),parts.end(),greater<int>());

        for(int &b:bins){
