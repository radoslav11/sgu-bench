Looking at this problem, I need to find the number of integer pairs (x, y) such that:
- ax + by + c = 0
- x1 ≤ x ≤ x2
- y1 ≤ y ≤ y2

So ax + by = -c.

Let me think about this carefully.

For each valid x in [x1, x2], I need by = -c - ax, so if b ≠ 0, y = (-c - ax) / b must be an integer and in [y1, y2].

But the range can be up to 2×10^8 which is too large to iterate over.

Let me think differently using the theory of linear Diophantine equations.

**Case 1: a = 0 and b = 0**
- If c = 0: infinite solutions → but what's the count? It would be (x2-x1+1)*(y2-y1+1)
- If c ≠ 0: no solutions → 0

**Case 2: a = 0, b ≠ 0**
- by = -c, so y = -c/b must be integer and in [y1,y2]. If so, count = x2-x1+1, else 0.

**Case 3: b = 0, a ≠ 0**
- ax = -c, x = -c/a must be integer and in [x1,x2]. If so, count = y2-y1+1, else 0.

**Case 4: a ≠ 0, b ≠ 0**
- ax + by = -c
- This has solutions iff gcd(a,b) | (-c), i.e., gcd(a,b) | c.
- Let g = gcd(|a|, |b|). If g doesn't divide c, answer is 0.
- Otherwise, divide through: (a/g)x + (b/g)y = -c/g.
- Find one particular solution (x0, y0). General solution: x = x0 + (b/g)*t, y = y0 - (a/g)*t.
- Count t values such that x1 ≤ x0 + (b/g)*t ≤ x2 and y1 ≤ y0 - (a/g)*t ≤ y2.

For the general solution, I need to find integer t satisfying both constraints simultaneously.

Let B = b/g, A = a/g (with sign).
- x = x0 + B*t → t in [ceil((x1-x0)/B), floor((x2-x0)/B)] (careful with sign of B)
- y = y0 - A*t → t in [ceil((y1-y0)/(-A)), floor((y2-y0)/(-A))] (careful with sign of -A)

The intersection of these two integer ranges gives the count.

I need to find a particular solution using extended GCD.

Let me implement this carefully using 64-bit integers to avoid overflow.

```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll gcd_ext(ll a, ll b, ll &x, ll &y) {
    if (b == 0) { x = 1; y = 0; return a; }
    ll x1, y1;
    ll g = gcd_ext(b, a % b, x1, y1);
    x = y1;
    y = x1 - (a / b) * y1;
    return g;
}

// floor division
ll floordiv(ll a, ll b) {
    return a / b - (a % b != 0 && (a ^ b) < 0);
}
ll ceildiv(ll a, ll b) {
    return a / b + (a % b != 0 && (a ^ b) > 0);
}

int main(){
    ll a,b,c,x1,x2,y1,y2;
    cin>>a>>b>>c>>x1>>x2>>y1>>y2;
    
    if(a==0 && b==0){

        else cout<<0;
        return 0;

    if(a==0){
        if((-c)%b==0){ ll y0=-c/b; if(y0>=y1&&y0<=y2) cout<<x2-x1+1; else cout<<0;}
        else cout<<0;
        return 0;
    }
    if(b==0){
        if((-c)%a==0){ ll x0=-c/a; if(x0>=x1&&x0<=x2) cout<<y2-y1+1; else cout<<0;}
        else cout<<0;
        return 0;
    }
    ll px,py;
    ll g=gcd_ext(a,b,px,py);
    if((-c)%g!=0){cout<<0;return 0;}
    ll rhs=-c/g, A=a/g, B=b/g;
    px*=rhs; py*=rhs;
    // x=px+B*t, y=py-A*t
    ll tlo=LLONG_MIN,thi=LLONG_MAX;
    ll lo,hi;
    if(B>0){lo=ceildiv(x1-px,B);hi=floordiv(x2-px,B);}
    else{lo=ceildiv(x2-px,B);hi=floordiv(x1-px,B);}
    tlo=max(tlo,lo);thi=min(thi,hi);
    if(A>0){lo=ceildiv(py-y2,A);hi=floordiv(py-y1,A);}
    else{lo=ceildiv(py-y1,A);hi=floordiv(py-y2,A);}
    tlo=max(tlo,lo);thi=min(thi,hi);
    cout<<max(0LL,thi-tlo+1);
}
```
