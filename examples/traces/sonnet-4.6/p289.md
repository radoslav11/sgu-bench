Looking at this problem, I need to:

1. Parse a Tic-Tac-Toe board
2. Determine if the position is legal
3. If legal, determine the optimal outcome with both playing perfectly

**Legality checks:**
- Count X's and 0's. Since X goes first, count_X must equal count_0 or count_X = count_0 + 1
- If X has already won (3 in a row), it must be X's turn that just ended, so count_X = count_0 + 1, and 0 must not have won
- If 0 has already won, count_X = count_0, and X must not have won
- Both can't have won simultaneously

**Game outcome with perfect play:**
- Use minimax

```cpp
#include <bits/stdc++.h>
using namespace std;

char board[9];

bool wins(char c) {
    int w[8][3] = {{0,1,2},{3,4,5},{6,7,8},{0,3,6},{1,4,7},{2,5,8},{0,4,8},{2,4,6}};
    for(auto& line : w) {
        if(board[line[0]]==c && board[line[1]]==c && board[line[2]]==c) return true;
    }
    return false;
}

// returns 1 if X wins, -1 if 0 wins, 0 draw
// turn: 1 = X's turn, -1 = 0's turn
int minimax(int turn) {
    if(wins('X')) return 1;
    if(wins('0')) return -1;
    
    // check empty
    bool empty = false;
    for(int i=0;i<9;i++) if(board[i]=='.') { empty=true; break; }
    if(!empty) return 0;
    
    char c = (turn==1) ? 'X' : '0';
    int best = -turn * 2; // worst for current player
    
    for(int i=0;i<9;i++) {
        if(board[i]=='.') {
            board[i] = c;
            int val = minimax(-turn);
            board[i] = '.';
            if(turn==1) best = max(best, val);
            else best = min(best, val);
        }
    }
    return best;
}

int main(){
    string line;
    while(true) {
        string rows[3];
        if(!getline(cin, rows[0])) break;
        if(rows[0] == "Qc") break;
        getline(cin, rows[1]);
        getline(cin, rows[2]);
        // read empty line
        getline(cin, line);
        
        for(int i=0;i<3;i++)
            for(int j=0;j<3;j++)
                board[i*3+j] = rows[i][j];
        
        int cx=0, co=0;
        for(int i=0;i<9;i++) {
            if(board[i]=='X') cx++;
            if(board[i]=='0') co++;
        }
        
        bool xwins = wins('X');
        bool owins = wins('0');
        
        bool legal = true;
        
        // counts
        if(cx != co && cx != co+1) legal = false;
        
        // both can't win
        if(xwins && owins) legal = false;
        
        // if X won, it was X's last move, so cx = co+1
        if(xwins && cx != co+1) legal = false;
        
        // if 0 won, it was 0's last move, so cx = co
        if(owins && cx != co) legal = false;
        
        if(!legal) {
            cout << "Illegal position.\n";
            continue;
        }
        
        // determine whose turn
        int turn = (cx == co) ? 1 : -1;
        
        // if game already over
        if(xwins) { cout << "X wins.\n"; continue; }
        if(owins) { cout << "0 wins.\n"; continue; }
        
        int result = minimax(turn);
        if(result == 1) cout << "X wins.\n";
        else if(result == -1) cout << "0 wins.\n";
        else cout << "Game is a draw.\n";
    }
}
```
