The number of trailing zeros in N! is given by Legendre's formula: sum of floor(N/5^k) for k=1,2,...

We need to find the minimal N such that this sum equals Q.

Key observations:
1. The function f(N) = number of trailing zeros in N! is non-decreasing
2. It can jump by more than 1 (e.g., at multiples of 25, 125, etc.), meaning some values of Q have no solution
3. We binary search for the smallest N where f(N) >= Q, then check if f(N) == Q

If f(N) == Q, output N. Otherwise output "No solution".

```cpp
#include<bits/stdc++.h>
using namespace std;

long long trailingZeros(long long n) {
    long long count = 0;
    long long p = 5;
    while (p <= n) {
        count += n / p;
        p *= 5;
    }
    return count;
}

int main(){
    long long Q;
    cin >> Q;
    
    if (Q == 0) {
        cout << 0 << endl;
        return 0;
    }
    
    // Binary search for smallest N with trailingZeros(N) >= Q
    long long lo = 0, hi = 5LL * Q + 5;
    while (lo < hi) {
        long long mid = (lo + hi) / 2;
        if (trailingZeros(mid) >= Q) {
            hi = mid;
        } else {
            lo = mid + 1;
        }
    }
    
    if (trailingZeros(lo) == Q) {
        cout << lo << endl;
    } else {
        cout << "No solution" << endl;
    }
    
    return 0;
}
```
