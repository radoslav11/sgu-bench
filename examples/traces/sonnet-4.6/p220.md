```cpp
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int n, k;

// count ways to place cnt bishops on given cells (of one color) with no two attacking
// cells sorted, use backtracking
ll countWays(vector<pair<int,int>>& cells, int cnt) {
    if(cnt == 0) return 1;
    int m = cells.size();
    if(m < cnt) return 0;
    // dp over cells
    // use bitmask for used slash and bslash diagonals
    // slash = i+j (0..2n-2), bslash = i-j+n-1 (0..2n-2)
    // but 2^19 too big; use map or process smartly
    // Since cells of one color all have same parity of (i+j),
    // slash values are 0,2,4,... or 1,3,5,...
    // For n=10, at most 10 "/" diagonals per color (ceil and floor of n)
    // Actually: for even color (i+j even), slash = 0,2,4,...,2(n-1): n values
    // for odd color: 1,3,...,2n-3: n-1 values
    // So at most n=10 "/" diagonals per color => at most 10 bishops per color
    // bslash = i-j+n-1: 0..2n-2 = 19 values for n=10
    // 2^19 too big. But since at most 10 bishops, and 10 "/" diags,
    // we can use map<int,ll> for sparse representation
    
    // DP: process cells in order, for each cell decide include or not
    // state: (bslash_mask) -> count
    // slash is handled by processing each "/" diagonal and picking at most 1
    
    // Group by slash diagonal
    map<int, vector<pair<int,int>>> bySlash;
    for(auto& [r,c]: cells) bySlash[r+c].push_back({r,c});
    
    // dp[j] = map from bslash_mask to count, for placing j bishops
    int nb = 2*n-1; // number of bslash diagonals
    // Use vector of unordered_map
    vector<unordered_map<int,ll>> dp(cnt+1);
    dp[0][0] = 1;
    
    for(auto& [d, dcells]: bySlash) {
        // process in reverse j to avoid reuse
        for(int j = min(cnt, (int)dp.size()-1); j >= 0; j--) {
            for(auto& [mask, ways]: dp[j]) {
                for(auto& [r,c]: dcells) {
                    int bd = r - c + n - 1;
                    if(!(mask & (1<<bd))) {
                        if(j+1 <= cnt) dp[j+1][mask|(1<<bd)] += ways;
                    }
                }
            }
        }
    }
    
    ll res = 0;
    for(auto& [mask, ways]: dp[cnt]) res += ways;
    return res;
}

int main(){

    vector<pair<int,int>> white, black;
    for(int i=0;i<n;i++) for(int j=0;j<n;j++) {
        if((i+j)%2==0) white.push_back({i,j});
        else black.push_back({i,j});
    }
    
    ll ans = 0;
    for(int i=0; i<=k; i++) {
        ll fw = countWays(white, i);
        ll fb = countWays(black, k-i);
        ans += fw * fb;
    }
