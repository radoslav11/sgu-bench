Looking at this problem, I need to place maximum number of Guardians on a grid such that no two Guardians are in the same or neighboring cells (where neighboring means sharing a common edge).

This is essentially a **Maximum Independent Set** problem on a grid graph where cells are adjacent if they share an edge. On a grid graph, this can be solved efficiently using **bipartite matching** (König's theorem: Maximum Independent Set = Total vertices - Maximum Matching).

The grid is bipartite: color cells like a checkerboard (black/white based on (i+j) % 2). Two adjacent cells always have different colors. So Maximum Independent Set = N_cells - Maximum_Matching.

Steps:
1. Build bipartite graph: "black" cells (i+j even) on left, "white" cells (i+j odd) on right
2. Add edges between adjacent valid cells
3. Find maximum bipartite matching
4. MIS = total valid cells - matching size
5. Reconstruct which cells are in the MIS using König's theorem

**König's theorem reconstruction:**
- Find minimum vertex cover = N - MIS size (equals matching size)
- The MIS is the complement of the minimum vertex cover
- To find min vertex cover from max matching:
  - Find all unmatched left vertices
  - Do alternating BFS/DFS: from unmatched left vertices, go to right via unmatched edges, then back to left via matched edges
  - Min vertex cover = (left vertices NOT in Z) ∪ (right vertices IN Z)
  - MIS = complement = (left vertices IN Z) ∪ (right vertices NOT in Z)

Let me implement this with standard Hopcroft-Karp or simple augmenting path matching.

Given N,M ≤ 200, we have up to 40000 cells. Simple augmenting path matching should work within time.

```cpp
#include <bits/stdc++.h>
using namespace std;

int M, N;
int grid[205][205];
int id[205][205]; // id for left nodes (even parity) or right nodes (odd parity)
vector<int> adj[40005]; // adjacency for left nodes
int matchL[40005], matchR[40005];
int nL, nR;
bool used[40005];

bool dfs(int u) {
    for (int v : adj[u]) {
        if (!used[v]) {
            used[v] = true;
            if (matchR[v] == -1 || dfs(matchR[v])) {
                matchL[u] = v;
                matchR[v] = u;
                return true;
            }
        }
    }
    return false;
}

int dx[] = {0,0,1,-1};
int dy[] = {1,-1,0,0};

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cin >> M >> N;
    for(int i=0;i<M;i++) for(int j=0;j<N;j++) cin >> grid[i][j];
    
    nL = nR = 0;
    memset(id, -1, sizeof(id));
    for(int i=0;i<M;i++) for(int j=0;j<N;j++)
        if(grid[i][j]==1) { if((i+j)%2==0) id[i][j]=nL++; else id[i][j]=nR++; }
    
    for(int i=0;i<M;i++) for(int j=0;j<N;j++) if(grid[i][j]==1 && (i+j)%2==0) {
        for(int d=0;d<4;d++) { int ni=i+dx[d], nj=j+dy[d];
            if(ni>=0&&ni<M&&nj>=0&&nj<N&&grid[ni][nj]==1) adj[id[i][j]].push_back(id[ni][nj]);
        }
    }
    
    memset(matchL,-1,sizeof(matchL)); memset(matchR,-1,sizeof(matchR));
    int matching=0;
    for(int u=0;u<nL;u++) { memset(used,false,sizeof(bool)*nR); if(dfs(u)) matching++; }
    
    // König: find MIS
    vector<bool> inZL(nL,false), inZR(nR,false);
    queue<int> q;
    for(int u=0;u<nL;u++) if(matchL[u]==-1) { inZL[u]=true; q.push(u); }
    while(!q.empty()) { int u=q.front();q.pop();
        for(int v:adj[u]) if(!inZR[v]) { inZR[v]=true;
            if(matchR[v]!=-1&&!inZL[matchR[v]]) { inZL[matchR[v]]=true; q.push(matchR[v]); }
        }
    }
    
    int total=0; for(int i=0;i<M;i++) for(int j=0;j<N;j++) if(grid[i][j]==1) total++;
    cout << total - matching << "\n";
    for(int i=0;i<M;i++) { for(int j=0;j<N;j++) {
        if(!grid[i][j]) cout<<'#';
        else { bool inMIS = ((i+j)%2==0) ? inZL[id[i][j]] : !inZR[id[i][j]];
            cout<<(inMIS?'G':'.'); }
    } cout<<"\n"; }
}
```
