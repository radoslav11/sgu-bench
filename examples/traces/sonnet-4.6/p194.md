This is a classic lower-bound flow problem. The approach is:

1. For each edge with lower bound `l` and upper bound `c`, transform it to have lower bound 0 and upper bound `c-l`.
2. Create a super source `S` and super sink `T`.
3. For each node, compute excess = sum of lower bounds of incoming edges - sum of lower bounds of outgoing edges.
   - If excess > 0: add edge from S to node with capacity = excess
   - If excess < 0: add edge from node to T with capacity = -excess
4. Run max flow from S to T. If all edges from S are saturated (and all edges to T), a feasible flow exists.
5. The actual flow on each original edge = flow in transformed network + lower bound.

```cpp
#include <bits/stdc++.h>
using namespace std;

const int INF = 1e9;

struct Edge {
    int to, cap, flow;
};

vector<Edge> edges;
vector<vector<int>> g;
int n_nodes;
vector<int> d, ptr_;

void addEdge(int from, int to, int cap) {
    g[from].push_back(edges.size());
    edges.push_back({to, cap, 0});
    g[to].push_back(edges.size());
    edges.push_back({from, 0, 0});
}

bool bfs(int s, int t) {
    d.assign(n_nodes, -1);
    d[s] = 0;
    queue<int> q;
    q.push(s);
    while (!q.empty()) {
        int v = q.front(); q.pop();
        for (int id : g[v]) {
            if (d[edges[id].to] == -1 && edges[id].cap > edges[id].flow) {
                d[edges[id].to] = d[v] + 1;
                q.push(edges[id].to);
            }
        }
    }
    return d[t] != -1;
}

int dfs(int v, int t, int pushed) {
    if (v == t || pushed == 0) return pushed;
    for (int& cid = ptr_[v]; cid < (int)g[v].size(); cid++) {
        int id = g[v][cid];
        int to = edges[id].to;
        if (d[v] + 1 != d[to] || edges[id].cap <= edges[id].flow)
            continue;
        int tr = dfs(to, t, min(pushed, edges[id].cap - edges[id].flow));
        if (tr > 0) {
            edges[id].flow += tr;
            edges[id ^ 1].flow -= tr;
            return tr;
        }
    }
    return 0;
}

int maxflow(int s, int t) {
    int flow = 0;
    while (bfs(s, t)) {
        ptr_.assign(n_nodes, 0);
        while (int pushed = dfs(s, t, INF)) {
            flow += pushed;
        }
    }
    return flow;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    // Skip "Test #X" lines if present - handle by reading carefully
    // Actually let's just parse integers
    int N, M;
    cin >> N >> M;
    
    // S = N, T = N+1
    int S = N, T = N + 1;
    n_nodes = N + 2;
    g.assign(n_nodes, {});
    
    vector<int> ei(M), ej(M), el(M), ec(M);
    vector<int> excess(N + 2, 0);
    // edge indices in 'edges' for original edges
    vector<int> edge_id(M);
    
    for (int k = 0; k < M; k++) {
        cin >> ei[k] >> ej[k] >> el[k] >> ec[k];
        ei[k]--; ej[k]--;
        // Add edge with capacity c-l
        edge_id[k] = edges.size();
        addEdge(ei[k], ej[k], ec[k] - el[k]);
        excess[ej[k]] += el[k];
        excess[ei[k]] -= el[k];
    }
    
    int total = 0;
    for (int i = 0; i < N; i++) {
        if (excess[i] > 0) {
            addEdge(S, i, excess[i]);
            total += excess[i];
        } else if (excess[i] < 0) {
            addEdge(i, T, -excess[i]);
        }
    }
    
    int fl = maxflow(S, T);
    
    if (fl != total) {
        cout << "NO" << endl;
    } else {
        cout << "YES" << endl;
        for (int k = 0; k < M; k++) {
            cout << edges[edge_id[k]].flow + el[k] << "\n";
        }
    }
    
    return 0;
}
```
