Looking at this problem, I need to assign n programs to n computers (one program per computer) to minimize the maximum unsafety value across all assignments.

This is a classic assignment problem where I want to minimize the maximum value (minimax assignment).

**Approach:** Binary search on the answer. For a given threshold T, check if there's a perfect matching where all selected edges have unsafety ≤ T. Use bipartite matching.

The possible values of the answer are among the values in the matrix, so I can binary search over sorted unique values.

**Steps:**
1. Collect all values, sort and deduplicate.
2. Binary search on threshold T.
3. For each T, build bipartite graph with edges where unsafety[i][j] ≤ T.
4. Check if perfect matching exists (using Hopcroft-Karp or simple augmenting path).
5. Find minimum T with perfect matching, then recover the matching.

With n=500, the matrix has 250,000 values. Binary search over them: log(250000) ≈ 18 iterations. Each matching check with Hopcroft-Karp is O(E√V) = O(250000 * 22) ≈ 5.5M operations. Total: ~100M, should be fine within 1.25s.

Let me implement this:

```cpp
#include <bits/stdc++.h>
using namespace std;

int n;
int a[500][500];
vector<int> adj[500]; // adjacency for left nodes (computers)
int matchL[500], matchR[500];
int dist[500];
const int INF = 1e9;

bool bfs() {
    queue<int> q;
    for (int u = 0; u < n; u++) {
        if (matchL[u] == -1) { dist[u] = 0; q.push(u); }
        else dist[u] = INF;
    }
    bool found = false;
    while (!q.empty()) {
        int u = q.front(); q.pop();
        for (int v : adj[u]) {
            int w = matchR[v];
            if (w == -1) found = true;
            else if (dist[w] == INF) { dist[w] = dist[u]+1; q.push(w); }
        }
    }
    return found;
}

bool dfs(int u) {
    for (int v : adj[u]) {
        int w = matchR[v];
        if (w == -1 || (dist[w] == dist[u]+1 && dfs(w))) {
            matchL[u] = v; matchR[v] = u;
            return true;
        }
    }
    dist[u] = INF;
    return false;
}

bool check(int T) {
    for (int i = 0; i < n; i++) {
        adj[i].clear();
        for (int j = 0; j < n; j++)
            if (a[i][j] <= T) adj[i].push_back(j);
    }
    fill(matchL, matchL+n, -1);
    fill(matchR, matchR+n, -1);
    int matched = 0;
    while (bfs()) {
        for (int u = 0; u < n; u++)
            if (matchL[u] == -1 && dfs(u)) matched++;
    }
    return matched == n;
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(0);
    cin >> n;
    vector<int> vals;
    for (int i = 0; i < n; i++)
        for (int j = 0; j < n; j++) {
            cin >> a[i][j];
            vals.push_back(a[i][j]);
        }
    sort(vals.begin(), vals.end());
    vals.erase(unique(vals.begin(), vals.end()), vals.end());
    
    int lo = 0, hi = (int)vals.size()-1, ans = hi;
    while (lo <= hi) {
        int mid = (lo+hi)/2;
        if (check(vals[mid])) { ans = mid; hi = mid-1; }
        else lo = mid+1;
    }
    
    check(vals[ans]);
    cout << vals[ans] << "\n";
    for (int i = 0; i < n; i++)
        cout << (i+1) << " " << (matchL[i]+1) << "\n";
    return 0;
}
```
