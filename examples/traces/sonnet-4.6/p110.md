Looking at this problem, I need to simulate ray reflections off spheres.

Key steps:
1. Parse spheres (position + radius)
2. Parse ray (starting point + direction)
3. Find the closest sphere intersection
4. Reflect the ray off that sphere
5. Repeat up to 10 times (or more, tracking if >10)

**Ray-Sphere Intersection:**
Ray: P + t*D (t > 0)
Sphere: center C, radius r
|P + t*D - C|² = r²
Let V = P - C
|V + t*D|² = r²
t²|D|² + 2t(V·D) + |V|² - r² = 0

a = D·D, b = 2*(V·D), c = V·V - r²
discriminant = b² - 4ac

If disc < 0: no intersection
t = (-b ± sqrt(disc)) / (2a)

We want the smallest positive t. But we need to be careful: the starting point is on the sphere after reflection, so we need t > small_epsilon.

**Reflection:**
Normal at intersection point = (P_hit - C) / r
D_new = D - 2*(D·N)*N (D normalized or not, N is unit normal)

Let me use non-normalized D but normalize when needed.

**Important:** After reflection, the new ray starts at the hit point, and we must exclude the current sphere we just reflected from (or use epsilon).

Actually, since after reflection we start exactly on the sphere surface, we should skip t values very close to 0 for all spheres. Using an epsilon like 1e-9.

#include <bits/stdc++.h>
using namespace std;

struct Vec3 { double x,y,z; };
Vec3 add(Vec3 a, Vec3 b){return {a.x+b.x,a.y+b.y,a.z+b.z};}
Vec3 sub(Vec3 a, Vec3 b){return {a.x-b.x,a.y-b.y,a.z-b.z};}
Vec3 scale(Vec3 a, double t){return {a.x*t,a.y*t,a.z*t};}
double dot(Vec3 a, Vec3 b){return a.x*b.x+a.y*b.y+a.z*b.z;}

int main(){
    int n;
    scanf("%d",&n);
    vector<double> cx(n),cy(n),cz(n),cr(n);
    for(int i=0;i<n;i++) scanf("%lf%lf%lf%lf",&cx[i],&cy[i],&cz[i],&cr[i]);
    double p1x,p1y,p1z,p2x,p2y,p2z;
    scanf("%lf%lf%lf%lf%lf%lf",&p1x,&p1y,&p1z,&p2x,&p2y,&p2z);
    
    Vec3 P={p1x,p1y,p1z};
    Vec3 D={p2x-p1x,p2y-p1y,p2z-p1z};
    
    vector<int> result;
    int lastSphere=-1;
    
    for(int iter=0;iter<11;iter++){
        double bestT=1e18;
        int bestI=-1;
        for(int i=0;i<n;i++){
            Vec3 C={cx[i],cy[i],cz[i]};
            Vec3 V=sub(P,C);
            double a=dot(D,D);
            double b=2*dot(V,D);
            double c=dot(V,V)-cr[i]*cr[i];
            double disc=b*b-4*a*c;
            if(disc<0) continue;
            double sq=sqrt(max(0.0,disc));
            double t1=(-b-sq)/(2*a);
            double t2=(-b+sq)/(2*a);
            double t=-1;
            if(t1>1e-9) t=t1;
            else if(t2>1e-9) t=t2;
            if(t>1e-9 && t<bestT){
                bestT=t;
                bestI=i;
            }
        }
        if(bestI==-1) break;
        result.push_back(bestI+1);
        if((int)result.size()>=11) break;
        // reflect
        Vec3 hit=add(P,scale(D,bestT));
        Vec3 C={cx[bestI],cy[bestI],cz[bestI]};
        Vec3 N=scale(sub(hit,C),1.0/cr[bestI]);
        double ddn=dot(D,N);
        // normalize D first? No, reflection formula works with unnormalized D
        D=sub(D,scale(N,2*ddn));
        P=hit;
        lastSphere=bestI;
    }
    
    bool etc=(result.size()>10);
    int show=min((int)result.size(),10);
    for(int i=0;i<show;i++){
        if(i) printf(" ");
        printf("%d",result[i]);
    }
    if(etc) printf(" etc.");
    printf("\n");
}
