#include <bits/stdc++.h>
using namespace std;

int n;
const int dx[8] = {-2,-2,-1,-1,1,1,2,2};
const int dy[8] = {-1,1,-2,2,-2,2,-1,1};

inline bool inside(int x, int y) {
    return x >= 0 && x < n && y >= 0 && y < n;
}

bool attempt(int sx, int sy, int mode, vector<vector<int>>& ans) {
    ans.assign(n, vector<int>(n, 0));
    vector<vector<int>> deg(n, vector<int>(n, 0));

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            for (int k = 0; k < 8; ++k) {
                int ni = i + dx[k], nj = j + dy[k];
                if (inside(ni, nj)) deg[i][j]++;
            }
        }
    }

    int x = sx, y = sy;

    for (int step = 1; step <= n * n; ++step) {
        ans[x][y] = step;

        for (int k = 0; k < 8; ++k) {
            int nx = x + dx[k], ny = y + dy[k];
            if (inside(nx, ny) && ans[nx][ny] == 0) deg[nx][ny]--;
        }

        if (step == n * n) return true;

        int bx = -1, by = -1;
        int bestDeg = 1000000;
        long long bestTie = LLONG_MIN;

        for (int k = 0; k < 8; ++k) {
            int nx = x + dx[k], ny = y + dy[k];
            if (!inside(nx, ny) || ans[nx][ny] != 0) continue;

            int d = deg[nx][ny];
            long long tie = 0;

            if (mode == 0) {
                long long cx = 2LL * nx - (n - 1);
                long long cy = 2LL * ny - (n - 1);
                tie = cx * cx + cy * cy;
            } else if (mode == 1) {
                long long cx = 2LL * nx - (n - 1);
                long long cy = 2LL * ny - (n - 1);
                tie = -(cx * cx + cy * cy);
            } else if (mode == 2) {
                tie = 1LL * nx * n + ny;
            } else {
                int sum = 0;
                for (int t = 0; t < 8; ++t) {
                    int tx = nx + dx[t], ty = ny + dy[t];
                    if (inside(tx, ty) && ans[tx][ty] == 0) sum += deg[tx][ty];
                }
                tie = -sum;
            }

            if (d < bestDeg || (d == bestDeg && tie > bestTie)) {
                bestDeg = d;
                bestTie = tie;
                bx = nx;
                by = ny;
            }
        }

        if (bx == -1) return false;

        x = bx;
        y = by;
    }

    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n;

    if (n == 1) {
        cout << "There is solution:\n1\n";
        return 0;
    }

    if (n < 5) {
        cout << "No solution.\n";
        return 0;
    }

    vector<pair<int,int>> starts;
    starts.push_back({0, 0});
    starts.push_back({0, n - 1});
    starts.push_back({n - 1, 0});
    starts.push_back({n - 1, n - 1});
    starts.push_back({n / 2, n / 2});
    starts.push_back({0, n / 2});
    starts.push_back({n / 2, 0});
    starts.push_back({n - 1, n / 2});
    starts.push_back({n / 2, n - 1});

    if (n == 5) {
        starts.insert(starts.begin(), {0, 0});
    }

    vector<vector<int>> ans;

    for (auto [sx, sy] : starts) {
        for (int mode = 0; mode < 4; ++mode) {
            if (attempt(sx, sy, mode, ans)) {
                cout << "There is solution:\n";
                for (int i = 0; i < n; ++i) {
                    for (int j = 0; j < n; ++j) {
                        if (j) cout << ' ';
                        cout << ans[i][j];
                    }
                    cout << '\n';
                }
                return 0;
            }
        }
    }

    cout << "No solution.\n";
    return 0;
}
