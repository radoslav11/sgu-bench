```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, f;
    cin >> n >> f;

    vector<int> e(n);
    vector<bool> active(101, false);

    for (int i = 0; i < n; ++i) {
        cin >> e[i];
        active[e[i]] = true;
    }

    vector<int> answer;
    int current = f;
    int remaining = n;

    while (remaining > 0) {
        int target = -1;

        for (int i = 0; i < n; ++i) {
            if (active[e[i]]) {
                target = e[i];
                break;
            }
        }

        int dir = (target > current) ? 1 : -1;

        for (int floor = current + dir; ; floor += dir) {
            if (active[floor]) {
                answer.push_back(floor);
                active[floor] = false;
                --remaining;
            }

            if (floor == target) break;
        }

        current = target;
    }

    for (int x : answer) {
        cout << x << ' ';
    }

    return 0;
}
```
