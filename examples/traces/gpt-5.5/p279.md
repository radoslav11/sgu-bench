```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    if (!(cin >> n)) return 0;

    vector<int> a(n + 1);
    for (int i = 1; i <= n; ++i) cin >> a[i];

    vector<int> p(n);
    iota(p.begin(), p.end(), 1);

    sort(p.begin(), p.end(), [&](int x, int y) {
        if (a[x] != a[y]) return a[x] > a[y];
        return x > y;
    });

    vector<int> posRank(n + 1);
    for (int i = 0; i < n; ++i) posRank[p[i]] = i;

    vector<int> needNegAfter(n + 1, 0);

    for (int x = 1; x <= n; ++x) {
        int positiveContribution = 0;
        for (int y = x + 1; y <= n; ++y) {
            if (posRank[x] < posRank[y]) positiveContribution++;
        }

        needNegAfter[x] = a[x] - positiveContribution;

        if (needNegAfter[x] < 0 || needNegAfter[x] >= x) {
            cout << "NO\n";
            return 0;
        }
    }

    vector<int> ans;
    vector<bool> seenPositive(n + 1, false), outputNegative(n + 1, false);

    int negPtr = 0;

    for (int x : p) {
        int currentlyAfter = 0;
        for (int y = 1; y < x; ++y) {
            if (!outputNegative[y]) currentlyAfter++;
        }

        int mustMoveBefore = currentlyAfter - needNegAfter[x];

        if (mustMoveBefore < 0) {
            cout << "NO\n";
            return 0;
        }

        while (mustMoveBefore > 0) {
            while (negPtr < n && outputNegative[p[negPtr]]) negPtr++;

            if (negPtr >= n || !seenPositive[p[negPtr]]) {
                cout << "NO\n";
                return 0;
            }

            int y = p[negPtr];
            outputNegative[y] = true;
            ans.push_back(-y);

            if (y < x) mustMoveBefore--;
            negPtr++;
        }

        seenPositive[x] = true;
        ans.push_back(x);
    }

    while (negPtr < n) {
        while (negPtr < n && outputNegative[p[negPtr]]) negPtr++;
        if (negPtr >= n) break;

        int y = p[negPtr];

        if (!seenPositive[y]) {
            cout << "NO\n";
            return 0;
        }

        outputNegative[y] = true;
        ans.push_back(-y);
        negPtr++;
    }

    if ((int)ans.size() != 2 * n) {
        cout << "NO\n";
        return 0;
    }

    vector<int> wherePos(n + 1), whereNeg(n + 1);
    vector<int> positiveOrder;

    for (int i = 0; i < 2 * n; ++i) {
        int v = ans[i];
        if (v > 0) {
            wherePos[v] = i;
            positiveOrder.push_back(v);
        } else {
            whereNeg[-v] = i;
        }
    }

    if ((int)positiveOrder.size() != n) {
        cout << "NO\n";
        return 0;
    }

    vector<int> rankPositive(n + 1);
    for (int i = 0; i < n; ++i) rankPositive[positiveOrder[i]] = i;

    for (int i = 1; i <= n; ++i) {
        if (wherePos[i] > whereNeg[i]) {
            cout << "NO\n";
            return 0;
        }
    }

    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            if ((rankPositive[i] < rankPositive[j]) != (whereNeg[i] < whereNeg[j])) {
                cout << "NO\n";
                return 0;
            }
        }
    }

    vector<int> got(n + 1, 0);

    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= n; ++j) {
            if (i == j) continue;

            if (j < i) {
                if (wherePos[i] < whereNeg[j]) got[i]++;
            } else {
                if (wherePos[i] < wherePos[j]) got[i]++;
            }
        }
    }

    for (int i = 1; i <= n; ++i) {
        if (got[i] != a[i]) {
            cout << "NO\n";
            return 0;
        }
    }

    cout << "YES\n";
    for (int x : ans) cout << x << ' ';
    cout << '\n';

    return 0;
}
```
