#include <bits/stdc++.h>
using namespace std;

using int64 = long long;

int64 modpow(int64 a, int64 e, int64 mod) {
    int64 r = 1;
    while (e) {
        if (e & 1) r = r * a % mod;
        a = a * a % mod;
        e >>= 1;
    }
    return r;
}

int64 extgcd(int64 a, int64 b, int64 &x, int64 &y) {
    if (!b) {
        x = 1; y = 0;
        return a;
    }
    int64 x1, y1;
    int64 g = extgcd(b, a % b, x1, y1);
    x = y1;
    y = x1 - y1 * (a / b);
    return g;
}

int64 modinv(int64 a, int64 mod) {
    int64 x, y;
    extgcd(a, mod, x, y);
    x %= mod;
    if (x < 0) x += mod;
    return x;
}

vector<int> factor_unique(int x) {
    vector<int> f;
    for (int d = 2; 1LL * d * d <= x; d++) {
        if (x % d == 0) {
            f.push_back(d);
            while (x % d == 0) x /= d;
        }
    }
    if (x > 1) f.push_back(x);
    return f;
}

int primitive_root(int p) {
    if (p == 2) return 1;
    vector<int> fac = factor_unique(p - 1);
    for (int g = 2; ; g++) {
        bool ok = true;
        for (int q : fac) {
            if (modpow(g, (p - 1) / q, p) == 1) {
                ok = false;
                break;
            }
        }
        if (ok) return g;
    }
}

int discrete_log_generator(int g, int target, int p) {
    int order = p - 1;
    int B = (int)sqrt(order) + 1;

    unordered_map<int, int> baby;
    baby.reserve(B * 2);

    int64 cur = 1;
    for (int j = 0; j < B; j++) {
        if (!baby.count((int)cur)) baby[(int)cur] = j;
        cur = cur * g % p;
    }

    int64 factor = modinv(modpow(g, B, p), p);
    cur = target;

    for (int i = 0; i <= B; i++) {
        auto it = baby.find((int)cur);
        if (it != baby.end()) {
            int ans = i * B + it->second;
            if (ans < order) return ans;
        }
        cur = cur * factor % p;
    }

    return -1;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int n, p;
        cin >> n >> p;

        if (p == 2) {
            cout << -1 << '\n';
            continue;
        }

        int g = std::gcd(n, p - 1);
        int m = (p - 1) / g;

        int root = primitive_root(p);

        if (g == 1) {
            int e = modinv(n % (p - 1), p - 1);
            int z = modpow(2, e, p);
            cout << 1 << ' ' << 1 << ' ' << z << '\n';
            continue;
        }

        bool solved = false;

        int tries = min(p - 1, 5000);
        for (int y = 1; y <= tries && !solved; y++) {
            int h = modpow(y, n, p);
            int s = h + 1;
            if (s >= p) s -= p;
            if (s == 0) continue;

            if (modpow(s, m, p) == 1) {
                int L = discrete_log_generator(root, s, p);
                if (L >= 0 && L % g == 0) {
                    int k = L / g;
                    int a = n / g;
                    int inva = modinv(a % m, m);
                    int exp = 1LL * k * inva % m;
                    int z = modpow(root, exp, p);

                    cout << 1 << ' ' << y << ' ' << z << '\n';
                    solved = true;
                }
            }
        }

        if (solved) continue;

        int64 mult = modpow(root, g, p);
        int64 cur = 1;

        unordered_map<int, int> pos;
        pos.reserve((size_t)m * 2 + 10);

        for (int k = 0; k < m; k++) {
            pos[(int)cur] = k;
            cur = cur * mult % p;
        }

        int inva = modinv((n / g) % m, m);

        cur = 1;
        for (int k = 0; k < m && !solved; k++) {
            int s = (int)cur + 1;
            if (s >= p) s -= p;

            if (s != 0) {
                auto it = pos.find(s);
                if (it != pos.end()) {
                    int ky = k;
                    int kz = it->second;

                    int ey = 1LL * ky * inva % m;
                    int ez = 1LL * kz * inva % m;

                    int y = modpow(root, ey, p);
                    int z = modpow(root, ez, p);

                    cout << 1 << ' ' << y << ' ' << z << '\n';
                    solved = true;
                    break;
                }
            }

            cur = cur * mult % p;
        }

        if (!solved) cout << -1 << '\n';
    }

    return 0;
}
