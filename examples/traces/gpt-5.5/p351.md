#include <bits/stdc++.h>
using namespace std;

const double PI = acos(-1.0);
const double EPS = 1e-10;

struct Point {
    double x, y;
};

Point A, B, C;
double R, Dfuel;

double normAng(double a) {
    while (a < 0) a += 2 * PI;
    while (a >= 2 * PI) a -= 2 * PI;
    return a;
}

double ccwDiff(double from, double to) {
    double d = normAng(to) - normAng(from);
    if (d < 0) d += 2 * PI;
    return d;
}

Point pointOnCircle(double ang) {
    return {C.x + R * cos(ang), C.y + R * sin(ang)};
}

double dist(Point p, Point q) {
    double dx = p.x - q.x, dy = p.y - q.y;
    return sqrt(dx * dx + dy * dy);
}

double chordLen(double delta) {
    delta = fmod(fabs(delta), 2 * PI);
    if (delta > 2 * PI - delta) delta = 2 * PI - delta;
    return 2.0 * R * sin(delta / 2.0);
}

double insideLengthOfSegment(Point p, Point q) {
    double vx = q.x - p.x, vy = q.y - p.y;
    double wx = p.x - C.x, wy = p.y - C.y;

    double a = vx * vx + vy * vy;
    double b = 2 * (vx * wx + vy * wy);
    double c = wx * wx + wy * wy - R * R;

    double disc = b * b - 4 * a * c;
    if (disc <= 0) return 0.0;

    disc = sqrt(disc);
    double t1 = (-b - disc) / (2 * a);
    double t2 = (-b + disc) / (2 * a);

    double l = max(0.0, t1);
    double r = min(1.0, t2);

    if (r <= l) return 0.0;
    return (r - l) * sqrt(a);
}

vector<double> tangentAngles(Point p) {
    double dx = p.x - C.x;
    double dy = p.y - C.y;
    double d = sqrt(dx * dx + dy * dy);
    double base = atan2(dy, dx);
    double alpha = acos(R / d);

    return {normAng(base - alpha), normAng(base + alpha)};
}

struct Part {
    char type;
    Point p, q;
};

void addSegment(vector<Part>& res, Point p, Point q) {
    if (dist(p, q) > 1e-8)
        res.push_back({'S', p, q});
}

void addArc(vector<Part>& res, double from, double to, int dir) {
    double len = (dir == 1) ? ccwDiff(from, to) : ccwDiff(to, from);
    if (len < 1e-10) return;

    int pieces = max(1, (int)ceil(len / (PI / 2.0)));

    double cur = from;
    for (int i = 1; i <= pieces; i++) {
        double nxt = from + dir * len * i / pieces;
        Point p = pointOnCircle(cur);
        Point q = pointOnCircle(nxt);
        res.push_back({'A', p, q});
        cur = nxt;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> A.x >> A.y >> B.x >> B.y;
    cin >> C.x >> C.y >> R;
    cin >> Dfuel;

    if (insideLengthOfSegment(A, B) <= Dfuel + 1e-9) {
        cout << 1 << "\n";
        cout << fixed << setprecision(8);
        cout << "S " << A.x << ' ' << A.y << ' ' << B.x << ' ' << B.y << "\n";
        return 0;
    }

    vector<double> ta = tangentAngles(A);
    vector<double> tb = tangentAngles(B);

    double da = dist(A, C);
    double db = dist(B, C);
    double tangentA = sqrt(da * da - R * R);
    double tangentB = sqrt(db * db - R * R);

    double bestCost = 1e100;
    double bestTa = 0, bestTb = 0, bestDelta = 0;
    int bestDir = 1;

    for (double aa : ta) {
        for (double bb : tb) {
            for (int dir : {1, -1}) {
                double L = (dir == 1) ? ccwDiff(aa, bb) : ccwDiff(bb, aa);

                double delta;
                if (Dfuel >= 2 * R - EPS) {
                    delta = L;
                } else {
                    double theta = 2.0 * asin(min(1.0, Dfuel / (2.0 * R)));

                    if (L >= 2 * PI - theta - EPS)
                        delta = L;
                    else
                        delta = min(L, theta);
                }

                double cost = tangentA + tangentB + R * (L - delta) + chordLen(delta);

                if (cost < bestCost) {
                    bestCost = cost;
                    bestTa = aa;
                    bestTb = bb;
                    bestDelta = delta;
                    bestDir = dir;
                }
            }
        }
    }

    vector<Part> ans;

    Point Pta = pointOnCircle(bestTa);
    Point Pq = pointOnCircle(bestTa + bestDir * bestDelta);
    Point Ptb = pointOnCircle(bestTb);

    addSegment(ans, A, Pta);
    addSegment(ans, Pta, Pq);
    addArc(ans, bestTa + bestDir * bestDelta, bestTb, bestDir);
    addSegment(ans, Ptb, B);

    cout << ans.size() << "\n";
    cout << fixed << setprecision(8);
    for (auto &e : ans) {
        cout << e.type << ' '
             << e.p.x << ' ' << e.p.y << ' '
             << e.q.x << ' ' << e.q.y << "\n";
    }

    return 0;
}
