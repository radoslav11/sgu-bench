#include <bits/stdc++.h>
#include <boost/multiprecision/cpp_int.hpp>

using namespace std;
using boost::multiprecision::cpp_int;

cpp_int countFixedLength(const string& N, int m) {
    int L = (int)N.size();

    int finalCarry;
    if (L == m) {
        finalCarry = 0;
    } else if (L == m + 1) {
        if (N[0] != '1') return 0;
        finalCarry = 1;
    } else {
        return 0;
    }

    vector<int> digit(m);
    for (int i = 0; i < m; ++i) {
        digit[i] = N[L - 1 - i] - '0';
    }

    int cntInner[19] = {};
    int cntOuter[19] = {};

    for (int a = 0; a <= 9; ++a) {
        for (int b = 0; b <= 9; ++b) {
            cntInner[a + b]++;
        }
        for (int b = 1; b <= 9; ++b) {
            cntOuter[a + b]++;
        }
    }

    cpp_int dp[2][2] = {};
    dp[0][finalCarry] = 1;

    int i = 0, j = m - 1;

    while (i < j) {
        cpp_int ndp[2][2] = {};
        int* cnt = (j == m - 1 ? cntOuter : cntInner);

        for (int cl = 0; cl <= 1; ++cl) {
            for (int crout = 0; crout <= 1; ++crout) {
                if (dp[cl][crout] == 0) continue;

                for (int s = 0; s <= 18; ++s) {
                    if (cnt[s] == 0) continue;

                    int leftSum = s + cl;
                    if (leftSum % 10 != digit[i]) continue;

                    int clNext = leftSum / 10;
                    if (clNext > 1) continue;

                    int crIn = digit[j] + 10 * crout - s;
                    if (crIn < 0 || crIn > 1) continue;

                    ndp[clNext][crIn] += dp[cl][crout] * cnt[s];
                }
            }
        }

        for (int a = 0; a < 2; ++a)
            for (int b = 0; b < 2; ++b)
                dp[a][b] = ndp[a][b];

        ++i;
        --j;
    }

    cpp_int ans = 0;

    if (i == j) {
        for (int cl = 0; cl <= 1; ++cl) {
            for (int crout = 0; crout <= 1; ++crout) {
                if (dp[cl][crout] == 0) continue;

                int from = (m == 1 ? 1 : 0);
                for (int d = from; d <= 9; ++d) {
                    if (2 * d + cl == digit[i] + 10 * crout) {
                        ans += dp[cl][crout];
                    }
                }
            }
        }
    } else {
        ans = dp[0][0] + dp[1][1];
    }

    return ans;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string N;
    while (cin >> N && N != "0") {
        int L = (int)N.size();

        cpp_int ans = 0;
        ans += countFixedLength(N, L);
        if (L > 1) ans += countFixedLength(N, L - 1);

        cout << ans << '\n';
    }

    return 0;
}
