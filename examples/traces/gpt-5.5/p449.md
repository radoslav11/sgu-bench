#include <bits/stdc++.h>
using namespace std;

struct DSU {
    vector<int> p, sz;
    DSU(int n = 0) { init(n); }

    void init(int n) {
        p.resize(n + 1);
        sz.assign(n + 1, 1);
        iota(p.begin(), p.end(), 0);
    }

    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }

    int unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return a;
        if (sz[a] < sz[b]) swap(a, b);
        p[b] = a;
        sz[a] += sz[b];
        return a;
    }
};

struct Line {
    long long y;
    int idx;
    vector<int> v;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M, Q;
    cin >> N >> M >> Q;

    vector<Line> lines(M);
    for (int i = 0; i < M; i++) {
        int L;
        cin >> lines[i].y >> L;
        lines[i].idx = i;
        lines[i].v.resize(L);
        for (int j = 0; j < L; j++) cin >> lines[i].v[j];
    }

    stable_sort(lines.begin(), lines.end(), [](const Line& a, const Line& b) {
        return a.y > b.y;
    });

    int maxNodes = N + M + 5;
    vector<vector<int>> children(maxNodes);
    vector<int> minLeaf(maxNodes);

    for (int i = 1; i <= N; i++) minLeaf[i] = i;

    DSU dsu(N);
    vector<int> compNode(N + 1);
    for (int i = 1; i <= N; i++) compNode[i] = i;

    int nodeCnt = N;

    for (const auto& line : lines) {
        vector<int> roots;
        vector<int> ch;

        for (int x : line.v) {
            int r = dsu.find(x);
            if (!roots.empty() && find(roots.begin(), roots.end(), r) != roots.end())
                continue;
            roots.push_back(r);
            ch.push_back(compNode[r]);
        }

        if (ch.empty()) continue;

        ++nodeCnt;

        sort(ch.begin(), ch.end(), [&](int a, int b) {
            return minLeaf[a] < minLeaf[b];
        });

        children[nodeCnt] = ch;
        minLeaf[nodeCnt] = minLeaf[ch[0]];

        int root = roots[0];
        for (int i = 1; i < (int)roots.size(); i++) {
            root = dsu.unite(root, roots[i]);
        }
        root = dsu.find(root);
        compNode[root] = nodeCnt;
    }

    vector<int> roots;
    vector<char> seen(N + 1, 0);

    for (int i = 1; i <= N; i++) {
        int r = dsu.find(i);
        if (!seen[r]) {
            seen[r] = 1;
            roots.push_back(compNode[r]);
        }
    }

    sort(roots.begin(), roots.end(), [&](int a, int b) {
        return minLeaf[a] < minLeaf[b];
    });

    vector<int> order;
    order.reserve(N);

    vector<int> st;
    for (int i = (int)roots.size() - 1; i >= 0; i--) st.push_back(roots[i]);

    while (!st.empty()) {
        int u = st.back();
        st.pop_back();

        if (u <= N) {
            order.push_back(u);
        } else {
            for (int i = (int)children[u].size() - 1; i >= 0; i--) {
                st.push_back(children[u][i]);
            }
        }
    }

    while (Q--) {
        int k;
        cin >> k;
        cout << order[k - 1] << '\n';
    }

    return 0;
}
