#include <bits/stdc++.h>
using namespace std;

using int64 = long long;
using uint64 = unsigned long long;

const long long LIMIT = 2000000LL;

int64 A, B, C;

static inline int64 wrap_to_int64(uint64 v) {
    int64 r;
    memcpy(&r, &v, sizeof(r));
    return r;
}

static inline int64 next_value(int64 x) {
    int64 r = x % B;
    uint64 v = (uint64)A * (uint64)x + (uint64)r;
    int64 y = wrap_to_int64(v);
    return y % C;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> A >> B >> C;

    int64 x0 = 1;

    long long power = 1, lambda = 1;
    int64 tortoise = x0;
    int64 hare = next_value(x0);
    long long hareIndex = 1;

    const long long DETECT_LIMIT = 4 * LIMIT + 100;

    while (tortoise != hare) {
        if (hareIndex > DETECT_LIMIT) {
            cout << -1 << '\n';
            return 0;
        }

        if (power == lambda) {
            tortoise = hare;
            power <<= 1;
            lambda = 0;
        }

        hare = next_value(hare);
        ++lambda;
        ++hareIndex;
    }

    if (lambda > LIMIT) {
        cout << -1 << '\n';
        return 0;
    }

    long long mu = 0;
    tortoise = x0;
    hare = x0;

    for (long long i = 0; i < lambda; ++i) {
        hare = next_value(hare);
    }

    while (tortoise != hare) {
        if (mu + lambda >= LIMIT) {
            cout << -1 << '\n';
            return 0;
        }

        tortoise = next_value(tortoise);
        hare = next_value(hare);
        ++mu;
    }

    long long answer = mu + lambda;
    if (answer <= LIMIT) cout << answer << '\n';
    else cout << -1 << '\n';

    return 0;
}
