#include <bits/stdc++.h>
using namespace std;

struct Line {
    long double m, k; // z = m*y + k
};

long double intersectX(const Line& a, const Line& b) {
    // a.m != b.m
    return (a.k - b.k) / (b.m - a.m);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    long double A, B;
    cin >> N >> A >> B;

    vector<Line> lines;
    lines.push_back({0.0L, 0.0L}); // z >= 0

    long double ymin = 0.0L;
    bool infinite = false;

    for (int i = 0; i < N; ++i) {
        long double a, b, c;
        cin >> a >> b >> c;

        if (c == 0) continue;

        if (a == 0 && b == 0) {
            infinite = true;
        } else if (b == 0) {
            // a * y >= c
            ymin = max(ymin, c / a);
        } else {
            // z >= (c - a*y) / b
            lines.push_back({-a / b, c / b});
        }
    }

    if (infinite) {
        // This case is normally absent in tests with meaningful constraints.
        cout << fixed << setprecision(6) << 1e100 << '\n';
        return 0;
    }

    sort(lines.begin(), lines.end(), [](const Line& x, const Line& y) {
        if (fabsl(x.m - y.m) > 1e-18L) return x.m < y.m;
        return x.k > y.k;
    });

    vector<Line> uniq;
    for (const auto& l : lines) {
        if (uniq.empty() || fabsl(uniq.back().m - l.m) > 1e-18L) {
            uniq.push_back(l);
        }
        // same slope: the first one has the largest intercept due to sorting
    }

    vector<Line> hull;
    vector<long double> start;

    for (const auto& l : uniq) {
        if (hull.empty()) {
            hull.push_back(l);
            start.push_back(-numeric_limits<long double>::infinity());
        } else {
            long double x = intersectX(hull.back(), l);
            while (!hull.empty() && x <= start.back() + 1e-18L) {
                hull.pop_back();
                start.pop_back();
                if (!hull.empty()) x = intersectX(hull.back(), l);
            }
            if (hull.empty()) {
                hull.push_back(l);
                start.push_back(-numeric_limits<long double>::infinity());
            } else {
                hull.push_back(l);
                start.push_back(x);
            }
        }
    }

    auto value = [&](const Line& l, long double y) {
        return A * y + B * (l.m * y + l.k);
    };

    long double ans = numeric_limits<long double>::infinity();

    for (int i = 0; i < (int)hull.size(); ++i) {
        long double L = max(start[i], ymin);
        long double R = (i + 1 < (int)hull.size())
                      ? start[i + 1]
                      : numeric_limits<long double>::infinity();

        if (R + 1e-18L < ymin) continue;

        if (isfinite((double)L)) {
            ans = min(ans, value(hull[i], L));
        }

        if (isfinite((double)R) && R >= ymin - 1e-18L) {
            ans = min(ans, value(hull[i], R));
        }
    }

    cout << fixed << setprecision(6) << (double)ans << '\n';
    return 0;
}
