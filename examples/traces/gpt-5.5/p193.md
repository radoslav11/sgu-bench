#include <bits/stdc++.h>
using namespace std;

string trimLeadingZeros(string s) {
    int i = 0;
    while (i + 1 < (int)s.size() && s[i] == '0') i++;
    return s.substr(i);
}

string div2(const string& s) {
    string res;
    int carry = 0;
    for (char c : s) {
        int x = carry * 10 + (c - '0');
        res.push_back(char('0' + x / 2));
        carry = x % 2;
    }
    return trimLeadingZeros(res);
}

string subSmall(string s, int x) {
    int i = (int)s.size() - 1;
    while (x > 0 && i >= 0) {
        int d = s[i] - '0';
        int need = x % 10;
        x /= 10;

        if (d < need) {
            s[i] = char('0' + d + 10 - need);
            int j = i - 1;
            while (s[j] == '0') {
                s[j] = '9';
                j--;
            }
            s[j]--;
        } else {
            s[i] = char('0' + d - need);
        }
        i--;
    }
    return trimLeadingZeros(s);
}

int mod4(const string& s) {
    int len = s.size();
    int val = 0;
    if (len >= 2) val = (s[len - 2] - '0') * 10 + (s[len - 1] - '0');
    else val = s[len - 1] - '0';
    return val % 4;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string token, n;
    while (cin >> token) {
        bool ok = !token.empty();
        for (char c : token) {
            if (!isdigit((unsigned char)c)) {
                ok = false;
                break;
            }
        }
        if (ok) n = token;
    }

    n = trimLeadingZeros(n);

    string half = div2(n);

    if ((n.back() - '0') % 2 == 1) {
        cout << half << '\n';
    } else {
        if (mod4(n) == 0) {
            cout << subSmall(half, 1) << '\n';
        } else {
            cout << subSmall(half, 2) << '\n';
        }
    }

    return 0;
}
