#include <bits/stdc++.h>
using namespace std;

struct Pt {
    double x, y;
};

struct Lin {
    double a, b; // a*p + b
};

vector<Pt> targetPoly, squarePoly;

double thetaGlobal;
double cu, su, cn, sn;

double dotu(const Pt& p) { return cu * p.x + su * p.y; }
double dotn(const Pt& p) { return cn * p.x + sn * p.y; }

bool getTwoLines(const vector<Pt>& poly, double pmid, vector<Lin>& res) {
    res.clear();
    int m = (int)poly.size();

    for (int i = 0; i < m; ++i) {
        Pt A = poly[i], B = poly[(i + 1) % m];
        double nA = dotn(A), nB = dotn(B);
        double sA = dotu(A), sB = dotu(B);

        if ((nA < pmid && pmid < nB) || (nB < pmid && pmid < nA)) {
            double den = nB - nA;
            if (fabs(den) < 1e-14) continue;

            double a = (sB - sA) / den;
            double b = sA - a * nA;
            res.push_back({a, b});
        }
    }

    if (res.size() != 2) return false;

    double v0 = res[0].a * pmid + res[0].b;
    double v1 = res[1].a * pmid + res[1].b;
    if (v0 > v1) swap(res[0], res[1]);

    return true;
}

double intCubeLinear(double a, double b, double l, double r) {
    // Integral of (a*p+b)^3 dp from l to r
    double r2 = r * r, r3 = r2 * r, r4 = r3 * r;
    double l2 = l * l, l3 = l2 * l, l4 = l3 * l;

    return (a * a * a) * (r4 - l4) / 4.0
         + (a * a * b) * (r3 - l3)
         + (1.5 * a * b * b) * (r2 - l2)
         + (b * b * b) * (r - l);
}

double innerIntegral(double theta) {
    cu = cos(theta);
    su = sin(theta);
    cn = -su;
    sn = cu;

    vector<double> ps;
    for (auto &p : squarePoly) ps.push_back(dotn(p));
    for (auto &p : targetPoly) ps.push_back(dotn(p));

    sort(ps.begin(), ps.end());

    vector<double> q;
    for (double x : ps) {
        if (q.empty() || fabs(q.back() - x) > 1e-10) q.push_back(x);
    }

    double ans = 0.0;

    for (int i = 0; i + 1 < (int)q.size(); ++i) {
        double l = q[i], r = q[i + 1];
        if (r - l < 1e-12) continue;

        double mid = (l + r) * 0.5;

        vector<Lin> sq, tg;
        if (!getTwoLines(squarePoly, mid, sq)) continue;
        if (!getTwoLines(targetPoly, mid, tg)) continue;

        // Square interval [sq[0], sq[1]], target interval [tg[0], tg[1]]
        // Contribution:
        // (L^3 - left^3 - right^3) / 6
        // L     = sq[1] - sq[0]
        // left  = tg[0] - sq[0]
        // right = sq[1] - tg[1]

        double La = sq[1].a - sq[0].a;
        double Lb = sq[1].b - sq[0].b;

        double Aa = tg[0].a - sq[0].a;
        double Ab = tg[0].b - sq[0].b;

        double Ra = sq[1].a - tg[1].a;
        double Rb = sq[1].b - tg[1].b;

        double val = intCubeLinear(La, Lb, l, r)
                   - intCubeLinear(Aa, Ab, l, r)
                   - intCubeLinear(Ra, Rb, l, r);

        ans += val / 6.0;
    }

    return ans;
}

double simpson(double l, double r, double fl, double fm, double fr) {
    return (r - l) * (fl + 4.0 * fm + fr) / 6.0;
}

double adaptive(double l, double r, double fl, double fm, double fr,
                double whole, double eps, int depth) {
    double m = (l + r) * 0.5;
    double lm = (l + m) * 0.5;
    double rm = (m + r) * 0.5;

    double flm = innerIntegral(lm);
    double frm = innerIntegral(rm);

    double left = simpson(l, m, fl, flm, fm);
    double right = simpson(m, r, fm, frm, fr);

    double diff = left + right - whole;

    if (depth <= 0 || fabs(diff) <= 15.0 * eps) {
        return left + right + diff / 15.0;
    }

    return adaptive(l, m, fl, flm, fm, left, eps * 0.5, depth - 1)
         + adaptive(m, r, fm, frm, fr, right, eps * 0.5, depth - 1);
}

double integrateTheta(double l, double r, double eps) {
    if (r - l < 1e-12) return 0.0;

    double m = (l + r) * 0.5;
    double fl = innerIntegral(l);
    double fm = innerIntegral(m);
    double fr = innerIntegral(r);
    double whole = simpson(l, r, fl, fm, fr);

    return adaptive(l, r, fl, fm, fr, whole, eps, 18);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    targetPoly.resize(N);
    for (int i = 0; i < N; ++i) {
        cin >> targetPoly[i].x >> targetPoly[i].y;
    }

    squarePoly = {
        {0, 0},
        {100, 0},
        {100, 100},
        {0, 100}
    };

    vector<Pt> all = squarePoly;
    for (auto &p : targetPoly) all.push_back(p);

    vector<double> cuts;
    cuts.push_back(0.0);
    cuts.push_back(M_PI);

    int M = all.size();
    for (int i = 0; i < M; ++i) {
        for (int j = i + 1; j < M; ++j) {
            double dx = all[j].x - all[i].x;
            double dy = all[j].y - all[i].y;
            double ang = atan2(dy, dx);
            while (ang < 0) ang += M_PI;
            while (ang >= M_PI) ang -= M_PI;
            cuts.push_back(ang);
        }
    }

    sort(cuts.begin(), cuts.end());

    vector<double> uniq;
    for (double x : cuts) {
        if (uniq.empty() || fabs(uniq.back() - x) > 1e-10) {
            uniq.push_back(x);
        }
    }

    double integral = 0.0;
    double totalEps = 1e-4;

    for (int i = 0; i + 1 < (int)uniq.size(); ++i) {
        double l = uniq[i], r = uniq[i + 1];
        if (r - l < 1e-10) continue;

        // Avoid evaluating exactly at critical directions.
        double delta = min(1e-9, (r - l) * 1e-4);
        integral += integrateTheta(l + delta, r - delta, totalEps / uniq.size());
    }

    // Oriented pairs give factor 2.
    double winningMeasure = 2.0 * integral;

    double probability = winningMeasure / 100000000.0;

    cout.setf(ios::fixed);
    cout << setprecision(10) << probability << '\n';

    return 0;
}
