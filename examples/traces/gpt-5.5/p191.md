```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    char start;
    string s;
    cin >> start >> s;

    auto other = [](char c) {
        return c == 'A' ? 'B' : 'A';
    };

    vector<char> st;
    st.push_back(start);

    int pos = 0;
    bool ok = true;

    while (!st.empty() && ok) {
        char c = st.back();
        st.pop_back();

        if (pos >= (int)s.size()) {
            ok = false;
            break;
        }

        if (s[pos] == other(c)) {
            pos++;
        } else if (s[pos] == c) {
            pos++;
            st.push_back(c);
            st.push_back(other(c));
        } else {
            ok = false;
        }
    }

    if (pos != (int)s.size()) ok = false;

    cout << (ok ? "YES" : "NO") << '\n';
    return 0;
}
```
