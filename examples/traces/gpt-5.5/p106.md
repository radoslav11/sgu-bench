#include <bits/stdc++.h>
using namespace std;

using int64 = long long;

int64 gcd_ext(int64 a, int64 b, int64 &x, int64 &y) {
    if (b == 0) {
        x = 1;
        y = 0;
        return a;
    }
    int64 x1, y1;
    int64 g = gcd_ext(b, a % b, x1, y1);
    x = y1;
    y = x1 - (a / b) * y1;
    return g;
}

int64 floor_div(int64 a, int64 b) {
    int64 q = a / b;
    int64 r = a % b;
    if (r != 0 && ((r > 0) != (b > 0))) q--;
    return q;
}

int64 ceil_div(int64 a, int64 b) {
    return -floor_div(-a, b);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int64 a, b, c, x1, x2, y1, y2;
    cin >> a >> b >> c >> x1 >> x2 >> y1 >> y2;

    int64 rhs = -c;

    if (a == 0 && b == 0) {
        if (c == 0) {
            cout << (x2 - x1 + 1) * (y2 - y1 + 1) << '\n';
        } else {
            cout << 0 << '\n';
        }
        return 0;
    }

    if (a == 0) {
        if (rhs % b != 0) {
            cout << 0 << '\n';
            return 0;
        }
        int64 y = rhs / b;
        if (y1 <= y && y <= y2) cout << x2 - x1 + 1 << '\n';
        else cout << 0 << '\n';
        return 0;
    }

    if (b == 0) {
        if (rhs % a != 0) {
            cout << 0 << '\n';
            return 0;
        }
        int64 x = rhs / a;
        if (x1 <= x && x <= x2) cout << y2 - y1 + 1 << '\n';
        else cout << 0 << '\n';
        return 0;
    }

    int64 x0, y0;
    int64 g = gcd_ext(llabs(a), llabs(b), x0, y0);

    if (rhs % g != 0) {
        cout << 0 << '\n';
        return 0;
    }

    if (a < 0) x0 = -x0;
    if (b < 0) y0 = -y0;

    int64 mul = rhs / g;
    x0 *= mul;
    y0 *= mul;

    int64 dx = b / g;
    int64 dy = -a / g;

    int64 l = LLONG_MIN / 4;
    int64 r = LLONG_MAX / 4;

    auto intersect = [&](int64 p, int64 q, int64 L, int64 R) {
        int64 nl, nr;
        if (q > 0) {
            nl = ceil_div(L - p, q);
            nr = floor_div(R - p, q);
        } else {
            nl = ceil_div(R - p, q);
            nr = floor_div(L - p, q);
        }
        l = max(l, nl);
        r = min(r, nr);
    };

    intersect(x0, dx, x1, x2);
    intersect(y0, dy, y1, y2);

    if (l > r) cout << 0 << '\n';
    else cout << r - l + 1 << '\n';

    return 0;
}
