#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<long double> f(n + 1, 0.0L);
    f[0] = 1.0L;

    long double numerator = 0.0L;
    long double denominator = 0.0L;

    for (int k = 2; k <= n; ++k) {
        vector<long double> g(n + 1, 0.0L);

        for (int i = 0; i <= n; ++i) {
            if (f[i] == 0.0L) continue;

            long double term = 1.0L;
            for (int t = 0; i + t * k <= n; ++t) {
                g[i + t * k] += f[i] * term;
                term /= (long double)k * (t + 1);
            }
        }

        f.swap(g);

        numerator += f[n - k];

        if (k == n) {
            denominator = f[n];
        }
    }

    long double answer = numerator / denominator;

    cout << fixed << setprecision(12) << (double)answer << '\n';

    return 0;
}
