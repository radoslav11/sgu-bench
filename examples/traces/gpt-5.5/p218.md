#include <bits/stdc++.h>
using namespace std;

int n;
vector<vector<int>> a;
vector<int> pairU, pairV, distU;
int limitValue;

bool bfs() {
    queue<int> q;
    bool foundFreeRight = false;

    for (int u = 0; u < n; ++u) {
        if (pairU[u] == -1) {
            distU[u] = 0;
            q.push(u);
        } else {
            distU[u] = -1;
        }
    }

    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (int v = 0; v < n; ++v) {
            if (a[u][v] <= limitValue) {
                int nextU = pairV[v];
                if (nextU == -1) {
                    foundFreeRight = true;
                } else if (distU[nextU] == -1) {
                    distU[nextU] = distU[u] + 1;
                    q.push(nextU);
                }
            }
        }
    }

    return foundFreeRight;
}

bool dfs(int u) {
    for (int v = 0; v < n; ++v) {
        if (a[u][v] <= limitValue) {
            int nextU = pairV[v];
            if (nextU == -1 || (distU[nextU] == distU[u] + 1 && dfs(nextU))) {
                pairU[u] = v;
                pairV[v] = u;
                return true;
            }
        }
    }

    distU[u] = -1;
    return false;
}

bool canMatch(int lim) {
    limitValue = lim;
    pairU.assign(n, -1);
    pairV.assign(n, -1);
    distU.assign(n, -1);

    int matching = 0;

    while (bfs()) {
        for (int u = 0; u < n; ++u) {
            if (pairU[u] == -1 && dfs(u)) {
                ++matching;
            }
        }
    }

    return matching == n;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n;

    a.assign(n, vector<int>(n));
    vector<int> values;
    values.reserve(n * n);

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cin >> a[i][j];
            values.push_back(a[i][j]);
        }
    }

    sort(values.begin(), values.end());
    values.erase(unique(values.begin(), values.end()), values.end());

    int l = 0, r = (int)values.size() - 1;

    while (l < r) {
        int mid = (l + r) / 2;
        if (canMatch(values[mid])) {
            r = mid;
        } else {
            l = mid + 1;
        }
    }

    int answer = values[l];
    canMatch(answer);

    cout << answer << '\n';
    for (int i = 0; i < n; ++i) {
        cout << i + 1 << ' ' << pairU[i] + 1 << '\n';
    }

    return 0;
}
