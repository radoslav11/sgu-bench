#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    long long catalan = 1;
    for (int i = 0; i < n; ++i) {
        catalan = catalan * (2LL * (2 * i + 1)) / (i + 2);
    }

    cout << catalan << '\n';
    return 0;
}
