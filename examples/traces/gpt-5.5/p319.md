#include <bits/stdc++.h>
using namespace std;

struct Rect {
    long long x1, y1, x2, y2;
    long long area;
};

struct Event {
    long long x;
    int type; // 0 = remove, 1 = add
    int id;
    bool operator<(const Event& other) const {
        if (x != other.x) return x < other.x;
        return type < other.type; // remove before add
    }
};

struct Edge {
    long long y;
    int type; // 0 = lower edge, 1 = upper edge
    int id;
    bool operator<(const Edge& other) const {
        if (y != other.y) return y < other.y;
        if (type != other.type) return type < other.type;
        return id < other.id;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    long long W, H;
    cin >> W >> H;

    vector<Rect> rect(N + 1);
    rect[0] = {0, 0, W, H, W * H};

    vector<Event> events;
    events.reserve(2 * N);

    for (int i = 1; i <= N; i++) {
        long long x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;

        if (x1 > x2) swap(x1, x2);
        if (y1 > y2) swap(y1, y2);

        rect[i] = {x1, y1, x2, y2, (x2 - x1) * (y2 - y1)};

        events.push_back({x1, 1, i});
        events.push_back({x2, 0, i});
    }

    sort(events.begin(), events.end());

    vector<int> parent(N + 1, 0);
    vector<long long> childSum(N + 1, 0);

    set<Edge> active;

    for (const Event& e : events) {
        int id = e.id;

        if (e.type == 0) {
            active.erase(Edge{rect[id].y1, 0, id});
            active.erase(Edge{rect[id].y2, 1, id});
        } else {
            long long y = rect[id].y1;

            auto it = active.lower_bound(Edge{y, -1, -1});

            if (it == active.begin()) {
                parent[id] = 0;
            } else {
                --it;
                if (it->type == 0) {
                    parent[id] = it->id;
                } else {
                    parent[id] = parent[it->id];
                }
            }

            childSum[parent[id]] += rect[id].area;

            active.insert(Edge{rect[id].y1, 0, id});
            active.insert(Edge{rect[id].y2, 1, id});
        }
    }

    vector<long long> answer;
    answer.reserve(N + 1);

    for (int i = 0; i <= N; i++) {
        answer.push_back(rect[i].area - childSum[i]);
    }

    sort(answer.begin(), answer.end());

    for (int i = 0; i <= N; i++) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';

    return 0;
}
