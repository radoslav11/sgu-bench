#include <bits/stdc++.h>
using namespace std;

const double EPS = 1e-12;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int X;
    cin >> X;

    /*
      First: Big Man's "solo" optimal strategy.

      For every possible stopped score s, compute the probability that a player
      with score s wins against an optimal croupier whose only goal is to
      minimize that player's win probability.
    */
    double soloWin[22];

    for (int s = 1; s <= 21; ++s) {
        double f[28];

        for (int t = 22; t <= 27; ++t) {
            f[t] = 1.0; // croupier busts, player wins
        }

        for (int t = 21; t >= 0; --t) {
            double roll = 0.0;
            for (int d = 1; d <= 6; ++d) roll += f[t + d];
            roll /= 6.0;

            if (t == 0) {
                f[t] = roll; // must roll at least once
            } else {
                double stop = (t < s) ? 1.0 : 0.0;
                // Croupier minimizes player's win probability.
                // On tie, prefers to roll.
                if (stop < roll - EPS) f[t] = stop;
                else f[t] = roll;
            }
        }

        soloWin[s] = f[0];
    }

    /*
      Big Man's DP and final-score distribution.
      Index 22 denotes bust.
    */
    double bmValue[28];

    for (int t = 22; t <= 27; ++t) bmValue[t] = 0.0;

    for (int t = 21; t >= 1; --t) {
        double roll = 0.0;
        for (int d = 1; d <= 6; ++d) roll += bmValue[t + d];
        roll /= 6.0;

        double stop = soloWin[t];

        // Big Man maximizes his own win probability.
        // On tie, prefers to roll.
        if (stop > roll + EPS) bmValue[t] = stop;
        else bmValue[t] = roll;
    }

    bmValue[0] = 0.0;
    for (int d = 1; d <= 6; ++d) bmValue[0] += bmValue[d];
    bmValue[0] /= 6.0;

    vector<double> bmDist(23, 0.0);

    function<void(int, double)> genBM = [&](int t, double prob) {
        if (t > 21) {
            bmDist[22] += prob;
            return;
        }

        if (t == 0) {
            for (int d = 1; d <= 6; ++d) genBM(t + d, prob / 6.0);
            return;
        }

        double roll = 0.0;
        for (int d = 1; d <= 6; ++d) roll += bmValue[t + d];
        roll /= 6.0;

        double stop = soloWin[t];

        if (stop > roll + EPS) {
            bmDist[t] += prob;
        } else {
            for (int d = 1; d <= 6; ++d) genBM(t + d, prob / 6.0);
        }
    };

    genBM(0, 1.0);

    auto playerWins = [](int score, int croupierScore) -> bool {
        if (score > 21) return false;
        if (croupierScore > 21) return true;
        return croupierScore < score;
    };

    double answer = 0.0;

    /*
      For every possible Big Man final result, Andrew plays optimally knowing it.
    */
    for (int bmScore = 1; bmScore <= 22; ++bmScore) {
        if (bmDist[bmScore] == 0.0) continue;

        /*
          andrewStopWin[a] = probability Andrew wins if he stops with a,
          while croupier uses the strategy minimizing total casino payout:
              X * BigManWin + 1 * AndrewWin
        */
        double andrewStopWin[22];

        for (int a = 1; a <= 21; ++a) {
            double cost[28];
            double awin[28];

            for (int t = 22; t <= 27; ++t) {
                bool bw = (bmScore <= 21);
                bool aw = true;

                cost[t] = X * (bw ? 1.0 : 0.0) + (aw ? 1.0 : 0.0);
                awin[t] = 1.0;
            }

            for (int t = 21; t >= 0; --t) {
                double rollCost = 0.0;
                double rollAWin = 0.0;

                for (int d = 1; d <= 6; ++d) {
                    rollCost += cost[t + d];
                    rollAWin += awin[t + d];
                }

                rollCost /= 6.0;
                rollAWin /= 6.0;

                if (t == 0) {
                    cost[t] = rollCost;
                    awin[t] = rollAWin;
                } else {
                    bool bw = playerWins(bmScore, t);
                    bool aw = playerWins(a, t);

                    double stopCost = X * (bw ? 1.0 : 0.0) + (aw ? 1.0 : 0.0);
                    double stopAWin = aw ? 1.0 : 0.0;

                    // Croupier minimizes total payout.
                    // On tie, prefers to roll.
                    if (stopCost < rollCost - EPS) {
                        cost[t] = stopCost;
                        awin[t] = stopAWin;
                    } else {
                        cost[t] = rollCost;
                        awin[t] = rollAWin;
                    }
                }
            }

            andrewStopWin[a] = awin[0];
        }

        /*
          Andrew's own optimal stopping DP.
        */
        double avalue[28];

        for (int t = 22; t <= 27; ++t) avalue[t] = 0.0;

        for (int t = 21; t >= 1; --t) {
            double roll = 0.0;
            for (int d = 1; d <= 6; ++d) roll += avalue[t + d];
            roll /= 6.0;

            double stop = andrewStopWin[t];

            // Andrew maximizes his own win probability.
            // On tie, prefers to roll.
            if (stop > roll + EPS) avalue[t] = stop;
            else avalue[t] = roll;
        }

        avalue[0] = 0.0;
        for (int d = 1; d <= 6; ++d) avalue[0] += avalue[d];
        avalue[0] /= 6.0;

        answer += bmDist[bmScore] * avalue[0];
    }

    cout << fixed << setprecision(6) << answer << '\n';

    return 0;
}
