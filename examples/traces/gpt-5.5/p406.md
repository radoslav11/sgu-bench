#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<vector<int>> sequences(n);
    vector<set<int>> present(n);

    for (int i = 0; i < n; ++i) {
        int k;
        cin >> k;
        sequences[i].resize(k);
        for (int j = 0; j < k; ++j) {
            cin >> sequences[i][j];
            present[i].insert(sequences[i][j]);
        }
    }

    for (int q = 0; q < m; ++q) {
        int l;
        cin >> l;

        vector<int> mustHave, mustNotHave;

        for (int i = 0; i < l; ++i) {
            int b;
            cin >> b;
            if (b > 0) mustHave.push_back(b);
            else mustNotHave.push_back(-b);
        }

        vector<int> answer;

        for (int i = 0; i < n; ++i) {
            bool ok = true;

            for (int x : mustHave) {
                if (!present[i].count(x)) {
                    ok = false;
                    break;
                }
            }

            if (!ok) continue;

            for (int x : mustNotHave) {
                if (present[i].count(x)) {
                    ok = false;
                    break;
                }
            }

            if (ok) answer.push_back(i);
        }

        cout << answer.size() << '\n';

        for (int idx : answer) {
            cout << sequences[idx]..size();
            for (int x : sequences[idx]) {
                cout << ' ' << x;
            }
            cout << '\n';
        }
    }

    return 0;
}
