```cpp
#include <bits/stdc++.h>
using namespace std;

struct Rec {
    char type;
    string s;
    int x;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<Rec> a(N);
    set<string> usedNames;

    for (int i = 0; i < N; i++) {
        cin >> a[i].type;
        if (a[i].type == '=') {
            cin >> a[i].x;
        } else {
            cin >> a[i].s;
            usedNames.insert(a[i].s);
        }
    }

    auto genName = [&]() {
        static int id = 0;
        while (true) {
            int v = id++;
            string t = "a";
            do {
                t.push_back(char('a' + v % 26));
                v /= 26;
            } while (v);
            if (!usedNames.count(t)) {
                usedNames.insert(t);
                return t;
            }
        }
    };

    auto nextEvent = [&](const string& name, int pos) {
        for (int i = pos + 1; i < N; i++) {
            if (a[i].type != '=' && a[i].s == name) {
                return a[i].type;
            }
        }
        return '?';
    };

    vector<string> ans;
    set<string> insideNamed;
    vector<string> insideAnon;

    int cnt = 0;

    auto addLine = [&](char c, const string& name) {
        string line;
        line.push_back(c);
        line.push_back(' ');
        line += name;
        ans.push_back(line);
    };

    auto addEq = [&](int x) {
        ans.push_back("= " + to_string(x));
    };

    for (int i = 0; i < N; i++) {
        if (a[i].type == '+') {
            string name = a[i].s;

            if (insideNamed.count(name)) {
                addLine('-', name);
                insideNamed.erase(name);
                cnt--;
            }

            addLine('+', name);
            insideNamed.insert(name);
            cnt++;
        } else if (a[i].type == '-') {
            string name = a[i].s;

            if (!insideNamed.count(name)) {
                addLine('+', name);
                insideNamed.insert(name);
                cnt++;
            }

            addLine('-', name);
            insideNamed.erase(name);
            cnt--;
        } else {
            int need = a[i].x;

            while (cnt < need) {
                string name = genName();
                addLine('+', name);
                insideAnon.push_back(name);
                cnt++;
            }

            while (cnt > need) {
                if (!insideAnon.empty()) {
                    string name = insideAnon.back();
                    insideAnon.pop_back();
                    addLine('-', name);
                    cnt--;
                } else {
                    string best;
                    int bestType = 3;

                    for (const string& name : insideNamed) {
                        char ne = nextEvent(name, i);
                        int tp;
                        if (ne == '?' || ne == '+') tp = 0;
                        else tp = 1;

                        if (tp < bestType) {
                            bestType = tp;
                            best = name;
                        }
                    }

                    addLine('-', best);
                    insideNamed.erase(best);
                    cnt--;
                }
            }

            addEq(need);
        }
    }

    cout << ans.size() << '\n';
    for (string &s : ans) {
        cout << s << '\n';
    }

    return 0;
}
```
