#include <bits/stdc++.h>
using namespace std;

using int64 = long long;

int64 A, B;
vector<int> primes;
vector<int64> path;
unordered_set<int64> used;
int nodes = 0;
const int NODE_LIMIT = 20000;
const int DEPTH_LIMIT = 500;

void sieve() {
    const int N = 1000000;
    vector<bool> is(N + 1, true);
    is[0] = is[1] = false;
    for (int i = 2; i <= N; ++i) {
        if (is[i]) {
            primes.push_back(i);
            if ((int64)i * i <= N) {
                for (int64 j = (int64)i * i; j <= N; j += i)
                    is[(int)j] = false;
            }
        }
    }
}

bool isPrime(int64 x) {
    if (x < 2) return false;
    for (int p : primes) {
        if ((int64)p * p > x) break;
        if (x % p == 0) return x == p;
    }
    return true;
}

vector<pair<int64,int>> factorize(int64 x) {
    vector<pair<int64,int>> f;
    for (int p : primes) {
        if ((int64)p * p > x) break;
        if (x % p == 0) {
            int c = 0;
            while (x % p == 0) {
                x /= p;
                ++c;
            }
            f.push_back({p, c});
        }
    }
    if (x > 1) f.push_back({x, 1});
    return f;
}

void genDivsRec(const vector<pair<int64,int>>& f, int idx, int64 cur, vector<int64>& divs) {
    if (idx == (int)f.size()) {
        divs.push_back(cur);
        return;
    }
    int64 val = 1;
    for (int i = 0; i <= f[idx].second; ++i) {
        genDivsRec(f, idx + 1, cur * val, divs);
        val *= f[idx].first;
    }
}

vector<int64> goodPrev(int64 x) {
    auto f = factorize(x);
    vector<int64> divs;
    genDivsRec(f, 0, 1, divs);

    vector<int64> res;
    for (int64 d : divs) {
        if (d <= 1) continue;
        if (d >= x / 2) continue;
        if (x - d < A) continue;
        res.push_back(x - d);
    }

    sort(res.begin(), res.end(), greater<int64>());
    res.erase(unique(res.begin(), res.end()), res.end());

    sort(res.begin(), res.end(), [](int64 u, int64 v) {
        return u < v;
    });

    return res;
}

bool dfs(int64 x) {
    if (++nodes > NODE_LIMIT) return false;
    if ((int)path.size() > DEPTH_LIMIT) return false;
    if (x == A) return true;
    if (x < A) return false;

    used.insert(x);

    vector<int64> nxt = goodPrev(x);

    sort(nxt.begin(), nxt.end(), [](int64 u, int64 v) {
        int64 du = llabs(u - A);
        int64 dv = llabs(v - A);
        if (du != dv) return du < dv;
        return u > v;
    });

    for (int64 y : nxt) {
        if (used.count(y)) continue;
        path.push_back(y);
        if (dfs(y)) return true;
        path.pop_back();
    }

    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    sieve();

    cin >> A >> B;

    if (isPrime(A) || isPrime(B)) {
        cout << "Impossible\n";
        return 0;
    }

    path.push_back(B);

    if (!dfs(B)) {
        cout << "Impossible\n";
        return 0;
    }

    reverse(path.begin(), path.end());

    if ((int)path.size() > 500) {
        cout << "Impossible\n";
        return 0;
    }

    for (int64 x : path) {
        cout << x << '\n';
    }

    return 0;
}
