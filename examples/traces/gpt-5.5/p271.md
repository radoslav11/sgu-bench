#include <bits/stdc++.h>
using namespace std;

struct Node {
    string val;
    unsigned pr;
    int sz;
    bool rev;
    Node *l, *r;

    Node(const string& v, unsigned p)
        : val(v), pr(p), sz(1), rev(false), l(nullptr), r(nullptr) {}
};

int sz(Node* t) {
    return t ? t->sz : 0;
}

void upd(Node* t) {
    if (t) t->sz = 1 + sz(t->l) + sz(t->r);
}

void push(Node* t) {
    if (!t || !t->rev) return;
    t->rev = false;
    swap(t->l, t->r);
    if (t->l) t->l->rev ^= true;
    if (t->r) t->r->rev ^= true;
}

uint32_t rng_state = 123456789;

uint32_t rng() {
    rng_state ^= rng_state << 13;
    rng_state ^= rng_state >> 17;
    rng_state ^= rng_state << 5;
    return rng_state;
}

void split(Node* t, int leftSize, Node*& a, Node*& b) {
    if (!t) {
        a = b = nullptr;
        return;
    }

    push(t);

    if (sz(t->l) >= leftSize) {
        split(t->l, leftSize, a, t->l);
        b = t;
        upd(b);
    } else {
        split(t->r, leftSize - sz(t->l) - 1, t->r, b);
        a = t;
        upd(a);
    }
}

Node* merge(Node* a, Node* b) {
    if (!a) return b;
    if (!b) return a;

    if (a->pr > b->pr) {
        push(a);
        a->r = merge(a->r, b);
        upd(a);
        return a;
    } else {
        push(b);
        b->l = merge(a, b->l);
        upd(b);
        return b;
    }
}

void output(Node* t) {
    if (!t) return;
    push(t);
    output(t->l);
    cout << t->val << '\n';
    output(t->r);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M, K;
    cin >> N >> M >> K;

    Node* root = nullptr;

    for (int i = 0; i < N; ++i) {
        string s;
        cin >> s;
        root = merge(root, new Node(s, rng()));
    }

    for (int i = 0; i < M; ++i) {
        string op;
        cin >> op;

        if (op[0] == 'A') {
            int l = op.find('(');
            int r = op.find(')');
            string name = op.substr(l + 1, r - l - 1);
            root = merge(new Node(name, rng()), root);
        } else {
            int cnt = min(K, sz(root));
            if (cnt > 1) {
                Node *a, *b;
                split(root, cnt, a, b);
                a->rev ^= true;
                root = merge(a, b);
            }
        }
    }

    output(root);

    return 0;
}
