#include <bits/stdc++.h>
using namespace std;

long long parseMicrometers(const string& s) {
    size_t dot = s.find('.');
    long long whole = 0;
    string frac;

    if (dot == string::npos) {
        whole = stoll(s);
        frac = "";
    } else {
        whole = stoll(s.substr(0, dot));
        frac = s.substr(dot + 1);
    }

    while ((int)frac.size() < 6) frac += '0';
    if ((int)frac.size() > 6) frac = frac.substr(0, 6);

    long long f = frac.empty() ? 0 : stoll(frac);
    return whole * 1000000LL + f;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<pair<long long, int>> pos, neg;
    vector<int> zero;

    long long total = 0;
    const long long BASE = 2000000LL;

    for (int i = 1; i <= N; ++i) {
        string s;
        cin >> s;

        long long h = parseMicrometers(s);
        long long d = h - BASE;
        total += d;

        if (d > 0) pos.push_back({d, i});
        else if (d < 0) neg.push_back({d, i});
        else zero.push_back(i);
    }

    if (total != 0) {
        cout << "no\n";
        return 0;
    }

    vector<int> ans;
    long long cur = 0;

    while (!pos.empty() || !neg.empty()) {
        if (cur <= 0) {
            if (pos.empty()) break;
            cur += pos.back().first;
            ans.push_back(pos.back().second);
            pos.pop_back();
        } else {
            if (neg.empty()) break;
            cur += neg.back().first;
            ans.push_back(neg.back().second);
            neg.pop_back();
        }
    }

    for (int id : zero) ans.push_back(id);

    if ((int)ans.size() != N) {
        cout << "no\n";
        return 0;
    }

    cout << "yes\n";
    for (int i = 0; i < N; ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';

    return 0;
}
