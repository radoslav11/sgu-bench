#include <bits/stdc++.h>
using namespace std;

struct Point {
    double x, y;
};

static inline double norm(Point p) {
    return sqrt(p.x * p.x + p.y * p.y);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    cout << fixed << setprecision(6);

    const double EPS = 1e-12;

    while (N--) {
        double X0, Y0, R, XA, YA, XB, YB;
        cin >> X0 >> Y0 >> R >> XA >> YA >> XB >> YB;

        Point A{XA - X0, YA - Y0};
        Point B{XB - X0, YB - Y0};

        Point AB{B.x - A.x, B.y - A.y};
        double lenAB = norm(AB);

        if (lenAB < EPS) {
            cout << X0 + R << ' ' << Y0 << '\n';
            continue;
        }

        Point M{(A.x + B.x) / 2.0, (A.y + B.y) / 2.0};
        double h = lenAB / 2.0;

        Point e{-AB.y / lenAB, AB.x / lenAB};

        auto f = [&](double s) {
            Point P{M.x + s * e.x, M.y + s * e.y};
            return norm(P) + sqrt(h * h + s * s);
        };

        if (fabs(f(0.0) - R) < 1e-10) {
            Point P = M;
            double d = norm(P);
            Point C{R * P.x / d, R * P.y / d};
            cout << C.x + X0 << ' ' << C.y + Y0 << '\n';
            continue;
        }

        auto find_root = [&](double sign) {
            double lo = 0.0, hi = 1.0;
            while (f(sign * hi) < R) hi *= 2.0;

            for (int i = 0; i < 100; ++i) {
                double mid = (lo + hi) / 2.0;
                if (f(sign * mid) < R)
                    lo = mid;
                else
                    hi = mid;
            }
            return sign * hi;
        };

        double s1 = find_root(1.0);
        double s2 = find_root(-1.0);

        double s = (fabs(s1) < fabs(s2)) ? s1 : s2;

        Point P{M.x + s * e.x, M.y + s * e.y};
        double d = norm(P);

        Point C{R * P.x / d, R * P.y / d};

        cout << C.x + X0 << ' ' << C.y + Y0 << '\n';
    }

    return 0;
}
