#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int to, id;
    long long w;
};

const long long INF = (1LL << 62);

vector<long long> dijkstra(int src, const vector<vector<Edge>>& g) {
    int n = (int)g.size() - 1;
    vector<long long> dist(n + 1, INF);
    priority_queue<pair<long long,int>, vector<pair<long long,int>>, greater<pair<long long,int>>> pq;

    dist[src] = 0;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, v] = pq.top();
        pq.pop();

        if (d != dist[v]) continue;

        for (const auto& e : g[v]) {
            if (dist[e.to] > d + e.w) {
                dist[e.to] = d + e.w;
                pq.push({dist[e.to], e.to});
            }
        }
    }

    return dist;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<vector<Edge>> g(n + 1);

    for (int i = 1; i <= m; i++) {
        int a, b;
        long long t;
        cin >> a >> b >> t;

        g[a].push_back({b, i, t});
        g[b].push_back({a, i, t});
    }

    int k;
    cin >> k;

    vector<int> special(k);
    for (int i = 0; i < k; i++) {
        cin >> special[i];
    }

    if (k == 1) {
        cout << 0 << '\n' << '\n';
        return 0;
    }

    vector<long long> d0 = dijkstra(special[0], g);

    int a = special[0];
    for (int v : special) {
        if (d0[v] > d0[a]) {
            a = v;
        }
    }

    vector<long long> distA = dijkstra(a, g);

    int b = a;
    for (int v : special) {
        if (distA[v] > distA[b]) {
            b = v;
        }
    }

    vector<long long> distB = dijkstra(b, g);
    long long totalDist = distA[b];

    sort(special.begin(), special.end(), [&](int x, int y) {
        return distA[x] < distA[y];
    });

    vector<int> terminalIndex(n + 1, -1);
    for (int i = 0; i < k; i++) {
        terminalIndex[special[i]] = i;
    }

    vector<char> onPath(n + 1, 0);
    vector<int> vertices;
    vertices.reserve(n);

    for (int v = 1; v <= n; v++) {
        if (distA[v] + distB[v] == totalDist) {
            onPath[v] = 1;
            vertices.push_back(v);
        }
    }

    sort(vertices.begin(), vertices.end(), [&](int x, int y) {
        return distA[x] > distA[y];
    });

    const int BIG = 1e9;
    vector<int> nearestTerminal(n + 1, BIG);

    for (int v : vertices) {
        if (terminalIndex[v] != -1) {
            nearestTerminal[v] = terminalIndex[v];
        }

        for (const auto& e : g[v]) {
            int to = e.to;
            if (!onPath[to]) continue;

            if (distA[v] + e.w == distA[to]) {
                nearestTerminal[v] = min(nearestTerminal[v], nearestTerminal[to]);
            }
        }
    }

    vector<int> answer;

    for (int i = 0; i + 1 < k; i++) {
        int cur = special[i];
        int target = special[i + 1];

        while (cur != target) {
            bool found = false;

            for (const auto& e : g[cur]) {
                int to = e.to;

                if (!onPath[to]) continue;

                if (distA[cur] + e.w == distA[to] && nearestTerminal[to] == i + 1) {
                    answer.push_back(e.id);
                    cur = to;
                    found = true;
                    break;
                }
            }

            if (!found) {
                return 0;
            }
        }
    }

    cout << answer.size() << '\n';
    for (int i = 0; i < (int)answer.size(); i++) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';

    return 0;
}
