```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    if (!(cin >> n >> k)) return 0;

    int total = 1 << n;
    int prev = 0;
    bool first = true;
    bool ok = true;

    string ans;
    ans.reserve((size_t)total * 5);

    auto add_number = [&](int x) {
        ans += to_string(x);
    };

    auto emit = [&](int from, int to) {
        int diff = from ^ to;
        int cnt = __builtin_popcount((unsigned)diff);

        if (cnt == 0) return;

        if (cnt == 1) {
            int p = __builtin_ctz((unsigned)diff) + 1;
            if (to & diff) {
                ans += '+';
                add_number(p);
            } else {
                ans += '-';
                add_number(p);
            }
        } else if (cnt == 2) {
            int leaveMask = from & diff;
            int enterMask = to & diff;

            int a = __builtin_ctz((unsigned)leaveMask) + 1;
            int b = __builtin_ctz((unsigned)enterMask) + 1;

            if (b == a + 1) {
                ans += "++";
                add_number(a);
            } else if (b == a - 1) {
                ans += "--";
                add_number(a);
            } else {
                ok = false;
            }
        } else {
            ok = false;
        }
    };

    for (int i = 0; i < total; ++i) {
        int g = i ^ (i >> 1);
        if (__builtin_popcount((unsigned)g) > k) continue;

        if (first) {
            prev = g;
            first = false;
        } else {
            emit(prev, g);
            prev = g;
        }
    }

    emit(prev, 0);

    if (!ok) {
        cout << 0 << '\n';
    } else {
        cout << ans << '\n';
    }

    return 0;
}
```
