#include <bits/stdc++.h>
using namespace std;

struct Point {
    long long x, y;
};

struct Segment {
    Point a, b;
    long long pref, len;
    bool vertical() const { return a.x == b.x; }
};

static inline bool between(long long v, long long l, long long r) {
    if (l > r) swap(l, r);
    return l <= v && v <= r;
}

long long offsetOnSegment(const Segment& s, long long x, long long y) {
    return llabs(x - s.a.x) + llabs(y - s.a.y);
}

void updPoint(const Segment& s1, const Segment& s2, long long x, long long y, long long& ans) {
    long long t1 = s1.pref + offsetOnSegment(s1, x, y);
    long long t2 = s2.pref + offsetOnSegment(s2, x, y);
    long long d = llabs(t1 - t2);
    if (d > 0) ans = min(ans, d);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<Point> p(N);
    for (int i = 0; i < N; ++i) {
        cin >> p[i].x >> p[i].y;
    }

    vector<Segment> seg;
    long long total = 0;

    for (int i = 0; i + 1 < N; ++i) {
        Segment s;
        s.a = p[i];
        s.b = p[i + 1];
        s.pref = total;
        s.len = llabs(s.a.x - s.b.x) + llabs(s.a.y - s.b.y);
        total += s.len;
        if (s.len > 0) seg.push_back(s);
    }

    int M = (int)seg.size();
    long long ans = total;

    for (int i = 0; i < M; ++i) {
        for (int j = i + 1; j < M; ++j) {
            if (j == i + 1) continue;

            const Segment& A = seg[i];
            const Segment& B = seg[j];

            if (A.vertical() && B.vertical()) {
                if (A.a.x != B.a.x) continue;

                long long l1 = min(A.a.y, A.b.y), r1 = max(A.a.y, A.b.y);
                long long l2 = min(B.a.y, B.b.y), r2 = max(B.a.y, B.b.y);
                long long l = max(l1, l2), r = min(r1, r2);

                if (l <= r) {
                    updPoint(A, B, A.a.x, l, ans);
                    updPoint(A, B, A.a.x, r, ans);
                }
            } else if (!A.vertical() && !B.vertical()) {
                if (A.a.y != B.a.y) continue;

                long long l1 = min(A.a.x, A.b.x), r1 = max(A.a.x, A.b.x);
                long long l2 = min(B.a.x, B.b.x), r2 = max(B.a.x, B.b.x);
                long long l = max(l1, l2), r = min(r1, r2);

                if (l <= r) {
                    updPoint(A, B, l, A.a.y, ans);
                    updPoint(A, B, r, A.a.y, ans);
                }
            } else {
                const Segment& V = A.vertical() ? A : B;
                const Segment& H = A.vertical() ? B : A;

                long long x = V.a.x;
                long long y = H.a.y;

                if (between(x, H.a.x, H.b.x) && between(y, V.a.y, V.b.y)) {
                    updPoint(A, B, x, y, ans);
                }
            }
        }
    }

    cout << ans << '\n';
    return 0;
}
