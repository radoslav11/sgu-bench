#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    vector<string> mask(N, string(M, '0'));

    for (int user = 0; user < M; ++user) {
        int k;
        cin >> k;
        for (int j = 0; j < k; ++j) {
            int r;
            cin >> r;
            --r;
            mask[r][user] = '1';
        }
    }

    set<string> groups;
    for (int r = 0; r < N; ++r) {
        if (mask[r].find('1') != string::npos) {
            groups.insert(mask[r]);
        }
    }

    cout << groups.size() << '\n';

    return 0;
}
