#include <bits/stdc++.h>
using namespace std;

static const long long MOD = 1000000009LL;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;

    vector<pair<long long, int>> players;
    players.reserve(2 * n);

    for (int i = 0; i < n; ++i) {
        long long x;
        cin >> x;
        players.push_back({x, 0}); // Berland player
    }

    for (int i = 0; i < n; ++i) {
        long long x;
        cin >> x;
        players.push_back({x, 1}); // Byteland player
    }

    int sum = n + k;
    if (sum % 2 != 0) {
        cout << 0 << '\n';
        return 0;
    }

    int need = sum / 2;
    if (need < 0 || need > n) {
        cout << 0 << '\n';
        return 0;
    }

    sort(players.begin(), players.end());

    int W = need + 1;
    vector<long long> dp((n + 1) * W, 0), ndp((n + 1) * W, 0);

    auto id = [&](int openA, int wins) {
        return openA * W + wins;
    };

    dp[id(0, 0)] = 1;

    int seenA = 0, seenB = 0;

    for (auto [level, team] : players) {
        fill(ndp.begin(), ndp.end(), 0);

        for (int openA = 0; openA <= n; ++openA) {
            int openB = openA - (seenA - seenB);
            if (openB < 0 || openB > n) continue;

            for (int wins = 0; wins <= need; ++wins) {
                long long cur = dp[id(openA, wins)];
                if (!cur) continue;

                if (team == 0) {
                    // Current Berland player stays open, waiting to lose to a stronger Byteland player.
                    if (openA + 1 <= n) {
                        long long &to = ndp[id(openA + 1, wins)];
                        to += cur;
                        if (to >= MOD) to -= MOD;
                    }

                    // Current Berland player closes one open weaker Byteland player: Berland wins.
                    if (openB > 0 && wins + 1 <= need) {
                        long long add = cur * openB % MOD;
                        long long &to = ndp[id(openA, wins + 1)];
                        to += add;
                        if (to >= MOD) to -= MOD;
                    }
                } else {
                    // Current Byteland player stays open, waiting to lose to a stronger Berland player.
                    {
                        long long &to = ndp[id(openA, wins)];
                        to += cur;
                        if (to >= MOD) to -= MOD;
                    }

                    // Current Byteland player closes one open weaker Berland player: Byteland wins.
                    if (openA > 0) {
                        long long add = cur * openA % MOD;
                        long long &to = ndp[id(openA - 1, wins)];
                        to += add;
                        if (to >= MOD) to -= MOD;
                    }
                }
            }
        }

        dp.swap(ndp);

        if (team == 0) ++seenA;
        else ++seenB;
    }

    cout << dp[id(0, need)] % MOD << '\n';
    return 0;
}
