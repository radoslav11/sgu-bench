#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<string> s(N);
    for (int i = 0; i < N; ++i) cin >> s[i];

    sort(s.begin(), s.end(), [](const string& a, const string& b) {
        return a + b < b + a;
    });

    for (const string& str : s) cout << str;
    cout << '\n';

    return 0;
}
