```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    vector<int> a;
    int blankRow = -1;

    for (int i = 0; i < 4; ++i) {
        for (int j = 0; j < 4; ++j) {
            int x;
            cin >> x;
            if (x == 0) blankRow = i;
            a.push_back(x);
        }
    }

    int inversions = 0;
    for (int i = 0; i < 16; ++i) {
        if (a[i] == 0) continue;
        for (int j = i + 1; j < 16; ++j) {
            if (a[j] != 0 && a[i] > a[j]) {
                ++inversions;
            }
        }
    }

    int rowFromBottom = 4 - blankRow;

    if ((inversions + rowFromBottom) % 2 == 1)
        cout << "YES\n";
    else
        cout << "NO\n";

    return 0;
}
```
