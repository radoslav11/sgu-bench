#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, N1, N2;
    cin >> N >> N1 >> N2;

    double x1, y1, x2, y2;
    cin >> x1 >> y1 >> x2 >> y2;

    int k = (N2 - N1 + N) % N;

    const double PI = acos(-1.0);
    double phi = -2.0 * PI * k / N;

    double c = cos(phi);
    double s = sin(phi);

    double rx = c * x1 - s * y1;
    double ry = s * x1 + c * y1;

    double dx = x2 - rx;
    double dy = y2 - ry;

    double A = 1.0 - c;
    double B = s;
    double det = A * A + B * B;

    double cx = (A * dx - B * dy) / det;
    double cy = (B * dx + A * dy) / det;

    cout << fixed << setprecision(6);

    for (int i = 1; i <= N; i++) {
        int step = (i - N1 + N) % N;
        double ang = -2.0 * PI * step / N;

        double cc = cos(ang);
        double ss = sin(ang);

        double vx = x1 - cx;
        double vy = y1 - cy;

        double x = cx + cc * vx - ss * vy;
        double y = cy + ss * vx + cc * vy;

        cout << x << ' ' << y << '\n';
    }

    return 0;
}
