#include <bits/stdc++.h>
using namespace std;

struct Operation {
    int p, a1, b1, a2, b2;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<vector<char>> cur(N + 1, vector<char>(N + 1, 0));
    vector<vector<char>> target(N + 1, vector<char>(N + 1, 0));
    vector<set<int>> adj(N + 1);

    for (int i = 0; i < N - 1; ++i) {
        int a, b;
        cin >> a >> b;
        cur[a][b] = cur[b][a] = 1;
        adj[a].insert(b);
        adj[b].insert(a);
    }

    vector<pair<int, int>> targetEdges;
    for (int i = 0; i < N - 1; ++i) {
        int a, b;
        cin >> a >> b;
        target[a][b] = target[b][a] = 1;
        targetEdges.push_back({a, b});
    }

    vector<Operation> ans;

    for (auto [u, v] : targetEdges) {
        if (cur[u][v]) continue;

        vector<int> parent(N + 1, -1);
        queue<int> q;
        q.push(u);
        parent[u] = 0;

        while (!q.empty()) {
            int x = q.front();
            q.pop();

            if (x == v) break;

            for (int to : adj[x]) {
                if (parent[to] == -1) {
                    parent[to] = x;
                    q.push(to);
                }
            }
        }

        int remA = -1, remB = -1;
        int x = v;
        while (x != u) {
            int y = parent[x];
            if (!target[x][y]) {
                remA = x;
                remB = y;
                break;
            }
            x = y;
        }

        ans.push_back({1, u, v, remA, remB});

        cur[u][v] = cur[v][u] = 1;
        adj[u].insert(v);
        adj[v].insert(u);

        cur[remA][remB] = cur[remB][remA] = 0;
        adj[remA].erase(remB);
        adj[remB].erase(remA);
    }

    cout << ans.size() << '\n';
    for (const auto &op : ans) {
        cout << op.p << ' ' << op.a1 << ' ' << op.b1 << ' '
             << op.a2 << ' ' << op.b2 << '\n';
    }

    return 0;
}
