```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    const int MAXN = 10000;

    vector<bool> isPrime(MAXN + 1, true);
    isPrime[0] = isPrime[1] = false;

    for (int i = 2; i * i <= MAXN; ++i) {
        if (isPrime[i]) {
            for (int j = i * i; j <= MAXN; j += i) {
                isPrime[j] = false;
            }
        }
    }

    vector<int> primes;
    for (int i = 2; i <= MAXN; ++i) {
        if (isPrime[i]) primes.push_back(i);
    }

    vector<int> superPrimes;
    for (int i = 0; i < (int)primes.size(); ++i) {
        int index = i + 1;
        if (isPrime[index]) {
            superPrimes.push_back(primes[i]);
        }
    }

    const int INF = 1e9;
    vector<int> dp(n + 1, INF);
    vector<int> prevCoin(n + 1, -1);

    dp[0] = 0;

    for (int sum = 1; sum <= n; ++sum) {
        for (int coin : superPrimes) {
            if (coin > sum) break;
            if (dp[sum - coin] + 1 < dp[sum]) {
                dp[sum] = dp[sum - coin] + 1;
                prevCoin[sum] = coin;
            }
        }
    }

    if (dp[n] == INF) {
        cout << 0 << '\n';
        return 0;
    }

    vector<int> answer;
    int cur = n;
    while (cur > 0) {
        int coin = prevCoin[cur];
        answer.push_back(coin);
        cur -= coin;
    }

    sort(answer.rbegin(), answer.rend());

    cout << answer.size() << '\n';
    for (int x : answer) {
        cout << x << ' ';
    }
    cout << '\n';

    return 0;
}
```
