#include <bits/stdc++.h>
using namespace std;

const double EPS = 1e-10;

struct Point {
    double x, y;
    Point() {}
    Point(double x_, double y_) : x(x_), y(y_) {}

    Point operator + (const Point& other) const {
        return Point(x + other.x, y + other.y);
    }
    Point operator - (const Point& other) const {
        return Point(x - other.x, y - other.y);
    }
    Point operator * (double k) const {
        return Point(x * k, y * k);
    }
};

double cross(Point a, Point b) {
    return a.x * b.y - a.y * b.x;
}

double norm(Point a) {
    return sqrt(a.x * a.x + a.y * a.y);
}

struct Line {
    Point p, d;

    Line() {}
    Line(Point p_, Point d_) : p(p_), d(d_) {}
};

bool inside(const Line& l, const Point& q) {
    return cross(l.d, q - l.p) >= -EPS;
}

Point intersection(const Line& a, const Line& b) {
    Point u = b.p - a.p;
    double t = cross(u, b.d) / cross(a.d, b.d);
    return a.p + a.d * t;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<Point> p(N);
    for (int i = 0; i < N; i++) {
        cin >> p[i].x >> p[i].y;
    }

    vector<double> ang(N);
    for (int i = 0; i < N; i++) {
        Point d = p[(i + 1) % N] - p[i];
        ang[i] = atan2(d.y, d.x);
    }

    int start = 0;
    for (int i = 1; i < N; i++) {
        if (ang[i] < ang[start]) start = i;
    }

    auto feasible = [&](double r) -> bool {
        deque<Line> dq;

        for (int k = 0; k < N; k++) {
            int i = (start + k) % N;
            Point a = p[i];
            Point b = p[(i + 1) % N];
            Point d = b - a;

            double len = norm(d);
            Point inward(-d.y / len, d.x / len);

            Line cur(a + inward * r, d);

            while (dq.size() >= 2) {
                Point q = intersection(dq[dq.size() - 2], dq[dq.size() - 1]);
                if (inside(cur, q)) break;
                dq.pop_back();
            }

            while (dq.size() >= 2) {
                Point q = intersection(dq[0], dq[1]);
                if (inside(cur, q)) break;
                dq.pop_front();
            }

            dq.push_back(cur);
        }

        while (dq.size() >= 3) {
            Point q = intersection(dq[dq.size() - 2], dq[dq.size() - 1]);
            if (inside(dq[0], q)) break;
            dq.pop_back();
        }

        while (dq.size() >= 3) {
            Point q = intersection(dq[0], dq[1]);
            if (inside(dq.back(), q)) break;
            dq.pop_front();
        }

        return dq.size() >= 3;
    };

    double lo = 0.0, hi = 1e9;

    for (int it = 0; it < 70; it++) {
        double mid = (lo + hi) / 2.0;
        if (feasible(mid)) lo = mid;
        else hi = mid;
    }

    cout << fixed << setprecision(10) << lo << '\n';

    return 0;
}
