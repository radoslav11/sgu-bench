#include <bits/stdc++.h>
using namespace std;

struct Group {
    long long t, w, e;
    int p;
};

struct Departure {
    long long time;
    int counter, l, r;
    bool operator>(const Departure& other) const {
        return time > other.time;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    long long Tclose;
    cin >> N >> M >> Tclose;

    vector<int> C(N);
    for (int i = 0; i < N; i++) cin >> C[i];

    vector<Group> groups(M);
    long long totalCustomers = 0;
    for (int i = 0; i < M; i++) {
        cin >> groups[i].t >> groups[i].p >> groups[i].w >> groups[i].e;
        totalCustomers += groups[i].p;
    }

    vector<vector<int>> occupied(N);
    for (int i = 0; i < N; i++) occupied[i].assign(C[i], 0);

    priority_queue<Departure, vector<Departure>, greater<Departure>> deps;
    deque<int> waitq;

    long double satisfactionSum = 0.0L;

    auto findSeat = [&](int p) {
        const int INF = 1e9;

        bool found = false;
        int bestMin = -1, bestMax = -1;
        int bestCounter = -1, bestL = -1, bestR = -1;

        for (int c = 0; c < N; c++) {
            int i = 0;
            while (i < C[c]) {
                while (i < C[c] && occupied[c][i]) i++;
                if (i == C[c]) break;

                int a = i;
                while (i < C[c] && !occupied[c][i]) i++;
                int b = i - 1;

                if (b - a + 1 >= p) {
                    for (int s = a; s + p - 1 <= b; s++) {
                        int l = s;
                        int r = s + p - 1;

                        int SL = (a == 0 ? INF : l - a);
                        int SR = (b == C[c] - 1 ? INF : b - r);

                        int mn = min(SL, SR);
                        int mx = max(SL, SR);

                        bool better = false;
                        if (!found) better = true;
                        else if (mn != bestMin) better = mn > bestMin;
                        else if (mx != bestMax) better = mx > bestMax;
                        else if (c != bestCounter) better = c < bestCounter;
                        else better = l < bestL;

                        if (better) {
                            found = true;
                            bestMin = mn;
                            bestMax = mx;
                            bestCounter = c;
                            bestL = l;
                            bestR = r;
                        }
                    }
                }
            }
        }

        return tuple<bool, int, int, int>(found, bestCounter, bestL, bestR);
    };

    int nextArrival = 0;

    while (nextArrival < M || !waitq.empty()) {
        long long cur = LLONG_MAX;

        if (nextArrival < M) cur = min(cur, groups[nextArrival].t);
        if (!deps.empty()) cur = min(cur, deps.top().time);
        if (!waitq.empty()) {
            int g = waitq.front();
            cur = min(cur, groups[g].t + groups[g].w);
            cur = min(cur, Tclose);
        }

        while (!deps.empty() && deps.top().time == cur) {
            auto d = deps.top();
            deps.pop();
            for (int i = d.l; i <= d.r; i++) {
                occupied[d.counter][i] = 0;
            }
        }

        while (nextArrival < M && groups[nextArrival].t == cur) {
            waitq.push_back(nextArrival);
            nextArrival++;
        }

        if (cur >= Tclose) {
            while (!waitq.empty()) {
                int g = waitq.front();
                waitq.pop_front();
                satisfactionSum -= groups[g].p;
            }
            continue;
        }

        while (!waitq.empty()) {
            int g = waitq.front();
            Group &gr = groups[g];
            long long deadline = gr.t + gr.w;

            auto [ok, counter, l, r] = findSeat(gr.p);

            if (ok && cur <= deadline && cur < Tclose) {
                for (int i = l; i <= r; i++) occupied[counter][i] = 1;

                long long waited = cur - gr.t;
                long double sat = (long double)(gr.w - waited) / (long double)gr.w;
                satisfactionSum += sat * gr.p;

                deps.push({cur + gr.e, counter, l, r});
                waitq.pop_front();
            } else {
                if (cur >= deadline) {
                    satisfactionSum -= gr.p;
                    waitq.pop_front();
                } else {
                    break;
                }
            }
        }
    }

    long double ans = satisfactionSum / (long double)totalCustomers;
    cout << fixed << setprecision(16) << (double)ans << '\n';

    return 0;
}
