#include <bits/stdc++.h>
using namespace std;

using ll = long long;

struct FastScanner {
    static const int BUFSIZE = 1 << 20;
    int idx = 0, size = 0;
    char buf[BUFSIZE];

    inline char getChar() {
        if (idx >= size) {
            size = (int)fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T &out) {
        char c;
        T sign = 1;
        T num = 0;

        c = getChar();
        if (!c) return false;

        while (c != '-' && (c < '0' || c > '9')) {
            c = getChar();
            if (!c) return false;
        }

        if (c == '-') {
            sign = -1;
            c = getChar();
        }

        while (c >= '0' && c <= '9') {
            num = num * 10 + (c - '0');
            c = getChar();
        }

        out = num * sign;
        return true;
    }
};

int main() {
    FastScanner fs;

    int N, M, L;
    fs.readInt(N);
    fs.readInt(M);
    fs.readInt(L);

    vector<int> level(N + 1);
    int source = -1, target = -1;

    for (int i = 1; i <= N; ++i) {
        fs.readInt(level[i]);
        if (level[i] == 1) source = i;
        if (level[i] == L) target = i;
    }

    vector<vector<int>> adj(N + 1);
    vector<int> to(M);
    vector<ll> cap(M), flow(M, 0);

    for (int i = 0; i < M; ++i) {
        int a, b;
        ll c;
        fs.readInt(a);
        fs.readInt(b);
        fs.readInt(c);

        to[i] = b;
        cap[i] = c;
        adj[a].push_back(i);
    }

    vector<int> ptr(N + 1, 0);

    function<ll(int, ll)> dfs = [&](int v, ll pushed) -> ll {
        if (v == target) return pushed;

        for (int &i = ptr[v]; i < (int)adj[v].size(); ++i) {
            int id = adj[v][i];
            if (flow[id] == cap[id]) continue;

            ll canPush = min(pushed, cap[id] - flow[id]);
            ll pushedFurther = dfs(to[id], canPush);

            if (pushedFurther > 0) {
                flow[id] += pushedFurther;
                return pushedFurther;
            }
        }

        return 0;
    };

    const ll INF = (1LL << 62);

    while (dfs(source, INF) > 0) {
    }

    string output;
    output.reserve((size_t)M * 12);

    for (int i = 0; i < M; ++i) {
        output += to_string(flow[i]);
        output += '\n';
    }

    cout << output;

    return 0;
}
