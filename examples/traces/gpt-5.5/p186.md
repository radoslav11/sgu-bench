```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<int> L(N);
    for (int i = 0; i < N; ++i) cin >> L[i];

    sort(L.begin(), L.end());

    int ans = N - 1;
    int sum = 0;

    for (int k = 0; k <= N - 1; ++k) {
        if (k > 0) sum += L[k - 1];

        int need = N - k - 1;
        ans = min(ans, max(sum, need));
    }

    cout << ans << '\n';
    return 0;
}
```
