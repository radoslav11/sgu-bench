```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    vector<deque<int>> q(M);

    for (int i = 0; i < N; ++i) {
        int k, t;
        cin >> k >> t;

        for (int j = 0; j < M; ++j) {
            while (!q[j].empty() && q[j].front() <= k) {
                q[j].pop_front();
            }
        }

        int chosen = 0;
        for (int j = 1; j < M; ++j) {
            if (q[j].size() < q[chosen].size()) {
                chosen = j;
            }
        }

        int startTime = k;
        if (!q[chosen].empty()) {
            startTime = q[chosen].back();
        }

        int leaveTime = startTime + t;
        q[chosen].push_back(leaveTime);

        cout << chosen + 1 << ' ' << leaveTime << '\n';
    }

    return 0;
}
```
