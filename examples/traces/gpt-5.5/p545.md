#include <bits/stdc++.h>
using namespace std;

const double EPS = 1e-10;
const double INF = 1e100;

struct Rope {
    double x, l;
};

struct Piece {
    double a, b;
    int id;
};

struct Envelope {
    vector<Piece> pieces;
    vector<double> bp;
    double L, R;
};

struct Segment {
    // type 0 = vertical, type 1 = circular arc
    int type;
    double t0, len;
    double x1, y1, x2, y2;
    int rope;
};

int n, m;
vector<Rope> ropes;
vector<Segment> segs;
double finalX, finalY;

double fval(int i, double x) {
    double dx = x - ropes[i].x;
    double v = ropes[i].l * ropes[i].l - dx * dx;
    if (v < 0 && v > -1e-8) v = 0;
    return -sqrt(max(0.0, v));
}

double hval(const vector<int>& alive, double x) {
    double res = -INF;
    for (int id : alive) res = max(res, fval(id, x));
    return res;
}

Envelope buildEnvelope(const vector<int>& alive) {
    Envelope e;
    e.L = -INF;
    e.R = INF;

    for (int id : alive) {
        e.L = max(e.L, ropes[id].x - ropes[id].l);
        e.R = min(e.R, ropes[id].x + ropes[id].l);
    }

    vector<double> pts;
    pts.push_back(e.L);
    pts.push_back(e.R);

    for (int a = 0; a < (int)alive.size(); a++) {
        for (int b = a + 1; b < (int)alive.size(); b++) {
            int i = alive[a], j = alive[b];
            double xi = ropes[i].x, xj = ropes[j].x;
            double li = ropes[i].l, lj = ropes[j].l;

            if (fabs(xi - xj) < EPS) continue;

            double x = (xj * xj - xi * xi + li * li - lj * lj) / (2.0 * (xj - xi));
            if (x > e.L + EPS && x < e.R - EPS) pts.push_back(x);
        }
    }

    sort(pts.begin(), pts.end());
    vector<double> u;
    for (double x : pts) {
        if (u.empty() || fabs(x - u.back()) > 1e-8) u.push_back(x);
    }
    e.bp = u;

    for (int k = 0; k + 1 < (int)u.size(); k++) {
        double a = u[k], b = u[k + 1];
        if (b - a < EPS) continue;
        double mid = (a + b) / 2.0;
        int best = alive[0];
        double by = fval(best, mid);
        for (int id : alive) {
            double y = fval(id, mid);
            if (y > by) {
                by = y;
                best = id;
            }
        }
        e.pieces.push_back({a, b, best});
    }

    return e;
}

pair<double, double> envelopeMinimum(const vector<int>& alive, const Envelope& e) {
    double bx = e.L, by = hval(alive, bx);

    auto relax = [&](double x) {
        if (x < e.L - EPS || x > e.R + EPS) return;
        x = max(e.L, min(e.R, x));
        double y = hval(alive, x);
        if (y < by) {
            by = y;
            bx = x;
        }
    };

    for (double x : e.bp) relax(x);

    for (auto p : e.pieces) {
        double cx = ropes[p.id].x;
        if (cx >= p.a - EPS && cx <= p.b + EPS) relax(cx);
    }

    return {bx, by};
}

int activePiece(const Envelope& e, double a, double b) {
    double mid = (a + b) / 2.0;
    for (auto p : e.pieces) {
        if (mid >= p.a - 1e-8 && mid <= p.b + 1e-8) return p.id;
    }
    return e.pieces.empty() ? -1 : e.pieces[0].id;
}

void addVertical(double& curTime, double x, double y1, double y2) {
    double len = fabs(y2 - y1);
    if (len < 1e-9) return;
    segs.push_back({0, curTime, len, x, y1, x, y2, -1});
    curTime += len;
}

void addArc(double& curTime, int id, double x1, double x2) {
    if (fabs(x1 - x2) < 1e-9) return;

    double y1 = fval(id, x1);
    double y2 = fval(id, x2);

    double th1 = atan2(y1, x1 - ropes[id].x);
    double th2 = atan2(y2, x2 - ropes[id].x);

    double dth = fabs(th2 - th1);
    double len = ropes[id].l * dth;

    if (len < 1e-9) return;

    segs.push_back({1, curTime, len, x1, y1, x2, y2, id});
    curTime += len;
}

void addPathAlongEnvelope(double& curTime, const vector<int>& alive, const Envelope& e,
                          double xStart, double xEnd) {
    if (fabs(xStart - xEnd) < 1e-9) return;

    vector<double> pts;
    pts.push_back(xStart);
    pts.push_back(xEnd);

    double lo = min(xStart, xEnd), hi = max(xStart, xEnd);
    for (double x : e.bp) {
        if (x > lo + 1e-8 && x < hi - 1e-8) pts.push_back(x);
    }

    sort(pts.begin(), pts.end());
    if (xStart > xEnd) reverse(pts.begin(), pts.end());

    for (int i = 0; i + 1 < (int)pts.size(); i++) {
        double a = pts[i], b = pts[i + 1];
        int id = activePiece(e, a, b);
        addArc(curTime, id, a, b);
    }
}

pair<double, double> positionAt(const Segment& s, double t) {
    double d = t - s.t0;
    d = max(0.0, min(s.len, d));

    if (s.type == 0) {
        double k = d / s.len;
        return {s.x1, s.y1 + (s.y2 - s.y1) * k};
    } else {
        int id = s.rope;
        double r = ropes[id].l;
        double th1 = atan2(s.y1, s.x1 - ropes[id].x);
        double th2 = atan2(s.y2, s.x2 - ropes[id].x);
        double dir = (th2 > th1 ? 1.0 : -1.0);
        double th = th1 + dir * d / r;
        return {ropes[id].x + r * cos(th), r * sin(th)};
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    ropes.resize(n);

    for (int i = 0; i < n; i++) cin >> ropes[i].x;
    for (int i = 0; i < n; i++) cin >> ropes[i].l;

    vector<int> cuts(n - 1);
    for (int i = 0; i < n - 1; i++) {
        cin >> cuts[i];
        cuts[i]--;
    }

    vector<double> queries(m);
    for (int i = 0; i < m; i++) cin >> queries[i];

    vector<int> alive;
    for (int i = 0; i < n; i++) alive.push_back(i);

    Envelope e0 = buildEnvelope(alive);
    auto start = envelopeMinimum(alive, e0);

    double curX = start.first;
    double curY = start.second;
    double curTime = 0.0;

    for (int cut : cuts) {
        alive.erase(find(alive.begin(), alive.end(), cut));

        Envelope e = buildEnvelope(alive);
        double newH = hval(alive, curX);

        if (curY > newH + 1e-9) {
            addVertical(curTime, curX, curY, newH);
            curY = newH;
        }

        auto mn = envelopeMinimum(alive, e);
        addPathAlongEnvelope(curTime, alive, e, curX, mn.first);

        curX = mn.first;
        curY = mn.second;
    }

    finalX = curX;
    finalY = curY;

    cout.setf(ios::fixed);
    cout << setprecision(10);

    int ptr = 0;
    for (double t : queries) {
        while (ptr < (int)segs.size() && t > segs[ptr].t0 + segs[ptr].len + 1e-9) ptr++;

        pair<double, double> ans;
        if (ptr < (int)segs.size() && t >= segs[ptr].t0 - 1e-9) {
            ans = positionAt(segs[ptr], t);
        } else {
            ans = {finalX, finalY};
            if (ptr < (int)segs.size() && t < segs[ptr].t0 - 1e-9) {
                if (ptr == 0) ans = {start.first, start.second};
                else ans = {segs[ptr - 1].x2, segs[ptr - 1].y2};
            }
        }

        cout << ans.first << ' ' << ans.second << '\n';
    }

    return 0;
}
