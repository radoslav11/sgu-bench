```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string secret, guess;
    cin >> secret >> guess;

    int bulls = 0;
    vector<int> cntSecret(10, 0), cntGuess(10, 0);

    for (int i = 0; i < 4; ++i) {
        if (secret[i] == guess[i]) bulls++;
        cntSecret[secret[i] - '0']++;
        cntGuess[guess[i] - '0']++;
    }

    int common = 0;
    for (int d = 0; d < 10; ++d) {
        common += min(cntSecret[d], cntGuess[d]);
    }

    int cows = common - bulls;

    cout << bulls << ' ' << cows << '\n';

    return 0;
}
```
