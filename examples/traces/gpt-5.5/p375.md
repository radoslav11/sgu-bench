#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    cin >> N;

    if (N % 2 == 0) {
        cout << "No solution\n";
        return 0;
    }

    long long m = (N - 1) / 2;

    if (m == 0) {
        cout << 0 << "\n\n";
        return 0;
    }

    string bits;
    while (m > 0) {
        bits.push_back(char('0' + (m & 1)));
        m >>= 1;
    }
    reverse(bits.begin(), bits.end());

    cout << bits.size() << '\n';
    for (size_t i = 0; i < bits.size(); ++i) {
        if (i) cout << ' ';
        cout << (bits[i] == '0' ? 1 : 2);
    }
    cout << '\n';

    return 0;
}
