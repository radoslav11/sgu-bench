#include <bits/stdc++.h>
using namespace std;

const int INF = 1e9;

struct Atom {
    // type: '*' for star, '?' for any single char, 'a'..'z' for fixed char
    char type;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string s;
    if (!(cin >> s)) return 0;

    vector<Atom> a;
    for (char c : s) {
        if (c == '!') {
            a.push_back({'?'});
            a.push_back({'?'});
            a.push_back({'?'});
        } else {
            a.push_back({c});
        }
    }

    int n = (int)a.size();

    auto isStar = [&](int i) {
        return a[i].type == '*';
    };

    auto isSingle = [&](int i) {
        return a[i].type != '*';
    };

    auto compatible = [&](int i, int j, char &c) {
        char x = a[i].type;
        char y = a[j].type;

        if (x == '?' && y == '?') {
            c = 'a';
            return true;
        }
        if (x == '?') {
            c = y;
            return true;
        }
        if (y == '?') {
            c = x;
            return true;
        }
        if (x == y) {
            c = x;
            return true;
        }
        return false;
    };

    auto singleBest = [&](int i) {
        if (a[i].type == '?') return 'a';
        return a[i].type;
    };

    vector<vector<int>> dp(n + 1, vector<int>(n + 1, INF));

    auto get = [&](int l, int r) -> int {
        if (l > r) return 0;
        return dp[l][r];
    };

    for (int len = 1; len <= n; len++) {
        for (int l = 0; l + len - 1 < n; l++) {
            int r = l + len - 1;
            int best = INF;

            if (l == r && isSingle(l)) {
                best = min(best, 1);
            }

            if (isStar(l)) {
                best = min(best, get(l + 1, r));
            }

            if (isStar(r)) {
                best = min(best, get(l, r - 1));
            }

            if (l < r && isSingle(l) && isSingle(r)) {
                char c;
                if (compatible(l, r, c)) {
                    int inner = get(l + 1, r - 1);
                    if (inner < INF) best = min(best, inner + 2);
                }
            }

            if (isStar(l) && isSingle(r)) {
                int inner = get(l, r - 1);
                if (inner < INF) best = min(best, inner + 2);
            }

            if (isSingle(l) && isStar(r)) {
                int inner = get(l + 1, r);
                if (inner < INF) best = min(best, inner + 2);
            }

            dp[l][r] = best;
        }
    }

    if (n == 0 || get(0, n - 1) >= INF) {
        cout << "NO\n";
        return 0;
    }

    unordered_map<long long, string> memo;

    function<string(int, int)> solve = [&](int l, int r) -> string {
        if (l > r) return "";

        long long key = 1LL * l * (n + 1) + r;
        if (memo.count(key)) return memo[key];

        int need = get(l, r);
        vector<string> cand;

        if (l == r && isSingle(l) && need == 1) {
            cand.push_back(string(1, singleBest(l)));
        }

        if (isStar(l) && get(l + 1, r) == need) {
            cand.push_back(solve(l + 1, r));
        }

        if (isStar(r) && get(l, r - 1) == need) {
            cand.push_back(solve(l, r - 1));
        }

        if (l < r && isSingle(l) && isSingle(r)) {
            char c;
            if (compatible(l, r, c)) {
                int inner = get(l + 1, r - 1);
                if (inner < INF && inner + 2 == need) {
                    string mid = solve(l + 1, r - 1);
                    cand.push_back(string(1, c) + mid + string(1, c));
                }
            }
        }

        if (isStar(l) && isSingle(r)) {
            int inner = get(l, r - 1);
            if (inner < INF && inner + 2 == need) {
                char c = singleBest(r);
                string mid = solve(l, r - 1);
                cand.push_back(string(1, c) + mid + string(1, c));
            }
        }

        if (isSingle(l) && isStar(r)) {
            int inner = get(l + 1, r);
            if (inner < INF && inner + 2 == need) {
                char c = singleBest(l);
                string mid = solve(l + 1, r);
                cand.push_back(string(1, c) + mid + string(1, c));
            }
        }

        string best = cand[0];
        for (string &x : cand) {
            if (x < best) best = x;
        }

        memo[key] = best;
        return best;
    };

    cout << "YES\n";
    cout << solve(0, n - 1) << '\n';

    return 0;
}
