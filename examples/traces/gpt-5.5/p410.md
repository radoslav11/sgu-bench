#include <bits/stdc++.h>
using namespace std;

using int64 = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<int64> a(N + 1);
    int64 M = 0;

    for (int i = 1; i <= N; ++i) {
        cin >> a[i];
        M = max(M, a[i]);
    }

    int64 answer = M;

    for (int i = 1; i <= N; ++i) {
        int64 cur = a[i];
        while (cur < M) {
            cur *= 2;
            ++answer;
        }
    }

    cout << answer << '\n';

    if (answer > 1000) {
        return 0;
    }

    vector<vector<int>> science((int)M + 1);

    function<void(int64, int64, int)> build = [&](int64 start, int64 target, int planet) {
        if (start == target) return;

        int64 prev = (target + 1) / 2;
        if (prev < start) prev = start;

        build(start, prev, planet);

        int64 flights_before_science = 2 * prev - target;
        science[(int)flights_before_science].push_back(planet);
    };

    for (int i = 1; i <= N; ++i) {
        build(a[i], M, i);
    }

    for (int t = 0; t < M; ++t) {
        for (int planet : science[t]) {
            cout << "science mission to the planet " << planet << '\n';
        }
        cout << "flying mission\n";
    }

    return 0;
}
