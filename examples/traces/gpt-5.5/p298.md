#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int to, w;
};

const int K = 10000;
const int LOW = -10000;
const int HIGH = 10000;

int N, M;
vector<vector<Edge>> g;

bool feasible(int t, vector<int>* answer = nullptr) {
    vector<int> dist(N + 1, LOW);
    vector<int> cnt(N + 1, 0);
    vector<char> inq(N + 1, true);
    queue<int> q;

    for (int i = 1; i <= N; ++i) q.push(i);

    auto relax_edge = [&](int u, int v, int w) -> bool {
        if (dist[v] < dist[u] + w) {
            dist[v] = dist[u] + w;
            if (dist[v] > HIGH) return false;

            if (!inq[v]) {
                q.push(v);
                inq[v] = true;
            }

            cnt[v]++;
            if (cnt[v] > N) return false;
        }
        return true;
    };

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        inq[u] = false;

        for (const Edge& e : g[u]) {
            if (!relax_edge(u, e.to, e.w)) return false;
        }

        if (u == N) {
            if (!relax_edge(N, 1, t)) return false;
        }
    }

    if (answer) *answer = dist;
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> N >> M;
    g.assign(N + 1, {});

    for (int i = 0; i < M; ++i) {
        int x, y, c;
        cin >> x >> y >> c;
        g[y].push_back({x, c});
    }

    if (!feasible(-20000)) {
        cout << -1 << '\n';
        return 0;
    }

    int lo = -20000, hi = 20000;

    while (lo < hi) {
        int mid = (lo + hi + 1) / 2;
        if (feasible(mid)) lo = mid;
        else hi = mid - 1;
    }

    vector<int> ans;
    feasible(lo, &ans);

    for (int i = 1; i <= N; ++i) {
        if (i > 1) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';

    return 0;
}
