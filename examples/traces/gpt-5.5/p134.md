#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<vector<int>> adj(N + 1);
    for (int i = 0; i < N - 1; ++i) {
        int a, b;
        cin >> a >> b;
        adj[a].push_back(b);
        adj[b].push_back(a);
    }

    vector<int> parent(N + 1, 0), order;
    order.reserve(N);

    stack<int> st;
    st.push(1);
    parent[1] = -1;

    while (!st.empty()) {
        int v = st.top();
        st.pop();
        order.push_back(v);

        for (int to : adj[v]) {
            if (to != parent[v]) {
                parent[to] = v;
                st.push(to);
            }
        }
    }

    vector<int> subtree(N + 1, 1);

    for (int i = N - 1; i >= 0; --i) {
        int v = order[i];
        for (int to : adj[v]) {
            if (parent[to] == v) {
                subtree[v] += subtree[to];
            }
        }
    }

    int best = N;
    vector<int> centroids;

    for (int v = 1; v <= N; ++v) {
        int value = N - subtree[v];

        for (int to : adj[v]) {
            if (parent[to] == v) {
                value = max(value, subtree[to]);
            }
        }

        if (value < best) {
            best = value;
            centroids.clear();
            centroids.push_back(v);
        } else if (value == best) {
            centroids.push_back(v);
        }
    }

    sort(centroids.begin(), centroids.end());

    cout << best << ' ' << centroids.size() << '\n';
    for (size_t i = 0; i < centroids.size(); ++i) {
        if (i) cout << ' ';
        cout << centroids[i];
    }
    cout << '\n';

    return 0;
}
