#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int K;
    string s;

    if (!(cin >> K)) return 0;
    cin >> s;

    int n = (int)s.size();
    string d = s + s;

    int bestM = 0;
    vector<int> bestStarts;

    int maxM = n / 2;

    for (int m = 1; m <= maxM; ++m) {
        int mismatches = 0;

        for (int j = 0; j < m; ++j) {
            if (d[j] != d[j + m]) ++mismatches;
        }

        vector<int> starts;

        for (int i = 0; i < n; ++i) {
            if (mismatches <= K) {
                starts.push_back(i);
            }

            if (i + 1 < n) {
                if (d[i] != d[i + m]) --mismatches;
                if (d[i + m] != d[i + 2 * m]) ++mismatches;
            }
        }

        if (!starts.empty()) {
            bestM = m;
            bestStarts = starts;
        }
    }

    if (bestM == 0) {
        cout << '\n';
        return 0;
    }

    int len = 2 * bestM;
    int bestStart = bestStarts[0];

    for (int st : bestStarts) {
        bool smaller = false;
        for (int i = 0; i < len; ++i) {
            if (d[st + i] < d[bestStart + i]) {
                smaller = true;
                break;
            } else if (d[st + i] > d[bestStart + i]) {
                break;
            }
        }
        if (smaller) bestStart = st;
    }

    cout << d.substr(bestStart, len) << '\n';

    return 0;
}
