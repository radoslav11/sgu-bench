#include <bits/stdc++.h>
using namespace std;

long long egcd(long long a, long long b, long long &x, long long &y) {
    if (b == 0) {
        x = 1; y = 0;
        return a;
    }
    long long x1, y1;
    long long g = egcd(b, a % b, x1, y1);
    x = y1;
    y = x1 - (a / b) * y1;
    return g;
}

long long modInv(long long a, long long mod) {
    long long x, y;
    egcd(a, mod, x, y);
    x %= mod;
    if (x < 0) x += mod;
    return x;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    const long long A = 987654321LL;
    const long long MOD2 = 512;
    const long long MOD5 = 1953125;
    const long long MOD = 1000000000LL;

    vector<long long> roots2;
    for (long long x = 0; x < MOD2; x++) {
        if ((x * x - A) % MOD2 == 0) {
            roots2.push_back(x);
        }
    }

    vector<long long> roots5;
    for (long long r0 : {1LL, 4LL}) {
        long long r = r0;
        long long mod = 5;
        while (mod < MOD5) {
            long long nextMod = mod * 5;
            for (long long t = 0; t < 5; t++) {
                long long cand = r + t * mod;
                if (((__int128)cand * cand - A) % nextMod == 0) {
                    r = cand;
                    break;
                }
            }
            mod = nextMod;
        }
        roots5.push_back(r);
    }

    long long inv5mod2 = modInv(MOD5 % MOD2, MOD2);

    vector<long long> roots;
    for (long long r2 : roots2) {
        for (long long r5 : roots5) {
            long long t = ((r2 - r5) % MOD2 + MOD2) % MOD2;
            t = (t * inv5mod2) % MOD2;
            long long x = r5 + MOD5 * t;
            x %= MOD;
            roots.push_back(x);
        }
    }

    sort(roots.begin(), roots.end());
    roots.erase(unique(roots.begin(), roots.end()), roots.end());

    if (N >= 10) {
        cout << "72";
        for (int i = 0; i < N - 10; i++) cout << '0';
        cout << '\n';
        return 0;
    }

    long long pow10[10];
    pow10[0] = 1;
    for (int i = 1; i <= 9; i++) pow10[i] = pow10[i - 1] * 10;

    long long L = pow10[N - 1];
    long long R = pow10[N] - 1;

    long long ans = 0;
    for (long long x : roots) {
        if (L <= x && x <= R) ans++;
    }

    cout << ans << '\n';
    return 0;
}
