#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int to, rev;
    long long cap;
    bool disabled;
};

struct Dinic {
    int n;
    vector<vector<Edge>> g;
    vector<int> level, it;

    Dinic(int n = 0) : n(n), g(n), level(n), it(n) {}

    int addEdge(int v, int to, long long cap, bool disabled = false) {
        Edge a{to, (int)g[to].size(), cap, disabled};
        Edge b{v, (int)g[v].size(), 0, disabled};
        g[v].push_back(a);
        g[to].push_back(b);
        return (int)g[v].size() - 1;
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);

        while (!q.empty()) {
            int v = q.front();
            q.pop();

            for (const Edge &e : g[v]) {
                if (e.disabled) continue;
                if (e.cap > 0 && level[e.to] == -1) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }

        return level[t] != -1;
    }

    long long dfs(int v, int t, long long pushed) {
        if (v == t || pushed == 0) return pushed;

        for (int &cid = it[v]; cid < (int)g[v].size(); cid++) {
            Edge &e = g[v][cid];

            if (e.disabled) continue;
            if (e.cap <= 0 || level[e.to] != level[v] + 1) continue;

            long long tr = dfs(e.to, t, min(pushed, e.cap));
            if (tr == 0) continue;

            e.cap -= tr;
            g[e.to][e.rev].cap += tr;
            return tr;
        }

        return 0;
    }

    long long maxflow(int s, int t) {
        long long flow = 0;
        const long long INF = (long long)4e18;

        while (bfs(s, t)) {
            fill(it.begin(), it.end(), 0);

            while (true) {
                long long pushed = dfs(s, t, INF);
                if (pushed == 0) break;
                flow += pushed;
            }
        }

        return flow;
    }

    void disableEdgePair(int v, int idx) {
        g[v][idx].disabled = true;
        int to = g[v][idx].to;
        int rev = g[v][idx].rev;
        g[to][rev].disabled = true;
    }
};

struct Pipe {
    int u, v;
    long long z;
    int c;
    int edgeIndex;
    long long residualCapacity;
    long long lower;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    int S = 1;
    int T = N;
    int SS = N + 1;
    int TT = N + 2;

    Dinic din(N + 3);
    vector<Pipe> pipes(M);
    vector<long long> balance(N + 1, 0);

    for (int i = 0; i < M; i++) {
        int u, v, c;
        long long z;
        cin >> u >> v >> z >> c;

        long long lower = (c == 1 ? z : 0);
        long long residualCapacity = z - lower;

        int idx = din.addEdge(u, v, residualCapacity);

        pipes[i] = {u, v, z, c, idx, residualCapacity, lower};

        balance[u] -= lower;
        balance[v] += lower;
    }

    const long long INF = (long long)4e18;

    int backEdgeIndex = din.addEdge(T, S, INF);

    long long need = 0;

    for (int v = 1; v <= N; v++) {
        if (balance[v] > 0) {
            din.addEdge(SS, v, balance[v]);
            need += balance[v];
        } else if (balance[v] < 0) {
            din.addEdge(v, TT, -balance[v]);
        }
    }

    long long got = din.maxflow(SS, TT);

    if (got != need) {
        cout << "Impossible\n";
        return 0;
    }

    din.disableEdgePair(T, backEdgeIndex);

    din.maxflow(T, S);

    vector<long long> flow(M);
    long long answer = 0;

    for (int i = 0; i < M; i++) {
        Edge &e = din.g[pipes[i].u][pipes[i].edgeIndex];
        long long usedResidual = pipes[i].residualCapacity - e.cap;
        flow[i] = pipes[i].lower + usedResidual;

        if (pipes[i].u == S) answer += flow[i];
        if (pipes[i].v == S) answer -= flow[i];
    }

    cout << answer << '\n';
    for (int i = 0; i < M; i++) {
        if (i) cout << ' ';
        cout << flow[i];
    }
    cout << '\n';

    return 0;
}
