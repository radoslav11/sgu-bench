#include <bits/stdc++.h>
using namespace std;

struct Dinic {
    struct Edge {
        int to, rev;
        int cap;
    };

    int n;
    vector<vector<Edge>> g;
    vector<int> level, it;

    Dinic(int n = 0) : n(n), g(n), level(n), it(n) {}

    int addEdge(int v, int to, int cap) {
        Edge a{to, (int)g[to].size(), cap};
        Edge b{v, (int)g[v].size(), 0};
        g[v].push_back(a);
        g[to].push_back(b);
        return (int)g[v].size() - 1;
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);

        while (!q.empty()) {
            int v = q.front();
            q.pop();

            for (auto &e : g[v]) {
                if (e.cap > 0 && level[e.to] == -1) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }

        return level[t] != -1;
    }

    int dfs(int v, int t, int pushed) {
        if (v == t) return pushed;

        for (int &cid = it[v]; cid < (int)g[v].size(); cid++) {
            Edge &e = g[v][cid];

            if (e.cap <= 0 || level[e.to] != level[v] + 1) continue;

            int tr = dfs(e.to, t, min(pushed, e.cap));
            if (tr == 0) continue;

            e.cap -= tr;
            g[e.to][e.rev].cap += tr;
            return tr;
        }

        return 0;
    }

    int maxflow(int s, int t) {
        int flow = 0;
        const int INF = 1e9;

        while (bfs(s, t)) {
            fill(it.begin(), it.end(), 0);

            while (true) {
                int pushed = dfs(s, t, INF);
                if (pushed == 0) break;
                flow += pushed;
            }
        }

        return flow;
    }
};

struct Road {
    int x, y, w, t;
    int idx;
    int low;
};

struct OutEdge {
    int to;
    int cnt;
    int id;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, A, B;
    cin >> n >> m >> A >> B;
    --A;
    --B;

    const int INF = 1000000000;

    int S = n;
    int T = n + 1;

    Dinic din(n + 2);
    vector<Road> roads;
    vector<int> balance(n, 0);

    for (int i = 0; i < m; i++) {
        int x, y, w, t;
        cin >> x >> y >> w >> t;
        --x;
        --y;

        int low = (t == 1 ? w : 0);
        int cap = w - low;

        int idx = din.addEdge(x, y, cap);

        balance[x] -= low;
        balance[y] += low;

        roads.push_back({x, y, w, t, idx, low});
    }

    int backIdx = din.addEdge(B, A, INF);

    int need = 0;
    for (int v = 0; v < n; v++) {
        if (balance[v] > 0) {
            din.addEdge(S, v, balance[v]);
            need += balance[v];
        } else if (balance[v] < 0) {
            din.addEdge(v, T, -balance[v]);
        }
    }

    int got = din.maxflow(S, T);

    if (got != need) {
        cout << 0 << '\n';
        return 0;
    }

    din.maxflow(A, B);

    int days = din.g[B][backIdx].cap;

    if (days == 0) {
        cout << 0 << '\n';
        return 0;
    }

    vector<vector<OutEdge>> adj(n);
    int edgeId = 0;

    for (auto &r : roads) {
        int usedExtra = din.g[r.y][din.g[r.x][r.idx].rev].cap;
        int flow = r.low + usedExtra;

        if (flow > 0) {
            adj[r.x].push_back({r.y, flow, edgeId++});
        }
    }

    int artificialId = edgeId++;
    adj[B].push_back({A, days, artificialId});

    vector<int> ptr(n, 0);
    vector<int> stV, stE;
    vector<int> circuitV, circuitE;

    stV.push_back(A);
    stE.push_back(-1);

    while (!stV.empty()) {
        int v = stV.back();

        while (ptr[v] < (int)adj[v].size() && adj[v][ptr[v]].cnt == 0) {
            ptr[v]++;
        }

        if (ptr[v] == (int)adj[v].size()) {
            circuitV.push_back(v);
            circuitE.push_back(stE.back());
            stV.pop_back();
            stE.pop_back();
        } else {
            OutEdge &e = adj[v][ptr[v]];
            e.cnt--;
            stV.push_back(e.to);
            stE.push_back(e.id);
        }
    }

    reverse(circuitV.begin(), circuitV.end());
    reverse(circuitE.begin(), circuitE.end());

    cout << days << '\n';

    vector<int> route;
    route.push_back(A);

    for (int i = 1; i < (int)circuitV.size(); i++) {
        int eid = circuitE[i];

        if (eid == artificialId) {
            for (int v : route) {
                cout << v + 1 << ' ';
            }
            cout << '\n';

            route.clear();
            route.push_back(A);
        } else {
            route.push_back(circuitV[i]);
        }
    }

    return 0;
}
