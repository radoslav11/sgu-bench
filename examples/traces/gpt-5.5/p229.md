#include <bits/stdc++.h>
using namespace std;

pair<int,int> rotCell(int r, int x, int y) {
    if (r == 0) return {x, y};
    if (r == 1) return {-y, x};
    if (r == 2) return {-x, -y};
    return {y, -x};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;

    vector<string> grid(N);
    for (int i = 0; i < N; i++) cin >> grid[i];

    vector<pair<int,int>> cells;
    int id[20][20];
    memset(id, -1, sizeof(id));

    for (int i = 0; i < N; i++) {
        for (int j = 0; j < N; j++) {
            if (grid[i][j] == '1') {
                id[i][j] = (int)cells.size();
                cells.push_back({i, j});
            }
        }
    }

    int M = (int)cells.size();

    if (M == 0) {
        cout << "YES\n";
        for (int i = 0; i < N; i++) {
            cout << string(N, '0') << '\n';
        }
        return 0;
    }

    if (M % 2 == 1) {
        cout << "NO\n";
        return 0;
    }

    vector<int> answer(M, 0);

    for (int r = 0; r < 4; r++) {
        for (auto p : cells) {
            auto rp = rotCell(r, p.first, p.second);

            for (auto q : cells) {
                int dx = q.first - rp.first;
                int dy = q.second - rp.second;

                vector<int> out(M, -1), in(M, -1);
                bool bad = false;

                for (int i = 0; i < M; i++) {
                    auto [x, y] = cells[i];
                    auto rr = rotCell(r, x, y);
                    int nx = rr.first + dx;
                    int ny = rr.second + dy;

                    if (0 <= nx && nx < N && 0 <= ny && ny < N && id[nx][ny] != -1) {
                        int j = id[nx][ny];
                        out[i] = j;
                        if (in[j] != -1) bad = true;
                        in[j] = i;
                    }
                }

                if (bad) continue;

                vector<int> color(M, -1);
                vector<int> visited(M, 0);
                bool ok = true;

                for (int i = 0; i < M && ok; i++) {
                    if (visited[i]) continue;

                    if (in[i] == -1) {
                        vector<int> path;
                        int v = i;

                        while (v != -1) {
                            if (visited[v]) {
                                ok = false;
                                break;
                            }
                            visited[v] = 1;
                            path.push_back(v);
                            v = out[v];
                        }

                        if (!ok) break;

                        if ((int)path.size() % 2 == 1) {
                            ok = false;
                            break;
                        }

                        for (int k = 0; k < (int)path.size(); k++) {
                            color[path[k]] = (k % 2 == 0 ? 1 : 0);
                        }
                    }
                }

                for (int i = 0; i < M && ok; i++) {
                    if (visited[i]) continue;

                    vector<int> cyc;
                    int v = i;

                    while (!visited[v]) {
                        visited[v] = 1;
                        cyc.push_back(v);
                        v = out[v];

                        if (v == -1) {
                            ok = false;
                            break;
                        }
                    }

                    if (!ok) break;

                    if (v != i) {
                        ok = false;
                        break;
                    }

                    if ((int)cyc.size() % 2 == 1) {
                        ok = false;
                        break;
                    }

                    for (int k = 0; k < (int)cyc.size(); k++) {
                        color[cyc[k]] = (k % 2 == 0 ? 1 : 0);
                    }
                }

                if (!ok) continue;

                for (int i = 0; i < M && ok; i++) {
                    if (color[i] == 1) {
                        if (out[i] == -1 || color[out[i]] != 0) ok = false;
                    } else {
                        if (in[i] == -1 || color[in[i]] != 1) ok = false;
                    }
                }

                if (ok) {
                    answer = color;

                    vector<string> res(N, string(N, '0'));
                    for (int i = 0; i < M; i++) {
                        if (answer[i] == 1) {
                            auto [x, y] = cells[i];
                            res[x][y] = '1';
                        }
                    }

                    cout << "YES\n";
                    for (int i = 0; i < N; i++) cout << res[i] << '\n';
                    return 0;
                }
            }
        }
    }

    cout << "NO\n";
    return 0;
}
