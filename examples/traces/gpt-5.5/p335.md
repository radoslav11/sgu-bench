#include <bits/stdc++.h>
using namespace std;

using int64 = long long;

int64 H, W, Rc, Cc, Rt, Ct;
char First;

long long mdist(long long r1, long long c1, long long r2, long long c2) {
    return llabs(r1 - r2) + llabs(c1 - c2);
}

vector<pair<int64,int64>> neigh2D(int64 r, int64 c) {
    vector<pair<int64,int64>> v;
    if (r > 1) v.push_back({r - 1, c});
    if (r < H) v.push_back({r + 1, c});
    if (c > 1) v.push_back({r, c - 1});
    if (c < W) v.push_back({r, c + 1});
    return v;
}

bool thief_has_safe_move_2D(int64 cr, int64 cc, int64 tr, int64 tc) {
    for (auto [nr, nc] : neigh2D(tr, tc)) {
        if (nr == cr && nc == cc) continue;
        if (mdist(nr, nc, cr, cc) > 1) return true;
    }
    return false;
}

long long solve_path(long long N, long long c, long long t, char first) {
    if (c > t) {
        c = N + 1 - c;
        t = N + 1 - t;
    }

    long long d = t - c;
    long long e = N - t; // distance from thief to the endpoint away from cop

    if (first == 'C') {
        if (d == 1) return 1;
        return 2 * e + d;
    } else {
        if (e > 0) {
            return 2 * e + d;
        } else {
            if (d == 1) return 1;
            if (d == 2) return 2;
            return d + 2;
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> H >> W;
    cin >> Rc >> Cc;
    cin >> Rt >> Ct;
    cin >> First;

    if (H == 1 || W == 1) {
        long long N, c, t;
        if (H == 1) {
            N = W;
            c = Cc;
            t = Ct;
        } else {
            N = H;
            c = Rc;
            t = Rt;
        }
        cout << solve_path(N, c, t, First) << '\n';
        return 0;
    }

    long long dist = mdist(Rc, Cc, Rt, Ct);

    if (First == 'C') {
        if (dist == 1) {
            cout << 1 << '\n';
            return 0;
        }

        for (auto [ncr, ncc] : neigh2D(Rc, Cc)) {
            if (ncr == Rt && ncc == Ct) {
                cout << 1 << '\n';
                return 0;
            }
            if (!thief_has_safe_move_2D(ncr, ncc, Rt, Ct)) {
                cout << 2 << '\n';
                return 0;
            }
        }

        cout << 0 << '\n';
    } else {
        if (!thief_has_safe_move_2D(Rc, Cc, Rt, Ct)) {
            cout << 2 << '\n';
        } else {
            cout << 0 << '\n';
        }
    }

    return 0;
}
