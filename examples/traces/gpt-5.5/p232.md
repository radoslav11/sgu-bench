#include <bits/stdc++.h>
using namespace std;

int max_rotation(const string& s) {
    int n = (int)s.size();
    if (n == 1) return 0;

    int i = 0, j = 1, k = 0;

    while (i < n && j < n && k < n) {
        char a = s[(i + k) % n];
        char b = s[(j + k) % n];

        if (a == b) {
            ++k;
        } else if (a < b) {
            i = i + k + 1;
            if (i == j) ++i;
            k = 0;
        } else {
            j = j + k + 1;
            if (i == j) ++j;
            k = 0;
        }
    }

    return min(i, j);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    long long K;
    cin >> N >> K;

    string D;
    cin >> D;

    int step = (int)(K % N);
    int g = std::gcd(N, step);
    int L = N / g;

    string best;
    string cur;
    cur.resize(L);

    for (int r = 0; r < g; ++r) {
        int pos = r;

        for (int i = 0; i < L; ++i) {
            cur[i] = D[pos];
            pos += step;
            if (pos >= N) pos -= N;
        }

        int p = max_rotation(cur);

        bool better = best.empty();

        if (!better) {
            for (int i = 0; i < L; ++i) {
                char c = cur[(p + i) % L];
                if (c > best[i]) {
                    better = true;
                    break;
                } else if (c < best[i]) {
                    break;
                }
            }
        }

        if (better) {
            best.resize(L);
            for (int i = 0; i < L; ++i) {
                best[i] = cur[(p + i) % L];
            }
        }
    }

    string ans;
    ans.resize(N);

    for (int i = 0; i < N; ++i) {
        ans[i] = best[i % L];
    }

    cout << ans << '\n';

    return 0;
}
