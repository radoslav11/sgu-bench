#include <bits/stdc++.h>
using namespace std;

struct Fenwick {
    int n;
    vector<int> bit;

    Fenwick(int n = 0) : n(n), bit(n + 2, 0) {}

    void add(int idx, int val) {
        for (; idx <= n + 1; idx += idx & -idx) bit[idx] += val;
    }

    void range_add(int l, int r, int val) {
        if (l > r) return;
        add(l, val);
        add(r + 1, -val);
    }

    int sum(int idx) {
        int res = 0;
        for (; idx > 0; idx -= idx & -idx) res += bit[idx];
        return res;
    }
};

struct Hut {
    long long u, v, w;
    int idx;
};

struct Event {
    long long u;
    int l, r, delta;

    bool operator<(const Event& other) const {
        return u < other.u;
    }
};

struct Query {
    long long u;
    int vpos, idx;

    bool operator<(const Query& other) const {
        return u < other.u;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;

    vector<Hut> huts(N);
    vector<long long> allV;

    for (int i = 0; i < N; i++) {
        long long x, y, w;
        cin >> x >> y >> w;

        huts[i].u = x + y;
        huts[i].v = x - y;
        huts[i].w = w;
        huts[i].idx = i + 1;

        allV.push_back(huts[i].v);
    }

    sort(allV.begin(), allV.end());
    allV.erase(unique(allV.begin(), allV.end()), allV.end());

    vector<Event> events;
    vector<Query> queries;

    for (int i = 0; i < N; i++) {
        long long ul = huts[i].u - huts[i].w;
        long long ur = huts[i].u + huts[i].w;
        long long vl = huts[i].v - huts[i].w;
        long long vr = huts[i].v + huts[i].w;

        int l = int(lower_bound(allV.begin(), allV.end(), vl) - allV.begin()) + 1;
        int r = int(upper_bound(allV.begin(), allV.end(), vr) - allV.begin());

        if (l <= r) {
            events.push_back({ul, l, r, +1});
            events.push_back({ur + 1, l, r, -1});
        }

        int vpos = int(lower_bound(allV.begin(), allV.end(), huts[i].v) - allV.begin()) + 1;
        queries.push_back({huts[i].u, vpos, i});
    }

    sort(events.begin(), events.end());
    sort(queries.begin(), queries.end());

    Fenwick fw((int)allV.size() + 2);
    vector<int> cnt(N, 0);

    int ptr = 0;
    for (const Query& q : queries) {
        while (ptr < (int)events.size() && events[ptr].u <= q.u) {
            fw.range_add(events[ptr].l, events[ptr].r, events[ptr].delta);
            ptr++;
        }

        cnt[q.idx] = fw.sum(q.vpos);
    }

    vector<int> answer;
    for (int i = 0; i < N; i++) {
        if (cnt[i] - 1 >= K) {
            answer.push_back(i + 1);
        }
    }

    cout << answer.size() << '\n';
    if (!answer.empty()) {
        for (int i = 0; i < (int)answer.size(); i++) {
            if (i) cout << ' ';
            cout << answer[i];
        }
        cout << '\n';
    }

    return 0;
}
