#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    long long m;
    cin >> n >> m;

    if (n == 1) {
        cout << (1 % m) << '\n';
        return 0;
    }

    vector<long long> up(n + 2, 0), down(n + 2, 0);
    up[1] = down[1] = 1 % m;

    for (int len = 2; len <= n; ++len) {
        vector<long long> newUp(n + 2, 0), newDown(n + 2, 0);

        long long pref = 0;
        for (int j = 1; j <= len; ++j) {
            newUp[j] = pref;
            if (j <= len - 1) {
                pref += down[j];
                pref %= m;
            }
        }

        long long suf = 0;
        for (int j = len; j >= 1; --j) {
            if (j <= len - 1) {
                suf += up[j];
                suf %= m;
            }
            newDown[j] = suf;
        }

        up.swap(newUp);
        down.swap(newDown);
    }

    long long ans = 0;
    for (int j = 1; j <= n; ++j) {
        ans += up[j] + down[j];
        ans %= m;
    }

    cout << ans << '\n';
    return 0;
}
