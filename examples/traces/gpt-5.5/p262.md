#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M, K;
    cin >> N >> M >> K;

    int P = N * M;
    vector<string> sym(K, string(P, '0'));

    for (int k = 0; k < K; ++k) {
        for (int i = 0; i < N; ++i) {
            string row;
            cin >> row;
            for (int j = 0; j < M; ++j) {
                sym[k][i * M + j] = row[j];
            }
        }
    }

    vector<pair<int, int>> pairs;
    for (int i = 0; i < K; ++i) {
        for (int j = i + 1; j < K; ++j) {
            pairs.push_back({i, j});
        }
    }

    int Q = pairs.size();
    int FULL = (1 << Q) - 1;

    vector<int> cover(P, 0);

    for (int p = 0; p < P; ++p) {
        int mask = 0;
        for (int q = 0; q < Q; ++q) {
            int a = pairs[q].first;
            int b = pairs[q].second;
            if (sym[a][p] != sym[b][p]) {
                mask |= (1 << q);
            }
        }
        cover[p] = mask;
    }

    const int INF = 1e9;
    vector<int> dp(1 << Q, INF);
    vector<int> prevMask(1 << Q, -1);
    vector<int> chosenPixel(1 << Q, -1);

    dp[0] = 0;

    for (int p = 0; p < P; ++p) {
        if (cover[p] == 0) continue;

        for (int mask = 0; mask <= FULL; ++mask) {
            if (dp[mask] == INF) continue;

            int newMask = mask | cover[p];
            if (dp[newMask] > dp[mask] + 1) {
                dp[newMask] = dp[mask] + 1;
                prevMask[newMask] = mask;
                chosenPixel[newMask] = p;
            }
        }
    }

    vector<int> answer(P, 0);
    int mask = FULL;

    while (mask != 0) {
        int p = chosenPixel[mask];
        answer[p] = 1;
        mask = prevMask[mask];
    }

    cout << dp[FULL] << '\n';

    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < M; ++j) {
            cout << answer[i * M + j];
        }
        cout << '\n';
    }

    return 0;
}
