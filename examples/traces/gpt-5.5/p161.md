#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100;

int N, M;
bitset<MAXN> reachability[MAXN], comparableMask[MAXN], predMask[MAXN];

struct Key {
    unsigned long long a, b;
    bool operator==(const Key& other) const {
        return a == other.a && b == other.b;
    }
};

struct KeyHash {
    size_t operator()(const Key& k) const {
        return k.a ^ (k.b * 1000003ULL);
    }
};

Key makeKey(const bitset<MAXN>& bs) {
    unsigned long long a = 0, b = 0;
    for (int i = 0; i < min(64, N); i++) {
        if (bs[i]) a |= (1ULL << i);
    }
    for (int i = 64; i < N; i++) {
        if (bs[i]) b |= (1ULL << (i - 64));
    }
    return {a, b};
}

vector<bitset<MAXN>> elems;
unordered_map<Key, int, KeyHash> idOf;
vector<bitset<MAXN>> downClosure;

void addElement(const bitset<MAXN>& bs) {
    int id = (int)elems.size();
    elems.push_back(bs);
    idOf[makeKey(bs)] = id;
}

void enumerateAntichains(bitset<MAXN>& cand, bitset<MAXN>& cur) {
    int v = -1;
    for (int i = 0; i < N; i++) {
        if (cand[i]) {
            v = i;
            break;
        }
    }

    if (v == -1) {
        addElement(cur);
        return;
    }

    cand.reset(v);
    enumerateAntichains(cand, cur);

    bitset<MAXN> oldCand = cand;
    cur.set(v);
    cand &= ~comparableMask[v];
    enumerateAntichains(cand, cur);
    cand = oldCand;
    cur.reset(v);

    cand.set(v);
}

bitset<MAXN> MaxSet(bitset<MAXN> s) {
    bitset<MAXN> res;
    for (int x = 0; x < N; x++) {
        if (!s[x]) continue;
        bitset<MAXN> tmp = reachability[x] & s;
        tmp.reset(x);
        if (!tmp.any()) res.set(x);
    }
    return res;
}

int getId(const bitset<MAXN>& bs) {
    return idOf[makeKey(bs)];
}

vector<vector<int>> opAnd, opOr, opImp, opEq;
int idZero, idOne;

int calcAnd(int a, int b) {
    return getId(MaxSet(elems[a] | elems[b]));
}

int calcOr(int a, int b) {
    bitset<MAXN> s = downClosure[a] & downClosure[b];
    return getId(MaxSet(s));
}

int calcImp(int a, int b) {
    bitset<MAXN> res;
    for (int x = 0; x < N; x++) {
        if (!elems[b][x]) continue;
        bitset<MAXN> tmp = reachability[x] & elems[a];
        if (!tmp.any()) res.set(x);
    }
    return getId(res);
}

enum TokenType {
    T_END, T_ZERO, T_ONE, T_VAR, T_LP, T_RP, T_NOT, T_AND, T_OR, T_IMP, T_EQ
};

struct Token {
    TokenType type;
    char var;
};

vector<Token> tokenize(const string& s) {
    vector<Token> t;
    for (int i = 0; i < (int)s.size(); ) {
        if (isspace((unsigned char)s[i])) {
            i++;
        } else if (s[i] == '0') {
            t.push_back({T_ZERO, 0});
            i++;
        } else if (s[i] == '1') {
            t.push_back({T_ONE, 0});
            i++;
        } else if ('A' <= s[i] && s[i] <= 'Z') {
            t.push_back({T_VAR, s[i]});
            i++;
        } else if (s[i] == '(') {
            t.push_back({T_LP, 0});
            i++;
        } else if (s[i] == ')') {
            t.push_back({T_RP, 0});
            i++;
        } else if (s[i] == '~') {
            t.push_back({T_NOT, 0});
            i++;
        } else if (s[i] == '&') {
            t.push_back({T_AND, 0});
            i++;
        } else if (s[i] == '|') {
            t.push_back({T_OR, 0});
            i++;
        } else if (s[i] == '=') {
            if (i + 1 < (int)s.size() && s[i + 1] == '>') {
                t.push_back({T_IMP, 0});
                i += 2;
            } else {
                t.push_back({T_EQ, 0});
                i++;
            }
        } else {
            i++;
        }
    }
    t.push_back({T_END, 0});
    return t;
}

struct Node {
    int type;
    int val;
    Node *l, *r;
    Node(int type, int val = 0, Node* l = nullptr, Node* r = nullptr)
        : type(type), val(val), l(l), r(r) {}
};

enum NodeType {
    N_CONST, N_VAR, N_NOT, N_AND, N_OR, N_IMP, N_EQ
};

struct Parser {
    vector<Token> tok;
    int pos = 0;

    Parser(const vector<Token>& tok) : tok(tok) {}

    Node* parse() {
        return parseEq();
    }

    Node* parseEq() {
        vector<Node*> parts;
        parts.push_back(parseImp());
        while (tok[pos].type == T_EQ) {
            pos++;
            parts.push_back(parseImp());
        }

        if (parts.size() == 1) return parts[0];

        Node* res = new Node(N_EQ, 0, parts[0], parts[1]);
        for (int i = 1; i + 1 < (int)parts.size(); i++) {
            Node* eq = new Node(N_EQ, 0, parts[i], parts[i + 1]);
            res = new Node(N_AND, 0, res, eq);
        }
        return res;
    }

    Node* parseImp() {
        Node* left = parseOr();
        if (tok[pos].type == T_IMP) {
            pos++;
            Node* right = parseImp();
            return new Node(N_IMP, 0, left, right);
        }
        return left;
    }

    Node* parseOr() {
        Node* res = parseAnd();
        while (tok[pos].type == T_OR) {
            pos++;
            res = new Node(N_OR, 0, res, parseAnd());
        }
        return res;
    }

    Node* parseAnd() {
        Node* res = parseUnary();
        while (tok[pos].type == T_AND) {
            pos++;
            res = new Node(N_AND, 0, res, parseUnary());
        }
        return res;
    }

    Node* parseUnary() {
        if (tok[pos].type == T_NOT) {
            pos++;
            return new Node(N_NOT, 0, parseUnary());
        }

        if (tok[pos].type == T_LP) {
            pos++;
            Node* res = parseEq();
            pos++;
            return res;
        }

        if (tok[pos].type == T_ZERO) {
            pos++;
            return new Node(N_CONST, idZero);
        }

        if (tok[pos].type == T_ONE) {
            pos++;
            return new Node(N_CONST, idOne);
        }

        char v = tok[pos].var;
        pos++;
        return new Node(N_VAR, v - 'A');
    }
};

bool usedVar[26];

void collectVars(Node* node) {
    if (!node) return;
    if (node->type == N_VAR) usedVar[node->val] = true;
    collectVars(node->l);
    collectVars(node->r);
}

int eval(Node* node, const int assign[26]) {
    switch (node->type) {
        case N_CONST:
            return node->val;
        case N_VAR:
            return assign[node->val];
        case N_NOT:
            return opImp[eval(node->l, assign)][idZero];
        case N_AND:
            return opAnd[eval(node->l, assign)][eval(node->r, assign)];
        case N_OR:
            return opOr[eval(node->l, assign)][eval(node->r, assign)];
        case N_IMP:
            return opImp[eval(node->l, assign)][eval(node->r, assign)];
        case N_EQ: {
            int a = eval(node->l, assign);
            int b = eval(node->r, assign);
            return opEq[a][b];
        }
    }
    return idZero;
}

bool checkValid(Node* root) {
    memset(usedVar, 0, sizeof(usedVar));
    collectVars(root);

    vector<int> vars;
    for (int i = 0; i < 26; i++) {
        if (usedVar[i]) vars.push_back(i);
    }

    int D = elems.size();
    int assign[26];
    for (int i = 0; i < 26; i++) assign[i] = idOne;

    bool valid = true;

    function<void(int)> dfs = [&](int p) {
        if (!valid) return;

        if (p == (int)vars.size()) {
            if (eval(root, assign) != idOne) valid = false;
            return;
        }

        for (int x = 0; x < D && valid; x++) {
            assign[vars[p]] = x;
            dfs(p + 1);
        }
    };

    dfs(0);
    return valid;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> N >> M;

    for (int i = 0; i < N; i++) reachability[i].set(i);

    for (int i = 0; i < M; i++) {
        int s, t;
        cin >> s >> t;
        --s;
        --t;
        reachability[s].set(t);
    }

    for (int k = 0; k < N; k++) {
        for (int i = 0; i < N; i++) {
            if (reachability[i][k]) reachability[i] |= reachability[k];
        }
    }

    for (int i = 0; i < N; i++) {
        comparableMask[i].reset();
        for (int j = 0; j < N; j++) {
            if (reachability[i][j] || reachability[j][i]) comparableMask[i].set(j);
            if (reachability[j][i]) predMask[i].set(j);
        }
    }

    bitset<MAXN> cand, cur;
    for (int i = 0; i < N; i++) cand.set(i);
    enumerateAntichains(cand, cur);

    int D = elems.size();

    bitset<MAXN> emptySet;
    idOne = getId(emptySet);

    bitset<MAXN> allSet;
    for (int i = 0; i < N; i++) allSet.set(i);
    idZero = getId(MaxSet(allSet));

    downClosure.assign(D, bitset<MAXN>());
    for (int id = 0; id < D; id++) {
        for (int y = 0; y < N; y++) {
            if (elems[id][y]) downClosure[id] |= predMask[y];
        }
    }

    opAnd.assign(D, vector<int>(D));
    opOr.assign(D, vector<int>(D));
    opImp.assign(D, vector<int>(D));
    opEq.assign(D, vector<int>(D));

    for (int i = 0; i < D; i++) {
        for (int j = 0; j < D; j++) {
            opAnd[i][j] = calcAnd(i, j);
            opOr[i][j] = calcOr(i, j);
            opImp[i][j] = calcImp(i, j);
        }
    }

    for (int i = 0; i < D; i++) {
        for (int j = 0; j < D; j++) {
            opEq[i][j] = opAnd[opImp[i][j]][opImp[j][i]];
        }
    }

    int K;
    cin >> K;
    string line;
    getline(cin, line);

    while (K--) {
        getline(cin, line);
        vector<Token> tokens = tokenize(line);
        Parser parser(tokens);
        Node* root = parser.parse();

        cout << (checkValid(root) ? "valid" : "invalid") << '\n';
    }

    return 0;
}
