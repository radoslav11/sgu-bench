#include <bits/stdc++.h>
using namespace std;

struct FenwickMax {
    int n;
    vector<int> bit;

    FenwickMax(int n = 0) {
        init(n);
    }

    void init(int n_) {
        n = n_;
        bit.assign(n + 1, 0);
    }

    void update(int pos, int val) {
        for (; pos <= n; pos += pos & -pos)
            bit[pos] = max(bit[pos], val);
    }

    int query(int pos) const {
        int res = 0;
        for (; pos > 0; pos -= pos & -pos)
            res = max(res, bit[pos]);
        return res;
    }
};

struct City {
    int x, y, idx, yr;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<City> cities(n);
    vector<int> ys;

    for (int i = 0; i < n; ++i) {
        cin >> cities[i].x >> cities[i].y;
        cities[i].idx = i;
        ys.push_back(cities[i].y);
    }

    sort(ys.begin(), ys.end());
    ys.erase(unique(ys.begin(), ys.end()), ys.end());

    int m = (int)ys.size();

    for (auto &c : cities) {
        c.yr = int(lower_bound(ys.begin(), ys.end(), c.y) - ys.begin()) + 1;
    }

    vector<int> f(n), g(n);
    vector<int> ord(n);
    iota(ord.begin(), ord.end(), 0);

    sort(ord.begin(), ord.end(), [&](int a, int b) {
        if (cities[a].x != cities[b].x) return cities[a].x < cities[b].x;
        return cities[a].y < cities[b].y;
    });

    FenwickMax fw(m);

    for (int i = 0; i < n;) {
        int j = i;
        while (j < n && cities[ord[j]].x == cities[ord[i]].x) ++j;

        for (int k = i; k < j; ++k) {
            int id = ord[k];
            f[id] = fw.query(cities[id].yr - 1) + 1;
        }

        for (int k = i; k < j; ++k) {
            int id = ord[k];
            fw.update(cities[id].yr, f[id]);
        }

        i = j;
    }

    sort(ord.begin(), ord.end(), [&](int a, int b) {
        if (cities[a].x != cities[b].x) return cities[a].x > cities[b].x;
        return cities[a].y > cities[b].y;
    });

    fw.init(m);

    for (int i = 0; i < n;) {
        int j = i;
        while (j < n && cities[ord[j]].x == cities[ord[i]].x) ++j;

        for (int k = i; k < j; ++k) {
            int id = ord[k];
            int rev = m - cities[id].yr + 1;
            g[id] = fw.query(rev - 1) + 1;
        }

        for (int k = i; k < j; ++k) {
            int id = ord[k];
            int rev = m - cities[id].yr + 1;
            fw.update(rev, g[id]);
        }

        i = j;
    }

    int L = 0;
    for (int x : f) L = max(L, x);

    vector<char> possible(n, false);
    vector<int> layerCnt(L + 1, 0);

    for (int i = 0; i < n; ++i) {
        if (f[i] + g[i] - 1 == L) {
            possible[i] = true;
            layerCnt[f[i]]++;
        }
    }

    vector<int> A, B;

    for (int i = 0; i < n; ++i) {
        if (possible[i]) {
            A.push_back(i + 1);
            if (layerCnt[f[i]] == 1) {
                B.push_back(i + 1);
            }
        }
    }

    cout << A.size();
    for (int id : A) cout << ' ' << id;
    cout << '\n';

    cout << B.size();
    for (int id : B) cout << ' ' << id;
    cout << '\n';

    return 0;
}
