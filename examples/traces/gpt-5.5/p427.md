#include <bits/stdc++.h>
using namespace std;

struct P3 {
    double x, y, z;
    string sx, sy, sz;
};

struct P2 {
    double x, y;
    int id;
};

static double cross2(const P2& a, const P2& b, const P2& c) {
    return (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
}

static P3 cross3(const P3& a, const P3& b) {
    P3 r;
    r.x = a.y * b.z - a.z * b.y;
    r.y = a.z * b.x - a.x * b.z;
    r.z = a.x * b.y - a.y * b.x;
    return r;
}

static double dot3(const P3& a, const P3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static double norm3(const P3& a) {
    return sqrt(dot3(a, a));
}

static P3 normalize(P3 a) {
    double n = norm3(a);
    a.x /= n;
    a.y /= n;
    a.z /= n;
    return a;
}

static vector<int> convex_hull(vector<P2> p) {
    sort(p.begin(), p.end(), [](const P2& a, const P2& b) {
        if (fabs(a.x - b.x) > 1e-12) return a.x < b.x;
        return a.y < b.y;
    });

    vector<P2> h;
    const double EPS = 1e-10;

    for (const auto& pt : p) {
        while (h.size() >= 2 && cross2(h[h.size() - 2], h.back(), pt) <= EPS)
            h.pop_back();
        h.push_back(pt);
    }

    int lower = (int)h.size();

    for (int i = (int)p.size() - 2; i >= 0; --i) {
        const auto& pt = p[i];
        while ((int)h.size() > lower && cross2(h[h.size() - 2], h.back(), pt) <= EPS)
            h.pop_back();
        h.push_back(pt);
    }

    if (!h.empty()) h.pop_back();

    vector<int> res;
    for (auto &pt : h) res.push_back(pt.id);
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int f;
    if (!(cin >> f)) return 0;

    vector<P3> vertices;
    unordered_map<string, int> id;
    set<pair<int,int>> edgeSet;

    auto get_id = [&](string sx, string sy, string sz) -> int {
        string key = sx + "#" + sy + "#" + sz;
        auto it = id.find(key);
        if (it != id.end()) return it->second;

        int nid = (int)vertices.size();
        id[key] = nid;

        P3 p;
        p.sx = sx; p.sy = sy; p.sz = sz;
        p.x = stod(sx);
        p.y = stod(sy);
        p.z = stod(sz);
        vertices.push_back(p);

        return nid;
    };

    for (int i = 0; i < f; ++i) {
        int k;
        cin >> k;
        vector<int> face(k);

        for (int j = 0; j < k; ++j) {
            string sx, sy, sz;
            cin >> sx >> sy >> sz;
            face[j] = get_id(sx, sy, sz);
        }

        for (int j = 0; j < k; ++j) {
            int a = face[j];
            int b = face[(j + 1) % k];
            if (a > b) swap(a, b);
            edgeSet.insert({a, b});
        }
    }

    int n = (int)vertices.size();

    auto is_edge = [&](int a, int b) -> bool {
        if (a > b) swap(a, b);
        return edgeSet.count({a, b});
    };

    vector<P3> directions;

    directions.push_back({1, 0, 0, "", "", ""});
    directions.push_back({0, 1, 0, "", "", ""});
    directions.push_back({0, 0, 1, "", "", ""});
    directions.push_back({1, 1, 1, "", "", ""});
    directions.push_back({1, 2, 3, "", "", ""});
    directions.push_back({3, 5, 7, "", "", ""});

    mt19937_64 rng(427427);
    normal_distribution<double> nd(0.0, 1.0);

    for (int i = 0; i < 5000; ++i) {
        P3 d{nd(rng), nd(rng), nd(rng), "", "", ""};
        if (norm3(d) > 1e-12) directions.push_back(d);
    }

    for (P3 dir : directions) {
        dir = normalize(dir);

        P3 tmp;
        if (fabs(dir.x) < 0.8) tmp = {1, 0, 0, "", "", ""};
        else tmp = {0, 1, 0, "", "", ""};

        P3 u = cross3(dir, tmp);
        if (norm3(u) < 1e-12) continue;
        u = normalize(u);

        P3 v = cross3(dir, u);
        v = normalize(v);

        vector<P2> projected;
        projected.reserve(n);

        for (int i = 0; i < n; ++i) {
            projected.push_back({dot3(vertices[i], u), dot3(vertices[i], v), i});
        }

        vector<int> hull = convex_hull(projected);

        if ((int)hull.size() != n) continue;

        bool ok = true;
        for (int i = 0; i < n; ++i) {
            if (!is_edge(hull[i], hull[(i + 1) % n])) {
                ok = false;
                break;
            }
        }

        if (!ok) continue;

        cout << "Yes\n";
        cout << n << '\n';
        for (int idx : hull) {
            cout << vertices[idx].sx << ' ' << vertices[idx].sy << ' ' << vertices[idx].sz << '\n';
        }
        return 0;
    }

    cout << "No\n";
    return 0;
}
