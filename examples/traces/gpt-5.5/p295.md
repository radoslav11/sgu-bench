```cpp
#include <bits/stdc++.h>
#include <boost/multiprecision/cpp_int.hpp>

using namespace std;
using boost::multiprecision::cpp_int;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int q, c;
    cin >> q >> c;

    string s;
    getline(cin, s); // end of first line
    getline(cin, s);

    string similar = "ABCEHKMOPTXaceopxy";

    int originalLen = (int)s.size();
    int lettersLen = 0;
    int changeable = 0;
    int words = 0;

    bool inWord = false;
    for (char ch : s) {
        if (ch == ' ') {
            inWord = false;
        } else {
            lettersLen++;
            if (!inWord) {
                words++;
                inWord = true;
            }
            if (similar.find(ch) != string::npos) {
                changeable++;
            }
        }
    }

    vector<vector<cpp_int>> C(70, vector<cpp_int>(70));
    for (int i = 0; i < 70; i++) {
        C[i][0] = C[i][i] = 1;
        for (int j = 1; j < i; j++) {
            C[i][j] = C[i - 1][j - 1] + C[i - 1][j];
        }
    }

    cpp_int letterVariants = cpp_int(1) << changeable;
    cpp_int spaceVariants = 0;

    if (words == 0) {
        for (int len = q; len <= c; len++) {
            spaceVariants += 1;
        }
    } else {
        int gaps = words - 1;

        for (int len = q; len <= c; len++) {
            int spaces = len - lettersLen;
            if (spaces < gaps) continue;

            // leading + trailing are nonnegative,
            // every internal gap must contain at least one space.
            // Number of ways: C(spaces + 1, gaps + 1)
            spaceVariants += C[spaces + 1][gaps + 1];
        }
    }

    cpp_int answer = letterVariants * spaceVariants;

    // The original nickname itself is not considered its mult.
    if (q <= originalLen && originalLen <= c) {
        answer -= 1;
    }

    cout << answer << '\n';

    return 0;
}
```
