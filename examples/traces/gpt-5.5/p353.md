#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N, k1, k2, p1, p2, p3;
    cin >> N >> k1 >> k2 >> p1 >> p2 >> p3;

    if (N < p1) {
        cout << 0 << '\n';
        return 0;
    }

    long long ans = k1;
    long long balance = N - p1;

    if (balance <= 0) {
        cout << ans << '\n';
        return 0;
    }

    long long canTalkSecondPart = (balance + p2 - 1) / p2;

    if (canTalkSecondPart <= k2) {
        ans += canTalkSecondPart;
        cout << ans << '\n';
        return 0;
    }

    ans += k2;
    balance -= k2 * p2;

    if (balance <= 0) {
        cout << ans << '\n';
        return 0;
    }

    ans += (balance + p3 - 1) / p3;

    cout << ans << '\n';

    return 0;
}
