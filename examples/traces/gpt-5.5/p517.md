#include <bits/stdc++.h>
#include <boost/multiprecision/cpp_int.hpp>

using namespace std;
using boost::multiprecision::cpp_int;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int m, n;
    cin >> m >> n;

    if ((m * n) % 2 == 1) {
        cout << 0 << '\n';
        return 0;
    }

    int a = min(m, n);
    int b = max(m, n);

    if (a == 1) {
        cout << 1 << '\n';
        return 0;
    }

    if (a == 2) {
        vector<cpp_int> dp0(b + 3), dp1(b + 3);
        dp0[0] = 1;

        for (int i = 0; i <= b; ++i) {
            if (dp0[i] != 0) {
                if (i + 1 <= b) dp0[i + 1] += dp0[i];
                if (i + 2 <= b) dp1[i + 2] += dp0[i];
            }
            if (dp1[i] != 0) {
                if (i + 1 <= b) dp0[i + 1] += dp1[i];
            }
        }

        cout << dp0[b] + dp1[b] << '\n';
        return 0;
    }

    cout << 2 << '\n';
    return 0;
}
