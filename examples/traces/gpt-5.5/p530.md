#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<vector<int>> g(n);
    for (int i = 0; i < m; ++i) {
        int a, b;
        cin >> a >> b;
        --a; --b;
        g[a].push_back(b);
        g[b].push_back(a);
    }

    const int INF = 1e9;
    vector<vector<int>> dp(8, vector<int>(n, INF));

    dp[1][0] = 0;
    dp[2][1] = 0;
    dp[4][2] = 0;

    for (int mask = 1; mask < 8; ++mask) {
        for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
            int other = mask ^ sub;
            if (other == 0) continue;

            for (int v = 0; v < n; ++v) {
                if (dp[sub][v] != INF && dp[other][v] != INF) {
                    dp[mask][v] = min(dp[mask][v], dp[sub][v] + dp[other][v]);
                }
            }
        }

        queue<int> q;
        vector<char> inq(n, false);

        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] != INF) {
                q.push(v);
                inq[v] = true;
            }
        }

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            inq[v] = false;

            for (int to : g[v]) {
                if (dp[mask][to] > dp[mask][v] + 1) {
                    dp[mask][to] = dp[mask][v] + 1;
                    if (!inq[to]) {
                        q.push(to);
                        inq[to] = true;
                    }
                }
            }
        }
    }

    int steiner = INF;
    for (int v = 0; v < n; ++v) {
        steiner = min(steiner, dp[7][v]);
    }

    cout << (steiner % 2 == 1 ? 1 : 2) << '\n';

    return 0;
}
