#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    int i, j;
    cin >> i >> j;
    --i; --j;

    vector<vector<bool>> destroyed(N, vector<bool>(M, false));

    int di[4] = {1, 0, -1, 0};
    int dj[4] = {0, -1, 0, 1};
    char mv[4] = {'D', 'L', 'U', 'R'};

    string answer;

    while (true) {
        destroyed[i][j] = true;

        bool moved = false;
        for (int k = 0; k < 4; ++k) {
            int ni = i + di[k];
            int nj = j + dj[k];

            if (ni >= 0 && ni < N && nj >= 0 && nj < M && !destroyed[ni][nj]) {
                answer += mv[k];
                i = ni;
                j = nj;
                moved = true;
                break;
            }
        }

        if (!moved) break;
    }

    cout << answer;

    return 0;
}
