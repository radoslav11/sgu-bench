#include <bits/stdc++.h>
using namespace std;

const int N = 1024;

struct Parser {
    string s;
    int pos = 0;
    bitset<N> var[10];

    Parser(const string& str) : s(str) {
        for (int m = 0; m < N; ++m) {
            for (int i = 0; i < 10; ++i) {
                if (m & (1 << i)) var[i].set(m);
            }
        }
    }

    bitset<N> parseExpression() {
        return parseLow();
    }

    bitset<N> parseLow() {
        bitset<N> res = parseMid();

        while (pos < (int)s.size()) {
            string op;

            if (s.compare(pos, 2, "||") == 0) {
                op = "||";
                pos += 2;
            } else if (s.compare(pos, 3, "<=>") == 0) {
                op = "<=>";
                pos += 3;
            } else if (s.compare(pos, 2, "=>") == 0) {
                op = "=>";
                pos += 2;
            } else if (s[pos] == '#') {
                op = "#";
                pos++;
            } else {
                break;
            }

            bitset<N> rhs = parseMid();

            if (op == "||") {
                res |= rhs;
            } else if (op == "<=>") {
                res = ~(res ^ rhs);
            } else if (op == "=>") {
                res = (~res) | rhs;
            } else if (op == "#") {
                res ^= rhs;
            }
        }

        return res;
    }

    bitset<N> parseMid() {
        bitset<N> res = parseUnary();

        while (pos < (int)s.size() && s[pos] == '&') {
            pos++;
            bitset<N> rhs = parseUnary();
            res &= rhs;
        }

        return res;
    }

    bitset<N> parseUnary() {
        if (s[pos] == '!') {
            pos++;
            return ~parseUnary();
        }

        if (s[pos] == '(') {
            pos++;
            bitset<N> res = parseExpression();
            pos++;
            return res;
        }

        int id = s[pos] - 'a';
        pos++;
        return var[id];
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string expr;
    getline(cin, expr);

    Parser parser(expr);
    bitset<N> value = parser.parseExpression();

    bool any = false;
    for (int m = 0; m < N; ++m) {
        if (value[m]) {
            any = true;
            break;
        }
    }

    if (!any) {
        cout << "a&!a\n";
        return 0;
    }

    bool firstTerm = true;

    for (int m = 0; m < N; ++m) {
        if (!value[m]) continue;

        if (!firstTerm) cout << "||";
        firstTerm = false;

        for (int i = 0; i < 10; ++i) {
            if (i) cout << '&';

            if (!(m & (1 << i))) cout << '!';
            cout << char('a' + i);
        }
    }

    cout << '\n';
    return 0;
}
