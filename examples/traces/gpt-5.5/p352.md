#include <bits/stdc++.h>
using namespace std;

using ll = long long;
const ll INF = (ll)4e18;

struct Edge {
    int a, b;
    ll w;
    int t;
};

struct TreeEdge {
    int to;
    ll w;
};

struct SegTree {
    int n;
    vector<ll> lazy;

    SegTree(int n = 0) : n(n), lazy(4 * n + 5, INF) {}

    void update(int v, int tl, int tr, int l, int r, ll val) {
        if (l > r) return;
        if (l == tl && r == tr) {
            lazy[v] = min(lazy[v], val);
            return;
        }
        int tm = (tl + tr) / 2;
        update(v * 2, tl, tm, l, min(r, tm), val);
        update(v * 2 + 1, tm + 1, tr, max(l, tm + 1), r, val);
    }

    void update(int l, int r, ll val) {
        if (l <= r) update(1, 1, n, l, r, val);
    }

    ll query(int v, int tl, int tr, int pos) {
        if (tl == tr) return lazy[v];
        int tm = (tl + tr) / 2;
        ll res = lazy[v];
        if (pos <= tm) res = min(res, query(v * 2, tl, tm, pos));
        else res = min(res, query(v * 2 + 1, tm + 1, tr, pos));
        return res;
    }

    ll query(int pos) {
        return query(1, 1, n, pos);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    vector<Edge> edges;
    vector<vector<TreeEdge>> tree(N + 1);

    for (int i = 0; i < M; i++) {
        int a, b, t;
        ll l;
        cin >> a >> b >> l >> t;
        edges.push_back({a, b, l, t});
        if (t == 1) {
            tree[a].push_back({b, l});
            tree[b].push_back({a, l});
        }
    }

    vector<int> parent(N + 1), depth(N + 1), sz(N + 1), heavy(N + 1, 0);
    vector<ll> dist(N + 1, 0);

    vector<int> order;
    order.reserve(N);

    stack<int> st;
    st.push(1);
    parent[1] = 0;

    while (!st.empty()) {
        int v = st.top();
        st.pop();
        order.push_back(v);

        for (auto e : tree[v]) {
            int to = e.to;
            if (to == parent[v]) continue;
            parent[to] = v;
            depth[to] = depth[v] + 1;
            dist[to] = dist[v] + e.w;
            st.push(to);
        }
    }

    for (int i = N - 1; i >= 0; i--) {
        int v = order[i];
        sz[v] = 1;
        int bestSize = 0;

        for (auto e : tree[v]) {
            int to = e.to;
            if (parent[to] != v) continue;
            sz[v] += sz[to];
            if (sz[to] > bestSize) {
                bestSize = sz[to];
                heavy[v] = to;
            }
        }
    }

    vector<int> head(N + 1), pos(N + 1);
    int timer = 0;

    for (int v = 1; v <= N; v++) {
        if (v == 1 || heavy[parent[v]] != v) {
            for (int u = v; u != 0; u = heavy[u]) {
                head[u] = v;
                pos[u] = ++timer;
            }
        }
    }

    auto lca = [&](int a, int b) {
        while (head[a] != head[b]) {
            if (depth[head[a]] > depth[head[b]]) a = parent[head[a]];
            else b = parent[head[b]];
        }
        return depth[a] < depth[b] ? a : b;
    };

    SegTree seg(N);

    auto updatePathExcludingAncestor = [&](int u, int anc, ll val) {
        while (head[u] != head[anc]) {
            seg.update(pos[head[u]], pos[u], val);
            u = parent[head[u]];
        }
        if (pos[anc] + 1 <= pos[u]) {
            seg.update(pos[anc] + 1, pos[u], val);
        }
    };

    for (const auto &e : edges) {
        if (e.t == 1) continue;

        int a = e.a;
        int b = e.b;
        int c = lca(a, b);

        ll val = dist[a] + dist[b] + e.w;

        updatePathExcludingAncestor(a, c, val);
        updatePathExcludingAncestor(b, c, val);
    }

    for (int v = 2; v <= N; v++) {
        ll best = seg.query(pos[v]);

        if (v > 2) cout << ' ';

        if (best >= INF / 2) cout << -1;
        else cout << best - dist[v];
    }

    cout << '\n';

    return 0;
}
