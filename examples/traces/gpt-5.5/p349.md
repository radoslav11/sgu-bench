#include <bits/stdc++.h>
using namespace std;

struct Interval {
    long double l, r;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    vector<Interval> wolves, sheep;

    auto getInterval = [](int x1, int y1, int x2, int y2) {
        long double a1 = atan2((long double)y1, (long double)x1);
        long double a2 = atan2((long double)y2, (long double)x2);
        if (a1 > a2) swap(a1, a2);
        return Interval{a1, a2};
    };

    for (int i = 0; i < N; ++i) {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        wolves.push_back(getInterval(x1, y1, x2, y2));
    }

    for (int i = 0; i < M; ++i) {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        sheep.push_back(getInterval(x1, y1, x2, y2));
    }

    const long double EPS = 1e-15L;
    const long double SHIFT = 1e-12L;

    sort(sheep.begin(), sheep.end(), [](const Interval& a, const Interval& b) {
        if (fabsl(a.l - b.l) > 1e-18L) return a.l < b.l;
        return a.r < b.r;
    });

    vector<Interval> forbid;
    for (auto s : sheep) {
        if (forbid.empty() || s.l > forbid.back().r + EPS) {
            forbid.push_back(s);
        } else {
            forbid.back().r = max(forbid.back().r, s.r);
        }
    }

    vector<long double> starts;
    for (auto f : forbid) starts.push_back(f.l);

    auto insideForbidden = [&](long double x) -> int {
        int id = int(upper_bound(starts.begin(), starts.end(), x + EPS) - starts.begin()) - 1;
        if (id >= 0 && x >= forbid[id].l - EPS && x <= forbid[id].r + EPS)
            return id;
        return -1;
    };

    auto chooseRightmostAllowed = [&](long double l, long double r, long double& shot) -> bool {
        int id = insideForbidden(r);

        if (id == -1) {
            shot = r;
            return shot >= l - EPS;
        }

        long double boundary = forbid[id].l;
        if (boundary <= l + EPS) return false;

        shot = boundary - SHIFT;

        if (shot < l - EPS) return false;

        int check = insideForbidden(shot);
        if (check != -1) return false;

        return true;
    };

    sort(wolves.begin(), wolves.end(), [](const Interval& a, const Interval& b) {
        if (fabsl(a.r - b.r) > 1e-18L) return a.r < b.r;
        return a.l < b.l;
    });

    int answer = 0;
    bool hasShot = false;
    long double lastShot = 0;

    for (auto w : wolves) {
        if (hasShot && lastShot >= w.l - EPS && lastShot <= w.r + EPS) {
            continue;
        }

        long double shot;
        if (!chooseRightmostAllowed(w.l, w.r, shot)) {
            cout << "No solution\n";
            return 0;
        }

        lastShot = shot;
        hasShot = true;
        ++answer;
    }

    cout << answer << '\n';
    return 0;
}
