#include <bits/stdc++.h>
#include <boost/multiprecision/cpp_int.hpp>

using namespace std;
using boost::multiprecision::cpp_int;

long long N, M, W, H, K;
long long R[3], C[3];

vector<pair<int,int>> blocks = {
    {0,0}, {0,1},
    {1,0}, {1,1}, {1,2},
    {2,1}, {2,2}
};

cpp_int fact[45];
cpp_int fallR[3][45], fallC[3][45];

int usedR[3], usedC[3];
int xval[7];

cpp_int answer = 0;

void dfs(int idx, int rem) {
    if (idx == (int)blocks.size()) {
        if (rem != 0) return;

        cpp_int term = 1;

        for (int i = 0; i < 3; ++i) {
            term *= fallR[i][usedR[i]];
            term *= fallC[i][usedC[i]];
        }

        for (int i = 0; i < 7; ++i) {
            term /= fact[xval[i]];
        }

        answer += term;
        return;
    }

    int r = blocks[idx].first;
    int c = blocks[idx].second;

    int lim = rem;
    lim = min<long long>(lim, R[r] - usedR[r]);
    lim = min<long long>(lim, C[c] - usedC[c]);

    for (int x = 0; x <= lim; ++x) {
        xval[idx] = x;
        usedR[r] += x;
        usedC[c] += x;

        dfs(idx + 1, rem - x);

        usedR[r] -= x;
        usedC[c] -= x;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> N >> M >> W >> H >> K)) return 0;

    if (K > 40) {
        cout << 0 << '\n';
        return 0;
    }

    long long rowOverlap = max(0LL, min(N, H + M) - max(0LL, H));
    long long colOverlap = max(0LL, min(N, W + M) - max(0LL, W));

    R[0] = N - rowOverlap;
    R[1] = rowOverlap;
    R[2] = M - rowOverlap;

    C[0] = N - colOverlap;
    C[1] = colOverlap;
    C[2] = M - colOverlap;

    long long totalRows = R[0] + R[1] + R[2];
    long long totalCols = C[0] + C[1] + C[2];

    if (K > min(totalRows, totalCols)) {
        cout << 0 << '\n';
        return 0;
    }

    fact[0] = 1;
    for (int i = 1; i <= 40; ++i) fact[i] = fact[i - 1] * i;

    for (int g = 0; g < 3; ++g) {
        fallR[g][0] = 1;
        fallC[g][0] = 1;

        for (int i = 1; i <= 40; ++i) {
            if (i <= R[g]) fallR[g][i] = fallR[g][i - 1] * (R[g] - i + 1);
            else fallR[g][i] = 0;

            if (i <= C[g]) fallC[g][i] = fallC[g][i - 1] * (C[g] - i + 1);
            else fallC[g][i] = 0;
        }
    }

    dfs(0, (int)K);

    cout << answer << '\n';

    return 0;
}
