#include <bits/stdc++.h>
using namespace std;

int pw3[10];
bool legalState[20000];
int memo[20000];

vector<array<int, 3>> lines = {
    array<int,3>{0,1,2}, {3,4,5}, {6,7,8},
    {0,3,6}, {1,4,7}, {2,5,8},
    {0,4,8}, {2,4,6}
};

void decode(int code, int a[9]) {
    for (int i = 0; i < 9; ++i) {
        a[i] = code % 3;
        code /= 3;
    }
}

bool wins(int code, int player) {
    int a[9];
    decode(code, a);
    for (auto &ln : lines) {
        if (a[ln[0]] == player && a[ln[1]] == player && a[ln[2]] == player)
            return true;
    }
    return false;
}

bool full(int code) {
    int a[9];
    decode(code, a);
    for (int i = 0; i < 9; ++i) {
        if (a[i] == 0) return false;
    }
    return true;
}

pair<int,int> counts(int code) {
    int a[9];
    decode(code, a);
    int x = 0, o = 0;
    for (int i = 0; i < 9; ++i) {
        if (a[i] == 1) ++x;
        else if (a[i] == 2) ++o;
    }
    return {x, o};
}

void generateLegal(int code) {
    if (legalState[code]) return;
    legalState[code] = true;

    if (wins(code, 1) || wins(code, 2) || full(code)) return;

    auto [x, o] = counts(code);
    int player = (x == o ? 1 : 2);

    int a[9];
    decode(code, a);

    for (int i = 0; i < 9; ++i) {
        if (a[i] == 0) {
            generateLegal(code + player * pw3[i]);
        }
    }
}

int solve(int code) {
    int &res = memo[code];
    if (res != 2) return res;

    if (wins(code, 1)) return res = 1;
    if (wins(code, 2)) return res = -1;
    if (full(code)) return res = 0;

    auto [x, o] = counts(code);
    int player = (x == o ? 1 : 2);

    int a[9];
    decode(code, a);

    if (player == 1) {
        res = -2;
        for (int i = 0; i < 9; ++i) {
            if (a[i] == 0) {
                res = max(res, solve(code + pw3[i]));
                if (res == 1) break;
            }
        }
    } else {
        res = 2;
        for (int i = 0; i < 9; ++i) {
            if (a[i] == 0) {
                res = min(res, solve(code + 2 * pw3[i]));
                if (res == -1) break;
            }
        }
    }

    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    pw3[0] = 1;
    for (int i = 1; i <= 9; ++i) pw3[i] = pw3[i - 1] * 3;

    generateLegal(0);
    fill(memo, memo + 20000, 2);

    string row;
    while (getline(cin, row)) {
        if (row == "Qc") break;

        vector<string> board(3);
        board[0] = row;
        getline(cin, board[1]);
        getline(cin, board[2]);

        string emptyLine;
        getline(cin, emptyLine);

        int code = 0;
        for (int i = 0; i < 9; ++i) {
            char c = board[i / 3][i % 3];
            if (c == 'X') code += pw3[i];
            else if (c == '0') code += 2 * pw3[i];
        }

        if (!legalState[code]) {
            cout << "Illegal position.\n";
        } else {
            int result = solve(code);
            if (result == 1) cout << "X wins.\n";
            else if (result == -1) cout << "0 wins.\n";
            else cout << "Game is a draw.\n";
        }
    }

    return 0;
}
