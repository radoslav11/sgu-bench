#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string s;
    cin >> s;

    int n = (int)s.size();
    vector<int> digit(n);
    for (int i = 0; i < n; ++i) digit[i] = s[i] - '0';

    int total = 1 << n;

    vector<array<unsigned char, 17>> dp(total);
    for (int i = 0; i < total; ++i) dp[i].fill(0);

    dp[0][0] = 1;

    for (int mask = 0; mask < total; ++mask) {
        int len = __builtin_popcount((unsigned)mask);

        for (int rem = 0; rem < 17; ++rem) {
            if (!dp[mask][rem]) continue;

            for (int i = 0; i < n; ++i) {
                if (mask & (1 << i)) continue;
                if (len == 0 && digit[i] == 0) continue;

                int nmask = mask | (1 << i);
                int nrem = (rem * 10 + digit[i]) % 17;
                dp[nmask][nrem] = 1;
            }
        }
    }

    int full = total - 1;

    if (!dp[full][0]) {
        cout << -1 << '\n';
        return 0;
    }

    string ans;
    int mask = full;
    int rem = 0;

    while (mask) {
        bool found = false;

        for (int i = 0; i < n && !found; ++i) {
            if (!(mask & (1 << i))) continue;

            int pmask = mask ^ (1 << i);

            for (int prem = 0; prem < 17; ++prem) {
                if (!dp[pmask][prem]) continue;

                if ((prem * 10 + digit[i]) % 17 == rem) {
                    ans.push_back(char('0' + digit[i]));
                    mask = pmask;
                    rem = prem;
                    found = true;
                    break;
                }
            }
        }
    }

    reverse(ans.begin(), ans.end());
    cout << ans << '\n';

    return 0;
}
