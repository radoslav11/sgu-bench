#include <bits/stdc++.h>
using namespace std;

struct Point {
    double x, y;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<Point> m(N + 1), v(N + 1), w(N + 1);
    for (int i = 1; i <= N; i++) {
        cin >> m[i].x >> m[i].y;
    }

    w[1] = {0.0, 0.0};
    for (int i = 1; i <= N - 1; i++) {
        w[i + 1].x = 2.0 * m[i].x - w[i].x;
        w[i + 1].y = 2.0 * m[i].y - w[i].y;
    }

    const double EPS = 1e-7;

    if (N % 2 == 0) {
        double needX = 2.0 * m[N].x;
        double needY = 2.0 * m[N].y;

        if (fabs(w[N].x - needX) > EPS || fabs(w[N].y - needY) > EPS) {
            cout << "NO\n";
            return 0;
        }

        v = w;
    } else {
        Point first;
        first.x = (2.0 * m[N].x - w[N].x) / 2.0;
        first.y = (2.0 * m[N].y - w[N].y) / 2.0;

        for (int i = 1; i <= N; i++) {
            if (i % 2 == 1) {
                v[i].x = w[i].x + first.x;
                v[i].y = w[i].y + first.y;
            } else {
                v[i].x = w[i].x - first.x;
                v[i].y = w[i].y - first.y;
            }
        }
    }

    cout << "YES\n";
    cout << fixed << setprecision(3);
    for (int i = 1; i <= N; i++) {
        cout << v[i].x << ' ' << v[i].y << '\n';
    }

    return 0;
}
