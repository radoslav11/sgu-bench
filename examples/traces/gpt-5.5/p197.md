#include <bits/stdc++.h>
using namespace std;

using Matrix = vector<vector<int>>;

int MOD, SZ;

Matrix multiply(const Matrix& A, const Matrix& B) {
    Matrix C(SZ, vector<int>(SZ, 0));
    for (int i = 0; i < SZ; ++i) {
        for (int k = 0; k < SZ; ++k) {
            if (A[i][k] == 0) continue;
            for (int j = 0; j < SZ; ++j) {
                C[i][j] = (C[i][j] + A[i][k] * B[k][j]) % MOD;
            }
        }
    }
    return C;
}

bool isZero(const string& s) {
    for (char c : s) if (c != '0') return false;
    return true;
}

string subtractOne(string s) {
    int i = (int)s.size() - 1;
    while (i >= 0 && s[i] == '0') {
        s[i] = '9';
        --i;
    }
    if (i >= 0) --s[i];

    int pos = 0;
    while (pos + 1 < (int)s.size() && s[pos] == '0') ++pos;
    return s.substr(pos);
}

vector<int> decimalToBinary(string s) {
    vector<int> bits;
    while (!isZero(s)) {
        int carry = 0;
        string nxt;
        for (char c : s) {
            int cur = carry * 10 + (c - '0');
            int q = cur / 2;
            carry = cur % 2;
            if (!nxt.empty() || q != 0) nxt.push_back(char('0' + q));
        }
        bits.push_back(carry);
        s = nxt.empty() ? "0" : nxt;
    }
    return bits;
}

bool compatible(int a, int b, int m) {
    for (int j = 0; j + 1 < m; ++j) {
        int x1 = (a >> j) & 1;
        int x2 = (a >> (j + 1)) & 1;
        int y1 = (b >> j) & 1;
        int y2 = (b >> (j + 1)) & 1;

        if (x1 == x2 && x1 == y1 && x1 == y2)
            return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string N;
    int M;
    cin >> N >> M >> MOD;

    SZ = 1 << M;

    if (MOD == 1) {
        cout << 0 << '\n';
        return 0;
    }

    Matrix T(SZ, vector<int>(SZ, 0));
    for (int a = 0; a < SZ; ++a) {
        for (int b = 0; b < SZ; ++b) {
            if (compatible(a, b, M)) {
                T[a][b] = 1 % MOD;
            }
        }
    }

    string expStr = subtractOne(N);
    vector<int> bits = decimalToBinary(expStr);

    Matrix result(SZ, vector<int>(SZ, 0));
    for (int i = 0; i < SZ; ++i) result[i][i] = 1 % MOD;

    Matrix base = T;

    for (int bit : bits) {
        if (bit) result = multiply(result, base);
        base = multiply(base, base);
    }

    int answer = 0;
    for (int i = 0; i < SZ; ++i) {
        for (int j = 0; j < SZ; ++j) {
            answer = (answer + result[i][j]) % MOD;
        }
    }

    cout << answer << '\n';
    return 0;
}
