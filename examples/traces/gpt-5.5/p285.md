#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<double> p(13);
    for (int i = 0; i < 12; ++i) cin >> p[i];

    double sumInternet = 0.0;
    for (int i = 0; i < N; ++i) {
        double x;
        cin >> x;
        sumInternet += x;
    }
    p[12] = sumInternet / N;

    const int S = 13;
    const int FULL = (1 << S) - 1;

    vector<array<array<double, 6>, 6>> dp(1 << S);
    for (int m = 0; m <= FULL; ++m)
        for (int i = 0; i < 6; ++i)
            for (int j = 0; j < 6; ++j)
                dp[m][i][j] = 0.0;

    dp[FULL][0][0] = 1.0;

    vector<double> ans(12, 0.0);

    for (int mask = FULL; mask >= 1; --mask) {
        for (int club = 0; club < 6; ++club) {
            for (int viewers = 0; viewers < 6; ++viewers) {
                double cur = dp[mask][club][viewers];
                if (cur == 0.0) continue;

                for (int sec = 0; sec < S; ++sec) {
                    if (!(mask & (1 << sec))) continue;

                    int prev = (sec + S - 1) % S;
                    while (!(mask & (1 << prev))) {
                        prev = (prev + S - 1) % S;
                    }

                    int dist = (sec - prev + S) % S;
                    if (dist == 0) dist = S;

                    double chooseProb = dist / 13.0;
                    int newMask = mask ^ (1 << sec);

                    double correct = p[sec];

                    int nc = club + 1;
                    if (nc == 6) {
                        ans[viewers] += cur * chooseProb * correct;
                    } else {
                        dp[newMask][nc][viewers] += cur * chooseProb * correct;
                    }

                    int nv = viewers + 1;
                    if (nv == 6) {
                        ans[6 + (5 - club)] += cur * chooseProb * (1.0 - correct);
                    } else {
                        dp[newMask][club][nv] += cur * chooseProb * (1.0 - correct);
                    }
                }
            }
        }
    }

    cout << fixed << setprecision(3);
    for (int i = 0; i < 12; ++i) {
        cout << ans[i] << '\n';
    }

    return 0;
}
