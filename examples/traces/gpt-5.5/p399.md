```cpp
#include <bits/stdc++.h>
using namespace std;

const int INF = 1e9;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    vector<string> g(N);
    for (int i = 0; i < N; i++) cin >> g[i];

    vector<vector<int>> dotId(N, vector<int>(M, -1));
    vector<pair<int,int>> dots;

    for (int i = 0; i < N; i++) {
        for (int j = 0; j < M; j++) {
            if (g[i][j] == '.') {
                dotId[i][j] = (int)dots.size();
                dots.push_back({i, j});
            }
        }
    }

    int V = dots.size();

    vector<vector<int>> craterId(N, vector<int>(M, -1));
    vector<vector<int>> craterAdj;
    int K = 0;

    int dx[4] = {-1, 1, 0, 0};
    int dy[4] = {0, 0, -1, 1};

    for (int i = 0; i < N; i++) {
        for (int j = 0; j < M; j++) {
            if (g[i][j] == 'X' && craterId[i][j] == -1) {
                queue<pair<int,int>> q;
                q.push({i, j});
                craterId[i][j] = K;

                set<int> adj;

                while (!q.empty()) {
                    auto [x, y] = q.front();
                    q.pop();

                    for (int d = 0; d < 4; d++) {
                        int nx = x + dx[d];
                        int ny = y + dy[d];

                        if (nx < 0 || nx >= N || ny < 0 || ny >= M) continue;

                        if (g[nx][ny] == 'X' && craterId[nx][ny] == -1) {
                            craterId[nx][ny] = K;
                            q.push({nx, ny});
                        } else if (g[nx][ny] == '.') {
                            adj.insert(dotId[nx][ny]);
                        }
                    }
                }

                craterAdj.push_back(vector<int>(adj.begin(), adj.end()));
                K++;
            }
        }
    }

    vector<vector<int>> graph(V);
    for (int id = 0; id < V; id++) {
        auto [x, y] = dots[id];
        for (int d = 0; d < 4; d++) {
            int nx = x + dx[d];
            int ny = y + dy[d];
            if (nx >= 0 && nx < N && ny >= 0 && ny < M && dotId[nx][ny] != -1) {
                graph[id].push_back(dotId[nx][ny]);
            }
        }
    }

    vector<int> oceanSources;
    for (int id = 0; id < V; id++) {
        auto [x, y] = dots[id];
        if (x == 0 || x == N - 1 || y == 0 || y == M - 1) {
            oceanSources.push_back(id);
        }
    }

    int S = K + 1;
    vector<vector<int>> dist(S, vector<int>(V, INF));
    vector<vector<int>> parent(S, vector<int>(V, -1));

    auto bfs = [&](int sidx, const vector<int>& sources) {
        queue<int> q;
        vector<char> used(V, false);

        for (int s : sources) {
            if (!used[s]) {
                used[s] = true;
                dist[sidx][s] = 1;
                parent[sidx][s] = -1;
                q.push(s);
            }
        }

        while (!q.empty()) {
            int v = q.front();
            q.pop();

            for (int to : graph[v]) {
                if (dist[sidx][to] > dist[sidx][v] + 1) {
                    dist[sidx][to] = dist[sidx][v] + 1;
                    parent[sidx][to] = v;
                    q.push(to);
                }
            }
        }
    };

    bfs(0, oceanSources);
    for (int c = 0; c < K; c++) {
        bfs(c + 1, craterAdj[c]);
    }

    vector<int> bestToOcean(K, INF), bestOceanNode(K, -1);
    for (int c = 0; c < K; c++) {
        for (int v : craterAdj[c]) {
            if (dist[0][v] < bestToOcean[c]) {
                bestToOcean[c] = dist[0][v];
                bestOceanNode[c] = v;
            }
        }
    }

    int bestCost = INF;
    int bestType = -1;
    int bestA = -1, bestB = -1, bestMeet = -1;

    for (int a = 0; a < K; a++) {
        if (bestToOcean[a] >= INF) continue;

        for (int b = a + 1; b < K; b++) {
            if (bestToOcean[b] >= INF) continue;

            int independentCost = bestToOcean[a] + bestToOcean[b];
            if (independentCost < bestCost) {
                bestCost = independentCost;
                bestType = 0;
                bestA = a;
                bestB = b;
                bestMeet = -1;
            }

            for (int v = 0; v < V; v++) {
                if (dist[0][v] >= INF || dist[a + 1][v] >= INF || dist[b + 1][v] >= INF) continue;

                int cost = dist[0][v] + dist[a + 1][v] + dist[b + 1][v] - 2;
                if (cost < bestCost) {
                    bestCost = cost;
                    bestType = 1;
                    bestA = a;
                    bestB = b;
                    bestMeet = v;
                }
            }
        }
    }

    vector<string> ans = g;

    auto markPath = [&](int sidx, int v) {
        while (v != -1) {
            auto [x, y] = dots[v];
            ans[x][y] = '*';
            v = parent[sidx][v];
        }
    };

    if (bestType == 0) {
        markPath(0, bestOceanNode[bestA]);
        markPath(0, bestOceanNode[bestB]);
    } else {
        markPath(0, bestMeet);
        markPath(bestA + 1, bestMeet);
        markPath(bestB + 1, bestMeet);
    }

    for (int i = 0; i < N; i++) {
        cout << ans[i] << '\n';
    }

    return 0;
}
```
