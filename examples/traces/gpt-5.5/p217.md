#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long double R1, R2;
    cin >> R1 >> R2;

    long double a = min(R1, R2);
    long double b = max(R1, R2);

    const int N = 100000; // even
    const long double PI = acosl(-1.0L);
    long double l = 0.0L, r = PI / 2.0L;
    long double h = (r - l) / N;

    auto f = [&](long double t) -> long double {
        long double s = sinl(t);
        long double c = cosl(t);
        return a * a * c * c * sqrtl(b * b - a * a * s * s);
    };

    long double sum = f(l) + f(r);
    for (int i = 1; i < N; ++i) {
        long double x = l + h * i;
        sum += (i & 1 ? 4.0L : 2.0L) * f(x);
    }

    long double integral = sum * h / 3.0L;
    long double volume = 8.0L * integral;

    cout << fixed << setprecision(10) << (double)volume << '\n';

    return 0;
}
