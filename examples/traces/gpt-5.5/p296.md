#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string N;
    int K;
    cin >> N >> K;

    string res;
    res.reserve(N.size());

    for (char c : N) {
        while (K > 0 && !res.empty() && res.back() < c) {
            res.pop_back();
            --K;
        }
        res.push_back(c);
    }

    while (K > 0) {
        res.pop_back();
        --K;
    }

    cout << res << '\n';
    return 0;
}
