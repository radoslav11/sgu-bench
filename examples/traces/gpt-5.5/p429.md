#include <bits/stdc++.h>
using namespace std;

using int64 = long long;

int64 rankExcluding(int64 x, int64 skip) {
    // Rank of x among positive integers excluding skip. Assumes x != skip.
    return x - (x > skip ? 1 : 0);
}

int64 unrankExcluding(int64 r, int64 skip) {
    // r-th positive integer among positive integers excluding skip.
    return (r < skip ? r : r + 1);
}

bool lose3(int64 x, int64 m, int64 y) {
    return x == y && x != m;
}

bool lose4(int64 x, int64 b, int64 c, int64 y) {
    if (b == c) return x == y;

    if (x == c || y == b) return false;
    return rankExcluding(x, c) == rankExcluding(y, b);
}

int64 countLose4FirstUpTo(int64 X, int64 b, int64 c, int64 y) {
    // Count t <= X such that lose4(t, b, c, y).
    if (X <= 0) return 0;

    if (b == c) {
        return (y <= X ? 1 : 0);
    }

    if (y == b) return 0;

    int64 r = rankExcluding(y, b);
    int64 t = unrankExcluding(r, c);
    return (t <= X ? 1 : 0);
}

int64 countLose4LastUpTo(int64 X, int64 x, int64 b, int64 c) {
    // Count t <= X such that lose4(x, b, c, t).
    if (X <= 0) return 0;

    if (b == c) {
        return (x <= X ? 1 : 0);
    }

    if (x == c) return 0;

    int64 r = rankExcluding(x, c);
    int64 t = unrankExcluding(r, b);
    return (t <= X ? 1 : 0);
}

bool lose5(int64 a, int64 b, int64 c, int64 d, int64 e) {
    // A position is losing iff both boundary-removal moves go to winning
    // positions and the two end heaps have equal ranks among such allowed ends.

    bool rowAllowed = !lose4(a, b, c, d);
    bool colAllowed = !lose4(b, c, d, e);

    if (!rowAllowed || !colAllowed) return false;

    int64 rowRank = a - countLose4FirstUpTo(a, b, c, d);
    int64 colRank = e - countLose4LastUpTo(e, b, c, d);

    return rowRank == colRank;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<int64> a(n);
    for (int i = 0; i < n; ++i) cin >> a[i];

    bool losing = false;

    if (n == 1) {
        losing = false;
    } else if (n == 2) {
        losing = ((a[0] ^ a[1]) == 0);
    } else if (n == 3) {
        losing = lose3(a[0], a[1], a[2]);
    } else if (n == 4) {
        losing = lose4(a[0], a[1], a[2], a[3]);
    } else {
        losing = lose5(a[0], a[1], a[2], a[3], a[4]);
    }

    cout << (losing ? "SERGEY" : "FEDOR") << '\n';

    return 0;
}
