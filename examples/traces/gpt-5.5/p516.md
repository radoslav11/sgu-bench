#include <bits/stdc++.h>
using namespace std;

const int YEAR = 365 * 24 * 60;

int mdays[12] = {31,28,31,30,31,30,31,31,30,31,30,31};

int parseDateTime(const string& d, const string& t) {
    int mon = stoi(d.substr(5,2));
    int day = stoi(d.substr(8,2));
    int hh = stoi(t.substr(0,2));
    int mm = stoi(t.substr(3,2));

    int days = 0;
    for (int i = 0; i < mon - 1; ++i) days += mdays[i];
    days += day - 1;

    return days * 24 * 60 + hh * 60 + mm;
}

struct Scheduler {
    int L, B, nb;
    int limit;
    vector<int> owner;
    vector<int> lazy;
    vector<int> week;
    vector<char> nonwork;

    vector<unordered_map<long long,int>> hist;
    unordered_map<long long,int> total;
    long long bad = 0;

    Scheduler() {}

    Scheduler(const vector<int>& init, const vector<int>& w, const vector<char>& nw, int lim)
        : L((int)init.size()), limit(lim), owner(init), week(w), nonwork(nw) {
        B = 750;
        nb = (L + B - 1) / B;
        lazy.assign(nb, -1);
        hist.resize(nb);
        buildAll();
    }

    long long key(int g, int w) {
        return ( (long long)g << 6 ) | w;
    }

    void addTotal(long long k, int delta) {
        if (delta == 0) return;
        int before = 0;
        auto it = total.find(k);
        if (it != total.end()) before = it->second;

        bool wasBad = before > limit;
        int after = before + delta;
        bool isBad = after > limit;

        if (!wasBad && isBad) bad++;
        if (wasBad && !isBad) bad--;

        if (after == 0) {
            if (it != total.end()) total.erase(it);
        } else {
            total[k] = after;
        }
    }

    void buildBlock(int b) {
        hist[b].clear();
        int l = b * B;
        int r = min(L, l + B);
        for (int i = l; i < r; ++i) {
            if (nonwork[i]) {
                hist[b][key(owner[i], week[i])]++;
            }
        }
    }

    void buildAll() {
        for (int b = 0; b < nb; ++b) {
            buildBlock(b);
            for (auto &p : hist[b]) addTotal(p.first, p.second);
        }
    }

    void materialize(int b) {
        if (lazy[b] == -1) return;
        int l = b * B;
        int r = min(L, l + B);
        for (int i = l; i < r; ++i) owner[i] = lazy[b];
        lazy[b] = -1;
    }

    void assignFullBlock(int b, int g) {
        for (auto &p : hist[b]) addTotal(p.first, -p.second);
        hist[b].clear();

        int l = b * B;
        int r = min(L, l + B);

        unordered_map<long long,int> nh;
        for (int i = l; i < r; ++i) {
            if (nonwork[i]) {
                nh[key(g, week[i])]++;
            }
        }

        hist[b].swap(nh);
        for (auto &p : hist[b]) addTotal(p.first, p.second);

        lazy[b] = g;
    }

    void assignPartial(int b, int lq, int rq, int g) {
        materialize(b);

        for (auto &p : hist[b]) addTotal(p.first, -p.second);

        int l = b * B;
        int r = min(L, l + B);

        for (int i = max(l, lq); i < min(r, rq); ++i) {
            owner[i] = g;
        }

        buildBlock(b);
        for (auto &p : hist[b]) addTotal(p.first, p.second);
    }

    void assignRange(int l, int r, int g) {
        if (l >= r) return;

        int bl = l / B;
        int br = (r - 1) / B;

        if (bl == br) {
            assignPartial(bl, l, r, g);
            return;
        }

        assignPartial(bl, l, r, g);
        assignPartial(br, l, r, g);

        for (int b = bl + 1; b <= br - 1; ++b) {
            assignFullBlock(b, g);
        }
    }

    bool valid() const {
        return bad == 0;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M, T;
    cin >> N >> M >> T;

    string d1, tm1, d2, tm2;
    cin >> d1 >> tm1 >> d2 >> tm2;
    int T1 = parseDateTime(d1, tm1);
    int T2 = parseDateTime(d2, tm2);
    int L = T2 - T1;

    unordered_map<string,int> id;
    auto getId = [&](const string& s) -> int {
        auto it = id.find(s);
        if (it != id.end()) return it->second;
        int v = (int)id.size();
        id[s] = v;
        return v;
    };

    vector<int> D(N), G(N);
    for (int i = 0; i < N; ++i) {
        string ds, ts, name;
        cin >> ds >> ts >> name;
        D[i] = parseDateTime(ds, ts);
        G[i] = getId(name);
    }

    vector<int> init(L, 0);

    for (int i = 0; i < N; ++i) {
        int s = max(D[i], T1);
        int e = (i + 1 < N ? D[i + 1] : T2);
        e = min(e, T2);
        if (s < e) {
            for (int x = s; x < e; ++x) {
                init[x - T1] = G[i];
            }
        }
    }

    vector<int> week(L);
    vector<char> nonwork(L);

    for (int i = 0; i < L; ++i) {
        int t = T1 + i;
        int day = t / (24 * 60);
        int minute = t % (24 * 60);
        int dow = (day + 3) % 7; // Monday = 0, Jan 1 2009 was Thursday
        int hour = minute / 60;

        week[i] = (day + 3) / 7;
        bool working = (dow < 5 && hour >= 9 && hour < 18);
        nonwork[i] = !working;
    }

    Scheduler sched(init, week, nonwork, T * 60);

    long long validMinutes = 0;
    int cur = T1;

    for (int i = 0; i < M; ++i) {
        string ads, ats, bds, bts, eds, ets, name;
        cin >> ads >> ats >> bds >> bts >> eds >> ets >> name;

        int A = parseDateTime(ads, ats);
        int B = parseDateTime(bds, bts);
        int E = parseDateTime(eds, ets);
        int g = getId(name);

        if (A > cur) {
            int to = min(A, T2);
            if (cur < to && sched.valid()) validMinutes += to - cur;
            cur = to;
        }

        if (A >= T2) continue;

        int l = max(B, T1);
        int r = min(E, T2);
        if (l < r) {
            sched.assignRange(l - T1, r - T1, g);
        }
    }

    if (cur < T2 && sched.valid()) {
        validMinutes += T2 - cur;
    }

    cout.setf(ios::fixed);
    cout << setprecision(15) << (double)validMinutes / (double)(T2 - T1) << "\n";

    return 0;
}
