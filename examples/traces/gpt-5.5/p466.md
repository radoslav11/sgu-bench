```cpp
#include <bits/stdc++.h>
using namespace std;

struct Allocation {
    int start = -1;
    int len = 0;
    bool ok = false;
};

int N, Q;

int nxt(int x) {
    return x == N ? 1 : x + 1;
}

int prv(int x) {
    return x == 1 ? N : x - 1;
}

int advance_pos(int x, int d) {
    return (int)(((long long)x - 1 + d) % N) + 1;
}

int cluster_end(int head, int len) {
    return advance_pos(head, len - 1);
}

map<int, int> clusters;              // head -> size
map<int, int> end_to_head;           // end -> head
set<pair<int, int>> by_size;         // {size, head}

void add_cluster(int head, int len) {
    if (len <= 0) return;
    if (len == N) head = 1;
    clusters[head] = len;
    end_to_head[cluster_end(head, len)] = head;
    by_size.insert({len, head});
}

void remove_cluster(int head) {
    auto it = clusters.find(head);
    if (it == clusters.end()) return;
    int len = it->second;
    clusters.erase(it);
    end_to_head.erase(cluster_end(head, len));
    by_size.erase({len, head});
}

string interval_str(int l, int r) {
    if (l == r) return to_string(l);
    return to_string(l) + "-" + to_string(r);
}

string format_allocation(int start, int len) {
    int end = cluster_end(start, len);
    if (start <= end) {
        return interval_str(start, end);
    } else {
        return interval_str(1, end) + "," + interval_str(start, N);
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> N >> Q;

    string s;
    cin >> s;

    bool all_free = true;
    for (char c : s) {
        if (c == 'X') {
            all_free = false;
            break;
        }
    }

    if (all_free) {
        add_cluster(1, N);
    } else {
        for (int i = 1; i <= N; i++) {
            int p = (i == 1 ? N : i - 1);
            if (s[i - 1] == '.' && s[p - 1] == 'X') {
                int len = 0;
                int cur = i;
                while (s[cur - 1] == '.') {
                    len++;
                    cur++;
                    if (cur > N) cur = 1;
                }
                add_cluster(i, len);
            }
        }
    }

    vector<Allocation> alloc(Q + 1);

    for (int qi = 1; qi <= Q; qi++) {
        string type;
        int x;
        cin >> type >> x;

        if (type == "PARK") {
            int need = x;

            auto it = by_size.lower_bound({need, 0});
            if (it == by_size.end()) {
                cout << "NO ROOM\n";
                alloc[qi].ok = false;
                continue;
            }

            int len = it->first;
            int head = it->second;

            remove_cluster(head);

            int start = head;
            if (len > need) {
                int new_head = advance_pos(head, need);
                add_cluster(new_head, len - need);
            }

            alloc[qi] = {start, need, true};
            cout << format_allocation(start, need) << '\n';
        } else {
            int ref = x;
            if (!alloc[ref].ok) continue;

            int start = alloc[ref].start;
            int len = alloc[ref].len;
            int end = cluster_end(start, len);

            int total = len;
            int new_head = start;

            int after_head = nxt(end);
            bool has_after = clusters.count(after_head);

            int before_end = prv(start);
            bool has_before = end_to_head.count(before_end);
            int before_head = has_before ? end_to_head[before_end] : -1;

            if (has_after) {
                total += clusters[after_head];
                remove_cluster(after_head);
            }

            if (has_before && before_head != after_head) {
                total += clusters[before_head];
                new_head = before_head;
                remove_cluster(before_head);
            } else if (has_before && before_head == after_head) {
                new_head = before_head;
            }

            if (total == N) new_head = 1;
            add_cluster(new_head, total);
        }
    }

    return 0;
}
```
