#include <bits/stdc++.h>
using namespace std;

struct Node {
    int pushes, moves, id;
    bool operator<(const Node& other) const {
        if (pushes != other.pushes) return pushes > other.pushes;
        return moves > other.moves;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    vector<string> grid;
    string line;
    while (getline(cin, line)) {
        if (!line.empty() && line.back() == '\r') line.pop_back();
        grid.push_back(line);
    }

    if (grid.empty()) return 0;

    int H = (int)grid.size();
    int W = (int)grid[0].size();
    int total = H * W;

    vector<char> passable(total, 0);
    int startPlayer = -1, startBox = -1, targetBox = -1;

    for (int r = 0; r < H; r++) {
        for (int c = 0; c < W; c++) {
            char ch = grid[r][c];
            int id = r * W + c;

            if (ch != '#') passable[id] = 1;

            if (ch == '@') startPlayer = id;
            else if (ch == '.') targetBox = id;
            else if (ch != '#' && ch != ' ') {
                startBox = id;
            }
        }
    }

    const int dr[4] = {-1, 0, 1, 0};
    const int dc[4] = {0, 1, 0, -1};
    const char low[4] = {'u', 'r', 'd', 'l'};
    const char up[4] = {'U', 'R', 'D', 'L'};

    auto inside = [&](int r, int c) {
        return r >= 0 && r < H && c >= 0 && c < W;
    };

    auto better = [](int p1, int m1, int p2, int m2) {
        return p1 < p2 || (p1 == p2 && m1 < m2);
    };

    int stateCount = total * 4;
    int initialState = stateCount;

    const int INF = 1e9;

    vector<int> distPush(stateCount + 1, INF), distMove(stateCount + 1, INF);
    vector<int> parent(stateCount + 1, -1);
    vector<string> parentPath(stateCount + 1);

    priority_queue<Node> pq;

    distPush[initialState] = 0;
    distMove[initialState] = 0;
    pq.push({0, 0, initialState});

    vector<int> bfsDist(total);
    vector<int> bfsPrev(total);
    vector<char> bfsMove(total);

    auto buildPath = [&](int src, int dst) {
        string res;
        int cur = dst;
        while (cur != src) {
            res.push_back(bfsMove[cur]);
            cur = bfsPrev[cur];
        }
        reverse(res.begin(), res.end());
        return res;
    };

    int answerState = -1;

    while (!pq.empty()) {
        Node cur = pq.top();
        pq.pop();

        if (cur.pushes != distPush[cur.id] || cur.moves != distMove[cur.id]) continue;

        int box, player;

        if (cur.id == initialState) {
            box = startBox;
            player = startPlayer;
        } else {
            box = cur.id / 4;
            int side = cur.id % 4;
            int br = box / W, bc = box % W;
            player = (br - dr[side]) * W + (bc - dc[side]);
        }

        if (box == targetBox) {
            answerState = cur.id;
            break;
        }

        fill(bfsDist.begin(), bfsDist.end(), -1);
        queue<int> q;

        bfsDist[player] = 0;
        bfsPrev[player] = -1;
        q.push(player);

        while (!q.empty()) {
            int v = q.front();
            q.pop();

            int r = v / W, c = v % W;

            for (int d = 0; d < 4; d++) {
                int nr = r + dr[d], nc = c + dc[d];
                if (!inside(nr, nc)) continue;

                int ni = nr * W + nc;
                if (!passable[ni] || ni == box || bfsDist[ni] != -1) continue;

                bfsDist[ni] = bfsDist[v] + 1;
                bfsPrev[ni] = v;
                bfsMove[ni] = low[d];
                q.push(ni);
            }
        }

        int br = box / W, bc = box % W;

        for (int d = 0; d < 4; d++) {
            int pr = br - dr[d], pc = bc - dc[d];
            int nr = br + dr[d], nc = bc + dc[d];

            if (!inside(pr, pc) || !inside(nr, nc)) continue;

            int stand = pr * W + pc;
            int newBox = nr * W + nc;

            if (!passable[stand] || !passable[newBox]) continue;
            if (bfsDist[stand] == -1) continue;

            int nid = newBox * 4 + d;
            int np = cur.pushes + 1;
            int nm = cur.moves + bfsDist[stand] + 1;

            if (better(np, nm, distPush[nid], distMove[nid])) {
                distPush[nid] = np;
                distMove[nid] = nm;
                parent[nid] = cur.id;

                string seg = buildPath(player, stand);
                seg.push_back(up[d]);
                parentPath[nid] = seg;

                pq.push({np, nm, nid});
            }
        }
    }

    if (answerState == -1) {
        cout << "Impossible.\n";
        return 0;
    }

    vector<string> parts;
    int cur = answerState;

    while (cur != initialState) {
        parts.push_back(parentPath[cur]);
        cur = parent[cur];
    }

    reverse(parts.begin(), parts.end());

    string answer;
    for (const string& s : parts) answer += s;

    cout << answer << '\n';

    return 0;
}
