```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    cin >> N >> K;

    vector<vector<bool>> pattern(N, vector<bool>(N, false));

    int cnt = 0;
    for (int i = 0; i < N && cnt < K; ++i) {
        for (int j = 0; j < N && cnt < K; ++j) {
            pattern[i][j] = true;
            ++cnt;
        }
    }

    int M = N * N;

    for (int r = 0; r < M; ++r) {
        int blockRow = r / N;
        int innerRow = r % N;

        for (int c = 0; c < M; ++c) {
            int blockCol = c / N;
            int innerCol = c % N;

            int x = (innerRow - blockCol + N) % N;
            int y = (innerCol - blockRow + N) % N;

            cout << (pattern[x][y] ? '*' : '.');
        }
        cout << '\n';
    }

    return 0;
}
```
