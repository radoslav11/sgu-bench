#include <bits/stdc++.h>
using namespace std;

struct Segment {
    long long x1, y1, x2, y2;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<Segment> s(N);
    for (int i = 0; i < N; ++i) {
        cin >> s[i].x1 >> s[i].y1 >> s[i].x2 >> s[i].y2;
    }

    long long ans = 0;

    for (int i = 0; i < N; ++i) {
        long long dx1 = s[i].x2 - s[i].x1;
        long long dy1 = s[i].y2 - s[i].y1;

        for (int j = i + 1; j < N; ++j) {
            bool shareEndpoint =
                (s[i].x1 == s[j].x1 && s[i].y1 == s[j].y1) ||
                (s[i].x1 == s[j].x2 && s[i].y1 == s[j].y2) ||
                (s[i].x2 == s[j].x1 && s[i].y2 == s[j].y1) ||
                (s[i].x2 == s[j].x2 && s[i].y2 == s[j].y2);

            if (!shareEndpoint) continue;

            long long dx2 = s[j].x2 - s[j].x1;
            long long dy2 = s[j].y2 - s[j].y1;

            __int128 dot = (__int128)dx1 * dx2 + (__int128)dy1 * dy2;

            if (dot == 0) {
                ++ans;
            }
        }
    }

    cout << ans << '\n';

    return 0;
}
