#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<bool> isPrime(N + 1, true);
    if (N >= 0) isPrime[0] = false;
    if (N >= 1) isPrime[1] = false;

    for (int i = 2; 1LL * i * i <= N; ++i) {
        if (isPrime[i]) {
            for (long long j = 1LL * i * i; j <= N; j += i) {
                isPrime[(int)j] = false;
            }
        }
    }

    vector<int> ans;
    for (int p = 3; p + 2 <= N; ++p) {
        if (isPrime[p] && isPrime[p + 2]) {
            ans.push_back(p);
        }
    }

    cout << ans.size() << '\n';
    for (int p : ans) {
        cout << 2 << ' ' << p << '\n';
    }

    return 0;
}
