#include <bits/stdc++.h>
using namespace std;

using u128 = unsigned __int128;

static void print_u128(u128 x) {
    if (x == 0) {
        cout << 0 << '\n';
        return;
    }
    string s;
    while (x > 0) {
        int digit = x % 10;
        s.push_back(char('0' + digit));
        x /= 10;
    }
    reverse(s.begin(), s.end());
    cout << s << '\n';
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, k;
    cin >> n >> k;

    int M = 1 << n;
    int full = M - 1;

    vector<int> popcnt(M);
    for (int i = 0; i < M; ++i) popcnt[i] = __builtin_popcount((unsigned)i);

    unordered_map<int, vector<u128>> dp;
    dp.reserve(20000);

    dp[0] = vector<u128>(1, 1); // state: previous-previous row mask, previous row mask

    for (int row = 0; row < n; ++row) {
        unordered_map<int, vector<u128>> ndp;
        ndp.reserve(20000);

        int oldMaxDeg = min(k, row * n);
        int newMaxDeg = min(k, (row + 1) * n);

        for (auto &it : dp) {
            int key = it.first;
            int prevprev = key / M;
            int prev = key % M;
            const vector<u128> &ways = it.second;

            int forbidden = 0;

            forbidden |= ((prev << 2) & full);
            forbidden |= (prev >> 2);

            forbidden |= ((prevprev << 1) & full);
            forbidden |= (prevprev >> 1);

            int avail = full & ~forbidden;

            for (int mask = avail;; mask = (mask - 1) & avail) {
                int cnt = popcnt[mask];
                if (cnt <= k) {
                    int nkey = prev * M + mask;
                    auto jt = ndp.find(nkey);
                    if (jt == ndp.end()) {
                        jt = ndp.emplace(nkey, vector<u128>(newMaxDeg + 1, 0)).first;
                    }

                    vector<u128> &to = jt->second;
                    for (int used = 0; used < (int)ways.size(); ++used) {
                        if (used + cnt <= k) {
                            to[used + cnt] += ways[used];
                        }
                    }
                }

                if (mask == 0) break;
            }
        }

        dp.swap(ndp);
    }

    u128 answer = 0;
    for (auto &it : dp) {
        if (k < (int)it.second.size()) {
            answer += it.second[k];
        }
    }

    print_u128(answer);

    return 0;
}
