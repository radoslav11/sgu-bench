```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<int> T(N), L(N);
    for (int i = 0; i < N; ++i) cin >> T[i];
    for (int i = 0; i < N; ++i) cin >> L[i];

    vector<pair<int, int>> jobs;
    for (int i = 0; i < N; ++i) {
        jobs.push_back({L[i], T[i]});
    }

    sort(jobs.begin(), jobs.end(), greater<pair<int, int>>());

    int currentTime = 0;
    int answer = 0;

    for (auto [delivery, printTime] : jobs) {
        currentTime += printTime;
        answer = max(answer, currentTime + delivery);
    }

    cout << answer << '\n';

    return 0;
}
```
