#include <bits/stdc++.h>
using namespace std;

using ll = long long;

ll fact[20], pow10v[20];
int cnt[10];
string Ls, Rs;
int N;

int permCap(int rem) {
    ll res = fact[rem];
    for (int d = 0; d < 10; d++) res /= fact[cnt[d]];
    return (res >= 3 ? 3 : (int)res);
}

int fullPermCap(int n) {
    int sum = 0;
    for (int d = 1; d <= 9; d++) {
        if (cnt[d]) {
            cnt[d]--;
            sum += permCap(n - 1);
            cnt[d]++;
            if (sum >= 3) return 3;
        }
    }
    return sum;
}

int rangeDfs(int pos, bool tl, bool tr) {
    if (pos == N) return 1;
    if (!tl && !tr) return permCap(N - pos);

    int lo = tl ? Ls[pos] - '0' : 0;
    int hi = tr ? Rs[pos] - '0' : 9;

    int sum = 0;
    for (int d = lo; d <= hi; d++) {
        if (cnt[d] == 0) continue;
        cnt[d]--;
        sum += rangeDfs(pos + 1, tl && d == lo, tr && d == hi);
        cnt[d]++;
        if (sum >= 3) return 3;
    }
    return sum;
}

ll answer = 0;

void genCounts(int digit, int rem, int len, bool isFull) {
    if (digit == 9) {
        cnt[9] = rem;

        int ways;
        if (isFull) ways = fullPermCap(len);
        else ways = rangeDfs(0, true, true);

        if (ways == 2) answer += 2;
        return;
    }

    for (int x = 0; x <= rem; x++) {
        cnt[digit] = x;
        genCounts(digit + 1, rem - x, len, isFull);
    }
}

int digits(ll x) {
    return to_string(x).size();
}

string toFixedLen(ll x, int len) {
    string s = to_string(x);
    while ((int)s.size() < len) s = "0" + s;
    return s;
}

void processRange(ll l, ll r, int len) {
    if (l > r) return;

    N = len;
    Ls = toFixedLen(l, len);
    Rs = toFixedLen(r, len);

    memset(cnt, 0, sizeof(cnt));
    genCounts(0, len, len, false);
}

void processFull(int len) {
    memset(cnt, 0, sizeof(cnt));
    genCounts(0, len, len, true);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    fact[0] = 1;
    for (int i = 1; i <= 16; i++) fact[i] = fact[i - 1] * i;

    pow10v[0] = 1;
    for (int i = 1; i <= 16; i++) pow10v[i] = pow10v[i - 1] * 10;

    ll l, r;
    cin >> l >> r;

    int dl = digits(l), dr = digits(r);

    if (dl == dr) {
        processRange(l, r, dl);
    } else {
        processRange(l, pow10v[dl] - 1, dl);

        for (int len = dl + 1; len <= dr - 1; len++) {
            processFull(len);
        }

        processRange(pow10v[dr - 1], r, dr);
    }

    cout << answer << '\n';
    return 0;
}
