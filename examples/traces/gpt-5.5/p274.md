#include <bits/stdc++.h>
using namespace std;

bool isLetter(char c) {
    return ('a' <= c && c <= 'z') || ('A' <= c && c <= 'Z');
}

bool isSymbol(char c) {
    return isLetter(c) || ('0' <= c && c <= '9') || c == '_' || c == '-';
}

bool validPrefix(const string& s) {
    if (s.empty()) return false;

    bool hasSymbolInCurrentWord = false;

    for (char c : s) {
        if (c == '.') {
            if (!hasSymbolInCurrentWord) return false;
            hasSymbolInCurrentWord = false;
        } else {
            if (!isSymbol(c)) return false;
            hasSymbolInCurrentWord = true;
        }
    }

    return hasSymbolInCurrentWord;
}

bool validDomain(const string& s) {
    if (s.size() != 2 && s.size() != 3) return false;

    for (char c : s) {
        if (!isLetter(c)) return false;
    }

    return true;
}

bool validSuffix(const string& s) {
    size_t pos = s.rfind('.');
    if (pos == string::npos) return false;

    string pref = s.substr(0, pos);
    string dom = s.substr(pos + 1);

    return validPrefix(pref) && validDomain(dom);
}

bool validAddress(const string& s) {
    size_t at = s.find('@');
    if (at == string::npos) return false;
    if (s.find('@', at + 1) != string::npos) return false;

    string pref = s.substr(0, at);
    string suff = s.substr(at + 1);

    return validPrefix(pref) && validSuffix(suff);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    string line;
    getline(cin, line);

    while (N--) {
        getline(cin, line);
        cout << (validAddress(line) ? "YES" : "NO") << '\n';
    }

    return 0;
}
