#include <bits/stdc++.h>
using namespace std;

struct Edge {
    int u, v, w;
};

const int INF = 1e9;

pair<int, vector<int>> stoerWagner(vector<vector<int>> g, vector<int> vertices) {
    int n = (int)g.size();
    vector<vector<int>> groups(n);

    for (int i = 0; i < n; i++) {
        groups[i].push_back(vertices[i]);
    }

    int best = INF;
    vector<int> bestSide;

    while (n > 1) {
        vector<int> weight(n, 0);
        vector<bool> used(n, false);
        int prev = -1, sel = -1;

        for (int it = 0; it < n; it++) {
            sel = -1;
            for (int i = 0; i < n; i++) {
                if (!used[i] && (sel == -1 || weight[i] > weight[sel])) {
                    sel = i;
                }
            }

            if (it == n - 1) {
                if (weight[sel] < best) {
                    best = weight[sel];
                    bestSide = groups[sel];
                }

                if (prev == -1) break;

                for (int i = 0; i < n; i++) {
                    if (i != sel && i != prev) {
                        g[prev][i] += g[sel][i];
                        g[i][prev] = g[prev][i];
                    }
                }

                groups[prev].insert(groups[prev].end(), groups[sel].begin(), groups[sel].end());

                g.erase(g.begin() + sel);
                for (auto &row : g) {
                    row.erase(row.begin() + sel);
                }
                groups.erase(groups.begin() + sel);

                n--;
                break;
            }

            used[sel] = true;
            prev = sel;

            for (int i = 0; i < n; i++) {
                if (!used[i]) {
                    weight[i] += g[sel][i];
                }
            }
        }
    }

    return {best, bestSide};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    vector<Edge> edges(M);

    for (int i = 0; i < M; i++) {
        cin >> edges[i].u >> edges[i].v >> edges[i].w;
    }

    int answerCost = INF;
    vector<int> answerEdges;

    for (int captured = 1; captured <= N; captured++) {
        vector<int> verts;
        vector<int> id(N + 1, -1);

        for (int v = 1; v <= N; v++) {
            if (v == captured) continue;
            id[v] = (int)verts.size();
            verts.push_back(v);
        }

        int sz = (int)verts.size();
        vector<vector<int>> g(sz, vector<int>(sz, 0));

        for (const Edge &e : edges) {
            if (e.u == captured || e.v == captured) continue;

            int a = id[e.u];
            int b = id[e.v];

            g[a][b] += e.w;
            g[b][a] += e.w;
        }

        auto [cost, sideVertices] = stoerWagner(g, verts);

        if (cost < answerCost) {
            vector<bool> side(N + 1, false);
            for (int v : sideVertices) {
                side[v] = true;
            }

            vector<int> destroyed;
            int realCost = 0;

            for (int i = 0; i < M; i++) {
                int u = edges[i].u;
                int v = edges[i].v;

                if (u == captured || v == captured) continue;

                if (side[u] != side[v]) {
                    destroyed.push_back(i + 1);
                    realCost += edges[i].w;
                }
            }

            answerCost = realCost;
            answerEdges = destroyed;
        }
    }

    cout << answerCost << '\n';
    cout << answerEdges.size() << '\n';

    for (int i = 0; i < (int)answerEdges.size(); i++) {
        if (i) cout << ' ';
        cout << answerEdges[i];
    }
    cout << '\n';

    return 0;
}
