```cpp
#include <bits/stdc++.h>
using namespace std;

const double INF = 1e100;
const double EPS = 1e-9;

struct Vec {
    int x, y;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    vector<string> g(N + 1);
    for (int y = 1; y <= N; y++) {
        string s;
        cin >> s;
        g[y] = " " + s;
    }

    int px, py, bx, by;
    cin >> px >> py;
    cin >> bx >> by;

    auto id = [&](int x, int y) {
        return (y - 1) * M + (x - 1);
    };

    auto flow = [&](int x, int y) -> Vec {
        char c = g[y][x];
        if (c == '1') return {0, -1};
        if (c == '2') return {0, 1};
        if (c == '3') return {1, 0};
        if (c == '4') return {-1, 0};
        return {0, 0};
    };

    vector<Vec> moves = {{0, 0}, {0, -1}, {0, 1}, {1, 0}, {-1, 0}};

    auto velocities = [&](int x, int y) {
        vector<Vec> res;
        Vec f = flow(x, y);
        for (auto m : moves) {
            Vec v{f.x + m.x, f.y + m.y};
            res.push_back(v);
        }
        return res;
    };

    int Vcnt = N * M;
    vector<double> dist(Vcnt, INF);
    priority_queue<pair<double,int>, vector<pair<double,int>>, greater<pair<double,int>>> pq;

    dist[id(px, py)] = 0;
    pq.push({0, id(px, py)});

    double ans = INF;

    auto centerTime = [&](Vec v) -> double {
        if (v.x == 0 && v.y == 0) return INF;
        if (abs(v.x) == 2 || abs(v.y) == 2) return 0.5;
        return 1.0;
    };

    auto shoreTime = [&](int x, int y, Vec v) -> double {
        if (v.x == 0 && v.y == 0) return INF;
        double t = INF;
        if (v.x > 0) t = min(t, (M + 0.5 - x) / v.x);
        if (v.x < 0) t = min(t, (0.5 - x) / v.x);
        if (v.y > 0) t = min(t, (N + 0.5 - y) / v.y);
        if (v.y < 0) t = min(t, (0.5 - y) / v.y);
        return t;
    };

    while (!pq.empty()) {
        auto [d, cur] = pq.top();
        pq.pop();
        if (d > dist[cur] + EPS) continue;

        int x = cur % M + 1;
        int y = cur / M + 1;

        for (Vec v : velocities(x, y)) {
            if (v.x == 0 && v.y == 0) continue;

            double tc = centerTime(v);
            double ts = shoreTime(x, y, v);

            if (ts <= tc + EPS) {
                ans = min(ans, d + ts);
                continue;
            }

            int nx = x + int(round(v.x * tc));
            int ny = y + int(round(v.y * tc));

            if (1 <= nx && nx <= M && 1 <= ny && ny <= N) {
                int nid = id(nx, ny);
                if (dist[nid] > d + tc + EPS) {
                    dist[nid] = d + tc;
                    pq.push({dist[nid], nid});
                }
            }
        }
    }

    double maxDist = 0;
    for (double d : dist) {
        if (d < INF / 2) maxDist = max(maxDist, d);
    }

    auto checkMeetSegment = [&](double b0x, double b0y, Vec bd, double L, double startTime) {
        int minx = max(1, (int)floor(min(b0x, b0x + bd.x * L)) - 2);
        int maxx = min(M, (int)ceil (max(b0x, b0x + bd.x * L)) + 2);
        int miny = max(1, (int)floor(min(b0y, b0y + bd.y * L)) - 2);
        int maxy = min(N, (int)ceil (max(b0y, b0y + bd.y * L)) + 2);

        for (int y = miny; y <= maxy; y++) {
            for (int x = minx; x <= maxx; x++) {
                double dp = dist[id(x, y)];
                if (dp >= INF / 2) continue;

                for (Vec pv : velocities(x, y)) {
                    if (pv.x == 0 && pv.y == 0) {
                        if (fabs(x - b0x) < EPS && fabs(y - b0y) < EPS && dp <= startTime + EPS)
                            ans = min(ans, startTime);
                        continue;
                    }

                    double tmax = min(centerTime(pv), shoreTime(x, y, pv));

                    double rx = b0x - x;
                    double ry = b0y - y;

                    double cross = pv.x * bd.y - pv.y * bd.x;

                    if (fabs(cross) > EPS) {
                        double tau = (rx * bd.y - ry * bd.x) / cross;
                        double lam;
                        if (bd.x != 0) lam = (x + tau * pv.x - b0x) / bd.x;
                        else lam = (y + tau * pv.y - b0y) / bd.y;

                        if (tau > -EPS && tau < tmax + EPS && lam > -EPS && lam < L + EPS) {
                            double T = startTime + lam;
                            if (T - tau + EPS >= dp) ans = min(ans, T);
                        }
                    } else {
                        if (fabs(rx * pv.y - ry * pv.x) > EPS) continue;

                        double a, b;
                        if (bd.x != 0) {
                            a = x - b0x;
                            b = pv.x;
                        } else {
                            a = y - b0y;
                            b = pv.y;
                        }

                        double lo = 0, hi = tmax;

                        auto addLower = [&](double val) { lo = max(lo, val); };
                        auto addUpper = [&](double val) { hi = min(hi, val); };

                        if (fabs(b) < EPS) {
                            if (a < -EPS || a > L + EPS) continue;
                        } else if (b > 0) {
                            addLower((-a) / b);
                            addUpper((L - a) / b);
                        } else {
                            addLower((L - a) / b);
                            addUpper((-a) / b);
                        }

                        double c = startTime + a - dp;
                        double coef = b - 1.0;
                        if (fabs(coef) < EPS) {
                            if (c < -EPS) continue;
                        } else if (coef > 0) {
                            addLower(-c / coef);
                        } else {
                            addUpper(-c / coef);
                        }

                        if (lo <= hi + EPS) {
                            double tau = (b >= 0 ? lo : hi);
                            double lam = a + b * tau;
                            if (lam > -EPS && lam < L + EPS)
                                ans = min(ans, startTime + lam);
                        }
                    }
                }
            }
        }
    };

    vector<int> seen(Vcnt, -1);
    int cx = bx, cy = by;
    double bt = 0;
    int step = 0;
    double limit = maxDist + 2.0 * Vcnt + 10.0;

    while (bt <= limit + EPS) {
        int cid = id(cx, cy);
        if (dist[cid] <= bt + EPS) ans = min(ans, bt);

        Vec f = flow(cx, cy);

        if (f.x == 0 && f.y == 0) {
            if (dist[cid] < INF / 2) ans = min(ans, max(bt, dist[cid]));
            break;
        }

        double ts = shoreTime(cx, cy, f);

        if (ts <= 0.5 + EPS) {
            checkMeetSegment(cx, cy, f, ts, bt);
            break;
        } else {
            checkMeetSegment(cx, cy, f, 1.0, bt);
            int nx = cx + f.x;
            int ny = cy + f.y;
            bt += 1.0;
            cx = nx;
            cy = ny;
        }

        step++;
        int nid = id(cx, cy);
        if (seen[nid] != -1 && bt > maxDist + Vcnt + 5.0) break;
        if (seen[nid] == -1) seen[nid] = step;
    }

    if (ans >= INF / 2) {
        cout << "SOS\n";
    } else {
        ans = ceil((ans - 1e-9) * 100.0) / 100.0;
        cout.setf(ios::fixed);
        cout << setprecision(2) << ans << "\n";
    }

    return 0;
}
```
