#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<double> x(n);
    for (int i = 0; i < n; ++i) cin >> x[i];

    auto calc = [&](double d, double *bestA = nullptr) -> double {
        vector<double> v(n);
        for (int i = 0; i < n; ++i) {
            v[i] = x[i] - i * d;
        }

        sort(v.begin(), v.end());
        double a = v[n / 2];

        if (bestA) *bestA = a;

        double res = 0.0;
        for (int i = 0; i < n; ++i) {
            res += fabs(x[i] - (a + i * d));
        }

        return res;
    };

    double l = 0.0, r = 20000.0;

    for (int it = 0; it < 200; ++it) {
        double m1 = l + (r - l) / 3.0;
        double m2 = r - (r - l) / 3.0;

        if (calc(m1) < calc(m2)) {
            r = m2;
        } else {
            l = m1;
        }
    }

    double d = (l + r) / 2.0;
    double a;
    double ans = calc(d, &a);

    cout << fixed << setprecision(10);
    cout << ans << '\n';

    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << a + i * d;
    }
    cout << '\n';

    return 0;
}
