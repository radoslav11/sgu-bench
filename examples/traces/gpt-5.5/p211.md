#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    int M;
    cin >> M;

    vector<int> a(N + 10, 0);

    auto hasTripleAt = [&](int s) -> bool {
        return s >= 0 && s + 2 < (int)a.size() && a[s] == 2 && a[s + 1] == 2 && a[s + 2] == 2;
    };

    for (int q = 0; q < M; q++) {
        int p;
        cin >> p;

        map<int, int> ch;

        auto setDigit = [&](int pos, int val) {
            if (a[pos] != val) {
                a[pos] = val;
                ch[pos] = val;
            }
        };

        if (a[p] < 2) {
            setDigit(p, a[p] + 1);
        } else {
            int pos = p;
            while (a[pos] == 2) {
                setDigit(pos, 1);
                pos++;
            }
            setDigit(pos, a[pos] + 1);
        }

        bool fixed = true;
        while (fixed) {
            fixed = false;
            int L = max(0, p - 5);
            int R = min((int)a.size() - 3, p + 8);

            for (int s = L; s <= R; s++) {
                if (hasTripleAt(s)) {
                    setDigit(s, 0);
                    setDigit(s + 1, 1);
                    setDigit(s + 2, 1);
                    setDigit(s + 3, a[s + 3] + 1);
                    fixed = true;
                    break;
                }
            }
        }

        cout << ch.size();
        for (auto [pos, val] : ch) {
            cout << ' ' << pos << ' ' << val;
        }
        cout << '\n';
    }

    return 0;
}
