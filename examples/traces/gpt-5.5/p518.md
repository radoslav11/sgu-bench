#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<vector<int>> l(n, vector<int>(n));
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cin >> l[i][j];
        }
    }

    int k;
    cin >> k;

    vector<int> r(k);
    for (int i = 0; i < k; ++i) cin >> r[i];

    vector<bool> cur(n, false), nxt(n, false);
    cur[0] = true;

    for (int step = 0; step < k; ++step) {
        fill(nxt.begin(), nxt.end(), false);

        for (int v = 0; v < n; ++v) {
            if (!cur[v]) continue;

            for (int to = 0; to < n; ++to) {
                if (l[v][to] == r[step]) {
                    nxt[to] = true;
                }
            }
        }

        cur.swap(nxt);
    }

    vector<int> ans;
    for (int i = 0; i < n; ++i) {
        if (cur[i]) ans.push_back(i + 1);
    }

    if (ans.empty()) {
        cout << 0 << '\n';
    } else {
        cout << ans.size() << '\n';
        for (int i = 0; i < (int)ans.size(); ++i) {
            if (i) cout << ' ';
            cout << ans[i];
        }
        cout << '\n';
    }

    return 0;
}
