#include <bits/stdc++.h>
#include <boost/multiprecision/cpp_int.hpp>

using namespace std;
using boost::multiprecision::cpp_int;

int phi(int x) {
    int res = x;
    for (int p = 2; p * p <= x; ++p) {
        if (x % p == 0) {
            while (x % p == 0) x /= p;
            res -= res / p;
        }
    }
    if (x > 1) res -= res / x;
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    cpp_int ans = 0;

    for (int d = 1; d * d <= n; ++d) {
        if (n % d == 0) {
            int d1 = d;
            ans += cpp_int(phi(d1)) * (cpp_int(1) << (n / d1));

            int d2 = n / d;
            if (d2 != d1) {
                ans += cpp_int(phi(d2)) * (cpp_int(1) << (n / d2));
            }
        }
    }

    ans /= n;
    cout << ans << '\n';

    return 0;
}
