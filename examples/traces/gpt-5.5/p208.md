#include <bits/stdc++.h>
#include <boost/multiprecision/cpp_int.hpp>

using namespace std;
using boost::multiprecision::cpp_int;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    int S = N * M;

    vector<cpp_int> pow2(S + 1);
    pow2[0] = 1;
    for (int i = 1; i <= S; ++i) pow2[i] = pow2[i - 1] * 2;

    set<vector<int>> perms;

    auto id = [&](int i, int j) {
        return i * M + j;
    };

    auto add_perm = [&](int type, int a, int b) {
        vector<int> p(S);

        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < M; ++j) {
                int ni, nj;

                if (type == 0) {
                    ni = (i + a) % N;
                    nj = (j + b) % M;
                } else if (type == 1) {
                    ni = (a - i) % N;
                    nj = (b - j) % M;
                    if (ni < 0) ni += N;
                    if (nj < 0) nj += M;
                } else if (type == 2) {
                    ni = (j + a) % N;
                    nj = (i + b) % M;
                } else {
                    ni = (a - j) % N;
                    nj = (b - i) % M;
                    if (ni < 0) ni += N;
                    if (nj < 0) nj += M;
                }

                p[id(i, j)] = id(ni, nj);
            }
        }

        perms.insert(p);
    };

    for (int a = 0; a < N; ++a) {
        for (int b = 0; b < M; ++b) {
            add_perm(0, a, b);
            add_perm(1, a, b);
        }
    }

    if (N == M) {
        for (int a = 0; a < N; ++a) {
            for (int b = 0; b < M; ++b) {
                add_perm(2, a, b);
                add_perm(3, a, b);
            }
        }
    }

    cpp_int sum = 0;

    for (const auto& p : perms) {
        vector<char> used(S, false);
        int cycles = 0;

        for (int i = 0; i < S; ++i) {
            if (!used[i]) {
                ++cycles;
                int v = i;
                while (!used[v]) {
                    used[v] = true;
                    v = p[v];
                }
            }
        }

        sum += pow2[cycles];
    }

    cpp_int answer = sum / perms.size();
    cout << answer << '\n';

    return 0;
}
