#include <bits/stdc++.h>
using namespace std;

struct P {
    double x, y;
    P() {}
    P(double x, double y) : x(x), y(y) {}
    P operator + (const P& o) const { return P(x + o.x, y + o.y); }
    P operator - (const P& o) const { return P(x - o.x, y - o.y); }
    P operator * (double k) const { return P(x * k, y * k); }
};

double cross(P a, P b) {
    return a.x * b.y - a.y * b.x;
}

double area2(const vector<P>& v) {
    double s = 0;
    for (int i = 0; i < (int)v.size(); ++i)
        s += cross(v[i], v[(i + 1) % v.size()]);
    return fabs(s);
}

P between(P a, P b, double ka, double kb) {
    // Point X on AB such that AX : XB = ka : kb
    return (a * kb + b * ka) * (1.0 / (ka + kb));
}

void printPoly(const vector<P>& p) {
    cout << p.size();
    cout << fixed << setprecision(15);
    for (auto q : p) {
        cout << ' ' << q.x << ' ' << q.y;
    }
    cout << "\n";
}

vector<int> parseRatio(string s) {
    vector<int> k;
    string cur;
    for (char c : s) {
        if (c == ':') {
            k.push_back(stoi(cur));
            cur.clear();
        } else cur += c;
    }
    k.push_back(stoi(cur));
    return k;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<P> v(N);
    for (int i = 0; i < N; ++i) cin >> v[i].x >> v[i].y;

    string ratio;
    cin >> ratio;
    vector<int> k = parseRatio(ratio);

    if (N == 3) {
        double sum = k[0] + k[1] + k[2];

        // Area near side i equals barycentric coordinate of opposite vertex.
        double l0 = k[1] / sum; // vertex 0 is opposite side 1
        double l1 = k[2] / sum; // vertex 1 is opposite side 2
        double l2 = k[0] / sum; // vertex 2 is opposite side 0

        P p = v[0] * l0 + v[1] * l1 + v[2] * l2;

        printPoly({v[0], v[1], p});
        printPoly({v[1], v[2], p});
        printPoly({v[2], v[0], p});
        return 0;
    }

    double total2 = area2(v);
    double sumK = k[0] + k[1] + k[2] + k[3];
    const double EPS = 1e-8;

    // Try diagonal v0-v2.
    {
        double a012 = area2({v[0], v[1], v[2]});
        double need = total2 * (k[0] + k[1]) / sumK;

        if (fabs(a012 - need) < EPS) {
            P A = between(v[0], v[2], k[0], k[1]);
            P B = between(v[0], v[2], k[3], k[2]);

            printPoly({v[0], v[1], A});
            printPoly({v[1], v[2], A});
            printPoly({v[2], v[3], B});
            printPoly({v[3], v[0], B});
            return 0;
        }
    }

    // Try diagonal v1-v3.
    {
        double a123 = area2({v[1], v[2], v[3]});
        double need = total2 * (k[1] + k[2]) / sumK;

        if (fabs(a123 - need) < EPS) {
            P A = between(v[1], v[3], k[1], k[2]);
            P B = between(v[3], v[1], k[3], k[0]);

            printPoly({v[0], v[1], B});
            printPoly({v[1], v[2], A});
            printPoly({v[2], v[3], A});
            printPoly({v[3], v[0], B});
            return 0;
        }
    }

    cout << -1 << "\n";
    return 0;
}
