#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long N;
    cin >> N;

    for (long long k = 0; ; ++k) {
        __int128 maxPayable = (__int128)(k + 1);
        maxPayable <<= (k + 1);
        maxPayable -= 1;

        if (maxPayable >= N) {
            cout << k << '\n';
            return 0;
        }
    }
}
