```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K, P;
    cin >> N >> K >> P;

    vector<int> B(K + 1);
    for (int i = 1; i <= K; i++) cin >> B[i];

    vector<vector<pair<int, int>>> gums(K + 1); // {cost, index}

    for (int i = 1; i <= N; i++) {
        int A, C;
        cin >> A >> C;
        gums[C].push_back({A, i});
    }

    for (int i = 1; i <= K; i++) {
        sort(gums[i].begin(), gums[i].end());
    }

    const int INF = 1e9;

    vector<int> dp(N + 1, INF), ndp(N + 1, INF);
    dp[0] = 0;

    vector<vector<int>> choice(K + 1, vector<int>(N + 1, -1));

    int processed = 0;

    for (int g = 1; g <= K; g++) {
        int m = (int)gums[g].size();

        vector<int> cost(m + 1, 0);
        int sum = 0;
        for (int t = 1; t <= m; t++) {
            sum += gums[g][t - 1].first;
            cost[t] = B[g] + sum;
        }

        fill(ndp.begin(), ndp.end(), INF);

        for (int cur = 0; cur <= processed; cur++) {
            if (dp[cur] == INF) continue;

            for (int t = 0; t <= m; t++) {
                int nxt = cur + t;
                int val = dp[cur] + cost[t];

                if (val < ndp[nxt]) {
                    ndp[nxt] = val;
                    choice[g][nxt] = t;
                }
            }
        }

        processed += m;
        dp.swap(ndp);
    }

    int S = 0;
    for (int i = N; i >= 0; i--) {
        if (dp[i] <= P) {
            S = i;
            break;
        }
    }

    vector<int> answer;
    int cnt = S;

    for (int g = K; g >= 1; g--) {
        int t = choice[g][cnt];
        if (t < 0) t = 0;

        for (int i = 0; i < t; i++) {
            answer.push_back(gums[g][i].second);
        }

        cnt -= t;
    }

    cout << S << '\n';
    for (int i = 0; i < (int)answer.size(); i++) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';

    return 0;
}
```
