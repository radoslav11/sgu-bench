#include <bits/stdc++.h>
using namespace std;

struct Person {
    long long s, b;
    int id;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<Person> a(n);
    for (int i = 0; i < n; i++) {
        cin >> a[i].s >> a[i].b;
        a[i].id = i + 1;
    }

    sort(a.begin(), a.end(), [](const Person& x, const Person& y) {
        if (x.s != y.s) return x.s < y.s;
        return x.b > y.b;
    });

    vector<long long> tail;
    vector<int> tailPos;
    vector<int> parent(n, -1);

    for (int i = 0; i < n; i++) {
        long long val = a[i].b;
        int pos = lower_bound(tail.begin(), tail.end(), val) - tail.begin();

        if (pos > 0) {
            parent[i] = tailPos[pos - 1];
        }

        if (pos == (int)tail.size()) {
            tail.push_back(val);
            tailPos.push_back(i);
        } else {
            tail[pos] = val;
            tailPos[pos] = i;
        }
    }

    int len = tail.size();
    vector<int> answer;
    int cur = tailPos[len - 1];

    while (cur != -1) {
        answer.push_back(a[cur].id);
        cur = parent[cur];
    }

    reverse(answer.begin(), answer.end());

    cout << len << '\n';
    for (int id : answer) {
        cout << id << ' ';
    }
    cout << '\n';

    return 0;
}
