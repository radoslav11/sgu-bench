#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    vector<int> C(N + 1);
    for (int i = 1; i <= N; ++i) cin >> C[i];

    const int INF = 1'000'000'000;
    int R = M + 1;

    vector<vector<int>> dp(R, vector<int>(R, INF));

    int ans = INF;

    for (int i = 1; i <= N - 1; ++i) {
        int col = i % R;

        deque<int> dq;
        int hStart = max(1, i - M);

        for (int h = hStart; h <= i - 1; ++h) {
            int val = dp[col][h % R];
            while (!dq.empty() && dp[col][dq.back() % R] >= val)
                dq.pop_back();
            dq.push_back(h);
        }

        int maxJ = min(N, i + M);
        for (int j = i + 1; j <= maxJ; ++j) {
            int best = INF;

            int low = max(1, j - M);
            while (!dq.empty() && dq.front() < low)
                dq.pop_front();

            if (!dq.empty()) {
                best = min(best, dp[col][dq.front() % R] + C[j]);
            }

            if (j <= M) {
                best = min(best, C[i] + C[j]);
            }

            dp[j % R][i % R] = best;

            if (i >= N - M + 1) {
                ans = min(ans, best);
            }
        }

        fill(dp[col].begin(), dp[col].end(), INF);
    }

    cout << ans << '\n';
    return 0;
}
