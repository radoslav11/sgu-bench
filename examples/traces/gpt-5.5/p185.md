#include <bits/stdc++.h>
using namespace std;

const long long INF = (1LL << 60);
const int MAXN = 405;

int n, m;
int len[MAXN][MAXN];
signed char cap[MAXN][MAXN];
signed char flow_[MAXN][MAXN];

vector<long long> dijkstra(int s) {
    vector<long long> dist(n + 1, INF);
    vector<char> used(n + 1, 0);
    dist[s] = 0;

    for (int it = 1; it <= n; ++it) {
        int v = -1;
        for (int i = 1; i <= n; ++i) {
            if (!used[i] && (v == -1 || dist[i] < dist[v])) {
                v = i;
            }
        }

        if (v == -1 || dist[v] == INF) break;
        used[v] = 1;

        for (int to = 1; to <= n; ++to) {
            if (len[v][to] > 0 && dist[to] > dist[v] + len[v][to]) {
                dist[to] = dist[v] + len[v][to];
            }
        }
    }

    return dist;
}

bool augment() {
    vector<int> parent(n + 1, -1);
    queue<int> q;

    parent[1] = 0;
    q.push(1);

    while (!q.empty() && parent[n] == -1) {
        int v = q.front();
        q.pop();

        for (int to = 1; to <= n; ++to) {
            if (parent[to] == -1 && cap[v][to] - flow_[v][to] > 0) {
                parent[to] = v;
                q.push(to);
                if (to == n) break;
            }
        }
    }

    if (parent[n] == -1) return false;

    int v = n;
    while (v != 1) {
        int p = parent[v];
        flow_[p][v]++;
        flow_[v][p]--;
        v = p;
    }

    return true;
}

vector<int> extract_path() {
    vector<int> path;
    int v = 1;
    path.push_back(v);

    while (v != n) {
        int nxt = -1;
        for (int to = 1; to <= n; ++to) {
            if (flow_[v][to] > 0) {
                nxt = to;
                break;
            }
        }

        if (nxt == -1) break;

        flow_[v][nxt]--;
        v = nxt;
        path.push_back(v);
    }

    return path;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;

    for (int i = 0; i < m; ++i) {
        int x, y, l;
        cin >> x >> y >> l;
        len[x][y] = len[y][x] = l;
    }

    vector<long long> d1 = dijkstra(1);
    vector<long long> dn = dijkstra(n);

    long long shortest = d1[n];

    if (shortest == INF) {
        cout << "No solution\n";
        return 0;
    }

    for (int u = 1; u <= n; ++u) {
        for (int v = 1; v <= n; ++v) {
            if (len[u][v] > 0) {
                long long w = len[u][v];
                if (d1[u] + w == d1[v] && d1[u] + w + dn[v] == shortest) {
                    cap[u][v] = 1;
                }
            }
        }
    }

    int maxflow = 0;
    while (maxflow < 2 && augment()) {
        maxflow++;
    }

    if (maxflow < 2) {
        cout << "No solution\n";
        return 0;
    }

    vector<int> p1 = extract_path();
    vector<int> p2 = extract_path();

    if (p1.empty() || p2.empty() || p1.back() != n || p2.back() != n) {
        cout << "No solution\n";
        return 0;
    }

    for (int i = 0; i < (int)p1.size(); ++i) {
        if (i) cout << ' ';
        cout << p1[i];
    }
    cout << '\n';

    for (int i = 0; i < (int)p2.size(); ++i) {
        if (i) cout << ' ';
        cout << p2[i];
    }
    cout << '\n';

    return 0;
}
