Looking at this problem, I need to:

1. Read two sets of n points representing the same constellation in two photos
2. Find the rotation angle between them
3. Return the smallest angle in [0, π]

The key insight is that the constellations have the same shape but are rotated. I need to find the rotation angle.

Approach:
1. For each photo, compute relative vectors from one point to all others
2. Try to match points between the two photos by finding which point in photo 2 corresponds to which point in photo 1
3. Calculate the rotation angle by comparing corresponding vectors
4. Return the smallest angle in [0, π]

Since the ordering doesn't matter, I need to:
- Pick a reference point in photo 1 (say point 0)
- For each potential corresponding point in photo 2, calculate what rotation would align them
- Verify this rotation works for other points
- Track the smallest valid rotation angle in [0, π]

Let me think about this differently:
- For point i in photo 1 and point j in photo 2, if they correspond, then:
  - Vector from point 0 to point i in photo 1 should match (when rotated) the vector from point j_0 to point j in photo 2
  - Where j_0 is the correspondence of point 0 in photo 2

Strategy:
1. Fix point 0 in photo 1
2. Try each point in photo 2 as the correspondence for point 0
3. For this assumption, calculate rotation angle from vector (0→1) in photo 1 to vector (j_0→j_1) in photo 2
4. Verify this rotation works for all other point pairs
5. Keep track of the best (smallest in [0,π]) rotation angle

```cpp
#include <bits/stdc++.h>
using namespace std;

const double EPS = 1e-9;
const double PI = acos(-1.0);

struct Point {
    double x, y;
    Point() {}
    Point(double x, double y) : x(x), y(y) {}
    Point operator-(const Point& p) const {
        return Point(x - p.x, y - p.y);
    }
    double dot(const Point& p) const {
        return x * p.x + y * p.y;
    }
    double cross(const Point& p) const {
        return x * p.y - y * p.x;
    }
    double norm() const {
        return sqrt(x * x + y * y);
    }
};

double getAngle(Point v) {
    return atan2(v.y, v.x);
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    
    vector<Point> photo1(n), photo2(n);
    
    for(int i = 0; i < n; i++){
        cin >> photo1[i].x >> photo1[i].y;
    }
    
    for(int i = 0; i < n; i++){
        cin >> photo2[i].x >> photo2[i].y;
    }
    
    double bestAngle = PI;
    
    // Try each point in photo2 as correspondence for point 0 in photo1
    for(int j0 = 0; j0 < n; j0++){
        // Try each point in photo2 as correspondence for point 1 in photo1
        for(int j1 = 0; j1 < n; j1++){
            if(j0 == j1) continue;
            
            Point v1 = photo1[1] - photo1[0];
            Point v2 = photo2[j1] - photo2[j0];
            
            double angle1 = getAngle(v1);
            double angle2 = getAngle(v2);
            double rotation = angle2 - angle1;
            
            // Normalize to [0, 2π)
            while(rotation < 0) rotation += 2 * PI;
            while(rotation >= 2 * PI) rotation -= 2 * PI;
            
            // Convert to [0, π]
            if(rotation > PI) rotation = 2 * PI - rotation;
            
            // Verify this rotation works for all points
            bool valid = true;
            for(int i = 0; i < n && valid; i++){
                bool found = false;
                for(int j = 0; j < n; j++){
                    Point vi = photo1[i] - photo1[0];
                    Point vj = photo2[j] - photo2[j0];
                    
                    double a1 = getAngle(vi);
                    double a2 = getAngle(vj);
                    double rot = a2 - a1;
                    while(rot < 0) rot += 2 * PI;
                    while(rot >= 2 * PI) rot -= 2 * PI;
                    if(rot > PI) rot = 2 * PI - rot;
                    
                    if(abs(rot - rotation) < EPS && abs(vi.norm() - vj.norm()) < EPS){
                        found = true;
                        break;
                    }
                }
                if(!found) valid = false;
            }
            
            if(valid){
                bestAngle = min(bestAngle, rotation);
            }
        }
    }
    
    cout << fixed << setprecision(10) << bestAngle << endl;
    
    return 0;
}
```
