Looking at this problem, I need to simulate bacterial growth in a grid where:

1. There are `n` cultures of bacteria
2. Each second, cultures divide in order (1, 2, ..., n)
3. Each bacterium divides into 5 (itself + 4 directions)
4. A bacterium dies if:
   - It tries to go outside the grid
   - It tries to occupy a cell with a different culture
   - It's older than another bacterium trying to take the same cell
5. We need to count cells occupied by each culture after `t` seconds

Key observations:
- With t up to 10^9, we can't simulate each second naively
- However, bacteria growth is limited by the grid size (max 1000x1000)
- After some point, the state stabilizes or reaches a cycle
- We need to detect when the grid state stops changing

Let me implement a solution that simulates until the state stabilizes or we reach t seconds:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int q, c, n;
    long long t;
    cin >> q >> c >> n >> t;
    
    // Store current state: grid[x][y] = {culture_id, generation}
    map<pair<int,int>, pair<int,int>> grid;
    
    vector<pair<int,int>> initial(n);
    for(int i = 0; i < n; i++){
        cin >> initial[i].first >> initial[i].second;
        grid[initial[i]] = {i, 0};
    }
    
    int dx[] = {0, -1, 1, 0, 0};
    int dy[] = {0, 0, 0, -1, 1};
    
    long long iteration = 0;
    
    while(iteration < t){
        map<pair<int,int>, pair<int,int>> newGrid = grid;
        bool changed = false;
        
        // Each culture divides in order
        for(int culture = 0; culture < n; culture++){
            map<pair<int,int>, vector<pair<int,int>>> candidates;
            
            // Find all cells of current culture and their bacteria divide
            for(auto& [pos, info] : grid){
                if(info.first != culture) continue;
                
                int x = pos.first;
                int y = pos.second;
                int gen = info.second;
                
                // Try to place 5 bacteria
                for(int dir = 0; dir < 5; dir++){
                    int nx = x + dx[dir];
                    int ny = y + dy[dir];
                    
                    // Check bounds
                    if(nx < 0 || nx >= q || ny < 0 || ny >= c) continue;
                    
                    // Check if occupied by different culture
                    if(newGrid.count({nx, ny}) && newGrid[{nx, ny}].first != culture) continue;
                    
                    candidates[{nx, ny}].push_back({culture, gen + 1});
                }
            }
            
            // Process candidates
            for(auto& [pos, bacteria] : candidates){
                // Remove older bacteria of same culture
                int maxGen = -1;
                for(auto& b : bacteria){
                    maxGen = max(maxGen, b.second);
                }
                
                // Keep only max generation
                if(newGrid.count(pos)){
                    if(newGrid[pos].first == culture && newGrid[pos].second < maxGen){
                        newGrid[pos] = {culture, maxGen};
                        changed = true;
                    }
                } else {
                    newGrid[pos] = {culture, maxGen};
                    changed = true;
                }
            }
        }
        
        if(!changed) break;
        
        grid = newGrid;
        iteration++;
    }
    
    vector<int> count(n, 0);
    for(auto& [pos, info] : grid){
        count[info.first]++;
    }
    
    for(int i = 0; i < n; i++){
        cout << count[i] << "\n";
    }
    
    return 0;
}
```

Wait, I need to reconsider the logic. When a culture divides, I need to track which cells currently have that culture, and those bacteria divide. Let me refine:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int q, c, n;
    long long t;
    cin >> q >> c >> n >> t;
    
    map<pair<int,int>, pair<int,int>> grid; // {x,y} -> {culture, generation}
    
    for(int i = 0; i < n; i++){
        int x, y;
        cin >> x >> y;
        grid[{x, y}] = {i, 0};
    }
    
    int dx[] = {0, -1, 1, 0, 0};
    int dy[] = {0, 0, 0, -1, 1};
    
    for(long long iter = 0; iter < t; iter++){
        map<pair<int,int>, pair<int,int>> newGrid = grid;
        bool changed = false;
        
        for(int culture = 0; culture < n; culture++){
            map<pair<int,int>, int> newCells;
            
            for(auto& [pos, info] : grid){
                if(info.first != culture) continue;
                
                int x = pos.first, y = pos.second, gen = info.second;
                
                for(int d = 0; d < 5; d++){
                    int nx = x + dx[d];
                    int ny = y + dy[d];
                    
                    if(nx < 0 || nx >= q || ny < 0 || ny >= c) continue;
                    if(newGrid.count({nx, ny}) && newGrid[{nx, ny}].first != culture) continue;
                    
                    newCells[{nx, ny}] = max(newCells[{nx, ny}], gen + 1);
                }
            }
            
            for(auto& [pos, gen] : newCells){
                if(!newGrid.count(pos) || newGrid[pos].first == culture){
                    newGrid[pos] = {culture, gen};
                    changed = true;
                }
            }
        }
        
        if(!changed) break;
        grid = newGrid;
    }
    
    vector<int> count(n, 0);
    for(auto& [pos, info] : grid) count[info.first]++;
    
    for(int i = 0; i < n; i++) cout << count[i] << "\n";
    
    return 0;
}
```
