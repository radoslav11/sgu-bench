Looking at this problem, I need to find four non-negative integers P1, N1, P2, N2 such that:
1. P1*x1 - N1*x1 + P2*x2 - N2*x2 = P
2. P1 + N1 + P2 + N2 = K

Let me rearrange:
- Let a = P1 - N1 (net jumps of x1)
- Let b = P2 - N2 (net jumps of x2)
- Then: a*x1 + b*x2 = P
- And: P1 + N1 + P2 + N2 = K

From a = P1 - N1 and P1 + N1 = some value, I can derive:
- If we use j1 total jumps of x1: P1 + N1 = j1, so P1 = (j1 + a)/2, N1 = (j1 - a)/2
- If we use j2 total jumps of x2: P2 + N2 = j2, so P2 = (j2 + b)/2, N2 = (j2 - b)/2

We need: j1 + j2 = K

For P1, N1, P2, N2 to be non-negative integers:
- j1 ≥ |a| and (j1 + a) must be even
- j2 ≥ |b| and (j2 + b) must be even

Strategy:
1. Try all possible values of a such that a*x1 = P - b*x2 for some valid b
2. Since x1, x2 < 40000 and P < 40000, a and b won't be too large
3. For each valid (a, b) pair, check if we can find valid j1, j2 with j1 + j2 = K

Let me iterate through possible values of a and b:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    long long x1, x2, P, K;
    cin >> x1 >> x2 >> P >> K;
    
    // Try all reasonable values of a (P1 - N1)
    // a*x1 + b*x2 = P, so b = (P - a*x1) / x2
    
    long long limit = 100000; // reasonable search limit
    
    for(long long a = -limit; a <= limit; a++){
        long long remainder = P - a * x1;
        if(remainder % x2 != 0) continue;
        
        long long b = remainder / x2;
        
        // Now we need j1, j2 such that:
        // j1 + j2 = K
        // j1 >= |a|, (j1 + a) even
        // j2 >= |b|, (j2 + b) even
        
        long long min_j1 = abs(a);
        long long min_j2 = abs(b);
        
        if(min_j1 + min_j2 > K) continue;
        
        // We need (j1 + a) and (j2 + b) to be even
        // This means j1 and a must have same parity, j2 and b must have same parity
        
        long long j1 = min_j1;
        if((j1 + a) % 2 != 0) j1++;
        
        long long j2 = K - j1;
        
        if(j2 < min_j2) continue;
        if((j2 + b) % 2 != 0) {
            j1 += 2;
            j2 = K - j1;
        }
        
        if(j1 < min_j1 || j2 < min_j2) continue;
        if(j1 + j2 != K) continue;
        
        long long P1 = (j1 + a) / 2;
        long long N1 = (j1 - a) / 2;
        long long P2 = (j2 + b) / 2;
        long long N2 = (j2 - b) / 2;
        
        if(P1 >= 0 && N1 >= 0 && P2 >= 0 && N2 >= 0){
            cout << "YES\n";
            cout << P1 << " " << N1 << " " << P2 << " " << N2 << "\n";
            return 0;
        }
    }
    
    cout << "NO\n";
    return 0;
}
```

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    long long x1, x2, P, K;
    cin >> x1 >> x2 >> P >> K;
    
    // Try all reasonable values of a
    for(long long a = -80000; a <= 80000; a++){
        long long remainder = P - a * x1;
        if(remainder % x2 != 0) continue;
        
        long long b = remainder / x2;
        
        long long min_j1 = abs(a);
        long long min_j2 = abs(b);
        
        if(min_j1 + min_j2 > K) continue;
        
        long long j1 = min_j1;
        if((j1 + a) % 2 != 0) j1++;
        
        long long j2 = K - j1;
        
        if(j2 < min_j2 || (j2 + b) % 2 != 0){
            if(j2 >= min_j2 + 1 && (j2 - 1 + b) % 2 == 0){
                j1 += 2; j2 -= 2;
            } else continue;
        }
        
        if(j2 < 0) continue;
        
        long long P1 = (j1 + a) / 2;
        long long N1 = (j1 - a) / 2;
        long long P2 = (j2 + b) / 2;
        long long N2 = (j2 - b) / 2;
        
        if(P1 >= 0 && N1 >= 0 && P2 >= 0 && N2 >= 0){

}
```
