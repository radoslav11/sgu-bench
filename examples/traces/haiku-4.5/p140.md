Looking at this problem, I need to:

1. Find a sequence X of length N such that A₁*X₁ + A₂*X₂ + ... + Aₙ*Xₙ ≡ B (mod P)
2. All elements Xᵢ should be non-negative integers

This is a linear Diophantine equation modulo P. The key insight is:
- We need to find if there exists a solution to the congruence equation
- The equation has a solution if and only if gcd(A₁, A₂, ..., Aₙ, P) divides B

Let me think about the approach:
1. Calculate g = gcd(A₁, A₂, ..., Aₙ, P)
2. If B % g ≠ 0, then there's no solution
3. If there is a solution, we need to find one

For finding a solution, I can use a constructive approach:
- Start with all Xᵢ = 0
- Try to find values that satisfy the equation
- We can greedily assign values or use extended GCD properties

A simpler approach:
- Calculate g = gcd of all elements in A and P
- If g doesn't divide B, answer is NO
- Otherwise, we can try values: for each position, try different values of Xᵢ until we find a solution
- Since we're working modulo P, we only need to check values 0 to P-1 for each variable

Let me implement a solution that:
1. Checks if a solution exists using GCD
2. If yes, constructs a solution by trying values

#include<bits/stdc++.h>
using namespace std;

long long gcd(long long a, long long b) {
    if (b == 0) return a;
    return gcd(b, a % b);
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, P, B;
    cin >> N >> P >> B;
    
    vector<long long> A(N);
    for(int i = 0; i < N; i++) {
        cin >> A[i];
    }
    
    // Calculate gcd of all A[i] and P
    long long g = P;
    for(int i = 0; i < N; i++) {
        g = gcd(g, A[i] % P);
    }
    
    // Check if solution exists
    if(B % g != 0) {
        cout << "NO\n";
        return 0;
    }
    
    // Solution exists, try to find one
    vector<long long> X(N, 0);
    
    // Try to construct a solution greedily
    for(int i = 0; i < N; i++) {
        // Try values for X[i] from 0 to P-1
        bool found = false;
        for(long long val = 0; val < P; val++) {
            X[i] = val;
            
            // Check if we can complete the solution
            long long sum = 0;
            for(int j = 0; j <= i; j++) {
                sum = (sum + A[j] * X[j]) % P;
            }
            
            // For the last element, check if it matches B
            if(i == N - 1) {
                if(sum == B) {
                    found = true;
                    break;
                }
            } else {
                // For intermediate elements, just try this value
                found = true;
                break;
            }
        }
        
        if(i == N - 1 && !found) {
            cout << "NO\n";
            return 0;
        }
        
        if(!found && i < N - 1) {
            X[i] = 0;
        }
    }
    
    // Verify solution
    long long result = 0;
    for(int i = 0; i < N; i++) {
        result = (result + (A[i] % P) * (X[i] % P)) % P;
    }
    
    if(result == B) {
        cout << "YES\n";
        for(int i = 0; i < N; i++) {
            if(i > 0) cout << " ";
            cout << X[i];
        }
        cout << "\n";
    } else {
        cout << "NO\n";
    }
    
    return 0;
}
