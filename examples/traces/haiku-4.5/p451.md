Looking at this problem, I need to:

1. Parse a relationship chain from A to C
2. For each relation, track possible distances (min and max)
3. Some relations have multiple possibilities (e.g., uncle could be father's brother or mother's brother)
4. Calculate the degree of kinship (shortest path considering all possibilities)

Key insights:
- Distance increases by 1 for: father, mother, son, daughter
- Distance stays 0 for: husband, wife
- Relations with multiple paths need to track both min and max possibilities
- Brother/sister, uncle/aunt, etc. have alternative paths

Let me trace through the first example:
- A's father: distance 1
- father's brother: distance 1+1=2 (could be father's brother, distance 1 more)
- brother's son: distance 2+1=3 (nephew, distance 1 more)
- son's aunt: This is tricky - the nephew's aunt could be:
  - The nephew's father's sister (path through father side)
  - The nephew's mother's sister (path through mother side)

I'll use a BFS-like approach tracking all possible distances as I traverse the relationship chain.

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    string line;
    getline(cin, line);
    
    // Parse the input
    vector<string> relations;
    size_t pos = line.find("'s");
    pos += 2;
    
    while(pos < line.length()) {
        while(pos < line.length() && (line[pos] == ' ' || line[pos] == '\'')) pos++;
        if(pos >= line.length()) break;
        
        size_t end = pos;
        while(end < line.length() && isalpha(line[end])) end++;
        
        string rel = line.substr(pos, end - pos);
        if(rel != "is" && rel != "A" && rel != "C") {
            relations.push_back(rel);
        }
        pos = end;
    }
    
    // Start with A at distance 0
    set<int> distances;
    distances.insert(0);
    
    // Process each relation
    for(const string& rel : relations) {
        set<int> newDistances;
        
        for(int d : distances) {
            if(rel == "father" || rel == "mother" || rel == "son" || rel == "daughter") {
                newDistances.insert(d + 1);
            }
            else if(rel == "husband" || rel == "wife") {
                newDistances.insert(d);
            }
            else if(rel == "brother" || rel == "sister") {
                // brother/sister: go to parent then to sibling = +2, but could be +0 if same person
                newDistances.insert(d + 2);
            }
            else if(rel == "grandfather" || rel == "grandmother") {
                newDistances.insert(d + 2);
            }
            else if(rel == "grandson" || rel == "granddaughter") {
                newDistances.insert(d + 2);
            }
            else if(rel == "uncle" || rel == "aunt") {
                // uncle/aunt: parent's brother/sister = +1 (to parent) + option of +1 or +0
                newDistances.insert(d + 2);
            }
            else if(rel == "nephew" || rel == "niece") {
                // nephew/niece: sibling's child = +2
                newDistances.insert(d + 2);
            }
        }
        
        distances = newDistances;
    }
    
    int minDist = *distances.begin();
    int maxDist = *distances.rbegin();
    
    cout << maxDist << " " << minDist << endl;
    
    return 0;
}
```
