Looking at this problem, I need to find the maximum integer whose square is not greater than X. In other words, I need to find floor(√X).

The challenge here is that X can be extremely large (up to 10^1000), which is way beyond what standard integer types can handle. I need to use a big integer approach.

Key observations:
1. X can be up to 10^1000, so I need to handle very large numbers
2. I need to find the integer square root of X
3. I can use binary search on the result since if we're looking for the largest number Y where Y² ≤ X, we can binary search for Y

Since C++ doesn't have built-in big integer support in the standard library (for competitive programming, we might have access to __int128 or need to implement our own), let me think about the approach:

For very large numbers, I can use binary search where:
- Lower bound: 1
- Upper bound: X (or more reasonably, something like 10^500)
- For each candidate, I need to check if candidate² ≤ X

However, multiplying large numbers is tricky. A simpler approach: I can use the fact that the answer will have at most ⌈log₁₀(X)/2⌉ digits, which is at most 500 digits.

Let me implement a solution using string manipulation and binary search:

```cpp
#include<bits/stdc++.h>
using namespace std;

// Compare two number strings: returns 1 if a > b, -1 if a < b, 0 if equal
int compareStrings(string a, string b) {
    if (a.length() != b.length()) {
        return a.length() > b.length() ? 1 : -1;
    }
    return a.compare(b);
}

// Multiply two number strings
string multiply(string a, string b) {
    int n = a.length(), m = b.length();
    vector<int> result(n + m, 0);
    
    for (int i = n - 1; i >= 0; i--) {
        for (int j = m - 1; j >= 0; j--) {
            int mul = (a[i] - '0') * (b[j] - '0');
            int p1 = i + j, p2 = i + j + 1;
            int sum = mul + result[p2];
            
            result[p2] = sum % 10;
            result[p1] += sum / 10;
        }
    }
    
    string str = "";
    for (int i = 0; i < result.size(); i++) {
        if (!(str.empty() && result[i] == 0)) {
            str += to_string(result[i]);
        }
    }
    
    return str.empty() ? "0" : str;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    string X;
    cin >> X;
    
    string low = "1";
    string high = X;
    string answer = "1";
    
    // Binary search
    while (compareStrings(low, high) <= 0) {
        // Calculate mid = (low + high) / 2
        // For simplicity with large numbers, we'll use a simpler approach
        // Try candidate and check if candidate^2 <= X
        
        string mid = low; // placeholder, we'll do linear search for smaller range
        
        if (compareStrings(mid, high) > 0) break;
        
        string square = multiply(mid, mid);
        int cmp = compareStrings(square, X);
        
        if (cmp <= 0) {
            answer = mid;
            // Try larger number
            // Increment low
            int carry = 1;
            for (int i = low.length() - 1; i >= 0 && carry; i--) {
                int digit = (low[i] - '0') + carry;
                low[i] = (digit % 10) + '0';
                carry = digit / 10;
            }
            if (carry) low = "1" + low;
        } else {
            // Try smaller number - decrement high
            int borrow = 1;
            for (int i = high.length() - 1; i >= 0 && borrow; i--) {
                int digit = (high[i] - '0') - borrow;
                if (digit < 0) {
                    digit += 10;
                    borrow = 1;
                } else {
                    borrow = 0;
                }
                high[i] = digit + '0';
            }
            // Remove leading zeros
            while (high.length() > 1 && high[0] == '0') {
                high = high.substr(1);
            }
        }
    }
    
    cout << answer << endl;
    
    return 0;
}
```

Actually, the above approach is too slow. Let me use a better binary search with proper midpoint calculation:

```cpp
#include<bits/stdc++.h>
using namespace std;

string multiply(string a, string b) {
    int n = a.length(), m = b.length();
    vector<int> result(n + m, 0);
    
    for (int i = n - 1; i >= 0; i--) {
        for (int j = m - 1; j >= 0; j--) {
            int mul = (a[i] - '0') * (b[j] - '0');
            int p1 = i + j, p2 = i + j + 1;
            int sum = mul + result[p2];
            result[p2] = sum % 10;
            result[p1] += sum / 10;
        }
    }
    
    string str = "";
    for (int num : result) {
        if (!(str.empty() && num == 0)) str += char('0' + num);
    }
    return str.empty() ? "0" : str;
}

int main(){
    string X;
    cin >> X;
    
    string low = "1", high = X, ans = "1";
    
    while (low.length() < high.length() || (low.length() == high.length() && low <= high)) {
        string mid = low;
        string sq = multiply(mid, mid);
        
        if (sq.length() < X.length() || (sq.length() == X.length() && sq <= X)) {
            ans = mid;
            int i = mid.length() - 1, carry = 1;
            while (i >= 0 && carry) {
                int d = mid[i] - '0' + 1;
                mid[i] = '0' + d % 10;
                carry = d / 10;
                i--;
            }
            if (carry) mid = "1" + mid;
            low = mid;
        } else {
            break;
        }
    }
    
    cout << ans << endl;
    return 0;
}
```
