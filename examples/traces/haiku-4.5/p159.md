Looking at this problem, I need to find all n-digit self-replicating numbers in base b, where a number is self-replicating if its square's last n digits (in base b) equal the number itself.

Key insights:
1. An n-digit number x in base b satisfies: b^(n-1) ≤ x < b^n
2. x is self-replicating if x² ≡ x (mod b^n)
3. This means x(x-1) ≡ 0 (mod b^n)

Since we're looking for x such that x² ≡ x (mod b^n), I can use a lifting approach:
- Start with solutions modulo b, then lift them to modulo b², b³, etc.
- For each solution x_k modulo b^k, find solutions x_{k+1} modulo b^{k+1}

Algorithm:
1. Find all solutions x where 0 ≤ x < b and x² ≡ x (mod b)
2. For each solution at level k, lift it to level k+1 by checking x_k + j*b^k for j = 0 to b-1
3. Filter solutions that are n-digit numbers in base b

Let me implement this:

```cpp
#include<bits/stdc++.h>
using namespace std;

string toBase(long long num, int base) {
    if (num == 0) return "0";
    string result = "";
    while (num > 0) {
        int digit = num % base;
        if (digit < 10) {
            result = char('0' + digit) + result;
        } else {
            result = char('A' + digit - 10) + result;
        }
        num /= base;
    }
    return result;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int b, n;
    cin >> b >> n;
    
    vector<long long> solutions;
    
    // Find initial solutions modulo b
    vector<long long> current;
    for (long long x = 0; x < b; x++) {
        long long sq = (x * x) % b;
        if (sq == x % b) {
            current.push_back(x);
        }
    }
    
    // Lift solutions from modulo b^k to modulo b^(k+1)
    long long modulus = b;
    for (int k = 1; k < n; k++) {
        vector<long long> next;
        long long new_modulus = modulus * b;
        
        for (long long x : current) {
            // Try x + j * modulus for j = 0 to b-1
            for (int j = 0; j < b; j++) {
                long long candidate = x + (long long)j * modulus;
                long long sq = (candidate * candidate) % new_modulus;
                if (sq == candidate % new_modulus) {
                    next.push_back(candidate);
                }
            }
        }
        
        current = next;
        modulus = new_modulus;
    }
    
    // Filter for n-digit numbers
    long long min_val = 1;
    for (int i = 1; i < n; i++) min_val *= b;
    long long max_val = min_val * b;
    
    for (long long x : current) {
        if (x >= min_val && x < max_val) {
            solutions.push_back(x);
        }
    }
    
    sort(solutions.begin(), solutions.end());
    
    cout << solutions.size() << "\n";
    for (long long sol : solutions) {
        cout << toBase(sol, b) << "\n";
    }
    
    return 0;
}
```

The algorithm:
1. First finds all single-digit self-replicating numbers (where x² ≡ x mod b)
2. Then iteratively lifts these solutions to higher moduli using Hensel lifting
3. At each step k, for each solution x modulo b^k, we try adding multiples of b^k to find solutions modulo b^(k+1)
4. Finally filters the results to keep only n-digit numbers (those in range [b^(n-1), b^n))
5. Converts results back to the given base for output

Time complexity is manageable because the number of solutions at each level is typically small and bounded.
