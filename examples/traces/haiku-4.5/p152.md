Looking at this problem, I need to:

1. Read N candidates and their vote counts
2. Convert each candidate's votes to a percentage of total votes
3. Round each percentage to an integer such that:
   - Each rounded value is within ±0.5 of the true percentage (standard rounding)
   - The sum of all rounded values equals exactly 100

This is a classic rounding problem where we need to distribute 100% among candidates while respecting rounding rules.

My approach:
1. Calculate the true percentage for each candidate: `percent[i] = 100.0 * votes[i] / total_votes`
2. Start with the floor of each percentage
3. Calculate remainders (fractional parts)
4. Distribute the remaining percentage points to candidates with the largest remainders (greedy approach)
5. Verify that each final value is within ±0.5 of its true percentage (valid rounding)

Let me trace through the example:
- Input: 2 candidates with 10, 10 votes
- Total: 20 votes
- True percentages: 50.0%, 50.0%
- Floor values: 50, 50
- Sum: 100 ✓
- Output: 50 50

Here's my solution:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    
    vector<long long> votes(n);
    long long total = 0;
    
    for(int i = 0; i < n; i++){
        cin >> votes[i];
        total += votes[i];
    }
    
    // Calculate exact percentages
    vector<double> exact(n);
    for(int i = 0; i < n; i++){
        exact[i] = 100.0 * votes[i] / total;
    }
    
    // Start with floor values
    vector<int> result(n);
    int sum = 0;
    vector<pair<double, int>> remainders; // (fractional part, index)
    
    for(int i = 0; i < n; i++){
        result[i] = (int)floor(exact[i]);
        sum += result[i];
        remainders.push_back({exact[i] - floor(exact[i]), i});
    }
    
    // Distribute remaining percentage points to candidates with largest remainders
    int remaining = 100 - sum;
    
    // Sort by remainder in descending order
    sort(remainders.begin(), remainders.end(), greater<pair<double, int>>());
    
    for(int i = 0; i < remaining; i++){
        result[remainders[i].second]++;
    }
    
    // Verify solution: each value should be floor or ceil of exact percentage
    for(int i = 0; i < n; i++){
        int floor_val = (int)floor(exact[i]);
        int ceil_val = (int)ceil(exact[i]);
        if(result[i] != floor_val && result[i] != ceil_val){
            cout << "No solution" << endl;
            return 0;
        }
    }
    
    // Verify sum
    int final_sum = 0;
    for(int i = 0; i < n; i++){
        final_sum += result[i];
    }
    
    if(final_sum != 100){
        cout << "No solution" << endl;
        return 0;
    }
    
    for(int i = 0; i < n; i++){
        if(i > 0) cout << " ";
        cout << result[i];
    }
    cout << endl;
    
    return 0;
}
```

The algorithm:
1. Calculate total votes
2. For each candidate, compute their exact percentage
3. Initialize result with floor of each percentage
4. Calculate how many additional percentage points we need to distribute (to reach 100)
5. Distribute these points greedily to candidates with largest fractional parts
6. Verify that each rounded value is either floor or ceil of the exact percentage (valid rounding)
7. Verify the sum is exactly 100

This greedy approach based on remainders ensures we get a valid solution if one exists.
