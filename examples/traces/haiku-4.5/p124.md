Looking at this problem, I need to:

1. Read a closed broken line consisting of K segments (axis-aligned)
2. Determine if a point (X0, Y0) is INSIDE, OUTSIDE, or on the BORDER of this closed polygon

The approach:
- First, check if the point lies on any segment (BORDER case)
- If not on border, use a ray casting algorithm to determine if inside or outside
  - Cast a ray from the point to infinity (typically horizontal to the right)
  - Count how many times it crosses the polygon boundary
  - Odd number of crossings = INSIDE, Even = OUTSIDE

For axis-aligned segments, I can cast a horizontal ray to the right and count crossings with vertical segments, or cast a vertical ray upward and count crossings with horizontal segments.

Let me implement this:

```cpp
#include<bits/stdc++.h>
using namespace std;

struct Point {
    int x, y;
};

struct Segment {
    Point p1, p2;
};

bool onSegment(Point p, Segment s) {
    int minX = min(s.p1.x, s.p2.x);
    int maxX = max(s.p1.x, s.p2.x);
    int minY = min(s.p1.y, s.p2.y);
    int maxY = max(s.p1.y, s.p2.y);
    
    if (p.x >= minX && p.x <= maxX && p.y >= minY && p.y <= maxY) {
        // Check if point is actually on the segment
        // For axis-aligned segments, if point is in bounding box, it's on segment
        if (s.p1.x == s.p2.x && p.x == s.p1.x) return true; // vertical line
        if (s.p1.y == s.p2.y && p.y == s.p1.y) return true; // horizontal line
    }
    return false;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int k;
    cin >> k;
    
    vector<Segment> segments(k);
    for(int i = 0; i < k; i++) {
        cin >> segments[i].p1.x >> segments[i].p1.y >> segments[i].p2.x >> segments[i].p2.y;
    }
    
    int x0, y0;
    cin >> x0 >> y0;
    
    Point p = {x0, y0};
    
    // Check if point is on border
    for(int i = 0; i < k; i++) {
        if(onSegment(p, segments[i])) {
            cout << "BORDER" << endl;
            return 0;
        }
    }
    
    // Ray casting algorithm - cast ray to the right (positive x direction)
    int crossings = 0;
    
    for(int i = 0; i < k; i++) {
        Segment s = segments[i];
        
        // Only consider vertical segments for horizontal ray
        if(s.p1.x == s.p2.x) {
            int segX = s.p1.x;
            int minY = min(s.p1.y, s.p2.y);
            int maxY = max(s.p1.y, s.p2.y);
            
            // Ray goes from (x0, y0) to the right (increasing x)
            // Check if this vertical segment crosses the ray
            if(segX > x0 && y0 >= minY && y0 <= maxY) {
                // Need to be careful about endpoints
                if(y0 == minY || y0 == maxY) {
                    // Point is at endpoint of vertical segment
                    // Count it only once by checking neighboring segments
                    // For now, count it as 0.5
                    // This is handled by checking if both endpoints are on same side
                    int otherY = (s.p1.y == minY) ? maxY : minY;
                    if((otherY > y0 && minY > y0) || (otherY < y0 && minY < y0)) {
                        crossings++;
                    }
                } else {
                    crossings++;
                }
            }
        }
    }
    
    if(crossings % 2 == 1) {
        cout << "INSIDE" << endl;
    } else {
        cout << "OUTSIDE" << endl;
    }
    
    return 0;
}
```
