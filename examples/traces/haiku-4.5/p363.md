Actually, I think simpler approach: track when each window becomes free. When person i arrives:
1. Determine which window has fewest people waiting (need to count)
2. That person starts service at max(arrival_time, window_free_time)

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
  ios_base::sync_with_stdio(false);
  cin.tie(nullptr);
  
  int N, M;
  cin >> N >> M;
  
  vector<int> window_free(M + 1, 0);  // when each window becomes free (1-indexed)
  vector<int> queue_size(M + 1, 0);   // number of people in each queue
  
  for(int i = 0; i < N; i++){
    int k, t;
    cin >> k >> t;
    
    // Find which window to join
    int best_window = 1;
    int min_queue_size = queue_size[1];
    
    for(int w = 2; w <= M; w++){
      if(queue_size[w] < min_queue_size){
        min_queue_size = queue_size[w];
        best_window = w;
      }
    }
    
    // Person joins the queue
    queue_size[best_window]++;
    
    // Calculate when they leave
    int start_service = max(k, window_free[best_window]);
    int leave_time = start_service + t;
    
    // Update window free time
    window_free[best_window] = leave_time;
    
    // Person has been served, remove from queue
    queue_size[best_window]--;
    
    cout << best_window << " " << leave_time << "\n";
  }
  
  return 0;
}
```
