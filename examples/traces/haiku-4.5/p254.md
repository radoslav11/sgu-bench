Looking at this problem, I need to simulate a circular elimination process where:

1. Numbers 1 to N are arranged in a circle
2. Start at 1, moving clockwise
3. Count Q positions in current direction and erase that number
4. Move to next number (in clockwise direction)
5. If next number is odd → direction becomes clockwise
6. If next number is even → direction becomes counter-clockwise
7. Find the last erased number

Let me trace through example 2 (N=5, Q=3):
- Circle: 1, 2, 3, 4, 5
- Start at 1, moving clockwise
- Count 3 clockwise from 1: 1→2→3, erase 3
- Move to next clockwise after 3: 4 (even) → direction = counter-clockwise
- Count 3 counter-clockwise from 4: 4→2→1, erase 1
- Move to next clockwise after 1: 2 (even) → direction = counter-clockwise
- Count 3 counter-clockwise from 2: 2→5→4, erase 4
- Move to next clockwise after 4: 5 (odd) → direction = clockwise
- Count 3 clockwise from 5: 5→2→5, erase 5
- Move to next clockwise after 5: 2 (even) → direction = counter-clockwise
- Count 3 counter-clockwise from 2: 2→2→2, erase 2
- No numbers left, answer is 2 ✓

#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, Q;
    cin >> N >> Q;
    
    list<int> circle;
    for(int i = 1; i <= N; i++){
        circle.push_back(i);
    }
    
    auto it = circle.begin(); // start at 1
    bool clockwise = true;
    int lastErased = -1;
    
    while(!circle.empty()){
        // Count Q positions in current direction
        int count = Q;
        while(count > 1){
            if(clockwise){
                it++;
                if(it == circle.end()) it = circle.begin();
            } else {
                if(it == circle.begin()) it = circle.end();
                it--;
            }
            count--;
        }
        
        // Erase current position
        lastErased = *it;
        auto next_it = it;
        
        // Move to next position clockwise before erasing
        next_it++;
        if(next_it == circle.end()) next_it = circle.begin();
        
        // Erase the counted position
        circle.erase(it);
        
        if(circle.empty()) break;
        
        // Set iterator to the next clockwise position
        it = next_it;
        if(it == circle.end()) it = circle.begin();
        
        // Determine new direction based on the next number
        int nextNum = *it;
        if(nextNum % 2 == 1){
            clockwise = true;
        } else {
            clockwise = false;
        }
    }
    
    cout << lastErased << endl;
    
    return 0;
}
