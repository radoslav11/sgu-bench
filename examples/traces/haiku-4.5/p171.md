Looking at this problem, I need to:

1. Assign N students to K zones
2. Each zone i must receive exactly N[i] students
3. A student with olympiad level P gets invited to All-Russian olympiad if P > Q (zone level)
4. Maximize the total weight of students who get invited (where P > Q for their assigned zone)

Key insight: This is an assignment problem. For each student, I need to decide which zone to assign them to. A student contributes to the total weight only if their olympiad level exceeds the zone level they're assigned to.

Strategy:
- For each student-zone pair, calculate the "benefit": if the student's level > zone's level, benefit = student's weight, else benefit = 0
- We want to maximize total benefit while respecting zone capacity constraints
- This is a maximum weight bipartite matching problem, but with capacity constraints

A greedy approach might work:
1. For each student, calculate which zones would qualify them (P > Q)
2. Try to assign students to zones greedily, prioritizing assignments that maximize weight gain

Better approach - use a greedy assignment:
1. Create a list of (student_index, zone_index, weight_gain) tuples for all valid assignments (where P > Q)
2. Sort by weight_gain in descending order
3. Greedily assign while respecting capacity constraints
4. Fill remaining slots with any valid assignment

Let me implement this:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int K;
    cin >> K;
    
    vector<int> N(K);
    int total_students = 0;
    for(int i = 0; i < K; i++){
        cin >> N[i];
        total_students += N[i];
    }
    
    vector<int> Q(K);
    for(int i = 0; i < K; i++){
        cin >> Q[i];
    }
    
    vector<int> P(total_students);
    for(int i = 0; i < total_students; i++){
        cin >> P[i];
    }
    
    vector<int> w(total_students);
    for(int i = 0; i < total_students; i++){
        cin >> w[i];
    }
    
    vector<int> assignment(total_students, -1);
    vector<int> zone_count(K, 0);
    
    // Create list of (weight, student_idx, zone_idx) for beneficial assignments
    vector<tuple<int, int, int>> beneficial;
    for(int i = 0; i < total_students; i++){
        for(int j = 0; j < K; j++){
            if(P[i] > Q[j]){
                beneficial.push_back(make_tuple(w[i], i, j));
            }
        }
    }
    
    // Sort by weight in descending order
    sort(beneficial.rbegin(), beneficial.rend());
    
    // Greedily assign beneficial assignments
    for(auto [weight, student, zone] : beneficial){
        if(zone_count[zone] < N[zone] && assignment[student] == -1){
            assignment[student] = zone;
            zone_count[zone]++;
        }
    }
    
    // Fill remaining slots with any valid zone
    for(int i = 0; i < total_students; i++){
        if(assignment[i] == -1){
            for(int j = 0; j < K; j++){
                if(zone_count[j] < N[j]){
                    assignment[i] = j;
                    zone_count[j]++;
                    break;
                }
            }
        }
    }
    
    for(int i = 0; i < total_students; i++){
        cout << assignment[i] + 1;
        if(i < total_students - 1) cout << " ";
    }
    cout << "\n";
    
    return 0;
}
```

The algorithm:
1. Read all input data
2. For each student-zone pair where the student qualifies (P > Q), add to beneficial list with weight
3. Sort beneficial assignments by weight (descending)
4. Greedily pick assignments from best weight down, respecting capacity constraints
5. For unassigned students, fill remaining zone slots arbitrarily

This greedy approach works because we prioritize high-weight beneficial assignments first, then fill remaining capacity with any assignment.
