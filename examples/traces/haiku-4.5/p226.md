Looking at this problem, I need to find the shortest path from vertex 1 to vertex N in a directed graph where consecutive edges must have different colors.

Key observations:
1. This is a shortest path problem with an additional constraint: no two consecutive edges can have the same color
2. Standard Dijkstra's/BFS won't work directly because the state needs to track not just which vertex we're at, but also what color the last edge was
3. I need to use a state (vertex, last_color) where last_color is the color of the edge we used to reach that vertex

Approach:
- Use BFS with state = (vertex, last_color_used)
- For each state, try all outgoing edges from that vertex
- Only take an edge if its color is different from last_color_used
- Track distances to each (vertex, color) pair
- The answer is the minimum distance to reach vertex N with any color

Let me trace through Test #1:
- Start at vertex 1, no previous color
- Take edge 1→2 (color 1), state = (2, 1), distance = 1
- From (2, 1), can take edge 2→3 (color 2, different from 1), state = (3, 2), distance = 2
- From (3, 2), can take edge 3→4 (color 3, different from 2), state = (4, 3), distance = 3
- From (2, 1), can take edge 2→4 (color 1, same as last), can't take
- Answer: 3 ✓

Test #2:
- Start at vertex 1
- Take edge 1→2 (color 1), state = (2, 1), distance = 1
- From (2, 1), edge 2→3 has color 1 (same as last), can't take
- Answer: -1 ✓

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, M;
    cin >> N >> M;
    
    vector<vector<pair<int, int>>> graph(N + 1); // graph[u] = {(v, color), ...}
    
    for(int i = 0; i < M; i++){
        int x, y, c;
        cin >> x >> y >> c;
        graph[x].push_back({y, c});
    }
    
    // State: (vertex, last_color)
    // dist[v][c] = shortest distance to vertex v where last edge had color c
    // c = 0 means no previous edge (for start state)
    vector<vector<int>> dist(N + 1, vector<int>(4, INT_MAX));
    
    queue<pair<int, int>> q; // (vertex, last_color)
    q.push({1, 0}); // start at vertex 1 with no previous color
    dist[1][0] = 0;
    
    while(!q.empty()){
        auto [u, last_color] = q.front();
        q.pop();
        
        int current_dist = dist[u][last_color];
        
        // Try all outgoing edges
        for(auto [v, color] : graph[u]){
            // Can only take this edge if color is different from last_color
            if(color != last_color){
                int new_dist = current_dist + 1;
                if(new_dist < dist[v][color]){
                    dist[v][color] = new_dist;
                    q.push({v, color});
                }
            }
        }
    }
    
    // Find minimum distance to vertex N with any last color
    int ans = INT_MAX;
    for(int c = 0; c <= 3; c++){
        ans = min(ans, dist[N][c]);
    }
    
    if(ans == INT_MAX){
        cout << -1 << endl;
    } else {
        cout << ans << endl;
    }
    
    return 0;
}
```
