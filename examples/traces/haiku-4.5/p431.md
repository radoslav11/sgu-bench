Looking at this problem, I need to find a wildcard pattern that matches all files in the first list but doesn't match any files in the second list.

Let me think about the approach:

1. A pattern with wildcards can match files in various ways
2. I need to find a pattern that:
   - Matches all `n` files (positive set)
   - Doesn't match any of `m` files (negative set)

Key observations:
- `?` matches exactly one character
- `*` matches zero or more characters
- I can try patterns based on the positive examples
- The simplest approach: try prefixes, suffixes, and combinations with wildcards

Strategy:
1. Start with simple patterns from the positive examples
2. For each positive example, generate candidate patterns (with wildcards)
3. Check if each candidate pattern matches all positive files and no negative files
4. Return the first valid pattern found

Let me implement a function to check if a pattern matches a filename:

```cpp
bool matches(const string& pattern, const string& filename) {
    int p = 0, f = 0;
    int p_star = -1, f_star = -1;
    
    while (f < filename.length()) {
        if (p < pattern.length() && (pattern[p] == '?' || pattern[p] == filename[f])) {
            p++;
            f++;
        } else if (p < pattern.length() && pattern[p] == '*') {
            p_star = p;
            f_star = f;
            p++;
        } else if (p_star != -1) {
            p = p_star + 1;
            f_star++;
            f = f_star;
        } else {
            return false;
        }
    }
    
    while (p < pattern.length() && pattern[p] == '*') {
        p++;
    }
    
    return p == pattern.length();
}
```

```cpp
    int p = 0, f = 0;
    int p_star = -1, f_star = -1;
    
    while (f < (int)filename.length()) {
        if (p < (int)pattern.length() && (pattern[p] == '?' || pattern[p] == filename[f])) {

            f++;

bool isValid(const string& pattern, const vector<string>& yes, const vector<string>& no) {
    for (const auto& s : yes) {
        if (!matches(pattern, s)) return false;
    }
    for (const auto& s : no) {
        if (matches(pattern, s)) return false;
    }
    return true;
}

int main(){

    int n, m;
    cin >> n >> m;
    
    vector<string> yes(n), no(m);
    for (int i = 0; i < n; i++) cin >> yes[i];
    for (int i = 0; i < m; i++) cin >> no[i];
    
    // Try patterns based on positive examples
    // Try each positive example with wildcards added
    for (const auto& base : yes) {
        // Try the base itself
        if (isValid(base, yes, no)) {
            cout << base << endl;
            return 0;
        }
        
        // Try with * at start and end
        if (isValid("*" + base + "*", yes, no)) {

            return 0;
        }
        
        // Try with * at start
        if (isValid("*" + base, yes, no)) {
            cout << "*" + base << endl;
            return 0;
        }
        
        // Try with * at end
        if (isValid(base + "*", yes, no)) {
            cout << base + "*" << endl;

        string pattern = base;
        for (int i = 0; i < (int)pattern.length(); i++) {

            pattern[i] = orig;
```
