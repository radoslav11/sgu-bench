Looking at this problem, I need to:

1. Parse a Pascal top-down parser
2. Extract the grammar rules by analyzing how each parsing routine can execute
3. Output the grammar in Backus-Naur form (BNF)

Key observations:
- Each parsing routine becomes a nonterminal in the grammar
- Each possible execution path through a routine's body (considering all If/Else branches) generates one production rule
- We need to track what gets "consumed" (terminals via Skip) and what nonterminals are called
- Multiple paths through conditionals create multiple productions for the same nonterminal

Let me write a parser for the Pascal code:

```cpp
#include<bits/stdc++.h>
using namespace std;

string input;
int pos;
set<string> routines;
map<string, set<string>> grammar;

void skipWhitespace() {
    while(pos < input.size() && isspace(input[pos])) pos++;
}

string readWord() {
    skipWhitespace();
    string word;
    while(pos < input.size() && isalnum(input[pos])) {
        word += tolower(input[pos]);
        pos++;
    }
    return word;
}

char readChar() {
    skipWhitespace();
    if(input[pos] == '\'') {
        pos++;
        char c = input[pos];
        pos++;
        pos++; // closing quote
        return c;
    }
    return 0;
}

void parseSegments(string& rhs);

void parseSegment(string& rhs) {
    skipWhitespace();
    if(input[pos] == 'I' || input[pos] == 'i') {
        // Conditional
        pos += 2; // skip "If"
        skipWhitespace();
        pos += 4; // skip "Peek"
        skipWhitespace();
        pos++; // skip "="
        char c = readChar();
        skipWhitespace();
        pos += 4; // skip "Then"
        skipWhitespace();
        pos += 5; // skip "Begin"
        
        string ifRhs = rhs + "'";
        ifRhs += c;
        ifRhs += "'";
        parseSegments(ifRhs);
        
        skipWhitespace();
        if(input[pos] == 'E' || input[pos] == 'e') {
            pos += 4; // "Else"
            skipWhitespace();
            if(input[pos] == 'I' || input[pos] == 'i') {
                pos = pos - 4; // back up
                string elseRhs = rhs;
                parseSegment(elseRhs);
            } else if(input[pos] == 'E' || input[pos] == 'e') {
                pos += 5; // "Error"
                skipWhitespace();
                pos++; // ";"
            }
        }
        skipWhitespace();
        pos++; // ";"
    } else {
        // Unconditional - routine call
        string name = readWord();
        rhs += "<" + name + ">";
        skipWhitespace();
        pos++; // ";"
    }
}

void parseSegments(string& rhs) {
    skipWhitespace();
    while(input[pos] != 'E' && input[pos] != 'e') {
        parseSegment(rhs);
        skipWhitespace();
    }
}

int main(){
    string line;
    while(getline(cin, line)) input += line + "\n";
    
    pos = 0;
    
    // Find all routine names
    while(pos < input.size()) {
        skipWhitespace();
        string word = readWord();
        if(word == "procedure") {
            string name = readWord();
            if(name != "skip" && name != "error") {
                routines.insert(name);
            }
        }
    }
    
    // Parse each routine's body
    pos = 0;
    while(pos < input.size()) {
        skipWhitespace();
        string word = readWord();
        if(word == "procedure") {
            string name = readWord();
            if(routines.count(name)) {
                skipWhitespace();
                pos++; // ";"
                skipWhitespace();
                // Skip to Begin
                while(!(input[pos] == 'B' && (input[pos+1] == 'e' || input[pos+1] == 'E'))) pos++;
                pos += 5; // "Begin"
                
                string rhs;
                parseSegments(rhs);
                grammar[name].insert(rhs);
            }
        }
    }
    
    // Output
    for(auto& p : grammar) {
        cout << "<" << p.first << ">::=";
        bool first = true;
        for(auto& prod : p.second) {
            if(!first) cout << "|";
            cout << prod;
            first = false;
        }
        cout << "\n";
    }
    
    return 0;
}
```

```cpp
void parseSegment(string rhs, string routine);

void parseSegment(string rhs, string routine) {
    skipWhitespace();
    string word = readWord();
    
    if(word == "if") {
        readWord(); // Peek
        skipWhitespace();
        pos++; // =
        char c = readChar();
        readWord(); // Then
        readWord(); // Begin
        
        string ifRhs = rhs + "'";
        ifRhs += c;
        ifRhs += "'";
        parseSegments(ifRhs, routine);
        skipWhitespace();
        pos++; // End
        
        skipWhitespace();
        if(pos < input.size() && input[pos] == 'E') {
            readWord(); // Else
            skipWhitespace();
            if(input[pos] == 'I') {
                parseSegment(rhs, routine);
            } else if(input[pos] == 'E') {
                readWord(); // Error
            }
        }
        skipWhitespace();
        pos++; // ;
    } else {
        rhs += "<" + word + ">";
        skipWhitespace();
        pos++; // ;
        grammar[routine].insert(rhs);
    }
}

void parseSegments(string rhs, string routine) {
    skipWhitespace();
    while(pos < input.size()) {
        string word = readWord();
        if(word == "end") {
            pos--; 
            for(int i = 0; i < 3; i++) pos--;
            break;
        }
        pos -= word.length();
        parseSegment(rhs, routine);
    }
    skipWhitespace();
    pos++; // E
    pos++; // n
    pos++; // d
    skipWhitespace();
    pos++; // ;
}
```

Actually, let me implement this more carefully with proper tokenization:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    string input, line;
    while(getline(cin, line)) input += " " + line;
    
    map<string, set<string>> grammar;
    
    int n = input.size();
    int i = 0;
    
    auto skipWS = [&]() {
        while(i < n && isspace(input[i])) i++;
    };
    
    auto readWord = [&]() {
        skipWS();
        string s;
        while(i < n && isalnum(input[i])) s += tolower(input[i++]);
        return s;
    };
    
    auto readChar = [&]() {
        skipWS();
        i++; // '
        char c = input[i++];
        i++; // '
        return c;
    };
    
    function<void(string, string)> parseSegs = [&](string rhs, string routine) {
        skipWS();
        while(i < n) {
            string w = readWord();
            if(w == "end") break;
            if(w == "if") {
                readWord(); // Peek
                skipWS();
                i++; // =
                char c = readChar();
                readWord(); // Then
                readWord(); // Begin
                parseSegs(rhs + "'" + c + "'", routine);
                skipWS();
                i++; // End
                skipWS();
                if(input[i] == 'E') {
                    readWord(); // Else
                    skipWS();
                    if(input[i] == 'I') {
                        parseSegs(rhs, routine);
                    } else {
                        readWord(); // Error
                    }
                }
                skipWS();
                i++; // ;
            } else {
                rhs += "<" + w + ">";
                skipWS();
                i++; // ;
                grammar[routine].insert(rhs);
            }
        }
        skipWS();
        i += 3; // End
        skipWS();
        i++; // ;
    };
    
    i = 0;
    while(i < n) {
        string w = readWord();
        if(w == "procedure") {
            string name = readWord();
            skipWS();
            i++; // ;
            skipWS();
            string next = readWord();
            if(next != "forward") {
                i -= next.length();
                readWord(); // Begin
                parseSegs("", name);
            }
        }
    }
    
    for(auto& [k, v] : grammar) {
        cout << "<" << k << ">::=";
        bool first = true;
        for(auto& s : v) {
            if(!first) cout << "|";
            cout << s;
            first = false;
        }
        cout << "\n";
    }
    
    return 0;
}
```
