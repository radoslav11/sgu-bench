Looking at this problem, I need to find the minimum size `d` such that 3 squares of side length `d` can cover all `n` points.

Key observations:
1. We have 3 identical squares with sides parallel to axes
2. We need to find the minimum `d` (side length) such that all points can be covered
3. A square at position (x0, y0) covers points with coordinates in range [x0, x0+d] × [y0, y0+d]

Strategy:
1. Use binary search on `d`
2. For each candidate `d`, check if 3 squares can cover all points
3. To check if 3 squares suffice for a given `d`:
   - Try different placements of squares
   - For each point, a square covering it can be positioned in multiple ways
   - We need to find if there exists a placement of 3 squares that covers everything

For the feasibility check with a given `d`:
- One approach: try placing the first square at different positions (anchored to specific points)
- For each placement of first square, try placements of second square
- Check if remaining points can be covered by third square

Actually, a better approach for small n (≤20000):
- Try all possible positions for the first square (anchor it so one point is at a corner)
- Try all possible positions for the second square
- Check if the third square can cover remaining points

Since we only have 3 squares, we can:
1. Binary search on `d`
2. For each `d`, try all combinations:
   - First square: try anchoring at each point (or slightly before it)
   - Second square: try anchoring at each uncovered point
   - Check if third square can cover rest

```cpp
#include<bits/stdc++.h>
using namespace std;

struct Point {
    long long x, y;
};

int n;
vector<Point> points;

bool canCoverWithSquare(vector<int>& uncovered, long long x0, long long y0, long long d) {
    for(int idx : uncovered) {
        if(points[idx].x < x0 || points[idx].x > x0 + d ||
           points[idx].y < y0 || points[idx].y > y0 + d) {
            return false;
        }
    }
    return true;
}

bool canCoverWith3Squares(long long d) {
    // Try first square at various positions
    for(int i = 0; i < n; i++) {
        // Position first square so point i is at bottom-left corner
        long long x1 = points[i].x;
        long long y1 = points[i].y;
        
        vector<int> uncovered1;
        for(int j = 0; j < n; j++) {
            if(points[j].x < x1 || points[j].x > x1 + d ||
               points[j].y < y1 || points[j].y > y1 + d) {
                uncovered1.push_back(j);
            }
        }
        
        if(uncovered1.empty()) return true;
        
        // Try second square at various positions
        for(int j : uncovered1) {
            long long x2 = points[j].x;
            long long y2 = points[j].y;
            
            vector<int> uncovered2;
            for(int k : uncovered1) {
                if(points[k].x < x2 || points[k].x > x2 + d ||
                   points[k].y < y2 || points[k].y > y2 + d) {
                    uncovered2.push_back(k);
                }
            }
            
            if(uncovered2.empty()) return true;
            
            // Try third square
            for(int k : uncovered2) {
                long long x3 = points[k].x;
                long long y3 = points[k].y;
                
                bool ok = true;
                for(int m : uncovered2) {
                    if(points[m].x < x3 || points[m].x > x3 + d ||
                       points[m].y < y3 || points[m].y > y3 + d) {
                        ok = false;
                        break;
                    }
                }
                if(ok) return true;
            }
        }
    }
    return false;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n;
    points.resize(n);
    
    for(int i = 0; i < n; i++) {
        cin >> points[i].x >> points[i].y;
    }
    
    long long left = 0, right = 2e9;
    long long answer = right;
    
    while(left <= right) {
        long long mid = left + (right - left) / 2;
        if(canCoverWith3Squares(mid)) {
            answer = mid;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    
    cout << answer << endl;
    
    return 0;
}
```
