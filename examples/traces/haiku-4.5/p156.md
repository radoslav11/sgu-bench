#include<bits/stdc++.h>
using namespace std;

int N, M;
vector<int> adj[10005];
bool visited[10005];
vector<int> path;

bool dfs(int v, int parent, int start, int depth) {
    visited[v] = true;
    path.push_back(v);
    
    if (depth == N) {
        // Check if we can return to start
        for (int u : adj[v]) {
            if (u == start) {
                return true;
            }
        }
        path.pop_back();
        visited[v] = false;
        return false;
    }
    
    for (int u : adj[v]) {
        if (u == parent) continue;
        if (!visited[u]) {
            if (dfs(u, v, start, depth + 1)) {
                return true;
            }
        }
    }
    
    path.pop_back();
    visited[v] = false;
    return false;
}
