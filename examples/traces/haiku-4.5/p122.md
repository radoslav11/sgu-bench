Looking at this problem, I need to find a Hamiltonian cycle starting and ending at person 1, where each consecutive pair are friends.

Key observations:
1. We have N people, each with at least ⌊(N+1)/2⌋ friends
2. Person 1 has the book and we need to find a path: 1 → ... → 1 that visits everyone exactly once
3. This is finding a Hamiltonian cycle in an undirected graph

By Dirac's theorem, if every vertex has degree ≥ N/2, a Hamiltonian cycle exists. Since each person has ≥ ⌊(N+1)/2⌋ friends, and ⌊(N+1)/2⌋ ≥ N/2 for N ≥ 2, a Hamiltonian cycle should almost always exist.

I'll use backtracking DFS to find the cycle:

```cpp
#include<bits/stdc++.h>
using namespace std;

int N;
vector<vector<int>> friends;
vector<int> path;
vector<bool> visited;

bool dfs(int current, int start, int depth) {
    // If we've visited all nodes
    if (depth == N) {
        // Check if current node is connected to start (node 1)
        for (int f : friends[current]) {
            if (f == start) {
                path.push_back(start);
                return true;
            }
        }
        return false;
    }
    
    // Try each friend of current node
    for (int next : friends[current]) {
        if (!visited[next]) {
            visited[next] = true;
            path.push_back(next);
            
            if (dfs(next, start, depth + 1)) {
                return true;
            }
            
            path.pop_back();
            visited[next] = false;
        }
    }
    
    return false;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> N;
    friends.resize(N + 1);
    visited.resize(N + 1, false);
    
    for (int i = 1; i <= N; i++) {
        int f;
        while (cin >> f) {
            if (cin.peek() == '\n') {
                friends[i].push_back(f);
                break;
            }
            friends[i].push_back(f);
        }
    }
    
    // Start DFS from node 1
    visited[1] = true;
    path.push_back(1);
    
    if (dfs(1, 1, 1)) {
        for (int i = 0; i < path.size(); i++) {
            if (i > 0) cout << " ";
            cout << path[i];
        }
        cout << "\n";
    } else {
        cout << "No solution\n";
    }
    
    return 0;
}
```
