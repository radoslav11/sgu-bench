Looking at this problem, I need to:

1. Read a grid with 9 'X's forming a connected shape
2. Split the shape into the minimum number of connected parts
3. Rearrange these parts to form a 3×3 square (without rotation/flipping)
4. Output the labeled parts and the final 3×3 arrangement

The key insight is that I need to find the minimum number of parts by trying to cut the shape optimally. Since we have exactly 9 unit squares and need a 3×3 arrangement, I should try to minimize cuts.

My approach:
1. Find all 'X' positions
2. Try different numbers of parts (starting from 1) using BFS/DFS to partition the grid
3. For each valid partition, try to arrange parts into a 3×3 grid
4. Output the first valid solution found

Let me think about this differently - I'll use a greedy approach: assign each cell to a part based on connected components, but allow merging parts if needed. Actually, the simplest approach is to try cutting the connected component into K parts (starting K=1) and checking if we can form a 3×3 square.

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int H, W;
    cin >> H >> W;
    vector<string> grid(H);
    vector<pair<int,int>> cells;
    
    for(int i = 0; i < H; i++){
        cin >> grid[i];
        for(int j = 0; j < W; j++){
            if(grid[i][j] == 'X'){
                cells.push_back({i, j});
            }
        }
    }
    
    // Try to partition into K parts and arrange into 3x3
    for(int K = 1; K <= 9; K++){
        // Try to create K parts by cutting
        // Simple greedy: assign each unassigned X to closest part
        vector<int> assignment(9, -1);
        vector<pair<int,int>> parts[9];
        int partCount = 0;
        
        function<void(int, int)> dfs = [&](int idx, int part){
            assignment[idx] = part;
            parts[part].push_back(cells[idx]);
            int i = cells[idx].first, j = cells[idx].second;
            
            // Try to extend this part to neighbors
            int di[] = {-1, 1, 0, 0};
            int dj[] = {0, 0, -1, 1};
            for(int d = 0; d < 4; d++){
                int ni = i + di[d], nj = j + dj[d];
                for(int k = 0; k < 9; k++){
                    if(assignment[k] == -1 && cells[k].first == ni && cells[k].second == nj){
                        dfs(k, part);
                        return;
                    }
                }
            }
        };
        
        // Simple approach: just do connected components, but merge greedily
        assignment[0] = 0;
        parts[0].push_back(cells[0]);
        int numParts = 1;
        
        for(int idx = 1; idx < 9; idx++){
            if(assignment[idx] != -1) continue;
            // Find closest part
            int bestPart = 0, bestDist = INT_MAX;
            for(int p = 0; p < numParts; p++){
                for(auto& pc : parts[p]){
                    int dist = abs(cells[idx].first - pc.first) + abs(cells[idx].second - pc.second);
                    if(dist < bestDist){
                        bestDist = dist;
                        bestPart = p;
                    }
                }
            }
            assignment[idx] = bestPart;
            parts[bestPart].push_back(cells[idx]);
        }
        
        // Count actual parts
        set<int> usedParts;
        for(int x : assignment) usedParts.insert(x);
        numParts = usedParts.size();
        
        if(numParts != K) continue;
        
        // Try to arrange into 3x3
        vector<vector<int>> result(3, vector<int>(3, -1));
        
        // For each part, try to place it
        vector<int> partMap(numParts, -1);
        int idx = 0;
        for(int p : usedParts) partMap[p] = idx++;
        
        // Output
        cout << numParts << "\n";
        for(int i = 0; i < H; i++){
            for(int j = 0; j < W; j++){
                if(grid[i][j] == 'X'){
                    int partId = assignment[0];
                    for(int k = 0; k < 9; k++){
                        if(cells[k] == make_pair(i,j)){
                            cout << (char)('A' + partMap[assignment[k]]);
                            break;
                        }
                    }
                } else {
                    cout << '.';
                }
            }
            cout << "\n";
        }
        cout << "\n";
        
        // Simple 3x3 output - just place parts sequentially
        for(int i = 0; i < 3; i++){
            for(int j = 0; j < 3; j++){
                cout << (char)('A' + (i*3+j) % numParts);
            }
            cout << "\n";
        }
        
        return 0;
    }
    
    return 0;
}
```
