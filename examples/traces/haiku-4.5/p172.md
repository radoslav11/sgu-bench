Looking at this problem, I need to:

1. Schedule exams into two days such that each pupil's two selected exams are on different days
2. Each subject is on exactly one day
3. Check if such a valid partition exists

This is a classic **2-coloring problem** (graph bipartition problem):
- Each subject is a node
- For each pupil selecting subjects i and j, add an edge between i and j
- We need to color the graph with 2 colors such that connected nodes have different colors
- This is possible if and only if the graph is bipartite

A graph is bipartite if and only if it contains no odd-length cycles. We can check this using BFS/DFS coloring.

Let me trace through the example:
- 4 subjects, 4 pupils
- Pupil 1: subjects 1,2 → edge(1,2)
- Pupil 2: subjects 3,4 → edge(3,4)
- Pupil 3: subjects 2,4 → edge(2,4)
- Pupil 4: subjects 1,3 → edge(1,3)

Graph edges: (1,2), (3,4), (2,4), (1,3)

Let me try to 2-color:
- Start with 1 → color 0
- 1 connects to 2,3 → color them 1
- 2 connects to 4 → color 0
- 3 connects to 4 → should be 0 ✓ (matches)

So: Day 1: {1,4}, Day 2: {2,3} ✓

Algorithm:
1. Build a graph where nodes are subjects and edges connect subjects that must be on different days
2. Try to 2-color the graph using BFS/DFS
3. If successful, output the coloring; otherwise output "no"

```cpp
#include<bits/stdc++.h>
using std::cin, std::cout, std::vector;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, M;
    cin >> N >> M;
    
    vector<vector<int>> adj(N + 1);
    
    for(int i = 0; i < M; i++){
        int a, b;
        cin >> a >> b;
        adj[a].push_back(b);
        adj[b].push_back(a);
    }
    
    vector<int> color(N + 1, -1);
    bool is_bipartite = true;
    
    // Try to 2-color the graph
    for(int start = 1; start <= N && is_bipartite; start++){
        if(color[start] == -1){
            queue<int> q;
            q.push(start);
            color[start] = 0;
            
            while(!q.empty() && is_bipartite){
                int u = q.front();
                q.pop();
                
                for(int v : adj[u]){
                    if(color[v] == -1){
                        color[v] = 1 - color[u];
                        q.push(v);
                    } else if(color[v] == color[u]){
                        is_bipartite = false;
                        break;
                    }
                }
            }
        }
    }
    
    if(!is_bipartite){
        cout << "no\n";
    } else {
        cout << "yes\n";
        vector<int> day1, day2;
        
        for(int i = 1; i <= N; i++){
            if(color[i] == 0){
                day1.push_back(i);
            } else {
                day2.push_back(i);
            }
        }
        
        cout << day1.size() << "\n";
        for(int i = 0; i < day1.size(); i++){
            if(i > 0) cout << " ";
            cout << day1[i];
        }
        cout << "\n";
    }
    
    return 0;
}
```

The solution:
1. Reads N subjects and M pupils
2. Builds a graph where each pupil's pair of subjects creates an edge
3. Uses BFS to attempt 2-coloring of the graph
4. If the graph is bipartite (2-colorable), outputs "yes" and the subjects for day 1
5. If not bipartite, outputs "no"

The time complexity is O(N + M) for the BFS coloring, which is efficient enough for the given constraints.
