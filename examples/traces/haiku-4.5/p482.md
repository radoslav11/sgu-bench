Let me think about the perimeter calculation:
- The perimeter includes the top (sum of heights), bottom (sum of heights), left side of first board, right side of last board, and the internal vertical segments where heights change.
- For a sequence of boards with heights h1, h2, ..., hn:
  - Perimeter = 2 * (sum of all heights) + 2 (left and right edges) + sum of |h[i+1] - h[i]| (internal height changes)

When we remove some boards, the perimeter of the remaining fence changes:
- The sum of heights decreases
- The height differences might change

- Original perimeter P = 2*S + 2 + D where S = sum of heights, D = sum of absolute differences
- We need remaining perimeter >= P/2

#include<bits/stdc++.h>
using namespace std;

int n;
vector<int> h;
int original_perimeter;

int calculate_perimeter(vector<int>& remaining_boards) {
    if (remaining_boards.empty()) return 0;
    
    int sum_heights = 0;
    int sum_diffs = 0;
    
    for (int i = 0; i < remaining_boards.size(); i++) {
        sum_heights += h[remaining_boards[i]];
    }
    
    for (int i = 1; i < remaining_boards.size(); i++) {
        sum_diffs += abs(h[remaining_boards[i]] - h[remaining_boards[i-1]]);
    }
    
    return 2 * sum_heights + 2 + sum_diffs;
}
