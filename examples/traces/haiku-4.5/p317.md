Looking at this problem, I need to find the minimum time for a messenger to travel from position 0 to position B, using horses at various stables along the way.

Key observations:
1. The messenger starts at position 0 and needs to reach position B
2. At each stable, they can pick any horse (instantaneously)
3. Each horse has a speed `v` and maximum distance `d` it can travel
4. We need to find the minimum time to reach B, or -1 if impossible

This is a shortest path problem. I can use Dijkstra's algorithm where:
- State: current position
- From each stable position, I can reach certain positions ahead using available horses
- Need to find the minimum time to reach position B

Let me think about the approach:
1. Add position 0 as a starting point and position B as the destination
2. For each stable, consider which horses can be used and where they can take us
3. From position X with a horse of speed v and distance d, we can reach any position in range [X, min(X+d, B)]
4. Use Dijkstra to find shortest time

Actually, a simpler greedy approach might work:
- At each stable, pick the horse that gets us furthest while still being able to reach the next stable or destination
- Or use Dijkstra with states being stables (and the final position B)

Let me implement Dijkstra's approach where I consider:
- Starting from position 0
- At each stable position, I can take any horse to potentially reach other stables or the destination
- Find minimum time to reach position B

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    long long B;
    int N;
    cin >> B >> N;
    
    vector<pair<long long, vector<pair<long long, long long>>>> stables; // position, horses (speed, distance)
    
    for(int i = 0; i < N; i++){
        long long X;
        int M;
        cin >> X >> M;
        vector<pair<long long, long long>> horses;
        for(int j = 0; j < M; j++){
            long long v, d;
            cin >> v >> d;
            horses.push_back({v, d});
        }
        stables.push_back({X, horses});
    }
    
    // Dijkstra: state is position, value is minimum time
    map<long long, double> dist;
    priority_queue<pair<double, long long>, vector<pair<double, long long>>, greater<pair<double, long long>>> pq;
    
    dist[0] = 0;
    pq.push({0, 0});
    
    while(!pq.empty()){
        auto [time, pos] = pq.top();
        pq.pop();
        
        if(pos == B){
            cout << fixed << setprecision(8) << time << endl;
            return 0;
        }
        
        if(dist.count(pos) && dist[pos] < time) continue;
        
        // Find stable at current position
        vector<pair<long long, long long>> available_horses;
        for(auto& stable : stables){
            if(stable.first == pos){
                for(auto& horse : stable.second){
                    available_horses.push_back(horse);
                }
            }
        }
        
        // If no horses at current position and pos != 0, skip
        if(available_horses.empty() && pos != 0) continue;
        
        // If at position 0, must find horses from first stable
        if(pos == 0){
            for(auto& stable : stables){
                for(auto& horse : stable.second){
                    available_horses.push_back(horse);
                }
            }
        }
        
        // Try each horse
        for(auto [speed, max_dist] : available_horses){
            long long reach = min(pos + max_dist, B);
            double new_time = time + (double)(reach - pos) / speed;
            
            if(reach == B){
                cout << fixed << setprecision(8) << new_time << endl;
                return 0;
            }
            
            // Check if we can reach any stable from here
            for(auto& stable : stables){
                long long stable_pos = stable.first;
                if(stable_pos > pos && stable_pos <= reach){
                    double time_to_stable = time + (double)(stable_pos - pos) / speed;
                    if(!dist.count(stable_pos) || dist[stable_pos] > time_to_stable){
                        dist[stable_pos] = time_to_stable;
                        pq.push({time_to_stable, stable_pos});
                    }
                }
            }
        }
    }
    
    cout << "-1" << endl;
    return 0;
}
```
