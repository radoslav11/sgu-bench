Looking at this problem, I need to find the minimum time to travel distance `d` on an N-lane road where each lane has a time-varying speed function.

Key observations:
1. Speed in lane i at time t: `v_i(t) = b_i + a_i * sin(t + δ_i)`
2. Changing lanes from x to y takes time `c * |x - y|` during which we don't move
3. We start at lane 1, time 0, and need to cover distance d
4. We can end at any lane

This is a shortest path problem that can be solved with dynamic programming or simulation with switching strategy.

Approach:
- Use a state-space search where state = (current_lane, current_distance, current_time)
- At each state, decide whether to stay in current lane or switch to another lane
- Use a greedy/DP approach with numerical integration

Since lanes change smoothly with sine functions, I'll simulate movement by:
1. Sampling time points and evaluating which lane gives best progress
2. Using Dijkstra-like approach or greedy simulation with backtracking for lane changes

Let me implement a solution using numerical simulation:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::precision(15);
    
    int N, d;
    double c;
    cin >> N >> d >> c;
    
    vector<double> a(N+1), b(N+1), delta(N+1);
    for(int i = 1; i <= N; i++){
        cin >> a[i] >> b[i] >> delta[i];
    }
    
    // Simulate with fine time steps, trying different lane strategies
    double best_time = 1e9;
    vector<pair<int, double>> best_switches;
    
    // Try different switching strategies using greedy approach
    int current_lane = 1;
    double current_time = 0;
    double current_dist = 0;
    vector<pair<int, double>> switches;
    
    double dt = 0.001; // time step
    
    while(current_dist < d){
        // Find best lane at current time
        int best_lane = current_lane;
        double best_speed = b[current_lane] + a[current_lane] * sin(current_time + delta[current_lane]);
        
        for(int lane = 1; lane <= N; lane++){
            double speed = b[lane] + a[lane] * sin(current_time + delta[lane]);
            // Consider switching cost
            double net_benefit = speed - (c * abs(lane - current_lane) / dt);
            double best_benefit = best_speed - (c * abs(best_lane - current_lane) / dt);
            
            if(net_benefit > best_benefit){
                best_lane = lane;
                best_speed = speed;
            }
        }
        
        // Switch if beneficial
        if(best_lane != current_lane){
            double switch_time = c * abs(best_lane - current_lane);
            switches.push_back({best_lane, current_time});
            current_lane = best_lane;
            current_time += switch_time;
        } else {
            // Move in current lane
            double speed = b[current_lane] + a[current_lane] * sin(current_time + delta[current_lane]);
            double dist_step = speed * dt;
            
            if(current_dist + dist_step >= d){
                // Reached destination
                double remaining = d - current_dist;
                double time_needed = remaining / speed;
                current_time += time_needed;
                current_dist = d;
                break;
            }
            
            current_dist += dist_step;
            current_time += dt;
        }
    }
    
    cout << fixed << current_time << "\n";
    cout << switches.size() << "\n";
    for(auto& sw : switches){
        cout << sw.first << " " << sw.second << "\n";
    }
    
    return 0;
}
```

Wait, the greedy approach above is too simplistic. Let me use a better approach with finer control:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::precision(15);
    
    int N, d;
    double c;
    cin >> N >> d >> c;
    
    vector<double> a(N+1), b(N+1), delta(N+1);
    for(int i = 1; i <= N; i++){
        cin >> a[i] >> b[i] >> delta[i];
    }
    
    int current_lane = 1;
    double current_time = 0;
    double current_dist = 0;
    vector<pair<int, double>> switches;
    
    double dt = 0.01;
    
    while(current_dist < d - 1e-9){
        double best_lane = current_lane;
        double best_score = -1e9;
        
        for(int lane = 1; lane <= N; lane++){
            double speed = b[lane] + a[lane] * sin(current_time + delta[lane]);
            double switch_cost = c * abs(lane - current_lane);
            double score = speed - switch_cost / dt;
            
            if(score > best_score){
                best_score = score;
                best_lane = lane;
            }
        }
        
        if(best_lane != current_lane){
            double switch_time = c * abs(best_lane - current_lane);
            switches.push_back({(int)best_lane, current_time});
            current_lane = best_lane;
            current_time += switch_time;
        } else {
            double speed = b[current_lane] + a[current_lane] * sin(current_time + delta[current_lane]);
            double dist_step = min(speed * dt, d - current_dist);
            current_dist += dist_step;
            current_time += dist_step / speed;
        }
    }
    
    cout << fixed << current_time << "\n";
    cout << switches.size() << "\n";
    for(auto& sw : switches){
        cout << sw.first << " " << sw.second << "\n";
    }
    
    return 0;
}
```
