Looking at this problem, I need to:

1. Paint bridges between islands with colors 1 (white) or 2 (black)
2. Each island with degree > 1 must have at least one bridge of each color
3. Islands with degree 1 can have either color

Key observations:
- If an island has degree 1, it can have either color
- If an island has degree 0, there's no constraint
- If an island has degree ≥ 2, it must have at least one edge colored 1 and at least one colored 2
- If an island has degree 1, it's impossible to satisfy the constraint (needs both colors but only has 1 edge) - **WAIT**, re-reading: "of course if an island has more than one bridge" - so degree 1 islands don't need both colors!

So the constraint is: **if degree ≥ 2, need at least one edge of each color**.

This is a 2-coloring problem on edges with constraints. Let me think about it differently:
- For each island with degree ≥ 2, at least one incident edge must be color 1 and at least one must be color 2.

A greedy approach:
1. First, check if any island has degree ≥ 2. If none exist, just color everything 1.
2. Try to color edges such that each island with degree ≥ 2 has both colors.

Actually, let's think simpler:
- If an island has degree 1, its single edge can be any color
- If an island has degree ≥ 2, edges incident to it must include both colors

A simple greedy strategy:
- For each edge, maintain which color it should be
- For islands with degree ≥ 2, ensure at least one neighbor is colored 1 and one is colored 2
- Start by coloring edges from high-degree vertices

Let me implement a solution:
1. Build adjacency list with edge indices
2. Check feasibility: for each island with degree 1, we don't care. For each with degree ≥ 2, we need both colors possible.
3. Try a greedy coloring: for each island with degree ≥ 2, assign colors to its edges ensuring both colors appear.

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    
    vector<vector<int>> adj(n + 1);
    vector<pair<int,int>> edges;
    map<pair<int,int>, int> edgeIdx;
    
    for(int i = 1; i <= n; i++){
        int j;
        while(cin >> j && j != 0){
            adj[i].push_back(j);
        }
    }
    
    // Build edge list
    for(int i = 1; i <= n; i++){
        for(int j : adj[i]){
            if(i < j){
                int idx = edges.size();
                edges.push_back({i, j});
                edgeIdx[{i, j}] = idx;
            }
        }
    }
    
    vector<int> color(edges.size(), 0); // 0 = uncolored, 1 = white, 2 = black
    
    // For each node with degree >= 2, ensure both colors
    for(int i = 1; i <= n; i++){
        if(adj[i].size() >= 2){
            bool has1 = false, has2 = false;
            vector<int> edgeIndices;
            
            for(int j : adj[i]){
                int u = i, v = j;
                if(u > v) swap(u, v);
                int idx = edgeIdx[{u, v}];
                edgeIndices.push_back(idx);
                if(color[idx] == 1) has1 = true;
                if(color[idx] == 2) has2 = true;
            }
            
            if(!has1 && edgeIndices.size() > 0){
                color[edgeIndices[0]] = 1;
                has1 = true;
            }
            if(!has2 && edgeIndices.size() > 1){
                color[edgeIndices[1]] = 2;
                has2 = true;
            }
            
            if(!has1 || !has2){
                cout << "No solution\n";
                return 0;
            }
        }
    }
    
    // Fill remaining with 1
    for(int i = 0; i < color.size(); i++){
        if(color[i] == 0) color[i] = 1;
    }
    
    // Output
    for(int i = 1; i <= n; i++){
        for(int j : adj[i]){
            int u = i, v = j;
            if(u > v) swap(u, v);
            int idx = edgeIdx[{u, v}];
            cout << color[idx] << " ";
        }
        cout << "0\n";
    }
    
    return 0;
}
```
