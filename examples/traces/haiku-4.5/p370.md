Key insight: A straight line from Petya's position (0,0) to a cell (i,j) passes through multiple cells. The number of cells on a line from (0,0) to (i,j) depends on the GCD of i and j.

Specifically, the line from (0,0) to (i,j) passes through exactly gcd(i,j) + 1 cells (including the starting point). But we don't count (0,0) itself since Petya is there.

The cells that lie on the line from (0,0) to (i,j) are at positions (k*i/g, k*j/g) where g = gcd(i,j) and k = 0, 1, ..., g.

Actually: count Euler's totient function! For each row i from 1 to N-1, count coprime j's with M-1. For each column j from 1 to M-1, count coprime i's with N-1.

```cpp
#include<bits/stdc++.h>
using namespace std;

long long gcd(long long a, long long b) {
    while(b) {
        long long temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

int main(){
    long long N, M;
    cin >> N >> M;
    
    long long count = 0;
    
    // Count all primitive directions
    for(long long i = 1; i < N; i++) {
        for(long long j = 1; j < M; j++) {
            if(gcd(i, j) == 1) {
                count++;
            }
        }
    }
    
    // Add horizontal directions (j varies, i=0 means same row)
    for(long long j = 1; j < M; j++) {
        if(gcd(0, j) == j) { // gcd(0,j) = j, so only j=1 counts as primitive
            count++;
        }
    }
    
    // Add vertical directions (i varies, j=0 means same column)  
    for(long long i = 1; i < N; i++) {
        if(gcd(i, 0) == i) { // gcd(i,0) = i, so only i=1 counts as primitive
            count++;
        }
    }
    
    cout << count << endl;
    return 0;
}
```

Wait, that's not efficient for large N,M. Let me use Euler's totient function approach properly.

```cpp
#include<bits/stdc++.h>
using namespace std;

long long gcd(long long a, long long b) {
    while(b) { long long t=b; b=a%b; a=t; }
    return a;
}

int main(){
    long long N, M;
    cin >> N >> M;
    
    long long count = 0;
    for(long long i = 1; i < N; i++) {
        for(long long j = 1; j < M; j++) {
            if(gcd(i, j) == 1) count++;
        }
    }
    count += (M - 1) + (N - 1);
    
    cout << count << endl;
    return 0;
}
```
