- S1 and S2 are both permutations of the first N letters
- I need to find a string containing both as substrings
- Every substring of length N+1 must have exactly one "extra" letter (or be missing one from the full set)

Key insight: If I concatenate S1 and S2 with maximum overlap, I might get a valid answer. Let me check if that works:
- Try overlapping S1 and S2 by 1, 2, ..., N-1 characters
- For each overlap, check if the resulting string is hyper almost permutative
- Return the shortest valid one

- Overlap by 0: "ABCACB" (length 6)
- Overlap by 1: "ABCACB"? No wait, overlap by 1 would be "ABCACB" if the last char of S1 matches first of S2... "C" vs "A" - no match
- Overlap by 2: "ABCB"? "BC" vs "AC" - no match

```cpp
#include <bits/stdc++.h>
using std/str;

int main(){
    int N;
    string S1, S2;
    cin >> N >> S1 >> S2;
    
    // Check if a string is almost permutative for length N
    auto isAlmostPermutative = [N](const string& s) -> bool {
        if(s.length() != N + 1) return false;
        
        // Try removing each character
        for(int i = 0; i < s.length(); i++) {
            set<char> chars;
            for(int j = 0; j < s.length(); j++) {
                if(i != j) {
                    chars.insert(s[j]);
                }
            }
            // Check if it's a permutation of first N letters
            if(chars.size() == N) {
                bool valid = true;
                for(int j = 0; j < N; j++) {
                    if(chars.find('A' + j) == chars.end()) {
                        valid = false;
                        break;
                    }
                }
                if(valid) return true;
            }
        }
        return false;
    };
    
    // Check if string is hyper almost permutative
    auto isHyperAlmostPermutative = [N, &isAlmostPermutative](const string& s) -> bool {
        if(s.length() < N + 1) return true; // vacuously true
        for(int i = 0; i + N + 1 <= (int)s.length(); i++) {
            if(!isAlmostPermutative(s.substr(i, N + 1))) {
                return false;
            }
        }
        return true;
    };
    
    // Try different overlaps
    string best = S1 + S2; // worst case: no overlap
    
    for(int overlap = 1; overlap < N; overlap++) {
        // Check if last 'overlap' chars of S1 match first 'overlap' chars of S2
        bool canOverlap = true;
        for(int i = 0; i < overlap; i++) {
            if(S1[N - overlap + i] != S2[i]) {
                canOverlap = false;
                break;
            }
        }
        
        if(canOverlap) {
            string candidate = S1 + S2.substr(overlap);
            if(isHyperAlmostPermutative(candidate)) {
                if(candidate.length() < best.length()) {
                    best = candidate;
                }
            }
        }
    }
    
    cout << best << endl;
    
    return 0;
}
```
