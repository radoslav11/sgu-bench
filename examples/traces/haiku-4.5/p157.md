- The winning condition is: all cards are in order (A, 2, 3, ..., K) and empty is at the end
- We can move a card to an empty cell only if the card is directly to the left and is the next card in sequence

using namespace std;

int n;
map<vector<int>, bool> memo;

// State: vector where 0 = empty, 1..n = card values (n=King, n-1=Queen, etc)
// Goal state: [1, 2, 3, ..., n, 0]

bool canWin(vector<int> state) {
    vector<int> goal(n + 1);
    for(int i = 0; i < n; i++) goal[i] = i + 1;
    goal[n] = 0;
    
    if(state == goal) return true;
    
    if(memo.count(state)) return memo[state];
    
    queue<vector<int>> q;
    set<vector<int>> visited;
    q.push(state);
    visited.insert(state);
    
    while(!q.empty()) {
        vector<int> curr = q.front();
        q.pop();
        
        if(curr == goal) return memo[state] = true;
        
        // Find empty cell (value 0)
        int emptyPos = -1;
        for(int i = 0; i < n; i++) {
            if(curr[i] == 0) {
                emptyPos = i;
                break;
            }
        }
        
        // Can we move from left?
        if(emptyPos > 0) {
            int leftCard = curr[emptyPos - 1];
            // We can move if leftCard is the card that should go in emptyPos

}
