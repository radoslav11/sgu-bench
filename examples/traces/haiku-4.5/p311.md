Looking at this problem, I need to:

1. Track ice-cream inventory with different prices
2. For ARRIVE queries, add n pieces at price c
3. For BUY queries, check if the n cheapest pieces cost ≤ t total
4. If yes, remove those pieces and output HAPPY; otherwise output UNHAPPY

The key insight is to maintain ice-cream sorted by price so I can always access the cheapest pieces first.

Let me trace through the example:
- ARRIVE 1 1: Have 1 piece at price 1
- ARRIVE 10 200: Have 1 piece at price 1, 10 pieces at price 200
- BUY 5 900: Need 5 cheapest pieces. Take 1 piece at price 1 (total: 1), then 4 pieces at price 200 (total: 801). Total = 802 ≤ 900. HAPPY. Inventory: 6 pieces at price 200
- BUY 5 900: Need 5 cheapest pieces. Take 5 pieces at price 200. Total = 1000 > 900. UNHAPPY. Inventory unchanged: 6 pieces at price 200
- BUY 5 1000: Need 5 cheapest pieces. Take 5 pieces at price 200. Total = 1000 ≤ 1000. HAPPY. Inventory: 1 piece at price 200

I'll use a multiset or priority queue to keep track of prices. A multiset would work well here since I need to iterate through the cheapest items.

Actually, I think a simpler approach using a multiset (which keeps elements sorted) would work:
- For ARRIVE: insert n copies of price c into the multiset
- For BUY: iterate through the multiset from smallest price, accumulating cost until we have n pieces or exceed t

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    multiset<long long> prices; // stores prices of all ice-cream pieces
    
    string query;
    while(cin >> query) {
        if(query == "ARRIVE") {
            long long n, c;
            cin >> n >> c;
            for(long long i = 0; i < n; i++) {
                prices.insert(c);
            }
        }
        else { // BUY
            long long n, t;
            cin >> n >> t;
            
            long long cost = 0;
            long long count = 0;
            vector<long long> bought;
            
            // Try to buy n cheapest pieces
            for(auto it = prices.begin(); it != prices.end() && count < n; ++it) {
                cost += *it;
                bought.push_back(*it);
                count++;
            }
            
            if(count == n && cost <= t) {
                // Purchase successful
                cout << "HAPPY\n";
                // Remove the bought items from prices
                for(long long price : bought) {
                    prices.erase(prices.find(price));
                }
            }
            else {
                // Purchase failed
                cout << "UNHAPPY\n";
            }
        }
    }
    
    return 0;
}
```

Wait, let me reconsider. The multiset approach with inserting individual elements might be slow if n is large (up to 10^6). Let me use a map instead to store count of each price:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    map<long long, long long> inventory; // price -> count
    
    string query;
    while(cin >> query) {
        if(query == "ARRIVE") {
            long long n, c;
            cin >> n >> c;
            inventory[c] += n;
        }
        else { // BUY
            long long n, t;
            cin >> n >> t;
            
            long long cost = 0;
            long long count = 0;
            vector<pair<long long, long long>> toBuy; // price, quantity
            
            // Try to buy n cheapest pieces
            for(auto& [price, avail] : inventory) {
                long long needed = n - count;
                long long take = min(needed, avail);
                cost += take * price;
                toBuy.push_back({price, take});
                count += take;
                if(count == n) break;
            }
            
            if(count == n && cost <= t) {
                // Purchase successful
                cout << "HAPPY\n";
                // Remove the bought items
                for(auto [price, qty] : toBuy) {
                    inventory[price] -= qty;
                    if(inventory[price] == 0) {
                        inventory.erase(price);
                    }
                }
            }
            else {
                // Purchase failed
                cout << "UNHAPPY\n";
            }
        }
    }
    
    return 0;
}
```

This solution uses a map to efficiently track inventory by price. The map keeps prices sorted automatically. For each BUY query, we iterate through prices in ascending order and accumulate the cost, then if successful, we remove the items.
