Looking at this problem, I need to:

1. Find the minimum cost to make all flights belong to a single airline
2. The final airline should be able to reach from any town to any other town
3. Output the chosen airline, number of operations, and which flights to transfer

Key insights:
- We need to keep the graph connected
- We want to choose one airline and transfer flights from other airlines to it
- For each airline, we can calculate the cost of making it the sole airline
- We need to ensure connectivity is maintained, which means we need a spanning tree that connects all towns
- The optimal solution is to pick an airline and then add the minimum cost flights from other airlines to maintain connectivity

Strategy:
1. For each possible final airline (1 to M), calculate the minimum cost to make the graph connected using only that airline plus transferred flights
2. We need to find a minimum spanning tree that connects all N towns
3. For each airline as the "final" airline, we start with its edges and greedily add minimum cost edges from other airlines to ensure connectivity
4. Pick the airline with minimum total transfer cost

Let me think differently:
- We need all N towns connected
- We pick one airline as base
- We need to add minimum cost edges to make it connected
- This is like finding a minimum spanning tree, but we prefer edges already belonging to the chosen airline (cost 0) over edges from other airlines (cost = given tax)

Algorithm:
1. Try each airline as the final one
2. For each airline, use Kruskal's algorithm where edges of that airline have weight 0, and edges of other airlines have their given weight
3. Find the minimum spanning tree cost and track which edges to transfer
4. Pick the airline with minimum cost

```cpp
#include<bits/stdc++.h>
using namespace std;

struct Edge {
    int u, v, cost, airline, idx;
};

class DSU {
public:
    vector<int> parent, rank;
    
    DSU(int n) {
        parent.resize(n + 1);
        rank.resize(n + 1, 0);
        for(int i = 0; i <= n; i++) parent[i] = i;
    }
    
    int find(int x) {
        if(parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }
    
    bool unite(int x, int y) {
        int px = find(x), py = find(y);
        if(px == py) return false;
        if(rank[px] < rank[py]) swap(px, py);
        parent[py] = px;
        if(rank[px] == rank[py]) rank[px]++;
        return true;
    }
};

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, M, K;
    cin >> N >> M >> K;
    
    vector<Edge> edges(K);
    for(int i = 0; i < K; i++) {
        cin >> edges[i].u >> edges[i].v >> edges[i].airline >> edges[i].cost;
        edges[i].idx = i + 1;
    }
    
    int bestAirline = -1;
    long long minCost = LLONG_MAX;
    vector<int> bestTransfers;
    
    // Try each airline as the final one
    for(int airline = 1; airline <= M; airline++) {
        // Create edge list with costs: 0 for this airline, original cost for others
        vector<pair<long long, int>> sortedEdges;
        for(int i = 0; i < K; i++) {
            long long cost = (edges[i].airline == airline) ? 0 : edges[i].cost;
            sortedEdges.push_back({cost, i});
        }
        
        sort(sortedEdges.begin(), sortedEdges.end());
        
        DSU dsu(N);
        long long totalCost = 0;
        vector<int> transfers;
        int edgesAdded = 0;
        
        for(auto [cost, idx] : sortedEdges) {
            if(dsu.unite(edges[idx].u, edges[idx].v)) {
                totalCost += cost;
                if(edges[idx].airline != airline) {
                    transfers.push_back(edges[idx].idx);
                }
                edgesAdded++;
                if(edgesAdded == N - 1) break;
            }
        }
        
        if(edgesAdded == N - 1 && totalCost < minCost) {
            minCost = totalCost;
            bestAirline = airline;
            bestTransfers = transfers;
        }
    }
    
    cout << minCost << " " << bestAirline << " " << bestTransfers.size() << "\n";
    for(int idx : bestTransfers) {
        cout << idx << "\n";
    }
    
    return 0;
}
```
