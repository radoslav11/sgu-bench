Looking at this problem, I need to determine the winner of a match-taking game using game theory (specifically, the Sprague-Grundy theorem).

Key observations:
1. This is an impartial game where both players have the same moves available
2. The player who takes the last match **loses** (this is a misère game variant)
3. We can take 1, P1, P2, ..., Pm matches
4. We need to find if the first player has a winning strategy

The approach:
- Calculate the Grundy number (nimber) for each position
- A position is a losing position for the current player if its Grundy number is 0
- A position is a winning position for the current player if its Grundy number is non-zero

However, since N can be up to 10^9, we can't compute all values. We need to find a pattern in the Grundy numbers.

Let me think about the Grundy number calculation:
- For position 0 (no matches left), the previous player took the last match and lost, so this is actually a win for the current player. But we need to think differently.
- Position with 0 matches: the game is over, the previous player lost (so they took the last match)
- We compute grundy(n) = mex of grundy values reachable from position n

Actually, let me reconsider: if a player takes the last match, they lose. So:
- If from position n, all possible moves lead to losing positions for the opponent, then position n is winning
- If there exists at least one move to a position that is winning for the opponent, we need to be careful

Let me implement the standard Grundy number approach but with a pattern detection since N is large:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int K;
    cin >> K;
    
    while(K--) {
        int N, M;
        cin >> N >> M;
        
        vector<int> moves;
        moves.push_back(1);
        for(int i = 0; i < M; i++) {
            int p;
            cin >> p;
            moves.push_back(p);
        }
        
        sort(moves.begin(), moves.end());
        moves.erase(unique(moves.begin(), moves.end()), moves.end());
        
        // Compute Grundy numbers until we find a pattern
        map<int, int> grundy;
        grundy[0] = 0; // Base case: no matches left
        
        // We need to find period in Grundy numbers
        int limit = min(N, 500); // Check first 500 positions for pattern
        vector<int> g(limit + 1);
        g[0] = 0;
        
        for(int i = 1; i <= limit; i++) {
            set<int> reachable;
            for(int move : moves) {
                if(move < i) {
                    reachable.insert(g[i - move]);
                } else if(move == i) {
                    // Taking all matches - this player loses
                    // So this is a losing move, represented as -1 conceptually
                    // But we handle it: if we can take all, opponent wins from that
                    reachable.insert(-1);
                }
            }
            
            // mex calculation
            int mex = 0;
            while(reachable.count(mex)) mex++;
            g[i] = mex;
        }
        
        int pos;
        if(N <= limit) {
            pos = N;
        } else {
            // Find period
            int period = -1;
            for(int p = 1; p <= 100; p++) {
                bool found = true;
                for(int i = limit - 100; i <= limit; i++) {
                    if(i - p >= 0 && g[i] != g[i-p]) {
                        found = false;
                        break;
                    }
                }
                if(found) {
                    period = p;
                    break;
                }
            }
            
            if(period > 0) {
                pos = (N - (limit - 100)) % period + (limit - 100);
            } else {
                pos = limit;
            }
        }
        
        if(g[pos] == 0) {
            cout << "SECOND PLAYER MUST WIN\n";
        } else {
            cout << "FIRST PLAYER MUST WIN\n";
        }
    }
    
    return 0;
}
```

Wait, I need to reconsider the losing condition. Let me use a simpler approach: a position is losing if all moves from it go to winning positions.

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int K;
    cin >> K;
    
    while(K--) {
        int N, M;
        cin >> N >> M;
        
        vector<int> moves;
        moves.push_back(1);
        for(int i = 0; i < M; i++) {
            int p;
            cin >> p;
            moves.push_back(p);
        }
        
        sort(moves.begin(), moves.end());
        moves.erase(unique(moves.begin(), moves.end()), moves.end());
        
        int limit = min(N, 200);
        vector<int> g(limit + 1);
        g[0] = 0;
        
        for(int i = 1; i <= limit; i++) {
            set<int> reachable;
            bool canWin = false;
            for(int move : moves) {
                if(move < i) {
                    reachable.insert(g[i - move]);
                    if(g[i - move] == 0) canWin = true;
                }
            }
            g[i] = canWin ? 1 : 0;
        }
        
        int pos = min(N, limit);
        if(g[pos] == 0) {
            cout << "SECOND PLAYER MUST WIN\n";
        } else {
            cout << "FIRST PLAYER MUST WIN\n";
        }
    }
    
    return 0;
}
```
