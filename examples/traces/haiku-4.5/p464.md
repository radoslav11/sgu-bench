Looking at this problem, I need to find the equilibrium strategy for two competing businessmen trying to get N approvals first.

Key observations:
1. Each day, a businessman can bribe one bureaucrat with amount `b`
2. Probability of approval given bribe `b` is `1 - 0.99^((1-F)*b)`
3. The first to get N approvals wins and gets profit V
4. Both play optimally knowing the other's strategy
5. This is a game theory problem requiring dynamic programming

The state is `(a1, a2)` where `a1` and `a2` are the number of approvals each businessman has. I need to find the expected value for each player from each state.

For each state, I need to find the optimal bribe amounts that form a Nash equilibrium. Given the complexity and that there's a unique equilibrium, I'll use backward induction with numerical optimization.

Let me think about the approach:
- Use DP with memoization for states (a1, a2)
- For each state, both players choose optimal bribes simultaneously
- The optimal bribe can be found by considering that in equilibrium, each player maximizes their expected payoff
- Simulate the day: both make their bribe choices, outcomes determined, move to next state

Since we need equilibrium and the problem guarantees uniqueness, I'll use an iterative approach to find optimal bribes at each state.

#include <bits/stdc++.h>
using namespace std;

int N;
double F1, F2, V;
map<pair<int,int>, pair<double,double>> memo;

// Returns (prob of player 1 winning, prob of player 2 winning) from state (a1, a2)
pair<double,double> solve(int a1, int a2);

double getProb(double F, double b) {
    if (b < 1e-9) return 0.01;
    return 1.0 - pow(0.99, (1.0 - F) * b);
}

pair<double,double> solve(int a1, int a2) {
    if (a1 == N) return {1.0, 0.0};
    if (a2 == N) return {0.0, 1.0};
    
    auto key = make_pair(a1, a2);
    if (memo.count(key)) return memo[key];
    
    // Binary search or numerical optimization for optimal bribes
    // For now, use a grid search to find approximate Nash equilibrium
    
    double bestEV1 = -1e9, bestEV2 = -1e9;
    double optB1 = 0, optB2 = 0;
    
    // Grid search for b1 and b2
    for (double b1 = 0; b1 <= 100; b1 += 0.5) {
        for (double b2 = 0; b2 <= 100; b2 += 0.5) {
            double p1 = getProb(F1, b1);  // prob player 1 gets approval
            double p2 = getProb(F2, b2);  // prob player 2 gets approval
            
            double ev1, ev2;
            
            // Four outcomes: (1 succeeds, 2 succeeds), (1 succeeds, 2 fails), etc.
            double case11 = p1 * p2; // both succeed same day
            double case10 = p1 * (1 - p2); // 1 succeeds, 2 fails
            double case01 = (1 - p1) * p2; // 1 fails, 2 succeeds
            double case00 = (1 - p1) * (1 - p2); // both fail
            
            auto [ev1_11, ev2_11] = solve(a1 + 1, a2 + 1);
            auto [ev1_10, ev2_10] = solve(a1 + 1, a2);
            auto [ev1_01, ev2_01] = solve(a1, a2 + 1);
            auto [ev1_00, ev2_00] = solve(a1, a2);
            
            ev1 = case11 * 0.5 * ev1_11 + case10 * ev1_10 + case01 * ev1_01 + case00 * ev1_00 - b1;
            ev2 = case11 * 0.5 * ev2_11 + case10 * ev2_10 + case01 * ev2_01 + case00 * ev2_00 - b2;
            
            if (ev1 > bestEV1 || (ev1 == bestEV1 && ev2 > bestEV2)) {
                bestEV1 = ev1;
                bestEV2 = ev2;
                optB1 = b1;
                optB2 = b2;
            }
        }
    }
    
    double p1 = getProb(F1, optB1);
    double p2 = getProb(F2, optB2);
    
    auto [ev1_11, ev2_11] = solve(a1 + 1, a2 + 1);
    auto [ev1_10, ev2_10] = solve(a1 + 1, a2);
    auto [ev1_01, ev2_01] = solve(a1, a2 + 1);
    auto [ev1_00, ev2_00] = solve(a1, a2);
    
    double res1 = p1 * p2 * 0.5 * ev1_11 + p1 * (1-p2) * ev1_10 + (1-p1) * p2 * ev1_01 + (1-p1) * (1-p2) * ev1_00 - optB1;
    double res2 = p1 * p2 * 0.5 * ev2_11 + p1 * (1-p2) * ev2_10 + (1-p1) * p2 * ev2_01 + (1-p1) * (1-p2) * ev2_00 - optB2;
    
    return memo[key] = {res1, res2};
}
