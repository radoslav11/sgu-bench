Looking at this problem, I need to find the minimum number of bombs to destroy all rocks, where each bomb destroys all rocks in its row and column.

Key observations:
1. Each bomb at position (i,j) destroys all rocks in row i and column j
2. This is a covering problem - I need to find the minimum set of bombs (cells) such that every rock is covered by at least one bomb
3. With n,m ≤ 25, the grid can have at most 625 cells and potentially many rocks
4. Since we need to cover all rocks optimally, this is a minimum set cover variant

The approach:
- Since n,m ≤ 25, I can use bitmask DP or greedy approaches
- Actually, this is equivalent to: find minimum number of (row, column) pairs such that every rock at position (r,c) has either its row or column (or both) selected
- This is a classic hitting set/vertex cover problem on a bipartite graph

Better approach:
- For small constraints, I can try a greedy approach or even brute force with bitmasks
- Greedy: repeatedly pick the bomb location that destroys the most remaining rocks
- Or use bitmask DP on rocks (if number of rocks is small)

Let me think differently:
- Find all rocks first
- For each possible bomb location, precompute which rocks it destroys
- Use BFS/DFS with bitmask to find minimum bombs needed to cover all rocks
- Since there can be up to 625 cells but likely much fewer rocks, use bitmask on rocks

Actually, simpler greedy approach might work:
- Greedy: while there are rocks left, pick the cell that destroys the most rocks, drop a bomb there, remove those rocks, repeat

Let me implement a solution using bitmask DP on rocks since the number of rocks might be manageable:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<string> grid(n);
    vector<pair<int,int>> rocks;
    
    for(int i = 0; i < n; i++){
        cin >> grid[i];
        for(int j = 0; j < m; j++){
            if(grid[i][j] == '*'){
                rocks.push_back({i, j});
            }
        }
    }
    
    int numRocks = rocks.size();
    
    if(numRocks == 0){
        cout << 0 << endl;
        return 0;
    }
    
    // For each cell (i,j), precompute which rocks it destroys (as a bitmask)
    vector<vector<long long>> destroys(n, vector<long long>(m, 0));
    
    for(int i = 0; i < n; i++){
        for(int j = 0; j < m; j++){
            long long mask = 0;
            for(int k = 0; k < numRocks; k++){
                if(rocks[k].first == i || rocks[k].second == j){
                    mask |= (1LL << k);
                }
            }
            destroys[i][j] = mask;
        }
    }
    
    // BFS to find minimum bombs
    map<long long, int> dist;
    queue<long long> q;
    
    long long allRocks = (1LL << numRocks) - 1;
    q.push(0);
    dist[0] = 0;
    
    while(!q.empty()){
        long long state = q.front();
        q.pop();
        
        if(state == allRocks){
            cout << dist[state] << endl;
            return 0;
        }
        
        int currDist = dist[state];
        
        // Try dropping a bomb at each cell
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
                long long newState = state | destroys[i][j];
                if(dist.find(newState) == dist.end()){
                    dist[newState] = currDist + 1;
                    q.push(newState);
                }
            }
        }
    }
    
    cout << 0 << endl;
    return 0;
}
```

This solution:
1. Reads the grid and finds all rock positions
2. For each cell, precomputes a bitmask of which rocks it would destroy
3. Uses BFS on states (represented as bitmasks of destroyed rocks)
4. Finds the minimum number of bombs (transitions) needed to reach the state where all rocks are destroyed

The time complexity is O(2^numRocks * n * m) which should be acceptable given the constraints.
