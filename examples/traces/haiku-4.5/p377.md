```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    int N, M;
    cin >> N >> M;
    
    // For a 2xM grid, we use DP
    // State: which cells in current column have students (0-3)
    // Transition: next column must ensure 2×2 has exactly 2
    
    if(N == 2) {
        // For 2xM: each 2x2 must have exactly 2
        // DP with states for column pattern
        vector<long long> dp(4, 0);
        dp[3] = 1; // both cells in first column
        
        for(int col = 1; col < M; col++) {
            vector<long long> ndp(4, 0);
            for(int prev = 0; prev < 4; prev++) {
                for(int curr = 0; curr < 4; curr++) {
                    int count = ((prev >> 0) & 1) + ((curr >> 0) & 1);
                    count += ((prev >> 1) & 1) + ((curr >> 1) & 1);
                    if(count == 2) {
                        ndp[curr] += dp[prev];
                    }
                }
            }
            dp = ndp;
        }
        
        long long ans = 0;
        for(int i = 0; i < 4; i++) ans += dp[i];
        cout << ans << endl;
    } else {
        // General case - would need more complex DP
        cout << 0 << endl;
    }
    
    return 0;
}
```

After more careful analysis, the answer is 6^(⌊N/2⌋×⌊M/2⌋):

```cpp
#include <bits/stdc++.h>
using namespace std;

long long power(long long a, long long b) {
    long long res = 1;
    for(long long i = 0; i < b; i++) res *= a;
    return res;
}

int main(){
    long long N, M;
    cin >> N >> M;
    cout << power(6, (N/2) * (M/2)) << endl;
    return 0;
}
```
