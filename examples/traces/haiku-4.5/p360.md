Looking at this problem, I need to:

1. Parse a B++ program with function definitions
2. Simulate a robot moving on a grid starting from position 'R'
3. Track which '*' cells are visited
4. Execute the entry point function `m()`

Key points:
- Robot starts facing up (direction 0)
- Commands: L (turn left), R (turn right), C (move forward)
- Functions can call other functions with arguments
- Arguments can be L, R, C or other arguments
- Need to handle function calls with argument substitution

```cpp
#include<bits/stdc++.h>
using namespace std;

int N, M;
vector<string> grid;
int robotX, robotY;
int dir = 0; // 0=up, 1=right, 2=down, 3=left
set<pair<int,int>> visited;
vector<pair<int,int>> visitOrder;

int dx[] = {-1, 0, 1, 0};
int dy[] = {0, 1, 0, -1};

map<string, pair<vector<string>, string>> functions; // name -> (args, body)

void executeCommand(char cmd);
void executeFunction(string fname, map<char,char>& args);

void executeCommand(char cmd) {
    if (cmd == 'L') {
        dir = (dir + 3) % 4;
    } else if (cmd == 'R') {
        dir = (dir + 1) % 4;
    } else if (cmd == 'C') {
        int nx = robotX + dx[dir];
        int ny = robotY + dy[dir];
        if (nx >= 0 && nx < N && ny >= 0 && ny < M) {
            robotX = nx;
            robotY = ny;
            if (grid[robotX][robotY] == '*' && visited.find({robotX, robotY}) == visited.end()) {
                visited.insert({robotX, robotY});
                visitOrder.push_back({robotX, robotY});
            }
        }
    }
}

void executeFunction(string fname, map<char,char>& args) {
    string body = functions[fname].second;
    
    int i = 0;
    while (i < body.length()) {
        if (body[i] == 'L' || body[i] == 'R' || body[i] == 'C') {
            char cmd = body[i];
            if (args.count(cmd)) {
                executeCommand(args[cmd]);
            } else {
                executeCommand(cmd);
            }
            i++;
        } else if (isalpha(body[i]) && islower(body[i])) {
            string funcName(1, body[i]);
            i++;
            // Skip '('
            i++;
            
            vector<char> callArgs;
            while (body[i] != ')') {
                if (body[i] == ',') {
                    i++;
                } else if (isalpha(body[i])) {
                    char arg = body[i];
                    if (args.count(arg)) {
                        callArgs.push_back(args[arg]);
                    } else {
                        callArgs.push_back(arg);
                    }
                    i++;
                } else {
                    i++;
                }
            }
            i++; // skip ')'
            
            map<char,char> newArgs;
            vector<string>& formalArgs = functions[funcName].first;
            for (int j = 0; j < callArgs.size(); j++) {
                newArgs[formalArgs[j][0]] = callArgs[j];
            }
            executeFunction(funcName, newArgs);
        } else {
            i++;
        }
    }
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> N >> M;
    grid.resize(N);
    
    for (int i = 0; i < N; i++) {
        cin >> grid[i];
        for (int j = 0; j < M; j++) {
            if (grid[i][j] == 'R') {
                robotX = i;
                robotY = j;
            }
        }
    }
    
    string program;
    char c;
    while (cin >> c) {
        program += c;
    }
    
    // Parse functions
    int i = 0;
    while (i < program.length()) {
        if (islower(program[i])) {
            string fname(1, program[i]);
            i++;
            i++; // skip '('
            
            vector<string> args;
            while (program[i] != ')') {
                if (isupper(program[i])) {
                    args.push_back(string(1, program[i]));
                    i++;
                } else {
                    i++;
                }
            }
            i++; // skip ')'
            i++; // skip '{'
            
            string body;
            int braceCount = 1;
            while (braceCount > 0) {
                if (program[i] == '{') braceCount++;
                else if (program[i] == '}') braceCount--;
                if (braceCount > 0) body += program[i];
                i++;
            }
            
            functions[fname] = {args, body};
        } else {
            i++;
        }
    }
    
    map<char,char> emptyArgs;
    executeFunction("m", emptyArgs);
    
    cout << visited.size() << "\n";
    for (auto& p : visitOrder) {
        cout << p.first + 1 << " " << p.second + 1 << "\n";
    }
    
    return 0;
}
```
