3. They need to meet at the same point (x, y) while always being at the same height
4. The mountain is defined by a polyline through points (1, y1), (2, y2), ..., (n, yn)

The key insight is that the alpinists move along the mountain edges. At any point in time, they must be at the same height.

So the question is: can they synchronize their movements such that they meet?

Key insight: If we denote the left alpinist's position as (x_l(t), y_l(t)) and right as (x_r(t), y_r(t)), they meet when x_l = x_r and y_l = y_r at some time t.

The constraint is: y_l(t) = y_r(t) at ALL times.

This means: at any point in the journey, if the left alpinist has progressed a certain fraction of their path, and the right alpinist has progressed the same fraction of their path, they must be at the same height.

So I need to check if there exists a common height value h such that:
- h is on the path from (1, y1) to (n, yn) 
- They can meet at height h

#include<bits/stdc++.h>
using namespace std;

int main(){
  ios_base::sync_with_stdio(false);
  cin.tie(nullptr);
  
  int n;
  cin >> n;
  vector<int> y(n);
  for(int i = 0; i < n; i++){
    cin >> y[i];
  }
  
  // Check all heights that appear in the polyline
  set<int> heights;
  for(int i = 0; i < n; i++){
    heights.insert(y[i]);
  }
  
  // For each segment, add intermediate heights
  for(int i = 0; i < n-1; i++){
    int minH = min(y[i], y[i+1]);
    int maxH = max(y[i], y[i+1]);
    for(int h = minH; h <= maxH; h++){
      heights.insert(h);
    }
  }
  
  // Check if there exists a height h where both paths can meet
  for(int h : heights){
    // Check if height h can be reached from both sides
    bool leftReach = false, rightReach = false;
    
    for(int i = 0; i < n-1; i++){
      int minH = min(y[i], y[i+1]);
      int maxH = max(y[i], y[i+1]);
      if(h >= minH && h <= maxH) leftReach = true;
    }
    
    for(int i = n-1; i > 0; i--){
      int minH = min(y[i], y[i-1]);
      int maxH = max(y[i], y[i-1]);
      if(h >= minH && h <= maxH) rightReach = true;
    }
    
    if(leftReach && rightReach){
      cout << "YES\n";
      return 0;
    }
  }
  
  cout << "NO\n";
  return 0;
}
