Looking at this problem, I need to:

1. Maximize total intensity of fuel
2. Subject to constraints:
   - Total volume ≤ A
   - Total cost ≤ B
   - Can buy any non-negative amount of each fuel type

For each fuel type i:
- Density: a_i (volume per kg)
- Cost: b_i (dollars per kg)
- Intensity: c_i (intensity per kg)

If I buy m_i kg of fuel type i:
- Volume used: m_i * a_i
- Cost: m_i * b_i
- Intensity gained: m_i * c_i

This is a linear programming problem. Since we want to maximize intensity subject to linear constraints, we should use a greedy approach or consider the efficiency metrics.

- Volume constraint: A

- Intensity per unit volume: c_i / a_i

using cmath;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, A, B;
    cin >> N >> A >> B;
    
    vector<tuple<double, int, int, int>> fuels; // (efficiency, a, b, c)
    
    for(int i = 0; i < N; i++){
        int a, b, c;
        cin >> a >> b >> c;
        
        // efficiency could be c/b (intensity per dollar)
        double eff = (b > 0) ? (double)c / b : 1e9;
        fuels.push_back({eff, a, b, c});
    }
    
    // Sort by efficiency (intensity per dollar) in descending order
    sort(fuels.rbegin(), fuels.rend());
    
    double volume_left = A;
    double budget_left = B;
    double total_intensity = 0;
    
    for(auto [eff, a, b, c] : fuels){
        if(volume_left <= 0 || budget_left <= 0) break;
        
        // How much can we buy of this fuel?
        // limited by volume_left/a and budget_left/b
        double max_by_volume = (a > 0) ? volume_left / a : 1e9;
        double max_by_budget = (b > 0) ? budget_left / b : 1e9;
        
        double amount = min(max_by_volume, max_by_budget);
        
        if(amount > 0){
            total_intensity += amount * c;
            volume_left -= amount * a;
            budget_left -= amount * b;
        }
    }
    
    cout << fixed << setprecision(6) << total_intensity << endl;
    
    return 0;
}
