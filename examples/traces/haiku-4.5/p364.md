Looking at this problem, I need to:

1. Simulate lemmings falling and walking on platforms
2. Try stopping different lemmings to see which strategy maximizes lemmings reaching home
3. Among solutions with max lemmings, minimize the time for the last lemming

Key observations:
- Lemmings start at (x, h) going right, appearing every s seconds
- They fall with velocity 1 cm/s or walk on platforms with velocity 1 cm/s
- Stopping a lemming reverses direction of all lemmings that touch it
- Goal is at (a, b)
- We can stop at most one lemming

Let me break down the solution:

```cpp
#include<bits/stdc++.h>
using namespace std;

struct Platform {
    int l, r, h;
};

int N, s, x, h, a, b, M;
vector<Platform> platforms;

// Find which platform a falling lemming at (px, ph) lands on
int findPlatform(int px, int ph) {
    for(int i = 0; i < M; i++) {
        if(platforms[i].h < ph && px >= platforms[i].l && px <= platforms[i].r) {
            return i;
        }
    }
    return -1;
}

// Simulate a single lemming starting at time startTime
// stoppedPos: (-1, -1) if no stopping, else (pos, platformIdx)
pair<bool, int> simulateLemming(int startTime, int stoppedPlatform, int stoppedPos) {
    int px = x, py = h;
    int dir = 1; // 1 for right, -1 for left
    int time = startTime;
    set<pair<int,int>> visited;
    
    while(true) {
        // Check if at home
        if(px == a && py == b) {
            return {true, time};
        }
        
        // Prevent infinite loops
        if(visited.count({px * 1000 + py, dir})) {
            return {false, time};
        }
        visited.insert({px * 1000 + py, dir});
        
        // Find platform below
        int platform = findPlatform(px, py);
        
        if(platform == -1) {
            // Fall
            int nextPlatform = -1;
            int fallDist = py;
            for(int i = 0; i < M; i++) {
                if(platforms[i].h < py && px >= platforms[i].l && px <= platforms[i].r) {
                    if(nextPlatform == -1 || platforms[i].h > platforms[nextPlatform].h) {
                        nextPlatform = i;
                    }
                }
            }
            
            if(nextPlatform == -1) {
                // Infinite fall
                return {false, time};
            }
            
            time += py - platforms[nextPlatform].h;
            py = platforms[nextPlatform].h;
        } else {
            // On platform, walk
            if(stoppedPlatform == platform && stoppedPos >= platforms[platform].l && stoppedPos <= platforms[platform].r) {
                if(dir == 1) {
                    if(px < stoppedPos) {
                        time += stoppedPos - px;
                        px = stoppedPos;
                        dir = -1;
                        continue;
                    }
                } else {
                    if(px > stoppedPos) {
                        time += px - stoppedPos;
                        px = stoppedPos;
                        dir = 1;
                        continue;
                    }
                }
            }
            
            int nextX;
            if(dir == 1) {
                nextX = min(px + 1, platforms[platform].r);
            } else {
                nextX = max(px - 1, platforms[platform].l);
            }
            
            if(nextX == platforms[platform].r && dir == 1) {
                px = nextX;
                time++;
                dir = -1;
            } else if(nextX == platforms[platform].l && dir == -1) {
                px = nextX;
                time++;
                dir = 1;
            } else {
                px = nextX;
                time++;
            }
        }
        
        if(time > 100000) return {false, time};
    }
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> N >> s;
    cin >> x >> h >> a >> b;
    cin >> M;
    
    platforms.resize(M);
    for(int i = 0; i < M; i++) {
        cin >> platforms[i].l >> platforms[i].r >> platforms[i].h;
    }
    
    int maxLemmings = 0;
    int minTime = INT_MAX;
    
    // Try no stopping
    int count = 0, lastTime = 0;
    for(int i = 0; i < N; i++) {
        auto [reached, time] = simulateLemming(i * s, -1, -1);
        if(reached) {
            count++;
            lastTime = max(lastTime, time);
        }
    }
    
    if(count > maxLemmings || (count == maxLemmings && lastTime < minTime)) {
        maxLemmings = count;
        minTime = lastTime;
    }
    
    cout << maxLemmings << " " << minTime << "\n";
    
    return 0;
}
```
