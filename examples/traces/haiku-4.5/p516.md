Looking at this problem, I need to:

1. Parse dates and track a schedule of guardians
2. Apply updates to the schedule at specific times
3. For each time interval, check if the schedule is valid (no guardian works more than T non-working hours per week)
4. Calculate the fraction of time [T1, T2] where the schedule is valid

Key observations:
- Working hours are Monday-Friday 9:00-18:00
- Non-working hours are everything else (evenings, nights, weekends)
- Need to track non-working hours per guardian per week
- A week is Monday 00:00 to Sunday 24:00
- Updates modify the schedule: a guardian works from Bi to Ei starting at time Ai

struct DateTime {
    int year, month, day, hour, minute;

        // Convert to minutes since 2009-01-01 00:00
        long long days = 0;
        int daysInMonth[] = {31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
        for(int m = 1; m < month; m++) days += daysInMonth[m-1];
        days += day - 1;
        return days * 24 * 60 + hour * 60 + minute;
    }
    
    int getDayOfWeek() const { // 0=Monday, 6=Sunday
        long long mins = toMinutes();
        return (mins / (24*60)) % 7;
    }
    
    bool isWorkingHour() const {
        int dow = getDayOfWeek();
        if(dow >= 5) return false; // Saturday, Sunday
        if(hour < 9 || hour >= 18) return false;
        return true;
    }
};
