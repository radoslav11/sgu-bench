Looking at this problem, I need to:

1. Find a subset of levers such that the product of their values modulo M is maximized
2. Output the maximum score and the levers that achieve it

This is a dynamic programming problem where I need to explore all possible products modulo M.

Let me think about the approach:
- Use DP where `dp[product]` = set of levers that give this product
- For each lever, I can either include it or not include it
- Track the best product found and which levers were used

Since M ≤ 1000 and N ≤ 10000, I can use a DP approach where I track which product values are achievable and how to achieve them.

Key insights:
- I need to try all possible subsets implicitly through DP
- For each product value mod M, I need to remember one way to achieve it
- The answer is the maximum product mod M that I can achieve

Let me implement this:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, M;
    cin >> N >> M;
    
    vector<int> a(N);
    for(int i = 0; i < N; i++){
        cin >> a[i];
    }
    
    // dp[product] = vector of lever indices that produce this product
    map<int, vector<int>> dp;
    dp[1] = vector<int>(); // empty set gives product 1
    
    // For each lever
    for(int i = 0; i < N; i++){
        map<int, vector<int>> new_dp = dp;
        
        // Try adding lever i to each existing product
        for(auto& [prod, levers] : dp){
            int new_prod = (prod * a[i]) % M;
            
            if(new_dp.find(new_prod) == new_dp.end()){
                // First time we see this product
                vector<int> new_levers = levers;
                new_levers.push_back(i + 1); // 1-indexed
                new_dp[new_prod] = new_levers;
            }
            // If we've seen it before, we can keep either solution
        }
        
        dp = new_dp;
    }
    
    // Find maximum product
    int max_prod = 0;
    vector<int> best_levers;
    
    for(auto& [prod, levers] : dp){
        if(prod > max_prod){
            max_prod = prod;
            best_levers = levers;
        }
    }
    
    cout << max_prod << "\n";
    
    if(best_levers.empty()){
        cout << "\n";
    } else {
        sort(best_levers.begin(), best_levers.end());
        for(int i = 0; i < best_levers.size(); i++){
            if(i > 0) cout << " ";
            cout << best_levers[i];
        }
        cout << "\n";
    }
    
    return 0;
}
```

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, M;
    cin >> N >> M;
    
    vector<int> a(N);
    for(int i = 0; i < N; i++){
        cin >> a[i];
    }
    
    // dp[product] = vector of lever indices that produce this product
    map<int, vector<int>> dp;
    dp[1] = vector<int>(); // empty set gives product 1
    
    // For each lever
    for(int i = 0; i < N; i++){
        vector<pair<int, vector<int>>> to_add;
        
        // Try adding lever i to each existing product
        for(auto& [prod, levers] : dp){
            int new_prod = (prod * a[i]) % M;
            vector<int> new_levers = levers;
            new_levers.push_back(i + 1);
            to_add.push_back({new_prod, new_levers});
        }
        
        for(auto& [prod, levers] : to_add){
            if(dp.find(prod) == dp.end()){
                dp[prod] = levers;
            }
        }
    }
    
    // Find maximum product
    int max_prod = 0;
    vector<int> best_levers;
    
    for(auto& [prod, levers] : dp){
        if(prod > max_prod){
            max_prod = prod;
            best_levers = levers;
        }
    }
    
    cout << max_prod << "\n";
    sort(best_levers.begin(), best_levers.end());
    for(int i = 0; i < best_levers.size(); i++){
        if(i > 0) cout << " ";
        cout << best_levers[i];
    }
    if(best_levers.size() > 0) cout << "\n";
    
    return 0;
}
```
