3. Each province has a capital (governor's headquarters) that must be reachable from all cities in the province through only cities in that province

Key insights:
- Since it's a tree, there's a unique path between any two cities
- Each province must be a connected subtree
- The capital of a province must be in that subtree (or reachable from it, but optimally in it)
- We need to partition the tree such that each component has size in [B, 3B]

Strategy:
- Use DFS to compute subtree sizes
- Greedily cut edges to create components of valid sizes
- For each component, pick any node as capital (or the root of the subtree)

void dfs_assign(int u, int parent) {
    int cur_size = 1;
    
    for(int v : adj[u]) {
        if(v != parent && province[v] == 0) {
            if(subtree_size[v] >= B && subtree_size[v] <= 3*B) {
                // Create new province from this subtree
                num_provinces++;
                int prov = num_provinces;
                capitals.push_back(v);
                function<void(int,int)> mark = [&](int x, int p) {
                    province[x] = prov;
                    for(int y : adj[x]) {
                        if(y != p && province[y] == 0) {
                            mark(y, x);
                        }
                    }
                };
                mark(v, u);
            } else if(subtree_size[v] < B) {
                // Include in current province
                cur_size += subtree_size[v];

                    province[x] = -1; // Mark for later assignment

                        if(y != p && province[y] == 0) {
                            mark(y, x);
                        }
                    }
                };
                mark(v, u);
            } else {
                // Recursively process
                dfs_assign(v, u);
            }
        }
    }
}
