    if(s.length() == 1) {
        return s[0] == '?' && currentPlayer == startingPlayer;
    }

    if(memo.find(key) != memo.end()) {

    char opponent = (currentPlayer == 'A') ? 'B' : 'A';
    bool result = false;
    
    // Option 1: Fill an empty stand with currentPlayer's production
