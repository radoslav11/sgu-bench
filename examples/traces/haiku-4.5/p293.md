Looking at this problem, I need to understand the game:

1. He writes 2n-1 letters total (alternating turns)
2. Qc moves on turns 2, 4, 6, ... (after He's moves 1, 2, 3, ...)
3. On Qc's turns, he can either swap two letters or pass
4. After He's moves 1, 3, 5, ... (odd moves), Qc needs the string to be a palindrome
5. After He's moves 2, 4, 6, ... (even moves), there's no constraint (but Qc moves before the next odd move)

Key insight: After He writes letter at position k (1-indexed), if k is odd, the string of length k must be a palindrome after Qc's move.

Let me trace through example 1:
- Initial: empty, Qc pass (move 1)
- He writes Q (pos 1): "Q", Qc does nothing (pass) - palindrome ✓
- He writes C (pos 2): "QC", Qc can swap to "CQ"
- He writes C (pos 3): "CQC", Qc swaps pos 1,2 → "QCC"→"CQC" - palindrome ✓
- And so on...

Strategy: Use BFS/DFS to explore all possible states. A state is: which letters are at which positions after processing some of He's moves. After each odd-numbered move by He, we need a palindrome.

Let me think of it differently:
- After He's move i (writing i-th letter), we're at a state
- If i is odd, this state must be a palindrome after Qc's move
- Qc can rearrange by swapping at most one pair before He's next move

I'll use BFS where each state tracks:
- Current position in string (how many letters He has written)
- Current arrangement of letters
- The moves Qc made

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    string s;
    cin >> n >> s;
    
    int total = 2*n - 1;
    
    // BFS: state = (position in string, current arrangement of letters)
    map<pair<int, string>, pair<int,int>> parent; // maps state to (prev_state, move)
    queue<pair<int, string>> q;
    
    string initial(total, '?');
    initial[0] = s[0];
    
    q.push({1, initial});
    parent[{1, initial}] = {-1, -1};
    
    bool found = false;
    pair<int, string> solution;
    
    while(!q.empty() && !found){
        auto [pos, current] = q.front();
        q.pop();
        
        if(pos == total + 1){
            found = true;
            solution = {pos-1, current};
            break;
        }
        
        // He writes next letter at position pos-1 (0-indexed)
        string next = current;
        next[pos-1] = s[pos-1];
        
        // Qc's turn: try all possible moves (including pass)
        vector<pair<int,int>> moves;
        moves.push_back({0, 0}); // pass
        
        for(int i = 0; i < pos; i++){
            for(int j = i+1; j < pos; j++){
                moves.push_back({i+1, j+1}); // 1-indexed
            }
        }
        
        for(auto [i, j] : moves){
            string candidate = next;
            if(i != 0){
                swap(candidate[i-1], candidate[j-1]);
            }
            
            // Check if valid: if pos is odd, must be palindrome
            bool valid = true;
            if(pos % 2 == 1){
                // Must be palindrome up to pos-1
                for(int k = 0; k < pos/2; k++){
                    if(candidate[k] != candidate[pos-1-k]){
                        valid = false;
                        break;
                    }
                }
            }
            
            if(valid){
                if(parent.find({pos, candidate}) == parent.end()){
                    parent[{pos, candidate}] = {-1, -1};
                    q.push({pos, candidate});
                    
                    if(pos == total){
                        found = true;
                        solution = {pos, candidate};
                        break;
                    }
                }
            }
        }
    }
    
    if(!found){
        cout << "He\n";
    } else {
        cout << "Qc\n";
        // Reconstruct solution
        // Trace back...
        cout << "0 0\n";
    }
    
    return 0;
}
```

Let me reconsider with a clearer approach using recursion with memoization:

```cpp
#include<bits/stdc++.h>
using namespace std;

int n;
string s;
map<pair<int,string>, vector<pair<int,int>>> memo;

vector<pair<int,int>> solve(int pos, string current){
    if(pos == 2*n){
        return {};
    }
    
    auto key = make_pair(pos, current);
    if(memo.count(key)) return memo[key];
    
    // Add next character
    current[pos] = s[pos];
    
    // Try all swaps
    for(int i = 0; i <= pos; i++){
        for(int j = i; j <= pos; j++){
            string next = current;
            if(i < j) swap(next[i], next[j]);
            
            bool ok = true;
            if((pos+1) % 2 == 1){
                int len = pos + 1;
                for(int k = 0; k < len/2; k++){
                    if(next[k] != next[len-1-k]){
                        ok = false;
                        break;
                    }
                }
            }
            
            if(ok){
                auto res = solve(pos+1, next);
                if(res.size() || pos+1 == 2*n){
                    auto ans = res;
                    ans.insert(ans.begin(), {i == j ? 0 : i+1, j+1});
                    return memo[key] = ans;
                }
            }
        }
    }
    
    return memo[key] = vector<pair<int,int>>();
}

int main(){
    cin >> n >> s;
    auto res = solve(0, string(2*n, '?'));
    if(res.empty()){
        cout << "He\n";
    } else {
        cout << "Qc\n";
        for(auto [a,b] : res) cout << a << " " << b << "\n";
    }
    return 0;
}
```
