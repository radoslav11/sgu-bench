2. Build the set H of all antichains (sets of pairwise incomparable elements)
3. Implement logical operations on H according to the given definitions
4. Parse and evaluate logical formulas to check if they're valid under all possible variable assignments

    memset(reach, false, sizeof(reach));
    for (int i = 0; i < N; i++) reach[i][i] = true;

        for (int j : adj[i]) reach[i][j] = true;
    }

            for (int j = 0; j < N; j++)
                if (reach[i][k] && reach[k][j]) reach[i][j] = true;
}

void generateH() {
    for (int mask = 0; mask < (1 << N); mask++) {
        set<int> s;
        for (int i = 0; i < N; i++)
            if (mask & (1 << i)) s.insert(i);
        
        bool isAntichain = true;
        for (int x : s)
            for (int y : s)
                if (x != y && reach[x][y]) {
                    isAntichain = false;
                    break;
                }
        
        if (isAntichain) {
            H.insert(s);
            hIndex[s] = hIndex.size();
        }
    }
}

set<int> Max(set<int> M) {
    set<int> result;
    for (int x : M) {
        bool maximal = true;
        for (int y : M)
            if (x != y && reach[x][y]) {
                maximal = false;
                break;
            }
        if (maximal) result.insert(x);
    }
    return result;
}

set<int> implies(set<int> a, set<int> b) {
    set<int> result;
    for (int x : b) {
        bool add = true;
        for (int y : a)
            if (reach[y][x]) {
                add = false;

    return result;

set<int> conjoin(set<int> a, set<int> b) {
    set<int> u;
    for (int x : a) u.insert(x);
    for (int x : b) u.insert(x);
    return Max(u);
}

        for (int y : a) if (reach[y][x]) add = true;
        for (int z : b) if (reach[z][x]) add = true;
        if (add) u.insert(x);

    return Max(u);

ExprPtr parseExpr(vector<string>& tokens, int& pos) {
    if (tokens[pos] == "1") { pos++; return make_shared<Const>(true); }
    if (tokens[pos] == "0") { pos++; return make_shared<Const>(false); }
    if (tokens[pos].length() == 1 && tokens[pos][0] >= 'A' && tokens[pos][0] <= 'Z') {
        return make_shared<Var>(tokens[pos++][0]);

    if (tokens[pos] == "(") {

        auto e = parseExpr(tokens, pos);
        pos++;
        return e;
    }
    return nullptr;
}
