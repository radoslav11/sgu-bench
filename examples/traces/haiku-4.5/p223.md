Looking at this problem, I need to count the number of ways to place k non-attacking kings on an n×n chessboard.

Key observations:
1. Two kings attack each other if they're on adjacent cells (horizontally, vertically, or diagonally)
2. This is a constraint satisfaction problem that can be solved with dynamic programming
3. With n ≤ 10, we can use bitmask DP where we process row by row

The approach:
- Use DP where we process the board row by row
- For each row, we use a bitmask to represent which cells have kings
- We need to ensure:
  - No two kings in the same row are adjacent
  - No two kings in adjacent rows are adjacent (including diagonally)
- Track the number of kings placed so far

Let me think about valid masks for a single row:
- A mask is valid if no two consecutive bits are set (no two kings adjacent in the same row)

For transitions between rows:
- If row i has mask A and row i+1 has mask B, they're compatible if:
  - No bit is set in both A and B (vertical adjacency)
  - No bit in A is adjacent to a bit in B (diagonal adjacency)

```cpp
#include <bits/stdc++.h>
using namespace std;

int n, k;
vector<int> validMasks;
map<pair<int,int>, bool> compatible;

bool isSingleRowValid(int mask) {
    // Check no two adjacent bits are set
    return (mask & (mask << 1)) == 0;
}

int popcount(int mask) {
    return __builtin_popcount(mask);
}

bool areCompatible(int mask1, int mask2) {
    // mask1 is in row i, mask2 is in row i+1
    // Check: no vertical adjacency
    if (mask1 & mask2) return false;
    
    // Check: no diagonal adjacency
    // Shift mask1 left and right to check diagonals
    if (mask1 & (mask2 << 1)) return false;
    if (mask1 & (mask2 >> 1)) return false;
    
    return true;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n >> k;
    
    // Generate all valid masks for a single row
    for (int mask = 0; mask < (1 << n); mask++) {
        if (isSingleRowValid(mask)) {
            validMasks.push_back(mask);
        }
    }
    
    // DP: dp[row][mask][numKings] = number of ways
    // We use map for flexibility
    map<tuple<int,int,int>, long long> dp;
    
    // Initialize first row
    for (int mask : validMasks) {
        int cnt = popcount(mask);
        if (cnt <= k) {
            dp[{0, mask, cnt}] = 1;
        }
    }
    
    // Fill DP for remaining rows
    for (int row = 0; row < n - 1; row++) {
        map<tuple<int,int,int>, long long> newDp;
        
        for (auto& [state, ways] : dp) {
            auto [r, prevMask, numKings] = state;
            if (r != row) continue;
            
            // Try all valid masks for next row
            for (int nextMask : validMasks) {
                if (!areCompatible(prevMask, nextMask)) continue;
                
                int nextKings = numKings + popcount(nextMask);
                if (nextKings <= k) {
                    newDp[{row + 1, nextMask, nextKings}] += ways;
                }
            }
        }
        
        for (auto& [state, ways] : newDp) {
            dp[state] = ways;
        }
    }
    
    // Sum up all configurations with exactly k kings at the last row
    long long answer = 0;
    for (auto& [state, ways] : dp) {
        auto [row, mask, numKings] = state;
        if (row == n - 1 && numKings == k) {
            answer += ways;
        }
    }
    
    cout << answer << endl;
    
    return 0;
}
```

Wait, let me reconsider the DP structure. It's cleaner to use a 3D DP array directly:

```cpp
#include <bits/stdc++.h>
using namespace std;

int n, k;
vector<int> validMasks;

bool isSingleRowValid(int mask) {
    return (mask & (mask << 1)) == 0;
}

int popcount(int mask) {
    return __builtin_popcount(mask);
}

bool areCompatible(int mask1, int mask2) {
    if (mask1 & mask2) return false;
    if (mask1 & (mask2 << 1)) return false;
    if (mask1 & (mask2 >> 1)) return false;
    return true;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    cin >> n >> k;
    
    for (int mask = 0; mask < (1 << n); mask++) {
        if (isSingleRowValid(mask)) {
            validMasks.push_back(mask);
        }
    }
    
    // dp[row][mask][count] = number of ways
    vector<vector<vector<long long>>> dp(n, vector<vector<long long>>(1 << n, vector<long long>(k + 1, 0)));
    
    for (int mask : validMasks) {
        int cnt = popcount(mask);
        if (cnt <= k) {
            dp[0][mask][cnt] = 1;
        }
    }
    
    for (int row = 0; row < n - 1; row++) {
        for (int prevMask : validMasks) {
            for (int prevCnt = 0; prevCnt <= k; prevCnt++) {
                if (dp[row][prevMask][prevCnt] == 0) continue;
                
                for (int nextMask : validMasks) {
                    if (!areCompatible(prevMask, nextMask)) continue;
                    
                    int nextCnt = prevCnt + popcount(nextMask);
                    if (nextCnt <= k) {
                        dp[row + 1][nextMask][nextCnt] += dp[row][prevMask][prevCnt];
                    }
                }
            }
        }
    }
    
    long long answer = 0;
    for (int mask : validMasks) {
        answer += dp[n - 1][mask][k];
    }
    
    cout << answer << endl;
    
    return 0;
}
```
