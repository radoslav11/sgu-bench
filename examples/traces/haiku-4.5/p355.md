Looking at this problem, I need to color numbers 1 to N such that if A is divisible by B, they have different colors, and I want to minimize the number of colors.

Let me think about this:
- If A is divisible by B (and A ≠ B), they must have different colors
- This is a graph coloring problem where there's an edge between A and B if one divides the other

Key insight: The minimum number of colors needed equals the length of the longest chain of divisibility. For example:
- 1 divides 2, 2 divides 4, 4 divides 8 - this is a chain of length 4
- So we need at least 4 colors

A good strategy is to color each number based on its "level" in the divisibility hierarchy. One way to think about it:
- Color a number based on how many prime factors it has (counting multiplicity)
- For example: 1 → 0 factors → color 1
             2, 3, 5, 7... → 1 factor → color 2
             4, 6, 9, 10... → 2 factors → color 3
             8, 12, 18... → 3 factors → color 4

The algorithm:
1. For each number from 1 to N, count the total number of prime factors (with multiplicity)
2. The color is this count + 1
3. The maximum count + 1 gives us M

```cpp
#include<bits/stdc++.h>
using namespace std;

int countPrimeFactors(int n) {
    int count = 0;
    
    // Count factor 2
    while (n % 2 == 0) {
        count++;
        n = n / 2;
    }
    
    // Count odd factors from 3 onwards
    for (int i = 3; i * i <= n; i += 2) {
        while (n % i == 0) {
            count++;
            n = n / i;
        }
    }
    
    // If n is still greater than 1, it's a prime factor
    if (n > 1) {
        count++;
    }
    
    return count;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N;
    cin >> N;
    
    vector<int> colors(N + 1);
    int maxColor = 0;
    
    for (int i = 1; i <= N; i++) {
        int primeFactorCount = countPrimeFactors(i);
        colors[i] = primeFactorCount + 1;
        maxColor = max(maxColor, colors[i]);
    }
    
    cout << maxColor << "\n";
    for (int i = 1; i <= N; i++) {
        cout << colors[i];
        if (i < N) cout << " ";
    }
    cout << "\n";
    
    return 0;
}
```
