Looking at this problem, I need to understand what's happening:

1. We have n wires, each with two endpoints (a_i, b_i), a reliability r_i, and a cost p_i
2. We solder wires in some order
3. When adding a wire creates a cycle, the least reliable wire in that cycle burns out (with ties broken by earlier soldering order)
4. We want to maximize the total cost of wires that remain

Key insight: The final graph must be a forest (no cycles). When we add a wire that would create a cycle, the weakest wire in that cycle is removed.

Strategy:
- We want to end up with a maximum cost forest
- This is similar to finding a maximum spanning forest, but we need to be careful about which wires burn out
- If we think about it differently: we want to keep wires with high cost, and sacrifice wires with low reliability when cycles form
- We should try to order wires so that when cycles form, the wires we remove have low cost

A greedy approach:
- Sort wires by cost in descending order (we want to keep high-cost wires)
- For each wire, try to add it. If it creates a cycle, remove the least reliable wire in that cycle
- But we need the right ordering...

Actually, let me reconsider: 
- We want maximum total cost of remaining wires
- When a cycle forms, the least reliable wire burns out
- So we want high-cost wires and high-reliability wires to survive

Better approach:
- Use a modified Kruskal's algorithm
- Sort wires by cost (descending) - we prefer to keep expensive wires
- When adding a wire would create a cycle, we remove the wire with minimum reliability in that cycle
- But for ties in reliability, earlier soldered wires burn first

This is essentially finding a maximum spanning forest where edges with lower reliability are preferred to be removed from cycles.

Let me implement this:

```cpp
#include<bits/stdc++.h>
using namespace std;

struct Wire {
    int a, b, r, p, idx;
};

class DSU {
public:
    vector<int> parent, rank;
    vector<int> minRelIdx; // index of minimum reliability wire in component
    
    DSU(int n) : parent(n), rank(n, 0), minRelIdx(n) {
        iota(parent.begin(), parent.end(), 0);
        iota(minRelIdx.begin(), minRelIdx.end(), 0);
    }
    
    int find(int x) {
        if (parent[x] != x) parent[x] = find(parent[x]);
        return parent[x];
    }
    
    bool unite(int x, int y, int wireIdx, const vector<Wire>& wires) {
        x = find(x);
        y = find(y);
        if (x == y) return false; // cycle detected
        
        if (rank[x] < rank[y]) swap(x, y);
        parent[y] = x;
        if (rank[x] == rank[y]) rank[x]++;
        
        int minIdx = minRelIdx[x];
        int minIdx2 = minRelIdx[y];
        if (wires[minIdx2].r < wires[minIdx].r || 
            (wires[minIdx2].r == wires[minIdx].r && minIdx2 < minIdx)) {
            minIdx = minIdx2;
        }
        minRelIdx[x] = minIdx;
        
        return true;
    }
};

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    
    vector<Wire> wires(n);
    for(int i = 0; i < n; i++) {
        cin >> wires[i].a >> wires[i].b >> wires[i].r >> wires[i].p;
        wires[i].idx = i + 1;
    }
    
    // Sort by cost descending (we want to keep expensive wires)
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int i, int j) {
        return wires[i].p > wires[j].p;
    });
    
    DSU dsu(1000001);
    vector<int> result;
    long long totalCost = 0;
    vector<bool> burned(n, false);
    
    for(int idx : order) {
        Wire& w = wires[idx];
        int u = dsu.find(w.a);
        int v = dsu.find(w.b);
        
        if(u == v) {
            // Would create cycle - this wire burns
            burned[idx] = true;
        } else {
            dsu.unite(u, v, idx, wires);
            result.push_back(w.idx);
            totalCost += w.p;
        }
    }
    
    cout << totalCost << "\n";
    for(int i = 0; i < result.size(); i++) {
        if(i > 0) cout << " ";
        cout << result[i];
    }
    cout << "\n";
    
    return 0;
}
```
