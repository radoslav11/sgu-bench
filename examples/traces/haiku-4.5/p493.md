Looking at this problem, I need to:

1. Understand that we have buildings represented as rectangles with corners at (L, 0) and (R, H)
2. Light sources are placed on top edges of buildings at some position (x, H)
3. A light source illuminates points if the line segment to those points doesn't pass through the interior of any building
4. We need to illuminate both left and right side edges of each building completely with minimum light sources

Key observations:
- Each building has a left side edge from (L, 0) to (L, H) and right side edge from (R, 0) to (R, H)
- A light source at position (x, y) on a building's top edge can illuminate points on side edges of other buildings if there's a clear line of sight
- For a building's left side at x-coordinate L, we need light sources that can "see" every point on that side
- We need to check which light source positions can illuminate which points

The strategy:
1. For each side edge of each building, determine which positions on other buildings' top edges can illuminate it
2. This becomes an interval covering problem - we need to cover all points on each side edge
3. For each side edge, find the range of x-coordinates on all top edges that can illuminate each point, then find minimum set of light sources

Let me think about when a light source at (x1, H1) can illuminate a point (x2, y2) on a side edge:
- The line segment between them shouldn't pass through building interiors
- We need to check geometric visibility

Actually, a simpler approach: for each side edge (either left or right of a building), we need to find which positions on top edges of any building can see any point on that side edge. This is a geometric visibility problem.

For practical purposes in a competitive programming context, I'll implement:
1. For each building's side edges
2. Try each possible position on top edges as a light source
3. Check which side edge points it can illuminate
4. Use greedy interval covering to find minimum lights

```cpp
#include<bits/stdc++.h>
using namespace std;

struct Building {
    double L, R, H;
};

bool canIlluminate(double sx, double sy, double px, double py, vector<Building>& buildings) {
    // Check if light at (sx, sy) can illuminate point (px, py)
    // Line segment from (sx, sy) to (px, py) shouldn't intersect building interiors
    
    double minX = min(sx, px), maxX = max(sx, px);
    double minY = min(sy, py), maxY = max(sy, py);
    
    for(auto& b : buildings) {
        // Check if building interior intersects the line segment
        // A building interior is (L, R) x (0, H) (excluding boundary)
        
        if(b.L >= maxX || b.R <= minX) continue; // No horizontal overlap
        
        // Check if line segment passes through building interior
        // Parametric form: (sx + t(px-sx), sy + t(py-sy)), t in [0,1]
        
        double x1 = sx, y1 = sy, x2 = px, y2 = py;
        double dx = x2 - x1, dy = y2 - y1;
        
        // Check intersection with (L, R) x (0, H)
        if(dx == 0) {
            // Vertical line
            if(x1 > b.L && x1 < b.R && ((y1 > 0 && y1 < b.H) || (y2 > 0 && y2 < b.H))) {
                double ymin = min(y1, y2), ymax = max(y1, y2);
                if(ymax > 0 && ymin < b.H) return false;
            }
        } else {
            // Check parametric intersection
            for(double t = 0; t <= 1; t += 0.01) {
                double x = x1 + t * dx, y = y1 + t * dy;
                if(x > b.L && x < b.R && y > 0 && y < b.H) return false;
            }
        }
    }
    return true;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int T;
    cin >> T;
    
    while(T--) {
        int N;
        cin >> N;
        
        vector<Building> buildings(N);
        for(int i = 0; i < N; i++) {
            cin >> buildings[i].L >> buildings[i].R >> buildings[i].H;
        }
        
        int lights = 0;
        
        // For each side edge that needs illumination
        for(int i = 0; i < N; i++) {
            // Left side at x = L, y in [0, H]
            // Right side at x = R, y in [0, H]
            
            // Check left side
            bool leftCovered = false;
            // Try placing light at right end of rightmost building that can see it
            for(int j = 0; j < N && !leftCovered; j++) {
                if(canIlluminate(buildings[j].R, buildings[j].H, buildings[i].L, buildings[i].H, buildings)) {
                    lights++;
                    leftCovered = true;
                }
            }
            
            // Check right side  
            bool rightCovered = false;
            for(int j = 0; j < N && !rightCovered; j++) {
                if(canIlluminate(buildings[j].L, buildings[j].H, buildings[i].R, buildings[i].H, buildings)) {
                    lights++;
                    rightCovered = true;
                }
            }
        }
        
        cout << lights << "\n";
    }
    
    return 0;
}
```
