Looking at this problem, I need to:

1. Find the minimum flow speed at source (node 1) such that all pipes marked with `Ci = 1` are completely filled
2. The flow must be valid (no accumulation at intermediate nodes)
3. Output the minimum speed and the flow through each pipe, or "Impossible"

Key insights:
- This is a flow problem where certain edges must be fully utilized (Ci = 1)
- We need to find minimum total flow from node 1 to node N
- Pipes with Ci = 1 must have flow equal to their capacity
- Pipes with Ci = 0 can have flow from 0 to their capacity
- Flow conservation must hold at each node (except source produces flow, sink consumes it)

Approach:
1. First, set all pipes with Ci = 1 to their full capacity
2. Check if flow conservation can be satisfied with these mandatory flows
3. Use a flow algorithm to find the minimum production speed

Let me use a simpler approach:
- Binary search or direct calculation on the total flow needed
- For a given total flow F, check if we can route it such that:
  - All Ci = 1 pipes are fully filled
  - Flow conservation holds at all nodes
  - All capacities are respected

Actually, a better approach:
1. Set all required pipes (Ci = 1) to their capacity
2. Calculate the net flow at each node (what's produced/consumed at that node)
3. Find the minimum flow F such that we can balance all nodes while respecting capacities

Let me implement using flow constraints:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, M;
    cin >> N >> M;
    
    vector<int> u(M), v(M), z(M), c(M);
    vector<vector<int>> adj(N+1);
    
    for(int i = 0; i < M; i++){
        cin >> u[i] >> v[i] >> z[i] >> c[i];
        adj[u[i]].push_back(i);
    }
    
    // Try to find minimum flow
    // Start with sum of all required capacities
    long long minFlow = 0;
    for(int i = 0; i < M; i++){
        if(c[i]) minFlow += z[i];
    }
    
    // Try flow values from minFlow down to find minimum
    for(long long flow = minFlow; flow <= 200000; flow++){
        vector<long long> f(M, 0);
        vector<long long> balance(N+1, 0);
        
        // Set required flows
        bool valid = true;
        for(int i = 0; i < M; i++){
            if(c[i]){
                f[i] = z[i];
                balance[u[i]] -= z[i];
                balance[v[i]] += z[i];
            }
        }
        
        balance[1] = flow;  // source produces
        balance[N] = -flow; // sink consumes
        
        // Check if we can balance with remaining capacity
        // Use greedy/network flow approach
        // For now, use DFS to check feasibility
        
        function<bool(int, long long)> dfs = [&](int node, long long needed) -> bool {
            if(node == N) return needed == 0;
            if(needed <= 0) return true;
            
            for(int idx : adj[node]){
                if(f[idx] < z[idx]){
                    long long can_add = z[idx] - f[idx];
                    long long add = min(can_add, needed);
                    f[idx] += add;
                    needed -= add;
                    if(needed == 0) return true;
                }
            }
            return needed == 0;
        };
        
        // Check flow conservation
        bool feasible = true;
        for(int i = 2; i < N; i++){
            long long in_flow = 0, out_flow = 0;
            for(int j = 0; j < M; j++){
                if(v[j] == i) in_flow += f[j];
                if(u[j] == i) out_flow += f[j];
            }
            if(in_flow != out_flow){
                feasible = false;
                break;
            }
        }
        
        if(feasible){
            cout << flow << "\n";
            for(int i = 0; i < M; i++){
                cout << f[i];
                if(i < M-1) cout << " ";
            }
            cout << "\n";
            return 0;
        }
    }
    
    cout << "Impossible\n";
    return 0;
}
```

After reflection, let me implement a cleaner max-flow based solution:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    int N, M;
    cin >> N >> M;
    
    vector<array<int,4>> edges(M);
    for(int i = 0; i < M; i++){
        cin >> edges[i][0] >> edges[i][1] >> edges[i][2] >> edges[i][3];
    }
    
    // Binary search on flow
    int left = 0, right = 100000;
    int ans = -1;
    vector<long long> ansFlow;
    
    while(left <= right){
        int mid = (left + right) / 2;
        vector<long long> flow(M);
        bool ok = true;
        
        for(int i = 0; i < M; i++){
            if(edges[i][3] == 1) flow[i] = edges[i][2];
            else flow[i] = 0;
        }
        
        vector<long long> balance(N+1);
        for(int i = 0; i < M; i++){
            balance[edges[i][0]] -= flow[i];
            balance[edges[i][1]] += flow[i];
        }
        
        balance[1] += mid;
        balance[N] -= mid;
        
        for(int i = 2; i < N; i++){
            if(balance[i] != 0) ok = false;
        }
        
        if(ok && balance[1] == 0 && balance[N] == 0){
            ans = mid;
            ansFlow = flow;
            right = mid - 1;
        } else {
            left = mid + 1;
        }
    }
    
    if(ans == -1) cout << "Impossible\n";
    else {
        cout << ans << "\n";
        for(int i = 0; i < M; i++){
            cout << ansFlow[i];
            if(i < M-1) cout << " ";
        }
        cout << "\n";
    }
    
    return 0;
}
```
