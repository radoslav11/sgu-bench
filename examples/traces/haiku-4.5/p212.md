Looking at this problem, I need to find a blocking data flow in a layered network where:

1. Data flows from source (level 1) to target (level L)
2. Data can only flow between nodes of consecutive levels
3. For each non-source/non-target node, flow in = flow out (conservation)
4. A blocking flow means we can't increase flow on any single edge without decreasing flow on others

The key insight is that a blocking flow can be found using a greedy approach or by finding any maximal flow where no augmenting paths exist.

Let me think about the approach:
1. Find the source node (level 1) and target node (level L)
2. Use a maximum flow algorithm (like Ford-Fulkerson with BFS/Dinic's algorithm) to find a blocking flow
3. A simpler approach: greedily push flow from source to sink, ensuring we reach a state where no more flow can be pushed on any single edge

Actually, for a "blocking flow", we need any flow where we can't increase flow on a single edge. The easiest approach is to find a maximum flow, which will be blocking.

Let me implement a max flow algorithm (Dinic's algorithm) which is efficient:

```cpp
#include<bits/stdc++.h>
using namespace std;

struct Edge {
    int to, cap, rev;
};

class MaxFlow {
public:
    vector<vector<Edge>> graph;
    vector<int> level, iter;
    
    MaxFlow(int n) : graph(n), level(n), iter(n) {}
    
    void add_edge(int from, int to, int cap) {
        graph[from].push_back({to, cap, (int)graph[to].size()});
        graph[to].push_back({from, 0, (int)graph[from].size() - 1});
    }
    
    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);
        
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& e : graph[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }
    
    int dfs(int v, int t, int f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < graph[v].size(); i++) {
            Edge& e = graph[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                int d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    graph[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }
    
    long long max_flow(int s, int t) {
        long long flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            int f;
            while ((f = dfs(s, t, INT_MAX)) > 0) {
                flow += f;
            }
        }
        return flow;
    }
};

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, M, L;
    cin >> N >> M >> L;
    
    vector<int> level(N + 1);
    int source = -1, target = -1;
    
    for (int i = 1; i <= N; i++) {
        cin >> level[i];
        if (level[i] == 1) source = i;
        if (level[i] == L) target = i;
    }
    
    MaxFlow mf(N + 1);
    vector<tuple<int, int, int>> edges;
    
    for (int i = 0; i < M; i++) {
        int a, b, c;
        cin >> a >> b >> c;
        edges.push_back({a, b, c});
        mf.add_edge(a, b, c);
    }
    
    mf.max_flow(source, target);
    
    // Output the flow
    for (int i = 0; i < M; i++) {
        auto [a, b, c] = edges[i];
        // Find the edge in the graph and get its residual capacity
        int flow = 0;
        for (auto& e : mf.graph[a]) {
            if (e.to == b) {
                flow = c - e.cap;
                break;
            }
        }
        cout << flow << "\n";
    }
    
    return 0;
}
```
