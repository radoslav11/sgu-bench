Looking at this problem, I need to find the probability that p < q given the observations.

Key insights:
1. We have two coins with unknown probabilities p and q, both uniformly distributed on [0,1]
2. Coin 1: tossed n1 times, got m1 heads
3. Coin 2: tossed n2 times, got m2 heads
4. We need to find P(p < q | observations)

Using Bayesian inference:
- The likelihood of observing m1 heads in n1 tosses with probability p is: p^m1 * (1-p)^(n1-m1)
- The likelihood of observing m2 heads in n2 tosses with probability q is: q^m2 * (1-q)^(n2-m2)

Since p and q are uniform on [0,1], the posterior distribution is proportional to the likelihood.

The joint posterior density is proportional to:
f(p,q) ∝ p^m1 * (1-p)^(n1-m1) * q^m2 * (1-q)^(n2-m2)

Since p and q are independent in the posterior, we can compute:
P(p < q) = ∫∫_{p<q} f(p,q) dp dq / ∫∫ f(p,q) dp dq

This can be simplified. The normalizing constant involves Beta function values:
- For p: Beta(m1+1, n1-m1+1)
- For q: Beta(m2+1, n2-m2+1)

So we need to compute:
P(p < q) = ∫₀¹ ∫₀^q p^m1(1-p)^(n1-m1) * q^m2(1-q)^(n2-m2) dp dq / (Beta(m1+1, n1-m1+1) * Beta(m2+1, n2-m2+1))

Using numerical integration or the fact that we can compute this with the incomplete beta function, we can evaluate this integral.

A practical approach is to use numerical integration (Monte Carlo or direct numerical integration).

```cpp
#include<bits/stdc++.h>
using namespace std;

double beta_func(int a, int b) {
    // Beta(a,b) = Gamma(a)*Gamma(b)/Gamma(a+b)
    // = (a-1)! * (b-1)! / (a+b-1)!
    double result = 1.0;
    for(int i = 1; i < a; i++) result *= i;
    for(int i = 1; i < b; i++) result *= i;
    for(int i = 1; i < a+b; i++) result /= i;
    return result;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int T;
    cin >> T;
    
    while(T--) {
        int n1, m1, n2, m2;
        cin >> n1 >> m1 >> n2 >> m2;
        
        double beta1 = beta_func(m1+1, n1-m1+1);
        double beta2 = beta_func(m2+1, n2-m2+1);
        
        // Numerical integration for P(p < q)
        double prob = 0.0;
        int steps = 10000;
        
        for(int i = 0; i < steps; i++) {
            double q = (i + 0.5) / steps;
            double q_contrib = pow(q, m2) * pow(1-q, n2-m2);
            
            // Integrate p from 0 to q
            for(int j = 0; j < steps * q; j++) {
                double p = (j + 0.5) / (steps * q) * q;
                double p_contrib = pow(p, m1) * pow(1-p, n1-m1);
                prob += p_contrib * q_contrib * (1.0/steps) * (1.0/steps);
            }
        }
        
        prob /= (beta1 * beta2);
        
        cout << fixed << setprecision(10) << prob << "\n";
    }
    
    return 0;
}
```

Actually, let me use a more efficient numerical integration approach:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int T;
    cin >> T;
    
    while(T--) {
        int n1, m1, n2, m2;
        cin >> n1 >> m1 >> n2 >> m2;
        
        double numerator = 0.0;
        double denominator = 0.0;
        int steps = 5000;
        double dq = 1.0 / steps;
        double dp = 1.0 / steps;
        
        for(int iq = 0; iq < steps; iq++) {
            double q = (iq + 0.5) * dq;
            double q_factor = pow(q, m2) * pow(1-q, n2-m2);
            
            for(int ip = 0; ip < steps; ip++) {
                double p = (ip + 0.5) * dp;
                double p_factor = pow(p, m1) * pow(1-p, n1-m1);
                double joint = p_factor * q_factor;
                
                denominator += joint * dp * dq;
                if(p < q) {
                    numerator += joint * dp * dq;
                }
            }
        }
        
        double prob = numerator / denominator;
        cout << fixed << setprecision(10) << prob << "\n";
    }
    
    return 0;
}
```
