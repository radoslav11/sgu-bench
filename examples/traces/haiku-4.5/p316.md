Looking at this problem, I need to simulate a tank battle:

1. Each tank starts with 100 hit points and 0 score points
2. For each shot:
   - Reduce target's HP by 8
   - If target's HP was > 0 before the shot, shooter gets 3 points
   - Only tanks with HP > 0 can shoot
3. At the end, each tank with HP > 0 gets bonus points = HP/2 (integer division)

Let me trace through the example:
- N=3, M=3
- Initial: Tank 1: HP=100, Score=0; Tank 2: HP=100, Score=0; Tank 3: HP=100, Score=0

Shot 1: Tank 2 shoots Tank 1
- Tank 1: HP = 100 - 8 = 92, Tank 2 was active, so Tank 2 gets 3 points
- Tank 2: Score = 3

Shot 2: Tank 1 shoots Tank 2
- Tank 2: HP = 100 - 8 = 92, Tank 1 was active, so Tank 1 gets 3 points
- Tank 1: Score = 3

Shot 3: Tank 2 shoots Tank 1
- Tank 1: HP = 92 - 8 = 84, Tank 2 was active, so Tank 2 gets 3 points
- Tank 2: Score = 3 + 3 = 6

End of match bonus (HP > 0):
- Tank 1: HP = 84, bonus = 84/2 = 42, Total Score = 3 + 42 = 45 ✓

```cpp
#include <iostream>
using namespace std;

int main() {
    int N, M;
    cin >> N >> M;
    
    vector<int> hp(N + 1, 100);      // hit points for tanks 1 to N
    vector<int> score(N + 1, 0);     // score for tanks 1 to N
    
    for (int i = 0; i < M; i++) {
        int shooter, target;
        cin >> shooter >> target;
        
        // Check if target had HP > 0 before the shot
        if (hp[target] > 0) {
            // Shooter gets 3 points
            score[shooter] += 3;
        }
        
        // Reduce target's HP by 8
        hp[target] -= 8;
        
        // HP cannot go below 0
        if (hp[target] < 0) {
            hp[target] = 0;
        }
    }
    
    // Add bonus points for tanks with HP > 0
    for (int i = 1; i <= N; i++) {
        if (hp[i] > 0) {
            score[i] += hp[i] / 2;
        }
    }
    
    // Output results
    for (int i = 1; i <= N; i++) {
        cout << hp[i] << " " << score[i] << "\n";
    }
    
    return 0;
}
```
