Looking at this problem, I need to:

1. Understand that we have a bidirected graph where each edge has two endpoints with orientations (+1 or -1)
2. An elementary transformation at a vertex negates all edge orientations at that vertex
3. We need to determine if we can transform the graph into a "directed" graph (where each edge has one endpoint with -1 and one with +1) using minimum transformations
4. A directed graph property: for each edge, one endpoint should have -1 and the other should have +1

Key insight: This is essentially a 2-coloring problem. We need to assign each vertex a state (transformed or not transformed). For each edge (u,v) with orientations (d1, d2):
- If we apply transformation to u but not v: orientations become (-d1, d2)
- If we apply transformation to v but not u: orientations become (d1, -d2)
- If we apply to both: orientations become (-d1, -d2)
- If we apply to neither: orientations stay (d1, d2)

For a valid directed graph, we need exactly one endpoint to have -1 and the other to have +1.

This is a 2-coloring/bipartition problem on a graph where:
- Vertices are the original vertices
- We need to assign each vertex a color (0 = no transform, 1 = transform)
- For each edge, based on the colors and current orientations, check if it results in exactly one -1

Let me think differently: For each edge (u, v) with ω(e,u)=d1 and ω(e,v)=d2:
- If d1 == d2 (both +1 or both -1), we need exactly one transformation between u and v
- If d1 != d2 (one +1, one -1), we need zero or two transformations between u and v

This becomes a graph coloring problem that we can solve with BFS/DFS!

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<tuple<int,int,int,int>> edges(m);
    vector<vector<pair<int,int>>> adj(n+1);
    
    for(int i = 0; i < m; i++){
        int a, b, d1, d2;
        cin >> a >> b >> d1 >> d2;
        edges[i] = {a, b, d1, d2};
        adj[a].push_back({b, i});
        adj[b].push_back({a, i});
    }
    
    vector<int> color(n+1, -1);
    vector<int> transformations;
    bool possible = true;
    
    for(int start = 1; start <= n && possible; start++){
        if(color[start] != -1) continue;
        
        queue<int> q;
        q.push(start);
        color[start] = 0;
        
        while(!q.empty() && possible){
            int u = q.front();
            q.pop();
            
            for(auto [v, edgeIdx] : adj[u]){
                auto [a, b, d1, d2] = edges[edgeIdx];
                
                int da = (a == u) ? d1 : d2;
                int db = (b == u) ? d2 : d1;
                
                // After transformation: if color[u]=1, da becomes -da, etc.
                int final_da = (color[u] == 1) ? -da : da;
                int final_db = (color[v] == -1) ? db : ((color[v] == 1) ? -db : db);
                
                // We need final_da and final_db to have opposite signs
                int needed_color_v;
                if(da == db){
                    // Need opposite final signs: need different colors
                    needed_color_v = 1 - color[u];
                } else {
                    // Need same final signs: need same colors
                    needed_color_v = color[u];
                }
                
                if(color[v] == -1){
                    color[v] = needed_color_v;
                    q.push(v);
                } else if(color[v] != needed_color_v){
                    possible = false;
                    break;
                }
            }
        }
    }
    
    if(!possible){
        cout << "NO\n";
    } else {
        cout << "YES\n";
        for(int i = 1; i <= n; i++){
            if(color[i] == 1){
                transformations.push_back(i);
            }
        }
        cout << transformations.size() << "\n";
        for(int v : transformations){
            cout << v << "\n";
        }
    }
    
    return 0;
}
```
