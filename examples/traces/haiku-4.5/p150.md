Looking at this problem, I need to:

1. Track a beetle moving from point (x1, y1) to point (x2, y2)
2. Number each cell it passes through
3. Find the lower-left corner of the n-th cell

The key insight is that the beetle moves in a straight line, and I need to find which cell it's in after passing through n cells.

Let me think about the geometry:
- The beetle starts at (x1, y1) and ends at (x2, y2)
- A cell is identified by its lower-left corner coordinates (floor values)
- As the beetle moves from start to end, it crosses through different cells
- I need to find the n-th distinct cell it enters

The approach:
1. Calculate the direction vector (dx, dy) from start to end
2. Normalize or parametrize the movement
3. Track which cells are visited in order
4. Return the lower-left corner of the n-th cell

For a line from (x1, y1) to (x2, y2), I can parametrize it as:
- x(t) = x1 + t*(x2-x1)
- y(t) = y1 + t*(y2-y1)
where t goes from 0 to 1.

The cell containing point (x, y) has lower-left corner at (floor(x), floor(y)).

I'll use a parametric approach, finding all the cell transitions by tracking when either x or y crosses an integer boundary.

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    long long x1, y1, x2, y2, n;
    cin >> x1 >> y1 >> x2 >> y2 >> n;
    
    long long dx = x2 - x1;
    long long dy = y2 - y1;
    
    // If start and end are in the same cell
    if (dx == 0 && dy == 0) {
        cout << "no solution\n";
        return 0;
    }
    
    set<pair<long long, long long>> visited;
    long long cx = x1, cy = y1;
    
    // Get initial cell
    long long cell_x = (long long)floor((double)x1);
    long long cell_y = (long long)floor((double)y1);
    if (x1 < 0 && x1 != (long long)x1) cell_x--;
    if (y1 < 0 && y1 != (long long)y1) cell_y--;
    
    visited.insert({cell_x, cell_y});
    
    // Parametric line walking
    long long steps = abs(dx) + abs(dy);
    if (steps == 0) {
        cout << "no solution\n";
        return 0;
    }
    
    // Use a small step to walk through cells
    double step_size = 1.0 / (steps + 1);
    
    for (int i = 1; i <= steps && visited.size() < n; i++) {
        double t = i * step_size;
        double x = x1 + t * dx;
        double y = y1 + t * dy;
        
        long long new_cell_x = (long long)floor(x);
        long long new_cell_y = (long long)floor(y);
        
        if (x < 0 && x != (long long)x) new_cell_x--;
        if (y < 0 && y != (long long)y) new_cell_y--;
        
        if (visited.find({new_cell_x, new_cell_y}) == visited.end()) {
            visited.insert({new_cell_x, new_cell_y});
        }
    }
    
    if (visited.size() < n) {
        cout << "no solution\n";
    } else {
        auto it = visited.begin();
        advance(it, n - 1);
        cout << it->first << " " << it->second << "\n";
    }
    
    return 0;
}
```

Wait, this approach with floating point might have precision issues. Let me use a better approach with exact arithmetic by tracking grid crossings:

```cpp
#include <bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    long long x1, y1, x2, y2, n;
    cin >> x1 >> y1 >> x2 >> y2 >> n;
    
    long long dx = x2 - x1;
    long long dy = y2 - y1;
    
    if (dx == 0 && dy == 0) {
        cout << "no solution\n";
        return 0;
    }
    
    set<pair<long long, long long>> cells;
    
    // Walk along the line using Bresenham-like approach
    long long steps = abs(dx) + abs(dy);
    
    for (long long i = 0; i < steps; i++) {
        double t = (double)(i) / steps;
        double x = x1 + t * dx;
        double y = y1 + t * dy;
        
        long long cell_x = (long long)floor(x);
        long long cell_y = (long long)floor(y);
        
        cells.insert({cell_x, cell_y});
        
        if (cells.size() == n) {
            cout << cell_x << " " << cell_y << "\n";
            return 0;
        }
    }
    
    cout << "no solution\n";
    return 0;
}
```
