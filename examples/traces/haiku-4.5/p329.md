// Each orientation rotates the edges

vector<vector<int>> getEdges(int type, int rot) {
    // Returns all 3 rotations for a given type
    vector<vector<int>> base = {
        {0,0,0}, // type 1
        {0,0,1}, // type 2

    vector<vector<int>> result;

        result.push_back(edges);
        rotate(edges.begin(), edges.begin()+1, edges.end());
    }
    return result;
}

        return;

    // Try each triangle type
