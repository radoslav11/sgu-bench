Looking at this problem, I need to:

1. Start with an undirected graph representing a road system
2. Handle two types of queries:
   - `DELETE x`: Remove the x-th road
   - `LENGTH x y`: Find the shortest distance between cities x and y

The key insight is that after each deletion, I need to recompute distances in the remaining graph.

For finding shortest paths, I'll use BFS since we're looking for unweighted shortest paths.

Let me trace through the example:
- Initial graph: 3 cities, 3 roads
  - Road 1: 1-2
  - Road 2: 2-3
  - Road 3: 3-1
  
- Query "LENGTH 1 2": Direct path exists, distance = 1
- Query "DELETE 1": Remove road 1 (1-2)
- Query "LENGTH 1 2": Path must be 1-3-2, distance = 2
- Query "LENGTH 2 3": Direct path exists (road 2), distance = 1
- Query "DELETE 3": Remove road 3 (3-1)
- Query "LENGTH 1 2": No path exists (1 is isolated), distance = -1

Here's my solution:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<pair<int,int>> edges(m + 1);
    vector<bool> deleted(m + 1, false);
    
    for(int i = 1; i <= m; i++){
        int x, y;
        cin >> x >> y;
        edges[i] = {x, y};
    }
    
    string command;
    while(cin >> command){
        if(command == "DELETE"){
            int x;
            cin >> x;
            deleted[x] = true;
        }
        else { // LENGTH
            int x, y;
            cin >> x >> y;
            
            // Build current graph
            vector<vector<int>> adj(n + 1);
            for(int i = 1; i <= m; i++){
                if(!deleted[i]){
                    int u = edges[i].first;
                    int v = edges[i].second;
                    adj[u].push_back(v);
                    adj[v].push_back(u);
                }
            }
            
            // BFS from x to y
            vector<int> dist(n + 1, -1);
            queue<int> q;
            q.push(x);
            dist[x] = 0;
            
            while(!q.empty()){
                int u = q.front();
                q.pop();
                
                for(int v : adj[u]){
                    if(dist[v] == -1){
                        dist[v] = dist[u] + 1;
                        q.push(v);
                    }
                }
            }
            
            cout << dist[y] << "\n";
        }
    }
    
    return 0;
}
```

The algorithm:
1. Read n cities and m roads, storing each edge with its index
2. Mark edges as deleted when DELETE commands are processed
3. For each LENGTH query:
   - Build the current graph by including only non-deleted edges
   - Run BFS from city x to find shortest distance to city y
   - Output the distance (or -1 if unreachable)

Time complexity: O(Q * (n + m)) where Q is the number of queries, which should be acceptable given the constraints.
