Looking at this problem, I need to:

1. Find a cyclic path (a cycle) in a directed graph
2. Maximize the greedy function f(G) = total_cost / total_time
3. Output the cycle with maximum f(G)

Key observations:
- We need to find cycles in a directed graph
- For each cycle, calculate the ratio of total cost to total time
- Return the cycle with the maximum ratio

Approach:
- Use DFS from each node to find all cycles
- For each cycle found, calculate its ratio
- Keep track of the best cycle found

Let me implement this:

```cpp
#include<bits/stdc++.h>
using namespace std;

int N, M;
vector<tuple<int, int, int>> adj[55]; // {destination, cost, time}
vector<int> bestPath;
double bestRatio = -1;

void dfs(int start, int current, vector<int>& path, vector<bool>& visited, int totalCost, int totalTime) {
    // Check if we can return to start (forming a cycle)
    for(auto& [next, cost, time] : adj[current]) {
        if(next == start && path.size() >= 2) {
            // Found a cycle
            double ratio = (double)(totalCost + cost) / (totalTime + time);
            if(ratio > bestRatio) {
                bestRatio = ratio;
                bestPath = path;
                bestPath.push_back(start);
            }
        }
        // Continue DFS if not visited
        if(!visited[next] && path.size() < 50) {
            visited[next] = true;
            path.push_back(next);
            dfs(start, next, path, visited, totalCost + cost, totalTime + time);
            path.pop_back();
            visited[next] = false;
        }
    }
}
```
