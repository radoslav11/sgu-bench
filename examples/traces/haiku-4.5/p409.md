Looking at this problem, I need to:

1. Create an N² × N² grid
2. Fill it with '*' (blue) and '.' (red) such that:
   - Each row has exactly K blue cells
   - Each column has exactly K blue cells
   - Each N × N sub-grid has exactly K blue cells

Let me think about the constraints:
- Total blue cells needed: N² × K (from row requirement)
- Total blue cells needed: N² × K (from column requirement)
- Total blue cells needed: N² × K (from N×N sub-grid requirement, since there are N² sub-grids each needing K)

So the constraints are consistent if we can satisfy them simultaneously.

A key insight: We can try to place cells in a pattern. One approach is to use a Latin square-like pattern where we place K blue cells in each N×N block systematically.

For each N×N block, we need to place exactly K cells such that when we consider all blocks, each row and column of the overall grid has exactly K blue cells.

Let me think of it differently: I'll iterate through the grid and for each cell (i,j), I need to decide if it's blue or red. I can use a greedy or constructive approach.

Actually, a simpler approach: 
- For position (i, j), it belongs to block (i/N, j/N)
- I can use a pattern where in block (bi, bj), I place blue cells at positions where (i%N + j%N) % N equals some value that depends on the block position.

Let me try a simpler construction: place a blue cell at (i,j) if `(i/N)*K + ((i%N + j%N) % N) < K` and `j%N == ((i/N)*K + ((i%N + j%N) % N)) % N`... This is getting complex.

Better idea: For each row, I need exactly K blues. For each N×N block in that row, I distribute the K cells across the N blocks. I can place them in a shifted diagonal pattern within each block.

Let me use this approach:
- For row i (0-indexed), in block column bc (0 to N-1), place K/N complete sets + (K%N) partial cells
- Use a diagonal offset pattern to ensure column and block constraints

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, K;
    cin >> N >> K;
    
    // Check if solution is possible
    if(K > N*N || K < 0){
        cout << "NO SOLUTION\n";
        return 0;
    }
    
    int size = N * N;
    vector<string> grid(size, string(size, '.'));
    
    // Try to fill greedily
    // For each cell (i,j), track constraints
    vector<int> row_count(size, 0);  // blues in each row
    vector<int> col_count(size, 0);  // blues in each column
    vector<int> block_count(N*N, 0); // blues in each N×N block
    
    // Place cells using pattern: for row i, place blues at columns with offset
    for(int i = 0; i < size; i++){
        for(int k = 0; k < K; k++){
            int col = (i * K + k) % size;
            if(row_count[i] < K && col_count[col] < K){
                int block_r = i / N;
                int block_c = col / N;
                int block_id = block_r * N + block_c;
                
                if(block_count[block_id] < K){
                    grid[i][col] = '*';
                    row_count[i]++;
                    col_count[col]++;
                    block_count[block_id]++;
                }
            }
        }
    }
    
    // Verify solution
    bool valid = true;
    for(int i = 0; i < size && valid; i++){
        if(row_count[i] != K) valid = false;
        if(col_count[i] != K) valid = false;
        if(block_count[i] != K) valid = false;
    }
    
    if(!valid){
        cout << "NO SOLUTION\n";
    } else {
        for(int i = 0; i < size; i++){
            cout << grid[i] << "\n";
        }
    }
    
    return 0;
}
```

Wait, my greedy approach might not work. Let me use a simpler pattern based on the examples - it seems like a diagonal-ish pattern works.

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, K;
    cin >> N >> K;
    
    int size = N * N;
    vector<string> grid(size, string(size, '.'));
    
    // Place K blues in each row using a pattern
    for(int i = 0; i < size; i++){
        for(int k = 0; k < K; k++){
            int col = (i + k * N) % size;
            grid[i][col] = '*';
        }
    }
    
    // Verify
    for(int i = 0; i < size; i++){
        int rc = 0, cc = 0;
        for(int j = 0; j < size; j++){
            if(grid[i][j] == '*') rc++;
            if(grid[j][i] == '*') cc++;
        }
        if(rc != K || cc != K){
            cout << "NO SOLUTION\n";
            return 0;
        }
    }
    
    for(int bi = 0; bi < N; bi++){
        for(int bj = 0; bj < N; bj++){
            int bc = 0;
            for(int i = bi*N; i < (bi+1)*N; i++){
                for(int j = bj*N; j < (bj+1)*N; j++){
                    if(grid[i][j] == '*') bc++;
                }
            }
            if(bc != K){
                cout << "NO SOLUTION\n";
                return 0;
            }
        }
    }
    
    for(int i = 0; i < size; i++){
        cout << grid[i] << "\n";
    }
    
    return 0;
}
```
