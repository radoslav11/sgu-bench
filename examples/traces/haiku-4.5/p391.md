Looking at this problem, I need to determine if we can fold a piece of paper so that all marked cells end up in the same position.

Key insights:
1. When we fold paper, cells that were at different positions can overlap
2. After folding along a line between rows i and i+1, rows get reflected/mirrored
3. Similarly for columns
4. We need to check if there's a sequence of folds that can bring all marked cells to the same location

The crucial observation is that when we fold:
- If we fold between rows i and i+1, rows 1..i stay as is, and rows (i+1)..n get mirrored around row i (row i+1 becomes i, row i+2 becomes i-1, etc.)
- Similarly for columns

For all marked cells to be at the same position after folding, they must be able to map to the same cell through a series of reflections.

A simpler way to think about it: 
- After folding, each cell (r, c) can be mapped to some cell (r', c') based on the fold lines
- We need to check if all marked cells can be mapped to the same (r', c')

#include<bits/stdc++.h>
using namespace std;

bool canFold(vector<int> positions, int maxPos){
    if(positions.empty()) return true;
    sort(positions.begin(), positions.end());
    
    // Try each possible fold line (between integers 0.5, 1.5, ..., maxPos-0.5)
    for(int foldBefore = 1; foldBefore <= maxPos; foldBefore++){
        set<int> mapped;
        bool valid = true;
        for(int pos : positions){
            int newPos;
            if(pos <= foldBefore){
                newPos = pos;
            } else {
                newPos = 2 * foldBefore + 1 - pos;
                if(newPos < 1) valid = false;
            }
            mapped.insert(newPos);
        }
        if(valid && mapped.size() == 1) return true;
    }
    return false;
}

int main(){
    int n, m, k;
    cin >> n >> m >> k;
    
    vector<int> rows, cols;
    for(int i = 0; i < k; i++){
        int x, y;
        cin >> x >> y;
        rows.push_back(x);
        cols.push_back(y);
    }
    
    if(canFold(rows, n) && canFold(cols, m)){
        cout << "YES\n";
    } else {
        cout << "NO\n";
    }
    
    return 0;
}
