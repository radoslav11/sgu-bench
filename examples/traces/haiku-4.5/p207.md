Looking at this problem, I need to distribute M coins among N robbers to minimize total unfairness.

Key observations:
1. Robber i should ideally get `Xi/Y * M` coins
2. The unfairness for robber i is `|Xi/Y - Ki/M|` = `|Xi*M - Ki*Y| / (Y*M)`
3. To minimize total unfairness, I need to minimize `sum(|Xi*M - Ki*Y|)`
4. The sum of all Ki must equal M

This is a classic problem that can be solved greedily. The idea is:
- Calculate the ideal amount each robber should get: `ideal_i = Xi * M / Y`
- Start by giving each robber the floor of their ideal amount
- Distribute the remaining coins one by one to the robbers who would benefit most (i.e., those with the largest fractional parts)

The fractional part for robber i when they get floor of ideal amount is important. If we give robber i one more coin, the unfairness contribution changes. We should greedily give extra coins to those robbers where adding one more coin reduces unfairness the most.

For robber i with ideal share `ideal_i = Xi*M/Y`:
- Let `base_i = floor(ideal_i)` 
- The fractional part is `ideal_i - base_i = (Xi*M mod Y) / Y`
- We should give extra coins to robbers with the largest fractional parts

Let me implement this approach:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N, M, Y;
    cin >> N >> M >> Y;
    
    vector<int> X(N);
    for(int i = 0; i < N; i++){
        cin >> X[i];
    }
    
    // Calculate base allocation (floor of ideal amount)
    vector<long long> K(N);
    long long total = 0;
    vector<pair<double, int>> fractions; // (fractional part, index)
    
    for(int i = 0; i < N; i++){
        long long ideal = (long long)X[i] * M / Y;
        K[i] = ideal;
        total += K[i];
        
        // Calculate fractional part
        long long remainder = ((long long)X[i] * M) % Y;
        double frac = (double)remainder / Y;
        fractions.push_back({frac, i});
    }
    
    // Distribute remaining coins to those with largest fractional parts
    long long remaining = M - total;
    
    // Sort by fractional part in descending order
    sort(fractions.rbegin(), fractions.rend());
    
    for(int i = 0; i < remaining; i++){
        K[fractions[i].second]++;
    }
    
    // Output result
    for(int i = 0; i < N; i++){
        if(i > 0) cout << " ";
        cout << K[i];
    }
    cout << "\n";
    
    return 0;
}
```

int main(){

    long long remaining = M - total;

    sort(fractions.rbegin(), fractions.rend());
    
    for(int i = 0; i < remaining; i++){
        K[-fractions[i].second]++;
    }
    
    // Output result
    for(int i = 0; i < N; i++){
