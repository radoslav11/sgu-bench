Looking at this problem, I need to:

1. Parse wiki-text lines to identify list markers (`*` for unordered, `#` for ordered)
2. Group consecutive lines with the same marker into lists
3. Handle nested lists recursively
4. Convert to HTML with proper formatting

Key observations:
- A line starts with a list marker if it begins with `*` or `#`
- Multiple consecutive markers at the start indicate nesting level
- Lines with the same marker at the same position form a list
- Non-list lines are printed as-is
- Need to track nesting depth and marker type

Let me trace through example 2:
- `*ab` - starts with `*` (1 level), content is `ab`
- `*x#x` - starts with `*` (1 level), content is `x#x` (the `#` is part of content)
- `*#1` - starts with `*` (1 level), followed by `#` which marks a nested list, content is `1`
- `*#2` - starts with `*` (1 level), followed by `#`, content is `2`
- `#3` - starts with `#` (1 level), content is `3`

So the algorithm:
1. For each line, count leading `*` and `#` markers
2. Group lines by marker type at each depth level
3. Recursively process nested structures

```cpp
#include<bits/stdc++.h>
using namespace std;

struct Line {
    int depth;
    char marker;
    string content;
};

vector<Line> lines;

vector<Line> parseInput() {
    vector<Line> result;
    string input;
    while(getline(cin, input)) {
        if(input.empty()) continue;
        
        int depth = 0;
        char marker = '\0';
        int pos = 0;
        
        while(pos < input.length() && (input[pos] == '*' || input[pos] == '#')) {
            if(depth == 0) {
                marker = input[pos];
            } else if(input[pos] != marker) {
                break;
            }
            depth++;
            pos++;
        }
        
        if(depth == 0) {
            // Not a list item
            result.push_back({0, '\0', input});
        } else {
            result.push_back({depth, marker, input.substr(pos)});
        }
    }
    return result;
}

void process(vector<Line>& lines, int start, int end, int minDepth) {
    int i = start;
    while(i < end) {
        if(lines[i].depth == 0) {
            cout << lines[i].content << "\n";
            i++;
        } else if(lines[i].depth < minDepth) {
            break;
        } else if(lines[i].depth == minDepth) {
            // Start of a list
            char marker = lines[i].marker;
            string tag = (marker == '*') ? "ul" : "ol";
            cout << "<" << tag << ">\n";
            
            int j = i;
            while(j < end && lines[j].depth == minDepth && lines[j].marker == marker) {
                cout << "<li>\n";
                cout << lines[j].content << "\n";
                
                // Check for nested content
                int k = j + 1;
                if(k < end && lines[k].depth > minDepth) {
                    process(lines, k, end, minDepth + 1);
                    // Find where nested list ends
                    while(k < end && lines[k].depth > minDepth) {
                        k++;
                    }
                    j = k - 1;
                }
                
                cout << "</li>\n";
                j++;
            }
            
            cout << "</" << tag << ">\n";
            i = j;
        } else {
            i++;
        }
    }
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    lines = parseInput();
    process(lines, 0, lines.size(), 0);
    
    return 0;
}
```

```cpp
struct Line {

    char marker;
    string content;

    vector<Line> result;
    string input;
    while(getline(cin, input)) {
        int pos = 0;
        int depth = 0;
        char marker = '\0';
        
        while(pos < input.length() && (input[pos] == '*' || input[pos] == '#')) {
            if(marker == '\0') marker = input[pos];
            if(input[pos] != marker) break;
            depth++;
            pos++;
        }
        
        if(depth == 0) {
            result.push_back({0, '\0', input});
        } else {
            result.push_back({depth, marker, input.substr(pos)});
        }
    }
    return result;
}

    while(idx < lines.size()) {
        if(lines[idx].depth == 0 || lines[idx].depth > maxDepth) {
            break;
        }
        
        if(lines[idx].depth < maxDepth) {
            break;
        }

            cout << "<li>\n";
            cout << lines[idx].content << "\n";
            idx++;
            
            if(idx < lines.size() && lines[idx].depth > maxDepth) {
```
