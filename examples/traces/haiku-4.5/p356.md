Looking at this problem, I need to find the probability of guessing exactly K married couples correctly out of N couples.

Let me think through the combinatorics:
- Total ways to match N women to N men: N!
- Ways to match exactly K couples correctly: We need to:
  1. Choose which K couples are correct: C(N, K)
  2. Arrange the remaining (N-K) couples such that NONE of them are correct (derangements): D(N-K)

A derangement D(n) is a permutation where no element appears in its original position.

The formula for derangements is:
- D(0) = 1
- D(1) = 0
- D(n) = (n-1) * (D(n-1) + D(n-2))

Or using the formula: D(n) = n! * Σ((-1)^i / i!) for i from 0 to n

So the number of ways to get exactly K correct matches is: C(N, K) * D(N-K)

The probability is: [C(N, K) * D(N-K)] / N!

Let me verify with the example:
- N=5, K=2
- C(5,2) = 10
- D(3) = 2 * (D(2) + D(1)) = 2 * (1 + 0) = 2
- Ways = 10 * 2 = 20
- Total = 5! = 120
- Probability = 20/120 = 1/6 ✓

For N=10, K=9:
- D(1) = 0 (can't have exactly 9 correct, because if 9 couples are correct, the 10th must also be correct)
- So probability = 0 ✓

I need to:
1. Calculate derangements
2. Calculate combinations
3. Compute the probability as a fraction
4. Reduce it to irreducible form using GCD

```cpp
#include<bits/stdc++.h>
using namespace std;

long long gcd(long long a, long long b) {
    while(b) {
        long long temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

long long derangement(int n) {
    if(n == 0) return 1;
    if(n == 1) return 0;
    
    vector<long long> D(n + 1);
    D[0] = 1;
    D[1] = 0;
    
    for(int i = 2; i <= n; i++) {
        D[i] = (long long)(i - 1) * (D[i - 1] + D[i - 2]);
    }
    
    return D[n];
}

long long factorial(int n) {
    long long result = 1;
    for(int i = 2; i <= n; i++) {
        result *= i;
    }
    return result;
}

long long binomial(int n, int k) {
    if(k > n) return 0;
    if(k == 0 || k == n) return 1;
    
    long long numerator = 1;
    long long denominator = 1;
    
    for(int i = 0; i < k; i++) {
        numerator *= (n - i);
        denominator *= (i + 1);
    }
    
    return numerator / denominator;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int K, N;
    cin >> K >> N;
    
    long long numerator = binomial(N, K) * derangement(N - K);
    long long denominator = factorial(N);
    
    if(numerator == 0) {
        cout << "0\n";
    } else {
        long long g = gcd(numerator, denominator);
        numerator /= g;
        denominator /= g;
        cout << numerator << "/" << denominator << "\n";
    }
    
    return 0;
}
```
