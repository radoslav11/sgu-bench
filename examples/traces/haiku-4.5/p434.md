Looking at this problem, I need to find the minimum number of pouring operations to transform the current state S into the desired state D.

Key observations:
1. If the total liquid in S doesn't equal total in D, it's impossible (return -1)
2. A tube that already has the correct amount doesn't need any pouring
3. We need to find the minimum number of pour operations

This is a state-space search problem. I can use BFS to find the shortest path from state S to state D, where each edge represents one pouring operation.

However, with N ≤ 21 and continuous amounts, we can't do straightforward BFS on exact states. Instead, I'll think about it differently:

A key insight: if a tube already has the correct amount, we don't need to touch it. For tubes that need adjustment, we can model this as a graph problem where we need to figure out the minimum operations.

Actually, let me reconsider: we can use BFS where the state is the current configuration of all tubes (represented as a tuple of amounts), and we explore all possible single pour operations.

But with continuous values, this won't work perfectly. However, we can represent states as vectors and use a map to track visited states.

Let me implement a BFS solution:

```cpp
#include<bits/stdc++.h>
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    
    vector<long long> s(n), d(n);
    for(int i = 0; i < n; i++) cin >> s[i];
    for(int i = 0; i < n; i++) cin >> d[i];
    
    // Check if total is same
    long long sum_s = 0, sum_d = 0;
    for(int i = 0; i < n; i++){
        sum_s += s[i];
        sum_d += d[i];
    }
    
    if(sum_s != sum_d){
        cout << -1 << endl;
        return 0;
    }
    
    // Check if already correct
    if(s == d){
        cout << 0 << endl;
        return 0;
    }
    
    // BFS
    map<vector<long long>, int> dist;
    queue<vector<long long>> q;
    
    q.push(s);
    dist[s] = 0;
    
    while(!q.empty()){
        vector<long long> cur = q.front();
        q.pop();
        
        if(cur == d){
            cout << dist[cur] << endl;
            return 0;
        }
        
        int cur_dist = dist[cur];
        
        // Try pouring from tube i to tube j
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++){
                if(i == j) continue;
                if(cur[i] == 0) continue;
                
                // Pour from i to j (pour until i is empty or j reaches d[j])
                vector<long long> next = cur;
                
                // Pour all from i to j
                next[j] += next[i];
                next[i] = 0;
                
                if(dist.find(next) == dist.end()){
                    dist[next] = cur_dist + 1;
                    q.push(next);
                }
                
                // Also try pouring to make j reach exactly d[j] if beneficial
                if(cur[j] < d[j]){
                    vector<long long> next2 = cur;
                    long long need = d[j] - cur[j];
                    long long pour_amount = min(need, cur[i]);
                    next2[i] -= pour_amount;
                    next2[j] += pour_amount;
                    
                    if(dist.find(next2) == dist.end()){
                        dist[next2] = cur_dist + 1;
                        q.push(next2);
                    }
                }
            }
        }
    }
    
    cout << -1 << endl;
    return 0;
}
```

Wait, this approach might be too slow or might not explore enough states. Let me reconsider: we only need to pour amounts that matter - either emptying a tube or filling to the target. This limits the state space significantly.

```cpp
int main(){

    for(int i = 0; i < n; i++){
        sum_s += s[i];
        sum_d += d[i];
    }
    
    if(sum_s != sum_d){
        cout << -1 << endl;
        return 0;
    }
    
    if(s == d){
        cout << 0 << endl;
        return 0;
    }
    
    map<vector<long long>, int> dist;
    queue<vector<long long>> q;
    
    q.push(s);
    dist[s] = 0;
    
    while(!q.empty()){
        vector<long long> cur = q.front();
        q.pop();
        
        if(cur == d){
            cout << dist[cur] << endl;
            return 0;
        }
        
        int cur_dist = dist[cur];
        if(cur_dist > 100) continue; // Pruning
        
        for(int i = 0; i < n; i++){
            for(int j = 0; j < n; j++){
                if(i == j || cur[i] == 0) continue;
                
                vector<long long> next = cur;
                next[j] += next[i];
                next[i] = 0;
                
                if(dist.find(next) == dist.end()){
                    dist[next] = cur_dist + 1;
                    q.push(next);
                }
            }
        }
    }
    
    cout << -1 << endl;
    return 0;
}
```
