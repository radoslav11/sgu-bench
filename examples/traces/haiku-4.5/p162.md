Looking at this problem, I need to find the volume of a tetrahedron (triangular pyramid) given all 6 edge lengths.

Given edges: AB, AC, AD, BC, BD, CD

Let me denote:
- a = AB, b = AC, c = AD, d = BC, e = BD, f = CD

One approach is to use the Cayley-Menger determinant formula for the volume of a tetrahedron given all 6 edge lengths.

The volume V of a tetrahedron with vertices A, B, C, D can be calculated using:

288V² = |Cayley-Menger determinant|

The Cayley-Menger determinant is:
```
| 0  1  1  1  1 |
| 1  0  d₁² d₂² d₃² |
| 1  d₁² 0  d₄² d₅² |
| 1  d₂² d₄² 0  d₆² |
| 1  d₃² d₅² d₆² 0  |
```

Where the distances are between pairs of 4 vertices.

Let me map the edges:
- AB = a (distance between vertices 0 and 1)
- AC = b (distance between vertices 0 and 2)  
- AD = c (distance between vertices 0 and 3)
- BC = d (distance between vertices 1 and 2)
- BD = e (distance between vertices 1 and 3)
- CD = f (distance between vertices 2 and 3)

The Cayley-Menger determinant becomes:
```
| 0  1   1   1   1  |
| 1  0   a²  b²  c² |
| 1  a²  0   d²  e² |
| 1  b²  d²  0   f² |
| 1  c²  e²  f²  0  |
```

I'll compute this 5x5 determinant and then V = sqrt(determinant / 288).

```cpp
#include <bits/stdc++.h>
using namespace std;

double det5x5(double m[5][5]) {
    // Compute determinant of 5x5 matrix using cofactor expansion
    double result = 0;
    
    // Expanding along first row
    for (int i = 0; i < 5; i++) {
        // Create 4x4 submatrix
        double sub[4][4];
        for (int row = 1; row < 5; row++) {
            int sub_col = 0;
            for (int col = 0; col < 5; col++) {
                if (col != i) {
                    sub[row-1][sub_col++] = m[row][col];
                }
            }
        }
        
        // Compute 4x4 determinant
        double det4 = 0;
        for (int j = 0; j < 4; j++) {
            double sub3[3][3];
            for (int row = 1; row < 4; row++) {
                int sub_col = 0;
                for (int col = 0; col < 4; col++) {
                    if (col != j) {
                        sub3[row-1][sub_col++] = sub[row][col];
                    }
                }
            }
            
            // Compute 3x3 determinant
            double det3 = sub3[0][0] * (sub3[1][1] * sub3[2][2] - sub3[1][2] * sub3[2][1])
                        - sub3[0][1] * (sub3[1][0] * sub3[2][2] - sub3[1][2] * sub3[2][0])
                        + sub3[0][2] * (sub3[1][0] * sub3[2][1] - sub3[1][1] * sub3[2][0]);
            
            det4 += (j % 2 == 0 ? 1 : -1) * sub[0][j] * det3;
        }
        
        result += (i % 2 == 0 ? 1 : -1) * m[0][i] * det4;
    }
    
    return result;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    double a, b, c, d, e, f;
    cin >> a >> b >> c >> d >> e >> f;
    
    double m[5][5] = {
        {0, 1, 1, 1, 1},
        {1, 0, a*a, b*b, c*c},
        {1, a*a, 0, d*d, e*e},
        {1, b*b, d*d, 0, f*f},
        {1, c*c, e*e, f*f, 0}
    };
    
    double det = det5x5(m);
    double volume = sqrt(abs(det) / 288.0);
    
    cout << fixed << setprecision(4) << volume << endl;
    
    return 0;
}
```
