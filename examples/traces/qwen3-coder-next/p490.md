                for (int j = 2; j < W-1 && spots_placed < N; j++) {
                    if (grid[i][j] == '#') continue;
                    // Check if surrounded by black
                    if (grid[i-1][j] == '.' && grid[i+1][j] == '.' && 
                        grid[i][j-1] == '.' && grid[i][j+1] == '.') {
                        grid[i][j] = '#';
                        spots_placed++;
                    }
                }
            }
        }
        
        if (spots_placed < N) {
            cout << "Impossible\n";
        } else {
            for (int i = 0; i < H; i++) {
                cout << grid[i] << "\n";
            }
        }
    }
    return 0;
}
