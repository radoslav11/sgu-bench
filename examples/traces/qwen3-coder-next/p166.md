#include <iostream>
#include <vector>
#include <string>
#include <cctype>

int main() {
    std::string commands;
    std::getline(std::cin, commands);

    // We need to simulate a multi-line text editor
    std::vector<std::string> lines;
    int current_line = 0; // index of current line (0-based)
    int cursor_pos = 0;   // position in the current line (0-based, after last char means end of line)

    for (size_t i = 0; i < commands.size(); ++i) {
        char cmd = commands[i];
        
        if (cmd == 'Q') {
            break;
        } else if (cmd == 'L') {
            // move cursor left
            if (cursor_pos > 0) {
                cursor_pos--;
            } else if (current_line > 0) {
                // move to end of previous line
                current_line--;
                cursor_pos = lines[current_line].size();
            }
        } else if (cmd == 'R') {
            // move cursor right
            int line_len = lines[current_line].size();
            if (cursor_pos < line_len) {
                cursor_pos++;
            } else if (current_line < static_cast<int>(lines.size()) - 1) {
                // move to beginning of next line
                current_line++;
                cursor_pos = 0;
            } else {
                // after end of last line, move beyond (but we don't have position beyond size, so just increment cursor_pos)
                cursor_pos++;
            }
        } else if (cmd == 'U') {
            // move up
            if (current_line > 0) {
                current_line--;
                cursor_pos = std::min(cursor_pos, static_cast<int>(lines[current_line].size()));
            }
        } else if (cmd == 'D') {
            // move down
            if (current_line < static_cast<int>(lines.size()) - 1) {
                current_line++;
                cursor_pos = std::min(cursor_pos, static_cast<int>(lines[current_line].size()));
            }
        } else if (cmd == 'N') {
            // insert new line after current
            std::string rest = "";
            if (cursor_pos < static_cast<int>(lines[current_line].size())) {
                rest = lines[current_line].substr(cursor_pos);
            }
            lines[current_line] = lines[current_line].substr(0, cursor_pos);
            
            // Insert new line after current
            lines.insert(lines.begin() + current_line + 1, rest);
            current_line++;
            cursor_pos = 0;
        } else if (cmd == 'E') {
            // move to end of current line
            cursor_pos = lines[current_line].size();
        } else if (cmd == 'H') {
            // move to beginning of current line
            cursor_pos = 0;
        } else if (cmd == 'X') {
            // delete current symbol
            int line_len = lines[current_line].size();
            if (cursor_pos < line_len) {
                // delete at cursor_pos
                lines[current_line].erase(cursor_pos, 1);
            } else if (cursor_pos == line_len && current_line < static_cast<int>(lines.size()) - 1) {
                // after end of line, attach next line
                lines[current_line] += lines[current_line + 1];
                lines.erase(lines.begin() + current_line + 1);
            } else if (cursor_pos > line_len && lines.size() > 0) {
                // beyond end of last line: if not at end, add spaces to make up for cursor position
                int extra_spaces = cursor_pos - line_len;
                lines[current_line].append(extra_spaces, ' ');
            }
        } else if (cmd == 'B') {
            // delete symbol before cursor
            if (cursor_pos > 0) {
                lines[current_line].erase(cursor_pos - 1, 1);
                cursor_pos--;
            } else if (current_line > 0) {
                // attach to previous line
                int prev_len = lines[current_line - 1].size();
                lines[current_line - 1] += lines[current_line];
                lines.erase(lines.begin() + current_line);
                current_line--;
                cursor_pos = prev_len;
            }
        } else if (cmd == ' ') {
            // space character: insert space at cursor
            lines[current_line].insert(cursor_pos, 1, ' ');
            cursor_pos++;
        } else if (std::isalpha(cmd)) {
            // lowercase letter: insert at cursor
            lines[current_line].insert(cursor_pos, 1, cmd);
            cursor_pos++;
        } else {
            // skip other commands (like in the sample input, there are random letters after Q)
        }
        
        // Ensure lines are never empty: if a line is empty, we still have one character (space?) but problem doesn't say.
        // Actually, the problem says length of each string does not exceed 1000 chars, but empty strings are possible.
        // We won't enforce minimum length unless needed.
        
        // After each command, ensure we handle cursor beyond end (for R, X) etc.
    }

    // Output the result
    for (size_t i = 0; i < lines.size(); ++i) {
        std::cout << lines[i];
        if (i != lines.size() - 1) {
            std::cout << "\n";
        }
    }

    return 0;
}
