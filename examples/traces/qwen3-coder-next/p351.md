    return a.x * b.x + a.y * b.y;
}

double cross(const Point& a, const Point& b) {
    return a.x * b.y - a.y * b.x;

double dist(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;

}

// Rotate vector (dx, dy) by angle theta counterclockwise
Point rotate(const Point& p, double theta) {
    double c = cos(theta);
    double s = sin(theta);
    return Point(p.x * c - p.y * s, p.x * s + p.y * c);
}

// Compute angle between two vectors (from v1 to v2), in range [0, 2*PI)
double angleBetween(const Point& v1, const Point& v2) {
    double a1 = atan2(v1.y, v1.x);
    double a2 = atan2(v2.y, v2.x);
    double diff = a2 - a1;
    while (diff < 0) diff += 2 * M_PI;
    while (diff >= 2 * M_PI) diff -= 2 * M_PI;
    return diff;
}

// Returns 2 tangent points: first tangent point on the left, second on the right when looking from P to C
void tangentPoints(const Point& P, const Point& C, double R, Point& T1, Point& T2) {
    double d = dist(P, C);
    if (d < R) {
        // Shouldn't happen per problem statement
        T1 = T2 = P;
        return;
    }
    double angle = atan2(C.y - P.y, C.x - P.x);
    double theta = asin(R / d);
    
    // Left tangent point
    double angle1 = angle + theta;
    T1 = Point(C.x + R * cos(angle1), C.y + R * sin(angle1));
    
    // Right tangent point
    double angle2 = angle - theta;
    T2 = Point(C.x + R * cos(angle2), C.y + R * sin(angle2));
}
