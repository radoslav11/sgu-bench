struct Platform {
    int l, r, h;
    Platform(int l=0, int r=0, int h=0) : l(l), r(r), h(h) {}
    bool containsX(int x) const { return l <= x && x <= r; }
};

struct Lemming {
    int start_time;
    double start_x, start_y;
    int dir; // 1 for right, -1 for left

    int final_time;
    bool reached_home;
};

struct State {

int N, s;

// Check if point (px, py) is on a platform
int findPlatformAt(double px, double py) {
    for (int i = 0; i < M; i++) {
        if (platforms[i].h == py && platforms[i].containsX(px)) {

        }
    }
    return -1;

// Simulate a lemming's path, considering only stopped lemmings that affect it via collisions
// We'll process events in chronological order: drops, platform arrivals, stops, collisions
struct Event {
    double time;
    int type; // 0: drop to platform, 1: reach end of platform, 2: collision with stopped lemming
    int idx; // lemming index
    double x, y;
};

// Helper to compare doubles with tolerance
const double EPS = 1e-9;
bool eq(double a, double b) { return fabs(a - b) < EPS; }
int sgn(double x) { return (x > EPS) - (x < EPS); }

// Simulate a single lemming starting at (sx, sh) at time st, with given dir
// Consider stopped lemmings (positions and times) to compute collisions and direction changes
double simulateLemming(int idx, int st, double sx, double sh, int dir, 
                      const vector<pair<double, double>>& stopped_positions) {
    double cur_time = st;
    double cur_x = sx;
    double cur_y = sh;
    int cur_dir = dir;
    
    // We need to simulate the entire path, handling collisions with stopped lemmings
    // A stopped lemming at (tx, ty) at time tt will cause this lemming to reverse direction
    // when it reaches (tx, ty) at or after time tt
    
    // For efficiency, we'll track events: falling and walking, and check for collisions
    // with stopped lemmings
    while (true) {
        // Check if already at home
        if (eq(cur_y, b) && a == (int)cur_x) {
            return cur_time;
        }
        
        // Check if infinitely falling
        int plat_idx = findPlatformAt(cur_x, cur_y);
        if (plat_idx == -1) {
            // Falling forever
            return -1;
        }
        
        // Check if home is on this platform
        if (platforms[plat_idx].h == b && platforms[plat_idx].containsX(a)) {
            // Home is on this platform, check if we can reach it
            // From current position to home on this platform
            double dist = fabs(a - cur_x);
            double time_to_home = dist;
            return cur_time + time_to_home;
        }
        
        // Find next platform below
        double next_y = -1;
        int next_plat_idx = -1;
        for (int i = 0; i < M; i++) {
            if (platforms[i].h >= cur_y) continue; // must be below
            if (!platforms[i].containsX(cur_x)) continue;
            if (next_y < platforms[i].h) {
                next_y = platforms[i].h;
                next_plat_idx = i;
            }
        }
        
        // Fall to next platform
        double fall_dist = cur_y - next_y;
        double fall_time = fall_dist;
        double land_time = cur_time + fall_time;
        
        // Check for collisions with stopped lemmings during fall
        bool hit_stopped = false;
        double collision_time = land_time;
        double collision_x = cur_x;
        
        for (auto [stop_t, stop_pos] : stopped_positions) {
            double stop_t_real = stop_t;
            double stop_x = stop_pos;
            // If stopped lemming is at (stop_x, cur_y) and we fall from above at time t_stop,
            // we hit it if we reach it at or after stop_t_real
            // But we're falling straight down at current x, so only if x == stop_x
            // Actually, since we're falling vertically from cur_x, only stopped lemmings at (cur_x, y) for y in (next_y, cur_y) matter
            // But the input says "All numbers in the input file are integer" and platforms are at integer heights
            // Since we start at integer coordinates and fall 1 cm/sec, our x stays same during fall
            
            // A stopped lemming at (stop_x, stop_y) at time stop_t will be hit if:
            // - stop_y is between next_y and cur_y (exclusive of cur_y, inclusive of next_y? no, we hit before landing)
            // - stop_x == cur_x
            // - stop_t <= cur_time + (cur_y - stop_y)
            // But note: the stopped lemming stays at (stop_x, stop_y) forever, so if we fall past it, we hit it at time cur_time + (cur_y - stop_y)
            
            if (eq(stop_x, cur_x)) {
                double hit_time = cur_time + (cur_y - stop_y);
                if (hit_time >= stop_t - EPS && hit_time < land_time - EPS) {
                    hit_stopped = true;
                    collision_time = hit_time;
                    collision_x = stop_x;
                    break; // earliest collision wins
                }
            }
        }
        
        if (hit_stopped) {
            // Reverse direction at collision point
            cur_time = collision_time;
            cur_x = collision_x;
            cur_y = cur_y; // still at collision height before falling to next platform
            cur_dir = -cur_dir;
            continue; // restart simulation from collision point
        }
        
        // No collision, land on platform
        cur_time = land_time;
        cur_y = next_y;
        
        // Now walk on platform
        double plat_l = platforms[next_plat_idx].l;
        double plat_r = platforms[next_plat_idx].r;
        
        // Check if home is on this platform and reachable
        if (platforms[next_plat_idx].h == b && platforms[next_plat_idx].containsX(a)) {
            double dist = fabs(a - cur_x);
            return cur_time + dist;
        }
        
        // Walk to the edge in current direction
        double walk_dist;
        if (cur_dir == 1) {
            walk_dist = plat_r - cur_x;
            cur_x = plat_r;
        } else {
            walk_dist = cur_x - plat_l;
            cur_x = plat_l;
        }
        double walk_time = walk_dist;
        cur_time += walk_time;
        
        // Check for collisions during walk
        // A stopped lemming at (stop_x, stop_y) at time stop_t will be hit during walk if:
        // - stop_y == current platform height
        // - stop_x is between cur_x (start) and next_x (end)
        // - stop_t <= cur_time - walk_time + |stop_x - start_x|
        
        bool walk_hit_stopped = false;
        for (auto [stop_t, stop_pos] : stopped_positions) {
            if (eq(platforms[next_plat_idx].h, stop_t)) continue; // wrong height
            if (!eq(platforms[next_plat_idx].h, stop_t)) continue;
            
            double stop_x = stop_pos;
            // Check if stop_x is on this platform
            if (!platforms[next_plat_idx].containsX(stop_x)) continue;
            
            // Check if we would pass it
            if ((cur_dir == 1 && stop_x < cur_x - EPS) || (cur_dir == -1 && stop_x > cur_x + EPS)) continue;
            if ((cur_dir == 1 && stop_x > plat_r + EPS) || (cur_dir == -1 && stop_x < plat_l - EPS)) continue;
            
            // We will hit it at time cur_time - walk_dist + fabs(stop_x - (cur_x - walk_dist * cur_dir))
            double dist_to_stop = fabs(stop_x - (cur_x - (cur_dir == 1 ? plat_r - cur_x : cur_x - plat_l)));
            // Actually simpler: we started at original cur_x, walking to edge
            // The position at time t after starting walk is cur_x + (t - walk_start) * cur_dir
            double walk_start_time = cur_time - walk_dist;
            double hit_time = walk_start_time + fabs(stop_x - (cur_dir == 1 ? cur_x - walk_dist : cur_x + walk_dist));
            // Wait, let me recalculate:
            // At time walk_start_time, we are at pos = cur_x - (cur_dir == 1 ? plat_r - cur_x : cur_x - plat_l)
            // Actually, no: before walk, we were at cur_x, then we walk for walk_dist in direction cur_dir
            // So at time walk_start_time + t, position = cur_x + t * cur_dir (for t in [0, walk_dist])
            // Wait, that's only if we walk from left to right: from plat_l to plat_r, cur_x in [plat_l, plat_r]
            // So if cur_dir == 1: pos = cur_x + t, t in [0, plat_r - cur_x]
            // If cur_dir == -1: pos = cur_x - t, t in [0, cur_x - plat_l]
            
            double t_hit;
            if (cur_dir == 1) {
                t_hit = stop_x - cur_x;
            } else {
                t_hit = cur_x - stop_x;
            }
            double hit_time_walk = walk_start_time + t_hit;
            
            if (hit_time_walk >= stop_t - EPS && hit_time_walk <= cur_time + EPS) {
                walk_hit_stopped = true;
                cur_time = hit_time_walk;
                cur_x = stop_x;
                cur_dir = -cur_dir;
                break;
            }
        }
        
        if (walk_hit_stopped) {
            continue; // restart simulation from new position
        }
        
        // No collision, continue falling from platform end
        cur_y = next_y;
    }
}
