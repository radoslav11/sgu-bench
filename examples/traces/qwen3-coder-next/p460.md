#include <iostream>
#include <string>
#include <vector>

int main() {
    int n;
    std::cin >> n;
    
    for (int i = 0; i < n; i++) {
        std::string word;
        std::cin >> word;
        
        int len = word.length();
        
        // Check if the word ends with 'ch', 'x', 's', or 'o'
        if (len >= 1 && (word[len-1] == 'ch' || word[len-1] == 'x' || word[len-1] == 's' || word[len-1] == 'o')) {
            // But wait, 'ch' is two characters, so need to check properly
            if (len >= 2 && word.substr(len-2) == "ch") {
                word += "es";
            } else if (word[len-1] == 'x' || word[len-1] == 's' || word[len-1] == 'o') {
                word += "es";
            } else {
                word += 's';
            }
        }
        // Check if the word ends with 'f' or 'fe'
        else if (len >= 2 && word.substr(len-2) == "fe") {
            word = word.substr(0, len-2) + "ves";
        } else if (len >= 1 && word[len-1] == 'f') {
            word = word.substr(0, len-1) + "ves";
        }
        // Check if the word ends with 'y'
        else if (len >= 1 && word[len-1] == 'y') {
            word = word.substr(0, len-1) + "ies";
        }
        // Default case: add 's'
        else {
            word += 's';
        }
        
        std::cout << word << std::endl;
    }
    
    return 0;
}
