#include <algorithm>
#include <cctype>
using namespace std;

const int NEG_INF = -1000000000;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int N;
    cin >> N;

    vector<vector<int>> children(N+1);
    if (N > 1) {
        for (int i = 0; i < N-1; i++) {
            int chief;
            cin >> chief;
            children[chief].push_back(i+2);
        }
    }

    // dp0[u]: max matching in subtree u, u not matched to any child
    // dp1[u]: max matching in subtree u, u matched to exactly one child
    vector<int> dp0(N+1, 0);
    vector<int> dp1(N+1, NEG_INF); // initially impossible
    vector<int> best_child(N+1, -1); // for reconstruction

    // Process nodes from N down to 1
    for (int u = N; u >= 1; u--) {
        if (!children[u].empty()) {
            // Compute dp0[u]
            for (int v : children[u]) {
                dp0[u] += max(dp0[v], dp1[v]);
            }

            // Compute dp1[u]: try matching u with each child v
            for (int v : children[u]) {
                int gain = dp0[v] - max(dp0[v], dp1[v]);
                int candidate = dp0[u] + 1 + gain;
                if (candidate > dp1[u]) {
                    dp1[u] = candidate;
                    best_child[u] = v;
                }
            }
        }
        // If no children, dp1[u] remains NEG_INF (impossible)
    }

    int total_matching = max(dp0[1], dp1[1]);
    cout << total_matching * 1000 << '\n';

    // Reconstruct the grant receivers
    vector<bool> grant(N+1, false);
    vector<int> state(N+1, 0); // 0: not matched to parent (in state 0 for the node), 1: matched to parent (in state 1 for the node)

    // For root
    if (dp1[1] > dp0[1]) {
        state[1] = 1;
    } else {
        state[1] = 0;
    }

    queue<int> q;
    q.push(1);

    while (!q.empty()) {
        int u = q.front(); q.pop();
        if (state[u] == 1) {
            // u is matched to best_child[u]
            int v0 = best_child[u];
            grant[v0] = true;
            state[v0] = 0; // v0 is not matched to any of its children
            q.push(v0);
            for (int v : children[u]) {
                if (v == v0) continue;
                if (dp0[v] >= dp1[v]) {
                    state[v] = 0;
                } else {
                    state[v] = 1;
                }
                q.push(v);
            }
        } else {
            for (int v : children[u]) {
                if (dp0[v] >= dp1[v]) {
                    state[v] = 0;
                } else {
                    state[v] = 1;
                }
                q.push(v);
            }
        }
    }

    // Collect grant receivers
    vector<int> recipients;
    for (int i = 1; i <= N; i++) {
        if (grant[i]) {
            recipients.push_back(i);
        }
    }

    // Output in ascending order (already sorted by i)
    for (int i = 0; i < recipients.size(); i++) {
        if (i > 0) cout << " ";
        cout << recipients[i];
    }
    if (recipients.size() > 0) {
        cout << '\n';
    } else {
        cout << '\n';
    }

    return 0;
}
