#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int T;
    cin >> T;

    while (T--) {
        int N;
        cin >> N;
        vector<long long> a(N);
        for (int i = 0; i < N; i++) {
            cin >> a[i];
        }

        // If N < 3, no hills or dales possible
        if (N < 3) {
            cout << "0 0\n";
            continue;
        }

        long long max_hill_height = 0;
        long long max_dale_depth = 0;

        // Scan for hills and dales
        int i = 0;
        while (i < N - 1) {
            // Skip equal adjacent elements
            while (i < N - 1 && a[i] == a[i + 1]) {
                i++;
            }
            if (i >= N - 1) break;

            // Check for potential hill start: increasing sequence
            int j = i;
            // Climb up
            while (j < N - 1 && a[j] < a[j + 1]) {
                j++;
            }
            // Check if we can go down after this climb
            if (j > i && j < N - 1 && a[j] > a[j + 1]) {
                // Found a hill top at j, now descend
                int k = j;
                while (k < N - 1 && a[k] > a[k + 1]) {
                    k++;
                }
                // Calculate hill height: peak - min(start, end)
                long long height = a[j] - min(a[i], a[k]);
                if (height > max_hill_height) {
                    max_hill_height = height;
                }
                i = k; // Continue from end of this hill
            } else {
                // Not a hill, check for potential dale start: decreasing sequence
                j = i;
                // Climb down
                while (j < N - 1 && a[j] > a[j + 1]) {
                    j++;
                }
                // Check if we can go up after this descent
                if (j > i && j < N - 1 && a[j] < a[j + 1]) {
                    // Found a dale bottom at j, now ascend
                    int k = j;
                    while (k < N - 1 && a[k] < a[k + 1]) {
                        k++;
                    }
                    // Calculate dale depth: min(start, end) - bottom
                    long long depth = a[j] - min(a[i], a[k]);
                    // Depth is positive, so use absolute difference
                    depth = abs(depth); // Actually: (min(start, end) - bottom) is positive
                    // But since a[j] is the minimum, depth = min(a[i], a[k]) - a[j]
                    depth = min(a[i], a[k]) - a[j];
                    if (depth > max_dale_depth) {
                        max_dale_depth = depth;
                    }
                    i = k; // Continue from end of this dale
                } else {
                    i++;
                }
            }
        }

        // Alternative approach: simpler state machine
        // Let's redo with a more standard approach: traverse once tracking increasing/decreasing runs
        max_hill_height = 0;
        max_dale_depth = 0;

        for (int i = 0; i < N; i++) {
            // Skip equal elements at start
            if (i > 0 && a[i] == a[i-1]) continue;
            
            // Try to extend as a hill
            if (i + 2 < N) {
                int j = i;
                // Go up
                while (j + 1 < N && a[j] < a[j + 1]) j++;
                if (j > i) { // Have ascending part
                    int k = j;
                    // Go down
                    while (k + 1 < N && a[k] > a[k + 1]) k++;
                    if (k > j) { // Have descending part
                        long long height = a[j] - min(a[i], a[k]);
                        if (height > max_hill_height) max_hill_height = height;
                        i = k - 1;
                        continue;
                    }
                }
            }
        }

        // Try to extend as a dale
        for (int i = 0; i < N; i++) {
            if (i > 0 && a[i] == a[i-1]) continue;
            
            if (i + 2 < N) {
                int j = i;
                // Go down
                while (j + 1 < N && a[j] > a[j + 1]) j++;
                if (j > i) { // Have descending part
                    int k = j;
                    // Go up
                    while (k + 1 < N && a[k] < a[k + 1]) k++;
                    if (k > j) { // Have ascending part
                        long long depth = min(a[i], a[k]) - a[j];
                        if (depth > max_dale_depth) max_dale_depth = depth;
                        i = k - 1;
                        continue;
                    }
                }
            }
        }

        cout << max_hill_height << " " << max_dale_depth << "\n";
    }

    return 0;
}
