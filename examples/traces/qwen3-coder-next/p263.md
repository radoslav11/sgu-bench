#include <bits/stdc++.h>
using namespace std;

// Fast I/O
inline void read(int &x) {
    x = 0;
    char c = getchar();
    while (c < '0' || c > '9') c = getchar();
    while (c >= '0' && c <= '9') {
        x = x * 10 + (c - '0');
        c = getchar();
    }
}

inline void read(long long &x) {
    x = 0;
    char c = getchar();
    while (c < '0' || c > '9') c = getchar();
    while (c >= '0' && c <= '9') {
        x = x * 10 + (c - '0');
        c = getchar();
    }
}

inline void read_string(char *s, int &len) {
    len = 0;
    char c = getchar();
    while (c == ' ' || c == '\n') c = getchar();
    while (c != ' ' && c != '\n' && c != '\r' && c != EOF) {
        s[len++] = c;
        c = getchar();
    }
    s[len] = '\0';
}

const int MAXN = 1000000;
const int MAXcells = 1000001;

// Segment tree for union-find style merging: track leftmost and rightmost cell with cubes in range
struct DSU {
    int par[MAXcells], lft[MAXcells], rgt[MAXcells];
    long long cubes[MAXcells]; // cubes on cell i

    void init() {
        for (int i = 0; i < MAXcells; ++i) {
            par[i] = i;
            lft[i] = i;
            rgt[i] = i;
            cubes[i] = 0;
        }
    }

    int find(int x) {
        if (par[x] != x) par[x] = find(par[x]);
        return par[x];
    }

    void merge(int x, int y) {
        x = find(x), y = find(y);
        if (x == y) return;
        if (x > y) swap(x, y);
        par[y] = x;
        lft[x] = min(lft[x], lft[y]);
        rgt[x] = max(rgt[x], rgt[y]);
        cubes[x] += cubes[y];
        cubes[y] = 0; // not used anymore
    }

    void add_cubes(int x, long long c) {
        cubes[x] += c;
        if (x > 1 && cubes[x - 1] > 0) merge(x - 1, x);
        if (x < MAXcells - 1 && cubes[x + 1] > 0) merge(x, x + 1);
    }
} dsu;

struct Tower {
    int l, r;
    long long total;
};

// Mapping from tower id to actual tower info
vector<Tower> towers;
map<int, int> cell_to_tower_id; // cell -> tower id (representative)
vector<int> tower_id_to_index; // tower id -> index in towers vector

int main() {
    dsu.init();
    
    char cmd[20];
    int N;
    read(N);
    
    // Pre-allocate towers vector
    towers.reserve(N / 2);
    tower_id_to_index.resize(MAXcells, -1);
    
    auto rebuild_towers_list = [&]() {
        towers.clear();
        tower_id_to_index.assign(MAXcells, -1);
        for (int i = 1; i < MAXcells; ++i) {
            if (dsu.cubes[i] > 0) {
                int rep = dsu.find(i);
                if (tower_id_to_index[rep] == -1) {
                    tower_id_to_index[rep] = (int)towers.size();
                    towers.push_back({dsu.lft[rep], dsu.rgt[rep], dsu.cubes[rep]});
                }
            }
        }
    };
    
    auto get_tower_info = [&](int t) -> pair<int, int> {
        // t is 1-indexed; return (left, right) cell indices
        if (t <= 0 || t > (int)towers.size()) return {-1, -1};
        const auto &tow = towers[t-1];
        return {tow.l, tow.r};
    };
    
    auto get_cell_from_tower_col = [&](int t, int col) -> int {
        auto [l, r] = get_tower_info(t);
        if (l == -1) return -1;
        return l + col - 1;
    };
    
    for (int iter = 0; iter < N; ++iter) {
        int len;
        read_string(cmd, len);
        
        if (cmd[0] == 'p' && cmd[3] == 't') {
            // tput <t> <x> <c>
            int t, x;
            long long c;
            read(t); read(x); read(c);
            int cell = get_cell_from_tower_col(t, x);
            if (cell != -1) {
                dsu.add_cubes(cell, c);
            }
        } else if (cmd[0] == 'p') {
            // put <x> <c>
            int x;
            long long c;
            read(x); read(c);
            dsu.add_cubes(x, c);
        } else if (cmd[0] == 't' && cmd[1] == 'o') {
            // towers
            rebuild_towers_list();
            printf("%d towers\n", (int)towers.size());
        } else if (cmd[0] == 'c' && cmd[1] == 'u') {
            // cubes <t>
            int t;
            read(t);
            rebuild_towers_list();
            if (t <= (int)towers.size()) {
                printf("%lld cubes in %dth tower\n", towers[t-1].total, t);
            }
        } else if (cmd[0] == 'l' && cmd[1] == 'e') {
            // length <t>
            int t;
            read(t);
            rebuild_towers_list();
            if (t <= (int)towers.size()) {
                int len = towers[t-1].r - towers[t-1].l + 1;
                printf("length of %dth tower is %d\n", t, len);
            }
        } else if (cmd[0] == 't' && cmd[1] == 'c') {
            // tcubes <t> <x>
            int t, x;
            read(t); read(x);
            rebuild_towers_list();
            int cell = get_cell_from_tower_col(t, x);
            if (cell != -1 && dsu.find(cell) == dsu.find(cell)) {
                printf("%lld cubes in %dth column of %dth tower\n", dsu.cubes[cell], x, t);
            }
        }
    }
    
    return 0;
}
