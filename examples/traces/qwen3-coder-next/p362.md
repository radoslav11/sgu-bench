```cpp
#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main() {
    int N, M;
    cin >> N >> M;
    
    int is, js;
    cin >> is >> js;
    
    // Convert to 0-indexed for internal use
    is--; 
    js--;
    
    // Grid to track destroyed buildings: true if destroyed
    vector<vector<bool>> destroyed(N, vector<bool>(M, false));
    
    string moves;
    
    // Directions in order: Down, Left, Up, Right
    // Corresponding to: (i+1, j), (i, j-1), (i-1, j), (i, j+1)
    vector<pair<int, int>> dirs = {{1, 0}, {0, -1}, {-1, 0}, {0, 1}};
    vector<char> dirChars = {'D', 'L', 'U', 'R'};
    
    int i = is, j = js;
    
    while (true) {
        // Mark current building as destroyed (will explode after robot leaves)
        destroyed[i][j] = true;
        
        bool moved = false;
        
        // Try directions in order: D, L, U, R
        for (int k = 0; k < 4; k++) {
            int ni = i + dirs[k].first;
            int nj = j + dirs[k].second;
            
            // Check if neighbor is within bounds and not destroyed
            if (ni >= 0 && ni < N && nj >= 0 && nj < M && !destroyed[ni][nj]) {
                // Move to this neighbor
                moves += dirChars[k];
                i = ni;
                j = nj;
                moved = true;
                break;
            }
        }
        
        // If no move possible, robot stays and explodes with the building
        if (!moved) {
            break;
        }
    }
    
    cout << moves << endl;
    
    return 0;
}
```
