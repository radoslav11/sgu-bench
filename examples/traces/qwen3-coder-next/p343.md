```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
#include <algorithm>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);
    
    int T, N;
    std::cin >> T >> N;
    
    std::vector<int> quantities(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> quantities[i];
    }
    
    // Read prices: T+1 lines, each with N prices
    std::vector<std::vector<double>> prices(T + 1, std::vector<double>(N));
    for (int t = 0; t <= T; ++t) {
        for (int i = 0; i < N; ++i) {
            std::cin >> prices[t][i];
        }
    }
    
    // Calculate today's portfolio value (VP)
    double VP = 0.0;
    for (int i = 0; i < N; ++i) {
        VP += quantities[i] * prices[0][i];
    }
    
    // Calculate asset returns: for t from 1 to T
    // r_t,i = (price_{t-1,i} - price_t,i) / price_t,i
    std::vector<std::vector<double>> returns(T, std::vector<double>(N));
    for (int t = 1; t <= T; ++t) {
        for (int i = 0; i < N; ++i) {
            returns[t-1][i] = (prices[t-1][i] - prices[t][i]) / prices[t][i];
        }
    }
    
    // Calculate mean returns for each asset (m_i)
    std::vector<double> mean_returns(N, 0.0);
    for (int i = 0; i < N; ++i) {
        double sum = 0.0;
        for (int t = 0; t < T; ++t) {
            sum += returns[t][i];
        }
        mean_returns[i] = sum / T;
    }
    
    // Calculate covariance matrix S (N x N)
    std::vector<std::vector<double>> cov_matrix(N, std::vector<double>(N, 0.0));
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            double sum = 0.0;
            for (int t = 0; t < T; ++t) {
                double diff_i = returns[t][i] - mean_returns[i];
                double diff_j = returns[t][j] - mean_returns[j];
                sum += diff_i * diff_j;
            }
            cov_matrix[i][j] = sum / (T - 1); // sample covariance
        }
    }
    
    // Calculate weights w_i = V_i / V_P
    std::vector<double> weights(N);
    for (int i = 0; i < N; ++i) {
        weights[i] = (quantities[i] * prices[0][i]) / VP;
    }
    
    // Calculate portfolio mean return: m_P = sum(w_i * m_i)
    double mP = 0.0;
    for (int i = 0; i < N; ++i) {
        mP += weights[i] * mean_returns[i];
    }
    
    // Calculate portfolio std deviation: s_P = sqrt(W^T * S * W)
    // First compute S * W
    std::vector<double> Sw(N, 0.0);
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            Sw[i] += cov_matrix[i][j] * weights[j];
        }
    }
    
    // Then compute W^T * (S * W)
    double WtSw = 0.0;
    for (int i = 0; i < N; ++i) {
        WtSw += weights[i] * Sw[i];
    }
    
    double sP = std::sqrt(WtSw);
    
    // Calculate VaR = -VP * (mP - 1.644854 * sP)
    double VaR = -VP * (mP - 1.644854 * sP);
    
    // Ensure non-negative (though mathematically it could be negative, in practice VaR is positive)
    if (VaR < 0) {
        VaR = 0.0;
    }
    
    std::cout << std::fixed << std::setprecision(2) << VaR << std::endl;
    
    return 0;
}
```
