#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <cctype>
#include <cmath>

using namespace std;

int main() {
    string n;
    cin >> n;
    
    // Sort digits to get the smallest permutation first
    sort(n.begin(), n.end());
    
    // Try all permutations until we find one divisible by 17
    // Use next_permutation which generates in lexicographic order
    do {
        // Skip permutations that start with '0'
        if (n[0] == '0') {
            continue;
        }
        
        // Convert string to number and check divisibility by 17
        long long num = stoll(n);
        if (num % 17 == 0) {
            cout << n << endl;
            return 0;
        }
    } while (next_permutation(n.begin(), n.end()));
    
    // If no permutation found
    cout << -1 << endl;
    return 0;
}
