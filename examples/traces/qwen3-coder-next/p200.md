#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <map>

using namespace std;

// Generate first t primes
vector<int> generatePrimes(int t) {
    vector<int> primes;
    if (t == 0) return primes;
    primes.push_back(2);
    int num = 3;
    while (primes.size() < t) {
        bool isPrime = true;
        for (int p : primes) {
            if (p * p > num) break;
            if (num % p == 0) {
                isPrime = false;
                break;
            }
        }
        if (isPrime) {
            primes.push_back(num);
        }
        num += 2;
    }
    return primes;
}

// Factorize x using the first t primes and return exponent vector mod 2
vector<int> getExponentVector(long long x, const vector<int>& primes) {
    int t = primes.size();
    vector<int> expVec(t, 0);
    for (int i = 0; i < t; i++) {
        int p = primes[i];
        while (x % p == 0) {
            expVec[i]++;
            x /= p;
        }
        expVec[i] %= 2;
    }
    // Note: if x > 1 after factorization, then it has a prime factor not in the first t primes.
    // But according to the problem, all prime factors are from the first t primes, so x should be 1.
    return expVec;
}

int main() {
    int t, m;
    cin >> t >> m;
    
    vector<int> primes = generatePrimes(t);
    
    vector<vector<int>> expVectors;
    for (int i = 0; i < m; i++) {
        long long b;
        cin >> b;
        expVectors.push_back(getExponentVector(b, primes));
    }
    
    // We need to find the number of non-empty subsets S such that the XOR (sum mod 2) of
    // the exponent vectors in S is the zero vector.
    // This is equivalent to finding the number of non-empty solutions to a system of linear equations over GF(2).
    // Let A be a matrix where each row is an exponent vector.
    // We want the number of subsets S such that the sum (mod 2) of rows indexed by S is 0.
    // This is 2^(m - rank) - 1, where rank is the rank of the matrix over GF(2).
    
    // Create augmented matrix for Gaussian elimination over GF(2)
    // Each row is an exponent vector (length t), and we'll do elimination to find rank.
    vector<vector<int>> mat = expVectors; // m rows, t columns
    
    int rank = 0;
    // For each column j from 0 to t-1
    for (int col = 0, row = 0; col < t && row < m; col++) {
        // Find pivot row with 1 in column col
        int pivot = -1;
        for (int i = row; i < m; i++) {
            if (mat[i][col] == 1) {
                pivot = i;
                break;
            }
        }
        if (pivot == -1) continue;
        
        // Swap rows pivot and row
        swap(mat[pivot], mat[row]);
        
        // Eliminate all other rows
        for (int i = 0; i < m; i++) {
            if (i != row && mat[i][col] == 1) {
                for (int j = 0; j < t; j++) {
                    mat[i][j] ^= mat[row][j];
                }
            }
        }
        
        row++;
        rank++;
    }
    
    // Number of solutions = 2^(m - rank)
    // But we need non-empty subsets, so subtract 1 (the empty subset)
    long long solutions = 1LL << (m - rank);
    solutions -= 1;
    
    cout << solutions << endl;
    
    return 0;
}
