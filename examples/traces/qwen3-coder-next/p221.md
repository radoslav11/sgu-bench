        *                i=1: j=2: dp_b[2] = 0 + dp_b[1]*(1-1)=0; j=1: dp_b[1] = 1 + dp_b[0]*(1-0)=1+1=2 -> dp_b=[1,2,0] -> but we only need up to size, so dp_b[0]=1, dp_b[1]=2, dp_b[2]=0.
        *                Actually, after i=1, dp_b[2] = 0 + 1*0 = 0, and dp_b[1] = 1 (from previous) + 1*1 = 2.
        *                So dp_b = [1,2,0] -> but we can use a vector of size 3.
        *            white: s=1 -> lengths: min(1, 2)+1 = 1+1=2? -> min(1, 2*2-2-1=1)+1 = 1+1=2.
        *                white = [2]
        *                dp_w[0]=1
        *                i=0: j=1: dp_w[1] = 0 + 1*(2-0)=2.
        *                dp_w = [1,2]
        *            ans for k=2: i=0: dp_b[0]*dp_w[2] -> dp_w[2]=0; i=1: dp_b[1]*dp_w[1]=2*2=4; i=2: dp_b[2]*dp_w[0]=0.
        *            -> 4. Correct.
        *
        *        So we'll implement accordingly.
        *
        *        Note: the inner loop must be from high to low for j to avoid using the updated dp_b[j-1] in the same iteration.
        *        We did: for j from i+1 down to 1.
        *
        *        Let's code.
        */
        
        #include <iostream>
        #include <vector>
        #include <algorithm>
        using namespace std;
        
        int main() {
            int n, k;
            cin >> n >> k;
            
            // If k is 0, there's exactly 1 way.
            // But our method will handle it, so no need for special case.
            
            // Generate lengths for black squares (even i+j)
            vector<int> black;
            for (int s = 0; s <= 2*n-2; s += 2) {
                int len = min(s, 2*n-2 - s) + 1;
                black.push_back(len);
            }
            
            // Generate lengths for white squares (odd i+j)
            vector<int> white;
            for (int s = 1; s <= 2*n-2; s += 2) {
                int len = min(s, 2*n-2 - s) + 1;
                white.push_back(len);
            }
            
            // Sort both
            sort(black.begin(), black.end());
            sort(white.begin(), white.end());
            
            // dp_b[i] = number of ways to place i non-attacking bishops on black squares
            vector<long long> dp_b(black.size() + 1, 0);
            dp_b[0] = 1;
            for (int i = 0; i < black.size(); i++) {
                // update dp_b from i+1 down to 1
                for (int j = i+1; j >= 1; j--) {
                    // dp_b[j] += dp_b[j-1] * (black[i] - (j-1))
                    dp_b[j] += dp_b[j-1] * (black[i] - (j-1));
                }
            }
            
            // dp_w for white
            vector<long long> dp_w(white.size() + 1, 0);
            dp_w[0] = 1;
            for (int i = 0; i < white.size(); i++) {
                for (int j = i+1; j >= 1; j--) {
                    dp_w[j] += dp_w[j-1] * (white[i] - (j-1));
                }
            }
            
            long long ans = 0;
            for (int i = 0; i <= k; i++) {
                if (i < (int)dp_b.size() && (k - i) < (int)dp_w.size()) {
                    ans += dp_b[i] * dp_w[k - i];
                }
            }
            
            cout << ans << endl;
            
            return 0;
        }
