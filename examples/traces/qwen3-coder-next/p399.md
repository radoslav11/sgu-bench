To solve this problem, we need to connect at least two craters on an island to the ocean using the minimal number of pipe segments. The island is represented as a grid where 'X' marks craters and '.' marks non-crater areas. The solution involves finding the optimal way to connect two craters (either separately or via a shared pipe) to the ocean (grid boundaries) with the least number of pipe segments.

### Approach
1. **Identify Craters**: Use BFS to identify connected components of 'X' (craters) in the grid.
2. **Compute Adjacent Cells**: For each crater, determine adjacent '.' cells that could connect to the pipe.
3. **Ocean Distance Calculation**: Perform a multi-source BFS from all boundary '.' cells to compute the minimal distance from each '.' cell to the ocean.
4. **Minimal Distance for Craters**: For each crater, find the minimal distance from any of its adjacent cells to the ocean.
5. **Pairwise Crater Connection**: For every pair of craters, compute:
   - **Option 1**: Sum of their individual distances to the ocean.
   - **Option 2**: Distance between the craters plus the minimal distance of either crater to the ocean.
6. **Optimal Connection**: Select the pair and option with the minimal total pipe segments.
7. **Reconstruct Pipe**: Based on the optimal choice, reconstruct the pipe segments on the grid.

#include <queue>
#include <climits>
#include <algorithm>
#include <set>
#include <tuple>

                queue<pair<int, int>> q;
                q.push({i, j});
                visited[i][j] = true;
                while (!q.empty()) {
                    auto [x, y] = q.front();
                    q.pop();
                    comp.push_back({x, y});
                    for (int d = 0; d < 4; d++) {
                        int nx = x + dx[d];

                            visited[nx][ny] = true;
                            q.push({nx, ny});
                        }
                    }
                }
                components.push_back(comp);
                int comp_idx = components.size() - 1;
                for (auto [x, y] : comp) {
                    comp_id[x][y] = comp_idx;
                }
            }

                    adj_cells[idx].insert({nx, ny});
                }

            if (i == 0 || i == n-1 || j == 0 || j == m-1) {
                if (grid[i][j] == '.') {

                    q_0.push({i, j, 1});

                    q_0.push({nx, ny, d+1});
                }

            }

            queue<pair<int, int>> q;

                for (int d_dir = 0; d_dir < 4; d_dir++) {

                        q.push({nx, ny});
                    }
                }

        }

    int best_i = -1, best_j = -1;
    int best_option = 0;

        for (int j = i+1; j < K; j++) {
            if (d_val[j] == INF) continue;
            int option1 = d_val[i] + d_val[j];
            int option2 = dist_between[i][j] + min(d_val[i], d_val[j]);
            int cost = min(option1, option2);
            if (cost < best_cost) {
                best_cost = cost;
                best_i = i;
                best_j = j;
                best_option = (option1 <= option2) ? 1 : 2;
            }
        }

    if (best_option == 1) {

        pair<int, int> cell_i = {-1, -1};
        pair<int, int> cell_j = {-1, -1};

            if (distOcean[x][y] == d_val[i]) {

                break;

            if (distOcean[x][y] == d_val[j]) {
                cell_j = {x, y};
                break;

            if (x == 0 || x == n-1 || y == 0 || y == m-1) break;
            int nx = -1, ny = -1;
            for (int d = 0; d < 4; d++) {

                    nx = cx;
                    ny = cy;
                    break;
                }

            if (nx == -1) break;
            x = nx;
            y = ny;
        }

        x = cell_j.first, y = cell_j.second;
        while (true) {
            pipe[x][y] = true;
            if (x == 0 || x == n-1 || y == 0 || y == m-1) break;
            int nx = -1, ny = -1;
            for (int d = 0; d < 4; d++) {
                int cx = x + dx[d];
                int cy = y + dy[d];
                if (cx >= 0 && cx < n && cy >= 0 && cy < m && grid[cx][cy] == '.' && distOcean[cx][cy] == distOcean[x][y] - 1) {

                    ny = cy;
                    break;

            for (auto [x, y] : adj_cells[i]) {
                for (auto [a, b] : adj_cells[j]) {
                    if (dist_between[i][j] == dist_from_i[a][b]) {
                        if (dist_from_i[x][y] + d_val[i] < min_dist) {
                            min_dist = dist_from_i[x][y] + d_val[i];
                            cell_i = {x, y};
                            cell_j = {a, b};
                        }
                    }
                }
            }
        } else {
            for (auto [x, y] : adj_cells[j]) {
                for (auto [a, b] : adj_cells[i]) {
                    if (dist_between[i][j] == dist_from_i[a][b]) {
                        if (dist_from_i[x][y] + d_val[j] < min_dist) {
                            min_dist = dist_from_i[x][y] + d_val[j];
                            cell_i = {a, b};
                            cell_j = {x, y};
                        }
                    }
                }
            }
        }

        int x = cell_i.first, y = cell_i.second;
        while (true) {
            pipe[x][y] = true;
            if (x == 0 || x == n-1 || y == 0 || y == m-1) break;
            int nx = -1, ny = -1;
            for (int d = 0; d < 4; d++) {
                int cx = x + dx[d];
                int cy = y + dy[d];
                if (cx >= 0 && cx < n && cy >= 0 && cy < m && grid[cx][cy] == '.' && distOcean[cx][cy] == distOcean[x][y] - 1) {
                    nx = cx;
                    ny = cy;
                    break;
                }

            if (nx == -1) break;
            x = nx;

        dist_from_cell_i[cell_i.first][cell_i.second] = 0;
        q.push(cell_i);
        while (!q.empty()) {
            auto [x, y] = q.front();
            q.pop();
            for (int d = 0; d < 4; d++) {

                    dist_from_cell_i[nx][ny] = dist_from_cell_i[x][y] + 1;
                    q.push({nx, ny});
                }
            }
        }
        x = cell_j.first, y = cell_j.second;
        while (true) {
            pipe[x][y] = true;
            if (x == cell_i.first && y == cell_i.second) break;
            int nx = -1, ny = -1;
            for (int d = 0; d < 4; d++) {
                int cx = x + dx[d];
                int cy = y + dy[d];
                if (cx >= 0 && cx < n && cy >= 0 && cy < m && grid[cx][cy] == '.' && dist_from_cell_i[cx][cy] == dist_from_cell_i[x][y] - 1) {

                    ny = cy;
                    break;

    for (int i = 0; i < n; i++) {

            } else if (pipe[i][j]) {
                cout << '*';
            } else {
                cout << '.';
            }
        }
        cout << '\n';
    }

    return 0;
}
