```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    
    vector<vector<int>> grid(n, vector<int>(m));
    for (int i = 0; i < n; i++) {
        string s;
        cin >> s;
        for (int j = 0; j < m; j++) {
            grid[i][j] = s[j] - '0';
        }
    }
    
    // Calculate total surface area
    // Each building contributes:
    // - Top: 1 if height > 0
    // - Bottom: 1 if height > 0
    // - Sides: for each direction, the exposed surface is max(0, height - neighbor_height)
    
    int surface = 0;
    
    // Directions: up, right, down, left
    vector<int> dx = {-1, 0, 1, 0};
    vector<int> dy = {0, 1, 0, -1};
    
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            if (grid[i][j] == 0) continue;
            
            // Top and bottom
            surface += 2; // top and bottom faces
            
            // Check 4 neighbors
            for (int d = 0; d < 4; d++) {
                int ni = i + dx[d];
                int nj = j + dy[d];
                
                if (ni >= 0 && ni < n && nj >= 0 && nj < m) {
                    // Exposed side = current height - neighbor height (if positive)
                    surface += max(0, grid[i][j] - grid[ni][nj]);
                } else {
                    // At the boundary, entire side is exposed
                    surface += grid[i][j];
                }
            }
        }
    }
    
    cout << surface << endl;
    
    return 0;
}
```
