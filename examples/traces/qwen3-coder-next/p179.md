    //                //        How to compute balance after [0, p-1]? 
    //                //            We can compute it in a forward pass up to p-1.
    //                //
    //                //   However, note: in the backward pass we did, we can also compute the balance at p-1.
    //                //        Let's recompute: 
    //                //            We can do a forward pass to get the balance at p-1.
    //                //
    //                //   Steps:
    //                //        Step 1: Compute bal0 = balance after [0, p-1] (by traversing from 0 to p-1).
    //                //        Then b = bal0 - 1.
    //                //
    //                //   But note: the entire sequence balance is 0, and we are changing the sequence.
    //                //        Original balance at p: 
    //                //            bal0 + 1 (because s[p] was '(')
    //                //        Then the suffix [p+1, n-1] has balance change = - (bal0+1) to get to 0.
    //                //        Now, after change, balance at p: bal0 - 1, so the suffix needs to have balance change = - (bal0 - 1) = 1 - bal0.
    //                //        And the number of '(' and ')' in the suffix must satisfy: 
    //                //            (# '(' in suffix) - (# ')' in suffix) = 1 - bal0.
    //                //        Also, (# '(' in suffix) + (# ')' in suffix) = n - p - 1.
    //                //        => # '(' = (n - p - 1 + 1 - bal0) / 2 = (n - p - bal0) / 2
    //                //           # ')' = (n - p - 1 - (1 - bal0)) / 2 = (n - p - 2 + bal0) / 2
    //                //        And we know these must be integers and non-negative.
    //                //
    //                //   However, we already know that a solution exists because we found p.
    //                //
    //                //   So in code:
    //                //        int bal0 = 0;
    //                //        for (int i = 0; i < p; i++) {
    //                //            if (s[i] == '(') bal0++; else bal0--;
    //                //        }
    //                //        int cur = bal0 - 1;  // balance after position p
    //                //        int rem_open = (n - p - bal0) / 2;   // because originally in [0,p] there were (p+1 - (# close in [0,p-1])) ... 
    //                //        But wait, we can compute rem_open from the original counts:
    //                //            total '(' in the whole sequence = n/2.
    //                //            '(' in [0, p-1] = (bal0 + (p)) / 2? 
    //                //                bal0 = open0 - close0, and open0 + close0 = p, so open0 = (p + bal0) / 2.
    //                //            Then at p, originally there was one '(', so total '(' in [0, p] = open0 + 1.
    //                //            Now, after change, we remove that '(', so remaining '(' to place = n/2 - open0.
    //                //            But in the suffix, we want to place open0_remaining = n/2 - open0.
    //                //            However, we also know that in the suffix [p+1, n-1] originally had:
    //                //                open_suf = n/2 - open0 - 1
    //                //                close_suf = n/2 - (p - open0)   [because close in [0,p-1] = p - open0, and at p was '(' so no close]
    //                //            Now, we want in the suffix: 
    //                //                open_suf' = open_suf + 1 = n/2 - open0 - 1 + 1 = n/2 - open0
    //                //                close_suf' = close_suf - 1 = n/2 - (p - open0) - 1
    //                //            So rem_open = n/2 - open0, rem_close = n/2 - p + open0 - 1.
    //                //        And indeed, 
    //                //            rem_open + rem_close = n/2 - open0 + n/2 - p + open0 - 1 = n - p - 1, correct.
    //                //            rem_open - rem_close = (n/2 - open0) - (n/2 - p + open0 - 1) = -2*open0 + p + 1.
    //                //        But we also have: 
    //                //            bal0 = open0 - (p - open0) = 2*open0 - p, so 2*open0 = bal0 + p.
    //                //            => rem_open - rem_close = - (bal0 + p) + p + 1 = 1 - bal0, which matches the requirement.
    //                //
    //                //   So we can compute:
    //                //        open0 = (p + bal0) / 2;   [must be integer]
    //                //        rem_open = n/2 - open0;
    //                //        rem_close = n/2 - (p - open0) - 1;
    //                //
    //                //   However, we can avoid this by simply using the counts from the original string for the suffix, and then adjust:
    //                //        In the original string, for [p+1, n-1]:
    //                //            open_orig = count of '(' in [p+1, n-1]
    //                //            close_orig = count of ')' in [p+1, n-1]
    //                //        Then rem_open = open_orig + 1, rem_close = close_orig - 1.
    //                //
    //                //   So we can do:
    //                //        int open_orig = 0, close_orig = 0;
    //                //        for (int i = p+1; i < n; i++) {
    //                //            if (s[i]=='(') open_orig++; else close_orig++;
    //                //        }
    //                //        int rem_open = open_orig + 1;
    //                //        int rem_close = close_orig - 1;
    //                //        int cur = bal0 - 1;
    //                //
    //                //   But note: we have to compute bal0, which is the balance after [0, p-1].
    //                //
    //                //   Alternatively, we can compute the balance after [0, p-1] in the original string by:
    //                //        int bal0 = 0;
    //                //        for (int i = 0; i < p; i++) {
    //                //            bal0 += (s[i]=='(' ? 1 : -1);
    //                //        }
    //                //
    //                //   Then we set the suffix.
    //            }
    //
    //        However, note: the problem constraints are small (n<=10000), so a few passes are acceptable.
    //
    //        Let's implement accordingly:
    //          Step 1: Find p (the rightmost index such that s[p]=='(' and the backward balance > 0).
    //          Step 2: If not found, output "No solution".
    //          Step 3: Change s[p] to ')'.
    //          Step 4: Compute bal0 = balance after [0, p-1] (in the original string).
    //          Step 5: cur = bal0 - 1.
    //          Step 6: Count open_orig and close_orig in [p+1, n-1] (original string).
    //          Step 7: rem_open = open_orig + 1, rem_close = close_orig - 1.
    //          Step 8: For i from p+1 to n-1:
    //                   if (rem_open > 0 && rem_close >= cur + 1) {
    //                       s[i] = '(';
    //                       cur++;
    //                       rem_open--;
    //                   } else {
    //                       s[i] = ')';
    //                       cur--;
    //                       rem_close--;
    //                   }
    //
    //        Why the condition "rem_close >= cur + 1"? 
    //            Because after placing '(', the current balance becomes cur+1, and we need to have at least cur+1 closing brackets to close them all (since each closing bracket reduces balance by 1, and we need to get to 0, so we need exactly cur+1 closing brackets in the remaining part? But not exactly, because we might have more open brackets that we will close later, but the condition is: the maximum drop we can do is rem_close, so we require cur+1 <= rem_close.
    //            However, note: we also have rem_open open brackets that will add to the balance, so the net drop is rem_close - rem_open, but we require that at no point the balance drops below 0.
    //            The safe condition is: after placing '(', the current balance is cur+1, and the remaining part must be able to be balanced without dropping below 0. The necessary and sufficient condition is that the number of closing brackets available (rem_close) is at least the current balance (cur+1), because even if we use all closing brackets next, the balance would drop to cur+1 - rem_close, and we require that to be >=0? But no, because we might interleave.
    //            Actually, the known greedy for valid parentheses generation: 
    //                We can place '(' if the number of remaining closing brackets is at least the current balance + 1.
    //            Why? 
    //                After placing '(', balance becomes cur+1.
    //                Then, to be able to close all open brackets, we need at least cur+1 closing brackets in the entire suffix.
    //                But we have rem_close closing brackets available, so we require rem_close >= cur+1.
    //            And for ')', we can only place if cur > 0.
    //            But in our algorithm, we try '(' first, and only if the condition for '(' fails, we try ')', and the condition for ')' is cur > 0.
    //            However, note: if rem_close < cur+1, then we cannot place '(' because even if we use all closing brackets, we cannot close the current open brackets (cur+1) since rem_close < cur+1.
    //            And if cur > 0, we can place ')', because then balance becomes cur-1 which is nonnegative.
    //
    //        But what if both conditions fail? 
    //            rem_open>0 and rem_close>=cur+1 fails -> then we check: if cur>0, we place ')'.
    //            But is it possible that rem_open>0, rem_close < cur+1, and cur==0? 
    //                Then we cannot place '(' (because rem_close < 1) and cannot place ')' (because cur==0), so it would be invalid.
    //            However, our earlier steps guarantee that a solution exists, so this should not happen.
    //
    //        Let's test with the sample:
    //            Input: "(())()"
    //            n=6.
    //            Find p from the right:
    //                i=5: s[5]=')' -> bal=1
    //                i=4: s[4]='(' -> bal=0 -> not >0, so bal becomes 0-1 = -1? 
    //                But our algorithm: 
    //                    bal=0
    //                    i=5: s[5]==')' -> bal=1
    //                    i=4: s[4]=='(' -> bal=1-1=0; then check: if s[4]=='(' and bal>0? -> 0>0 false.
    //                    i=3: s[3]==')' -> bal=0+1=1
    //                    i=2: s[2]==')' -> bal=1+1=2
    //                    i=1: s[1]=='(' -> bal=2-1=1, and bal>0 -> p=1.
    //                So p=1.
    //            Change s[1] to ')', so the string becomes: "())()()"
    //                But wait, the original s = "(())()": 
    //                    index0: '(', index1: '(', index2: ')', index3: ')', index4: '(', index5: ')'
    //                After changing index1 to ')': s = "())()()" -> but that's 7 characters? 
    //                Actually, n=6, so: 
    //                    original: [0:'(', 1:'(', 2:')', 3:')', 4:'(', 5:')']
    //                After: [0:'(', 1:')', 2:')', 3:')', 4:'(', 5:')'] -> "())())" -> but that's not valid and length 6.
    //                But the expected output is "()(())", which is different.
    //
    //        What's the lexicographic order:
    //            "(())()" -> 
    //            next: 
    //                We want the next valid sequence.
    //                "(" then we have 5 more.
    //                Possibilities:
    //                  "(()())" -> but length 6: 
    //                  "(())()" -> 
    //                  "()()()" -> 
    //                  "()(())" -> 
    //                  "(())()" < "()(())" lexicographically? 
    //                      first char: '('
    //                      second char: '(' vs ')' -> '(' < ')', so "(...)" < "()"... 
    //                  So "(())()" < "()()()" and "(())()" < "()(())", but is there a sequence between?
    //                  The next should be "()(())", but how?
    //
    //        Let me list all valid sequences of length 6 in lex order:
    //            1. ((()))
    //            2. (()())
    //            3. (())()
    //            4. ()(())
    //            5. ()()()
    //            So the next after 3 is 4: "()(())"
    //
    //        How to get from "(())()" to "()(())":
    //            We see that at position 1 (0-indexed), the original has '(', but in the next sequence it becomes ')', and then we adjust.
    //            Specifically:
    //                Original: ( ( ) ) ( ) 
    //                Next:     ( ) ( ( ) ) 
    //            So p=1 (the second character) is changed from '(' to ')', and then the suffix [2,5] is changed from ")()(" to "(())" but wait, the suffix is 4 characters.
    //
    //        In our algorithm for p=1:
    //            bal0 = balance after [0,0] = 1 (because s[0]=='(')
    //            cur = 1 - 1 = 0.
    //            In the original, suffix [2,5] = ")()(" -> 
    //                open_orig: count '(' in [2,5]: s[2]=')', s[3]=')', s[4]='(', s[5]=')' -> open_orig=1, close_orig=3.
    //            Then rem_open = 1+1=2, rem_close=3-1=2.
    //            Now, build the suffix:
    //                i=2: 
    //                    rem_open=2>0, and rem_close=2 >= cur+1=1 -> true, so put '(' -> s[2]='(', cur=1, rem_open=1.
    //                i=3:
    //                    rem_open=1>0, rem_close=2 >= 1+1=2 -> true, so put '(' -> s[3]='(', cur=2, rem_open=0.
    //                i=4:
    //                    rem_open=0, so put ')', s[4]=')', cur=1, rem_close=1.
    //                i=5:
    //                    put ')', s[5]=')', cur=0.

    //                s[0]='(', s[1]=')', s[2]='(', s[3]='(', s[4]=')', s[5]=')' -> "()(())", which is correct.
    //
    //        So the algorithm works.
    //
    //        Implementation:
    //          - Read the string.
    //          - If the string is empty, output "No solution"? But n>=0, and n is even.
    //          - Do the backward pass to find p.
    //          - If p not found, output "No solution".
    //          - Change s[p] to ')'.

    //                if (rem_open > 0 && rem_close >= cur+1) {

    //                    s[i] = ')';

    //                    rem_close--;

    //          - Output the string.
    //
    //        Note: We must use the original counts for the suffix, because the change at p hasn't been reflected in the counts for the suffix? 
    //              But we are counting in the original string, so open_orig and close_orig are for [p+1, n-1] in the original.
    //              And then we adjust by +1 open and -1 close.
    //
    //        Let's code accordingly.
    //
    //        Edge: if n=0, but the problem says length doesn't exceed 10000, and it's a correct bracket sequence, so n>=0, and n=0 is possible? 
    //              But n=0: then there is no next sequence? 
    //              However, the smallest sequence of length 0 is "" and there is no next, so output "No solution".
    //              But n=0: the input is empty string.
    //        Also, n must be even.
    //
    //        Steps:
    //          if (n==0) -> "No solution"
    //
    //        However, the problem says: "correct brackets sequence", so n is even and nonnegative.
    //        For n=0: 
    //            backward pass: loop from i=-1 down to 0 -> doesn't run, so p not found -> "No solution", which is correct.
    //
    //        Let's test n=2: 
    //            Input: "()"
    //            Backward:
    //                i=1: s[1]=')' -> bal=1
    //                i=0: s[0]='(' -> bal=1-1=0, and 0>0 false -> p not found -> "No solution", but the next of "()" doesn't exist? 
    //                Indeed, "()" is the only valid sequence of length 2, so next doesn't exist.
    //
    //        So it's correct.
    //
    //        Code:
    string s;
    getline(cin, s);
    int n = s.size();
    
    // If n is odd, but the problem says correct bracket sequence, so n should be even, but just in case.
    // However, the problem states it's a correct brackets sequence, so n is even.
    
    // Step 1: Find p from the right.
    int bal_rev = 0;
    int p = -1;
    for (int i = n-1; i >= 0; i--) {
        if (s[i] == ')') {
            bal_rev++;
        } else {
            bal_rev--;
            if (bal_rev > 0) {
                p = i;
                break;
            }
        }
    }
    
    if (p == -1) {
        cout << "No solution" << endl;
        return 0;
    }
    
    // Change s[p] to ')'
    s[p] = ')';
    
    // Step 2: Compute balance for [0, p-1] in the original string.
    int bal0 = 0;
    for (int i = 0; i < p; i++) {
        if (s[i] == '(') {
            bal0++;
        } else {
            bal0--;
        }
    }
    int cur = bal0 - 1; // balance after position p
    
    // Step 3: Count open and close in the suffix [p+1, n-1] of the original string.
    int open_orig = 0, close_orig = 0;
    for (int i = p+1; i < n; i++) {
        if (s[i] == '(') {
            open_orig++;
        } else {
            close_orig++;
        }
    }
    int rem_open = open_orig + 1;
    int rem_close = close_orig - 1;
    
    // Step 4: Build the suffix [p+1, n-1]
    for (int i = p+1; i < n; i++) {
        if (rem_open > 0 && rem_close >= cur + 1) {
            s[i] = '(';
            cur++;
            rem_open--;
        } else {
            s[i] = ')';
            cur--;
            rem_close--;
        }
    }
    
    cout << s << endl;
    return 0;
}
