We are given the number of rows N, and then N integers (each between 0 and 3) representing the numbers in the second column.

The field has 2 columns and N rows. The second column is fully opened and contains no mines. The first column is closed and may contain mines (each cell either has a mine or not).

For a cell in the second column at row i (0-indexed), the number shown is the total number of mines in the neighboring cells of the first column. Since the second column has no mines, the neighbors of the cell (i,1) [second column, row i] that lie in the first column are:
  - (i-1, 0) [top-left]
  - (i, 0)   [left]
  - (i+1, 0) [bottom-left]

But note: for the top row (i=0), there is no top neighbor (i-1), and for the bottom row (i=N-1), there is no bottom neighbor (i+1).

So the number at row i (0-indexed) is:
  a[i] = (mine at (i-1,0) ? 1 : 0) + (mine at (i,0) ? 1 : 0) + (mine at (i+1,0) ? 1 : 0)

But only the existing neighbors are considered (so for i=0, only (0,0) and (1,0); for i=N-1, only (N-2,0) and (N-1,0)).

#include <iostream>
#include <vector>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> a(N);
    for (int i = 0; i < N; i++) {
        cin >> a[i];
    }

    // dp[prev2][prev1]: number of ways for the state where the last two mines (x_{i-2}=prev2, x_{i-1}=prev1)
    // For the first cell (i=0), we use a virtual x_{-1}=0, so state is (0, x0)
    long long dp[2][2] = {{0}};
    dp[0][0] = 1;
    dp[0][1] = 1;

    // Process from the second cell (i=1) to the last cell (i=N-1)
    for (int i = 1; i < N; i++) {
        long long new_dp[2][2] = {{0}};
        for (int prev2 = 0; prev2 < 2; prev2++) {
            for (int prev1 = 0; prev1 < 2; prev1++) {
                for (int curr = 0; curr < 2; curr++) {
                    // Check constraint for row (i-1): a[i-1] = prev2 + prev1 + curr
                    if (prev2 + prev1 + curr == a[i-1]) {
                        new_dp[prev1][curr] += dp[prev2][prev1];
                    }
                }
            }
        }
        // Update dp to new_dp
        for (int i1 = 0; i1 < 2; i1++) {
            for (int i2 = 0; i2 < 2; i2++) {
                dp[i1][i2] = new_dp[i1][i2];
            }
        }
    }

    long long ans = 0;
    for (int prev1 = 0; prev1 < 2; prev1++) {
        for (int curr = 0; curr < 2; curr++) {
            // Check constraint for the last row (N-1): a[N-1] = prev1 + curr
            if (prev1 + curr == a[N-1]) {

            }
        }
    }
