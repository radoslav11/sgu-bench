#include <iostream>
#include <deque>
#include <string>
#include <vector>
#include <cctype>
#include <algorithm>

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(0);
    
    int N, M, K;
    std::cin >> N >> M >> K;
    
    if (K == 0) {
        std::deque<std::string> pile;
        std::string book;
        // Read initial N books
        for (int i = 0; i < N; i++) {
            std::cin >> book;
            pile.push_back(book);  // front of pile is top
        }
        
        std::string op;
        std::cin.ignore(); // ignore newline after last book
        for (int i = 0; i < M; i++) {
            std::getline(std::cin, op);
            if (op[0] == 'A') {
                // Parse the book name from ADD(S)
                // Find the opening parenthesis
                size_t pos1 = op.find('(');
                size_t pos2 = op.find(')');
                std::string s = op.substr(pos1 + 1, pos2 - pos1 - 1);
                pile.push_front(s);
            }
            // ROTATE does nothing for K=0
        }
        
        // Output
        for (const std::string& b : pile) {
            std::cout << b << '\n';
        }
    } else {
        std::deque<std::string> topPart;
        std::deque<std::string> bottomPart;
        bool reversed = false;
        
        std::string book;
        // Read initial N books
        for (int i = 0; i < N; i++) {
            std::cin >> book;
            if (topPart.size() < static_cast<size_t>(K)) {
                topPart.push_back(book);
            } else {
                bottomPart.push_back(book);
            }
        }
        
        // Read M operations
        std::string op;
        std::cin.ignore(); // ignore newline
        for (int i = 0; i < M; i++) {
            std::getline(std::cin, op);
            if (op[0] == 'A') {
                // Parse book name
                size_t pos1 = op.find('(');
                size_t pos2 = op.find(')');
                std::string s = op.substr(pos1 + 1, pos2 - pos1 - 1);
                
                if (topPart.size() < static_cast<size_t>(K)) {
                    if (!reversed) {
                        topPart.push_front(s);
                    } else {
                        topPart.push_back(s);
                    }
                } else {
                    // topPart is full (size == K)
                    if (!reversed) {
                        topPart.push_front(s);
                        // Move the back of topPart to front of bottomPart
                        bottomPart.push_front(topPart.back());
                        topPart.pop_back();
                    } else {
                        topPart.push_back(s);
                        // Move the front of topPart to front of bottomPart
                        bottomPart.push_front(topPart.front());
                        topPart.pop_front();
                    }
                }
            } else if (op == "ROTATE") {
                reversed = !reversed;
            }
        }
        
        // Output
        if (!reversed) {
            for (const std::string& b : topPart) {
                std::cout << b << '\n';
            }
        } else {
            // Output topPart in reverse order
            for (auto it = topPart.rbegin(); it != topPart.rend(); ++it) {
                std::cout << *it << '\n';
            }
        }
        for (const std::string& b : bottomPart) {
            std::cout << b << '\n';
        }
    }
    
    return 0;
}
