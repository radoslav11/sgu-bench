#include <iostream>
#include <vector>
#include <algorithm>
#include <cctype>
using namespace std;

struct Point {
    int x, y;
    Point() {}
    Point(int x_, int y_) : x(x_), y(y_) {}
};

// Check if point p lies on segment (a,b)
bool onSegment(const Point& a, const Point& b, const Point& p) {
    // Check if p is collinear with a and b
    long long cross = (long long)(b.x - a.x) * (p.y - a.y) - (long long)(b.y - a.y) * (p.x - a.x);
    if (cross != 0) return false;
    
    // Check if p is within the bounding box of segment (a,b)
    int minX = min(a.x, b.x), maxX = max(a.x, b.x);
    int minY = min(a.y, b.y), maxY = max(a.y, b.y);
    
    return (p.x >= minX && p.x <= maxX && p.y >= minY && p.y <= maxY);
}

// Check if point is on the polygon border
bool onBorder(const vector<Point>& poly, const Point& p) {
    int n = poly.size();
    for (int i = 0; i < n; ++i) {
        if (onSegment(poly[i], poly[(i + 1) % n], p)) {
            return true;
        }
    }
    return false;
}

// Ray casting algorithm to check if point is inside polygon
bool inside(const vector<Point>& poly, const Point& p) {
    int n = poly.size();
    bool inside = false;
    
    for (int i = 0; i < n; ++i) {
        int j = (i + 1) % n;
        
        // Check if point is on edge
        if (onSegment(poly[i], poly[j], p)) {
            return false; // will be handled separately
        }
        
        // Check if edge crosses horizontal ray to the right of point
        if ((poly[i].y > p.y) != (poly[j].y > p.y)) {
            // Compute x-coordinate of intersection of edge with horizontal line at y=p.y
            long double dx = (long double)(poly[j].x - poly[i].x);
            long double dy = (long double)(poly[j].y - poly[i].y);
            long double x_intersect = (long double)poly[i].x + dx * ((long double)p.y - poly[i].y) / dy;
            
            if (x_intersect > p.x) {
                inside = !inside;
            }
        }
    }
    
    return inside;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int K;
    cin >> K;
    
    vector<Point> points;
    // Read segments and build polygon
    for (int i = 0; i < K; ++i) {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        
        // Add first point of segment
        if (points.empty() || points.back().x != x1 || points.back().y != y1) {
            if (!points.empty() && points.back().x == x2 && points.back().y == y2) {
                // Segment is in reverse order
                points.push_back(Point(x1, y1));
            } else {
                points.push_back(Point(x1, y1));
            }
        } else if (points.back().x == x2 && points.back().y == y2) {
            // Already have the second point
        } else if (points.back().x != x2 || points.back().y != y2) {
            // Need to add both points
            points.push_back(Point(x1, y1));
        }
        
        // Add second point of segment
        if (points.empty() || points.back().x != x2 || points.back().y != y2) {
            points.push_back(Point(x2, y2));
        }
    }
    
    // We need to reconstruct the polygon properly
    // Read segments again and build adjacency list
    vector<vector<int>> adj(K);
    vector<pair<Point, Point>> segs;
    
    for (int i = 0; i < K; ++i) {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        segs.emplace_back(Point(x1, y1), Point(x2, y2));
        
        // Find which other segments share endpoints
        for (int j = 0; j < i; ++j) {
            if (segs[i].first.x == segs[j].first.x && segs[i].first.y == segs[j].first.y) {
                adj[i].push_back(j);
                adj[j].push_back(i);
            } else if (segs[i].first.x == segs[j].second.x && segs[i].first.y == segs[j].second.y) {
                adj[i].push_back(j);
                adj[j].push_back(i);
            } else if (segs[i].second.x == segs[j].first.x && segs[i].second.y == segs[j].first.y) {
                adj[i].push_back(j);
                adj[j].push_back(i);
            } else if (segs[i].second.x == segs[j].second.x && segs[i].second.y == segs[j].second.y) {
                adj[i].push_back(j);
                adj[j].push_back(i);
            }
        }
    }
    
    // Reconstruct polygon by following the chain
    vector<Point> poly;
    vector<bool> used(K, false);
    
    // Start from segment 0
    int cur = 0;
    poly.push_back(segs[cur].first);
    poly.push_back(segs[cur].second);
    used[cur] = true;
    
    // Build the polygon in order
    while (poly.size() < K + 1) {
        bool found = false;
        
        for (int next : adj[cur]) {
            if (used[next]) continue;
            
            // Check if segments connect
            Point p1 = segs[next].first;
            Point p2 = segs[next].second;
            
            if (poly.back().x == p1.x && poly.back().y == p1.y) {
                // Continue with p2
                if (poly.size() < K + 1 || !(p2.x == poly[0].x && p2.y == poly[0].y)) {
                    poly.push_back(p2);
                }
                used[next] = true;
                cur = next;
                found = true;
                break;
            } else if (poly.back().x == p2.x && poly.back().y == p2.y) {
                // Continue with p1
                if (poly.size() < K + 1 || !(p1.x == poly[0].x && p1.y == poly[0].y)) {
                    poly.push_back(p1);
                }
                used[next] = true;
                cur = next;
                found = true;
                break;
            }
        }
        
        if (!found) {
            // Try to connect with the beginning
            Point first = poly[0];
            for (int i = 0; i < K; ++i) {
                if (!used[i]) {
                    Point p1 = segs[i].first;
                    Point p2 = segs[i].second;
                    if ((p1.x == first.x && p1.y == first.y) || (p2.x == first.x && p2.y == first.y)) {
                        // Found the closing segment
                        Point other = (p1.x == first.x && p1.y == first.y) ? p2 : p1;
                        poly.push_back(other);
                        used[i] = true;
                        cur = i;
                        found = true;
                        break;
                    }
                }
            }
        }
        
        if (!found) break;
    }
    
    // Read point
    int X0, Y0;
    cin >> X0 >> Y0;
    Point p(X0, Y0);
    
    // Check if point is on border
    bool border = false;
    for (int i = 0; i < K; ++i) {
        if (onSegment(segs[i].first, segs[i].second, p)) {
            border = true;
            break;
        }
    }
    
    if (border) {
        cout << "BORDER" << endl;
    } else {
        // Use ray casting to check inside/outside
        bool in = false;
        for (int i = 0; i < K; ++i) {
            int j = (i + 1) % K;
            
            // Skip if point is on edge (already handled above)
            if (onSegment(segs[i].first, segs[i].second, p)) {
                in = true;
                break;
            }
            
            // Check if edge crosses horizontal ray to the right
            Point a = segs[i].first;
            Point b = segs[j].second;
            
            if ((a.y > Y0) != (b.y > Y0)) {
                // Compute x-coordinate of intersection
                long double dx = (long double)(b.x - a.x);
                long double dy = (long double)(b.y - a.y);
                long double x_intersect = (long double)a.x + dx * ((long double)Y0 - a.y) / dy;
                
                if (x_intersect > X0) {
                    in = !in;
                }
            }
        }
        
        if (in) {
            cout << "INSIDE" << endl;
        } else {
            cout << "OUTSIDE" << endl;
        }
    }
    
    return 0;
}
