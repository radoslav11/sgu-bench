To solve this problem, we need to determine the optimal set of problems a team can solve during a training session, given precedence constraints and a time limit. The goal is to maximize the number of problems solved and then minimize the penalty time, which is the sum of the completion times of all solved problems.

### Approach
1. **Problem Analysis**:
   - **Input**: The problem involves reading the number of problems (N), the training duration (T), the time required for each problem (t_i), the number of constraints (M), and the constraints themselves (each constraint (a, b) means problem a must be solved before problem b).
   - **Constraints Handling**: The constraints form a directed graph. We need to handle cycles in this graph because cycles make it impossible to solve any problems involved in the cycle (since solving any problem in the cycle requires solving all problems in the cycle, which is infeasible due to the cycle).
   - **SCC Identification**: Using Kosaraju's algorithm, we identify strongly connected components (SCCs) in the constraint graph. Components with more than one node are marked as "bad" because they contain cycles.
   - **Bad Propagation**: Any component reachable from a bad component in the condensation graph (DAG of SCCs) is also marked as bad, as solving any problem in such a component would require solving a bad component.
   - **Safe Nodes**: Nodes not in bad components and in singleton SCCs (size 1) are considered safe.
   - **Greedy Scheduling**: For safe nodes, we use a priority queue to schedule problems in the order of increasing time (Shortest Processing Time first) while respecting precedence constraints. This minimizes the penalty time for the scheduled set.

2. **Algorithm Selection**:
   - **Kosaraju's Algorithm**: To identify SCCs efficiently.
   - **Topological Sorting with Priority Queue**: To schedule safe nodes in an order that minimizes penalty time while respecting constraints.
   - **Prefix Sum Calculation**: To compute total time and penalty for prefixes of the scheduled order, allowing us to find the largest set of problems that fits within the time limit T.

3. **Complexity Analysis**:
   - **Time Complexity**: O(N + M) for Kosaraju's algorithm, O(N + M) for building the condensation graph, O(N log N) for the priority queue scheduling (due to heap operations), and O(N) for prefix sum calculations. Overall, O(N log N + M), which is efficient for N ≤ 1000 and M ≤ 10000.
   - **Space Complexity**: O(N + M) for storing the graph and auxiliary data structures.

            dfs2(v, visited, rev_graph, comp, comp_id, cid);
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N;
    long long T;
    cin >> N >> T;
    vector<int> t(N);
    for (int i = 0; i < N; i++) {
        cin >> t[i];
    }
    int M;
    cin >> M;
    vector<vector<int>> graph(N);
    vector<vector<int>> rev_graph(N);
    for (int i = 0; i < M; i++) {
        int a, b;
        cin >> a >> b;
        a--; b--;
        graph[a].push_back(b);
        rev_graph[b].push_back(a);
    }

    vector<bool> visited1(N, false);
    stack<int> st1;
    for (int i = 0; i < N; i++) {
        if (!visited1[i]) {
            dfs1(i, visited1, graph, st1);
        }
    }

    vector<int> comp_id(N, -1);
    vector<vector<int>> components;
    vector<bool> visited2(N, false);
    int comp_count = 0;

    while (!st1.empty()) {
        int node = st1.top();
        st1.pop();
        if (!visited2[node]) {
            vector<int> comp;
            dfs2(node, visited2, rev_graph, comp, comp_id, comp_count);
            components.push_back(comp);
            comp_count++;
        }
    }

    vector<bool> bad_comp(comp_count, false);
    for (int i = 0; i < comp_count; i++) {
        if (components[i].size() > 1) {
            bad_comp[i] = true;
        }
    }

    vector<unordered_set<int>> cond_set(comp_count);
    for (int u = 0; u < N; u++) {
        for (int v : graph[u]) {
            int c1 = comp_id[u];
            int c2 = comp_id[v];
            if (c1 != c2) {
                cond_set[c1].insert(c2);
            }
        }
    }

    vector<vector<int>> cond_graph(comp_count);
    for (int i = 0; i < comp_count; i++) {
        for (int neighbor : cond_set[i]) {
            cond_graph[i].push_back(neighbor);
        }
    }

    vector<bool> is_bad_comp(comp_count, false);
    queue<int> q;
    for (int i = 0; i < comp_count; i++) {
        if (bad_comp[i]) {
            is_bad_comp[i] = true;
            q.push(i);
        }
    }
    while (!q.empty()) {
        int c = q.front();
        q.pop();
        for (int neighbor : cond_graph[c]) {
            if (!is_bad_comp[neighbor]) {
                is_bad_comp[neighbor] = true;
                q.push(neighbor);
            }
        }
    }

    vector<bool> safe(N, false);
    for (int i = 0; i < N; i++) {
        int cid = comp_id[i];
        if (!is_bad_comp[cid] && components[cid].size() == 1) {
            safe[i] = true;
        }
    }

    vector<int> in_degree(N, 0);
    vector<vector<int>> safe_graph(N);
    for (int u = 0; u < N; u++) {
        if (!safe[u]) continue;
        for (int v : graph[u]) {
            if (safe[v]) {
                safe_graph[u].push_back(v);
                in_degree[v]++;
            }
        }
    }

    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    for (int i = 0; i < N; i++) {
        if (safe[i] && in_degree[i] == 0) {
            pq.push({t[i], i});
        }
    }

    vector<int> order;
    while (!pq.empty()) {
        auto node = pq.top();
        pq.pop();
        int u = node.second;
        order.push_back(u);
        for (int v : safe_graph[u]) {
            in_degree[v]--;
            if (in_degree[v] == 0) {
                pq.push({t[v], v});
            }
        }
    }

    vector<long long> total_time(order.size() + 1, 0);
    vector<long long> penalty(order.size() + 1, 0);
    for (int k = 1; k <= order.size(); k++) {
        total_time[k] = total_time[k-1] + t[order[k-1]];
        penalty[k] = penalty[k-1] + total_time[k];
    }

    int best_k = 0;
    long long best_penalty = 0;
    for (int k = order.size(); k >= 0; k--) {
        if (total_time[k] <= T) {
            best_k = k;
            best_penalty = penalty[k];
            break;
        }
    }

    cout << best_k << " " << best_penalty << "\n";
    if (best_k > 0) {
        for (int i = 0; i < best_k; i++) {
            if (i > 0) cout << " ";
            cout << order[i] + 1;
        }
    }
    cout << "\n";

    return 0;
}

### Explanation
1. **Input Reading**: The code reads the number of problems (N), training duration (T), time required for each problem, number of constraints (M), and the constraints themselves.
2. **SCC Identification**: Using Kosaraju's algorithm, the code identifies SCCs in the constraint graph. Components with more than one node are marked as "bad" due to cycles.
3. **Bad Propagation**: The code propagates the "bad" status to all components reachable from initially bad components in the condensation graph (DAG of SCCs).
4. **Safe Node Selection**: Nodes not in bad components and in singleton SCCs are marked as safe.
5. **Greedy Scheduling**: Safe nodes are scheduled using a priority queue (min-heap) that prioritizes problems with the shortest processing time, respecting precedence constraints.
6. **Prefix Calculation**: The code computes total time and penalty for prefixes of the scheduled order to determine the largest set of problems that fits within the time limit T.
7. **Output**: The solution outputs the maximum number of problems solved and the corresponding minimal penalty time, followed by the problem indices in the optimal order.

This approach efficiently handles precedence constraints and time limits to maximize problem count while minimizing penalty time, leveraging graph theory and greedy scheduling for optimal performance.
