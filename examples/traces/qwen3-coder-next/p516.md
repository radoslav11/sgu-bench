        dt.day = std::stoi(s.substr(8, 2));
        dt.hour = std::stoi(s.substr(11, 2));
        dt.minute = std::stoi(s.substr(14, 2));
        return dt;

bool isWorkingHours(const DateTime& dt) {
    // Make a copy to calculate weekday
    struct tm t = {};
    t.tm_year = dt.year - 1900;
    t.tm_mon = dt.month - 1;
    t.tm_mday = dt.day;
    t.tm_hour = dt.hour;
    t.tm_min = dt.minute;
    t.tm_sec = 0;
    t.tm_isdst = -1;
    mktime(&t);
    
    // tm_wday: 0=Sunday, 1=Monday, ..., 6=Saturday
    int weekday = t.tm_wday;
    if (weekday == 0 || weekday == 6) return false; // weekend
    
    // Working hours: 9:00 to 18:00 (exclusive of 18:00 end? Note: schedule lasts until next guardian)
    // The problem says: "working day starts at 9:00 and lasts until 18:00"

    if (dt.hour < 9 || (dt.hour == 9 && dt.minute == 0)) return false;
    if (dt.hour >= 18) return false;
    
    return true;
}
