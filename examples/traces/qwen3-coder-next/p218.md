          // Build adjacency list for left nodes (computers)
          vector<vector<int>> adj(n);
          for (int u = 0; u < n; u++) {
              for (int v = 0; v < n; v++) {
                  if (a[u][v] <= x) {
                      adj[u].push_back(v);
                  }

          vector<int> matchL(n, -1); // for left node u, the matched right node v
          vector<int> matchR(n, -1); // for right node v, the matched left node u

          vector<int> dist(n); // for left nodes

          auto bfs = [&]() -> bool {
              queue<int> q;
              for (int u = 0; u < n; u++) {
                  if (matchL[u] == -1) {
                      dist[u] = 0;
                      q.push(u);

                      dist[u] = -1;

                          dist[next_u] = dist[u] + 1;

                      matchL[u] = v;

              dist[u] = -1;
              return false;
          };

          int matching_size = 0;
          while (bfs()) {
              for (int u = 0; u < n; u++) {
                  if (matchL[u] == -1 && dfs(u)) {
                      matching_size++;
                  }
              }
          }

          if (matching_size != n) {
              return false;
          }
          matching = move(matchL);
          return true;
      }

To solve this problem, we need to assign each program to exactly one computer such that the maximum unsafety value among all assignments is minimized. This is a classic minimax assignment problem, which can be efficiently solved using binary search combined with a bipartite matching algorithm (specifically, the Hopcroft-Karp algorithm for maximum matching).

### Approach
1. **Binary Search on the Answer**: The minimal possible maximum unsafety value must lie between the minimum and maximum values in the given matrix. We perform binary search over this range to find the smallest value `x` such that there exists a perfect matching where all assigned unsafety values are ≤ `x`.
2. **Bipartite Matching Check**: For each candidate value `x` during binary search, we construct a bipartite graph where an edge from computer `i` to program `j` exists if the unsafety value `matrix[i][j] ≤ x`. We then check if this graph has a perfect matching using the Hopcroft-Karp algorithm.
3. **Hopcroft-Karp Algorithm**: This algorithm efficiently finds the maximum matching in bipartite graphs in \(O(\sqrt{n} \cdot m)\) time, where \(n\) is the number of nodes and \(m\) is the number of edges. Given the constraints (n ≤ 500), this is feasible and optimal for our needs.

### Solution Code
```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <functional>
#include <climits>
#include <algorithm>
using namespace std;

bool hopcroft_karp(int x, const vector<vector<int>>& a, int n, vector<int>& matching) {
    vector<vector<int>> adj(n);
    for (int u = 0; u < n; u++) {
        for (int v = 0; v < n; v++) {
            if (a[u][v] <= x) {
                adj[u].push_back(v);
            }
        }
    }

    vector<int> matchL(n, -1);
    vector<int> matchR(n, -1);
    vector<int> dist(n);

    auto bfs = [&]() -> bool {
        queue<int> q;
        for (int u = 0; u < n; u++) {
            if (matchL[u] == -1) {
                dist[u] = 0;
                q.push(u);
            } else {
                dist[u] = -1;
            }
        }
        bool found = false;
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int v : adj[u]) {
                int next_u = matchR[v];
                if (next_u != -1 && dist[next_u] == -1) {
                    dist[next_u] = dist[u] + 1;
                    q.push(next_u);
                } else if (next_u == -1) {
                    found = true;
                }
            }
        }
        return found;
    };

    function<bool(int)> dfs = [&](int u) {
        for (int v : adj[u]) {
            int next_u = matchR[v];
            if (next_u == -1 || (dist[next_u] != -1 && dist[next_u] == dist[u] + 1 && dfs(next_u))) {
                matchR[v] = u;
                matchL[u] = v;
                return true;
            }
        }
        dist[u] = -1;
        return false;
    };

    int matching_size = 0;
    while (bfs()) {
        for (int u = 0; u < n; u++) {
            if (matchL[u] == -1 && dfs(u)) {
                matching_size++;
            }
        }
    }

    if (matching_size != n) {
        return false;
    }
    matching = move(matchL);
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n;
    cin >> n;
    vector<vector<int>> matrix(n, vector<int>(n));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> matrix[i][j];
        }
    }

    int low = INT_MAX;
    int high = INT_MIN;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            low = min(low, matrix[i][j]);
            high = max(high, matrix[i][j]);
        }
    }

    int ans_val = high;
    vector<int> best_matching;

    while (low <= high) {
        int mid = low + (high - low) / 2;
        vector<int> matching;
        if (hopcroft_karp(mid, matrix, n, matching)) {
            ans_val = mid;
            best_matching = matching;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    cout << ans_val << '\n';
    for (int i = 0; i < n; i++) {
        cout << i + 1 << " " << best_matching[i] + 1 << '\n';
    }

    return 0;
}
```

### Explanation
1. **Binary Search Setup**: We determine the range of possible values for the maximum unsafety value by scanning the matrix for the minimum and maximum values.
2. **Binary Search Execution**: For each midpoint value `mid` in the current search range, we check if a perfect matching exists where all assigned values are ≤ `mid`.
3. **Graph Construction**: For each candidate `mid`, we build a bipartite graph where edges from computers to programs exist if their unsafety value is ≤ `mid`.
4. **Hopcroft-Karp Algorithm**: This algorithm efficiently checks for a perfect matching in the constructed bipartite graph. It uses BFS to build layered graphs and DFS to find augmenting paths, ensuring optimal performance.
5. **Result Extraction**: After binary search completes, the smallest valid `mid` value is our answer. The corresponding matching is then output, converting indices to 1-based as required.

This approach efficiently narrows down the solution space using binary search and leverages the Hopcroft-Karp algorithm for quick matching checks, ensuring optimal performance even for the upper constraint limits.
