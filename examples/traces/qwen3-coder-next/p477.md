// Function to compute the number of door openings for a given door parameter t
// and return the list of open intervals [start, end]
vector<pair<long long, long long>> computeIntervals(const vector<long long>& times, long long t) {
    if (times.empty()) return {};
    
    vector<pair<long long, long long>> intervals;
    int i = 0;
    while (i < (int)times.size()) {
        long long start = times[i] - t;
        long long end = times[i] + t;
        int j = i + 1;
        
        // Extend the current interval as long as the next person comes within 2*t of the previous one
        while (j < (int)times.size() && times[j] - times[j-1] <= 2 * t) {
            end = times[j] + t;
            j++;
        }
        
        intervals.push_back({start, end});
        i = j;
    }
    
    return intervals;
}

// Check if two sets of intervals have overlap of length > d
bool hasTooLongOverlap(const vector<pair<long long, long long>>& intervals1,
                       const vector<pair<long long, long long>>& intervals2,
                       long long d) {
    int i = 0, j = 0;
    while (i < (int)intervals1.size() && j < (int)intervals2.size()) {
        long long start = max(intervals1[i].first, intervals2[j].first);
        long long end = min(intervals1[i].second, intervals2[j].second);
        
        if (start < end) {
            long long overlap = end - start;
            if (overlap > d) {
                return true;
            }
        }
        
        if (intervals1[i].second < intervals2[j].second) {
            i++;
        } else {
            j++;
        }
    }
    return false;
}
