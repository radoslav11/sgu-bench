To solve this problem, we need to simulate a parking system where groups of users request contiguous parking slots in a circular arrangement. The system must efficiently handle parking and leaving queries while adhering to specific allocation rules.

### Approach
1. **Initialization**: Read the number of parking slots (N) and queries (Q). Initialize the initial state of parking slots based on the input string.
2. **Interval Management**: Maintain a set of intervals representing contiguous free parking slots. Each interval is stored as a pair (l, r) where l ≤ r, indicating slots from `l` to `r` (0-indexed).
3. **Handling Queries**:
   - **PARK Query (S)**: 
     - **Cluster Selection**: Find the smallest cluster (interval) with size ≥ S. If multiple exist, choose the one with the smallest head (starting slot). Also consider a wrapped cluster (if the first and last intervals in the set are adjacent in the circular arrangement).
     - **Allocation**: Allocate S contiguous slots starting from the head of the selected cluster. Update the interval set by removing the allocated segment and adding any remaining parts.
     - **Output**: Format the allocated slots as intervals (e.g., "1-4" or "1,8-10") in 1-indexed notation.
   - **LEAVE Query (Qi)**:
     - **Free Slots**: Retrieve the start and size of the slot allocation from query Qi. Add the freed slots back to the interval set.
     - **Merge Intervals**: Merge consecutive intervals to maintain contiguous free segments.
4. **Efficiency**: The solution efficiently handles queries by leveraging set operations for interval management, ensuring optimal performance for typical cases where the number of intervals remains manageable.

```cpp
                    if (size >= S) {
                        if (size < best_size || (size == best_size && l < best_head)) {
                            best_size = size;
                            best_head = l;
                            best_is_wrapped = false;
                            best_interval = interval;
                        }
                    }
                }
                
                if (intervals.size() >= 2) {
                    auto it_first = intervals.begin();
                    auto it_last = prev(intervals.end());
                    int first_l = it_first->first;
                    int last_r = it_last->second;
                    if (first_l == 0 && last_r == N-1) {
                        int last_size = last_r - last_l + 1;
                        int first_size = first_r - first_l + 1;
                        int size_wrapped = last_size + first_size;
                        int head_wrapped = last_l;
                        if (size_wrapped >= S) {
                            if (size_wrapped < best_size || (size_wrapped == best_size && head_wrapped < best_head)) {
                                best_size = size_wrapped;
                                best_head = head_wrapped;
                                best_is_wrapped = true;
                                best_first_interval = *it_first;
                                best_last_interval = *it_last;
                            }
                        }
                    }
                }
            }
            
            if (best_size == INT_MAX) {
                cout << "NO ROOM\n";
            } else {
                if (!best_is_wrapped) {
                    intervals.erase(best_interval);
                    if (best_size > S) {
                        intervals.insert({best_head + S, best_head + best_size - 1});
                    }
                } else {
                    intervals.erase(best_first_interval);
                    intervals.erase(best_last_interval);
                    if (best_head + S <= N) {
                        if (best_head + S <= N-1) {
                            intervals.insert({best_head + S, N-1});
                        }
                        intervals.insert({0, best_first_interval.second});
                    } else {
                        int new_start = best_head + S - N;
                        if (new_start <= best_first_interval.second) {
                            intervals.insert({new_start, best_first_interval.second});
                        }
                    }
                }
                
                assignment_for_query[idx] = {best_head, S};
                
                vector<pair<int, int>> segments;
                if (!best_is_wrapped) {
                    segments.push_back({best_head, best_head + S - 1});
                } else {
                    if (best_head + S <= N) {
                        segments.push_back({best_head, best_head + S - 1});
                    } else {
                        segments.push_back({best_head, N-1});
                        segments.push_back({0, best_head + S - 1 - N});
                    }
                }
                
                for (auto &seg : segments) {
                    seg.first++;
                    seg.second++;
                }
                
                sort(segments.begin(), segments.end());
                vector<pair<int, int>> merged;
                for (auto &seg : segments) {
                    if (merged.empty() || merged.back().second + 1 < seg.first) {
                        merged.push_back(seg);
                    } else {
                        merged.back().second = max(merged.back().second, seg.second);
                    }
                }
                
                for (int i = 0; i < merged.size(); i++) {
                    if (i > 0) {
                        cout << ",";
                    }
                    if (merged[i].first == merged[i].second) {
                        cout << merged[i].first;
                    } else {
                        cout << merged[i].first << '-' << merged[i].second;
                    }
                }
                cout << '\n';
            }
        } else if (command == "LEAVE") {
            int Qi;
            cin >> Qi;
            int start = assignment_for_query[Qi].first;
            int size = assignment_for_query[Qi].second;
            
            if (start + size <= N) {
                intervals.insert({start, start + size - 1});
            } else {
                intervals.insert({start, N-1});
                intervals.insert({0, start + size - 1 - N});
            }
            
            vector<pair<int, int>> vec(intervals.begin(), intervals.end());
            sort(vec.begin(), vec.end());
            vector<pair<int, int>> merged_vec;
            for (auto &intv : vec) {
                if (merged_vec.empty() || merged_vec.back().second + 1 < intv.first) {
                    merged_vec.push_back(intv);
                } else {
                    merged_vec.back().second = max(merged_vec.back().second, intv.second);
                }
            }
            intervals = set<pair<int, int>>(merged_vec.begin(), merged_vec.end());
        }
    }
    
    return 0;
}
```

### Explanation
1. **Initialization**: The system starts by reading the number of parking slots (N) and queries (Q). The initial state of slots is parsed to identify contiguous free intervals.
2. **PARK Query Handling**: For each parking request (S users), the system checks all free intervals and any potential wrapped cluster (if the first and last intervals are adjacent in the circular arrangement). The smallest valid cluster is selected for allocation.
3. **Allocation and Update**: The selected cluster is split to allocate S contiguous slots. Remaining parts of the cluster are added back to the interval set. The allocated slots are formatted and printed.
4. **LEAVE Query Handling**: When users leave, their slots are freed and added back to the interval set. Consecutive intervals are merged to maintain contiguous free segments.
5. **Efficiency**: Using a set for interval management ensures O(log N) complexity for insertions and deletions, while processing queries in linear time relative to the number of intervals (typically small in practice), making the solution efficient for large inputs.
