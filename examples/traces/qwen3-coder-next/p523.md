            active[3]=true.
            current=5, target=3 -> down.
            candidates: x from 4 down to 3: 
                x=4: active? no -> skip
                x=3: active -> push 3.
            stops = [3] -> output 3.

        Another: 
          Input: 
            2 5
            3 10

          Then: 
            active[3]=true, active[10]=true.
            Trip1: target=3 (first pressed) -> down: candidates = [3] -> stops=[3], current=3, active[3]=false.
            Trip2: target=10 -> up: candidates: from 4 to 10: active[10] is true -> so candidates = [10] -> stops=[3,10].

        But what if: 
          Input: 
            3 5
            10 3 7

          Then: 
            active: 10,3,7 -> true.
            Trip1: target=10 -> up: candidates: from 6 to 10: 
                x=6: no, 7: yes -> push 7, 8:no,9:no,10:yes -> push 10 -> candidates=[7,10]
            stops: [7,10], current=10, active[7]=false, active[10]=false.
            Trip2: target=3 -> down: from 9 down to 3: 
                active: only 3 is active -> so when x=3: push 3 -> candidates=[3]
            stops: [7,10,3]

        This matches.

        However, note: what if the target is 1 and current is 100? The loop for down would go from 99 down to 1 -> 99 iterations. Since n and floors are at most 100, and the total number of stops is at most 100, the total iterations would be at most 100*100 = 10000, which is acceptable.

        So we code accordingly.

        Let me write the code.


```cpp
#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, f;
    cin >> n >> f;
    vector<int> e(n);
    for (int i = 0; i < n; i++) {
        cin >> e[i];
    }

    vector<bool> active(101, false);
    for (int i = 0; i < n; i++) {
        active[e[i]] = true;
    }

    vector<int> stops;
    int current = f;

    while (true) {
        int target = -1;
        for (int i = 0; i < n; i++) {
            if (active[e[i]]) {
                target = e[i];
                break;
            }
        }
        if (target == -1) {
            break;
        }

        vector<int> candidates;
        if (target > current) {
            for (int x = current + 1; x <= target; x++) {
                if (active[x]) {
                    candidates.push_back(x);
                }
            }
        } else {
            for (int x = current - 1; x >= target; x--) {
                if (active[x]) {
                    candidates.push_back(x);
                }
            }
        }

        for (int stop : candidates) {
            stops.push_back(stop);
            active[stop] = false;
        }
        current = candidates.back();
    }

    for (int i = 0; i < stops.size(); i++) {
        if (i > 0) {
            cout << " ";
        }
        cout << stops[i];
    }
    cout << endl;

    return 0;
}
```
