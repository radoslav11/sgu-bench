I need to solve this competitive programming problem. Let me analyze it:

We have N subjects and M pupils. Each pupil selects 2 different subjects. We need to schedule all exams into 2 days such that:
- Each subject's exam happens on exactly one day (either day 1 or day 2)
- For each pupil, their two selected subjects must be on different days (so they can take both exams)

This is essentially a graph coloring problem:
- Create a graph where each subject is a vertex
- For each pupil who selected subjects a and b, add an edge between vertices a and b
- We need to check if this graph is bipartite (2-colorable)
- If yes, output one partition as day 1 subjects

Wait, but note: multiple pupils might select the same pair of subjects, but that's fine - we only need to ensure the constraint is satisfied.

Let me verify with the sample:
- Subjects: 1,2,3,4
- Pupils: (1,2), (3,4), (2,4), (1,3)
- Edges: 1-2, 3-4, 2-4, 1-3

This forms a cycle: 1-2-4-3-1 (length 4, which is even), so it's bipartite.
One valid partition: {1,4} and {2,3} - which matches the sample output.

Algorithm:
1. Build an undirected graph from the pupil selections
2. Check if the graph is bipartite using BFS/DFS coloring
3. If not bipartite, output "no"
4. If bipartite, output "yes", then the number of subjects in one color class, then the subjects themselves

Note: The graph might be disconnected, so we need to check all connected components.

Implementation details:
- Use 0-indexed internally, convert to 1-indexed for output
- Use BFS or DFS for bipartite checking
- Store the color assignment (0 or 1) for each vertex

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    cin >> N >> M;

    vector<vector<int>> graph(N + 1);

    for (int i = 0; i < M; i++) {
        int a, b;
        cin >> a >> b;
        graph[a].push_back(b);
        graph[b].push_back(a);
    }

    // color[i] = -1 means unvisited, 0 and 1 are the two colors
    vector<int> color(N + 1, -1);
    bool isBipartite = true;

    for (int i = 1; i <= N; i++) {
        if (color[i] == -1) {
            queue<int> q;
            q.push(i);
            color[i] = 0;

            while (!q.empty() && isBipartite) {
                int u = q.front();
                q.pop();

                for (int v : graph[u]) {
                    if (color[v] == -1) {
                        color[v] = color[u] ^ 1;
                        q.push(v);
                    } else if (color[v] == color[u]) {
                        isBipartite = false;
                        break;
                    }
                }
            }

            if (!isBipartite) break;
        }
    }

    if (!isBipartite) {
        cout << "no" << endl;
    } else {
        vector<int> day1;
        for (int i = 1; i <= N; i++) {
            if (color[i] == 0) {
                day1.push_back(i);
            }
        }

        cout << "yes" << endl;
        cout << day1.size() << endl;
        if (!day1.empty()) {
            for (int i = 0; i < day1.size(); i++) {
                if (i > 0) cout << " ";
                cout << day1[i];
            }
            cout << endl;
        } else {
            cout << endl;
        }
    }

    return 0;
}
```
