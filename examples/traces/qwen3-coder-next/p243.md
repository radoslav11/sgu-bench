#include <iostream>
#include <vector>

using namespace std;

// Directions for rook movement: up, right, down, left
const int dx[4] = {-1, 0, 1, 0};
const int dy[4] = {0, 1, 0, -1};

// Rotations: 0 = original, 1 = 90°, 2 = 180°, 3 = 270°
// To rotate a matrix 90° clockwise: (i, j) -> (j, n-1-i)
struct Piece {
    char id;
    vector<pair<int, int>> cells; // relative coordinates (minimized to start at (0,0))
    int min_i, min_j;
    
    Piece(char id_) : id(id_) {}
    
    // Normalize the piece so that top-left cell is at (0,0)
    void normalize() {
        if (cells.empty()) return;
        int min_i = INT_MAX, min_j = INT_MAX;
        for (auto [i, j] : cells) {
            min_i = min(min_i, i);
            min_j = min(min_j, j);
        }
        for (auto &p : cells) {
            p.first -= min_i;
            p.second -= min_j;
        }
        this->min_i = 0;
        this->min_j = 0;
    }
    
    // Get rotated version (90° clockwise)
    Piece rotate90() const {
        Piece res(id);
        if (cells.empty()) return res;
        int max_j = 0;
        for (auto [i, j] : cells) {
            max_j = max(max_j, j);
        }
        for (auto [i, j] : cells) {
            res.cells.push_back({j, max_j - i});
        }
        res.normalize();
        return res;
    }
};

int main() {
    int N;
    cin >> N;
    
    vector<string> grid(20);
    for (int i = 0; i < 20; i++) {
        cin >> grid[i];
    }
    
    // Find pieces: map from char to piece
    map<char, vector<pair<int, int>>> rawPieces;
    for (int i = 0; i < 20; i++) {
        for (int j = 0; j < 20; j++) {
            if (isupper(grid[i][j])) {
                rawPieces[grid[i][j]].push_back({i, j});
            }
        }
    }
    
    // Normalize and create pieces
    vector<Piece> pieces;
    for (auto &p : rawPieces) {
        Piece piece(p.first);
        piece.cells = p.second;
        piece.normalize();
        pieces.push_back(piece);
    }
    
    // Sort pieces by ID for consistent ordering
    sort(pieces.begin(), pieces.end(), [](const Piece &a, const Piece &b) {
        return a.id < b.id;
    });
    
    int K = pieces.size();
    
    // We need to try all permutations of pieces and all rotations
    vector<int> perm(K);
    for (int i = 0; i < K; i++) perm[i] = i;
    
    // For each piece, store all 4 rotations (precomputed)
    vector<vector<Piece>> allRotations(K);
    for (int i = 0; i < K; i++) {
        Piece p0 = pieces[i];
        allRotations[i].push_back(p0); // 0°
        allRotations[i].push_back(p0.rotate90()); // 90°
        allRotations[i].push_back(allRotations[i][1].rotate90()); // 180°
        allRotations[i].push_back(allRotations[i][2].rotate90()); // 270°
    }
    
    // Try all permutations and rotations
    do {
        // Try all combinations of rotations
        vector<int> rotIndex(K, 0);
        bool found = false;
        
        // Use DFS/backtracking over rotations
        function<bool(int)> dfs = [&](int idx) -> bool {
            if (idx == K) {
                // Try to place pieces in the N x N board
                vector<string> board(N, string(N, '.'));
                vector<vector<bool>> used(N, vector<bool>(N, false));
                
                // Place pieces in order
                for (int i = 0; i < K; i++) {
                    int pieceIdx = perm[i];
                    const Piece &p = allRotations[pieceIdx][rotIndex[i]];
                    
                    // Try to place the piece at each possible top-left position
                    for (int start_i = 0; start_i < N; start_i++) {
                        for (int start_j = 0; start_j < N; start_j++) {
                            bool canPlace = true;
                            // Check if all cells fit in the board
                            for (auto [di, dj] : p.cells) {
                                int ni = start_i + di, nj = start_j + dj;
                                if (ni < 0 || ni >= N || nj < 0 || nj >= N || used[ni][nj]) {
                                    canPlace = false;
                                    break;
                                }
                            }
                            if (canPlace) {
                                // Place the piece
                                for (auto [di, dj] : p.cells) {
                                    int ni = start_i + di, nj = start_j + dj;
                                    board[ni][nj] = pieces[pieceIdx].id;
                                    used[ni][nj] = true;
                                }
                                if (i == K - 1) {
                                    // All pieces placed, check if all cells filled
                                    bool full = true;
                                    for (int r = 0; r < N; r++) {
                                        for (int c = 0; c < N; c++) {
                                            if (!used[r][c]) full = false;
                                        }
                                    }
                                    if (full) {
                                        // Found solution
                                        for (int r = 0; r < N; r++) {
                                            cout << board[r] << endl;
                                        }
                                        found = true;
                                        return true;
                                    }
                                } else {
                                    break; // Move to next piece
                                }
                            }
                        }
                        if (found) break;
                    }
                    if (found) break;
                }
                return found;
            }
            
            for (int r = 0; r < 4; r++) {
                rotIndex[idx] = r;
                if (dfs(idx + 1)) return true;
            }
            return false;
        };
        
        if (dfs(0)) {
            return 0;
        }
        
    } while (next_permutation(perm.begin(), perm.end()));
    
    // According to problem statement, solution exists, so this should not be reached
    return 0;
}
