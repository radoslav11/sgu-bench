   #include <bits/stdc++.h>
   using namespace std;

   vector<string> match, avoid;
   int n, m;
   bool solve(int idx, string pat) {
     if (pat.length() > 100) return false;
     if (idx == match.size()) {
       // check avoid
       for (string s : avoid) {
         // check if pat matches s
         vector<vector<int>> dp(pat.size()+1, vector<int>(s.size()+1,0));
         dp[0][0] = 1;
         for (int i = 0; i <= pat.size(); i++) {
           for (int j = 0; j <= s.size(); j++) {
             if (dp[i][j]) {
               if (i < pat.size()) {
                 if (pat[i] == '*') {
                   dp[i+1][j] = 1;
                   if (j < s.size()) dp[i][j+1] = 1;
                   dp[i+1][j+1] = 1;
                 } else if (pat[i] == '?') {
                   if (j < s.size()) dp[i+1][j+1] = 1;
                 } else {
                   if (j < s.size() && pat[i] == s[j]) dp[i+1][j+1] = 1;
                 }
               }
             }
           }
         }
         if (dp[pat.size()][s.size()]) return false;
       }

To solve this problem, we need to generate a pattern string that matches all the given file names to match and avoids matching any of the file names to avoid. The pattern can include lowercase letters, asterisks ('*'), which match any sequence of characters (including empty), and question marks ('?'), which match exactly one character. The solution involves generating candidate patterns of increasing length and checking if they meet the criteria.

### Approach
1. **Problem Analysis**: The problem requires finding a pattern string that matches all specified files to match and none of the files to avoid. Given the constraints (up to 10 files to match, up to 10 files to avoid, and each file name up to 11 characters), a brute-force approach is feasible for small pattern lengths.
2. **Pattern Generation**: We generate patterns of lengths from 0 to 5 (inclusive). For each length, we use depth-first search (DFS) to explore all possible patterns composed of lowercase letters, '?', and '*'.
3. **Pruning**: During pattern generation, if a pattern matches any file to avoid, we prune that branch early to avoid unnecessary computations.
4. **Matching Check**: For each candidate pattern, we check if it matches all files to match and avoids all files to avoid. The matching is done using dynamic programming (DP) to handle wildcards efficiently.
5. **Output**: If a valid pattern is found within the length limit, it is printed; otherwise, "OOPS" is printed indicating no solution exists.

#include <iostream>
#include <vector>
#include <string>
using namespace std;

bool matches(const string& pattern, const string& s) {
    int m = pattern.size();
    int n = s.size();
    vector<vector<bool>> dp(m+1, vector<bool>(n+1, false));
    dp[0][0] = true;
    
    for (int i = 0; i < m; i++) {
        for (int j = 0; j <= n; j++) {
            if (!dp[i][j]) continue;
            if (pattern[i] == '*') {
                dp[i+1][j] = true;
                if (j < n) {
                    dp[i][j+1] = true;
                    dp[i+1][j+1] = true;
                }
            } else if (pattern[i] == '?') {
                if (j < n) {
                    dp[i+1][j+1] = true;
                }
            } else {
                if (j < n && pattern[i] == s[j]) {
                    dp[i+1][j+1] = true;
                }
            }
        }
    }
    return dp[m][n];
}

bool dfs(int depth, int max_depth, string& current, vector<string>& match, vector<string>& avoid) {
    if (depth > 0) {
        for (string s : avoid) {
            if (matches(current, s)) {
                return false;
            }
        }
    }
    
    if (depth == max_depth) {
        for (string s : match) {
            if (!matches(current, s)) {
                return false;
            }
        }
        return true;
    }
    
    for (char c = 'a'; c <= 'z'; c++) {
        current.push_back(c);
        if (dfs(depth+1, max_depth, current, match, avoid)) {
            return true;
        }
        current.pop_back();
    }
    
    current.push_back('?');
    if (dfs(depth+1, max_depth, current, match, avoid)) {
        return true;
    }
    current.pop_back();
    
    current.push_back('*');
    if (dfs(depth+1, max_depth, current, match, avoid)) {
        return true;
    }
    current.pop_back();
    
    return false;
}
