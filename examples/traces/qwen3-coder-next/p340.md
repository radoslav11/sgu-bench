#include <iostream>
#include <string>
#include <vector>
#include <cctype>
#include <cassert>

using namespace std;

// Removes all whitespace from the string
string removeWhitespace(const string& s) {
    string result;
    for (char c : s) {
        if (!isspace(c)) {
            result += c;
        }
    }
    return result;
}

// Recursively process a string that may contain subscripts, superscripts, letters, digits, operators
// Returns the HTML-converted string
string convert(const string& s, int& pos);

// Parse an atom: a letter (possibly with sub/superscripts) or a digit or parenthesized expression
string parseAtom(const string& s, int& pos);

// Parse subscript or superscript (including braces)
string parseSubscript(const string& s, int& pos);
string parseSuperscript(const string& s, int& pos);

// Get the next character, advancing pos if not at end
char peek(const string& s, int pos) {
    if (pos < (int)s.size()) return s[pos];
    return '\0';
}

string parseSubscript(const string& s, int& pos) {
    if (pos >= (int)s.size() || s[pos] != '_') return "";
    
    pos++; // consume '_'
    string content;
    
    if (pos < (int)s.size() && s[pos] == '{') {
        // Braced subscript
        pos++; // consume '{'
        int depth = 1;
        while (pos < (int)s.size() && depth > 0) {
            if (s[pos] == '{') depth++;
            else if (s[pos] == '}') depth--;
            if (depth > 0) {
                content += s[pos];
                pos++;
            }
        }
        if (depth == 0 && pos < (int)s.size()) pos++; // consume '}'
    } else if (pos < (int)s.size()) {
        // Single character subscript
        content = s[pos];
        pos++;
    }
    
    if (content.empty()) return "";
    
    // Recursively process the subscript content
    int tempPos = 0;
    string processed = convert(content, tempPos);
    return "<sub>" + processed + "</sub>";
}

string parseSuperscript(const string& s, int& pos) {
    if (pos >= (int)s.size() || s[pos] != '^') return "";
    
    pos++; // consume '^'
    string content;
    
    if (pos < (int)s.size() && s[pos] == '{') {
        // Braced superscript
        pos++; // consume '{'
        int depth = 1;
        while (pos < (int)s.size() && depth > 0) {
            if (s[pos] == '{') depth++;
            else if (s[pos] == '}') depth--;
            if (depth > 0) {
                content += s[pos];
                pos++;
            }
        }
        if (depth == 0 && pos < (int)s.size()) pos++; // consume '}'
    } else if (pos < (int)s.size()) {
        // Single character superscript
        content = s[pos];
        pos++;
    }
    
    if (content.empty()) return "";
    
    // Recursively process the superscript content
    int tempPos = 0;
    string processed = convert(content, tempPos);
    return "<sup>" + processed + "</sup>";
}

string parseAtom(const string& s, int& pos) {
    string result;
    
    // If current char is a letter, process it and potential sub/superscripts
    if (isalpha(s[pos])) {
        result += s[pos];
        pos++;
        
        string sub = parseSubscript(s, pos);
        string sup = parseSuperscript(s, pos);
        
        return "<i>" + result + "</i>" + sub + sup;
    }
    
    // Check for parentheses: if we see '(', we might be in a parenthesized expression
    if (s[pos] == '(') {
        int start = pos;
        int depth = 1;
        pos++;
        while (pos < (int)s.size() && depth > 0) {
            if (s[pos] == '(') depth++;
            else if (s[pos] == ')') depth--;
            pos++;
        }
        // Now process the content inside parentheses recursively
        string inner = s.substr(start + 1, pos - start - 2);
        int tempPos = 0;
        string converted = convert(inner, tempPos);
        return "(" + converted + ")";
    }
    
    // Otherwise, just a digit or other character
    result += s[pos];
    pos++;
    return result;
}

// Main conversion function: processes the whole string, handling operators, and wrapping letters
string convert(const string& s, int& pos) {
    string result;
    vector<string> parts;
    
    // We'll build parts of the expression (separated by operators)
    // and handle operators specially with non-breaking spaces
    
    while (pos < (int)s.size()) {
        // Check if this is an operator
        if (s[pos] == '+' || s[pos] == '-' || s[pos] == '*' || s[pos] == '/') {
            // Wrap operator with &nbsp;
            result += "&nbsp;" + string(1, s[pos]) + "&nbsp;";
            pos++;
            continue;
        }
        
        // Parse an atom (letter/digit/parenthesized)
        if (pos < (int)s.size()) {
            string atom = parseAtom(s, pos);
            result += atom;
        }
    }
    
    return result;
}

int main() {
    string line;
    while (getline(cin, line)) {
        // Skip empty lines
        if (line.empty()) continue;
        
        // Remove whitespace from the whole line first
        string cleanLine = removeWhitespace(line);
        
        // Check if it starts and ends with '$'
        if (cleanLine.size() >= 2 && cleanLine[0] == '$' && cleanLine.back() == '$') {
            // Extract content between dollar signs
            string formula = cleanLine.substr(1, cleanLine.size() - 2);
            
            // Process the formula
            int pos = 0;
            string html = convert(formula, pos);
            
            // Output
            cout << html << endl;
        }
    }
    return 0;
}
