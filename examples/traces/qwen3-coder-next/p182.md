// Parse "or" (lowest priority except for our custom operators)
// Operators: ||, <=>, =>, #
Expression parseOr() {
    Expression res = parseAnd();
    while (pos < expr.size()) {
        if (expr[pos] == '|' && pos + 1 < expr.size() && expr[pos + 1] == '|') {
            pos += 2;
            Expression rhs = parseAnd();
            Expression result;
            for (int i = 0; i < 1024; ++i) {
                result.values[i] = res.values[i] || rhs.values[i];
            }
            res = result;
        } else if (expr[pos] == '<' && pos + 2 < expr.size() && expr[pos+1] == '=' && expr[pos+2] == '>') {
            pos += 3;
            Expression rhs = parseAnd();
            Expression result;
            for (int i = 0; i < 1024; ++i) {
                result.values[i] = (res.values[i] == rhs.values[i]);
            }
            res = result;
        } else if (expr[pos] == '=' && pos + 1 < expr.size() && expr[pos+1] == '>') {
            pos += 2;
            Expression rhs = parseAnd();
            Expression result;
            for (int i = 0; i < 1024; ++i) {
                result.values[i] = (!res.values[i]) || rhs.values[i]; // A => B = !A || B
            }
            res = result;
        } else if (expr[pos] == '#') {
            pos++;
            Expression rhs = parseAnd();
            Expression result;
            for (int i = 0; i < 1024; ++i) {
                result.values[i] = res.values[i] != rhs.values[i]; // XOR
            }
            res = result;
        } else {
            break;
        }
    }
    return res;
}

// Parse "and"

Expression parseAnd() {

        Expression result;

            result.values[i] = res.values[i] && rhs.values[i];
        }
        res = result;
    }
    return res;
}

// Parse "not"

Expression parseNot() {
    if (pos < expr.size() && expr[pos] == '!') {
        pos++;
        Expression rhs = parseNot();
        Expression result;
        for (int i = 0; i < 1024; ++i) {
            result.values[i] = !rhs.values[i];
        }
        return result;
    }
    return parseAtom();
}

// Parse atom: variable, or parenthesized expression
Expression parseAtom() {
    if (expr[pos] == '(') {
        pos++; // skip '('
        Expression res = parseOr();
        if (pos < expr.size() && expr[pos] == ')') {
            pos++;
        }
        return res;
    } else if (islower(expr[pos])) {
        char var = expr[pos];
        pos++;
        Expression res;
        // Fill the truth table for this variable
        for (int i = 0; i < 1024; ++i) {
            // Variable index: var - 'a'
            // Bit representation: i is the assignment of variables a-j
            // We check if the bit corresponding to this variable is set
            int varIdx = getVarIndex(var);
            res.values[i] = (i >> varIdx) & 1;
        }
        return res;
    }
    // Should not reach here on valid input
    return Expression();
}

// Function to convert expression to DNF
// DNF: OR of ANDs of literals
// Algorithm: distributive law: A&(B|C) = (A&B)|(A&C)

// We'll write a function that takes a Node* and returns a Node* in DNF.
// DNF representation: 
//   - A literal is a VAR or NOT(VAR)
//   - A term is an AND of literals
//   - DNF is an OR of terms

// Steps for conversion:
// 1. Convert to negative normal form (NNF): move NOTs down to variables using De Morgan's laws.
// 2. Distribute AND over OR.

// We'll do:
// - NNF conversion: recursively push down NOTs.
// - Then distribute.

Node* pushNot(Node* node) {
    if (node->type == Node::VAR) {
        return node;
    } else if (node->type == Node::NOT) {
        Node* child = node->child;
        if (child->type == Node::VAR) {
            // Already a negative literal
            return node;
        } else if (child->type == Node::NOT) {
            // !!A = A
            return pushNot(child->child);
        } else if (child->type == Node::AND) {
            // !(A&B) = !A || !B
            Node* notLeft = new Node(Node::NOT);
            notLeft->child = child->left;
            Node* notRight = new Node(Node::NOT);
            notRight->child = child->right;
            Node* orNode = new Node(Node::OR);
            orNode->left = pushNot(notLeft);
            orNode->right = pushNot(notRight);
            delete notLeft;
            delete notRight;
            delete node;
            return orNode;
        } else if (child->type == Node::OR) {
            // !(A||B) = !A & !B
            Node* notLeft = new Node(Node::NOT);
            notLeft->child = child->left;
            Node* notRight = new Node(Node::NOT);
            notRight->child = child->right;
            Node* andNode = new Node(Node::AND);
            andNode->left = pushNot(notLeft);
            andNode->right = pushNot(notRight);
            delete notLeft;
            delete notRight;
            delete node;
            return andNode;
        } else if (child->type == Node::EQ) {
            // !(A<=>B) = A#B
            Node* xorNode = new Node(Node::XOR);
            xorNode->left = child->left;
            xorNode->right = child->right;
            delete node;
            return xorNode;
        } else if (child->type == Node::IMP) {
            // !(A=>B) = A & !B
            Node* notB = new Node(Node::NOT);
            notB->child = child->right;
            Node* andNode = new Node(Node::AND);
            andNode->left = child->left;
            andNode->right = pushNot(notB);
            delete notB;
            delete node;
            return andNode;
        } else if (child->type == Node::XOR) {
            // !(A#B) = A<=>B
            Node* eqNode = new Node(Node::EQ);
            eqNode->left = child->left;
            eqNode->right = child->right;
            delete node;
            return eqNode;
        }
    } else if (node->type == Node::AND) {
        node->left = pushNot(node->left);
        node->right = pushNot(node->right);
        return node;
    } else if (node->type == Node::OR) {
        node->left = pushNot(node->left);
        node->right = pushNot(node->right);
        return node;
    } else if (node->type == Node::EQ) {
        // A<=>B = (A& B) | (!A & !B)
        // We'll handle this in distribution step? Or convert to basic operators first?
        // To simplify, let's convert all operators to basic ones: !, &, |.
        // But we want to avoid that complexity.
        // Alternative: convert to NNF first for all operators.
        // However, for simplicity and since the problem has only 10 variables, 
        // we can use the truth table method and then generate the minimal DNF.
        // Given the time, and the constraints (max 1024 minterms), let's use the truth table method.
        return node;
    } else if (node->type == Node::IMP) {
        // A=>B = !A | B
        Node* notLeft = new Node(Node::NOT);
        notLeft->child = node->left;
        Node* orNode = new Node(Node::OR);
        orNode->left = pushNot(notLeft);
        orNode->right = node->right;
        delete notLeft;
        delete node;
        return orNode;
    } else if (node->type == Node::XOR) {
        // A#B = (A|B) & !(A&B) = (A|B) & (!A|!B) 
        // Or: A#B = (A & !B) | (!A & B)
        Node* notLeft = new Node(Node::NOT);
        notLeft->child = node->left;
        Node* notRight = new Node(Node::NOT);
        notRight->child = node->right;
        Node* term1 = new Node(Node::AND);
        term1->left = node->left;
        term1->right = pushNot(notRight);
        Node* term2 = new Node(Node::AND);
        term2->left = pushNot(notLeft);
        term2->right = node->right;
        Node* orNode = new Node(Node::OR);
        orNode->left = term1;
        orNode->right = term2;
        delete notLeft;
        delete notRight;
        delete node;
        return orNode;
    } else if (node->type == Node::EQ) {
        // A<=>B = (A & B) | (!A & !B)
        Node* notLeft = new Node(Node::NOT);
        notLeft->child = node->left;
        Node* notRight = new Node(Node::NOT);
        notRight->child = node->right;
        Node* term1 = new Node(Node::AND);
        term1->left = node->left;
        term1->right = node->right;
        Node* term2 = new Node(Node::AND);
        term2->left = pushNot(notLeft);
        term2->right = pushNot(notRight);
        Node* orNode = new Node(Node::OR);
        orNode->left = term1;
        orNode->right = term2;
        delete notLeft;
        delete notRight;
        delete node;
        return orNode;
    }
    return node;
}

// Function to distribute AND over OR to get DNF
Node* toDNF(Node* node) {
    if (node == nullptr) return nullptr;
    
    // First ensure children are in DNF
    node->left = toDNF(node->left);
    node->right = toDNF(node->right);
    node->child = toDNF(node->child);
    
    if (node->type == Node::AND) {
        Node* a = node->left;
        Node* b = node->right;
        // Case 1: a is OR, then distribute: (a1 | a2) & b = (a1&b) | (a2&b)
        if (a->type == Node::OR) {
            Node* a1 = a->left;
            Node* a2 = a->right;
            Node* and1 = new Node(Node::AND);
            and1->left = toDNF(a1);
            and1->right = toDNF(b);
            Node* and2 = new Node(Node::AND);
            and2->left = toDNF(a2);
            and2->right = toDNF(b);
            Node* orNode = new Node(Node::OR);
            orNode->left = toDNF(and1);
            orNode->right = toDNF(and2);
            delete a;
            delete b;
            delete node;
            return toDNF(orNode);
        }
        // Case 2: b is OR, then distribute: a & (b1 | b2) = (a&b1) | (a&b2)
        if (b->type == Node::OR) {
            Node* b1 = b->left;
            Node* b2 = b->right;
            Node* and1 = new Node(Node::AND);
            and1->left = toDNF(a);
            and1->right = toDNF(b1);
            Node* and2 = new Node(Node::AND);
            and2->left = toDNF(a);
            and2->right = toDNF(b2);
            Node* orNode = new Node(Node::OR);
            orNode->left = toDNF(and1);
            orNode->right = toDNF(and2);
            delete a;
            delete b;
            delete node;
            return toDNF(orNode);
        }
        // Case 3: both are not OR, then it's a term if they are literals, or recursively built
        return node;
    }
    if (node->type == Node::OR) {
        // If any child is OR, flatten
        // But for now, we assume children are DNF
        return node;
    }
    return node;
}

// Function to check if a node is a literal (VAR or NOT(VAR))
bool isLiteral(Node* node) {
    if (node == nullptr) return false;
    if (node->type == Node::VAR) return true;
    if (node->type == Node::NOT && node->child->type == Node::VAR) return true;
    return false;
}

// Function to check if a node is a term (AND of literals)
bool isTerm(Node* node) {
    if (node == nullptr) return false;
    if (isLiteral(node)) return true;
    if (node->type == Node::AND) {
        return isLiteral(node->left) && isLiteral(node->right);
    }
    return false;
}

// Function to convert a DNF node to string without brackets, but using precedence: & before ||
// We'll generate the string by adding parentheses only where necessary.
// But the problem says "free of brackets", so we must output without any brackets.
// So we rely on operator precedence: ! highest, then &, then ||, then others.
// In our DNF, we have only !, &, ||, and the terms are ANDs of literals, and the whole is ORs of terms.
// So we need to put parentheses around terms if a term is just a literal? No, because a literal has no operator that needs parentheses.
// But note: a literal like !a doesn't need parentheses because ! has highest precedence.
// And a term like a&b doesn't need parentheses because & has higher precedence than ||.
// So the whole DNF: term1 || term2 || ... 
// where each term is literal1 & literal2 & ... 
// will be: a&b||c&d, etc.

// However, what if a term has more than 2 literals? We need to make sure that the & operations are grouped correctly.
// Since & is associative, a&b&c can be written as a&b&c without parentheses.

// So algorithm for string generation:
// - If the node is OR: output left string + "||" + right string
// - If the node is AND: output left string + "&" + right string
// - If the node is NOT: output "!" + child string
// - If the node is VAR: output string(1, var)

// But wait: what if we have a term like !(a&b) in NNF? We converted to NNF, so we only have NOT on variables.
// So all NOTs are on variables, and then we have ANDs of literals, and ORs of terms.

// Therefore, the generated string for a term like a & !b & c is "a&!b&c", and for the whole DNF: "a&!b&c||d&e"

// However, the sample: a&(b||c) -> DNF: (a&b) || (a&c) -> "a&b||a&c"

// So we can do:

string nodeToString(Node* node) {
    if (node == nullptr) return "";
    
    if (node->type == Node::VAR) {
        return string(1, node->var);
    } else if (node->type == Node::NOT) {
        return "!" + nodeToString(node->child);
    } else if (node->type == Node::AND) {
        string left = nodeToString(node->left);
        string right = nodeToString(node->right);
        // Check if left or right need parentheses: 
        // For AND, if left is OR, then left needs parentheses? But in DNF, left should be a literal or AND, so no OR in left.
        // Similarly for right.
        // So no parentheses needed.
        return left + "&" + right;
    } else if (node->type == Node::OR) {
        string left = nodeToString(node->left);
        string right = nodeToString(node->right);
        return left + "||" + right;
    } else if (node->type == Node::EQ) {
        // Should not happen after conversion
        string left = nodeToString(node->left);
        string right = nodeToString(node->right);
        return left + "<=>" + right;
    } else if (node->type == Node::IMP) {
        string left = nodeToString(node->left);
        string right = nodeToString(node->right);
        return left + "=>" + right;
    } else if (node->type == Node::XOR) {
        string left = nodeToString(node->left);
        string right = nodeToString(node->right);
        return left + "#" + right;
    }
    return "";
}

Node* distribute(Node* node) {
    if (node == nullptr) return nullptr;
    
    // Recursively distribute children
    if (node->type == Node::AND) {
        node->left = distribute(node->left);
        node->right = distribute(node->right);
        
        // Check if left is OR
        if (node->left->type == Node::OR) {
            Node* a1 = node->left->left;
            Node* a2 = node->left->right;
            Node* b = node->right;
            
            Node* and1 = new Node(Node::AND);
            and1->left = a1;
            and1->right = b;
            
            Node* and2 = new Node(Node::AND);
            and2->left = a2;
            and2->right = b;
            
            Node* orNode = new Node(Node::OR);
            orNode->left = and1;
            orNode->right = and2;
            
            delete node;
            return distribute(orNode);
        }
        
        // Check if right is OR
        if (node->right->type == Node::OR) {
            Node* a = node->left;
            Node* b1 = node->right->left;
            Node* b2 = node->right->right;
            
            Node* and1 = new Node(Node::AND);
            and1->left = a;
            and1->right = b1;
            
            Node* and2 = new Node(Node::AND);
            and2->left = a;
            and2->right = b2;
            
            Node* orNode = new Node(Node::OR);
            orNode->left = and1;
            orNode->right = and2;
            
            delete node;
            return distribute(orNode);
        }
        
        return node;
    }
    
    if (node->type == Node::OR) {
        node->left = distribute(node->left);
        node->right = distribute(node->right);
        return node;
    }
    
    // For NOT, it should be only on variables in NNF, so no distribution needed.
    return node;
}

string nodeToStringFlat(Node* node, bool isOrContext) {
    // isOrContext: if true, we are inside an OR, so AND needs no parentheses; if false, inside an AND, so OR would need, but in DNF, OR is only at top.
    // In our DNF, the top is OR, and inside OR we have ANDs, and inside ANDs we have literals.
    // So for AND: if node->type is AND, we output left + "&" + right, but if left or right is AND, we need to flatten.
    // Similarly for OR.

    if (node->type == Node::VAR) {
        return string(1, node->var);
    } else if (node->type == Node::NOT) {
        return "!" + nodeToStringFlat(node->child, false);
    } else if (node->type == Node::AND) {
        vector<string> parts;
        stack<Node*> st;
        st.push(node);
        while (!st.empty()) {
            Node* cur = st.top(); st.pop();
            if (cur->type == Node::AND) {
                st.push(cur->right);
                st.push(cur->left);
            } else {
                parts.push_back(nodeToStringFlat(cur, true));
            }
        }
        string res;
        for (size_t i = 0; i < parts.size(); ++i) {
            if (i > 0) res += "&";
            res += parts[i];
        }
        return res;
    } else if (node->type == Node::OR) {
        vector<string> parts;
        stack<Node*> st;
        st.push(node);
        while (!st.empty()) {
            Node* cur = st.top(); st.pop();
            if (cur->type == Node::OR) {
                st.push(cur->right);
                st.push(cur->left);
            } else {
                parts.push_back(nodeToStringFlat(cur, false));
            }
        }
        string res;
        for (size_t i = 0; i < parts.size(); ++i) {
            if (i > 0) res += "||";
            res += parts[i];
        }
        return res;
    } else {
        // Other operators should not occur after conversion to DNF
        string left = nodeToStringFlat(node->left, node->type == Node::OR);
        string right = nodeToStringFlat(node->right, node->type == Node::OR);
        if (node->type == Node::EQ) return left + "<=>" + right;
        if (node->type == Node::IMP) return left + "=>" + right;
        if (node->type == Node::XOR) return left + "#" + right;
    }
    return "";
}
