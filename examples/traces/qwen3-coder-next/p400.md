int parseAttempt(const string& s) {
    if (s.empty()) return -1;
    if (s[0] == '+') return 0;
    if (s[0] == '-') return -(stoi(s.substr(1)));
    if (isdigit(s[0])) {
        int cnt = stoi(s);
        return -(cnt + 1); // because +1 means the accepted run
    }
    return -1;
}
