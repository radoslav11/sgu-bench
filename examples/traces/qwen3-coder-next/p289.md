            vector<pair<int,int>> moves;
            for (int i = 0; i < 3; i++) {
                for (int j = 0; j < 3; j++) {
                    if (board[i][j] == '.') {
                        moves.push_back({i, j});
                    }

                board[i][j] = player;

                if (checkWin(board, player)) {
                    board[i][j] = '.';
                    return 1;
                }

                bool full = true;
                for (int x = 0; x < 3; x++) {
                    for (int y = 0; y < 3; y++) {
                        if (board[x][y] == '.') {
                            full = false;
                            break;
                        }
                    }

                }

                    if (0 > best) best = 0;

                moves.push_back(make_pair(i, j));
            }
        }
    }

    int best = -2;
    for (auto& m : moves) {
        int i = m.first;
        int j = m.second;
        board[i][j] = player;

        if (checkWin(board, player)) {
            board[i][j] = '.';
            return 1;
        }

        bool full = true;
        for (int x = 0; x < 3; x++) {
            for (int y = 0; y < 3; y++) {
                if (board[x][y] == '.') {
                    full = false;
                    break;
                }
            }
            if (!full) break;
        }
        if (full) {
            board[i][j] = '.';
            if (0 > best) best = 0;
            continue;
        }

        char opponent = (player == 'X') ? '0' : 'X';
        int val = solve_game(board, opponent);
        int score = -val;
        board[i][j] = '.';

        if (score > best) best = score;
        if (best == 1) break;
    }

    if (best == -2) {
        best = 0;
    }
    return best;
}
