          vector<ll> dist(N+1, LLONG_MAX);
          priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
          parent.assign(N+1, -1);

          dist[S] = 0;

To solve this problem, we need to find the minimal maximum temperature encountered while escaping from a prison through one of the exits. The prison is modeled as a graph where corridors (edges) have initial temperatures and rates at which they heat up over time. The goal is to determine a path from the starting junction to any exit such that the maximum temperature experienced along the path is minimized, and this maximum temperature does not exceed the resistance limit of the suit (H degrees).

### Approach
1. **Problem Analysis**: The problem involves navigating a graph where each edge's temperature increases over time. The temperature at any point on an edge at time `t` is given by `R + P * t`, where `R` is the initial temperature and `P` is the rate of heating. The challenge is to find a path from the start node `S` to any exit node such that the maximum temperature encountered on any edge in the path is minimized.

2. **Binary Search on Maximum Temperature**: We perform a binary search on the possible values of the maximum temperature (from 0 to a sufficiently high upper bound). For each candidate maximum temperature `X`, we check if there exists a path from `S` to any exit where the temperature on every traversed edge does not exceed `X`.

3. **Dijkstra's Algorithm for Path Validation**: For each candidate `X`, we use a modified Dijkstra's algorithm to explore the graph. The algorithm prioritizes paths that minimize the arrival time at each node, as earlier arrival times allow for lower temperatures on subsequent edges. The relaxation step checks if traversing an edge `u -> v` at time `t` results in a maximum temperature `R + P * (t + T)` (where `T` is the traversal time of the edge) that is within the limit `X`.

4. **Path Reconstruction**: Once the minimal feasible maximum temperature `X` is found, we reconstruct the path from the start node to an exit using parent pointers recorded during the Dijkstra's run.

#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

struct Edge {
    int v, T, R, P;
    Edge(int v, int T, int R, int P) : v(v), T(T), R(R), P(P) {}
};

vector<vector<Edge>> adj;
vector<bool> is_exit;

bool check(ll X, int N, int S, vector<int>& parent, int& found_exit) {
    vector<ll> dist(N+1, LLONG_MAX);
    priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
    parent.assign(N+1, -1);

    dist[S] = 0;
    pq.push({0, S});

    while (!pq.empty()) {
        ll t = pq.top().first;
        int u = pq.top().second;
        pq.pop();
        if (t != dist[u]) continue;

        if (is_exit[u]) {
            found_exit = u;
            return true;
        }

        for (const Edge& e : adj[u]) {
            ll arrival = t + e.T;
            ll temp_at_arrival = e.R + (ll)e.P * arrival;
            if (temp_at_arrival <= X) {
                if (arrival < dist[e.v]) {
                    dist[e.v] = arrival;
                    parent[e.v] = u;
                    pq.push({arrival, e.v});
                }
            }
        }
    }
    return false;
}

    }

    vector<int> parent;
    int found_exit_node = -1;
    ll best_X = H + 1;

    if (is_exit[S]) {
        best_X = 0;
        parent.assign(N+1, -1);
        parent[S] = -1;
        found_exit_node = S;
    } else {
        ll low = 0, high = 20000000000LL;
        while (low < high) {

            if (check(mid, N, S, parent, found_exit_node)) {
                high = mid;
            } else {
                low = mid + 1;
            }
        }
