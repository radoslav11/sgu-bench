Implementation details:

1. Precompute binomial coefficients C[n][k] for n=0..50, k=0..n.

   We can compute with Pascal's triangle, using `long long` or `__int128` — `long long` is enough since C(50,25) ≈ 1.26e14 < 2^60 (~1e18).

2. For each n from 0 to N (input), compute I_n.

   Since we need I_0 to I_{n-1} to compute I_n, we'll compute iteratively from 0 up to input n.

3. Use `__int128` for numerators and denominators.

4. Write a gcd function for `__int128`.

5. Write a function to convert `__int128` to string for output.

Note: The problem says input n (0 ≤ n ≤ 50), so we compute up to that n.

- Read n
- Precompute binomials up to n (but n≤50, so up to 50)
- Compute I[0..n] iteratively
- Output I[n] as num/den

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cctype>
#include <cmath>
#include <climits>
using namespace std;

// gcd for long long (and we'll use for __int128)
__int128 gcd(__int128 a, __int128 b) {
    if (b == 0) return a < 0 ? -a : a; // make positive
    return gcd(b, a % b);
}

string to_string(__int128 x) {
    if (x == 0) return "0";
    string s = "";
    bool neg = false;
    if (x < 0) {
        neg = true;
        x = -x;
    }
    while (x > 0) {
        s = char('0' + (x % 10)) + s;
        x /= 10;
    }
    if (neg) s = '-' + s;
    return s;
}

int main() {
    int N;
    cin >> N;

    // Precompute binomial coefficients up to 50
    vector<vector<long long>> C(51, vector<long long>(51, 0));
    for (int i = 0; i <= 50; i++) {
        C[i][0] = 1;
        for (int j = 1; j <= i; j++) {
            C[i][j] = C[i-1][j-1] + C[i-1][j];
        }
    }

    // Store I[k] = {num, den} as __int128
    vector<pair<__int128, __int128>> I(N+1);

    I[0] = make_pair(1, 1); // I0 = 1/1

    for (int i = 1; i <= N; i++) {
        // Compute S = 1 + sum_{k=0}^{i-1} C[i][k] * I[k]
        __int128 sum_num = 1; // represents 1/1
        __int128 sum_den = 1;

        for (int k = 0; k < i; k++) {
            // term = C[i][k] * I[k].num / I[k].den
            __int128 term_num = (__int128)C[i][k] * I[k].first;
            __int128 term_den = I[k].second;

            // sum = sum + term
            sum_num = sum_num * term_den + term_num * sum_den;
            sum_den = sum_den * term_den;

            // Reduce sum
            __int128 g = gcd(sum_num, sum_den);
            if (g != 0) {
                sum_num /= g;
                sum_den /= g;
            } else {
                // both zero? shouldn't happen
                sum_num = 0;
                sum_den = 1;
            }
        }

        // d_i = 3 * 2^i - 2
        __int128 d_i = (__int128)3 * (1LL << i) - 2; // 1LL << i is 2^i

        // I_i = sum_num / (sum_den * d_i)
        __int128 num = sum_num;
        __int128 den = sum_den * d_i;

        // Reduce fraction
        __int128 g = gcd(num, den);
        if (g != 0) {
            num /= g;
            den /= g;
        } else {
            num = 0;
            den = 1;
        }

        // Ensure positive (should be)
        if (den < 0) {
            num = -num;
            den = -den;
        }

        I[i] = make_pair(num, den);
    }

    // Output I[N]
    string num_str = to_string(I[N].first);
    string den_str = to_string(I[N].second);
    cout << num_str << "/" << den_str << endl;

    return 0;
}
```

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

__int128 gcd(__int128 a, __int128 b) {
    a = a < 0 ? -a : a;
    b = b < 0 ? -b : b;
    while (b != 0) {
        a %= b;
        swap(a, b);
    }
    return a;
}

string to_string(__int128 x) {
    if (x == 0) return "0";
    string s = "";
    bool neg = false;
    if (x < 0) {
        neg = true;
        x = -x;
    }
    while (x > 0) {
        s = char('0' + (x % 10)) + s;
        x /= 10;
    }
    if (neg) s = '-' + s;
    return s;
}

int main() {
    int N;
    cin >> N;

    vector<vector<long long>> C(51, vector<long long>(51, 0));
    for (int i = 0; i <= 50; i++) {
        C[i][0] = 1;
        for (int j = 1; j <= i; j++) {

    I[0] = make_pair(1, 1);

    for (int i = 1; i <= N; i++) {
        __int128 sum_num = 1;
        __int128 sum_den = 1;

        for (int k = 0; k < i; k++) {
            __int128 term_num = (__int128)C[i][k] * I[k].first;

                sum_num /= g;
                sum_den /= g;
```
