We are given up to 250,000 queries, but at most 1000 segments present at any time.
 Since the number of segments is small (<=1000), we can simply store all current segments and for each '+' query, 
 count the number of existing segments (before adding the new one) that are inside the new segment [L, R].
 
 Note: A segment [a, b] is inside [L, R] (non-strictly) if and only if L <= a and b <= R.

 Steps:
 - We'll maintain a list (or vector) of current segments (each segment as a pair (L, R)).
 - For a '+' query:
        Before adding the new segment, count how many existing segments [a, b] satisfy: L <= a and b <= R.
        Then add the new segment (L, R) to the list.
        Output the count.

 - For a '-' query:
        Remove one occurrence of the segment (L, R) from the list. Since there might be duplicates, we remove the first occurrence.

 However, note: the problem says "remove one segment", and if there is no such segment, ignore.

 Important: We must remove exactly one segment for '-'.

 Code:


```cpp
#include <iostream>
#include <vector>
#include <cctype>
#include <string>
using namespace std;

int main() {
    vector<pair<int, int>> segments;
    char op;
    int L, R;

    while (cin >> op >> L >> R) {
        if (op == '+') {
            int count = 0;
            for (const auto& seg : segments) {
                if (L <= seg.first && seg.second <= R) {
                    count++;
                }
            }
            cout << count << '\n';
            segments.push_back({L, R});
        } else if (op == '-') {
            for (auto it = segments.begin(); it != segments.end(); ++it) {
                if (it->first == L && it->second == R) {
                    segments.erase(it);
                    break;
                }
            }
        }
    }

    return 0;
}
```
