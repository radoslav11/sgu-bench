typedef vector<vector<long long>> Matrix;

Matrix multiply(const Matrix& a, const Matrix& b, long long mod) {
    int n = a.size();
    int m = b[0].size();
    int k = b.size();
    Matrix res(n, vector<long long>(m, 0));
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            for (int l = 0; l < k; ++l) {
                res[i][j] = (res[i][j] + a[i][l] * b[l][j]) % mod;
            }
        }
    }
    return res;
}

Matrix matrix_power(const Matrix& base, const string& exp_str, long long mod) {

    // We'll use exponentiation by repeated squaring on big number represented as string
    // We'll do exponentiation by processing digits from left to right, using binary exponentiation
    // But since exponent can be huge (up to 10^100), we need to convert to binary representation or use digit DP
    
    // Alternative: use exponentiation by repeated squaring in binary, but first convert string to binary
    // However, N can be up to 10^100, so binary representation is about log2(10^100) ≈ 332 bits.
    // We can convert the string number to binary representation.
    
    // Convert decimal string to binary
    string binary = "";
    string num = exp_str;
    while (num != "0") {
        int carry = 0;
        string new_num = "";
        for (char c : num) {
            int digit = c - '0';
            int val = carry * 10 + digit;
            carry = val % 2;
            if (new_num.empty() && val / 2 == 0) {
                // skip leading zero
            } else {
                new_num += (char)('0' + val / 2);
            }
        }
        if (new_num.empty()) new_num = "0";
        num = new_num;
        binary = (char)('0' + carry) + binary;
    }
    
    // If number is 0, binary should be "0", but our N >= 1 so no problem
    if (binary.empty()) binary = "0";
    
    int n = base.size();
    Matrix result(n, vector<long long>(n, 0));
    for (int i = 0; i < n; ++i) {
        result[i][i] = 1;
    }
    
    Matrix current = base;
    // Process binary from most significant to least significant
    for (char bit : binary) {
        result = multiply(result, result, mod);
        if (bit == '1') {
            result = multiply(result, current, mod);
        }
    }
    
    return result;
}
