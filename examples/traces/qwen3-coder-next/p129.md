#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>

using namespace std;

struct Point {
    long long x, y;
    Point() {}
    Point(long long x, long long y) : x(x), y(y) {}
};

// Cross product of vectors OA and OB
long long cross(const Point& O, const Point& A, const Point& B) {
    return (A.x - O.x) * (B.y - O.y) - (A.y - O.y) * (B.x - O.x);
}

// Check if point p is strictly inside the convex polygon
bool pointInsideConvexPolygon(const vector<Point>& poly, const Point& p) {
    int n = poly.size();
    for (int i = 0; i < n; ++i) {
        int j = (i + 1) % n;
        if (cross(poly[i], poly[j], p) <= 0) {
            return false;
        }
    }
    return true;
}

// Check if segment ab intersects the polygon boundary strictly (excluding endpoints)
bool segmentIntersectsBoundary(const vector<Point>& poly, const Point& a, const Point& b) {
    int n = poly.size();
    
    // Check if segment ab crosses any edge of the polygon
    for (int i = 0; i < n; ++i) {
        int j = (i + 1) % n;
        
        // Check if segments ab and poly[i]poly[j] intersect properly
        long long d1 = cross(poly[i], poly[j], a);
        long long d2 = cross(poly[i], poly[j], b);
        
        // If both points are on the same side (or one is on the line), no proper intersection
        if ((d1 > 0 && d2 > 0) || (d1 < 0 && d2 < 0)) {
            continue;
        }
        
        // Check if points a and b are on opposite sides of edge i
        if (d1 == 0 || d2 == 0) {
            // One endpoint lies on the edge - not a proper intersection (boundary touch)
            continue;
        }
        
        // Now check if poly[i]poly[j] crosses ab
        long long d3 = cross(a, b, poly[i]);
        long long d4 = cross(a, b, poly[j]);
        
        if ((d3 > 0 && d4 > 0) || (d3 < 0 && d4 < 0)) {
            continue;
        }
        
        if (d3 == 0 || d4 == 0) {
            continue;
        }
        
        // Proper intersection
        return true;
    }
    
    return false;
}

// Get intersection points of segment ab with polygon edges
vector<pair<double, Point>> getIntersections(const vector<Point>& poly, const Point& a, const Point& b) {
    vector<pair<double, Point>> intersections;
    int n = poly.size();
    
    for (int i = 0; i < n; ++i) {
        int j = (i + 1) % n;
        
        // Compute intersection of segment ab and edge poly[i]poly[j]
        long long x1 = a.x, y1 = a.y;
        long long x2 = b.x, y2 = b.y;
        long long x3 = poly[i].x, y3 = poly[i].y;
        long long x4 = poly[j].x, y4 = poly[j].y;
        
        long long det = (x1 - x2) * (y3 - y4) - (y1 - y2) * (x3 - x4);
        if (det == 0) {
            continue; // Parallel segments
        }
        
        long long px = (long long)(x1 * y2 - y1 * x2);
        long long py = (long long)(x3 * y4 - y3 * x4);
        
        long long x = px * (x3 - x4) - py * (x1 - x2);
        long long y = px * (y3 - y4) - py * (y1 - y2);
        
        // Check if intersection point lies on both segments
        Point intersect(x / det, y / det);
        
        // Check bounds using cross products for robustness
        if (cross(a, b, intersect) == 0 && 
            min(x1, x2) <= intersect.x && intersect.x <= max(x1, x2) &&
            min(y1, y2) <= intersect.y && intersect.y <= max(y1, y2)) {
            
            if (cross(poly[i], poly[j], intersect) == 0 &&
                min(x3, x4) <= intersect.x && intersect.x <= max(x3, x4) &&
                min(y3, y4) <= intersect.y && intersect.y <= max(y3, y4)) {
                
                // Calculate parameter t in [0,1] for point on segment ab
                double t = 0.0;
                if (abs(x2 - x1) > abs(y2 - y1)) {
                    t = (double)(intersect.x - x1) / (x2 - x1);
                } else {
                    t = (double)(intersect.y - y1) / (y2 - y1);
                }
                
                if (t >= 0.0 && t <= 1.0) {
                    intersections.push_back({t, intersect});
                }
            }
        }
    }
    
    return intersections;
}

double segmentLength(const Point& a, const Point& b) {
    double dx = (double)a.x - b.x;
    double dy = (double)a.y - b.y;
    return sqrt(dx * dx + dy * dy);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int N;
    cin >> N;
    
    vector<Point> polygon(N);
    for (int i = 0; i < N; ++i) {
        cin >> polygon[i].x >> polygon[i].y;
    }
    
    // Compute convex hull to get the boundary in order
    vector<Point> pts = polygon;
    sort(pts.begin(), pts.end(), [](const Point& a, const Point& b) {
        return make_pair(a.x, a.y) < make_pair(b.x, b.y);
    });
    
    vector<Point> hull;
    // Lower hull
    for (int i = 0; i < N; ++i) {
        while (hull.size() >= 2 && cross(hull[hull.size()-2], hull.back(), pts[i]) <= 0) {
            hull.pop_back();
        }
        hull.push_back(pts[i]);
    }
    
    // Upper hull
    int lower_size = hull.size();
    for (int i = N-2; i >= 0; --i) {
        while (hull.size() > lower_size && cross(hull[hull.size()-2], hull.back(), pts[i]) <= 0) {
            hull.pop_back();
        }
        hull.push_back(pts[i]);
    }
    
    // Remove duplicate first point
    if (hull.size() > 1) {
        hull.pop_back();
    }
    
    int M;
    cin >> M;
    
    for (int j = 0; j < M; ++j) {
        int ax1, ay1, ax2, ay2;
        cin >> ax1 >> ay1 >> ax2 >> ay2;
        
        Point a(ax1, ay1);
        Point b(ax2, ay2);
        
        // Check if both endpoints are strictly inside
        bool aInside = pointInsideConvexPolygon(hull, a);
        bool bInside = pointInsideConvexPolygon(hull, b);
        
        // If both endpoints are inside and segment doesn't cross boundary
        if (aInside && bInside && !segmentIntersectsBoundary(hull, a, b)) {
            cout << fixed << setprecision(2) << segmentLength(a, b) << '\n';
            continue;
        }
        
        // Get intersection points
        vector<pair<double, Point>> intersections = getIntersections(hull, a, b);
        
        // If no intersection, segment is entirely outside or tangent
        if (intersections.size() == 0) {
            cout << fixed << setprecision(2) << 0.0 << '\n';
            continue;
        }
        
        // Sort intersection points by parameter t along the segment
        sort(intersections.begin(), intersections.end());
        
        // Find the sub-segment inside the polygon
        double lengthInside = 0.0;
        bool inPolygon = false;
        Point startPoint, endPoint;
        
        // Check endpoints first
        bool aStrictlyInside = pointInsideConvexPolygon(hull, a);
        bool bStrictlyInside = pointInsideConvexPolygon(hull, b);
        
        if (aStrictlyInside) {
            inPolygon = true;
            startPoint = a;
        }
        
        // Process intersection points
        for (size_t i = 0; i < intersections.size(); ++i) {
            if (!inPolygon && i + 1 < intersections.size()) {
                // Entering the polygon
                startPoint = intersections[i].second;
                endPoint = intersections[i+1].second;
                
                // Check if the midpoint between intersection points is inside
                double midT = (intersections[i].first + intersections[i+1].first) / 2.0;
                Point mid;
                mid.x = a.x + (long long)((b.x - a.x) * midT);
                mid.y = a.y + (long long)((b.y - a.y) * midT);
                
                if (pointInsideConvexPolygon(hull, mid)) {
                    lengthInside += segmentLength(startPoint, endPoint);
                }
                
                i++; // Skip next intersection as it's already processed
            } else if (inPolygon && i + 1 < intersections.size()) {
                // Already inside and encountered another intersection
                endPoint = intersections[i].second;
                
                // Check if endpoint of the segment is inside
                bool bStrictlyInside = pointInsideConvexPolygon(hull, b);
                
                // Calculate the part inside
                double segmentStartT = intersections[i-1].first;
                double segmentEndT = intersections[i].first;
                
                if (bStrictlyInside) {
                    lengthInside += segmentLength(intersections[i-1].second, b);
                } else {
                    lengthInside += segmentLength(intersections[i-1].second, intersections[i].second);
                }
                
                inPolygon = false;
            } else if (inPolygon && i + 1 == intersections.size()) {
                // Segment ends inside
                endPoint = b;
                lengthInside += segmentLength(intersections[i-1].second, b);
            }
        }
        
        // Handle special case: only one intersection and both endpoints inside
        if (intersections.size() == 1) {
            // This shouldn't happen for a convex polygon, but handle it anyway
            if (aStrictlyInside && bStrictlyInside) {
                lengthInside = segmentLength(a, b);
            } else if (aStrictlyInside) {
                // Segment starts inside and exits once
                lengthInside = segmentLength(a, intersections[0].second);
            } else if (bStrictlyInside) {
                // Segment enters and ends inside
                lengthInside = segmentLength(intersections[0].second, b);
            }
        } else if (intersections.size() == 2 && aStrictlyInside && bStrictlyInside) {
            // Segment enters and exits, but both endpoints are inside
            lengthInside = segmentLength(a, b);
        }
        
        cout << fixed << setprecision(2) << lengthInside << '\n';
    }
    
    return 0;
}
