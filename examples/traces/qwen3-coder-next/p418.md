    for (const string& rule : uniqueRules) {
        rules.push_back(rule);
    }
    
    // Sort lexicographically
    sort(rules.begin(), rules.end());
    
    return rules;
}

int main() {
    vector<string> lines;
    string line;
    while (getline(cin, line)) {
        lines.push_back(line);
    }
    
    // Extract routine names
    set<string> allRoutineNames = extractRoutineNames(lines);
    
    // Map routine name to its body
    map<string, string> routineBodies;
    for (const string& name : allRoutineNames) {
        routineBodies[name] = extractBody(lines, name);
    }
    
    // Get production rules for each routine
    map<string, vector<string>> rulesMap;
    for (const string& name : allRoutineNames) {
        rulesMap[name] = parseBody(routineBodies[name], allRoutineNames);
    }
    
    // Output in required format
    // Sort by routine name (lexicographically)
    vector<string> sortedNames(allRoutineNames.begin(), allRoutineNames.end());
    sort(sortedNames.begin(), sortedNames.end());
    
    for (const string& name : sortedNames) {
        cout << "<" << name << ">::=";
        
        const vector<string>& r = rulesMap[name];
        for (size_t i = 0; i < r.size(); i++) {
            if (i > 0) cout << "|";
            cout << r[i];
        }
        cout << "\n";
    }
    
    return 0;
}
