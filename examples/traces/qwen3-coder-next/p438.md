To solve this problem, we need to determine the minimal time required for all tourists to cross a river using floating piles of rubbish, where each pile has a capacity constraint and tourists can jump up to a maximum distance in any direction. The solution involves modeling the problem as a flow problem over time, where we use binary search to find the minimal time `T` such that all `M` tourists can reach the opposite bank within `T` seconds.

### Approach
1. **Binary Search on Time `T`**: We perform binary search over possible time values (from 1 to a reasonable upper bound, say 2000) to find the minimal `T` where all tourists can cross.
2. **Flow Over Time Model**: For each candidate time `T`, we construct a layered flow network where:
   - **Nodes**: Each layer corresponds to a time step (1 to `T`). Each pile is split into input and output nodes to enforce capacity constraints.
   - **Edges**: 
     - From the source (start bank) to input nodes of piles reachable from the start bank (distance ≤ `D`).
     - From input nodes to output nodes of the same pile with capacity equal to the pile's capacity.
     - From output nodes to the sink (opposite bank) if the pile can reach the opposite bank (distance ≤ `D`).
     - From output nodes of one pile to input nodes of any other pile (including itself) if the jump distance is ≤ `D`.
     - Direct edge from source to sink if the river width `W` ≤ `D`.
3. **Max Flow Calculation**: For each time step, we compute the maximum flow in the constructed network. The total flow accumulated over `T` steps must be at least `M` for `T` to be feasible.
4. **Feasibility Check**: The binary search checks each candidate `T` using the flow network to determine if all tourists can cross within `T` seconds.

### Solution Code
```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <cmath>
#include <climits>

using namespace std;

struct Pile {
    int x, y, c;
};

bool feasible(int T, int M, int D, int W, const vector<Pile>& piles) {
    int N = piles.size();
    const long long INF = 1000000;
    vector<long long> prev_cur(N, 0);
    long long total_flow = 0;

    for (int t = 1; t <= T; t++) {
        int nodes = 2 * N + 2;
        vector<vector<long long>> cap(nodes, vector<long long>(nodes, 0));
        int sink_node = nodes - 1;

        for (int i = 0; i < N; i++) {
            cap[0][i + 1] = M + prev_cur[i];
            cap[i + 1][N + 1 + i] = piles[i].c;
            if (W - piles[i].y <= D) {
                cap[N + 1 + i][sink_node] = INF;
            }
            for (int j = 0; j < N; j++) {
                long long dx = piles[i].x - piles[j].x;
                long long dy = piles[i].y - piles[j].y;
                if (dx * dx + dy * dy <= (long long)D * D) {
                    cap[N + 1 + i][j + 1] = INF;
                }
            }
        }

        if (W <= D) {
            cap[0][sink_node] = INF;
        }

        long long flow_value = 0;
        while (true) {
            vector<int> parent(nodes, -1);
            queue<int> q;
            q.push(0);
            parent[0] = -2;
            while (!q.empty() && parent[sink_node] == -1) {
                int u = q.front();
                q.pop();
                for (int v = 0; v < nodes; v++) {
                    if (parent[v] == -1 && cap[u][v] > 0) {
                        parent[v] = u;
                        q.push(v);
                    }
                }
            }
            if (parent[sink_node] == -1) break;
            long long path_flow = INF;
            int v = sink_node;
            while (v != 0) {
                int u = parent[v];
                path_flow = min(path_flow, cap[u][v]);
                v = u;
            }
            v = sink_node;
            while (v != 0) {
                int u = parent[v];
                cap[u][v] -= path_flow;
                cap[v][u] += path_flow;
                v = u;
            }
            flow_value += path_flow;
        }

        total_flow += flow_value;
        if (total_flow >= M) {
            return true;
        }

        vector<long long> next_cur(N, 0);
        for (int i = 0; i < N; i++) {
            for (int j = 0; j < N; j++) {
                if (cap[N + 1 + i][j + 1] < INF) {
                    long long flow_edge = INF - cap[N + 1 + i][j + 1];
                    next_cur[j] += flow_edge;
                }
            }
        }
        prev_cur = next_cur;
    }

    return total_flow >= M;
}

int main() {
    int N, M, D, W;
    cin >> N >> M >> D >> W;

    vector<Pile> piles(N);
    for (int i = 0; i < N; i++) {
        cin >> piles[i].x >> piles[i].y >> piles[i].c;
    }

    if (M == 0) {
        cout << 0 << endl;
        return 0;
    }

    int low = 1;
    int high = 2000;
    int ans = -1;

    while (low <= high) {
        int mid = (low + high) / 2;
        if (feasible(mid, M, D, W, piles)) {
            ans = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    if (ans == -1) {
        cout << "IMPOSSIBLE" << endl;
    } else {
        cout << ans << endl;
    }

    return 0;
}
```

### Explanation
1. **Binary Search**: The solution uses binary search to efficiently determine the minimal time `T` required for all tourists to cross the river.
2. **Flow Network Construction**: For each candidate time `T`, the algorithm constructs a flow network with layers representing time steps. Each pile is split into input and output nodes to enforce capacity constraints.
3. **Max Flow Calculation**: Using the Ford-Fulkerson algorithm (with BFS for shortest augmenting paths), the solution computes the maximum flow for each time step. The flow represents the number of tourists moving towards the opposite bank.
4. **Feasibility Check**: The algorithm checks if the accumulated flow over `T` steps is sufficient to transport all `M` tourists. If yes, `T` is feasible; otherwise, it searches for a larger `T`.
5. **Edge Cases**: Direct jumps from the start bank to the opposite bank (if `W ≤ D`) are handled by adding a direct edge from the source to the sink.

This approach efficiently checks feasibility for different time values using flow networks and binary search, ensuring optimal performance even for the upper constraint limits.
