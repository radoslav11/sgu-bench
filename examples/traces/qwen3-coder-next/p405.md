#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n, m;
    cin >> n >> m;
    
    vector<int> scores(n, 0);
    
    for (int game = 0; game < m; ++game) {
        int actual_a, actual_b;
        cin >> actual_a >> actual_b;
        
        for (int participant = 0; participant < n; ++participant) {
            int pred_a, pred_b;
            cin >> pred_a >> pred_b;
            
            // Check if winner (or tie) is guessed
            bool actual_win_first = actual_a > actual_b;
            bool actual_win_second = actual_a < actual_b;
            bool actual_tie = actual_a == actual_b;
            
            bool pred_win_first = pred_a > pred_b;
            bool pred_win_second = pred_a < pred_b;
            bool pred_tie = pred_a == pred_b;
            
            if ((actual_win_first && pred_win_first) || 
                (actual_win_second && pred_win_second) || 
                (actual_tie && pred_tie)) {
                scores[participant] += 2;
            }
            
            // Check if goal difference is guessed
            if (actual_a - actual_b == pred_a - pred_b) {
                scores[participant] += 3;
            }
            
            // Check if exact goals for first team
            if (actual_a == pred_a) {
                scores[participant] += 1;
            }
            
            // Check if exact goals for second team
            if (actual_b == pred_b) {
                scores[participant] += 1;
            }
        }
    }
    
    for (int i = 0; i < n; ++i) {
        if (i > 0) cout << " ";
        cout << scores[i];
    }
    cout << endl;
    
    return 0;
}
