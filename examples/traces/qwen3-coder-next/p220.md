```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>
using namespace std;

int main() {
    int n, k;
    cin >> n >> k;

    // For n=1, special case: only 1 square
    if (n == 1) {
        cout << (k <= 1 ? 1 : 0) << endl;
        return 0;
    }

    // The key observation: bishops attack along diagonals.
    // There are two sets of diagonals: 
    //  - diagonals running from top-left to bottom-right (let's call them "positive" diagonals)
    //  - diagonals running from top-right to bottom-left (let's call them "negative" diagonals)
    //
    // Each square belongs to exactly one positive and one negative diagonal.
    // Bishops attack each other if they share the same positive or same negative diagonal.
    //
    // Important: The number of positive diagonals is 2*n-1, same for negative.
    // But note: the center diagonal (for odd n) is shared by both sets in terms of length,
    // but we don't need that directly.
    //
    // Alternate approach: 
    // We can split the board into black and white squares (like a chessboard).
    // Bishops on black squares never attack bishops on white squares, because diagonals 
    // alternate colors. So we can solve independently for black and white squares.
    //
    // How many black squares? For an n x n board:
    //   If n is even: n*n/2 black, n*n/2 white.
    //   If n is odd: (n*n+1)/2 of one color, (n*n-1)/2 of the other.
    //
    // But actually, the diagonals: 
    //   Positive diagonals: indices from 0 to 2*n-2.
    //   For a square (i, j): positive diagonal index = i+j, negative = i-j + n-1.
    //
    // However, a more efficient known solution for small n (n<=10) is backtracking.
    // Since n is at most 10, the board has at most 100 squares, but k is at most 100.
    // But 100 choose k is too big for brute force.
    //
    // Instead, we can use backtracking with pruning: place bishops row by row, 
    // and for each row, try placing a bishop in each column if it doesn't conflict.
    // We need to track which positive and negative diagonals are occupied.
    // 
    // Positive diagonal: i+j (range 0 to 2*n-2)
    // Negative diagonal: i-j + n-1 (range 0 to 2*n-2)
    //
    // Since n<=10, 2*n-1 <= 19, so we can use bit masks or arrays of bool.
    //
    // We'll do DFS: process row by row. For each row, try columns that are not attacked.
    // We also need to consider that we only need to place k bishops.

    // Let's use:
    // diag1[i+j] = true if diagonal i+j is occupied
    // diag2[i-j+n-1] = true if diagonal i-j+n-1 is occupied

    vector<bool> diag1(2 * n, false); // index: i+j, max value = (n-1)+(n-1) = 2*n-2 -> size 2*n-1, but we use 2*n to be safe
    vector<bool> diag2(2 * n, false); // index: i-j + n-1, min i-j = -(n-1), max = n-1 -> so index from 0 to 2*n-2

    long long count = 0;

    // We'll do DFS row by row (0 to n-1)
    function<void(int, int)> dfs = [&](int row, int placed) {
        if (placed == k) {
            count++;
            return;
        }
        if (row == n) {
            return;
        }

        // Option 1: skip current row entirely
        dfs(row + 1, placed);

        // Option 2: try placing a bishop in each column of current row
        for (int col = 0; col < n; col++) {
            int d1 = row + col;
            int d2 = row - col + n - 1;
            if (!diag1[d1] && !diag2[d2]) {
                // place bishop
                diag1[d1] = true;
                diag2[d2] = true;
                dfs(row + 1, placed + 1);
                // backtrack
                diag1[d1] = false;
                diag2[d2] = false;
            }
        }
    };

    dfs(0, 0);

    cout << count << endl;

    return 0;
}
```
