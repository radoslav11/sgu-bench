To solve this problem, we need to determine if a control function exists for a given control matrix such that applying the function to every element of the matrix results in another control matrix. A control matrix is defined as a matrix where the first row differs from all other rows. The control function must map all distinct values in the matrix to non-negative integers not exceeding 50.

### Approach
1. **Problem Analysis**:
   - **Control Matrix Condition**: The first row must differ from every other row in at least one column.
   - **Control Function Condition**: After applying the function, the first row must still differ from every other row in at least one column.
   - **Key Insight**: For each row (other than the first), there must be at least one column where the function values of the corresponding elements in the first row and the current row are different.

2. **Critical Observation**:
   - For any value `x` in the first row, if there are more than 50 rows where every occurrence of `x` in the first row corresponds to a different value in those rows (i.e., no `x` in those rows where the first row has `x`), then we need more than 51 distinct values for the function (since `f(x)` must differ from each of these values), which violates the constraint (values ≤ 50). Hence, the solution is impossible in such cases.

3. **Algorithm**:
   - **Force Count Calculation**: For each value `x` in the first row, count the number of rows where every edge (difference between first row and current row in a column) involves `x`. If any value has a force count exceeding 50, output "No".
   - **Graph Construction**: For valid cases, construct a graph where edges represent chosen differences (one per row) that minimize the maximum degree of any node (to ensure ≤ 50 degrees, allowing coloring with 51 colors).
   - **Greedy Coloring**: Assign function values (colors) to values in the matrix such that adjacent nodes in the graph have different colors, using the smallest available color not used by neighbors.

### Solution Code
```cpp
#include <iostream>
#include <vector>
#include <set>
#include <map>
#include <algorithm>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n, m;
    cin >> n >> m;
    vector<vector<int>> T(n, vector<int>(m));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> T[i][j];
        }
    }

    if (n == 1) {
        set<int> vals;
        for (int j = 0; j < m; j++) {
            vals.insert(T[0][j]);
        }
        cout << "Yes" << endl;
        for (int x : vals) {
            cout << x << " -> " << 0 << '\n';
        }
        return 0;
    }

    set<int> all_vals_set;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            all_vals_set.insert(T[i][j]);
        }
    }
    vector<int> all_vals(all_vals_set.begin(), all_vals_set.end());

    vector<vector<pair<int, int>>> row_edges;
    map<int, int> force_count;

    for (int i = 1; i < n; i++) {
        vector<pair<int, int>> edges;
        for (int j = 0; j < m; j++) {
            if (T[0][j] != T[i][j]) {
                edges.push_back({T[0][j], T[i][j]});
            }
        }
        row_edges.push_back(edges);

        if (edges.empty()) {
            continue;
        }

        set<int> cand;
        cand.insert(edges[0].first);
        cand.insert(edges[0].second);
        for (int k = 1; k < edges.size(); k++) {
            set<int> new_cand;
            for (int x : cand) {
                if (x == edges[k].first || x == edges[k].second) {
                    new_cand.insert(x);
                }
            }
            cand = new_cand;
            if (cand.empty()) {
                break;
            }
        }
        for (int x : cand) {
            force_count[x]++;
        }
    }

    bool possible = true;
    for (auto &p : force_count) {
        if (p.second > 50) {
            possible = false;
            break;
        }
    }

    if (!possible) {
        cout << "No" << endl;
        return 0;
    }

    map<int, int> deg;
    for (int x : all_vals) {
        deg[x] = 0;
    }

    vector<pair<int, int>> chosen_edges;
    for (int i = 0; i < row_edges.size(); i++) {
        vector<pair<int, int>>& edges = row_edges[i];
        int best_index = 0;
        int best_val = 1000000000;
        for (int k = 0; k < edges.size(); k++) {
            int a = edges[k].first;
            int b = edges[k].second;
            int current = max(deg[a], deg[b]);
            if (current < best_val) {
                best_val = current;
                best_index = k;
            }
        }
        int a = edges[best_index].first;
        int b = edges[best_index].second;
        chosen_edges.push_back({a, b});
        deg[a]++;
        deg[b]++;
    }

    map<int, vector<int>> adj;
    for (int x : all_vals) {
        adj[x] = vector<int>();
    }
    for (auto &e : chosen_edges) {
        int a = e.first;
        int b = e.second;
        adj[a].push_back(b);
        adj[b].push_back(a);
    }

    map<int, int> f;
    for (int x : all_vals) {
        vector<bool> used(52, false);
        for (int neighbor : adj[x]) {
            if (f.find(neighbor) != f.end()) {
                if (f[neighbor] < 52) {
                    used[f[neighbor]] = true;
                }
            }
        }
        int color = 0;
        while (color < 52 && used[color]) {
            color++;
        }
        f[x] = color;
    }

    cout << "Yes" << endl;
    for (int x : all_vals) {
        cout << x << " -> " << f[x] << '\n';
    }

    return 0;
}
```

### Explanation
1. **Input Handling**: The matrix is read, and if there's only one row, any function (e.g., all zeros) suffices.
2. **Distinct Values Collection**: All unique values in the matrix are collected for processing.
3. **Force Count Calculation**: For each value in the first row, we count rows where every edge (difference) involves that value. If any value's count exceeds 50, the solution is impossible ("No").
4. **Graph Construction**: For valid cases, we select one edge per row (minimizing the current maximum degree of endpoints) to form a graph where each node's degree is ≤ 50.
5. **Greedy Coloring**: Nodes (values) are colored with the smallest available color not used by neighbors, ensuring adjacent nodes have different colors. The solution outputs the function mapping values to colors in increasing order.

This approach efficiently checks feasibility and constructs the control function while adhering to constraints, leveraging graph theory and greedy coloring for optimal results.
