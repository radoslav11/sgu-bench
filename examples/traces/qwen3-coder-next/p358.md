#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    std::vector<int> medians;
    
    for (int i = 0; i < 3; ++i) {
        std::vector<int> nums(3);
        std::cin >> nums[0] >> nums[1] >> nums[2];
        std::sort(nums.begin(), nums.end());
        medians.push_back(nums[1]);
    }
    
    std::sort(medians.begin(), medians.end());
    std::cout << medians[1] << std::endl;
    
    return 0;
}
