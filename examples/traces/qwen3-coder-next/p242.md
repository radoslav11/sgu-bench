        #include <bits/stdc++.h>

        const int INF = 1e9;

        struct Dinic {
            struct Edge {
                int from, to, cap, flow;
                Edge(int from, int to, int cap) : from(from), to(to), cap(cap), flow(0) {}
            };

            vector<Edge> edges;
            vector<vector<int>> adj;
            vector<int> level, ptr;
            int n;

            Dinic(int n) : n(n) {
                adj.resize(n);
                level.resize(n);
                ptr.resize(n);
            }

            void add_edge(int from, int to, int cap) {
                adj[from].push_back(edges.size());
                edges.push_back(Edge(from, to, cap));
                adj[to].push_back(edges.size());
                edges.push_back(Edge(to, from, 0));
            }

            bool bfs(int s, int t) {
                fill(level.begin(), level.end(), -1);
                queue<int> q;
                level[s] = 0;
                q.push(s);
                while (!q.empty()) {
                    int u = q.front(); q.pop();
                    for (int id : adj[u]) {
                        Edge& e = edges[id];
                        if (level[e.to] == -1 && e.flow < e.cap) {
                            level[e.to] = level[u] + 1;
                            q.push(e.to);
                        }
                    }
                }
                return level[t] != -1;
            }

            int dfs(int u, int t, int pushed) {
                if (pushed == 0) return 0;
                if (u == t) return pushed;
                for (int& i = ptr[u]; i < adj[u].size(); i++) {
                    int id = adj[u][i];
                    Edge& e = edges[id];
                    if (level[e.to] == level[u] + 1 && e.flow < e.cap) {
                        int how = dfs(e.to, t, min(pushed, e.cap - e.flow));
                        if (how) {
                            e.flow += how;
                            edges[id ^ 1].flow -= how;
                            return how;
                        }
                    }
                }
                return 0;
            }

            int max_flow(int s, int t) {
                int flow = 0;
                while (bfs(s, t)) {
                    fill(ptr.begin(), ptr.end(), 0);
                    while (int pushed = dfs(s, t, INF)) {
                        flow += pushed;
                    }
                }
                return flow;
            }
        };

#include <vector>
#include <queue>
#include <algorithm>
#include <set>
#include <map>

using namespace std;

const int INF = 1e9;

struct Dinic {
    struct Edge {
        int from, to, cap, flow;
        Edge(int from, int to, int cap) : from(from), to(to), cap(cap), flow(0) {}
    };

    vector<Edge> edges;
    vector<vector<int>> adj;
    vector<int> level, ptr;
    int n;

    Dinic(int n) : n(n) {
        adj.resize(n);
        level.resize(n);
        ptr.resize(n);
    }

    void add_edge(int from, int to, int cap) {
        adj[from].push_back(edges.size());
        edges.push_back(Edge(from, to, cap));
        adj[to].push_back(edges.size());
        edges.push_back(Edge(to, from, 0));
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int u = q.front(); q.pop();
            for (int id : adj[u]) {
                Edge& e = edges[id];
                if (level[e.to] == -1 && e.flow < e.cap) {
                    level[e.to] = level[u] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] != -1;
    }

    int dfs(int u, int t, int pushed) {
        if (pushed == 0) return 0;
        if (u == t) return pushed;
        for (int& i = ptr[u]; i < adj[u].size(); i++) {
            int id = adj[u][i];
            Edge& e = edges[id];
            if (level[e.to] == level[u] + 1 && e.flow < e.cap) {
                int how = dfs(e.to, t, min(pushed, e.cap - e.flow));
                if (how) {
                    e.flow += how;
                    edges[id ^ 1].flow -= how;
                    return how;
                }
            }
        }
        return 0;
    }

    int max_flow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(ptr.begin(), ptr.end(), 0);
            while (int pushed = dfs(s, t, INF)) {
                flow += pushed;
            }
        }
        return flow;
    }
};

    if (M < K || N < 2 * K) {

    }

    int source = 0;
    int sink = N + M + 2;
    int X_node = N + M + 1;
    int total_nodes = sink + 1;

    map<int, int> univ_to_idx;
    for (int i = 0; i < M; i++) {
        univ_to_idx[univ_list[i]] = i;

        for (int univ : prefs[i]) {

        int idx_before = dinic.edges.size();
        dinic.add_edge(univ_node, X_node, 2);
        edge_index_to_X[j] = idx_before;
    }

    for (int j = 0; j < M; j++) {
        int univ_node = N + 1 + j;
        dinic.add_edge(univ_node, sink, 1000);
    }

    int idx_edge_X_sink = dinic.edges.size();
    dinic.add_edge(X_node, sink, 2 * K);

    int total_flow = dinic.max_flow(source, sink);
    int flow_on_X_sink = dinic.edges[idx_edge_X_sink].flow;

    if (flow_on_X_sink == 2 * K) {
        cout << "YES" << endl;
        vector<vector<int>> assignment(M);

        for (int i = 0; i < N; i++) {
            int student_node = i+1;
            for (int id : dinic.adj[student_node]) {
                auto& e = dinic.edges[id];
                if (e.to >= N+1 && e.to <= N+M && e.flow == 1) {
                    int univ_idx = e.to - (N+1);

                }
            }
        }

        vector<pair<int, vector<int>>> output_univs;
        for (int j = 0; j < M; j++) {

        }

        for (auto& p : output_univs) {

            }
            cout << endl;
        }
    } else {
        cout << "NO" << endl;
    }

    return 0;
}

### Explanation
1. **Input Reading**: The code reads the number of students (N) and universities (K), followed by each student's preference list of universities.
2. **Initial Checks**: It checks if the number of distinct universities (M) is at least K and if there are enough students (N >= 2*K). If not, it outputs "NO".
3. **Flow Network Setup**: The flow network is constructed with:
   - **Source** connected to each student (capacity 1).
   - Each student connected to their preferred universities (capacity 1).
   - Each university connected to a special node X (capacity 2) to ensure at least two students if selected.
   - Each university connected directly to the sink (capacity 1000) for extra students.
   - Node X connected to the sink (capacity 2*K) to ensure exactly K universities are selected.
4. **Max Flow Computation**: Dinic's algorithm computes the maximum flow. If the flow to X is exactly 2*K, it means K universities can be selected with at least two students each.
5. **Output**: If feasible, it outputs "YES" followed by the list of students for each selected university. Otherwise, it outputs "NO".

This approach efficiently checks the feasibility of the assignment using flow network modeling and ensures optimal performance with Dinic's algorithm.
