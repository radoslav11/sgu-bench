void precompute_powers() {
    pow1[0] = 1;
    pow2[0] = 1;
    for (int i = 1; i < MAX; i++) {
        pow1[i] = (pow1[i-1] * base) % MOD1;
        pow2[i] = (pow2[i-1] * base) % MOD2;
