#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <climits>

using namespace std;

typedef long long ll;
typedef pair<int, ll> pll;

// Function to perform BFS/DFS from a source node and return the farthest node and its distance
pll bfs(int start, const vector<vector<pll>>& graph, int n) {
    vector<ll> dist(n + 1, -1);
    queue<int> q;
    
    q.push(start);
    dist[start] = 0;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (const auto& edge : graph[u]) {
            int v = edge.first;
            ll w = edge.second;
            
            if (dist[v] == -1) {
                dist[v] = dist[u] + w;
                q.push(v);
            }
        }
    }
    
    // Find the farthest node from start
    int farthestNode = start;
    ll maxDist = 0;
    
    for (int i = 1; i <= n; ++i) {
        if (dist[i] > maxDist) {
            maxDist = dist[i];
            farthestNode = i;
        }
    }
    
    return {farthestNode, maxDist};
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    
    vector<vector<pll>> graph(n + 1);
    
    // Build the tree (graph is undirected)
    for (int i = 2; i <= n; ++i) {
        int parent;
        ll weight;
        cin >> parent >> weight;
        
        graph[i].push_back({parent, weight});
        graph[parent].push_back({i, weight});
    }
    
    // Find the diameter endpoints
    pll firstEnd = bfs(1, graph, n);
    pll secondEnd = bfs(firstEnd.first, graph, n);
    
    // Now compute distances from both endpoints
    vector<ll> dist1(n + 1, -1), dist2(n + 1, -1);
    
    // BFS from first endpoint
    queue<int> q;
    q.push(firstEnd.first);
    dist1[firstEnd.first] = 0;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (const auto& edge : graph[u]) {
            int v = edge.first;
            ll w = edge.second;
            
            if (dist1[v] == -1) {
                dist1[v] = dist1[u] + w;
                q.push(v);
            }
        }
    }
    
    // BFS from second endpoint
    q.push(secondEnd.first);
    dist2[secondEnd.first] = 0;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (const auto& edge : graph[u]) {
            int v = edge.first;
            ll w = edge.second;
            
            if (dist2[v] == -1) {
                dist2[v] = dist2[u] + w;
                q.push(v);
            }
        }
    }
    
    // For each node, the maximum distance to any other node is max(dist1[i], dist2[i])
    // because in a tree, the farthest node from any vertex is one of the endpoints of the diameter
    for (int i = 1; i <= n; ++i) {
        cout << max(dist1[i], dist2[i]) << '\n';
    }
    
    return 0;
}
