#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

int main() {
    int N, x;
    std::cin >> N >> x;
    
    // Adjust x to 0-based index for internal use
    x--;
    
    std::vector<std::vector<int>> friends(N);
    
    for (int i = 0; i < N; ++i) {
        int d;
        std::cin >> d;
        friends[i].resize(d);
        for (int j = 0; j < d; ++j) {
            std::cin >> friends[i][j];
            friends[i][j]--; // Convert to 0-based
        }
    }
    
    // Set to store friends of friends (unique and sorted)
    std::set<int> foaf;
    
    // For each friend b of user x
    for (int b : friends[x]) {
        // For each friend c of user b
        for (int c : friends[b]) {
            // c is a friend of b
            // Skip if c is x or c is already a direct friend of x
            if (c == x) continue;
            if (std::find(friends[x].begin(), friends[x].end(), c) != friends[x].end()) {
                continue;
            }
            foaf.insert(c);
        }
    }
    
    // Output the count
    std::cout << foaf.size() << std::endl;
    
    // Output the friends of friends in increasing order (1-based)
    bool first = true;
    for (int c : foaf) {
        if (!first) std::cout << " ";
        std::cout << c + 1;
        first = false;
    }
    
    if (foaf.size() > 0) {
        std::cout << std::endl;
    }
    
    return 0;
}
