```cpp
#include <iostream>
#include <vector>
#include <iomanip>
#include <cmath>
#include <cstring>

using namespace std;

int main() {
    int N;
    cin >> N;
    
    vector<double> p(13);
    for (int i = 0; i < 12; ++i) {
        cin >> p[i];
    }
    
    double internetAvg = 0.0;
    for (int i = 0; i < N; ++i) {
        double prob;
        cin >> prob;
        internetAvg += prob;
    }
    internetAvg /= N;
    p[12] = internetAvg;
    
    // dp[sector_mask][club_score][viewer_score]
    // sector_mask: 13 bits, bit i set if envelope at sector i+1 is still on table
    // We only need to track states where neither team has reached 6 yet
    
    // Initialize dp
    // dp[mask][c][v] = probability of reaching state (mask, c, v)
    vector<vector<vector<double>>> dp(1 << 13, 
        vector<vector<double>>(7, vector<double>(7, 0.0)));
    
    // Initial state: all 13 envelopes present (mask = (1<<13)-1), 0-0 score
    dp[(1 << 13) - 1][0][0] = 1.0;
    
    // Process all possible states
    for (int mask = (1 << 13) - 1; mask >= 0; --mask) {
        for (int c = 0; c <= 6; ++c) {
            for (int v = 0; v <= 6; ++v) {
                if (c == 6 || v == 6) continue; // terminal states
                
                if (dp[mask][c][v] == 0.0) continue;
                
                // Count available sectors (envelopes still present)
                int available = 0;
                for (int i = 0; i < 13; ++i) {
                    if (mask & (1 << i)) {
                        available++;
                    }
                }
                
                if (available == 0) continue; // shouldn't happen before 6-6
                
                double prob_state = dp[mask][c][v];
                
                // For each possible starting sector (equally likely: 1/13)
                for (int start = 0; start < 13; ++start) {
                    // Find the first sector with envelope in counter-clockwise order from start
                    int selected = -1;
                    for (int offset = 0; offset < 13; ++offset) {
                        int sector = (start + offset) % 13;
                        if (mask & (1 << sector)) {
                            selected = sector;
                            break;
                        }
                    }
                    
                    if (selected == -1) continue;
                    
                    double selection_prob = 1.0 / 13.0;
                    
                    // After selection, update probability
                    double correct_prob = p[selected];
                    
                    // Club wins the point
                    if (selected != 12) {
                        // If it's an envelope from sectors 1-12 (indices 0-11) or empty (12)
                        // but we use p[selected] as probability of club answering correctly
                        
                        // Remove the envelope
                        int new_mask = mask ^ (1 << selected);
                        
                        if (selected == 12) {
                            // Empty envelope: use internet question
                            // We already stored internetAvg in p[12]
                        }
                        
                        // Club gets point
                        if (c + 1 == 6) {
                            // Club wins
                            // Accumulate to final score c+1 : v
                            // We'll collect final scores at the end
                            // For now, we just process the transition
                            // But we need to handle terminal states carefully
                            dp[new_mask][c + 1][v] += prob_state * selection_prob;
                        } else {
                            dp[new_mask][c + 1][v] += prob_state * selection_prob;
                        }
                        
                        // Club fails
                        if (v + 1 == 6) {
                            dp[new_mask][c][v + 1] += prob_state * (1.0 - selection_prob);
                        } else {
                            dp[new_mask][c][v + 1] += prob_state * (1.0 - selection_prob);
                        }
                    } else {
                        // This case is redundant with above since p[12] is already set
                        // Just use the same logic
                        int new_mask = mask ^ (1 << selected);
                        dp[new_mask][c + 1][v] += prob_state * selection_prob * correct_prob;
                        dp[new_mask][c][v + 1] += prob_state * selection_prob * (1.0 - correct_prob);
                    }
                }
            }
        }
    }
    
    // Instead, let's use a better DP approach: process states in order of increasing 
    // number of removed envelopes (i.e., decreasing mask weight)
    // Redo DP more carefully
    
    // Reinitialize
    dp.assign(1 << 13, vector<vector<double>>(7, vector<double>(7, 0.0)));
    dp[(1 << 13) - 1][0][0] = 1.0;
    
    // Process masks from high bit count to low (i.e., fewer available envelopes later)
    for (int available = 13; available >= 1; --available) {
        for (int mask = 0; mask < (1 << 13); ++mask) {
            if (__builtin_popcount(mask) != available) continue;
            
            for (int c = 0; c <= 6; ++c) {
                for (int v = 0; v <= 6; ++v) {
                    if (dp[mask][c][v] == 0.0) continue;
                    if (c == 6 || v == 6) continue; // already terminal
                    
                    double prob_state = dp[mask][c][v];
                    double selection_prob_per_start = 1.0 / 13.0;
                    
                    // For each possible starting sector
                    for (int start = 0; start < 13; ++start) {
                        int selected = -1;
                        for (int offset = 0; offset < 13; ++offset) {
                            int sector = (start + offset) % 13;
                            if (mask & (1 << sector)) {
                                selected = sector;
                                break;
                            }
                        }
                        if (selected == -1) continue;
                        
                        double p_correct = p[selected];
                        
                        int new_mask = mask ^ (1 << selected);
                        
                        // Club answers correctly
                        if (c + 1 == 6) {
                            // Terminal state, accumulate separately
                            // We'll handle terminal states at the end
                        } else {
                            dp[new_mask][c + 1][v] += prob_state * selection_prob_per_start * p_correct;
                        }
                        
                        // Club answers incorrectly
                        if (v + 1 == 6) {
                            // Terminal state
                        } else {
                            dp[new_mask][c][v + 1] += prob_state * selection_prob_per_start * (1.0 - p_correct);
                        }
                    }
                }
            }
        }
    }
    
    // Collect final scores
    vector<double> final_probs(12, 0.0);
    
    // Map final score to index:
    // '6:0', '6:1', '6:2', '6:3', '6:4', '6:5', '5:6', '4:6', '3:6', '2:6', '1:6', '0:6'
    // Club wins: c=6, v=0..5 -> index = v
    // Viewer wins: v=6, c=0..5 -> index = 6 + (5 - c)
    
    for (int mask = 0; mask < (1 << 13); ++mask) {
        for (int c = 0; c <= 6; ++c) {
            for (int v = 0; v <= 6; ++v) {
                if (c == 6 && v <= 5) {
                    final_probs[v] += dp[mask][c][v];
                } else if (v == 6 && c <= 5) {
                    final_probs[6 + (5 - c)] += dp[mask][c][v];
                }
            }
        }
    }
    
    // Also consider paths where the game ends exactly when reaching terminal state
    // Our DP already handles this if we accumulated transitions into terminal states
    // But we didn't store terminal states in dp[][][], so we need to adjust
    
    // Let's restructure DP to store terminal states as well
    // Redo with dp[mask][c][v] for all c,v including 6 (but not both 6)
    
    dp.assign(1 << 13, vector<vector<double>>(7, vector<double>(7, 0.0)));
    dp[(1 << 13) - 1][0][0] = 1.0;
    
    for (int available = 13; available >= 1; --available) {
        for (int mask = 0; mask < (1 << 13); ++mask) {
            if (__builtin_popcount(mask) != available) continue;
            
            for (int c = 0; c <= 6; ++c) {
                for (int v = 0; v <= 6; ++v) {
                    if (c == 6 || v == 6) {
                        // Terminal state, nothing to do
                        continue;
                    }
                    
                    if (dp[mask][c][v] == 0.0) continue;
                    
                    double prob_state = dp[mask][c][v];
                    
                    for (int start = 0; start < 13; ++start) {
                        int selected = -1;
                        for (int offset = 0; offset < 13; ++offset) {
                            int sector = (start + offset) % 13;
                            if (mask & (1 << sector)) {
                                selected = sector;
                                break;
                            }
                        }
                        if (selected == -1) continue;
                        
                        double p_correct = p[selected];
                        int new_mask = mask ^ (1 << selected);
                        
                        // Club correct
                        if (c + 1 == 6) {
                            final_probs[v] += prob_state * (1.0 / 13.0) * p_correct;
                        } else {
                            dp[new_mask][c + 1][v] += prob_state * (1.0 / 13.0) * p_correct;
                        }
                        
                        // Club incorrect
                        if (v + 1 == 6) {
                            final_probs[6 + (5 - c)] += prob_state * (1.0 / 13.0) * (1.0 - p_correct);
                        } else {
                            dp[new_mask][c][v + 1] += prob_state * (1.0 / 13.0) * (1.0 - p_correct);
                        }
                    }
                }
            }
        }
    }
    
    // Set precision and output
    cout << fixed << setprecision(3);
    for (int i = 0; i < 12; ++i) {
        if (i > 0) cout << " ";
        cout << final_probs[i];
    }
    cout << endl;
    
    return 0;
}
```
