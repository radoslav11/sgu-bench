#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

const double EPS = 1e-9;

int main() {
    double c, b, m;
    cin >> c >> b >> m;

    // Check triangle inequality for the median formula
    // For a median AM, where M is midpoint of BC:
    // m = (1/2) * sqrt(2*b^2 + 2*c^2 - a^2), where a = |BC|
    // Rearranged: 4*m^2 = 2*b^2 + 2*c^2 - a^2
    // So we need: 2*b^2 + 2*c^2 - 4*m^2 >= 0 (since a^2 >= 0)
    // Also, for a valid triangle ABC: |b - c| < a < b + c
    // Substituting a^2 = 2*b^2 + 2*c^2 - 4*m^2:
    // |b - c|^2 < 2*b^2 + 2*c^2 - 4*m^2 < (b + c)^2

    double four_m2 = 4 * m * m;
    double two_b2 = 2 * b * b;
    double two_c2 = 2 * c * c;

    // Check if median length is possible
    double a_squared = two_b2 + two_c2 - four_m2;

    // If a^2 < 0, impossible
    if (a_squared < -EPS) {
        cout << "Mission impossible" << endl;
        return 0;
    }

    // Clamp a_squared to 0 if it's very close to zero but negative
    if (a_squared < 0) a_squared = 0;

    double a = sqrt(a_squared);

    // Check triangle inequality: |b - c| < a < b + c
    if (a < fabs(b - c) - EPS || a > b + c + EPS) {
        cout << "Mission impossible" << endl;
        return 0;
    }

    // To simplify, place point A at (0, h) and BC horizontal on x-axis
    // Let B = (-d/2, 0), C = (d/2, 0), so M = midpoint of BC = (0, 0)
    // Then |AB| = c, |AC| = b, |AM| = m (distance from A(0,h) to M(0,0))
    // So |AM| = |h| = m => h = m (or -m, we'll use positive)
    
    // Then distance AB: sqrt((d/2)^2 + h^2) = c
    // Distance AC: sqrt((d/2)^2 + h^2) = b
    
    // Wait, this would imply b = c. But the sample input has b=c=5.
    // For general case, we need a different approach.

    // Alternative: Place A at (0, 0), B at (c, 0)
    // Then point C = (x, y) such that:
    // |AC| = b => x^2 + y^2 = b^2
    // M is midpoint of BC: ((c+x)/2, y/2)
    // |AM| = m => ((c+x)/2)^2 + (y/2)^2 = m^2
    // => (c+x)^2 + y^2 = 4*m^2
    
    // Subtracting: (c+x)^2 + y^2 - (x^2 + y^2) = 4*m^2 - b^2
    // c^2 + 2*c*x = 4*m^2 - b^2
    // => x = (4*m^2 - b^2 - c^2) / (2*c)
    
    // Then y^2 = b^2 - x^2
    double x = (4 * m * m - b * b - c * c) / (2 * c);
    double y_squared = b * b - x * x;

    // Check if y^2 >= 0
    if (y_squared < -EPS) {
        cout << "Mission impossible" << endl;
        return 0;
    }

    if (y_squared < 0) y_squared = 0;
    
    double y = sqrt(y_squared);

    // We can choose y >= 0 (triangle above x-axis)
    // A at (0,0), B at (c, 0), C at (x, y)

    // However, the sample output has A at (0, 3), B at (-4, 0), C at (4, 0)
    // Let's verify: AB = distance from (0,3) to (-4,0) = sqrt(16+9) = 5 = c
    // AC = distance from (0,3) to (4,0) = sqrt(16+9) = 5 = b
    // M = midpoint of BC = ((-4+4)/2, (0+0)/2) = (0, 0)
    // AM = distance from (0,3) to (0,0) = 3 = m
    // So in the sample, y=0 for B and C, A is at (0, m)

    // Let's try a different approach: Place M at origin (0, 0)
    // Then A is at (0, m) or (0, -m); use (0, m)
    // Let B = (-d, 0), C = (d, 0) so M is midpoint: ((-d+d)/2, (0+0)/2) = (0, 0)
    // Then |AB| = sqrt(d^2 + m^2) = c => d^2 + m^2 = c^2 => d = sqrt(c^2 - m^2)
    // |AC| = sqrt(d^2 + m^2) = b => d^2 + m^2 = b^2
    // So this only works when b = c.

    // For general case, place A at (0, 0), M at (m, 0)
    // Then B and C are symmetric with respect to M: if B = (x, y), then C = (2*m - x, -y)
    // |AB| = c => x^2 + y^2 = c^2
    // |AC| = b => (2*m - x)^2 + y^2 = b^2
    // Subtract: (2*m - x)^2 - x^2 = b^2 - c^2
    // 4*m^2 - 4*m*x = b^2 - c^2
    // => x = (4*m^2 + c^2 - b^2) / (4*m)
    // Then y^2 = c^2 - x^2

    double xm = (4 * m * m + c * c - b * b) / (4 * m);
    double y2 = c * c - xm * xm;

    if (y2 < -EPS) {
        cout << "Mission impossible" << endl;
        return 0;
    }

    if (y2 < 0) y2 = 0;
    
    double ym = sqrt(y2);

    // Points:
    // A: (0, 0)
    // M: (m, 0) but we don't need this
    // B: (xm, ym)
    // C: (2*m - xm, -ym)

    double bx = xm;
    double by = ym;
    double cx = 2 * m - xm;
    double cy = -ym;

    // Let's check if |AC| is b:
    // AC^2 = (cx - 0)^2 + (cy - 0)^2 = cx^2 + cy^2
    // = (2*m - xm)^2 + (-ym)^2 = 4*m^2 - 4*m*xm + xm^2 + ym^2
    // = 4*m^2 - 4*m*xm + c^2 (since xm^2 + ym^2 = c^2)
    // From xm = (4*m^2 + c^2 - b^2)/(4*m):
    // 4*m*xm = 4*m^2 + c^2 - b^2
    // So AC^2 = 4*m^2 - (4*m^2 + c^2 - b^2) + c^2 = b^2 ✓

    cout << fixed << setprecision(5);
    cout << "0.00000 0.00000" << endl;
    cout << bx << " " << by << endl;
    cout << cx << " " << cy << endl;

    return 0;
}
