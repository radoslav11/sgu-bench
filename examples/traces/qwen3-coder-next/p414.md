## Problem Understanding

Two circles are orthogonal if at their intersection points, the tangent lines are perpendicular. For two circles with centers (x₁,y₁), (x₂,y₂) and radii r₁, r₂, they are orthogonal if and only if:
```
distance²(center₁, center₂) = r₁² + r₂²
```

We need to find a circle (with center (x,y) and radius r) that is orthogonal to all given n circles.

For the i-th given circle with center (xᵢ, yᵢ) and radius rᵢ, the orthogonality condition is:
```
(x - xᵢ)² + (y - yᵢ)² = r² + rᵢ²
```

Expanding this:
```
x² - 2xxᵢ + xᵢ² + y² - 2yyᵢ + yᵢ² = r² + rᵢ²
```

Rearranging terms:
```
x² + y² - r² = 2xᵢx + 2yᵢy + (rᵢ² - xᵢ² - yᵢ²)
```

Let's define:
- A = x² + y² - r² (this is the same for all circles in the solution)
- Then for each circle i: `2xᵢx + 2yᵢy - A = xᵢ² + yᵢ² - rᵢ²`

Or equivalently:
```
2xᵢx + 2yᵢy - A = xᵢ² + yᵢ² - rᵢ²
```

This is a linear equation in variables x, y, and A.

## Mathematical Approach

We have n linear equations in 3 unknowns (x, y, A):
```
2xᵢ * x + 2yᵢ * y - 1 * A = xᵢ² + yᵢ² - rᵢ², for i = 1..n
```

This is a system of linear equations: M * [x, y, A]^T = b, where M is n×3 matrix.

Since we have 3 unknowns:
- If n < 3: infinitely many solutions (or none if inconsistent) → output -2
- If n = 3: unique solution if equations are independent, else no solution or infinitely many

After finding x, y, A, we compute r = sqrt(x² + y² - A), but we need x² + y² - A ≥ 0.

## Cramer's Rule for 3x3

For system:
```
a1*x + b1*y + c1*z = d1
a2*x + b2*y + c2*z = d2
a3*x + b3*y + c3*z = d3
```

Determinant D = a1*(b2*c3 - b3*c2) - b1*(a2*c3 - a3*c2) + c1*(a2*b3 - a3*b2)

Then Dx = d1*(b2*c3 - b3*c2) - b1*(d2*c3 - d3*c2) + c1*(d2*b3 - d3*b2)
     Dy = a1*(d2*c3 - d3*c2) - d1*(a2*c3 - a3*c2) + c1*(a2*d3 - a3*d2)
     Dz = a1*(b2*d3 - b3*d2) - b1*(a2*d3 - a3*d2) + d1*(a2*b3 - a3*b2)

Solution: x = Dx/D, y = Dy/D, z = Dz/D

In our case: z = A

```cpp
    }
    
    // We'll use the first 3 circles to set up the system
    // Equations: (x - x_i)^2 + (y - y_i)^2 - r_i^2 = r^2 for all i
    // Which gives: x^2 - 2x_i*x + x_i^2 + y^2 - 2y_i*y + y_i^2 - r_i^2 = r^2
    // Rearranged: -2x_i*x - 2y_i*y + (x^2 + y^2 - r^2) = -x_i^2 - y_i^2 + r_i^2
    // Let A = x^2 + y^2 - r^2, then: 2x_i*x + 2y_i*y - A = x_i^2 + y_i^2 - r_i^2
    
    // Build equations for first 3 circles
    vector<vector<double>> eqs(3, vector<double>(4));
    
    for (int i = 0; i < 3; i++) {
        double x_i = circles[i].first;
        double y_i = circles[i].second.first;
        double r_i = circles[i].second.second;
        
        eqs[i][0] = 2.0 * x_i;      // coefficient of x
        eqs[i][1] = 2.0 * y_i;      // coefficient of y
        eqs[i][2] = -1.0;           // coefficient of A
        eqs[i][3] = x_i * x_i + y_i * y_i - r_i * r_i; // right-hand side
    }
    
    // Try to find 3 equations with non-zero determinant
    // If first 3 have zero determinant, try other triplets
    int idx1 = 0, idx2 = 1, idx3 = 2;
    for (int i = 2; i < n; i++) {
        if (i >= 3) {
            idx3 = i;
            eqs[0] = vector<double>(4);
            eqs[1] = vector<double>(4);
            eqs[2] = vector<double>(4);
            
            for (int j = 0; j < 3; j++) {
                int idx = (j == 0 ? 0 : (j == 1 ? 1 : idx3));
                double x_i = circles[idx].first;
                double y_i = circles[idx].second.first;
                double r_i = circles[idx].second.second;
                
                eqs[j][0] = 2.0 * x_i;
                eqs[j][1] = 2.0 * y_i;
                eqs[j][2] = -1.0;
                eqs[j][3] = x_i * x_i + y_i * y_i - r_i * r_i;
            }
        }
        
        // Compute determinant
        double det = eqs[0][0] * (eqs[1][1] * eqs[2][2] - eqs[1][2] * eqs[2][1])
                   - eqs[0][1] * (eqs[1][0] * eqs[2][2] - eqs[1][2] * eqs[2][0])
                   + eqs[0][2] * (eqs[1][0] * eqs[2][1] - eqs[1][1] * eqs[2][0]);
        
        if (fabs(det) > EPS) {
            // Solve using Cramer's rule
            vector<double> rhs = {eqs[0][3], eqs[1][3], eqs[2][3]};
            
            // Dx
            vector<vector<double>> matDx = {
                {rhs[0], eqs[0][1], eqs[0][2]},
                {rhs[1], eqs[1][1], eqs[1][2]},
                {rhs[2], eqs[2][1], eqs[2][2]}
            };
            double D_x = matDx[0][0] * (matDx[1][1] * matDx[2][2] - matDx[1][2] * matDx[2][1])
                       - matDx[0][1] * (matDx[1][0] * matDx[2][2] - matDx[1][2] * matDx[2][0])
                       + matDx[0][2] * (matDx[1][0] * matDx[2][1] - matDx[1][1] * matDx[2][0]);
            
            // Dy
            vector<vector<double>> matDy = {
                {eqs[0][0], rhs[0], eqs[0][2]},
                {eqs[1][0], rhs[1], eqs[1][2]},
                {eqs[2][0], rhs[2], eqs[2][2]}
            };
            double D_y = matDy[0][0] * (matDy[1][1] * matDy[2][2] - matDy[1][2] * matDy[2][1])
                       - matDy[0][1] * (matDy[1][0] * matDy[2][2] - matDy[1][2] * matDy[2][0])
                       + matDy[0][2] * (matDy[1][0] * matDy[2][1] - matDy[1][1] * matDy[2][0]);
            
            // DA
            vector<vector<double>> matDA = {
                {eqs[0][0], eqs[0][1], rhs[0]},
                {eqs[1][0], eqs[1][1], rhs[1]},
                {eqs[2][0], eqs[2][1], rhs[2]}
            };
            double D_A = matDA[0][0] * (matDA[1][1] * matDA[2][2] - matDA[1][2] * matDA[2][1])
                       - matDA[0][1] * (matDA[1][0] * matDA[2][2] - matDA[1][2] * matDA[2][0])
                       + matDA[0][2] * (matDA[1][0] * matDA[2][1] - matDA[1][1] * matDA[2][0]);
            
            double x_sol = D_x / det;
            double y_sol = D_y / det;
            double A_sol = D_A / det;
            
            // Verify against all circles
            bool valid = true;
            for (int j = 0; j < n; j++) {
                double x_i = circles[j].first;
                double y_i = circles[j].second.first;
                double r_i = circles[j].second.second;
                
                double residual = 2.0 * x_i * x_sol + 2.0 * y_i * y_sol - A_sol - (x_i * x_i + y_i * y_i - r_i * r_i);
                if (fabs(residual) > 1e-6) {
                    valid = false;
                    break;
                }
            }
            
            if (valid) {
                double r_squared = x_sol * x_sol + y_sol * y_sol - A_sol;
                if (r_squared < -1e-8) {
                    cout << -1 << "\n";
                } else {
                    double r = sqrt(max(0.0, r_squared));
                    cout << fixed << setprecision(10);
                    cout << x_sol << " " << y_sol << " " << r << "\n";
                }
                return 0;
            }
        }
    }
    
    // If we reach here, no unique solution from any 3 circles
    // Check if there are infinitely many solutions or no solution
    // This happens when all circles are concentric or form a special configuration
    
    // Try to see if all circles have the same center (concentric)
    bool all_concentric = true;
    for (int i = 1; i < n; i++) {
        if (circles[i].first != circles[0].first || circles[i].second.first != circles[0].second.first) {
            all_concentric = false;
            break;
        }
    }
    
    if (all_concentric) {
        // For concentric circles, any circle with center at the same point has orthogonal condition:
        // 0 = r^2 + r_i^2 => r^2 = -r_i^2, impossible unless r_i = 0 (but r_i >= 1)
        // So no solution
        cout << -1 << "\n";
        return 0;
    }
    
    // Otherwise, likely infinitely many solutions or no solution
    // Check consistency using rank approach
    vector<vector<double>> mat;
    for (int i = 0; i < n; i++) {
        double x_i = circles[i].first;
        double y_i = circles[i].second.first;
        double r_i = circles[i].second.second;
        
        vector<double> row = {2.0 * x_i, 2.0 * y_i, -1.0, x_i * x_i + y_i * y_i - r_i * r_i};
        mat.push_back(row);
    }
    
    // Gaussian elimination to check rank and consistency
    int rows = n;
    int cols = 4;
    int rank = 0;
    
    for (int col = 0; col < 3 && rank < rows; col++) {
        int pivot = rank;
        for (int i = rank + 1; i < rows; i++) {
            if (fabs(mat[i][col]) > fabs(mat[pivot][col])) {
                pivot = i;
            }
        }
        
        if (fabs(mat[pivot][col]) < EPS) continue;
        
        swap(mat[rank], mat[pivot]);
        
        double div = mat[rank][col];
        for (int j = col; j < cols; j++) {
            mat[rank][j] /= div;
        }
        
        for (int i = 0; i < rows; i++) {
            if (i != rank && fabs(mat[i][col]) > EPS) {
                double factor = mat[i][col];
                for (int j = col; j < cols; j++) {
                    mat[i][j] -= factor * mat[rank][j];
                }
            }
        }
        
        rank++;
    }
    
    // Check consistency
    bool consistent = true;
    for (int i = 0; i < rows; i++) {
        if (fabs(mat[i][0]) < EPS && fabs(mat[i][1]) < EPS && fabs(mat[i][2]) < EPS) {
            if (fabs(mat[i][3]) > EPS) {
                consistent = false;
                break;
            }
        }
    }
    
    if (!consistent) {
        cout << -1 << "\n";
    } else if (rank < 3) {
        cout << -2 << "\n";
    } else {
        // This shouldn't happen, but output -1 as fallback
        cout << -1 << "\n";
    }
    
    return 0;
}
```

Actually, the O(n³) approach won't work for n=10^5. Let me think differently.

For 3 circles, there's typically a unique solution. For n > 3, we just need to verify the solution from the first 3 circles against all others. And for the first 3 circles, if they are not collinear, we get a unique solution.

The probability that the first 3 centers are collinear is very low for random inputs, and the problem constraints say coordinates are integers in [-10^6, 10^6].

So in practice, we can:
1. Take first 3 circles
2. If they are collinear (determinant near zero), try circles 0,1,4, then 0,1,5, etc. but limit to at most 100 attempts
3. If still no non-collinear triplet among first 100 circles, then likely all centers are collinear

For collinear centers, the system has rank at most 2, so either no solution or infinitely many.

Let me implement this more efficiently:

```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
#include <algorithm>

using namespace std;

const double EPS = 1e-9;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n;
    cin >> n;
    
    vector<tuple<long long, long long, long long>> circles;
    for (int i = 0; i < n; i++) {
        long long x, y, r;
        cin >> x >> y >> r;
        circles.emplace_back(x, y, r);
    }
    
    if (n <= 2) {
        cout << -2 << "\n";
        return 0;
    }
    
    // Try to find 3 non-collinear centers
    int idx1 = 0, idx2 = 1, idx3 = 2;
    bool found_triplet = false;
    
    for (int i = 2; i < n && !found_triplet; i++) {
        idx3 = i;
        long long x1 = get<0>(circles[0]), y1 = get<1>(circles[0]);
        long long x2 = get<0>(circles[1]), y2 = get<1>(circles[1]);
        long long x3 = get<0>(circles[idx3]), y3 = get<1>(circles[idx3]);
        
        // Check if centers are collinear: (x2-x1)*(y3-y1) - (x3-x1)*(y2-y1) != 0
        long long area2 = (x2 - x1) * (y3 - y1) - (x3 - x1) * (y2 - y1);
        if (area2 != 0) {
            found_triplet = true;
        }
    }
    
    // If no non-collinear triplet found in first n circles, all centers collinear
    if (!found_triplet) {
        // Check for collinear centers case
        // All centers lie on a line. The system has rank <= 2.
        // Check if there's a solution by verifying consistency
        double x1 = get<0>(circles[0]), y1 = get<1>(circles[0]);
        double r1 = get<2>(circles[0]);
        
        // Try to find a potential solution
        // For circles on a line, we can project the problem to 1D
        // But simpler: check if all equations are consistent
        
        // Equation for circle i: 2x_i*x + 2y_i*y - A = x_i^2 + y_i^2 - r_i^2
        
        // Since centers are collinear, we can rotate coordinates to make them collinear on x-axis
        // For simplicity, let's just check consistency using Gaussian elimination on first 3
        
        vector<vector<double>> mat(3, vector<double>(4));
        for (int i = 0; i < 3; i++) {
            double xi = get<0>(circles[i]), yi = get<1>(circles[i]), ri = get<2>(circles[i]);
            mat[i][0] = 2.0 * xi;
            mat[i][1] = 2.0 * yi;
            mat[i][2] = -1.0;
            mat[i][3] = xi * xi + yi * yi - ri * ri;
        }
        
        // Gaussian elimination to check rank
        int rank = 0;
        for (int col = 0; col < 3 && rank < 3; col++) {
            int pivot = rank;
            for (int i = rank + 1; i < 3; i++) {
                if (fabs(mat[i][col]) > fabs(mat[pivot][col])) pivot = i;
            }
            if (fabs(mat[pivot][col]) < EPS) continue;
            
            swap(mat[rank], mat[pivot]);
            double div = mat[rank][col];
            for (int j = col; j < 4; j++) mat[rank][j] /= div;
            
            for (int i = 0; i < 3; i++) {
                if (i != rank && fabs(mat[i][col]) > EPS) {
                    double factor = mat[i][col];
                    for (int j = col; j < 4; j++) mat[i][j] -= factor * mat[rank][j];
                }
            }
            rank++;
        }
        
        bool consistent = true;
        for (int i = 0; i < 3; i++) {
            if (fabs(mat[i][0]) < EPS && fabs(mat[i][1]) < EPS && fabs(mat[i][2]) < EPS) {
                if (fabs(mat[i][3]) > EPS) {
                    consistent = false;
                    break;
                }
            }
        }
        
        if (!consistent) {
            cout << -1 << "\n";
            return 0;
        }
        
        // Since rank < 3 (collinear centers), infinitely many solutions
        cout << -2 << "\n";
        return 0;
    }
    
    // Use the first 3 non-collinear circles
    vector<vector<double>> eqs(3, vector<double>(4));
    vector<int> indices = {0, 1, idx3};
    
    for (int i = 0; i < 3; i++) {
        int idx = indices[i];
        double xi = get<0>(circles[idx]), yi = get<1>(circles[idx]), ri = get<2>(circles[idx]);
        eqs[i][0] = 2.0 * xi;
        eqs[i][1] = 2.0 * yi;
        eqs[i][2] = -1.0;
        eqs[i][3] = xi * xi + yi * yi - ri * ri;
    }
    
    // Compute determinant
    double det = eqs[0][0] * (eqs[1][1] * eqs[2][2] - eqs[1][2] * eqs[2][1])
               - eqs[0][1] * (eqs[1][0] * eqs[2][2] - eqs[1][2] * eqs[2][0])
               + eqs[0][2] * (eqs[1][0] * eqs[2][1] - eqs[1][1] * eqs[2][0]);
    
    if (fabs(det) < EPS) {
        cout << -2 << "\n";
        return 0;
    }
    
    // Cramer's rule
    double dx = eqs[0][3] * (eqs[1][1] * eqs[2][2] - eqs[1][2] * eqs[2][1])
              - eqs[0][1] * (eqs[1][3] * eqs[2][2] - eqs[1][2] * eqs[2][3])
              + eqs[0][2] * (eqs[1][3] * eqs[2][1] - eqs[1][1] * eqs[2][3]);
    
    double dy = eqs[0][0] * (eqs[1][3] * eqs[2][2] - eqs[1][2] * eqs[2][3])
              - eqs[0][3] * (eqs[1][0] * eqs[2][2] - eqs[1][2] * eqs[2][0])
              + eqs[0][2] * (eqs[1][0] * eqs[2][3] - eqs[1][3] * eqs[2][0]);
    
    double dA = eqs[0][0] * (eqs[1][1] * eqs[2][3] - eqs[1][3] * eqs[2][1])
              - eqs[0][1] * (eqs[1][0] * eqs[2][3] - eqs[1][3] * eqs[2][0])
              + eqs[0][3] * (eqs[1][0] * eqs[2][1] - eqs[1][1] * eqs[2][0]);
    
    double x = dx / det;
    double y = dy / det;
    double A = dA / det;
    
    // Verify against all circles
    bool valid = true;
    for (int i = 0; i < n; i++) {
        double xi = get<0>(circles[i]), yi = get<1>(circles[i]), ri = get<2>(circles[i]);
        double residual = 2.0 * xi * x + 2.0 * yi * y - A - (xi * xi + yi * yi - ri * ri);
        if (fabs(residual) > 1e-6) {
            valid = false;
            break;
        }
    }
    
    if (!valid) {
        cout << -1 << "\n";
        return 0;
    }
    
    double r_squared = x * x + y * y - A;
    if (r_squared < -1e-8) {
        cout << -1 << "\n";
        return 0;
    }
    
    double r = sqrt(max(0.0, r_squared));
    
    cout << fixed << setprecision(10);
    cout << x << " " << y << " " << r << "\n";
    
    return 0;
}
```
