We can store the number as a vector of digits in reverse order (for easier carry).

Algorithm for multiplication:
  result = [1] (as a vector of digits, in reverse: [1])
  for i from 1 to k:
      carry = 0
      for each digit in result:
          temp = digit * base + carry
          digit = temp % 10
          carry = temp / 10
      while carry:
          result.push_back(carry % 10)
          carry /= 10

Code structure:

#include <iostream>
#include <vector>
#include <cctype>
#include <algorithm>
#include <cmath>
#include <string>
using namespace std;

int main() {
    int a, b, k;
    cin >> a >> b >> k;
    int base = a + b;

    // If base is 1, then result is 1. But our method works for base=1 too.
    vector<int> digits;
    digits.push_back(1);

    for (int i = 0; i < k; i++) {
        int carry = 0;
        for (int j = 0; j < digits.size(); j++) {
            int temp = digits[j] * base + carry;
            digits[j] = temp % 10;
            carry = temp / 10;
        }
        while (carry) {
            digits.push_back(carry % 10);
            carry /= 10;
        }
    }

    // Output: from the last digit to the first
    for (int i = digits.size() - 1; i >= 0; i--) {
        cout << digits[i];
    }
    cout << endl;

    return 0;
}
