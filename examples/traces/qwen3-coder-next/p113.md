#include <iostream>
#include <vector>
#include <cmath>
#include <cctype>

using namespace std;

bool isPrime(long long n) {
    if (n <= 1) return false;
    if (n <= 3) return true;
    if (n % 2 == 0 || n % 3 == 0) return false;
    
    for (long long i = 5; i * i <= n; i += 6) {
        if (n % i == 0 || n % (i + 2) == 0)
            return false;
    }
    return true;
}

bool isNearlyPrime(long long n) {
    if (n <= 1) return false;
    
    // Try all possible prime factors up to sqrt(n)
    for (long long i = 2; i * i <= n; ++i) {
        if (n % i == 0) {
            long long factor1 = i;
            long long factor2 = n / i;
            
            // Check if both factors are prime
            if (isPrime(factor1) && isPrime(factor2)) {
                return true;
            }
        }
    }
    
    // Special case: if n is a square of a prime, then it's not nearly prime
    // because we need two primes P1 and P2 (not necessarily distinct)
    // But note: the problem says "P1 and P2" without requiring distinctness.
    // Actually, if n = p * p where p is prime, then it's nearly prime.
    // Our loop above would have caught this case when i = p.
    
    return false;
}

int main() {
    int n;
    cin >> n;
    
    for (int i = 0; i < n; ++i) {
        long long num;
        cin >> num;
        
        if (isNearlyPrime(num)) {
            cout << "Yes" << endl;
        } else {
            cout << "No" << endl;
        }
    }
    
    return 0;
}
