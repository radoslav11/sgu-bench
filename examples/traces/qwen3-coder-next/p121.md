#include <iostream>
#include <vector>
#include <cstring>
using namespace std;

const int MAXN = 105;

int n;
vector<int> adj[MAXN];
int color[MAXN][MAXN]; // color[i][j] = color of bridge between i and j (1 or 2), 0 if not set

// Check if current coloring is valid
bool isValid() {
    for (int i = 1; i <= n; i++) {
        if (adj[i].size() < 2) continue; // No constraint if degree < 2
        bool hasWhite = false, hasBlack = false;
        for (int j : adj[i]) {
            if (color[i][j] == 1) hasWhite = true;
            if (color[i][j] == 2) hasBlack = true;
        }
        if (!hasWhite || !hasBlack) return false;
    }
    return true;
}

// Backtracking function to assign colors to bridges
bool backtrack(int idx, int u, int v) {
    // idx represents the bridge index we are processing (to avoid duplicate processing)
    // We iterate over all edges (u, v) where u < v to avoid duplicates
    
    // First, find the next edge to process
    while (u <= n) {
        while (v <= n) {
            // Check if there is an edge between u and v
            bool found = false;
            for (int neighbor : adj[u]) {
                if (neighbor == v) {
                    found = true;
                    break;
                }
            }
            if (found) {
                // Edge exists
                if (color[u][v] == 0) {
                    // Try color 1 first
                    color[u][v] = 1;
                    color[v][u] = 1;
                    if (backtrack(idx + 1, u, v + 1)) return true;
                    
                    // Try color 2
                    color[u][v] = 2;
                    color[v][u] = 2;
                    if (backtrack(idx + 1, u, v + 1)) return true;
                    
                    // Backtrack
                    color[u][v] = 0;
                    color[v][u] = 0;
                    return false;
                }
            }
            v++;
        }
        u++;
        v = u + 1;
    }
    
    // All edges processed, check validity
    return isValid();
}

int main() {
    cin >> n;
    
    for (int i = 1; i <= n; i++) {
        int x;
        while (cin >> x && x != 0) {
            adj[i].push_back(x);
        }
    }
    
    // Initialize color array to 0
    memset(color, 0, sizeof(color));
    
    // Try to find a valid coloring using backtracking
    bool found = backtrack(0, 1, 2);
    
    if (!found) {
        cout << "No solution" << endl;
        return 0;
    }
    
    // Output the result
    for (int i = 1; i <= n; i++) {
        for (int j = 0; j < adj[i].size(); j++) {
            int neighbor = adj[i][j];
            if (i < neighbor) {
                // This bridge will be output when we process the smaller index
                continue;
            }
            cout << color[neighbor][i];
            if (j < adj[i].size() - 1) {
                cout << " ";
            }
        }
        
        // Wait, this approach has an issue: we need to output the colors in the order
        // of the input for each island
        // Let's rebuild the output properly
    }
    
    // Better approach: output according to the input order
    // First, let's store the input order explicitly
    // Actually, we can re-read or just use the adj list structure
    
    // Since we processed edges in backtracking with i < j, but output order should follow input order
    // Let's store the order of neighbors
    vector<vector<int>> order(n + 1);
    for (int i = 1; i <= n; i++) {
        for (int neighbor : adj[i]) {
            if (i < neighbor) {
                order[i].push_back(neighbor);
            }
        }
    }
    
    // But the problem: we need to output colors in the order of the input list for each island
    // So let's use the original adj list structure and figure out the color based on the edge
    
    // Since color[i][j] = color[j][i], we can output color[i][neighbor] for each neighbor
    for (int i = 1; i <= n; i++) {
        for (int j = 0; j < adj[i].size(); j++) {
            int neighbor = adj[i][j];
            cout << color[i][neighbor];
            if (j < adj[i].size() - 1) {
                cout << " ";
            }
        }
        cout << " 0" << endl;
    }
    
    return 0;
}
