#include <iostream>
#include <string>
#include <algorithm>
#include <vector>

using namespace std;

// Compare two numbers represented as strings
int compare(const string& a, const string& b) {
    if (a.length() != b.length()) return a.length() < b.length() ? -1 : 1;
    for (int i = 0; i < a.length(); i++) {
        if (a[i] != b[i]) return a[i] < b[i] ? -1 : 1;
    }
    return 0;
}

// Add two numbers represented as strings
string add(const string& a, const string& b) {
    string result;
    int carry = 0;
    int i = a.length() - 1, j = b.length() - 1;
    
    while (i >= 0 || j >= 0 || carry) {
        int sum = carry;
        if (i >= 0) sum += a[i--] - '0';
        if (j >= 0) sum += b[j--] - '0';
        result.push_back((sum % 10) + '0');
        carry = sum / 10;
    }
    
    reverse(result.begin(), result.end());
    return result;
}

// Subtract two numbers represented as strings (a - b), assumes a >= b
string subtract(const string& a, const string& b) {
    string result;
    int borrow = 0;
    int i = a.length() - 1, j = b.length() - 1;
    
    while (i >= 0) {
        int digit_a = a[i] - '0';
        int digit_b = (j >= 0) ? b[j] - '0' : 0;
        int diff = digit_a - digit_b - borrow;
        
        if (diff < 0) {
            diff += 10;
            borrow = 1;
        } else {
            borrow = 0;
        }
        
        result.push_back(diff + '0');
        i--;
        j--;
    }
    
    // Remove trailing zeros
    while (result.length() > 1 && result.back() == '0') {
        result.pop_back();
    }
    
    reverse(result.begin(), result.end());
    return result;
}

// Multiply two numbers represented as strings
string multiply(const string& a, const string& b) {
    if (a == "0" || b == "0") return "0";
    
    vector<int> result(a.length() + b.length(), 0);
    
    for (int i = a.length() - 1; i >= 0; i--) {
        for (int j = b.length() - 1; j >= 0; j--) {
            int mul = (a[i] - '0') * (b[j] - '0');
            int pos1 = i + j, pos2 = i + j + 1;
            int sum = mul + result[pos2];
            
            result[pos2] = sum % 10;
            result[pos1] += sum / 10;
        }
    }
    
    // Convert to string
    string res_str = "";
    int i = 0;
    while (i < result.size() && result[i] == 0) i++;
    
    if (i == result.size()) return "0";
    
    for (; i < result.size(); i++) {
        res_str += (result[i] + '0');
    }
    
    return res_str;
}

// Integer square root using binary search
string integerSquareRoot(const string& x) {
    if (x == "0") return "0";
    if (x == "1") return "1";
    
    string low = "1";
    string high = x;
    
    // Optimization: start with a better upper bound
    if (x.length() > 1) {
        high = "";
        int half_len = (x.length() + 1) / 2;
        high += '1';
        for (int i = 0; i < half_len; i++) high += '0';
    }
    
    while (compare(low, high) <= 0) {
        string mid = add(low, high);
        // Divide by 2
        int i = 0;
        while (i < mid.length() && mid[i] == '0') i++;
        if (i == mid.length()) {
            mid = "0";
        } else {
            string temp = "";
            int carry = 0;
            for (int j = i; j < mid.length(); j++) {
                int digit = (mid[j] - '0') + carry * 10;
                temp += (digit / 2) + '0';
                carry = digit % 2;
            }
            mid = temp;
        }
        
        string sq = multiply(mid, mid);
        
        if (compare(sq, x) <= 0 && compare(x, sq) < 1) {
            string next = add(mid, "1");
            string next_sq = multiply(next, next);
            if (compare(next_sq, x) > 0) {
                return mid;
            } else {
                low = add(mid, "1");
            }
        } else if (compare(sq, x) > 0) {
            high = subtract(mid, "1");
        } else {
            low = add(mid, "1");
        }
    }
    
    return high;
}

int main() {
    string X;
    cin >> X;
    cout << integerSquareRoot(X) << endl;
    return 0;
}
