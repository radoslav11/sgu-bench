I need to solve this competitive programming problem.

Let me understand the problem:
- There are K guests and N types of tea
- Each round, all current guests get one bag of the same tea type
- After each round, one guest leaves (so round 1 has K guests, round 2 has K-1, ..., round K has 1 guest)
- Each tea type can be used at most once (all served tea types must be different)
- Any three consecutive rounds cannot use tea of the same kind (so no three consecutive rounds with all green or all black)
- Goal: minimize total cost

The output should be K different tea type indices (1-indexed) representing the tea used in each round.

Observations:
- We need exactly K rounds (since we start with K guests and one leaves after each round)
- Each round uses a different tea type, so we need at least K tea types available
- The constraint about three consecutive rounds not being the same kind means we cannot have three green or three black teas in a row

Since we need K different tea types and K rounds, we need to select K tea types from N available ones.

The key insight is to minimize cost while satisfying the "no three consecutive same kind" constraint.

Let me think about the constraint: we can have at most 2 consecutive teas of the same kind. This is similar to arranging items where we can have runs of at most 2 of the same type.

A greedy approach might work:
1. Sort tea types by cost
2. Try to select K cheapest teas that can be arranged to avoid 3 consecutive same kind

But the arrangement matters - we need to output the order (which tea in which round).

Actually, we need to choose K tea types AND decide their order of use.

Constraints on arrangement:
- If we have g green teas and b black teas (g + b = K), we need to arrange them so no 3 consecutive same kind.
- This is possible if and only if: max(g, b) ≤ 2 * min(g, b) + 2
  (because we can interleave: e.g., if more green, arrange as GG B GG B GG ...)
  Actually, the precise condition is: if max(g, b) ≤ 2 * min(g, b) + 2, then it's possible.
  For example:
  - If g=3, b=0: max=3, min=0, 3 > 2*0+2=2 → impossible
  - If g=3, b=1: max=3, min=1, 3 ≤ 2*1+2=4 → possible (GG B G or G GG B, etc.)
  - If g=3, b=2: max=3, min=2, 3 ≤ 2*2+2=6 → possible (GG B GG B or G B GG B G, etc.)

So the problem reduces to:
1. Check if there exists a selection of K teas such that if g = number of green teas selected, b = number of black teas selected (g+b=K), then max(g,b) ≤ 2*min(g,b)+2
2. Among all valid selections, find one with minimum total cost

#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

struct Tea {
    int cost;
    int type;
    int idx;
};

int main() {
    int K, N;
    cin >> K >> N;
    
    vector<Tea> green, black;
    
    for (int i = 0; i < N; i++) {
        int c, s;
        cin >> c >> s;
        Tea tea = {c, s, i + 1};
        if (s == 0) {
            green.push_back(tea);
        } else {
            black.push_back(tea);
        }
    }
    
    sort(green.begin(), green.end(), [](const Tea& a, const Tea& b) {
        return a.cost < b.cost;
    });
    
    sort(black.begin(), black.end(), [](const Tea& a, const Tea& b) {
        return a.cost < b.cost;
    });
    
    // Precompute prefix sums
    vector<long long> green_prefix(green.size() + 1, 0);
    for (int i = 0; i < green.size(); i++) {
        green_prefix[i + 1] = green_prefix[i] + green[i].cost;
    }
    
    vector<long long> black_prefix(black.size() + 1, 0);
    for (int i = 0; i < black.size(); i++) {
        black_prefix[i + 1] = black_prefix[i] + black[i].cost;
    }
    
    long long min_cost = LLONG_MAX;
    int best_g = -1, best_b = -1;
    
    // Try all possible g (number of green teas to use)
    for (int g = 0; g <= min(K, (int)green.size()); g++) {
        int b = K - g;
        if (b < 0 || b > (int)black.size()) continue;
        
        // Check constraint: no three consecutive same kind
        if (max(g, b) <= 2 * min(g, b) + 2) {
            long long cost = green_prefix[g] + black_prefix[b];
            if (cost < min_cost) {
                min_cost = cost;
                best_g = g;
                best_b = b;
            }
        }
    }
    
    if (best_g == -1) {
        cout << "Impossible" << endl;
        return 0;
    }
    
    // Extract the selected teas
    vector<Tea> selected_green(green.begin(), green.begin() + best_g);
    vector<Tea> selected_black(black.begin(), black.begin() + best_b);
    
    // Arrange them to avoid 3 consecutive same kind
    vector<Tea> result;
    int g_idx = 0, b_idx = 0;
    
    while (result.size() < K) {
        // If last two are same type, we must choose the other type
        if (result.size() >= 2 && 
            result[result.size() - 1].type == result[result.size() - 2].type) {
            int last_type = result[result.size() - 1].type;
            if (last_type == 0) { // last is green, need black
                if (b_idx < (int)selected_black.size()) {
                    result.push_back(selected_black[b_idx++]);
                } else {
                    // This should not happen if constraint is satisfied
                    result.push_back(selected_green[g_idx++]);
                }
            } else { // last is black, need green
                if (g_idx < (int)selected_green.size()) {
                    result.push_back(selected_green[g_idx++]);
                } else {
                    // This should not happen if constraint is satisfied
                    result.push_back(selected_black[b_idx++]);
                }
            }
        } else {
            // Choose the type with more remaining, or if equal, prefer any
            if (g_idx < (int)selected_green.size() && 
                (b_idx >= (int)selected_black.size() || 
                 (int)selected_green.size() - g_idx >= (int)selected_black.size() - b_idx)) {
                result.push_back(selected_green[g_idx++]);
            } else if (b_idx < (int)selected_black.size()) {
                result.push_back(selected_black[b_idx++]);
            } else {
                // This should not happen since g + b = K
                break;
            }
        }
    }
    
    // Output the indices
    for (int i = 0; i < K; i++) {
        if (i > 0) cout << " ";
        cout << result[i].idx;
    }
    cout << endl;
    
    return 0;
}
