#include <bits/stdc++.h>
using namespace std;

int v, n;
vector<vector<bool>> adj;
vector<int> assignment;
vector<int> count_clauses;

bool backtrack(int idx) {
    if (idx > v) {
        // Check cross-clause condition: for every pair of clauses, for each vertex in the second clause, 
        // the number of non-neighbors in the first clause is at most 1.
        for (int c1 = 0; c1 < n; c1++) {
            vector<int> clause1;
            for (int i = 1; i <= v; i++) {
                if (assignment[i] == c1) {
                    clause1.push_back(i);
                }
            }
            for (int c2 = c1 + 1; c2 < n; c2++) {
                vector<int> clause2;
                for (int i = 1; i <= v; i++) {
                    if (assignment[i] == c2) {
                        clause2.push_back(i);
                    }
                }
                for (int x : clause2) {
                    int cnt = 0;
                    for (int y : clause1) {
                        if (!adj[x][y]) {
                            cnt++;
                        }
                    }
                    if (cnt > 1) {
                        return false;
                    }
                }
            }
        }
        return true;
    }

    int next_clause = 0;
    while (next_clause < n && count_clauses[next_clause] > 0) {
        next_clause++;
    }

    for (int c = 0; c <= next_clause && c < n; c++) {
        if (count_clauses[c] < 3) {
            assignment[idx] = c;
            count_clauses[c]++;
            // If this clause is now full (size 3), check it's an independent set
            if (count_clauses[c] == 3) {
                vector<int> tri;
                for (int i = 1; i <= v; i++) {
                    if (assignment[i] == c) {
                        tri.push_back(i);
                    }
                }
                // Check no edges among the three
                if (adj[tri[0]][tri[1]] || adj[tri[0]][tri[2]] || adj[tri[1]][tri[2]]) {
                    count_clauses[c]--;
                    continue;
                }
            }
            if (backtrack(idx + 1)) {
                return true;
            }
            count_clauses[c]--;
        }
    }
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int e;
    cin >> v >> e;

    // Check divisibility by 3
    if (v % 3 != 0) {
        cout << "NO\n";
        return 0;
    }

    n = v / 3;

    // Initialize adjacency matrix (1-indexed)
    adj.assign(v + 1, vector<bool>(v + 1, false));
    assignment.assign(v + 1, -1);
    count_clauses.assign(n, 0);

    // Read edges
    for (int i = 0; i < e; i++) {
        int a, b;
        cin >> a >> b;
        adj[a][b] = true;
        adj[b][a] = true;
    }

    // If there are no edges, but v=3, then it's valid (one clause, no edges within clause)
    // Run backtracking
    if (backtrack(1)) {
        cout << "YES\n";
    } else {
        cout << "NO\n";
    }

    return 0;
}
