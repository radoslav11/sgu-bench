#include <iostream>
#include <string>
#include <stack>
#include <cctype>

int main() {
    std::string s;
    std::getline(std::cin, s);
    
    std::string result;
    std::stack<char> tagStack;
    size_t i = 0;
    
    while (i < s.length()) {
        if (s[i] == '<') {
            // Parse tag
            size_t j = s.find('>', i);
            if (j == std::string::npos) {
                break;
            }
            
            std::string tag = s.substr(i + 1, j - i - 1);
            
            // Check if it's a closing tag
            if (tag[0] == '/') {
                tagStack.pop();
            } else {
                // Opening tag
                if (tag == "UP") {
                    tagStack.push('U');
                } else if (tag == "DOWN") {
                    tagStack.push('D');
                }
            }
            
            i = j + 1;
        } else {
            // Regular character
            char c = s[i];
            
            if (!tagStack.empty()) {
                if (tagStack.top() == 'U') {
                    result += std::toupper(c);
                } else { // 'D'
                    result += std::tolower(c);
                }
            } else {
                result += c;
            }
            
            i++;
        }
    }
    
    std::cout << result << std::endl;
    
    return 0;
}
