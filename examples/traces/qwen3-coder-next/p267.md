The key insight is that the optimist and pessimist will select K pies such that the total geometric area is maximized. However, not all pies can be cut as required (each pie must be split into two equal areas with each piece containing one candle without the cut passing through any candle). For usable pies, the optimist and pessimist each get half the pie's area. The solution involves:

1. **Reading and Processing Pies**: For each pie, compute its geometric area and determine if it can be cut as required (usable).
2. **Sorting by Area**: Sort the pies in descending order of their geometric areas.
3. **Finding Maximum Area Subsets**: Identify the maximum possible total geometric area (M) for any K pies. This is the sum of the top K areas.
4. **Handling Ties in Area**: Pies with areas equal to the K-th largest area (r) form a group. The number of pies to select from this group (L) is K minus the number of pies with areas greater than r (m).
5. **Calculating Min and Max Usable Areas**: For the group of pies with area r, compute the minimum and maximum possible sum of usable areas when selecting L pies. The fixed part (from pies with area > r) is added to these sums.
6. **Output Results**: The pessimist's area is the minimum usable sum divided by 2, and the optimist's area is the maximum usable sum divided by 2, both formatted to three decimal places.

### Solution Code
```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <iomanip>
using namespace std;

struct Pie {
    int area;
    int usable_area;
};

int main() {
    int N, K;
    cin >> N >> K;

    vector<Pie> pies;
    for (int i = 0; i < N; i++) {
        vector<long long> coord(12);
        for (int j = 0; j < 12; j++) {
            cin >> coord[j];
        }

        long long x1 = coord[0], y1 = coord[1];
        long long x2 = coord[2], y2 = coord[3];
        long long x3 = coord[4], y3 = coord[5];
        long long x4 = coord[6], y4 = coord[7];
        long long cx1 = coord[8], cy1 = coord[9];
        long long cx2 = coord[10], cy2 = coord[11];

        vector<long long> d;
        d.push_back((x2 - x1) * (x2 - x1) + (y2 - y1) * (y2 - y1));
        d.push_back((x3 - x1) * (x3 - x1) + (y3 - y1) * (y3 - y1));
        d.push_back((x4 - x1) * (x4 - x1) + (y4 - y1) * (y4 - y1));
        sort(d.begin(), d.end());
        long long area_sq = d[0] * d[1];
        int area_val = static_cast<int>(round(sqrt(static_cast<double>(area_sq))));

        long long ox4 = x1 + x2 + x3 + x4;
        long long oy4 = y1 + y2 + y3 + y4;

        long long ax4 = 4 * cx1;
        long long ay4 = 4 * cy1;
        long long bx4 = 4 * cx2;
        long long by4 = 4 * cy2;

        long long vax = ax4 - ox4;
        long long vay = ay4 - oy4;
        long long vbx = bx4 - ox4;
        long long vby = by4 - oy4;

        long long cross = vax * vby - vay * vbx;
        long long dot = vax * vbx + vay * vby;

        bool usable = false;
        if ((ax4 == ox4 && ay4 == oy4) || (bx4 == ox4 && by4 == oy4)) {
            usable = false;
        } else if (cross != 0 || dot < 0) {
            usable = true;
        } else {
            usable = false;
        }

        int usable_area = usable ? area_val : 0;
        pies.push_back({area_val, usable_area});
    }

    if (K == 0) {
        cout << "0.000 0.000" << endl;
        return 0;
    }

    sort(pies.begin(), pies.end(), [](const Pie& a, const Pie& b) {
        return a.area > b.area;
    });

    int r = pies[K-1].area;
    int m = 0;
    for (int i = 0; i < K; i++) {
        if (pies[i].area > r) {
            m++;
        }
    }

    vector<int> usable_r;
    for (int i = 0; i < N; i++) {
        if (pies[i].area == r) {
            usable_r.push_back(pies[i].usable_area);
        }
    }

    int t = usable_r.size();
    int L = K - m;

    sort(usable_r.begin(), usable_r.end());
    long long minL = 0;
    long long maxL = 0;
    for (int i = 0; i < L; i++) {
        minL += usable_r[i];
        maxL += usable_r[t - 1 - i];
    }

    long long fixed_usable = 0;
    for (int i = 0; i < m; i++) {
        fixed_usable += pies[i].usable_area;
    }

    long long min_usable = fixed_usable + minL;
    long long max_usable = fixed_usable + maxL;

    double pessimist = static_cast<double>(min_usable) / 2.0;
    double optimist = static_cast<double>(max_usable) / 2.0;

    cout << fixed << setprecision(3) << pessimist << " " << optimist << endl;

    return 0;
}
```

### Explanation
1. **Reading Input**: The input is read for N pies, each defined by 12 integers (4 corner coordinates and 2 candle coordinates).
2. **Area Calculation**: For each pie, the geometric area is computed using the squared distances from one corner to the others, then derived as the square root of the product of the two smallest squared distances.
3. **Usable Check**: A pie is usable if the two candles are distinct, not at the center, and can be separated by a line through the center (checked via cross and dot products of vectors from the center to the candles).
4. **Sorting and Selection**: Pies are sorted by area in descending order. The maximum total area (M) for K pies is the sum of the top K areas.
5. **Handling Ties**: Pies with areas equal to the K-th largest area (r) are grouped. The minimum and maximum sums of usable areas for selecting L pies from this group are computed.
6. **Result Calculation**: The fixed usable area from pies with area > r is added to the min and max sums from the group. The results are divided by 2 (since each person gets half the usable area) and printed to three decimal places.

This approach efficiently handles the constraints by leveraging sorting and careful grouping to avoid exhaustive subset checks, ensuring optimal performance even for the upper limits of input sizes.
