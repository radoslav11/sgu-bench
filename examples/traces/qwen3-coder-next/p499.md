#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <climits>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<int> a(n);
    int max_val = 0;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        max_val = max(max_val, a[i]);
    }

    // Count frequency of each number
    vector<int> count(max_val + 1, 0);
    for (int x : a) {
        count[x]++;
    }

    // For each possible gcd value d (from max_val down to 1),
    // count how many multiples of d exist in the array.
    // If at least 2 multiples exist, then d is achievable as gcd of some pair.
    for (int d = max_val; d >= 1; d--) {
        int cnt = 0;
        for (int multiple = d; multiple <= max_val; multiple += d) {
            cnt += count[multiple];
            // Early break if we already have enough
            if (cnt >= 2) {
                cout << d << '\n';
                return 0;
            }
        }
    }

    // This line should never be reached given the constraints (n >= 2 and numbers >= 1)
    // But just in case, output 1 as gcd of any two numbers is at least 1.
    cout << 1 << '\n';
    return 0;
}
