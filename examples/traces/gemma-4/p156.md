```cpp
        int v = (dir == 1) ? edge_to_v[eid] : edge_to_u[eid];
        // Path is u -> path_of_edge[eid] -> v
        if (res.empty() || res.back() != u) res.push_back(u);
        else { /* u is already the last vertex, skip it */ }
        for (int pv : path_of_edge[eid]) res.push_back(pv);
        if (res.back() != v) res.push_back(v);
    }
    // Now res has all vertices, but some block vertices might be missing.
    // For each block, if any vertex is missing, insert it after the first time its block is visited.
    // This is still not quite right because we need to avoid repeats.
    // Let's use the property that a Hamiltonian cycle visits each vertex once.
    // Let's collect all vertices in the order they are first visited.
    // Then for each block, if any vertex is not in the list, insert it.
    
    vector<int> ham;
    vector<bool> ham_visited(N+1, false);
    for (int v : res) {
        if (!ham_visited[v]) {
            ham.push_back(v);
            ham_visited[v] = true;
        }
    }
    // The order in 'ham' is not necessarily correct for a cycle.
    // Let's use the 'res' list but only keep the first occurrence.
    // And for each block, insert missing vertices.
    // The 'res' list is actually the Hamiltonian cycle if we insert the missing vertices.
    
    // Let's use a better approach for the final Hamiltonian cycle:
    // For each block b, the entry/exit vertices are v_in1, v_out1, v_in2, v_out2...
    // The sequence of vertices in the Eulerian circuit is:
    // ... (v_in1) [path_v2] (v_out1) ... (v_in2) [path_v2] (v_out2) ...
    // All v_in, v_out are in the clique, so we can visit them in any order.
    // Let's just use 'res' and handle duplicates.
    
    vector<int> final_res;
    for (int v : res) {
        if (!ham_visited[v]) {
            final_res.push_back(v);
            ham_visited[v] = true;
        }
    }
    // The above is still not working. Let's use the simple idea:
    // For each block, the first time we visit it, we visit all its vertices.
    // Let's use the 'res' sequence and for each block, if we see a vertex from it,
    // we check if it's the first time we see this block. If so, we add all vertices of the block.
    // But we must add them such that we start and end with the correct vertices.
    
    // Let's reconsider: In block b, we have endpoints v_in1, v_out1, v_in2, v_out2, ..., v_in_k, v_out_k.
    // We can visit: v_in1, v_in2, ..., v_in_k, (all other vertices), v_out_k, ..., v_out_1.
    // This is a single path that starts at v_in1 and ends at v_out1, and also visits v_in2, ..., v_out_k.
    // This is NOT correct. The Eulerian circuit is a sequence of paths.
    // In block b, the paths are P1, P2, ..., Pk.
    // P1 ends at v_out1, P2 starts at v_in2.
    // So the circuit is: ... -> v_in1 -> [P1's path] -> v_out1 -> v_in2 -> [P2's path] -> v_out2 -> ...
    // This is perfect! We can visit all vertices by inserting them into the first path.
    
    // Let's re-trace everything.
    // ... (after careful re-thinking)
    // The simple 'res' with duplicates removed will work if we insert missing vertices correctly.
    // The correct way is: 
    // 1. For each block, find its unused vertices.
    // 2. For each block, the first time it appears in the circuit (at v_in), 
    //    immediately add all its unused vertices, then add the path.
    
    vector<int> real_ham;
    vector<bool> block_seen(num_blocks + 1, false);
    vector<bool> v_seen(N + 1, false);
    for (int i = 0; i < res.size(); ++i) {
        int v = res[i];
        int b = block_id[v];
        if (!block_seen[b]) {
            block_seen[b] = true;
            // To make it a single cycle, we need to be careful.
            // Let's add all vertices of block b.
            // Which ones? The ones not yet seen.
            // But we must add them such that we don't break the cycle.
            // The current vertex v is already in res.
            // Let's add all other vertices of the block here.
            // We need to find all vertices in block b.
            // Let's pre-calculate them.
        }
    }
    
    // Actually, the most robust way:
    // The 'res' sequence is: v_in1, [path1], v_out1, v_in2, [path2], v_out2, ...
    // This already visits all vertices of all paths.
    // We only need to add vertices of each block that are not in any path.
    // A vertex is in a path if it's in path_of_edge or it's an endpoint of an edge.
    // Let's just add all missing vertices of block b after the first time we visit any vertex of b.
    
    // To ensure it's a single cycle:
    // After 'res' is built, some vertices of a block might be missing.
    // For each block b, find its first appearance in 'res' at index idx.
    // Insert all missing vertices of block b into 'res' at index idx + 1.
    
    // Wait, the 'res' list can have duplicates. Let's first remove them.
    // But we can't just remove them. We need to keep the structure.
    // The sequence of vertices is: v_in1, [path1], v_out1, v_in2, [path2], v_out2, ...
    // Let's call this the 'raw_res'. 
    // From 'raw_res', we can get a Hamiltonian cycle:
    // 1. Remove duplicates, but keep the first occurrence.
    // 2. For each block, if it has missing vertices, insert them after the first occurrence of any of its vertices.
    
    vector<int> raw_res;
    for (int i = 0; i < res.size(); ++i) {
        if (i == 0 || res[i] != res[i-1]) {
            raw_res.push_back(res[i]);
        }

    // Some v_out1 and v_in2 might be the same. Let's fix that.
    vector<int> clean_res;
    for (int v : raw_res) {
        if (clean_res.empty() || v != clean_res.back()) clean_res.push_back(v);
    }
    if (clean_res.size() > 1 && clean_res.front() == clean_res.back()) clean_res.pop_back();
    
    vector<bool> visited_in_ham(N + 1, false);
    vector<int> ham_final;
    vector<int> block_first_idx(num_blocks + 1, -1);
    for (int i = 0; i < clean_res.size(); ++i) {
        int v = clean_res[i];
        if (block_id[v] > 0 && block_first_idx[block_id[v]] == -1) {
            block_first_idx[block_id[v]] = i;
        }
    }

        ham_final.push_back(clean_res[i]);
        visited_in_ham[clean_res[i]] = true;

    // The above is not quite right, the order might be broken. 
    // Let's use the 'clean_res' and insert missing vertices.
    // For each block, its missing vertices can be inserted after its first vertex in 'clean_res'.
    vector<int> block_members[MAXN];
    for (int i = 1; i <= N; ++i) if (deg[i] > 2) block_members[block_id[i]].push_back(i);
    
    vector<int> final_ham_correct;
    for (int i = 0; i < clean_res.size(); ++i) {
        final_ham_correct.push_back(clean_res[i]);
        int b = block_id[clean_res[i]];
        if (b > 0 && block_first_idx[b] == i) {
            for (int v : block_members[b]) {
                if (!visited_in_ham[v]) {
                    final_ham_correct.push_back(v);
                    visited_in_ham[v] = true;
                }
            }
        }

        cout << final_ham_correct[i] << (i == final_ham_correct.size() - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
```

Actually, the block members and the cycle reconstruction should be handled more carefully. But the logic above is the most likely to work for the given "strange graph" structure. Let's clean it up for the final version. (The logic of inserting all block members after the first time any member of the block is seen in the Eulerian circuit's vertex sequence is sound for a clique).

Final check on time complexity: Finding $u$ for each $v$ is $O(N \cdot \text{deg}(v)^2)$ in the worst case, but with bitsets or the given clique condition, it should be fast. The constraints $N=10,000, M=100,000$ are small enough.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <bitset>
#include <stack>

using namespace std;

const int MAXN = 10005;
vector<int> adj[MAXN];
int deg[MAXN];
bitset<MAXN> neighbors[MAXN];
int special_neighbor[MAXN];
bool is_v2[MAXN];
int block_id[MAXN], block_size[MAXN], num_blocks = 0;
bool visited[MAXN];

struct Edge {
    int to, id, u_to_v;
};
vector<Edge> s_adj[MAXN];
bool used_edge[MAXN * 10];
int edge_to_u[MAXN * 10], edge_to_v[MAXN * 10];
vector<int> path_of_edge[MAXN * 10];
vector<pair<int, int>> circuit;

void hierholzer(int u) {
    stack<int> st;
    st.push(u);
    while (!st.empty()) {
        int v = st.top();
        if (!s_adj[v].empty()) {
            Edge e = s_adj[v].back();
            s_adj[v].pop_back();
            if (used_edge[e.id]) continue;
            used_edge[e.id] = true;
            st.push(e.to);
            // Store edge and direction to reconstruct later
            // Instead of this, let's use the recursive version which is simpler.
        } else {
            st.pop();
        }
    }
}

// Using recursive for simplicity
void hierholzer_rec(int u) {
    while (!s_adj[u].empty()) {
        Edge e = s_adj[u].back();
        s_adj[u].pop_back();
        if (used_edge[e.id]) continue;
        used_edge[e.id] = true;
        hierholzer_rec(e.to);
        circuit.push_back({e.id, e.u_to_v});
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    if (!(cin >> N >> M)) return 0;
    for (int i = 0; i < M; ++i) {
        int u, v; cin >> u >> v;
        adj[u].push_back(v); adj[v].push_back(u);
        neighbors[u].set(v); neighbors[v].set(u);
        deg[u]++; deg[v]++;
    }

    bool all_deg2 = true;
    for (int i = 1; i <= N; ++i) if (deg[i] != 2) { all_deg2 = false; break; }
    if (all_deg2) {
        vector<int> cycle;
        int curr = 1, prev = 0;
        for (int i = 0; i < N; ++i) {
            cycle.push_back(curr);
            int next_node = (adj[curr][0] == prev) ? adj[curr][1] : adj[curr][0];
            prev = curr; curr = next_node;
        }
        for (int i = 0; i < N; ++i) cout << cycle[i] << (i == N - 1 ? "" : " ");
        cout << endl; return 0;
    }

    for (int i = 1; i <= N; ++i) if (deg[i] == 2) is_v2[i] = true;
    for (int i = 1; i <= N; ++i) {
        if (deg[i] > 2) {
            int found_u = -1;
            for (int u : adj[i]) {
                if (is_v2[u]) {
                    bool clique = true;
                    vector<int> others;
                    for (int w : adj[i]) if (w != u) others.push_back(w);
                    for (int j = 0; j < (int)others.size(); ++j) {
                        for (int k = j + 1; k < (int)others.size(); ++k) {
                            if (!neighbors[others[j]].test(others[k])) { clique = false; break; }
                        }
                        if (!clique) break;
                    }
                    if (clique) { found_u = u; break; }
                }
            }
            if (found_u == -1) { cout << -1 << endl; return 0; }
            special_neighbor[i] = found_u;
        }
    }

    for (int i = 1; i <= N; ++i) {
        if (deg[i] > 2 && !visited[i]) {
            num_blocks++;
            vector<int> q = {i}; visited[i] = true;
            int head = 0;
            while (head < (int)q.size()) {
                int u = q[head++];
                block_id[u] = num_blocks; block_size[num_blocks]++;
                for (int v : adj[u]) if (deg[v] > 2 && !visited[v]) { visited[v] = true; q.push_back(v); }
            }
        }
    }

    int edge_count = 0;
    for (int i = 1; i <= N; ++i) {
        if (deg[i] > 2) {
            int u = special_neighbor[i];
            int curr = u, prev = i;
            vector<int> path_v2;
            while (is_v2[curr]) {
                path_v2.push_back(curr);
                int next_node = (adj[curr][0] == prev) ? adj[curr][1] : adj[curr][0];
                prev = curr; curr = next_node;
            }
            edge_count++;
            edge_to_u[edge_count] = i; edge_to_v[edge_count] = curr;
            path_of_edge[edge_count] = path_v2;
        }
    }

    for (int i = 1; i <= edge_count; ++i) {
        int b1 = block_id[edge_to_u[i]], b2 = block_id[edge_to_v[i]];
        s_adj[b1].push_back({b2, i, 1});
        s_adj[b2].push_back({b1, i, 0});
    }

    int start_block = 0;
    for (int i = 1; i <= num_blocks; ++i) {
        if (!s_adj[i].empty()) { start_block = i; break; }
    }
    if (start_block == 0) { cout << -1 << endl; return 0; }

    hierholzer_rec(start_block);
    if (circuit.size() != (size_t)edge_count) { cout << -1 << endl; return 0; }
    reverse(circuit.begin(), circuit.end());

    vector<int> res;
    for (auto& p : circuit) {
        int eid = p.first, dir = p.second;
        int u = (dir == 1) ? edge_to_u[eid] : edge_to_v[eid];
        int v = (dir == 1) ? edge_to_v[eid] : edge_to_u[eid];
        if (res.empty() || res.back() != u) res.push_back(u);
        for (int pv : path_of_edge[eid]) res.push_back(pv);
        if (res.back() != v) res.push_back(v);
    }

    vector<int> clean_res;
    for (int v : res) if (clean_res.empty() || v != clean_res.back()) clean_res.push_back(v);
    if (clean_res.size() > 1 && clean_res.front() == clean_res.back()) clean_res.pop_back();

    vector<int> block_members[MAXN];
    for (int i = 1; i <= N; ++i) if (deg[i] > 2) block_members[block_id[i]].push_back(i);

    vector<int> block_first_idx(num_blocks + 1, -1);
    for (int i = 0; i < (int)clean_res.size(); ++i) {
        int b = block_id[clean_res[i]];
        if (b > 0 && block_first_idx[b] == -1) block_first_idx[b] = i;
    }

    vector<int> final_ham;
    vector<bool> ham_visited(N + 1, false);
    for (int i = 0; i < (int)clean_res.size(); ++i) {
        int v = clean_res[i];
        final_ham.push_back(v);
        ham_visited[v] = true;
        int b = block_id[v];
        if (b > 0 && block_first_idx[b] == i) {
            for (int mv : block_members[b]) {
                if (!ham_visited[mv]) {
                    final_ham.push_back(mv);
                    ham_visited[mv] = true;
                }
            }
        }
    }

    for (int i = 0; i < (int)final_ham.size(); ++i) cout << final_ham[i] << (i == (int)final_ham.size() - 1 ? "" : " ");
    cout << endl;

    return 0;
}
```
