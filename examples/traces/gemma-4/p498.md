#include <iostream>
#include <cmath>
#include <iomanip>

/**
 * Problem: Compute the probability P(p < q | observations) where:
 *   - p and q are independent and uniformly distributed on [0, 1].
 *   - Coin 1 is tossed n1 times, m1 results are heads.
 *   - Coin 2 is tossed n2 times, m2 results are heads.
 * 
 * Let E be the observations (m1, n1, m2, n2).
 * The joint density of (p, q) given E is:
 *   f(p, q | E) = [p^{m1} (1-p)^{n1-m1} * q^{m2} (1-q)^{n2-m2}] / [B(m1+1, n1-m1+1) * B(m2+1, n2-m2+1)]
 * where B(x, y) is the Beta function.
 * 
 * We want to compute:
 *   P(p < q | E) = \int_0^1 \int_0^q f(p, q | E) dp dq
 *   = \int_0^1 [ \int_0^q p^{m1} (1-p)^{n1-m1} dp / B(m1+1, n1-m1+1) ] * [q^{m2} (1-q)^{n2-m2} / B(m2+1, n2-m2+1)] dq
 * 
 * The term in the first bracket is the regularized incomplete beta function I_q(a, b)
 * where a = m1+1 and b = n1-m1+1.
 * For integer a, b: I_q(a, b) = \sum_{j=a}^{a+b-1} \binom{a+b-1}{j} q^j (1-q)^{a+b-1-j}.
 * 
 * P(p < q | E) = \sum_{j=a}^{a+b-1} \binom{a+b-1}{j} \int_0^1 \frac{q^{j+m2} (1-q)^{a+b-1-j+n2-m2}}{B(m2+1, n2-m2+1)} dq
 *   = \sum_{j=a}^{a+b-1} \binom{a+b-1}{j} \frac{B(j+m2+1, a+b-j+n2-m2)}{B(m2+1, n2-m2+1)}
 * Let c = m2+1 and d = n2-m2+1. The sum is:
 *   \sum_{j=a}^{a+b-1} \binom{a+b-1}{j} \frac{B(j+c, a+b-j+d-1)}{B(c, d)}
 */

using namespace std;

// Function to compute the Beta function log value
double log_beta(double x, double y) {
    return lgamma(x) + lgamma(y) - lgamma(x + y);
}

void solve() {
    int n1, m1, n2, m2;
    if (!(cin >> n1 >> m1 >> n2 >> m2)) return;

    double a = m1 + 1.0;
    double b = n1 - m1 + 1.0;
    double c = m2 + 1.0;
    double d = n2 - m2 + 1.0;
    double K = a + b; // which is n1 + 1

    // Precompute the constant log(1/B(c, d))
    double log_inv_Bc_d = lgamma(c) + lgamma(d) - lgamma(c + d);

    double total_prob = 0.0;

    // We compute each term in the sum using the log-gamma function to ensure numerical stability.
    // T_j = \binom{a+b-1}{j} * B(j+c, a+b-j+d-1) / B(c, d)
    // T_j = \exp( lgamma(a+b) - lgamma(j+1) - lgamma(a+b-j) 
    //             + lgamma(j+c) + lgamma(a+b-j+d-1) - lgamma(a+b+c+d-1)
    //             + log_inv_Bc_d )
    for (int j = (int)a; j <= (int)(a + b - 1.0); ++j) {
        double log_T_j = lgamma(K) - lgamma(j + 1.0) - lgamma(K - j)
                         + lgamma(j + c) + lgamma(K - j + d - 1.0) - lgamma(K + c + d - 1.0)
                         + log_inv_Bc_d;
        total_prob += exp(log_T_j);
    }

    cout << fixed << setprecision(10) << total_prob << endl;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int T;
    if (!(cin >> T)) return 0;
    while (T--) {
        solve();
    }
    return 0;
}
