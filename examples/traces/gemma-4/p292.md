#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

/**
 * BigInt class to handle very large integers (up to 10^1000).
 * Uses schoolbook arithmetic for addition, subtraction, multiplication, and division.
 * Time complexity for D digits:
 * - Addition, subtraction, comparison: O(D)
 * - Multiplication: O(D^2)
 * - Division: O(D^2)
 */
struct BigInt {
    vector<int> d;

    BigInt() {}
    BigInt(long long v) {
        if (v == 0) d.push_back(0);
        while (v > 0) {
            d.push_back(v % 10);
            v /= 10;
        }
    }
    BigInt(string s) {
        for (int i = (int)s.size() - 1; i >= 0; --i) {
            d.push_back(s[i] - '0');
        }
        trim();
    }

    void trim() {
        while (d.size() > 1 && d.back() == 0) d.pop_back();
        if (d.empty()) d.push_back(0);
    }

    bool operator<(const BigInt& other) const {
        if (d.size() != other.d.size()) return d.size() < other.d.size();
        for (int i = (int)d.size() - 1; i >= 0; --i) {
            if (d[i] != other.d[i]) return d[i] < other.d[i];
        }
        return false;
    }

    bool operator>=(const BigInt& other) const {
        return !(*this < other);
    }

    BigInt operator+(const BigInt& other) const {
        BigInt res;
        res.d.clear();
        int carry = 0;
        int n = d.size(), m = other.d.size();
        for (int i = 0; i < max(n, m) || carry; ++i) {
            int sum = carry + (i < n ? d[i] : 0) + (i < m ? other.d[i] : 0);
            res.d.push_back(sum % 10);
            carry = sum / 10;
        }
        return res;
    }

    BigInt operator-(const BigInt& other) const {
        BigInt res = *this;
        int borrow = 0;
        int n = d.size(), m = other.d.size();
        for (int i = 0; i < m || borrow; ++i) {
            int sub = (i < m ? other.d[i] : 0) + borrow;
            if (res.d[i] < sub) {
                res.d[i] = res.d[i] + 10 - sub;
                borrow = 1;
            } else {
                res.d[i] -= sub;
                borrow = 0;
            }
        }
        res.trim();
        return res;
    }

    BigInt operator*(const BigInt& other) const {
        if ((d.size() == 1 && d[0] == 0) || (other.d.size() == 1 && other.d[0] == 0)) return BigInt(0);
        BigInt res;
        res.d.assign(d.size() + other.d.size(), 0);
        for (int i = 0; i < (int)d.size(); ++i) {
            int carry = 0;
            for (int j = 0; j < (int)other.d.size() || carry; ++j) {
                long long cur = res.d[i + j] + d[i] * 1LL * (j < (int)other.d.size() ? other.d[j] : 0) + carry;
                res.d[i + j] = cur % 10;
                carry = (int)(cur / 10);
            }
        }
        res.trim();
        return res;
    }

    BigInt operator*(int v) const {
        if (v == 0) return BigInt(0);
        BigInt res;
        res.d.clear();
        int carry = 0;
        for (int i = 0; i < (int)d.size() || carry; ++i) {
            long long cur = carry + (i < (int)d.size() ? d[i] : 0) * 1LL * v;
            res.d.push_back(cur % 10);
            carry = (int)(cur / 10);
        }
        res.trim();
        return res;
    }

    BigInt operator+(int v) const {
        return *this + BigInt(v);
    }

    string to_string() const {
        string s = "";
        for (int i = (int)d.size() - 1; i >= 0; --i) {
            s += (char)(d[i] + '0');
        }
        return s;
    }
};

// Standard schoolbook division algorithm returning {quotient, remainder}
pair<BigInt, BigInt> divide(BigInt a, BigInt b) {
    if (a < b) return {BigInt(0), a};
    BigInt quotient, remainder;
    quotient.d.assign(a.d.size(), 0);
    remainder = BigInt(0);
    for (int i = (int)a.d.size() - 1; i >= 0; --i) {
        remainder = remainder * 10 + a.d[i];
        int count = 0;
        while (remainder >= b) {
            remainder = remainder - b;
            count++;
        }
        quotient.d[i] = count;
    }
    quotient.trim();
    remainder.trim();
    return {quotient, remainder};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string qs, cs, ns;
    // Input format is 3 lines, but we can read using cin for flexibility.
    if (!(cin >> qs >> cs >> ns)) return 0;

    BigInt q(qs), c(cs), n(ns);

    /**
     * To maximize the number of n x 1 or 1 x n graves in a q x c rectangle:
     * We can partition the q x c area into two parts:
     * 1. A (q/n * n) x c part, where we place (q/n) * c graves of size n x 1.
     * 2. A (q % n) x c part, where we can only place graves of size 1 x n.
     *    The number of such graves is (q % n) * (c / n).
     * 
     * Total graves = floor(q/n) * c + floor(c/n) * (q % n)
     * This covers both orientations and provides the maximum packing.
     */
    pair<BigInt, BigInt> qr = divide(q, n);
    pair<BigInt, BigInt> cr = divide(c, n);

    BigInt term1 = qr.first * c;
    BigInt term2 = cr.first * qr.second;

    BigInt ans = term1 + term2;

    cout << ans.to_string() << endl;

    return 0;
}
