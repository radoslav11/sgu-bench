```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Problem Analysis:
 * The game is a standard Nim game. The condition for the second player (Petya) to win
 * is that the XOR sum of all piles must be zero after he takes x stones from each pile.
 * Let the initial sizes be a_i. After taking x stones from each pile, the sizes become
 * b_i = a_i - x. We need to find the number of x such that 0 <= x < min(a_i) and
 * (a_1 - x) ^ (a_2 - x) ^ ... ^ (a_n - x) = 0.
 * 
 * Digit DP Approach:
 * Let's process the bits of x and (a_i - x) from least significant to most significant.
 * Let c_{i, k-1} be the carry from bit k-1 to bit k in the addition b_i + x = a_i.
 * The k-th bit of b_i is 1 if and only if the k-th bit of a_i - x is 1.
 * Let S_{k-1} be the set of indices i such that there is a carry (or borrow) from 
 * bit k-1 to bit k in the subtraction a_i - x. 
 * This is equivalent to a_i % 2^k < x % 2^k.
 * Let n_S be the size of S_{k-1}, and n_S_1 be the number of elements in S_{k-1} 
 * whose k-th bit of a_i is 1.
 * 
 * At bit k, we decide x_k (0 or 1). The number of b_{i,k} = 1 is:
 * If x_k = 0: n_{b,k} = (number of i in S_{k-1} with a_{i,k}=0) + (number of i not in S_{k-1} with a_{i,k}=1).
 *             n_{b,k} = (n_S - n_S_1) + (count_a_1[k] - n_S_1) = n_S - 2 * n_S_1 + count_a_1[k].
 * If x_k = 1: n_{b,k} = (number of i not in S_{k-1} with a_{i,k}=0) + (number of i in S_{k-1} with a_{i,k}=1).
 *             n_{b,k} = (count_a_0[k] - (n_S - n_S_1)) + n_S_1 = count_a_0[k] - n_S + 2 * n_S_1.
 * 
 * For a valid x, n_{b,k} must be even for all k.
 * We transition to k+1 by updating n_S and n_S_1:
 * If x_k = 0: n_{S_k} = n_S - n_S_1, n_{S_k, 1} = 0.
 * If x_k = 1: n_{S_k} = count_a_0[k] + n_S_1, n_{S_k, 1} = n_S_1.
 */

typedef long long ll;

int n;
ll a[200005];
ll min_a = 2e18; 
int count_a_1[62];
int count_a_0[62];
ll count_x_total = 0;

// k: current bit
// x: value of x chosen so far
// n_S: number of i where a_i % 2^k < x % 2^k
// n_S_1: number of i in S where the k-th bit of a_i is 1
void dfs(int k, ll x, int n_S, int n_S_1) {
    if (k == 61) {
        if (x < min_a) count_x_total++;
        return;
    }

    // Try choosing x_k = 0
    // n_b_0 is the number of i such that the k-th bit of b_i is 1.
    // We must have n_b_0 % 2 == 0 for the k-th bit of the XOR sum to be 0.
    ll n_b_0 = (ll)n_S - 2LL * n_S_1 + count_a_1[k];
    if (n_b_0 % 2 == 0) {
        // Transition for x_k = 0:
        // S_k = {i in S_{k-1} | a_{i,k} = 0}
        dfs(k + 1, x, n_S - n_S_1, 0);
    }

    // Try choosing x_k = 1
    ll n_b_1 = (ll)count_a_0[k] - n_S + 2LL * n_S_1;
    if (n_b_1 % 2 == 0) {
        // Transition for x_k = 1:
        // S_k = S_{k-1} U {i not in S_{k-1} | a_{i,k} = 0}
        // n_{S_k} = n_S + (count_a_0[k] - (n_S - n_S_1)) = count_a_0[k] + n_S_1
        // n_{S_k, 1} = n_{S_{k-1}, 1} (because if i not in S_{k-1} and a_{i,k}=1, i is not in S_k)
        dfs(k + 1, x | (1LL << k), count_a_0[k] + n_S_1, n_S_1);
    }
}

int main() {
    // Optimization for fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> n)) return 0;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        if (a[i] < min_a) min_a = a[i];
    }

    // Pre-calculate number of piles with 0 or 1 at each bit position
    for (int k = 0; k <= 60; k++) {
        count_a_1[k] = 0;
        for (int i = 0; i < n; i++) {
            if ((a[i] >> k) & 1) {
                count_a_1[k]++;
            }
        }
        count_a_0[k] = n - count_a_1[k];
    }

    // Start DFS from the 0-th bit
    dfs(0, 0, 0, 0);

    cout << count_x_total << endl;

    return 0;
}
```
