#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Function to count the number of new distinct substrings added when appending character 'c' to string 's'.
 * The number of new substrings is (length(s) + 1) - (length of the longest suffix of s+c that is also a substring of s).
 * 
 * @param s The current string.
 * @param c The character being appended.
 * @return The number of new distinct substrings.
 */
int count_new_substrings(const string& s, char c) {
    int n = s.length();
    if (n == 0) return 1;
    
    // We are looking for the largest L in [1, n] such that the suffix of length L of s+c
    // is already a substring of s. The suffix of length L is s.substr(n - L + 1) + c.
    for (int L = n; L >= 1; --L) {
        // We want to check if the suffix of length L, which is s[n-L+1 ... n-1] + c,
        // is a substring of s.
        int suffix_start_in_s = n - L + 1;
        for (int i = 0; i <= n - L; ++i) {
            // A possible match must have s[i + L - 1] == c
            if (s[i + L - 1] == c) {
                bool match = true;
                // Compare s[i ... i+L-2] with s[n-L+1 ... n-1]
                for (int j = 0; j < L - 1; ++j) {
                    if (s[i + j] != s[suffix_start_in_s + j]) {
                        match = false;
                        break;
                    }
                }
                if (match) return (n + 1) - L;
            }
        }
    }
    // If no suffix of length 1 to n of s+c is a substring of s, the only 
    // possible new substring is s+c itself (length n+1), but we already
    // checked all smaller suffixes. So the longest substring suffix of s+c in s is 0.
    return n + 1;
}

/**
 * Depth-first search to find the lexicographically smallest string of target_len with exactly n substrings.
 * 
 * @param current_len Current length of the string being built.
 * @param target_len The target length of the string.
 * @param current_n The number of distinct substrings in the current string.
 * @param s The current string being built.
 * @param n The target number of distinct substrings.
 * @return True if a valid string is found, false otherwise.
 */
bool dfs(int current_len, int target_len, int current_n, string& s, int n) {
    if (current_len == target_len) {
        return current_n == n;
    }
    
    int remaining = target_len - current_len;
    
    // Pruning 1: Each additional character adds at least 1 new substring.
    if (current_n + remaining > n) return false;
    
    // Pruning 2: Each additional character adds at most its current index (length + 1) new substrings.
    // The total possible number of substrings in target_len is target_len*(target_len+1)/2.
    // The maximum number of additional substrings that can be added is:
    // (target_len*(target_len+1)/2) - (current_len*(current_len+1)/2).
    if (current_n + (target_len * (target_len + 1) / 2 - current_len * (current_len + 1) / 2) < n) return false;

    for (char c = 'a'; c <= 'z'; ++c) {
        int added = count_new_substrings(s, c);
        // Only proceed if the number of substrings does not exceed n.
        if (current_n + added <= n) {
            s.push_back(c);
            if (dfs(current_len + 1, target_len, current_n + added, s, n)) return true;
            s.pop_back();
        }
    }
    return false;
}

int main() {
    // Speed up I/O operations
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    // We search for the shortest possible length len.
    // For a given len, we search for the lexicographically smallest string.
    // The maximum length for n=300 is n (e.g., "aaaa...a").
    for (int len = 1; len <= n; ++len) {
        string s = "";
        if (dfs(0, len, 0, s, n)) {
            cout << s << endl;
            return 0;
        }
    }

    return 0;
}
