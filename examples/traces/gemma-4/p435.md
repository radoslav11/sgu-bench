```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>

using namespace std;

/**
 * Problem Summary:
 * We are given N circles on a plane. Each circle application toggles the state of a point 
 * (from grain to burned or vice versa). The problem asks for:
 * 1. Area of burned fields (covered by an odd number of circles).
 * 2. Area of inverse grain (covered by an even number of circles that's >= 1).
 * 
 * Mathematical Insight:
 * Let N(x,y) be the number of circles covering point (x,y).
 * Burned area = { (x,y) : N(x,y) is odd }
 * Inverse grain area = { (x,y) : N(x,y) is even and N(x,y) > 0 }
 * 
 * We use Green's Theorem for area calculation: Area = 1/2 * Integral of (x dy - y dx) 
 * around the boundary. 
 * For any circle i, an arc of its circumference is part of the boundary of the odd-covered 
 * region if it's covered by an even number of other circles (the state changes from 
 * even to odd at this arc). If it's covered by an odd number of other circles, 
 * the arc also separates an odd-covered area from an even-covered area, 
 * so its contribution to the area integral is negated.
 * 
 * For the area of the union of circles, the boundary consists of arcs covered by 0 other circles.
 * Union area = Area(odd-covered) + Area(even-covered and N >= 2).
 * Thus, Area(even-covered and N >= 2) = Area(union) - Area(odd-covered).
 */

typedef long double ld;

// Using M_PI or its calculated version for precision.
const ld PI = acos(-1.0L);

struct Circle {
    ld x, y, r;
};

const ld EPS = 1e-11;

// Normalizes angles to the range [0, 2*PI]
ld normalize_angle(ld angle) {
    while (angle < 0) angle += 2 * PI;
    while (angle >= 2 * PI) angle -= 2 * PI;
    return angle;
}

// Integral of 0.5 * (x dy - y dx) along a circle arc
// The antiderivative is F(t) = 0.5 * (r^2 * t + x_c * r * sin(t) - y_c * r * cos(t))
ld arc_integral(const Circle& c, ld t1, ld t2) {
    auto f = [&](ld t) {
        return 0.5L * (c.r * c.r * t + c.x * c.r * sin(t) - c.y * c.r * cos(t));
    };
    return f(t2) - f(t1);
}

int main() {
    // Optimize I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    vector<Circle> circles(n);
    for (int i = 0; i < n; ++i) {
        double tx, ty, tr;
        cin >> tx >> ty >> tr;
        circles[i].x = tx;
        circles[i].y = ty;
        circles[i].r = tr;
    }

    ld area_odd = 0;
    ld area_union = 0;

    for (int i = 0; i < n; ++i) {
        vector<ld> angles;
        angles.push_back(0.0L);
        angles.push_back(2.0L * PI);

        for (int j = 0; j < n; ++j) {
            if (i == j) continue;

            ld dx = circles[j].x - circles[i].x;
            ld dy = circles[j].y - circles[i].y;
            ld d2 = dx * dx + dy * dy;
            ld d = sqrt(d2);

            // Check if circles intersect
            if (d > circles[i].r + circles[j].r || d < abs(circles[i].r - circles[j].r) || d < EPS) {
                continue;
            }

            // Distance from circle i center to the chord between intersection points
            ld a = (circles[i].r * circles[i].r - circles[j].r * circles[j].r + d2) / (2.0L * d);
            // Half-length of the chord
            ld h_val = circles[i].r * circles[i].r - a * a;
            ld h = sqrt(max(0.0L, h_val));

            ld phi = atan2(dy, dx);
            // Angle from center to intersection points
            ld alpha = acos(max(-1.0L, min(1.0L, a / circles[i].r)));

            angles.push_back(normalize_angle(phi - alpha));
            angles.push_back(normalize_angle(phi + alpha));
        }

        // Sort intersection angles to define arcs
        sort(angles.begin(), angles.end());

        for (size_t j = 0; j < angles.size() - 1; ++j) {
            ld t1 = angles[j];
            ld t2 = angles[j + 1];
            if (t2 - t1 < 1e-15L) continue; // Skip very small intervals

            ld mid_t = (t1 + t2) / 2.0L;
            ld mid_x = circles[i].x + circles[i].r * cos(mid_t);
            ld mid_y = circles[i].y + circles[i].r * sin(mid_t);

            // Count how many other circles cover the arc's midpoint
            int k = 0;
            for (int m = 0; m < n; ++m) {
                if (i == m) continue;
                ld dist2 = (mid_x - circles[m].x) * (mid_x - circles[m].x) + (mid_y - circles[m].y) * (mid_y - circles[m].y);
                if (dist2 < circles[m].r * circles[m].r - EPS) {
                    k++;
                }
            }

            ld term = arc_integral(circles[i], t1, t2);
            if (k % 2 == 0) {
                // Even coverage: arc separates even (outside) and odd (inside) regions.
                // Contribution is positive for the area of odd regions.
                area_odd += term;
                // If k=0, it is on the boundary of the union.
                if (k == 0) {
                    area_union += term;
                }
            } else {
                // Odd coverage: arc separates odd (outside) and even (inside) regions.
                // Contribution to the odd region is negative (from the integration direction).
                area_odd -= term;
            }
        }
    }

    // The problem asks for area_odd (burned) and area_union - area_odd (inverse grain).
    // Ensure we don't output slightly negative results due to precision.
    ld burned = max(0.0L, area_odd);
    ld inverse = max(0.0L, area_union - area_odd);

    cout << fixed << setprecision(5) << (double)burned << " " << (double)inverse << endl;

    return 0;
}
```
