#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <cctype>

using namespace std;

/**
 * Problem 432. XYZX 2009
 * The goal is to find all possible single-negation versions of the given problem statement.
 * A negation can be:
 * - "not" after "can", "may", "must", "should" (case-insensitive).
 * - "no" after "is", "are" (case-insensitive). If "a" or "the" follows "is" or "are", it must be removed.
 * "not" and "no" are case-sensitive.
 * Output the count of such single-negation versions, then each modified sentence on a new line.
 */

// Function to compare strings case-insensitively
bool is_case_insensitive(const string& s1, const string& s2) {
    if (s1.length() != s2.length()) return false;
    for (size_t i = 0; i < s1.length(); ++i) {
        if (tolower((unsigned char)s1[i]) != tolower((unsigned char)s2[i])) return false;
    }
    return true;
}

int main() {
    // Faster I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string input, line;
    // Read all input into a single string
    while (getline(cin, line)) {
        input += line + "\n";
    }

    // Removing the last newline added by getline if it's there
    if (!input.empty() && input.back() == '\n') {
        input.pop_back();
    }

    // Split the input into sentences using the dot '.' as a delimiter.
    // Each sentence ends with a dot.
    vector<vector<string>> sentences;
    string current_sentence_raw;
    for (size_t i = 0; i < input.length(); ++i) {
        current_sentence_raw += input[i];
        if (input[i] == '.') {
            // Extract words (latin letters only) from the sentence
            vector<string> words;
            string word;
            for (size_t j = 0; j < current_sentence_raw.length(); ++j) {
                if (isalpha((unsigned char)current_sentence_raw[j])) {
                    word += current_sentence_raw[j];
                } else {
                    if (!word.empty()) {
                        words.push_back(word);
                        word = "";
                    }
                }
            }
            sentences.push_back(words);
            current_sentence_raw = "";
        }
    }

    vector<string> results;
    // For each sentence, check for all possible negation points
    for (const auto& words : sentences) {
        for (size_t i = 0; i < words.size(); ++i) {
            bool can_negate = false;
            bool is_not_rule = false;
            string negation_word;

            // Rule for "not" (after can, may, must, should)
            if (is_case_insensitive(words[i], "can") || 
                is_case_insensitive(words[i], "may") || 
                is_case_insensitive(words[i], "must") || 
                is_case_insensitive(words[i], "should")) {
                can_negate = true;
                is_not_rule = true;
                negation_word = "not";
            } 
            // Rule for "no" (after is, are)
            else if (is_case_insensitive(words[i], "is") || 
                     is_case_insensitive(words[i], "are")) {
                can_negate = true;
                is_not_rule = false;
                negation_word = "no";
            }

            if (can_negate) {
                vector<string> new_words = words;
                if (is_not_rule) {
                    // Rule: Insert "not" after the word
                    new_words.insert(new_words.begin() + i + 1, negation_word);
                } else {
                    // Rule: Insert "no" after the word. If "a" or "the" is the next word, remove it.
                    if (i + 1 < new_words.size() && 
                        (is_case_insensitive(new_words[i+1], "a") || 
                         is_case_insensitive(new_words[i+1], "the"))) {
                        new_words.insert(new_words.begin() + i + 1, negation_word);
                        new_words.erase(new_words.begin() + i + 2);
                    } else {
                        new_words.insert(new_words.begin() + i + 1, negation_word);
                    }
                }

                // Reconstruct the sentence from modified words
                string res_sentence;
                for (size_t j = 0; j < new_words.size(); ++j) {
                    res_sentence += new_words[j];
                    if (j < new_words.size() - 1) res_sentence += " ";
                }
                res_sentence += ".";
                results.push_back(res_sentence);
            }
        }
    }

    // Output the number of ways and then each way
    cout << results.size() << "\n";
    for (const auto& res : results) {
        cout << res << "\n";
    }

    return 0;
}
