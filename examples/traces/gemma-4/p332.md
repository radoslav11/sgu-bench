#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <deque>
#include <cstdio>

using namespace std;

/**
 * Problem: Find the radius of the largest circle that can fit inside a convex N-gon.
 * Approach: 
 * 1. The problem is equivalent to finding the maximum R such that the intersection of 
 *    half-planes defined by the polygon's edges shifted inwards by distance R is non-empty.
 * 2. Since the polygon is convex, the intersection of shifted edges is also a convex polygon.
 * 3. We can binary search for the radius R. For a fixed R, we check if the intersection 
 *    of the shifted edges is non-empty using an O(N) half-plane intersection algorithm.
 * 4. Given that the edges of the original polygon are already sorted by angle, 
 *    the half-plane intersection check takes O(N).
 * 5. Total complexity: O(N * log(precision)).
 */

struct Point {
    double x, y;
};

struct Line {
    Point p, v;
    double ang;
};

// Function to find the intersection point of two lines.
Point intersect(const Line& a, const Line& b) {
    double cross = a.v.x * b.v.y - a.v.y * b.v.x;
    if (abs(cross) < 1e-15) return a.p; 
    double t = ((b.p.x - a.p.x) * b.v.y - (b.p.y - a.p.y) * b.v.x) / cross;
    return {a.p.x + t * a.v.x, a.p.y + t * a.v.y};
}

// Function to check if a point is to the right of a directed line.
bool is_right(const Line& l, const Point& p) {
    return l.v.x * (p.y - l.p.y) - l.v.y * (p.x - l.p.x) < -1e-10;
}

// Function to check if a radius R is feasible.
bool check(double R, int N, const vector<Line>& original_lines) {
    static vector<Line> lines;
    lines.resize(N);
    for (int i = 0; i < N; ++i) {
        double dx = original_lines[i].v.x;
        double dy = original_lines[i].v.y;
        double d = sqrt(dx * dx + dy * dy);
        // Normal vector pointing inwards for CCW polygon.
        double nx = -dy / d;
        double ny = dx / d;
        lines[i].p = {original_lines[i].p.x + R * nx, original_lines[i].p.y + R * ny};
        lines[i].v = original_lines[i].v;
        lines[i].ang = original_lines[i].ang;
    }

    // Find the starting index for sorted angles.
    int start = 0;
    for (int i = 0; i < N; ++i) {
        int prev = (i + N - 1) % N;
        if (lines[i].ang < lines[prev].ang) {
            start = i;
            break;
        }
    }

    deque<Line> dq;
    for (int i = 0; i < N; ++i) {
        const Line& cur = lines[(start + i) % N];
        // Standard O(N) half-plane intersection for pre-sorted lines.
        while (dq.size() >= 2 && is_right(cur, intersect(dq[dq.size() - 2], dq.back()))) {
            dq.pop_back();
        }
        while (dq.size() >= 2 && is_right(cur, intersect(dq[0], dq[1]))) {
            dq.pop_front();
        }
        dq.push_back(cur);
    }
    while (dq.size() >= 3 && is_right(dq[0], intersect(dq[dq.size() - 2], dq.back()))) {
        dq.pop_back();
    }
    while (dq.size() >= 3 && is_right(dq.back(), intersect(dq[0], dq[1]))) {
        dq.pop_front();
    }

    if (dq.size() < 3) return false;

    // Check if the resulting intersection is non-empty using its area.
    int sz = dq.size();
    vector<Point> pts(sz);
    for (int i = 0; i < sz; ++i) {
        pts[i] = intersect(dq[i], dq[(i + 1) % sz]);
    }
    double area = 0;
    for (int i = 0; i < sz; ++i) {
        area += (pts[i].x * pts[(i + 1) % sz].y - pts[(i + 1) % sz].x * pts[i].y);
    }
    // Use a small epsilon because area can be effectively zero for the maximal radius.
    return area > 1e-8;
}

int main() {
    int N;
    if (scanf("%d", &N) != 1) return 0;
    vector<Point> p(N);
    for (int i = 0; i < N; ++i) {
        scanf("%lf %lf", &p[i].x, &p[i].y);
    }

    // Pre-calculate original edges.
    vector<Line> original_lines(N);
    for (int i = 0; i < N; ++i) {
        Point p1 = p[i];
        Point p2 = p[(i + 1) % N];
        original_lines[i].p = p1;
        original_lines[i].v = {p2.x - p1.x, p2.y - p1.y};
        original_lines[i].ang = atan2(original_lines[i].v.y, original_lines[i].v.x);
    }

    // Binary search for the largest possible radius.
    double low = 0, high = 2e7; 
    for (int iter = 0; iter < 45; ++iter) {
        double mid = (low + high) / 2.0;
        if (check(mid, N, original_lines)) {
            low = mid;
        } else {
            high = mid;
        }
    }

    printf("%.10f\n", low);

    return 0;
}
