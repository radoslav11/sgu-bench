#include <cstdio>
#include <cstring>

using namespace std;

/**
 * Problem: 311. Ice-cream Tycoon
 * Goal: Maintain a stock of ice cream with varying costs and process BUY queries.
 * Each BUY query asks if the n cheapest pieces cost <= t.
 * Data Structure: Segment Tree on the range of possible prices [1, 10^6].
 * Each node in the tree stores:
 *   - count: Total number of ice cream pieces in this price range.
 *   - cost: Total cost of all ice cream pieces in this price range.
 * 
 * Time Complexity:
 *   - ARRIVE: O(log(MAX_P))
 *   - BUY: O(log(MAX_P)) for cost calculation and O(log(MAX_P)) for removal.
 *   - Overall: O(Q log(MAX_P)), where Q is number of queries and MAX_P = 10^6.
 * 
 * Space Complexity:
 *   - O(MAX_P), specifically 4 * 10^6 nodes.
 *   - Memory: 4,000,005 * 16 bytes (2 long longs) is approximately 61 MiB, which fits in 64 MiB limit.
 */

const int MAX_P = 1000000;

struct Node {
    long long count;
    long long cost;
} tree[4000005];

// Add n ice cream pieces of price 'price' to the stock.
void update(int node, int L, int R, int price, int quantity) {
    if (L == R) {
        tree[node].count += (long long)quantity;
        tree[node].cost += (long long)quantity * price;
        return;
    }
    int mid = L + (R - L) / 2;
    if (price <= mid) 
        update(2 * node, L, mid, price, quantity);
    else 
        update(2 * node + 1, mid + 1, R, price, quantity);
    
    tree[node].count = tree[2 * node].count + tree[2 * node + 1].count;
    tree[node].cost = tree[2 * node].cost + tree[2 * node + 1].cost;
}

// Calculate the cost of the n cheapest ice cream pieces.
long long query_cost(int node, int L, int R, long long n) {
    if (n <= 0) return 0;
    if (L == R) return n * L;
    int mid = L + (R - L) / 2;
    if (tree[2 * node].count >= n) {
        return query_cost(2 * node, L, mid, n);
    } else {
        return tree[2 * node].cost + query_cost(2 * node + 1, mid + 1, R, n - tree[2 * node].count);
    }
}

// Remove the n cheapest ice cream pieces from the stock.
void remove_cheapest(int node, int L, int R, long long n) {
    if (n <= 0) return;
    if (L == R) {
        tree[node].count -= n;
        tree[node].cost -= n * L;
        return;
    }
    int mid = L + (R - L) / 2;
    if (tree[2 * node].count >= n) {
        remove_cheapest(2 * node, L, mid, n);
    } else {
        long long left_n = tree[2 * node].count;
        // Optimizing removal of the entire left child:
        // We set the left child's count and cost to zero and then update the ancestor.
        // This keeps the complexity O(log(MAX_P)) because the descendants are not visited.
        // Subsequent updates or queries will correctly re-update/ignore them.
        tree[2 * node].count = 0;
        tree[2 * node].cost = 0;
        remove_cheapest(2 * node + 1, mid + 1, R, n - left_n);
    }
    tree[node].count = tree[2 * node].count + tree[2 * node + 1].count;
    tree[node].cost = tree[2 * node].cost + tree[2 * node + 1].cost;
}

int main() {
    char type[15];
    long long n, val;

    // Using scanf for faster I/O as the time limit is quite tight (0.5s).
    while (scanf("%s %lld %lld", type, &n, &val) != EOF) {
        if (type[0] == 'A') { // ARRIVE n c
            // price 'val' can be up to 10^6, fits in int.
            // quantity 'n' can be up to 10^6, fits in int.
            update(1, 1, MAX_P, (int)val, (int)n);
        } else { // BUY n t
            // If we have fewer than n pieces, the student cannot buy.
            if (tree[1].count < n) {
                printf("UNHAPPY\n");
            } else {
                // val is the budget t.
                long long min_cost = query_cost(1, 1, MAX_P, n);
                if (min_cost <= val) {
                    printf("HAPPY\n");
                    remove_cheapest(1, 1, MAX_P, n);
                } else {
                    printf("UNHAPPY\n");
                }
            }
        }
    }

    return 0;
}
