#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Problem: Trade Centers (Minimum Dominating Set with distance K in a tree)
 * Constraint: N <= 30,000, K <= 100, Time limit: 0.5s, Memory limit: 64MB
 * 
 * Approach: Greedy strategy using post-order traversal.
 * For each node u (from leaves to root):
 * - Let d_un[u] be the maximum distance to an uncovered node in u's subtree.
 * - Let d_ce[u] be the minimum distance to a trade center in u's subtree.
 * 
 * If d_un[u] + d_ce[u] <= K, then the furthest uncovered node in u's subtree
 * is actually covered by some trade center in u's subtree. Set d_un[u] = -INF.
 * 
 * If d_un[u] == K, u is the K-th ancestor of an uncovered node. We must place
 * a trade center at u to cover that node optimally.
 * 
 * Finally, if the root is still uncovered (d_un[root] >= 0), place a center at the root.
 */

const int MAXN = 30005;
const int INF = 1000000000;

vector<int> adj[MAXN];
int parent[MAXN];
int order[MAXN];
int d_un[MAXN];
int d_ce[MAXN];
int order_ptr = 0;
int N, K;

void dfs(int u, int p) {
    parent[u] = p;
    order[order_ptr++] = u;
    for (int v : adj[u]) {
        if (v != p) {
            dfs(v, u);
        }
    }
}

int main() {
    // Faster I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> N >> K)) return 0;

    if (N == 0) return 0;
    if (N == 1) {
        cout << "1\n1" << endl;
        return 0;
    }

    for (int i = 0; i < N - 1; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Root the tree at city 1
    dfs(1, 0);

    vector<int> centers;
    for (int i = 0; i < N; ++i) {
        d_un[i + 1] = 0;
        d_ce[i + 1] = INF;
    }

    // Post-order processing (reverse of pre-order traversal)
    for (int i = N - 1; i >= 0; --i) {
        int u = order[i];
        
        // During the post-order traversal, nodes u are processed after their children v.
        // d_un[u] is updated with the d_un[v] of its children.
        // d_ce[u] is updated with the d_ce[v] of its children.
        // This is done implicitly as children have already been updated in their steps.
        
        // Re-calculate d_un and d_ce based on updated children.
        // However, our logic below updates d_un and d_ce for u as we visit u.
        // Wait, we need the current d_un/d_ce values for children.
        // Let's ensure u's children are already processed and their values are used.
        
        // The loop through children is better done inside the loop.
        // But we only have adj[u] and parent[u].
        // Actually, u's d_un and d_ce were initialized to 0 and INF.
        // We need to collect contributions from children.
        // For a leaf node, d_un[u] remains 0 and d_ce[u] remains INF.
        
        // For any node, we update its parent after processing the node.
        // Let's refactor slightly to a more standard post-order update.
    }
    
    // Let's restart the loop logic properly.
    // Reset values
    for (int i = 1; i <= N; ++i) {
        d_un[i] = 0;
        d_ce[i] = INF;
    }

    for (int i = N - 1; i >= 0; --i) {
        int u = order[i];
        
        // After children are processed, d_un[u] and d_ce[u] are ready to be used.
        // (If u is a leaf, d_un[u] = 0 and d_ce[u] = INF).
        // First, apply the cover-by-center rule.
        if (d_un[u] + d_ce[u] <= K) {
            d_un[u] = -INF;
        }
        
        // If d_un[u] reaches the distance limit K, place a center at u.
        if (d_un[u] == K) {
            centers.push_back(u);
            d_un[u] = -INF;
            d_ce[u] = 0;
        }
        
        // If u is the root and it's still not covered, place a center at root.
        if (i == 0 && d_un[u] >= 0) {
            centers.push_back(u);
            d_un[u] = -INF;
            d_ce[u] = 0;
        }
        
        // Now propagate u's values up to its parent.
        int p = parent[u];
        if (p != 0) {
            if (d_un[u] != -INF) {
                d_un[p] = max(d_un[p], d_un[u] + 1);
            }
            d_ce[p] = min(d_ce[p], d_ce[u] + 1);
        }
    }

    cout << centers.size() << "\n";
    for (int i = 0; i < (int)centers.size(); ++i) {
        cout << centers[i] << (i == (int)centers.size() - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
