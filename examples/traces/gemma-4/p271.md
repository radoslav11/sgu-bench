#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <random>
#include <cstdio>
#include <cstring>

using namespace std;

/**
 * The problem asks for a data structure that supports adding an element to the top
 * and rotating the top K elements of a pile. Since the maximum number of books is
 * relatively small (40,000) and the number of operations is 100,000, a Treap 
 * (a balanced binary search tree) can handle these operations in O(log N) time.
 * 
 * - ADD(S): Add a node to the front of the Treap.
 * - ROTATE: Split the Treap into the first K elements and the rest, then further split 
 *   the first K elements into the first element and the other K-1 elements, and 
 *   re-merge them in the order (K-1 elements, 1st element).
 * 
 * The total number of nodes should be at most 140,000 (N + M).
 * Time limit: 0.25s, which is very tight, so we use efficient array-based Treap
 * and fast I/O.
 */

struct Node {
    char name[4];
    unsigned int priority;
    int size;
    int left, right;
} tree[140005];

int node_count = 0;
int root = 0;
mt19937 rng(42);

// Function to create a new Treap node
int new_node(const char* s) {
    int idx = ++node_count;
    int i = 0;
    while (s[i] != '\0' && i < 3) {
        tree[idx].name[i] = s[i];
        i++;
    }
    tree[idx].name[i] = '\0';
    tree[idx].priority = rng();
    tree[idx].size = 1;
    tree[idx].left = tree[idx].right = 0;
    return idx;
}

inline int get_size(int idx) {
    return idx ? tree[idx].size : 0;
}

inline void update_size(int idx) {
    if (idx) {
        tree[idx].size = 1 + get_size(tree[idx].left) + get_size(tree[idx].right);
    }
}

// Splits the Treap into two, one with k nodes and one with the rest
void split(int idx, int k, int& l, int& r) {
    if (!idx) {
        l = r = 0;
        return;
    }
    int left_size = get_size(tree[idx].left);
    if (left_size < k) {
        l = idx;
        split(tree[idx].right, k - left_size - 1, tree[idx].right, r);
    } else {
        r = idx;
        split(tree[idx].left, k, l, tree[idx].left);
    }
    update_size(idx);
}

// Merges two Treaps
int merge(int l, int r) {
    if (!l || !r) return l | r;
    if (tree[l].priority > tree[r].priority) {
        tree[l].right = merge(tree[l].right, r);
        update_size(l);
        return l;
    } else {
        tree[r].left = merge(l, tree[r].left);
        update_size(r);
        return r;
    }
}

// In-order traversal to print the book names
void print_treap(int idx) {
    if (!idx) return;
    if (tree[idx].left) print_treap(tree[idx].left);
    printf("%s\n", tree[idx].name);
    if (tree[idx].right) print_treap(tree[idx].right);
}

int main() {
    int N, M, K;
    if (scanf("%d %d %d", &N, &M, &K) != 3) return 0;

    // Initial pile
    for (int i = 0; i < N; ++i) {
        char name_buf[10];
        if (scanf("%s", name_buf) == 1) {
            root = merge(root, new_node(name_buf));
        }
    }

    // Performing M operations
    for (int i = 0; i < M; ++i) {
        char op_buf[100];
        if (scanf("%s", op_buf) == 1) {
            // ADD(S)
            if (op_buf[0] == 'A' && op_buf[1] == 'D' && op_buf[2] == 'D' && op_buf[3] == '(') {
                int len = strlen(op_buf);
                int name_len = len - 5;
                char name_buf[4];
                for (int j = 0; j < name_len; ++j) {
                    name_buf[j] = op_buf[4 + j];
                }
                name_buf[name_len] = '\0';
                root = merge(new_node(name_buf), root);
            } else {
                // ROTATE
                int L = get_size(root);
                int k = (K < L) ? K : L;
                if (k > 1) {
                    int T1, T2, T1_1, T1_2;
                    split(root, k, T1, T2);
                    split(T1, 1, T1_1, T1_2);
                    // The rotation: (a1, a2, ..., ak) becomes (a2, ..., ak, a1)
                    root = merge(merge(T1_2, T1_1), T2);
                }
            }
        }
    }

    // Final state of the pile
    print_treap(root);

    return 0;
}
