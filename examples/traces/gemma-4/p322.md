#include <iostream>
#include <vector>
#include <algorithm>
#include <set>

using namespace std;

/**
 * Problem Analysis:
 * The goal is to make two spanning trees, Berland (T_B) and Beerland (T_L), identical.
 * An operation in one country: add an edge, creating a cycle, and then remove an edge from that cycle.
 * Let E_B and E_L be the edge sets of the two trees.
 * The minimum number of months (operations) is Q = |E_B \ E_L| = |E_L \ E_B|.
 * We can achieve this by only performing operations in Beerland to make T_L equal to T_B.
 * For each edge (u, v) in E_B but not in E_L:
 * 1. Add (u, v) to T_L.
 * 2. This creates a cycle in the current T_L.
 * 3. Find an edge in this cycle that belongs to E_L \ E_B.
 * 4. Remove that edge from T_L.
 * This keeps T_L as a spanning tree and eventually makes T_L = T_B.
 */

struct Edge {
    int u, v;
};

const int MAXN = 2005;
Edge edges[2 * MAXN];
bool edge_active[2 * MAXN];
bool is_L_minus_B[2 * MAXN];
bool is_B_minus_L[2 * MAXN];
vector<int> adj[MAXN];

// DFS to find a path between u and target in the current Beerland tree
bool dfs(int u, int target, int parent_node, vector<int>& path) {
    if (u == target) return true;
    for (int idx : adj[u]) {
        if (!edge_active[idx]) continue;
        int v = (edges[idx].u == u) ? edges[idx].v : edges[idx].u;
        if (v == parent_node) continue;
        path.push_back(idx);
        if (dfs(v, target, u, path)) return true;
        path.pop_back();
    }
    return false;
}

struct Op {
    int p, a1, b1, a2, b2;
};

int main() {
    // Faster I/O
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    if (N <= 1) {
        cout << 0 << endl;
        return 0;
    }

    set<pair<int, int>> setB, setL;
    for (int i = 0; i < N - 1; ++i) {
        int u, v;
        cin >> u >> v;
        if (u > v) swap(u, v);
        edges[i] = {u, v};
        setB.insert({u, v});
    }
    for (int i = N - 1; i < 2 * N - 1; ++i) {
        int u, v;
        cin >> u >> v;
        if (u > v) swap(u, v);
        edges[i] = {u, v};
        setL.insert({u, v});
    }

    for (int i = 0; i < N - 1; ++i) {
        if (setL.find({edges[i].u, edges[i].v}) == setL.end()) {
            is_B_minus_L[i] = true;
        }
    }
    for (int i = N - 1; i < 2 * N - 1; ++i) {
        if (setB.find({edges[i].u, edges[i].v}) == setB.end()) {
            is_L_minus_B[i] = true;
        }
        edge_active[i] = true;
        adj[edges[i].u].push_back(i);
        adj[edges[i].v].push_back(i);
    }

    vector<Op> operations;
    for (int i = 0; i < N - 1; ++i) {
        if (is_B_minus_L[i]) {
            int u = edges[i].u;
            int v = edges[i].v;
            vector<int> path;
            // Find the cycle in the current T_L by finding the path between u and v
            if (dfs(u, v, -1, path)) {
                int idx_to_remove = -1;
                for (int idx : path) {
                    if (is_L_minus_B[idx]) {
                        idx_to_remove = idx;
                        break;
                    }
                }
                
                // Even if we can't find an edge in is_L_minus_B, we found a cycle.
                // However, our mathematical proof guarantees idx_to_remove will be found.
                if (idx_to_remove != -1) {
                    operations.push_back({2, u, v, edges[idx_to_remove].u, edges[idx_to_remove].v});
                    edge_active[idx_to_remove] = false;
                    is_L_minus_B[idx_to_remove] = false;
                    
                    edge_active[i] = true;
                    is_L_minus_B[i] = false;
                    adj[edges[i].u].push_back(i);
                    adj[edges[i].v].push_back(i);
                }
            }
        }
    }

    cout << operations.size() << "\n";
    for (const auto& op : operations) {
        cout << op.p << " " << op.a1 << " " << op.b1 << " " << op.a2 << " " << op.b2 << "\n";
    }

    return 0;
}
