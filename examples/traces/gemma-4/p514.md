#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>

using namespace std;

/**
 * Problem: Find the infimum of the areas of all simple polygons with given side lengths.
 * 
 * For n=3, the area is fixed and can be calculated using Heron's formula.
 * For n>=4, the infimum can be thought of as the area of a "nearly flat" polygon.
 * A nearly flat polygon can be thought of as having n-2 sides almost collinear on the x-axis,
 * and two sides (say k and l) that provide the necessary y-displacement to form a closed loop.
 * The minimum area is achieved by minimizing the area of a triangle whose sides are 
 * a_k, a_l, and |S|, where S is the net x-displacement of the remaining n-2 sides.
 * 
 * Since we want to minimize this area, we iterate through all possible pairs of sides 
 * (k, l) that will have non-zero y-displacements, and all possible combinations 
 * of signs for the other n-2 sides to find the minimum area.
 */

int main() {
    int n;
    if (!(cin >> n)) return 0;

    vector<double> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }

    if (n == 3) {
        // For n=3, the area is fixed.
        double s = (a[0] + a[1] + a[2]) / 2.0;
        double area_sq = s * (s - a[0]) * (s - a[1]) * (s - a[2]);
        // area_sq should be positive according to problem constraints.
        cout << fixed << setprecision(10) << sqrt(max(0.0, area_sq)) << endl;
    } else {
        double min_area = 1e18;

        // Iterate over all pairs of sides (k, l) to be the "non-flat" sides.
        for (int k = 0; k < n; ++k) {
            for (int l = k + 1; l < n; ++l) {
                // Collect the other n-2 sides.
                vector<double> others;
                for (int i = 0; i < n; ++i) {
                    if (i != k && i != l) {
                        others.push_back(a[i]);
                    }
                }

                int m = others.size();
                // Iterate through all possible combinations of directions for the other sides.
                for (int mask = 0; mask < (1 << m); ++mask) {
                    double S = 0;
                    for (int i = 0; i < m; ++i) {
                        if ((mask >> i) & 1) {
                            S += others[i];
                        } else {
                            S -= others[i];
                        }
                    }

                    // The sides a_k, a_l and |S| form a triangle.
                    // For a simple polygon to exist in the limit, the triangle must be valid.
                    double s1 = a[k], s2 = a[l], s3 = abs(S);
                    
                    // Check the triangle inequality for the "degenerate" triangle.
                    if (s3 >= abs(s1 - s2) - 1e-9 && s3 <= (s1 + s2) + 1e-9) {
                        double p = (s1 + s2 + s3) / 2.0;
                        double area_sq = p * (p - s1) * (p - s2) * (p - s3);
                        
                        if (area_sq < 0) area_sq = 0;
                        min_area = min(min_area, sqrt(area_sq));
                    }
                }
            }
        }
        cout << fixed << setprecision(10) << min_area << endl;
    }

    return 0;
}
