```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>

using namespace std;

/**
 * Problem analysis:
 * Two members i and j hate each other if:
 * (Si <= Sj AND Bi >= Bj) OR (Si >= Sj AND Bi <= Bj)
 * 
 * This means for them NOT to hate each other, they must satisfy:
 * (Si < Sj AND Bi < Bj) OR (Si > Sj AND Bi > Bj)
 * 
 * We want to find the maximum size of a set where no two members hate each other.
 * This is equivalent to finding the longest chain in the partial order (Si, Bi) < (Sj, Bj)
 * where (Si, Bi) < (Sj, Bj) iff Si < Sj and Bi < Bj.
 * 
 * This is the Longest Increasing Subsequence problem in 2D.
 * 1. Sort members by strength (S) in ascending order.
 * 2. If strengths are equal, sort by beauty (B) in descending order.
 *    This ensures that if Si = Sj, we can pick at most one of them because an 
 *    increasing subsequence in the B values will not pick two B values that are decreasing.
 * 3. Find the LIS of the B values.
 */

struct Member {
    int s, b, id;
};

// Custom comparator for sorting the members.
bool compareMembers(const Member& a, const Member& b) {
    if (a.s != b.s) return a.s < b.s;
    return a.b > b.b;
}

const int MAXN = 100005;
Member members[MAXN];
int B[MAXN];
int pos[MAXN];
int parent[MAXN];
int tails_indices[MAXN];

int main() {
    int N;
    if (scanf("%d", &N) != 1) return 0;
    for (int i = 0; i < N; ++i) {
        scanf("%d %d", &members[i].s, &members[i].b);
        members[i].id = i + 1;
    }

    // Step 1: Sort by S ascending, then by B descending.
    sort(members, members + N, compareMembers);

    // Step 2: Prepare beauty values for LIS.
    for (int i = 0; i < N; ++i) {
        B[i] = members[i].b;
    }

    // Step 3: Standard O(N log N) LIS algorithm.
    int tails_len = 0;
    for (int i = 0; i < N; ++i) {
        // Binary search to find the position where B[i] would be placed in the LIS.
        // We are looking for the first index p such that B[tails_indices[p]] >= B[i].
        int low = 0, high = tails_len - 1, p = tails_len;
        while (low <= high) {
            int mid = low + (high - low) / 2;
            if (B[tails_indices[mid]] >= B[i]) {
                p = mid;
                high = mid - 1;
            } else {
                low = mid + 1;
            }
        }

        // To reconstruct the LIS, record the predecessor of current element.
        if (p > 0) {
            parent[i] = tails_indices[p - 1];
        } else {
            parent[i] = -1;
        }

        // Update tails_indices with the current index.
        if (p == tails_len) {
            tails_indices[tails_len++] = i;
        } else {
            tails_indices[p] = i;
        }
        // Store the length of the LIS ending at index i.
        pos[i] = p + 1;
    }

    // Step 4: Output the result.
    printf("%d\n", tails_len);
    
    // Reconstruct the LIS by backtracking using the parent array.
    vector<int> result;
    int cur = -1;
    // Start from the last element added to an LIS of maximum length.
    for (int i = N - 1; i >= 0; --i) {
        if (pos[i] == tails_len) {
            cur = i;
            break;
        }
    }

    while (cur != -1) {
        result.push_back(members[cur].id);
        cur = parent[cur];
    }

    // Print the member IDs.
    for (int i = 0; i < (int)result.size(); ++i) {
        printf("%d%c", result[i], (i == (int)result.size() - 1 ? '\n' : ' '));
    }

    return 0;
}
```
