```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Problem: Playing with matches (Impartial Game, Misere Play)
 * The game is played with N matches. Possible moves: take 1, P1, P2, ..., Pm matches.
 * The player who takes the last match loses.
 * 
 * Misere play analysis for a single heap:
 * Let f(n) be the state of the game with n matches.
 * f(n) = W (Winning) if there is a move to an L (Losing) state.
 * f(n) = L (Losing) if all possible moves lead to a W state.
 * Since the player who takes the last match loses, we can define:
 * f(0) = W (the player who is faced with 0 matches has already won 
 * because the other player took the last match).
 * 
 * For a single heap, f(n) = 1 (W) if there exists x in S such that f(n-x) = 0 (L),
 * otherwise f(n) = 0 (L).
 * S = {1, P1, P2, ..., Pm}.
 * 
 * Since max(x) = 9, the state at n is determined by (f[n-1], f[n-2], ..., f[n-9]).
 * The sequence of W/L states must be periodic with a period at most 2^9 = 512.
 */

void solve() {
    long long N;
    int M;
    if (!(cin >> N >> M)) return;
    
    vector<int> S;
    S.push_back(1);
    for (int i = 0; i < M; ++i) {
        int p;
        cin >> p;
        S.push_back(p);
    }
    
    // Remove duplicates and sort S to simplify the logic
    sort(S.begin(), S.end());
    S.erase(unique(S.begin(), S.end()), S.end());

    vector<bool> f;
    f.push_back(true); // f(0) = W (true)

    // State consists of the last 9 results (up to 9 possible moves)
    // We use a bitmask to represent the state and an array to find the period.
    vector<int> seen(1 << 9, -1);

    for (int n = 1; (long long)n <= N; ++n) {
        bool can_to_L = false;
        for (int x : S) {
            if (x <= n && !f[n - x]) {
                can_to_L = true;
                break;
            }
        }
        f.push_back(can_to_L);

        // Once we have at least 9 elements, we check for a cycle
        if (n >= 8) {
            int state = 0;
            for (int i = 0; i < 9; ++i) {
                if (f[n - i]) {
                    state |= (1 << i);
                }
            }
            
            if (seen[state] != -1) {
                int prev_n = seen[state];
                int L = n - prev_n;
                
                // Period found. If N is beyond the current index, calculate the index in the period.
                if (N > (long long)n) {
                    long long res_idx = (long long)prev_n + 1 + (N - ((long long)prev_n + 1)) % L;
                    if (f[res_idx]) cout << "FIRST PLAYER MUST WIN" << endl;
                    else cout << "SECOND PLAYER MUST WIN" << endl;
                } else {
                    if (f[N]) cout << "FIRST PLAYER MUST WIN" << endl;
                    else cout << "SECOND PLAYER MUST WIN" << endl;
                }
                return;
            }
            seen[state] = n;
        }
    }

    // If no period was found or N was small, simply output the result for f[N]
    if (f[N]) cout << "FIRST PLAYER MUST WIN" << endl;
    else cout << "SECOND PLAYER MUST WIN" << endl;
}

int main() {
    // Speed up I/O for competitive programming
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int K;
    if (!(cin >> K)) return 0;
    while (K--) {
        solve();
    }
    return 0;
}
```
