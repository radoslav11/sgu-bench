#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <map>

using namespace std;

/**
 * Problem Analysis:
 * We want to represent A as a sum of integer powers of B: A = sum_{i=1}^n s_i B^{k_i},
 * where s_i is 1 or -1, and n is minimized.
 * This is equivalent to finding a base-B representation A = sum_{i=0}^m c_i B^i
 * where we minimize the total number of terms n = sum_{i=0}^m |c_i|.
 * 
 * The recursive formula to find the minimum cost f(A) is:
 * f(A) = 0 if A = 0
 * f(A) = min(d + f(A/B), (B-d) + f(A/B + 1)) where d = A mod B.
 * 
 * Given that A can be very large (3000 digits), we use a BigInt representation.
 * For a number with 3000 digits and a base B >= 2, the number of levels in recursion is at most ~10000.
 * At each level, there are at most two possible values of A, making the total number of distinct 
 * BigInts visited around 20,000.
 */

const int BASE = 1000000000;

typedef vector<int> BigIntVec;

// Convert string to vector of digits in base 10^9
BigIntVec stringToBigInt(const string& s) {
    BigIntVec digits;
    int n = (int)s.size();
    for (int i = n; i > 0; i -= 9) {
        int start = max(0, i - 9);
        int len = i - start;
        digits.push_back(stoi(s.substr(start, len)));
    }
    while (digits.size() > 1 && digits.back() == 0)
        digits.pop_back();
    if (digits.empty()) digits.push_back(0);
    return digits;
}

// Return the quotient and remainder of BigIntVec divided by int b
pair<BigIntVec, int> div_mod(const BigIntVec& a, int b) {
    BigIntVec res(a.size());
    long long rem = 0;
    for (int i = (int)a.size() - 1; i >= 0; i--) {
        long long cur = (long long)a[i] + rem * BASE;
        res[i] = (int)(cur / b);
        rem = cur % b;
    }
    int last = (int)res.size() - 1;
    while (last > 0 && res[last] == 0) {
        last--;
    }
    res.resize(last + 1);
    return {res, (int)rem};
}

// Increment a BigIntVec by 1
BigIntVec addOne(BigIntVec a) {
    int carry = 1;
    for (int i = 0; i < (int)a.size() && carry; i++) {
        long long cur = (long long)a[i] + carry;
        a[i] = (int)(cur % BASE);
        carry = (int)(cur / BASE);
    }
    if (carry) a.push_back(carry);
    return a;
}

// Check if BigIntVec represents 0
bool isZero(const BigIntVec& a) {
    return a.empty() || (a.size() == 1 && a[0] == 0);
}

map<BigIntVec, long long> memo;
int B_val;

// Recursive function with memoization to find the minimal number of terms
long long solve(BigIntVec a) {
    if (isZero(a)) return 0;
    auto it = memo.find(a);
    if (it != memo.end()) return it->second;

    pair<BigIntVec, int> dm = div_mod(a, B_val);
    BigIntVec q = dm.first;
    int d = dm.second;

    // We consider two choices: using d terms of B^0 or using (B-d) terms of B^0 and carrying 1 to the next power
    long long res = (long long)d + solve(q);
    long long res2 = (long long)(B_val - d) + solve(addOne(q));
    
    return memo[a] = min(res, res2);
}

int main() {
    // Faster standard I/O
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string a_str;
    int b;
    if (!(cin >> a_str >> b)) return 0;

    // Edge case B=1
    if (b == 1) {
        cout << a_str << endl;
        return 0;
    }

    B_val = b;
    BigIntVec a = stringToBigInt(a_str);
    
    // Calculate and output the minimum number of terms
    cout << solve(a) << endl;

    return 0;
}
