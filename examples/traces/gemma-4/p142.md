#include <iostream>
#include <string>
#include <vector>

using namespace std;

/**
 * Problem Analysis:
 * We are looking for the shortest non-empty string (made of 'a' and 'b') that is 
 * not a contiguous substring of the given key-string S.
 * The key-string length N is up to 500,000.
 * The shortest missing substring length L will be at most around log2(N) + 1.
 * For N = 500,000, log2(N) is approx 18.9, so L will be at most 19 or 20.
 * 
 * Complexity:
 * Time Complexity: O(N * L_max), where L_max is approximately 20.
 * For N=500,000, this is about 10^7 operations, which should comfortably fit 
 * within the 0.5s time limit in C++.
 * 
 * Space Complexity:
 * The key-string takes ~500,000 bytes (0.5 MB).
 * The 'present' boolean array (using vector<char>) for L=20 takes 2^20 bytes (1 MB).
 * This is well within the 16 MB memory limit.
 */

int main() {
    // Optimization for fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    if (!(cin >> N)) return 0;

    string S;
    cin >> S;

    // We test lengths L = 1, 2, 3, ... until we find a missing substring.
    // The maximum L we need to check for N=500,000 is at most 20.
    for (int L = 1; L <= 20; ++L) {
        int num_possible = 1 << L;
        
        // We use a vector of char instead of vector<bool> for potentially faster 
        // access, although vector<bool> is more memory efficient. 
        // 1 MB for 2^20 chars is well within 16 MB.
        vector<char> present(num_possible, 0);
        int count = 0;

        if (N >= L) {
            int current_val = 0;
            // Calculate the integer representation of the first substring of length L.
            // 'a' is treated as 0 and 'b' as 1.
            for (int i = 0; i < L; ++i) {
                current_val = (current_val << 1) | (S[i] == 'b' ? 1 : 0);
            }
            if (!present[current_val]) {
                present[current_val] = 1;
                count++;
            }

            // Use a rolling window to efficiently find all other substrings of length L.
            int mask = (1 << L) - 1;
            for (int i = L; i < N; ++i) {
                current_val = ((current_val << 1) | (S[i] == 'b' ? 1 : 0)) & mask;
                if (!present[current_val]) {
                    present[current_val] = 1;
                    count++;
                }
            }
        }

        // If the number of distinct substrings found is less than the total 
        // possible substrings of length L (2^L), then L is the length of the keyword.
        if (count < num_possible) {
            cout << L << "\n";
            // Find the lexicographically smallest missing integer representation.
            for (int i = 0; i < num_possible; ++i) {
                if (!present[i]) {
                    // Convert the integer back into its string representation.
                    string keyword = "";
                    for (int j = L - 1; j >= 0; --j) {
                        keyword += ((i >> j) & 1) ? 'b' : 'a';
                    }
                    cout << keyword << "\n";
                    return 0;
                }
            }
        }
    }

    return 0;
}
