#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>

using namespace std;

/**
 * Problem: Random Shooting
 * Goal: Find the probability that the segment between two independently uniformly 
 * distributed points in a 100x100 square intersects a convex polygon target.
 * 
 * Approach: Use the kinematic measure of lines.
 * The probability is P = (1/A^2) * integral_{0}^{pi} integral_{p_min}^{p_max} M(p, theta) dp dtheta,
 * where M(p, theta) is the integral over the line L(p, theta) of |x-y| for points x, y on the line 
 * such that the segment [x,y] intersects the polygon.
 * 
 * M(p, theta) = (1/3) * (l_S^3 - l_1^3 - l_2^3), where l_S is the length of the segment of the line within the board,
 * and l_1, l_2 are the lengths of the two parts of that segment that are outside the polygon intersection.
 */

struct Point {
    double x, y;
};

int N;
vector<Point> poly;

// Function to calculate M(L) for a given line parameterized by (p, theta)
double M_function(double p, double theta) {
    double cos_theta = cos(theta);
    double sin_theta = sin(theta);
    // Parameterize the line by x = x0 + t * dx, y = y0 + t * dy
    double dx = -sin_theta;
    double dy = cos_theta;

    double x0, y0;
    // Find an initial point (x0, y0) on the line x*cos(theta) + y*sin(theta) = p
    if (abs(cos_theta) > abs(sin_theta)) {
        x0 = p / cos_theta;
        y0 = 0;
    } else {
        x0 = 0;
        y0 = p / sin_theta;
    }

    // Lambda to find intersection of line and a segment (p1, p2)
    auto get_t_values = [&](Point p1, Point p2) {
        vector<double> ts;
        double dx_edge = p2.x - p1.x;
        double dy_edge = p2.y - p1.y;
        // Solve the system:
        // u * dx_edge - t * dx = x0 - p1.x
        // u * dy_edge - t * dy = y0 - p1.y
        // Using Cramer's rule:
        double D = -dx_edge * dy + dy_edge * dx;
        if (abs(D) > 1e-10) {
            double Du = -(x0 - p1.x) * dy + (y0 - p1.y) * dx;
            double Dt = dx_edge * (y0 - p1.y) - dy_edge * (x0 - p1.x);
            double u = Du / D;
            double t = Dt / D;
            if (u >= -1e-10 && u <= 1.0 + 1e-10) {
                ts.push_back(t);
            }
        }
        return ts;
    };

    // Find intersection points with the polygon edges
    vector<double> poly_ts;
    for (int i = 0; i < N; ++i) {
        vector<double> ts = get_t_values(poly[i], poly[(i + 1) % N]);
        for (double t : ts) poly_ts.push_back(t);
    }

    // If no intersection, the segment cannot intersect the polygon
    if (poly_ts.empty()) return 0.0;
    sort(poly_ts.begin(), poly_ts.end());
    double t_poly_min = poly_ts.front();
    double t_poly_max = poly_ts.back();

    // Find intersection points with the square edges (board boundaries)
    vector<double> sq_ts;
    Point sq_pts[4] = {{0, 0}, {100, 0}, {100, 100}, {0, 100}};
    for (int i = 0; i < 4; ++i) {
        vector<double> ts = get_t_values(sq_pts[i], sq_pts[(i + 1) % 4]);
        for (double t : ts) sq_ts.push_back(t);
    }

    if (sq_ts.empty()) return 0.0;
    sort(sq_ts.begin(), sq_ts.end());
    double t_sq_min = sq_ts.front();
    double t_sq_max = sq_ts.back();

    // Since the target is strictly inside the board: t_sq_min <= t_poly_min < t_poly_max <= t_sq_max
    double lp = t_poly_max - t_poly_min;
    double l1 = t_poly_min - t_sq_min;
    double l2 = t_sq_max - t_poly_max;
    double ls = t_sq_max - t_sq_min;

    // Use max(0.0, ...) to avoid negative values due to tiny precision errors
    lp = max(0.0, lp);
    l1 = max(0.0, l1);
    l2 = max(0.0, l2);
    ls = max(0.0, ls);

    return (ls * ls * ls - l1 * l1 * l1 - l2 * l2 * l2) / 3.0;
}

// Simpson's rule for the p-integral
double simpson_p(double theta, double p_min, double p_max, int Mp) {
    if (p_min >= p_max) return 0.0;
    double sum = 0;
    double h = (p_max - p_min) / Mp;
    for (int i = 0; i <= Mp; ++i) {
        double p = p_min + i * h;
        double weight = (i == 0 || i == Mp) ? 1 : (i % 2 == 1 ? 4 : 2);
        sum += weight * M_function(p, theta);
    }
    return sum * h / 3.0;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    if (!(cin >> N)) return 0;
    poly.resize(N);
    for (int i = 0; i < N; ++i) {
        cin >> poly[i].x >> poly[i].y;
    }

    // Using Simpson's rule for both theta and p integration
    // Higher Mp/Mtheta increases precision but also computation time.
    // 200x200 steps fits comfortably in the 0.25s limit.
    int Mtheta = 200;
    int Mp = 200;
    double total_integral = 0;
    double h_theta = M_PI / Mtheta;

    for (int i = 0; i <= Mtheta; ++i) {
        double theta = i * h_theta;
        double cos_theta = cos(theta);
        double sin_theta = sin(theta);

        // Determine the range of p where the line intersects the polygon
        double p_min = 1e18, p_max = -1e18;
        for (int j = 0; j < N; ++j) {
            double p = poly[j].x * cos_theta + poly[j].y * sin_theta;
            p_min = min(p_min, p);
            p_max = max(p_max, p);
        }

        double p_integral = simpson_p(theta, p_min, p_max, Mp);
        double weight = (i == 0 || i == Mtheta) ? 1 : (i % 2 == 1 ? 4 : 2);
        total_integral += weight * p_integral;
    }

    total_integral = total_integral * (h_theta / 3.0);
    // Probability = integral / (Board Area)^2
    double probability = total_integral / (10000.0 * 10000.0);

    cout << fixed << setprecision(10) << probability << endl;

    return 0;
}
