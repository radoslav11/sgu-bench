#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>

using namespace std;

typedef long long ll;

// Function to compute (base^exp) % mod
ll power(ll base, ll exp, ll mod) {
    ll res = 1;
    base %= mod;
    while (exp > 0) {
        if (exp % 2 == 1) res = (res * base) % mod;
        base = (base * base) % mod;
        exp /= 2;
    }
    return res;
}

// Extended Euclidean Algorithm to find modular inverse
ll extended_gcd(ll a, ll b, ll &x, ll &y) {
    if (a == 0) {
        x = 0; y = 1;
        return b;
    }
    ll x1, y1;
    ll gcd = extended_gcd(b % a, a, x1, y1);
    x = y1 - (b / a) * x1;
    y = x1;
    return gcd;
}

ll modInverse(ll a, ll m) {
    ll x, y;
    ll g = extended_gcd(a, m, x, y);
    if (g != 1) return -1;
    return (x % m + m) % m;
}

// Find a primitive root modulo p
ll findPrimitiveRoot(ll p) {
    if (p == 2) return 1;
    vector<ll> factors;
    ll n = p - 1;
    for (ll i = 2; i * i <= n; ++i) {
        if (n % i == 0) {
            factors.push_back(i);
            while (n % i == 0) n /= i;
        }
    }
    if (n > 1) factors.push_back(n);

    for (ll res = 2; res < p; ++res) {
        bool ok = true;
        for (ll f : factors) {
            if (power(res, (p - 1) / f, p) == 1) {
                ok = false;
                break;
            }
        }
        if (ok) return res;
    }
    return -1;
}

bool is_in_S[1000001];
int index_in_S[1000001];

void solve() {
    int n_in, p_in;
    if (!(cin >> n_in >> p_in)) return;
    ll n = n_in;
    ll p = p_in;

    if (p == 2) {
        cout << -1 << endl;
        return;
    }

    ll d = std::gcd(n, p - 1);
    ll g = findPrimitiveRoot(p);
    
    // S_d = { (g^d)^k mod p | k = 0, ..., (p-1)/d - 1 }
    ll size_S = (p - 1) / d;
    vector<int> used_indices;
    ll step = power(g, d, p);
    ll u = 1;
    for (int k = 0; k < size_S; ++k) {
        is_in_S[u] = true;
        index_in_S[u] = k;
        used_indices.push_back(u);
        u = (u * step) % p;
    }

    int ans_x = -1, ans_y = -1, ans_z = -1;

    for (int val_u : used_indices) {
        if (val_u == 1) continue; // skip u=1 because 1-1=0, and 0 is not in S_d
        ll val_v = (1 - val_u + p) % p;
        if (val_v != 0 && is_in_S[val_v]) {
            // Found u, v in S_d such that u + v = 1 (mod p)
            // u = (g^d)^k, v = (g^d)^m.
            // Then x^n = u and y^n = v, with z=1 works because x^n + y^n = u + v = 1 = 1^n
            // x = g^m where m satisfies m * (n/d) = k (mod (p-1)/d)
            ll k = index_in_S[val_u];
            ll m = index_in_S[val_v];
            
            ll n_prime = n / d;
            ll p_prime = (p - 1) / d;
            ll inv_n_prime = modInverse(n_prime, p_prime);
            
            ll x_exp = (k * inv_n_prime) % p_prime;
            ll y_exp = (m * inv_n_prime) % p_prime;
            
            ans_x = power(g, x_exp, p);
            ans_y = power(g, y_exp, p);
            ans_z = 1;
            break;
        }
    }

    if (ans_x != -1) {
        cout << ans_x << " " << ans_y << " " << ans_z << endl;
    } else {
        cout << -1 << endl;
    }

    // Cleanup the is_in_S array for the next testcase
    for (int val : used_indices) {
        is_in_S[val] = false;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t;
    if (!(cin >> t)) return 0;
    while (t--) {
        solve();
    }
    return 0;
}
