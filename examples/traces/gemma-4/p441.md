```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Problem Analysis:
 * N distinct photo albums are distributed among K friends.
 * Each friend must receive at least one photo album.
 * "Two distributions are considered the same if they differ only by order of albums 
 * in the gifts or by the persons receiving gifts."
 * This means:
 * 1. The order of albums within a gift doesn't matter (each gift is a set).
 * 2. The order of friends doesn't matter (the friends are indistinguishable).
 * 
 * This is the definition of Stirling numbers of the second kind, S(N, K), 
 * which counts the number of ways to partition a set of N elements into K non-empty 
 * unlabelled subsets.
 * 
 * The recurrence relation for S(n, k) is:
 * S(n, k) = k * S(n-1, k) + S(n-1, k-1)
 * Base cases:
 * S(n, n) = 1, S(n, 1) = 1, S(n, k) = 0 if k > n or k = 0 (for n > 0).
 * 
 * Given the tight time limit (0.25s) and a small modulus (2007), 
 * an O(NK) dynamic programming approach should be efficient enough.
 */

int main() {
    // Speed up I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long long N, K;
    // Reading input N and K
    if (!(cin >> N >> K)) return 0;

    // Basic constraints: if K > N, it's impossible to give at least one album to each friend.
    // If K <= 0, the problem setup is invalid for positive N.
    if (K <= 0 || K > N) {
        cout << 0 << endl;
        return 0;
    }

    // Special case for K=1
    if (K == 1) {
        cout << 1 % 2007 << endl;
        return 0;
    }

    // We'll use a 1D DP array to compute S(N, K) to save space.
    // dp[j] will store the value of S(i, j) for the current i.
    // The formula S(i, j) = j * S(i-1, j) + S(i-1, j-1) requires 
    // the values from the previous row (i-1). 
    // To update the row in-place, we iterate backwards from j=K down to 1.
    vector<int> dp(K + 1, 0);
    
    // Base case for i = 1: S(1, 1) = 1, others are 0.
    dp[1] = 1;

    // Iterate from i = 2 to N
    for (int i = 2; i <= (int)min(N, (long long)2000000); ++i) { // Using a practical limit if N is huge
        // We only need to compute up to j = min(i, K)
        int limit = (i < K) ? (int)i : (int)K;
        
        // Iterate backwards to use values from the previous i-1 iteration
        for (int j = limit; j >= 1; --j) {
            // S(i, j) = j * S(i-1, j) + S(i-1, j-1)
            // Since j can be up to 10^6 and dp[j] up to 2006, 
            // j * dp[j] can reach 2*10^9, which fits in a signed 32-bit integer.
            // Using 1LL to promote to long long during multiplication to ensure no overflow.
            long long term1 = (long long)j * dp[j];
            long long term2 = (j > 1) ? dp[j - 1] : 0;
            dp[j] = (int)((term1 + term2) % 2007);
        }
        
        // If N is extremely large, we'd need a different strategy like periodicity,
        // but O(NK) is standard for these constraints.
        if (i >= N) break;
    }

    // The answer is S(N, K) modulo 2007.
    cout << dp[K] << endl;

    return 0;
}
```
