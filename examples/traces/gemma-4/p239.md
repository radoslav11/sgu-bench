/**
 * Problem analysis:
 * A Minesweeper grid of size N x 2 is given.
 * The second column contains no mines and is fully opened.
 * The first column may contain mines and is not opened.
 * For each cell (i, 1) in the second column, the number a_i represents the number of mines
 * in its neighboring cells in the first column.
 * The neighbors of (i, 1) in the first column are (i-1, 0), (i, 0), and (i+1, 0).
 * Let x_i = 1 if there's a mine at (i, 0), and x_i = 0 otherwise.
 * The equations are:
 * For N = 1: a_0 = x_0
 * For N >= 2:
 *   a_0 = x_0 + x_1
 *   a_i = x_{i-1} + x_i + x_{i+1} for 0 < i < N-1
 *   a_{N-1} = x_{N-2} + x_{N-1}
 *
 * We need to find the number of valid binary sequences (x_0, ..., x_{N-1}).
 * Since N can be up to 1000, the answer can be as large as 2^1000.
 * We'll use dynamic programming with a custom BigInt class for large numbers.
 */

const long long BASE = 1e9;

struct BigInt {
    long long digits[45];
    int len;

    BigInt(long long v = 0) {
        for (int i = 0; i < 45; ++i) digits[i] = 0;
        if (v == 0) {
            len = 1;
        } else {
            len = 0;
            long long temp = v;
            while (temp > 0) {
                digits[len++] = temp % BASE;
                temp /= BASE;
            }
        }
    }

    void add(const BigInt& other) {
        int max_len = max(len, other.len);
        long long carry = 0;
        for (int i = 0; i < max_len || carry; ++i) {
            long long other_digit = (i < other.len ? other.digits[i] : 0);
            digits[i] += other_digit + carry;
            carry = digits[i] / BASE;
            digits[i] %= BASE;
            if (i >= len) len = i + 1;
        }
        while (len > 1 && digits[len - 1] == 0) len--;
    }

    void print() const {
        bool all_zero = true;
        for (int i = 0; i < len; ++i) {
            if (digits[i] != 0) {
                all_zero = false;
                break;
            }
        }
        if (all_zero) {
            cout << 0 << endl;
        } else {
            cout << digits[len - 1];
            for (int i = len - 2; i >= 0; --i) {
                cout << setfill('0') << setw(9) << digits[i];
            }
            cout << endl;
        }
    }
};
