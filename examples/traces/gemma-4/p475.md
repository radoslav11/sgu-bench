```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Problem Analysis:
 * - n participants (n <= 10), m riffles (m <= 1000).
 * - Each participant j has weight wj, walking time tj, and get-on/off time sj.
 * - Each riffle i has critical weight ci, capsize time Di, and non-capsize time di.
 * - At each riffle i, a non-empty subset of people Sk is on the raft.
 * - People not in Sk walk from pi-1 to pi.
 * - The raft and all walkers must arrive at pi before anyone can change places.
 * - The goal is to find the minimal time to reach the finish pm and have everyone on the bank.
 *
 * Dynamic Programming State:
 * - dp[i][S] = minimal time at point pi, after the raft and all walkers arrive at pi, 
 *   and S is the set of people on the raft during the i-th riffle.
 * 
 * Transitions:
 * - At p0, people in S1 get on the raft (cost sum(s_j for j in S1)).
 * - Raft starts moving at T = sum(s_j for j in S1), walkers start at T = 0.
 * - Arrival at p1: T_arr_p1 = max(sum(s_j for j in S1) + raft_time(S1, 1), max(t_j for j not in S1)).
 * - For i > 1, after arrivals at pi-1 and changes to Si, arrival at pi is:
 *   T_arr_pi = T_arr_p(i-1) + sum(s_j for j in S_{i-1} delta Si) + max(raft_time(Si, i), max(t_j for j not in Si)).
 * - At pm, all people on the raft Sm must get off (cost sum(s_j for j in Sm)).
 *
 * Complexity:
 * - O(m * 2^(2n)) = 1000 * 1024 * 1024 ≈ 10^9 operations.
 * - Given the tight time limit (1.25s) and simple inner loop, it should pass.
 */

const int INF = 1e9 + 7;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<int> w(n), t(n), s(n);
    for (int i = 0; i < n; ++i) {
        cin >> w[i] >> t[i] >> s[i];
    }

    vector<int> c(m), D(m), d(m);
    for (int i = 0; i < m; ++i) {
        cin >> c[i] >> D[i] >> d[i];
    }

    int num_subsets = (1 << n);
    vector<int> subset_w(num_subsets, 0);
    vector<int> subset_s(num_subsets, 0);
    vector<int> max_walking_t(num_subsets, 0);

    for (int i = 0; i < num_subsets; ++i) {
        int weight_sum = 0;
        int s_sum = 0;
        for (int j = 0; j < n; ++j) {
            if ((i >> j) & 1) {
                weight_sum += w[j];
                s_sum += s[j];
            }
        }
        subset_w[i] = weight_sum;
        subset_s[i] = s_sum;
        
        int max_t = 0;
        for (int j = 0; j < n; ++j) {
            if (!((i >> j) & 1)) {
                max_t = max(max_t, t[j]);
            }
        }
        max_walking_t[i] = max_t;
    }

    // max_RTW[i][S] = max(raft_time(S, i), max_walking_t(S))
    // Precomputing this saves time in the main DP loop.
    vector<vector<int>> max_RTW(m, vector<int>(num_subsets));
    for (int i = 0; i < m; ++i) {
        for (int mask = 0; mask < num_subsets; ++mask) {
            int raft_time = (subset_w[mask] <= c[i]) ? d[i] : D[i];
            max_RTW[i][mask] = max(raft_time, max_walking_t[mask]);
        }
    }

    // dp[i][mask] = min arrival time at pi with subset 'mask' on raft for riffle i
    vector<vector<int>> dp(m + 1, vector<int>(num_subsets, INF));

    // Base case: Riffle 1
    for (int mask = 1; mask < num_subsets; ++mask) {
        dp[1][mask] = max(subset_s[mask] + ((subset_w[mask] <= c[0]) ? d[0] : D[0]), max_walking_t[mask]);
    }

    // DP transitions for riffles 2 to m
    for (int i = 2; i <= m; ++i) {
        int r_idx = i - 1;
        for (int mask_prev = 1; mask_prev < num_subsets; ++mask_prev) {
            if (dp[i - 1][mask_prev] == INF) continue;
            
            int current_val = dp[i - 1][mask_prev];
            for (int mask_curr = 1; mask_curr < num_subsets; ++mask_curr) {
                // The change cost is sum of s_j for people in mask_prev XOR mask_curr
                int arrival_time = current_val + subset_s[mask_prev ^ mask_curr] + max_RTW[r_idx][mask_curr];
                if (arrival_time < dp[i][mask_curr]) {
                    dp[i][mask_curr] = arrival_time;
                }
            }
        }
    }

    // Final step: after riffle m, everyone on the raft must get off at pm.
    int min_total_time = INF;
    for (int mask = 1; mask < num_subsets; ++mask) {
        if (dp[m][mask] != INF) {
            min_total_time = min(min_total_time, dp[m][mask] + subset_s[mask]);
        }
    }

    cout << min_total_time << endl;

    return 0;
}
```
