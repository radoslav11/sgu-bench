#include <iostream>
#include <string>
#include <stack>
#include <cctype>

using namespace std;

/**
 * Problem Analysis:
 * The task is to implement a simple markup language parser.
 * There are two types of tags: <UP> and <DOWN>.
 * <UP> makes all characters in its body uppercase.
 * <DOWN> makes all characters in its body lowercase.
 * Tags are correctly nested, meaning they form a correct bracket sequence.
 * If a character is within multiple tags, the innermost tag determines its case.
 * 
 * Algorithm:
 * We'll iterate through the input string character by character.
 * We use a stack to keep track of the currently active tags.
 * When we encounter a '<', we identify the tag and either push its effect
 * onto the stack (for <UP> or <DOWN>) or pop from the stack (for </UP> or </DOWN>).
 * When we encounter a non-tag character, we consult the top of the stack
 * to determine its case (uppercase, lowercase, or no change).
 * 
 * Complexity:
 * Time Complexity: O(N), where N is the length of the input string (N <= 1000).
 * Space Complexity: O(N) for the stack and the result string.
 */

int main() {
    // Optimizing input and output operations
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string s;
    // Reading the entire input string line by line to handle spaces correctly
    if (!getline(cin, s)) {
        return 0;
    }

    // stack<bool> stores the mode of the current innermost tag:
    // true for UP, false for DOWN.
    stack<bool> tagStack;
    string result = "";
    // Pre-allocating space for the result string for efficiency
    result.reserve(s.length());

    int n = s.length();
    for (int i = 0; i < n; ++i) {
        if (s[i] == '<') {
            // Look for the closing '>' of the current tag
            int j = i + 1;
            while (j < n && s[j] != '>') {
                j++;
            }

            // If a closing '>' is found, extract the tag name
            if (j < n) {
                string tag = s.substr(i + 1, j - i - 1);
                if (tag == "UP") {
                    tagStack.push(true);
                } else if (tag == "DOWN") {
                    tagStack.push(false);
                } else if (tag == "/UP") {
                    if (!tagStack.empty()) {
                        tagStack.pop();
                    }
                } else if (tag == "/DOWN") {
                    if (!tagStack.empty()) {
                        tagStack.pop();
                    }
                }
                // Move the loop counter i to the position of '>'
                i = j;
            } else {
                // If no '>' is found, treat the '<' as a normal character
                result += s[i];
            }
        } else {
            // If we are not in a tag, process the character based on the stack top
            if (tagStack.empty()) {
                // No active tag: leave the character as it is
                result += s[i];
            } else {
                // Most inner tag dictates the case
                if (tagStack.top()) {
                    // The innermost tag is UP
                    result += (char)toupper((unsigned char)s[i]);
                } else {
                    // The innermost tag is DOWN
                    result += (char)tolower((unsigned char)s[i]);
                }
            }
        }
    }

    // Output the final processed string
    cout << result << endl;

    return 0;
}
