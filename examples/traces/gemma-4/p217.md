#include <iostream>
#include <cmath>
#include <iomanip>
#include <algorithm>

using namespace std;

/**
 * Problem analysis:
 * Two infinite cylinders with radii R1 and R2, whose axes are perpendicular and intersect.
 * Let the axis of the first cylinder be along the x-axis: y^2 + z^2 <= R1^2.
 * Let the axis of the second cylinder be along the y-axis: x^2 + z^2 <= R2^2.
 * The volume of their intersection is given by the triple integral of 1 over the region.
 * For a fixed z, the cross-section in the xy-plane is a rectangle with sides 
 * 2*sqrt(R1^2 - z^2) and 2*sqrt(R2^2 - z^2).
 * The range for z is -min(R1, R2) <= z <= min(R1, R2).
 * Let Rmin = min(R1, R2) and Rmax = max(R1, R2).
 * Volume V = \int_{-Rmin}^{Rmin} 4 * sqrt(R1^2 - z^2) * sqrt(R2^2 - z^2) dz
 *          = 8 * \int_{0}^{Rmin} sqrt(R1^2 - z^2) * sqrt(R2^2 - z^2) dz
 * 
 * Using the substitution z = Rmin * sin(theta), dz = Rmin * cos(theta) d(theta):
 * V = 8 * \int_{0}^{PI/2} sqrt(Rmin^2 - Rmin^2*sin^2(theta)) * sqrt(Rmax^2 - Rmin^2*sin^2(theta)) * Rmin * cos(theta) d(theta)
 *   = 8 * \int_{0}^{PI/2} (Rmin * cos(theta)) * (Rmax * sqrt(1 - (Rmin/Rmax)^2 * sin^2(theta))) * Rmin * cos(theta) d(theta)
 *   = 8 * Rmin^2 * Rmax * \int_{0}^{PI/2} cos^2(theta) * sqrt(1 - k^2 * sin^2(theta)) d(theta)
 * where k = Rmin / Rmax.
 * 
 * The integrand f(theta) = cos^2(theta) * sqrt(1 - k^2 * sin^2(theta)) is smooth on [0, PI/2].
 * We can use Simpson's Rule for numerical integration to find the volume.
 */

int main() {
    // Optimize standard I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    double R1, R2;
    // Read input R1 and R2
    if (!(cin >> R1 >> R2)) return 0;

    double Rmin = min(R1, R2);
    double Rmax = max(R1, R2);
    double k = Rmin / Rmax;
    double k2 = k * k;

    // Use acos(-1.0) to get the most accurate value of PI
    const double PI = acos(-1.0);
    
    // Number of sub-intervals for Simpson's Rule (must be even)
    const int N = 1000000; 
    double h = (PI / 2.0) / N;

    // Integrand function
    auto g = [&](double theta) {
        double s = sin(theta);
        double c = cos(theta);
        // Using max(0.0, ...) to prevent negative values due to floating-point precision
        return c * c * sqrt(max(0.0, 1.0 - k2 * s * s));
    };

    // Simpson's Rule Implementation
    // V = 8 * Rmin^2 * Rmax * integral
    // integral \approx (h/3) * (f(0) + 4*f(h) + 2*f(2h) + 4*f(3h) + ... + f(N*h))
    double sum = g(0.0) + g(PI / 2.0);
    for (int i = 1; i < N; ++i) {
        double theta = i * h;
        if (i % 2 == 0) {
            sum += 2.0 * g(theta);
        } else {
            sum += 4.0 * g(theta);
        }
    }

    double integral = (h / 3.0) * sum;
    double volume = 8.0 * Rmin * Rmin * Rmax * integral;

    // Output the volume with required precision
    cout << fixed << setprecision(4) << volume << endl;

    return 0;
}
