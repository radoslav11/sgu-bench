#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

// Use a structure for edges in the Dinic's algorithm
struct Edge {
    int to, cap, flow, rev;
    int id; // To identify original edges
    int L; // Lower bound
};

vector<Edge> adj[110];
int level[110], iter[110];

// Dinic's algorithm: Adding an edge
void add_edge(int from, int to, int cap, int id = -1, int L = 0) {
    adj[from].push_back({to, cap, 0, (int)adj[to].size(), id, L});
    adj[to].push_back({from, 0, 0, (int)adj[from].size() - 1, -1, 0});
}

// BFS for level graph
bool bfs(int s, int t) {
    fill(level, level + 110, -1);
    level[s] = 0;
    queue<int> q;
    q.push(s);
    while (!q.empty()) {
        int v = q.front();
        q.pop();
        for (auto &e : adj[v]) {
            if (e.cap - e.flow > 0 && level[e.to] < 0) {
                level[e.to] = level[v] + 1;
                q.push(e.to);
            }
        }
    }
    return level[t] != -1;
}

// DFS for finding augmenting paths
int dfs(int v, int t, int f) {
    if (v == t) return f;
    for (int &i = iter[v]; i < (int)adj[v].size(); ++i) {
        Edge &e = adj[v][i];
        if (e.cap - e.flow > 0 && level[v] < level[e.to]) {
            int d = dfs(e.to, t, min(f, e.cap - e.flow));
            if (d > 0) {
                e.flow += d;
                adj[e.to][e.rev].flow -= d;
                return d;
            }
        }
    }
    return 0;
}

// Dinic's algorithm: Find max flow
int max_flow(int s, int t) {
    int flow = 0;
    while (bfs(s, t)) {
        fill(iter, iter + 110, 0);
        int f;
        while ((f = dfs(s, t, 1e9)) > 0) {
            flow += f;
        }
    }
    return flow;
}
