#include <iostream>
#include <algorithm>

using namespace std;

/**
 * Problem Analysis:
 * Kate's tariff plan "Very Profitable" works as follows:
 * 1. At the very beginning, Kate is charged p1 cents for the first k1 seconds.
 * 2. If her account balance becomes non-positive (N <= 0), the call is automatically dropped.
 * 3. Each of the next k2 seconds costs p2 cents (charged at the end of each second).
 * 4. All remaining time costs p3 cents per second (charged at the end of each second).
 * 
 * Rule interpretation:
 * - If N < p1, she cannot even establish the call (0 seconds).
 * - If N >= p1, she starts the call, pays p1 immediately, and the first k1 seconds are covered.
 * - After k1 seconds, her balance is B = N - p1.
 * - If B <= 0, she can talk for exactly k1 seconds and then the call is dropped.
 * - If B > 0, she can start the (k1+1)-th second.
 * - She continues to complete seconds as long as she can start them (i.e., the balance 
 *   at the end of the previous second was > 0).
 * - For the next k2 seconds, she can complete 's' additional seconds where:
 *   s = min(k2, (B-1)/p2 + 1)
 *   Wait, let's re-verify: if B=5, p2=4, she can complete 2 seconds (4th and 5th).
 *   4th: B becomes 5-4=1. Since 1 > 0, she can start the 5th second.
 *   5th: B becomes 1-4=-3. Since -3 <= 0, she cannot start the 6th second.
 *   Total extra: 2 seconds. (5-1)/4 + 1 = 1 + 1 = 2. Correct.
 * - If she completes all k2 seconds (s = k2) and her remaining balance B_new = B - k2*p2 is > 0,
 *   she can continue with the p3 rate.
 * - The number of additional p3 seconds is:
 *   s3 = (B_new-1)/p3 + 1
 * 
 * Constraints:
 * N, k1, k2, p1, p2, p3 <= 10^9.
 * Total seconds can reach ~3 * 10^9, so we must use long long.
 * The intermediate value k2 * p2 can reach 10^18, so long long is necessary.
 */

int main() {
    // Optimization for competitive programming
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long long N, k1, k2, p1, p2, p3;
    if (!(cin >> N >> k1 >> k2 >> p1 >> p2 >> p3)) return 0;

    // Initial check if the call can be established
    if (N < p1) {
        cout << 0 << endl;
        return 0;
    }

    // After paying p1 for the first k1 seconds
    long long B = N - p1;
    long long T = k1;

    // If balance is still positive after p1, she can start the next phase
    if (B > 0) {
        // Phase 2: Each of the next k2 seconds costs p2 cents
        // Calculate how many seconds in the k2-block she can complete
        long long s = (B - 1) / p2 + 1;
        if (s > k2) {
            s = k2;
        }
        
        T += s;

        // If she managed to complete all k2 seconds, check if she can enter the p3 phase
        if (s == k2) {
            B = B - k2 * p2;
            // If balance is still positive, she can start the p3 phase
            if (B > 0) {
                long long s3 = (B - 1) / p3 + 1;
                T += s3;
            }
        }
    }

    cout << T << endl;

    return 0;
}
