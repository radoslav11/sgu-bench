#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

/**
 * Problem Summary:
 * A hijacker travels from junction X to junction Y using one of the shortest paths.
 * We need to find for each junction K, the number of points P (junctions or points on roads) 
 * such that P is on a shortest path from X to Y and the distance from X to P is equal 
 * to the shortest distance from X to junction K.
 * If junction K is not on any shortest path from X to Y, AK = 0.
 * 
 * Algorithm:
 * 1. Compute shortest distances from X to all junctions (distX) and from Y to all junctions (distY) using Dijkstra.
 * 2. Let D = distX[Y]. A junction K is on a shortest path if distX[K] + distY[K] == D.
 * 3. A point P on a road (u, v) of length L is on a shortest path if it's in a road direction 
 *    u -> v where distX[u] + L + distY[v] == D. For such a road, the points P satisfy 
 *    distX[u] <= distX[P] <= distX[u] + L.
 * 4. For each junction K on a shortest path, AK is the count of:
 *    - Junctions j on a shortest path where distX[j] == distX[K].
 *    - Points P strictly inside a road (u, v) such that distX[u] < distX[K] < distX[u] + L 
 *      and the road is part of a shortest path from X to Y.
 * 5. To efficiently count roads, we can use a difference array on the unique values of distX[K].
 */

const long long INF = 1e18;

struct Edge {
    int to;
    int weight;
};

struct Road {
    int u, v, l;
};

void dijkstra(int n, int start, const vector<vector<Edge>>& adj, vector<long long>& dist) {
    dist.assign(n + 1, INF);
    dist[start] = 0;
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
    pq.push({0, start});

    while (!pq.empty()) {
        long long d = pq.top().first;
        int u = pq.top().second;
        pq.pop();

        if (d > dist[u]) continue;

        for (const auto& edge : adj[u]) {
            if (dist[u] + edge.weight < dist[edge.to]) {
                dist[edge.to] = dist[u] + edge.weight;
                pq.push({dist[edge.to], edge.to});
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, x, y;
    if (!(cin >> n >> x >> y)) return 0;

    int m;
    cin >> m;
    vector<vector<Edge>> adj(n + 1);
    vector<Road> roads(m);
    for (int i = 0; i < m; ++i) {
        cin >> roads[i].u >> roads[i].v >> roads[i].l;
        adj[roads[i].u].push_back({roads[i].v, roads[i].l});
        adj[roads[i].v].push_back({roads[i].u, roads[i].l});
    }

    vector<long long> distX, distY;
    dijkstra(n, x, adj, distX);
    dijkstra(n, y, adj, distY);

    long long D = distX[y];
    if (D == INF) {
        // If no path exists, all AK should be 0.
        for (int i = 1; i <= n; ++i) {
            cout << (i == 1 ? "" : " ") << 0;
        }
        cout << endl;
        return 0;
    }

    vector<int> on_sp_junctions(n + 1, 0);
    vector<long long> sp_dists;
    for (int i = 1; i <= n; ++i) {
        if (distX[i] + distY[i] == D) {
            on_sp_junctions[i] = 1;
            sp_dists.push_back(distX[i]);
        }
    }

    // Prepare to count junctions per distance
    sort(sp_dists.begin(), sp_dists.end());
    sp_dists.erase(unique(sp_dists.begin(), sp_dists.end()), sp_dists.end());

    vector<int> junc_count_at_dist(sp_dists.size(), 0);
    for (int i = 1; i <= n; ++i) {
        if (on_sp_junctions[i]) {
            auto it = lower_bound(sp_dists.begin(), sp_dists.end(), distX[i]);
            if (it != sp_dists.end() && *it == distX[i]) {
                junc_count_at_dist[distance(sp_dists.begin(), it)]++;
            }
        }
    }

    // Prepare to count roads covering a specific distance
    vector<int> road_diff(sp_dists.size() + 1, 0);
    for (int i = 0; i < m; ++i) {
        int u = roads[i].u;
        int v = roads[i].v;
        int l = roads[i].l;

        // A road is on the shortest path if distX[u] + l + distY[v] == D or vice-versa
        if (distX[u] + l + distY[v] == D) {
            // Directed road u -> v
            long long d_start = distX[u];
            long long d_end = distX[u] + l;
            // Find range of indices in sp_dists such that d_start < sp_dists[idx] < d_end
            int idx1 = upper_bound(sp_dists.begin(), sp_dists.end(), d_start) - sp_dists.begin();
            int idx2 = lower_bound(sp_dists.begin(), sp_dists.end(), d_end) - sp_dists.begin() - 1;
            if (idx1 <= idx2) {
                road_diff[idx1]++;
                road_diff[idx2 + 1]--;
            }
        } else if (distX[v] + l + distY[u] == D) {
            // Directed road v -> u
            long long d_start = distX[v];
            long long d_end = distX[v] + l;
            int idx1 = upper_bound(sp_dists.begin(), sp_dists.end(), d_start) - sp_dists.begin();
            int idx2 = lower_bound(sp_dists.begin(), sp_dists.end(), d_end) - sp_dists.begin() - 1;
            if (idx1 <= idx2) {
                road_diff[idx1]++;
                road_diff[idx2 + 1]--;
            }
        }
    }

    vector<int> road_count_at_dist(sp_dists.size(), 0);
    int current_roads = 0;
    for (int i = 0; i < (int)sp_dists.size(); ++i) {
        current_roads += road_diff[i];
        road_count_at_dist[i] = current_roads;
    }

    for (int i = 1; i <= n; ++i) {
        if (on_sp_junctions[i]) {
            auto it = lower_bound(sp_dists.begin(), sp_dists.end(), distX[i]);
            int idx = distance(sp_dists.begin(), it);
            cout << junc_count_at_dist[idx] + road_count_at_dist[idx] << (i == n ? "" : " ");
        } else {
            cout << 0 << (i == n ? "" : " ");
        }
    }
    cout << endl;

    return 0;
}
