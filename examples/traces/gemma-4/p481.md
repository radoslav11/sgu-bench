#include <iostream>
#include <vector>
#include <iomanip>
#include <algorithm>

using namespace std;

/**
 * The problem asks for the number of connected undirected graphs with n vertices and n edges.
 * Such a graph contains exactly one cycle.
 * The number of such labeled graphs (Mescher trees) with n vertices is given by:
 * f(n) = (1/2) * sum_{m=0}^{n-3} [(n-1)! / m!] * n^m
 * This formula is derived from the number of ways to pick a cycle of length k (3 <= k <= n)
 * and then attach trees to each of its vertices such that the total number of vertices is n.
 * Let P_m = [(n-1)! / m!] * n^m.
 * Then f(n) = (1/2) * (P_0 + P_1 + ... + P_{n-3}).
 * P_0 = (n-1)!
 * P_m = P_{m-1} * (n / m)
 * All P_m are integers because m <= n-3 < n-1.
 */

struct BigInt {
    vector<long long> digits;
    static const long long BASE = 1000000000LL; // 10^9

    BigInt(long long v = 0) {
        if (v == 0) digits.push_back(0);
        while (v > 0) {
            digits.push_back(v % BASE);
            v /= BASE;
        }
    }

    // Multiplication by an integer
    void multiply(int v) {
        if (v == 0) {
            digits.assign(1, 0);
            return;
        }
        long long carry = 0;
        for (size_t i = 0; i < digits.size() || carry; ++i) {
            if (i == digits.size()) digits.push_back(0);
            long long cur = digits[i] * (long long)v + carry;
            digits[i] = cur % BASE;
            carry = cur / BASE;
        }
        while (digits.size() > 1 && digits.back() == 0) digits.pop_back();
    }

    // Division by an integer
    void divide(int v) {
        if (v == 0) return;
        long long rem = 0;
        for (int i = (int)digits.size() - 1; i >= 0; --i) {
            long long cur = digits[i] + rem * BASE;
            digits[i] = cur / v;
            rem = cur % v;
        }
        while (digits.size() > 1 && digits.back() == 0) digits.pop_back();
    }

    // Addition of another BigInt
    void add(const BigInt& other) {
        long long carry = 0;
        size_t n = max(digits.size(), other.digits.size());
        for (size_t i = 0; i < n || carry; ++i) {
            if (i == digits.size()) digits.push_back(0);
            long long other_digit = (i < other.digits.size() ? other.digits[i] : 0);
            long long cur = digits[i] + other_digit + carry;
            digits[i] = cur % BASE;
            carry = cur / BASE;
        }
    }

    // Division by 2
    void divideByTwo() {
        long long rem = 0;
        for (int i = (int)digits.size() - 1; i >= 0; --i) {
            long long cur = digits[i] + rem * BASE;
            digits[i] = cur / 2;
            rem = cur % 2;
        }
        while (digits.size() > 1 && digits.back() == 0) digits.pop_back();
    }

    // Output the BigInt
    void print() const {
        if (digits.empty()) {
            cout << 0 << endl;
            return;
        }
        cout << digits.back();
        for (int i = (int)digits.size() - 2; i >= 0; --i) {
            cout << setfill('0') << setw(9) << digits[i];
        }
        cout << endl;
    }
};

int main() {
    // Optimize I/O performance
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    if (n < 3) {
        // The problem states n >= 3
        cout << 0 << endl;
        return 0;
    }

    // P_0 = (n-1)!
    BigInt P(1);
    for (int i = 1; i < n; ++i) {
        P.multiply(i);
    }

    // Summing up P_m for m = 0 to n-3
    BigInt sum = P;
    for (int m = 1; m <= n - 3; ++m) {
        // P_m = P_{m-1} * n / m
        P.multiply(n);
        P.divide(m);
        sum.add(P);
    }

    // The final answer is sum / 2
    sum.divideByTwo();
    sum.print();

    return 0;
}
