#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
#include <algorithm>

using namespace std;

typedef long double ld;

struct Point {
    ld x, y;
};

ld area(const vector<Point>& p) {
    ld a = 0;
    for (int i = 0; i < p.size(); ++i) {
        a += (p[i].x * p[(i + 1) % p.size()].y - p[(i + 1) % p.size()].x * p[i].y);
    }
    return a / 2.0;
}

void solve_2x2(ld a1, ld b1, ld c1, ld a2, ld b2, ld c2, ld &x, ld &y, bool &singular) {
    ld det = a1 * b2 - a2 * b1;
    if (abs(det) < 1e-15) {
        singular = true;
        return;
    }
    x = (b1 * c2 - b2 * c1) / det;
    y = (a2 * c1 - a1 * c2) / det;
    singular = false;
}

void print_poly(const vector<Point>& p) {
    cout << p.size();
    for (const auto& pt : p) cout << " " << fixed << setprecision(10) << pt.x << " " << pt.y;
    cout << endl;
}

int main() {
    int N;
    if (!(cin >> N)) return 0;
    vector<Point> V(N);
    for (int i = 0; i < N; ++i) cin >> V[i].x >> V[i].y;
    string ratios;
    cin >> ratios;
    vector<int> K;
    string temp = "";
    for (char c : ratios) {
        if (c == ':') {
            K.push_back(stoi(temp));
            temp = "";
        } else temp += c;
    }
    K.push_back(stoi(temp));

    ld total_area = area(V);
    if (total_area < 0) {
        reverse(V.begin(), V.end());
        reverse(K.begin(), K.end());
        total_area = -total_area;
    }

    ld sum_K = 0;
    for (int k : K) sum_K += k;

    if (N == 3) {
        ld A1 = total_area * K[0] / sum_K;
        ld A2 = total_area * K[1] / sum_K;
        ld A3 = total_area * K[2] / sum_K;
        ld w1 = K[1] / sum_K;
        ld w2 = K[2] / sum_K;
        ld w3 = K[0] / sum_K;
        Point P = {w1 * V[0].x + w2 * V[1].x + w3 * V[2].x, w1 * V[0].y + w2 * V[1].y + w3 * V[2].y};
        print_poly({V[0], V[1], P});
        print_poly({V[1], V[2], P});
        print_poly({V[2], V[0], P});
    } else {
        vector<ld> A(4);
        for (int i = 0; i < 4; ++i) A[i] = total_area * K[i] / sum_K;

        // Case 1: Common Point P
        ld a[4], b[4], c[4];
        for (int i = 0; i < 4; ++i) {
            int next = (i + 1) % 4;
            a[i] = V[i].y - V[next].y;
            b[i] = V[next].x - V[i].x;
            c[i] = V[i].x * V[next].y - V[next].x * V[i].y - 2.0 * A[i];
        }
        ld px, py;
        bool singular;
        solve_2x2(a[0], b[0], -c[0], a[1], b[1], -c[1], px, py, singular);
        if (!singular) {
            bool ok = true;
            for (int i = 0; i < 4; ++i) {
                if (abs(a[i] * px + b[i] * py + c[i]) > 1e-7) ok = false;
            }
            if (ok) {
                // Verify P is inside by checking A_i > 0
                // For CCW, A_i(P, Vi, Vi+1) = 0.5 * (xi*yi+1 - yi*xi+1 + xi+1*py - yi+1*px + px*yi - py*xi)
                // We already used this to solve. Now check px, py is inside
                for(int i=0; i<4; ++i) {
                    int next = (i+1)%4;
                    ld area_part = (V[i].x * V[next].y - V[next].x * V[i].y + V[next].x * py - V[next].y * px + px * V[i].y - py * V[i].x) / 2.0;
                    if (area_part < -1e-7) ok = false;
                }
                if (ok) {
                    print_poly({V[0], V[1], P});
                    print_poly({V[1], V[2], P});
                    print_poly({V[2], V[3], P});
                    print_poly({V[3], V[0], P});
                    return 0;
                }
            }
        }

        // Case 2: Two points P1, P2
        for (int start_side : {0, 1}) {
            int s1 = start_side, s2 = (start_side + 1) % 4, s3 = (start_side + 2) % 4, s4 = (start_side + 3) % 4;
            // Triangles: s1, s3. Quads: s2, s4.
            // Part 1: V_s1, V_s1+1, P1
            // Part 3: V_s3, V_s3+1, P2
            // Part 2: V_s1+1, V_s1+2, P2, P1
            // Part 4: V_s3+1, V_s1, P1, P2
            int v1 = s1, v2 = (s1 + 1) % 4, v3 = (s1 + 2) % 4, v4 = (s1 + 3) % 4;
            // Triangle 1: V_v1, V_v2, P1
            // Triangle 3: V_v3, V_v4, P2
            // Quad 2: V_v2, V_v3, P2, P1
            // Quad 4: V_v4, V_v1, P1, P2

            ld a1 = V[v1].y - V[v2].y, b1 = V[v2].x - V[v1].x, c1 = V[v1].x * V[v2].y - V[v2].x * V[v1].y - 2.0 * A[s1];
            ld a3 = V[v3].y - V[v4].y, b3 = V[v4].x - V[v3].x, c3 = V[v3].x * V[v4].y - V[v4].x * V[v3].y - 2.0 * A[s3];

            Point p1, p2;
            if (abs(b1) > 1e-9) {
                p1.x = (V[v1].x + V[v2].x) / 2.0;
                p1.y = (-c1 - a1 * p1.x) / b1;
            } else {
                p1.x = -c1 / a1;
                p1.y = (V[v1].y + V[v2].y) / 2.0;
            }

            ld rhs = 2.0 * A[s2] - (V[v2].x * V[v3].y - V[v3].x * V[v2].y) - (p1.x * V[v2].y - p1.y * V[v2].x);
            // RHS = px_p2 * (py_p1 - vy3) + py_p2 * (vx3 - px_p1) 
            // But wait, my area formula is: 2*A2 = x2y3-y2x3 + x3yp2-y3xp2 + xp2yp1-yp2xp1 + xp1y2-yp1x2
            // 2*A2 - (x2y3-y2x3) - (xp1y2-yp1x2) = xp2(yp1-y3) + yp2(x3-xp1)
            ld a_prime = p1.y - V[v3].y, b_prime = V[v3].x - p1.x, c_prime = -rhs;
            
            solve_2x2(a3, b3, c3, a_prime, b_prime, c_prime, p2.x, p2.y, singular);
            if (!singular) {
                // Verify p1, p2 are inside and areas are ok
                bool ok = true;
                // Check P1 in Triangle 1 area: (x1y2-y1x2 + x2yp1-y2xp1 + xp1y1-yp1x1)/2 = A1
                // Check P2 in Triangle 3 area: (x3y4-y3x4 + x4yp2-y4xp2 + xp2y3-yp2x3)/2 = A3
                // Check Quad 2 area: (x2y3-y2x3 + x3yp2-y3xp2 + xp2yp1-yp2xp1 + xp1y2-yp1x2)/2 = A2
                // Check Quad 4 area: (x4y1-y4x1 + x1yp1-y1xp1 + xp1yp2-yp1xp2 + xp2y4-yp2x4)/2 = A4
                auto check_area = [&](const vector<Point>& poly, ld expected) {
                    return abs(area(poly) - expected) < 1e-7;
                };
                if (check_area({V[v1], V[v2], p1}, A[s1]) &&
                    check_area({V[v3], V[v4], p2}, A[s3]) &&
                    check_area({V[v2], V[v3], p2, p1}, A[s2]) &&
                    check_area({V[v4], V[v1], p1, p2}, A[s4])) {
                    // Also need to check if they are inside
                    // Since we already checked area sum and all A_i > 0, it should be ok for convex.
                    print_poly({V[v1], V[v2], p1});
                    print_poly({V[v2], V[v3], p2, p1});
                    print_poly({V[v3], V[v4], p2});
                    print_poly({V[v4], V[v1], p1, p2});
                    return 0;
                }
            }
        }
        cout << -1 << endl;
    }

    return 0;
}
