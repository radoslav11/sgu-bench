#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long ll;

/**
 * Problem Analysis:
 * To reach B from A using only allowed divisors:
 * 1. A and B must both be composite.
 * 2. If B is odd, we must reach B-d, where d is a divisor of B such that 1 < d < B/2.
 *    The largest such d is B/p, where p is the smallest prime factor of B.
 *    Since B is composite and odd, p >= 3, so d = B/p is at most B/3, which is < B/2.
 * 3. Once we have a target T (either B if B is even, or B-B/p if B is odd),
 *    T is always even.
 * 4. If A is odd, its first step must be adding its smallest prime factor to make it even.
 * 5. Once A is even, we can reach even T by choosing appropriate divisors.
 *    The strategy to minimize steps is to pick the largest divisor d <= T-A.
 *    Given the problem guarantee, this strategy will find a sequence within 500 steps.
 */

const int MAX_SQRT = 1000005;
vector<int> primes;
bool is_prime_sieve[MAX_SQRT];

// Precompute primes up to 10^6 for fast factorization and primality testing
void sieve() {
    fill(is_prime_sieve + 2, is_prime_sieve + MAX_SQRT, true);
    for (int p = 2; p * p < MAX_SQRT; p++) {
        if (is_prime_sieve[p]) {
            for (int i = p * p; i < MAX_SQRT; i += p)
                is_prime_sieve[i] = false;
        }
    }
    for (int p = 2; p < MAX_SQRT; p++) {
        if (is_prime_sieve[p]) primes.push_back(p);
    }
}

ll get_smallest_prime_factor(ll n) {
    for (int p : primes) {
        if (1LL * p * p > n) break;
        if (n % p == 0) return (ll)p;
    }
    return n;
}

bool is_prime(ll n) {
    if (n < MAX_SQRT) return is_prime_sieve[n];
    for (int p : primes) {
        if (1LL * p * p > n) break;
        if (n % p == 0) return false;
    }
    return true;
}

// Generate all divisors of a number given its prime factorization
void get_all_divisors(ll n, vector<ll>& divisors) {
    vector<pair<ll, int>> factors;
    ll temp = n;
    for (int p : primes) {
        if (1LL * p * p > temp) break;
        if (temp % p == 0) {
            int count = 0;
            while (temp % p == 0) {
                temp /= p;
                count++;
            }
            factors.push_back({(ll)p, count});
        }
    }
    if (temp > 1) factors.push_back({temp, 1});

    divisors.clear();
    divisors.push_back(1);
    for (auto& f : factors) {
        int sz = (int)divisors.size();
        ll p_pow = 1;
        for (int i = 1; i <= f.second; i++) {
            p_pow *= f.first;
            for (int j = 0; j < sz; j++) {
                divisors.push_back(divisors[j] * p_pow);
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    sieve();
    ll A, B;
    if (!(cin >> A >> B)) return 0;

    // If A or B is prime, no sequence can exist (unless A=B, but A < B)
    if (is_prime(A) || is_prime(B)) {
        cout << "Impossible" << endl;
        return 0;
    }

    ll T;
    bool B_is_odd = (B % 2 != 0);
    if (B_is_odd) {
        ll p = get_smallest_prime_factor(B);
        T = B - B / p;
    } else {
        T = B;
    }

    // If initial A is already past target T, it's impossible
    if (A > T) {
        cout << "Impossible" << endl;
        return 0;
    }

    vector<ll> seq;
    seq.push_back(A);

    while (A < T) {
        if (A % 2 != 0) {
            // If A is odd, its smallest prime factor must be added to make it even
            ll p = get_smallest_prime_factor(A);
            if (A + p > T) {
                cout << "Impossible" << endl;
                return 0;
            }
            A += p;
            seq.push_back(A);
        } else {
            // If A is even, try to find the largest divisor d that keeps us within target T
            vector<ll> divisors;
            get_all_divisors(A, divisors);
            ll best_d = -1;
            for (ll d : divisors) {
                if (d > 1 && d < A && A + d <= T) {
                    if (best_d == -1 || d > best_d) {
                        best_d = d;
                    }
                }
            }
            if (best_d == -1) {
                cout << "Impossible" << endl;
                return 0;
            }
            A += best_d;
            seq.push_back(A);
        }
    }

    // If B was odd, the last step is from target T to B
    if (B_is_odd) {
        seq.push_back(B);
    }

    // Print the sequence
    for (ll val : seq) {
        cout << val << "\n";
    }

    return 0;
}
