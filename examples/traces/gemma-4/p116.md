```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Problem: Index of super-prime
 * A super-prime number is a prime number whose index in the sequence of primes is also prime.
 * For example:
 * Primes: P1=2, P2=3, P3=5, P4=7, P5=11, P6=13, P7=17, ...
 * Indices: 1, 2, 3, 4, 5, 6, 7, ...
 * Super-primes (index is prime): P2=3 (index 2), P3=5 (index 3), P5=11 (index 5), P7=17 (index 7), ...
 * 
 * Task: Find the minimum number of super-primes that sum to n.
 * Output the minimum number and the super-primes in non-increasing order.
 * If no such sum exists, output 0.
 */

const int MAXN = 10000;
bool is_prime[MAXN + 1];
vector<int> primes;
vector<int> superPrimes;
int dp[MAXN + 1];
int parent[MAXN + 1];

// Precompute primes and super-primes up to MAXN
void sieve() {
    for (int i = 2; i <= MAXN; i++) is_prime[i] = true;
    for (int p = 2; p * p <= MAXN; p++) {
        if (is_prime[p]) {
            for (int i = p * p; i <= MAXN; i += p)
                is_prime[i] = false;
        }
    }
    for (int p = 2; p <= MAXN; p++) {
        if (is_prime[p]) {
            primes.push_back(p);
        }
    }
    for (int i = 0; i < (int)primes.size(); i++) {
        int idx = i + 1; // 1-based index
        // A prime is a super-prime if its 1-based index is also prime.
        if (idx <= MAXN && is_prime[idx]) {
            superPrimes.push_back(primes[i]);
        }
    }
}

int main() {
    // Speed up I/O
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    sieve();

    int n;
    // Read input (single integer as per problem statement)
    while (cin >> n) {
        // dp[i] stores the minimum number of super-primes needed to sum up to i.
        // parent[i] stores the last super-prime used to reach sum i to allow reconstruction.
        for (int i = 0; i <= n; i++) {
            dp[i] = MAXN + 7; // Initialize with a value larger than any possible answer
            parent[i] = -1;
        }

        dp[0] = 0;
        for (int i = 1; i <= n; i++) {
            for (int sp : superPrimes) {
                if (sp > i) break; // Optimization: super-primes are sorted
                if (dp[i - sp] != MAXN + 7 && dp[i - sp] + 1 < dp[i]) {
                    dp[i] = dp[i - sp] + 1;
                    parent[i] = sp;
                }
            }
        }

        // Check if the target number can be represented as a sum of super-primes
        if (dp[n] > MAXN) {
            cout << 0 << "\n";
        } else {
            cout << dp[n] << "\n";
            vector<int> res;
            int curr = n;
            // Reconstruct the sum using the parent array
            while (curr > 0) {
                res.push_back(parent[curr]);
                curr -= parent[curr];
            }
            // Sort the super-primes in non-increasing order
            sort(res.rbegin(), res.rend());
            for (int i = 0; i < (int)res.size(); i++) {
                cout << res[i] << (i == (int)res.size() - 1 ? "" : " ");
            }
            cout << "\n";
        }
    }

    return 0;
}
```
