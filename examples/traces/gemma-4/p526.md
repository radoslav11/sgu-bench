#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <iomanip>
#include <map>

using namespace std;

struct Interval {
    double l, r;
};

struct Stone {
    int x1, x2, t;
};

struct Instruction {
    double w, t;
};

// Intersect two sets of disjoint intervals
vector<Interval> intersect(const vector<Interval>& A, const vector<Interval>& B) {
    vector<Interval> res;
    size_t i = 0, j = 0;
    while (i < A.size() && j < B.size()) {
        double l = max(A[i].l, B[j].l);
        double r = min(A[i].r, B[j].r);
        if (l <= r + 1e-12) {
            res.push_back({l, r});
        }
        if (A[i].r < B[j].r) i++;
        else j++;
    }
    return res;
}

// Merge overlapping intervals
vector<Interval> merge_all(vector<Interval>& intervals) {
    if (intervals.empty()) return {};
    sort(intervals.begin(), intervals.end(), [](Interval a, Interval b) {
        return a.l < b.l;
    });
    vector<Interval> res;
    res.push_back(intervals[0]);
    for (size_t i = 1; i < intervals.size(); ++i) {
        if (intervals[i].l <= res.back().r + 1e-12) {
            res.back().r = max(res.back().r, intervals[i].r);
        } else {
            res.push_back(intervals[i]);
        }
    }
    return res;
}

// Compute the set of allowed positions at a given stone time
vector<Interval> get_Si(const vector<Interval>& stones, double M) {
    if (stones.empty()) return {{-M, M}};
    vector<Interval> sorted_stones = stones;
    sort(sorted_stones.begin(), sorted_stones.end(), [](Interval a, Interval b) {
        return a.l < b.l;
    });

    vector<Interval> open_intervals;
    double cur_l = sorted_stones[0].l;
    double cur_r = sorted_stones[0].r;
    for (size_t i = 1; i < sorted_stones.size(); ++i) {
        if (sorted_stones[i].l < cur_r - 1e-12) {
            cur_r = max(cur_r, sorted_stones[i].r);
        } else {
            open_intervals.push_back({cur_l, cur_r});
            cur_l = sorted_stones[i].l;
            cur_r = sorted_stones[i].r;
        }
    }
    open_intervals.push_back({cur_l, cur_r});

    vector<Interval> Si;
    double last_r = -M;
    for (auto& inter : open_intervals) {
        if (inter.l > last_r + 1e-12) {
            Si.push_back({last_r, inter.l});
        } else if (abs(inter.l - last_r) < 1e-12) {
            Si.push_back({last_r, inter.l});
        }
        last_r = inter.r;
    }
    if (last_r < M - 1e-12) {
        Si.push_back({last_r, M});
    }
    return Si;
}

bool check(double T, double v, double G, const vector<double>& times, const vector<vector<Interval>>& S, vector<vector<Interval>>& R_history) {
    R_history.clear();
    vector<Interval> R = {{0, 0}};
    R_history.push_back(R);

    for (size_t i = 1; i < times.size(); ++i) {
        if (times[i] >= T - 1e-12) break;
        double dt = times[i] - times[i - 1];
        vector<Interval> R_prime;
        for (auto& inter : R) {
            R_prime.push_back({inter.l - v * dt, inter.r + v * dt});
        }
        R = merge_all(R_prime);
        R = intersect(R, S[i]);
        if (R.empty()) {
            R_history.push_back({});
            return false;
        }
        R_history.push_back(R);
    }

    double last_t = (R_history.size() == 1) ? 0 : times[R_history.size() - 1];
    double dt = T - last_t;
    // Final check: exists x in R such that |G - x| <= v * dt
    for (auto& inter : R) {
        if (G >= inter.l - v * dt - 1e-10 && G <= inter.r + v * dt + 1e-10) {
            return true;
        }
    }
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    double v, G;
    int n;
    if (!(cin >> v >> G >> n)) return 0;

    map<int, vector<Interval>> stones_at_time;
    for (int i = 0; i < n; ++i) {
        int x1, x2, t;
        cin >> x1 >> x2 >> t;
        stones_at_time[t].push_back({(double)x1, (double)x2});
    }

    vector<double> times;
    times.push_back(0);
    for (auto const& [t, stones] : stones_at_time) {
        times.push_back((double)t);
    }
    sort(times.begin(), times.end());
    times.erase(unique(times.begin(), times.end()), times.end());

    int m = times.size();
    vector<vector<Interval>> S(m);
    double M = 1e6;
    for (int i = 0; i < m; ++i) {
        if (stones_at_time.count((int)times[i])) {
            S[i] = get_Si(stones_at_time[(int)times[i]], M);
        } else {
            S[i] = {{-M, M}};
        }
    }

    double low = abs(G) / v, high = 1e6;
    bool possible = false;
    for (int iter = 0; iter < 80; ++iter) {
        double mid = (low + high) / 2.0;
        vector<vector<Interval>> dummy;
        if (check(mid, v, G, times, S, dummy)) {
            high = mid;
            possible = true;
        } else {
            low = mid;
        }
    }

    if (!possible) {
        cout << -1 << endl;
        return 0;
    }

    double T = high;
    vector<vector<Interval>> R_history;
    check(T, v, G, times, S, R_history);

    // The check function only goes up to times[i] < T.
    // Let's find how many unique times we actually used.
    int last_idx = R_history.size() - 1;
    
    vector<double> actual_times;
    for (int i = 0; i <= last_idx; ++i) actual_times.push_back(times[i]);
    actual_times.push_back(T);

    vector<double> path_x(actual_times.size());
    path_x.back() = G;

    for (int i = last_idx; i >= 0; --i) {
        double dt = actual_times[i+1] - actual_times[i];
        double L = path_x[i+1] - v * dt;
        double R_limit = path_x[i+1] + v * dt;
        
        // Find x in R_history[i] such that x is in [L, R_limit]
        bool found = false;
        for (auto& inter : R_history[i]) {
            double intersect_l = max(inter.l, L);
            double intersect_r = min(inter.r, R_limit);
            if (intersect_l <= intersect_r + 1e-12) {
                path_x[i] = intersect_l;
                found = true;
                break;
            }
        }
        // This should not happen if T is correct
        if (!found) path_x[i] = R_history[i][0].l;
    }

    vector<Instruction> instructions;
    for (size_t i = 0; i < actual_times.size() - 1; ++i) {
        double dt = actual_times[i+1] - actual_times[i];
        double dx = path_x[i+1] - path_x[i];
        if (dt > 1e-13) {
            instructions.push_back({dx / dt, dt});
        }
    }

    // Merge consecutive instructions with same speed (optional, but can reduce output)
    vector<Instruction> merged_instructions;
    for (auto& inst : instructions) {
        if (!merged_instructions.empty() && abs(merged_instructions.back().w - inst.w) < 1e-9) {
            merged_instructions.back().t += inst.t;
        } else {
            merged_instructions.push_back(inst);
        }
    }

    cout << merged_instructions.size() << "\n";
    cout << fixed << setprecision(10);
    for (auto& inst : merged_instructions) {
        cout << inst.w << " " << inst.t << "\n";
    }

    return 0;
}
