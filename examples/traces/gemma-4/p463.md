```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

/**
 * Problem Analysis:
 * Berhattan is an n x m grid of blocks.
 * Streets (horizontal) and avenues (vertical) divide the grid.
 * There are n+1 streets (y = 0 to n) and m+1 avenues (x = 0 to m).
 * A block (i, j) is defined by the area between streets i and i+1, and avenues j and j+1.
 * Walking along a street or avenue "passes" a block if that street/avenue is one of its boundaries.
 * 
 * Satisfaction Rules:
 * - First time passing a block (i, j): +b[i][j]
 * - Subsequent times: +floor(b[i][j] / 2)
 * 
 * Movement:
 * - 'M': move one unit in the current direction.
 * - 'L': 90-degree left turn.
 * - 'R': 90-degree right turn.
 * - Starts at (0,0) facing East.
 * 
 * Complexity:
 * - n, m <= 100, path length <= 500.
 * - Time complexity O(path_length), which is well within 0.25s.
 * - Space complexity O(n*m), which fits in 64MB.
 */

int main() {
    // Speed up I/O
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    // Reading the beauty values of blocks.
    // b[i][j] stores the beauty of the block at row i and column j.
    vector<vector<int>> b(n, vector<int>(m));
    for (int i = 0; i < n; ++i) {
        string row;
        cin >> row;
        for (int j = 0; j < m; ++j) {
            b[i][j] = row[j] - '0';
        }
    }

    string path;
    cin >> path;

    // counts[i][j] stores how many times block (i, j) has been passed.
    vector<vector<int>> counts(n, vector<int>(m, 0));
    
    // Starting position and direction
    int x = 0, y = 0;
    int dir = 0; // 0: East, 1: South, 2: West, 3: North
    int dx[] = {1, 0, -1, 0};
    int dy[] = {0, 1, 0, -1};
    
    long long total_satisfaction = 0;

    for (char c : path) {
        if (c == 'L') {
            // Turn left: 90 degrees counter-clockwise
            dir = (dir + 3) % 4;
        } else if (c == 'R') {
            // Turn right: 90 degrees clockwise
            dir = (dir + 1) % 4;
        } else if (c == 'M') {
            // Move one step forward
            int nx = x + dx[dir];
            int ny = y + dy[dir];

            // Check whether the movement is horizontal or vertical
            if (dir == 0 || dir == 2) { // Horizontal movement (along a street)
                // The movement is on street y, between avenue min(x, nx) and max(x, nx)
                int x_min = min(x, nx);
                
                // If the street is not the very top boundary, we pass the block above it.
                if (y > 0) {
                    int r = y - 1, col = x_min;
                    if (counts[r][col] == 0) {
                        total_satisfaction += b[r][col];
                        counts[r][col] = 1;
                    } else {
                        total_satisfaction += b[r][col] / 2;
                        counts[r][col]++;
                    }
                }
                // If the street is not the very bottom boundary, we pass the block below it.
                if (y < n) {
                    int r = y, col = x_min;
                    if (counts[r][col] == 0) {
                        total_satisfaction += b[r][col];
                        counts[r][col] = 1;
                    } else {
                        total_satisfaction += b[r][col] / 2;
                        counts[r][col]++;
                    }
                }
            } else { // Vertical movement (along an avenue)
                // The movement is on avenue x, between street min(y, ny) and max(y, ny)
                int y_min = min(y, ny);

                // If the avenue is not the leftmost boundary, we pass the block to the left.
                if (x > 0) {
                    int r = y_min, col = x - 1;
                    if (counts[r][col] == 0) {
                        total_satisfaction += b[r][col];
                        counts[r][col] = 1;
                    } else {
                        total_satisfaction += b[r][col] / 2;
                        counts[r][col]++;
                    }
                }
                // If the avenue is not the rightmost boundary, we pass the block to the right.
                if (x < m) {
                    int r = y_min, col = x;
                    if (counts[r][col] == 0) {
                        total_satisfaction += b[r][col];
                        counts[r][col] = 1;
                    } else {
                        total_satisfaction += b[r][col] / 2;
                        counts[r][col]++;
                    }
                }
            }
            // Update position
            x = nx;
            y = ny;
        }
    }

    cout << total_satisfaction << endl;

    return 0;
}
```
