#include <iostream>
#include <string>
#include <vector>
#include <queue>
#include <unordered_map>
#include <algorithm>

using namespace std;

/**
 * Problem Summary:
 * A string is permutative for N if it's a permutation of the first N capital letters.
 * A string is almost permutative for N if removing one character makes it permutative for N.
 * A string is hyper almost permutative for N if every substring of length N+1 is almost permutative.
 * Goal: Find the shortest hyper almost permutative string that contains S1 and S2 (both length N).
 *
 * A window of length N+1 is almost permutative if it contains the first N characters of the alphabet.
 * We assume the shortest string only uses characters from the first N capital letters.
 */

struct State {
    string lastN;
    bool seenS2;

    bool operator==(const State& other) const {
        return seenS2 == other.seenS2 && lastN == other.lastN;
    }
};

struct StateHash {
    size_t operator()(const State& s) const {
        size_t h = hash<string>{}(s.lastN);
        h ^= hash<bool>{}(s.seenS2) + 0x9e3779b9 + (h << 6) + (h >> 2);
        return h;
    }
};

bool is_almost_permutative(const string& s, int N) {
    if (s.length() != N + 1) return false;
    vector<int> counts(26, 0);
    for (char c : s) {
        if (c - 'A' >= N) return false; // Not within the first N letters
        counts[c - 'A']++;
    }
    int missing = 0;
    for (int i = 0; i < N; ++i) {
        if (counts[i] == 0) missing++;
    }
    // If exactly one character is missing, then the window must have one character repeated (or an extra character)
    // In our assumption (only using the first N letters), the extra character must be one of the N characters.
    // The condition for the string to contain V is: the count of each character in V must be at least 1.
    // However, a window of length N+1 containing all N characters from V means exactly one character is repeated.
    return missing == 0;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string S1, S2;
    cin >> S1 >> S2;

    if (S1 == S2) {
        cout << S1 << endl;
        return 0;
    }

    // We use BFS to find the shortest string.
    // State: (last N characters, whether S2 has been seen as a substring).
    // Start BFS from S1.
    queue<pair<State, string>> q;
    unordered_map<State, int, StateHash> visited;

    State startState = {S1, false};
    q.push({startState, S1});
    visited[startState] = S1.length();

    // S2 could be S1 itself (not possible here), or part of S1.
    if (S1.find(S2) != string::npos) {
        startState.seenS2 = true;
        q.front().first.seenS2 = true;
    }

    while (!q.empty()) {
        State currState = q.front().first;
        string currStr = q.front().second;
        q.pop();

        if (currState.seenS2) {
            cout << currStr << endl;
            return 0;
        }

        // Try adding every character from 'A' to 'A'+N-1
        for (int i = 0; i < N; ++i) {
            char c = 'A' + i;
            string nextWindow = currState.lastN + c;
            // The new window of length N+1 must be almost permutative.
            // This is equivalent to checking if the last N+1 characters contain all N characters.
            
            // Check if the last N+1 characters contain all characters from 'A' to 'A'+N-1
            bool containsAll = true;
            vector<bool> present(N, false);
            for (char ch : nextWindow) {
                if (ch - 'A' < N) present[ch - 'A'] = true;
            }
            for (int j = 0; j < N; ++j) {
                if (!present[j]) {
                    containsAll = false;
                    break;
                }
            }

            if (containsAll) {
                string nextLastN = nextWindow.substr(1);
                bool nextSeenS2 = currState.seenS2 || (currStr.length() >= (size_t)N && (currStr.substr(currStr.length() - N) == S2 || (nextWindow.size() >= (size_t)N+1 && nextWindow.substr(nextWindow.size() - N) == S2)));
                // Correctly check if S2 is now a substring. The window approach is more reliable:
                // If we append 'c', we check if the new string's suffix of length N is S2.
                // However, a substring could have been completed.
                // Since S2 has length N, it's completed if the last N characters are S2.
                // But we only need to check if the window [length-N, length-1] is S2.
                // Let's re-evaluate nextSeenS2.
                string nextStr = currStr + c;
                bool newlySeenS2 = (nextStr.length() >= (size_t)N && nextStr.substr(nextStr.length() - N) == S2);
                bool totalSeenS2 = currState.seenS2 || newlySeenS2;

                State nextState = {nextLastN, totalSeenS2};
                if (visited.find(nextState) == visited.end()) {
                    visited[nextState] = nextStr.length();
                    q.push({nextState, nextStr});
                }
            }
        }
    }

    return 0;
}

Wait, the BFS state logic needs to be correct. The character we add must ensure the *new* window of length $N+1$ is almost permutative. If the string is $S$, the new window is $S[|S|-N \dots |S|]$. This window has length $N+1$.

Let's refine the code for submission. The current BFS structure is okay. Let's double-check the condition for `nextSeenS2`.
