#include <cstdio>
#include <vector>

/**
 * Problem: Matrix Multiplication
 * 
 * Let the incidence matrix A of an undirected graph with N vertices and M edges be an N x M matrix,
 * where A_{ij} = 1 if vertex i is an endpoint of edge j, and 0 otherwise.
 * The task is to find the sum of all elements of the matrix C = A^T * A.
 * 
 * Analysis:
 * C is an M x M matrix where the element C_{jk} is calculated as:
 * C_{jk} = sum_{i=1 to N} (A^T)_{ji} * A_{ik} = sum_{i=1 to N} A_{ij} * A_{ik}
 * 
 * C_{jk} counts the number of vertices common to both edge j and edge k.
 * For a simple graph (no loops and no multiple edges):
 * - If j = k: C_{jj} = 2 (each edge has exactly 2 endpoints).
 * - If j != k and edges j and k share one vertex: C_{jk} = 1.
 * - If j != k and edges j and k share zero vertices: C_{jk} = 0.
 * 
 * The sum of all elements in C is:
 * S = sum_{j=1 to M} sum_{k=1 to M} C_{jk}
 * S = sum_{j=1 to M} sum_{k=1 to M} sum_{i=1 to N} (A_{ij} * A_{ik})
 * Reordering the summation:
 * S = sum_{i=1 to N} sum_{j=1 to M} sum_{k=1 to M} (A_{ij} * A_{ik})
 * S = sum_{i=1 to N} (sum_{j=1 to M} A_{ij}) * (sum_{k=1 to M} A_{ik})
 * 
 * The term sum_{j=1 to M} A_{ij} is the degree of vertex i, denoted as d_i.
 * S = sum_{i=1 to N} d_i * d_i = sum_{i=1 to N} (d_i)^2.
 * 
 * Constraints:
 * N <= 10,000, M <= 100,000.
 * Max possible value for sum_{i=1 to N} (d_i)^2 is roughly M^2 / N * N = M^2, 
 * which for M = 100,000 is 10,000,000,000. 
 * This fits in a 64-bit long long.
 * 
 * Time complexity: O(N + M)
 * Space complexity: O(N)
 */

int main() {
    int n, m;
    // Using scanf for faster input to meet the tight 0.25s time limit.
    if (scanf("%d %d", &n, &m) != 2) {
        return 0;
    }

    // Degree array for each vertex.
    // Using a vector to handle N up to 10,000.
    std::vector<int> degree(n + 1, 0);

    for (int i = 0; i < m; ++i) {
        int u, v;
        if (scanf("%d %d", &u, &v) != 2) {
            break;
        }
        // Ensure vertices are within the valid range [1, n].
        if (u >= 1 && u <= n) {
            degree[u]++;
        }
        if (v >= 1 && v <= n) {
            degree[v]++;
        }
    }

    long long sum_squares = 0;
    for (int i = 1; i <= n; ++i) {
        // Sum the square of the degree of each vertex.
        // Cast to long long to avoid overflow before adding to total.
        sum_squares += static_cast<long long>(degree[i]) * degree[i];
    }

    // Output the final calculated sum.
    printf("%lld\n", sum_squares);

    return 0;
}
