```cpp
#include <iostream>
#include <vector>

using namespace std;

/**
 * Problem: 223. Little Kings
 * We need to place k kings on an n x n chessboard such that no two kings are in an attacking position.
 * Two kings are in an attacking position if they are on adjacent cells (including diagonals).
 * n is small (up to 10), so we can use dynamic programming with bitmask.
 * 
 * Let dp[r][kc][mask] be the number of ways to place kc kings in the first r rows, 
 * with the r-th row configuration represented by the bitmask 'mask'.
 * 
 * A mask is valid if no two kings in the same row are adjacent.
 * Two masks for adjacent rows are compatible if no kings attack each other.
 * 
 * Constraints: n <= 10, k <= n^2.
 * The number of valid masks for n=10 is F_{12} = 144.
 * The complexity will be O(n * k * (number of valid masks)^2).
 */

typedef long long ll;

int main() {
    // Faster I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, k;
    if (!(cin >> n >> k)) return 0;

    // If k is more than the number of cells, it's impossible.
    // Although k <= n^2 is given, safety check is fine.
    if (k < 0 || k > n * n) {
        cout << 0 << endl;
        return 0;
    }

    // Precompute valid masks for a single row of length n.
    // A mask is valid if no two kings are adjacent in the same row.
    vector<int> valid_masks;
    vector<int> popcounts;
    for (int m = 0; m < (1 << n); ++m) {
        if ((m & (m << 1)) == 0) {
            valid_masks.push_back(m);
            popcounts.push_back(__builtin_popcount(m));
        }
    }

    int num_masks = valid_masks.size();

    // Precompute compatible masks for adjacent rows.
    // A mask m1 in row r and m2 in row r+1 are compatible if:
    // (m1 & m2) == 0 (no king in the same column)
    // (m1 & (m2 << 1)) == 0 (no king in the diagonal column c-1)
    // (m1 & (m2 >> 1)) == 0 (no king in the diagonal column c+1)
    vector<vector<int>> compatible(num_masks);
    for (int i = 0; i < num_masks; ++i) {
        for (int j = 0; j < num_masks; ++j) {
            int m1 = valid_masks[i];
            int m2 = valid_masks[j];
            if ((m1 & m2) == 0 && (m1 & (m2 << 1)) == 0 && (m1 & (m2 >> 1)) == 0) {
                compatible[i].push_back(j);
            }
        }
    }

    // dp[kc][mask_idx] stores the number of ways to place kc kings in the rows 
    // processed so far, ending with the current row's configuration being valid_masks[mask_idx].
    // We use a 2D DP (space optimized for row index) to keep memory usage low.
    vector<vector<ll>> dp(k + 1, vector<ll>(num_masks, 0));

    // Base case: first row (row 0)
    for (int i = 0; i < num_masks; ++i) {
        if (popcounts[i] <= k) {
            dp[popcounts[i]][i] = 1;
        }
    }

    // DP transitions for rows 1 to n-1
    for (int r = 1; r < n; ++r) {
        vector<vector<ll>> next_dp(k + 1, vector<ll>(num_masks, 0));
        for (int kc = 0; kc <= k; ++kc) {
            for (int i = 0; i < num_masks; ++i) {
                if (dp[kc][i] == 0) continue;
                for (int j : compatible[i]) {
                    int new_kc = kc + popcounts[j];
                    if (new_kc <= k) {
                        next_dp[new_kc][j] += dp[kc][i];
                    }
                }
            }
        }
        dp = move(next_dp);
    }

    // Sum all ways where exactly k kings have been placed.
    ll ans = 0;
    for (int i = 0; i < num_masks; ++i) {
        ans += dp[k][i];
    }

    cout << ans << endl;

    return 0;
}
```
