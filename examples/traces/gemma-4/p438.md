```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <queue>

using namespace std;

/**
 * Problem: The Glorious Karlutka River =)
 * 
 * We want to find the minimum time T such that M tourists can cross a river of width W.
 * A tourist can move up to D meters in one jump, and each jump takes 1 second.
 * Each rubbish pile (Xi, Yi) has a capacity Ci (max tourists at that pile at the same time).
 * The starting bank is Y=0 and the destination bank is Y=W.
 * 
 * This can be modeled as a max flow problem in a time-expanded graph.
 * Since M is small (M <= 50) and the number of piles is small (N <= 50),
 * we can build the time-expanded graph layer by layer (one second at a time)
 * and use the Ford-Fulkerson algorithm to incrementally find the max flow.
 */

struct Edge {
    int to;
    int capacity;
    int flow;
    int rev;
};

vector<vector<Edge>> adj;
vector<int> parent_edge;
vector<int> parent_node;

void add_edge(int u, int v, int cap) {
    adj[u].push_back({v, cap, 0, (int)adj[v].size()});
    adj[v].push_back({u, 0, 0, (int)adj[u].size() - 1});
}

// BFS to find an augmenting path for Ford-Fulkerson
bool bfs(int s, int t) {
    fill(parent_node.begin(), parent_node.end(), -1);
    fill(parent_edge.begin(), parent_edge.end(), -1);
    queue<int> q;
    q.push(s);
    parent_node[s] = s;
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        if (u == t) return true;
        for (int i = 0; i < (int)adj[u].size(); ++i) {
            Edge &e = adj[u][i];
            if (parent_node[e.to] == -1 && e.capacity > e.flow) {
                parent_node[e.to] = u;
                parent_edge[e.to] = i;
                q.push(e.to);
            }
        }
    }
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M, D, W;
    if (!(cin >> N >> M >> D >> W)) return 0;

    // Special case: can jump across the river directly
    if (W <= D) {
        cout << 1 << endl;
        return 0;
    }

    struct Pile {
        int x, y, c;
    };
    vector<Pile> piles(N);
    for (int i = 0; i < N; ++i) {
        cin >> piles[i].x >> piles[i].y >> piles[i].c;
    }

    // Node indexing for time-expanded graph:
    // S = 0, T_sink = 1
    // Pile i at time t: 
    //   P_in(i, t) = 2 + (t-1) * 2 * N + 2 * i
    //   P_out(i, t) = 2 + (t-1) * 2 * N + 2 * i + 1
    // We'll set a maximum T value.
    const int MAX_T = 2000; 
    const int INF = 1e9;
    int S = 0, T_sink = 1;
    
    // Total possible nodes for T=2000: 2 + 2000 * 2 * 50 = 200,002
    adj.resize(2 + MAX_T * 2 * N + 2);
    parent_node.resize(2 + MAX_T * 2 * N + 2);
    parent_edge.resize(2 + MAX_T * 2 * N + 2);

    auto P_in = [&](int i, int t) { return 2 + (t - 1) * 2 * N + 2 * i; };
    auto P_out = [&](int i, int t) { return 2 + (t - 1) * 2 * N + 2 * i + 1; };

    int total_flow = 0;

    // Time t = 1: tourists arrive at piles after their first jump
    for (int i = 0; i < N; ++i) {
        if (piles[i].y <= D) {
            add_edge(S, P_in(i, 1), INF);
            add_edge(P_in(i, 1), P_out(i, 1), piles[i].c);
        }
    }

    // Note: No one reaches the opposite bank at T=1 since W > D.

    for (int T = 2; T <= MAX_T; ++T) {
        // 1. Add current time layer's pile nodes and their capacities
        for (int i = 0; i < N; ++i) {
            add_edge(P_in(i, T), P_out(i, T), piles[i].c);
        }

        // 2. Add edges from piles at T-1 to piles (or sink) at T
        for (int i = 0; i < N; ++i) {
            // A tourist at pile i at T-1 can jump to pile j at T
            for (int j = 0; j < N; ++j) {
                long long dx = piles[i].x - piles[j].x;
                long long dy = piles[i].y - piles[j].y;
                if (dx * dx + dy * dy <= (long long)D * D) {
                    add_edge(P_out(i, T - 1), P_in(j, T), INF);
                }
            }
            // A tourist at pile i at T-1 can jump to the opposite bank and arrive at T
            if (W - piles[i].y <= D) {
                add_edge(P_out(i, T - 1), T_sink, INF);
            }
        }

        // 3. Update max flow incrementally using Ford-Fulkerson/BFS
        while (bfs(S, T_sink)) {
            int path_flow = INF;
            for (int v = T_sink; v != S; v = parent_node[v]) {
                int u = parent_node[v];
                int idx = parent_edge[v];
                path_flow = min(path_flow, adj[u][idx].capacity - adj[u][idx].flow);
            }
            for (int v = T_sink; v != S; v = parent_node[v]) {
                int u = parent_node[v];
                int idx = parent_edge[v];
                int rev_idx = adj[u][idx].rev;
                adj[u][idx].flow += path_flow;
                adj[v][rev_idx].flow -= path_flow;
            }
            total_flow += path_flow;
            if (total_flow >= M) {
                cout << T << endl;
                return 0;
            }
        }
    }

    cout << "IMPOSSIBLE" << endl;

    return 0;
}
```
