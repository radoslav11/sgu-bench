#include <iostream>

using namespace std;

/**
 * Problem Summary:
 * A "nearly prime number" is a positive integer that can be expressed as the product of
 * exactly two prime numbers (e.g., 4 = 2 * 2, 6 = 2 * 3).
 * We need to check N such numbers, where each number is up to 10^9 and N is up to 10.
 * 
 * Complexity:
 * The prime factorization approach for each number takes O(sqrt(X)) time, where X is the number.
 * For X = 10^9, sqrt(X) is approximately 31622.
 * With N = 10, the total number of iterations is approximately 316,220.
 * This easily fits within the tight 0.25-second time limit.
 */

// Function to determine if a positive integer is nearly prime
bool isNearlyPrime(long long n) {
    // A nearly prime number must be at least 4 (2 * 2)
    if (n <= 1) return false;
    
    int count = 0;
    // Iterate through all potential prime factors up to sqrt(n)
    for (long long i = 2; i * i <= n; ++i) {
        // If i divides n, it is a prime factor
        while (n % i == 0) {
            count++;
            n /= i;
            // If we have already found more than 2 prime factors, it cannot be nearly prime
            if (count > 2) return false;
        }
    }
    
    // If n is still greater than 1, the remaining n is a prime factor
    if (n > 1) {
        count++;
    }
    
    // The number is nearly prime if and only if it has exactly two prime factors
    return count == 2;
}

int main() {
    // Faster I/O for competitive programming
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    // Read the number of integers to check
    if (!(cin >> N)) return 0;

    // Process N integers from the input
    for (int i = 0; i < N; ++i) {
        long long x;
        if (!(cin >> x)) break;
        
        if (isNearlyPrime(x)) {
            cout << "Yes" << "\n";
        } else {
            cout << "No" << "\n";
        }
    }

    return 0;
}
