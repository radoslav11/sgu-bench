#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Problem Analysis:
 * The rule about the least reliable wire burning out is like maintaining a Maximum Spanning Forest (MSF)
 * where the reliability r_i is the key. To maximize the cost, when multiple wires have the same
 * reliability, we should prefer the ones with higher cost p_i. This means we want to find
 * an MSF using (r_i, p_i) as the key in descending order.
 * 
 * Let S be the set of wires in this MSF, and O be the remaining wires.
 * We want to find an order of soldering such that all wires in S remain and all in O burn out.
 * For a wire w in O, it will form a cycle with some wires in S. All wires s in that cycle
 * satisfy r_s >= r_w. To ensure w burns out, it must be the earliest among those with
 * the minimum reliability in its cycle. This can be achieved by soldering all wires in O
 * before wires in S for the same reliability value.
 * 
 * Sorting Strategy:
 * 1. Use Kruskal's algorithm with (r_i, p_i) descending to determine which wires are in S.
 * 2. Sort all wires by reliability (r_i) descending.
 * 3. For wires with the same reliability, solder wires in O before wires in S.
 * 4. The order within O or within S does not affect the outcome.
 */

struct Wire {
    int a, b, r, p, id;
    int type; // 0 for O (not in MSF), 1 for S (in MSF)
};

struct DSU {
    vector<int> parent;
    DSU(int n) {
        parent.resize(n);
        for (int i = 0; i < n; ++i) parent[i] = i;
    }
    int find(int i) {
        if (parent[i] == i) return i;
        return parent[i] = find(parent[i]);
    }
    bool unite(int i, int j) {
        int root_i = find(i);
        int root_j = find(j);
        if (root_i != root_j) {
            parent[root_i] = root_j;
            return true;
        }
        return false;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    vector<Wire> wires(n);
    vector<int> coords;
    for (int i = 0; i < n; ++i) {
        cin >> wires[i].a >> wires[i].b >> wires[i].r >> wires[i].p;
        wires[i].id = i + 1;
        wires[i].type = 1; // Initially assume it's in S
        coords.push_back(wires[i].a);
        coords.push_back(wires[i].b);
    }

    // Coordinate compression for DSU
    sort(coords.begin(), coords.end());
    coords.erase(unique(coords.begin(), coords.end()), coords.end());
    auto get_coord = [&](int x) {
        return lower_bound(coords.begin(), coords.end(), x) - coords.begin();
    };

    for (int i = 0; i < n; ++i) {
        wires[i].a = get_coord(wires[i].a);
        wires[i].b = get_coord(wires[i].b);
    }

    // Sort wires for Kruskal's: r desc, then p desc
    vector<Wire*> kruskal_wires(n);
    for (int i = 0; i < n; ++i) kruskal_wires[i] = &wires[i];

    sort(kruskal_wires.begin(), kruskal_wires.end(), [](Wire* x, Wire* y) {
        if (x->r != y->r) return x->r > y->r;
        return x->p > y->p;
    });

    DSU dsu(coords.size());
    long long total_cost = 0;
    for (int i = 0; i < n; ++i) {
        if (dsu.unite(kruskal_wires[i]->a, kruskal_wires[i]->b)) {
            total_cost += kruskal_wires[i]->p;
        } else {
            kruskal_wires[i]->type = 0; // Wire is in O
        }
    }

    // Sort all wires for output: r desc, then O before S
    sort(wires.begin(), wires.end(), [](const Wire& x, const Wire& y) {
        if (x.r != y.r) return x.r > y.r;
        return x.type < y.type; // type 0 (O) comes before type 1 (S)
    });

    cout << total_cost << "\n";
    for (int i = 0; i < n; ++i) {
        cout << wires[i].id << (i == n - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
