#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <queue>
#include <iomanip>

using namespace std;

// Structure to represent each ghost's sphere
struct Ghost {
    double x, y, z, r;
};

// Global variables for asylum dimensions, number of ghosts, and the ghosts themselves
double W, H, D;
int N;
Ghost ghosts[11];

/**
 * Function to calculate the minimum distance to any boundary or old ghost.
 * This is the radius of the largest sphere centered at (x, y, z) that stays
 * within the asylum and doesn't intersect any old ghosts.
 */
inline double h(double x, double y, double z) {
    double res = x;
    if (W - x < res) res = W - x;
    if (y < res) res = y;
    if (H - y < res) res = H - y;
    if (z < res) res = z;
    if (D - z < res) res = D - z;
    for (int i = 0; i < N; ++i) {
        double dx = x - ghosts[i].x;
        double dy = y - ghosts[i].y;
        double dz = z - ghosts[i].z;
        double d = sqrt(dx * dx + dy * dy + dz * dz);
        double dist_to_sphere = d - ghosts[i].r;
        if (dist_to_sphere < res) res = dist_to_sphere;
    }
    return res;
}

/**
 * Represents a sub-box in the octree.
 * Includes the boundaries and an upper bound of h(x, y, z) within this box.
 */
struct Box {
    double x1, x2, y1, y2, z1, z2, ub;
    bool operator<(const Box& other) const {
        return ub < other.ub;
    }
};

int main() {
    // Optimizing I/O performance
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    // Read asylum dimensions
    if (!(cin >> W >> H >> D)) return 0;
    // Read the number of ghosts
    if (!(cin >> N)) N = 0;
    // Read each ghost's center and radius
    for (int i = 0; i < N; ++i) {
        cin >> ghosts[i].x >> ghosts[i].y >> ghosts[i].z >> ghosts[i].r;
    }

    // Priority queue for the octree search, prioritizing boxes with high potential upper bounds
    priority_queue<Box> pq;
    
    // Initial starting point: center of the asylum
    double mid_x = W / 2.0;
    double mid_y = H / 2.0;
    double mid_z = D / 2.0;
    double initial_h = h(mid_x, mid_y, mid_z);
    
    double max_r = initial_h;
    double best_x = mid_x, best_y = mid_y, best_z = mid_z;

    // Initial upper bound for the whole asylum box using the Lipschitz property of h(x,y,z)
    double initial_half_diag = sqrt(W * W + H * H + D * D) / 2.0;
    pq.push({0, W, 0, H, 0, D, initial_h + initial_half_diag});

    // Octree exploration using a best-first search approach (similar to A*)
    int iterations = 0;
    // A limit on iterations to ensure we stay within the time limit
    while (!pq.empty() && iterations < 70000) {
        Box curr = pq.top();
        pq.pop();

        // If the upper bound of the current box is less than or equal to our best radius, 
        // then no point in this box or any other box in the priority queue can give a better result.
        if (curr.ub <= max_r + 1e-8) break;

        double cx = (curr.x1 + curr.x2) / 2.0;
        double cy = (curr.y1 + curr.y2) / 2.0;
        double cz = (curr.z1 + curr.z2) / 2.0;
        double cur_h = h(cx, cy, cz);

        // Update the best center and radius if a better one is found
        if (cur_h > max_r) {
            max_r = cur_h;
            best_x = cx;
            best_y = cy;
            best_z = cz;
        }

        double dx = (curr.x2 - curr.x1) / 2.0;
        double dy = (curr.y2 - curr.y1) / 2.0;
        double dz = (curr.z2 - curr.z1) / 2.0;
        double half_diag = sqrt(dx * dx + dy * dy + dz * dz);

        // If the box is extremely small, stop splitting
        if (half_diag < 1e-8) continue;

        // Split the box into 8 sub-boxes and add them to the priority queue
        for (int i = 0; i < 8; ++i) {
            double nx1 = (i & 1) ? cx : curr.x1;
            double nx2 = (i & 1) ? curr.x2 : cx;
            double ny1 = (i & 2) ? cy : curr.y1;
            double ny2 = (i & 2) ? curr.y2 : cy;
            double nz1 = (i & 4) ? cz : curr.z1;
            double nz2 = (i & 4) ? curr.z2 : cz;

            double ncx = (nx1 + nx2) / 2.0;
            double ncy = (ny1 + ny2) / 2.0;
            double ncz = (nz1 + nz2) / 2.0;
            double nh = h(ncx, ncy, ncz);
            
            double ndx = (nx2 - nx1) / 2.0;
            double ndy = (ny2 - ny1) / 2.0;
            double ndz = (nz2 - nz1) / 2.0;
            double n_half_diag = sqrt(ndx * ndx + ndy * ndy + ndz * ndz);
            double n_ub = nh + n_half_diag;

            // Only add sub-boxes that have the potential to contain a better radius
            if (n_ub > max_r) {
                pq.push({nx1, nx2, ny1, ny2, nz1, nz2, n_ub});
            }
        }
        iterations++;
    }

    // Final output: coordinates of the point where the maximum radius ghost can stay
    cout << fixed << setprecision(10) << best_x << " " << best_y << " " << best_z << endl;

    return 0;
}
