#include <iostream>
#include <vector>
#include <string>
#include <limits>

using namespace std;

/**
 * Problem 368. Tests
 * 
 * The goal is to process Peter's test results and determine which of his 
 * university applications were accepted based on conversion rules for 
 * each faculty's subjects and the required sum of grades for each specialty.
 * 
 * Time complexity: O(N * S * K), where N is the number of faculties, 
 * S is the number of specialties per faculty, and K is the number of 
 * required subjects per specialty.
 * 
 * Constraints: N, M, S, K <= 100, which makes the complexity roughly 
 * 1,000,000 operations, well within the 0.25s limit.
 */

struct Application {
    string faculty;
    string specialty;
};

int main() {
    // Optimize standard I/O for faster execution
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    // Read Peter's test scores for M subjects
    vector<int> B(M);
    for (int i = 0; i < M; ++i) {
        cin >> B[i];
    }

    // To switch from cin >> to getline, we must consume the newline character
    string dummy;
    getline(cin, dummy);

    vector<Application> accepted;

    for (int i = 0; i < N; ++i) {
        // Read the faculty name
        string faculty_name;
        if (!getline(cin, faculty_name)) break;
        // Handle Windows-style \r character if present
        if (!faculty_name.empty() && faculty_name.back() == '\r') faculty_name.pop_back();

        // Read the number of exams K and the required subject numbers
        int K;
        if (!(cin >> K)) break;
        vector<int> subjects(K);
        for (int j = 0; j < K; ++j) {
            cin >> subjects[j];
        }

        // Read the K triplets of thresholds (Xi, Yi, Zi)
        vector<int> X(K), Y(K), Z(K);
        for (int j = 0; j < K; ++j) {
            cin >> X[j] >> Y[j] >> Z[j];
        }

        // Read the number of specialties S
        int S;
        cin >> S;
        // Consume the rest of the line after S before reading specialty names
        getline(cin, dummy);

        for (int j = 0; j < S; ++j) {
            // Read the specialty name
            string specialty_name;
            if (!getline(cin, specialty_name)) break;
            if (!specialty_name.empty() && specialty_name.back() == '\r') specialty_name.pop_back();

            // Read the required sum of grades
            int required_sum;
            cin >> required_sum;
            // Consume the rest of the line after required_sum
            getline(cin, dummy);

            // Calculate the sum of grades Peter gets for this specialty
            int current_sum = 0;
            for (int k = 0; k < K; ++k) {
                // Subjects are given as 1-indexed
                int score = B[subjects[k] - 1];
                int grade;
                // Apply conversion rules:
                // T < Xi: 2
                // Xi <= T < Yi: 3
                // Yi <= T < Zi: 4
                // Zi <= T: 5
                if (score < X[k]) grade = 2;
                else if (score < Y[k]) grade = 3;
                else if (score < Z[k]) grade = 4;
                else grade = 5;
                current_sum += grade;
            }

            // Check if Peter's application is accepted
            if (current_sum >= required_sum) {
                accepted.push_back({faculty_name, specialty_name});
            }
        }
    }

    // Final Output
    if (accepted.empty()) {
        cout << "Army" << endl;
    } else {
        for (const auto& app : accepted) {
            cout << app.faculty << " " << app.specialty << "\n";
        }
    }

    return 0;
}
