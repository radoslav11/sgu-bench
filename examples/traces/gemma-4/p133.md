```cpp
#include <cstdio>
#include <vector>
#include <algorithm>

/**
 * Problem Analysis:
 * An outpost i is redundant if there exists some outpost j such that 
 * Ak_j < Ak_i and Bk_i < Bk_j. This means the interval [Ak_i, Bk_i] 
 * is strictly contained within the interval [Ak_j, Bk_j].
 *
 * Since all Ak and Bk are distinct, the condition simplifies to finding
 * whether any outpost j with Ak_j < Ak_i has a Bk_j > Bk_i.
 *
 * Approach:
 * 1. Store all outposts as pairs (Ak, Bk).
 * 2. Sort the outposts based on Ak in ascending order.
 * 3. Iterate through the sorted outposts. For each outpost i, we want to 
 *    check if there is any j < i such that Bk_j > Bk_i.
 * 4. Since we are iterating in increasing order of Ak, any j < i will 
 *    automatically satisfy Ak_j < Ak_i.
 * 5. To check if there's any j < i with Bk_j > Bk_i, we only need to 
 *    track the maximum Bk seen so far among all outposts j < i.
 * 6. If max_Bk_seen_so_far > Bk_i, then the current outpost i is redundant.
 *
 * Complexity:
 * Time: O(N log N) due to sorting, where N is up to 16,000.
 * Memory: O(N) to store the outposts, well within the 4096 KB limit.
 */

using namespace std;

// Structure to represent an outpost interval [a, b]
struct Outpost {
    int a, b;
};

// Comparator to sort outposts by their starting point Ak in ascending order
bool compareOutposts(const Outpost& o1, const Outpost& o2) {
    return o1.a < o2.a;
}

int main() {
    int n;
    // Read the number of outposts
    if (scanf("%d", &n) != 1) {
        return 0;
    }

    // Using a vector to store the outposts
    vector<Outpost> outposts(n);
    for (int i = 0; i < n; ++i) {
        // Read Ak and Bk for each outpost
        if (scanf("%d %d", &outposts[i].a, &outposts[i].b) != 2) {
            break;
        }
    }

    // Sort outposts by Ak
    sort(outposts.begin(), outposts.end(), compareOutposts);

    int redundant_count = 0;
    // Initialize max_b to a value smaller than any possible Bk
    int max_b = -1;

    // Linear scan to count redundant outposts
    for (int i = 0; i < n; ++i) {
        // If the current outpost's Bk is smaller than the largest Bk 
        // seen so far, it means it's contained within an earlier outpost's interval.
        if (max_b > outposts[i].b) {
            redundant_count++;
        } else {
            // Otherwise, update the maximum Bk seen so far
            max_b = outposts[i].b;
        }
    }

    // Print the total count of redundant outposts
    printf("%d\n", redundant_count);

    return 0;
}
```
