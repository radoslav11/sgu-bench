#include <iostream>
#include <string>

using namespace std;

/**
 * Problem Analysis:
 * The task is to convert singular nouns to plural forms based on a specific set of provided rules.
 * 
 * Rules:
 * 1. If a noun ends with 'ch', 'x', 's', or 'o', add 'es' to the end.
 * 2. If a noun ends with 'f' or 'fe', replace the 'f' or 'fe' with 'ves'.
 * 3. If a noun ends with 'y', replace the 'y' with 'ies'.
 * 4. Otherwise, add 's' to the end.
 * 
 * Constraints:
 * - n (number of words) is between 1 and 10.
 * - Word length is between 2 and 25 lowercase Latin letters.
 * - Time limit: 0.25 seconds.
 * - Memory limit: 65536 kilobytes.
 * 
 * Complexity:
 * - Time Complexity: O(n * L), where n is the number of words and L is the maximum length of a word.
 *   Since n <= 10 and L <= 25, this is extremely efficient.
 * - Space Complexity: O(L) to store each word.
 */

int main() {
    // Optimize input/output
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    // Read the number of words
    if (!(cin >> n)) return 0;

    while (n--) {
        string s;
        // Read each noun
        if (!(cin >> s)) break;

        int len = s.length();

        // Applying Rule 1: Ends with ch, x, s, o
        // A noun ending in 'ch' is checked first to distinguish it from other endings.
        if (len >= 2 && s.substr(len - 2) == "ch") {
            cout << s << "es" << "\n";
        } 
        else if (s.back() == 'x' || s.back() == 's' || s.back() == 'o') {
            cout << s << "es" << "\n";
        }
        // Applying Rule 2: Ends with f or fe
        // We check for 'fe' before 'f' to ensure the correct replacement of the suffix.
        else if (len >= 2 && s.substr(len - 2) == "fe") {
            // Replace 'fe' with 'ves'
            cout << s.substr(0, len - 2) << "ves" << "\n";
        } 
        else if (s.back() == 'f') {
            // Replace 'f' with 'ves'
            cout << s.substr(0, len - 1) << "ves" << "\n";
        }
        // Applying Rule 3: Ends with y
        else if (s.back() == 'y') {
            // Replace 'y' with 'ies'
            cout << s.substr(0, len - 1) << "ies" << "\n";
        }
        // Applying Rule 4: All other cases
        else {
            // Add 's' to the end
            cout << s << "s" << "\n";
        }
    }

    return 0;
}
