#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>

using namespace std;

/**
 * Problem Analysis:
 * Two circles A and B move on a flat table. They only change speeds during a collision.
 * A collision changes the velocity only in the direction of the vector AB.
 * Total momentum (m_A*v_A + m_B*v_B) and total kinetic energy (1/2*m_A*|v_A|^2 + 1/2*m_B*|v_B|^2) 
 * are conserved. This means the collision is a standard 1D elastic collision along the normal 
 * direction (the line connecting the centers of the circles).
 * 
 * Algorithm:
 * 1. Find the earliest collision time t_coll within the remaining time t_rem.
 * 2. A collision occurs when the distance between centers equals r_A + r_B.
 *    Let P_0 = B_pos - A_pos and V = v_B - v_A.
 *    The condition |P_0 + V*t|^2 = (r_A + r_B)^2 leads to a quadratic equation:
 *    |V|^2 * t^2 + 2 * (P_0 . V) * t + (|P_0|^2 - (r_A + r_B)^2) = 0.
 * 3. Solve the quadratic equation for the smallest positive t_coll.
 * 4. If a collision occurs within t_rem:
 *    - Move circles to their positions at t_coll.
 *    - Update velocities according to the 1D elastic collision formulas.
 *    - Subtract t_coll from t_rem and repeat.
 * 5. If no collision occurs within t_rem:
 *    - Move circles to their final positions at t_rem.
 *    - End the process.
 * 
 * Since the relative velocity along the normal is reversed (with mass-dependent adjustment), 
 * the circles move apart after a collision. For two circles moving in straight lines, 
 * at most one collision can occur.
 */

struct Vec {
    long double x, y;
    Vec operator+(const Vec& other) const { return {x + other.x, y + other.y}; }
    Vec operator-(const Vec& other) const { return {x - other.x, y - other.y}; }
    Vec operator*(long double scalar) const { return {x * scalar, y * scalar}; }
    long double dot(const Vec& other) const { return x * other.x + y * other.y; }
    long double normSq() const { return x * x + y * y; }
    long double norm() const { return sqrt(normSq()); }
};

struct Circle {
    Vec pos, vel;
    long double r, m;
};

int main() {
    // Optimize input/output
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    Circle A, B;
    long double rA, mA, rB, mB;
    
    // Read data for Circle A
    if (!(cin >> A.pos.x >> A.pos.y >> A.vel.x >> A.vel.y >> rA >> mA)) return 0;
    A.r = rA; A.m = mA;
    
    // Read data for Circle B
    if (!(cin >> B.pos.x >> B.pos.y >> B.vel.x >> B.vel.y >> rB >> mB)) return 0;
    B.r = rB; B.m = mB;
    
    // Read the time period t
    long double t_target;
    if (!(cin >> t_target)) return 0;

    long double t_rem = t_target;
    while (t_rem > 1e-12) {
        Vec p0 = B.pos - A.pos;
        Vec v_rel = B.vel - A.vel;
        long double a = v_rel.normSq();
        long double b = 2.0 * p0.dot(v_rel);
        long double c = p0.normSq() - (A.r + B.r) * (A.r + B.r);

        bool collision = false;
        long double t_coll = 0;

        // Solve the quadratic equation at^2 + bt + c = 0 for t > 0
        if (a > 1e-20) {
            long double D = b * b - 4.0 * a * c;
            if (D < 0) D = 0; // Handle floating point precision
            // For the distance to be decreasing, we need b < 0.
            if (b < 0) {
                t_coll = (-b - sqrt(D)) / (2.0 * a);
                if (t_coll >= 0 && t_coll <= t_rem + 1e-13) {
                    collision = true;
                }
            }
        }

        if (collision) {
            // Smallest positive t_coll might be slightly negative due to precision
            if (t_coll < 0) t_coll = 0;
            
            // Move to collision position
            A.pos = A.pos + A.vel * t_coll;
            B.pos = B.pos + B.vel * t_coll;
            
            // Collision occurs along the line connecting centers. 
            // The normal vector is the unit vector between the centers.
            Vec diff = B.pos - A.pos;
            long double dist = diff.norm();
            Vec n_unit = (dist > 1e-15) ? diff * (1.0 / dist) : Vec{1, 0};

            // Component of velocities along the normal
            long double vAn = A.vel.dot(n_unit);
            long double vBn = B.vel.dot(n_unit);
            
            // 1D elastic collision formulas for normal velocity components
            long double vAn_new = (vAn * (A.m - B.m) + 2.0 * B.m * vBn) / (A.m + B.m);
            long double vBn_new = (vBn * (B.m - A.m) + 2.0 * A.m * vAn) / (A.m + B.m);

            // Update velocities by adding the change in normal velocity
            A.vel = A.vel + n_unit * (vAn_new - vAn);
            B.vel = B.vel + n_unit * (vBn_new - vBn);

            t_rem -= t_coll;
        } else {
            // No collision within remaining time, move to end
            A.pos = A.pos + A.vel * t_rem;
            B.pos = B.pos + B.vel * t_rem;
            t_rem = 0;
        }
    }

    // Final output with required precision
    cout << fixed << setprecision(4) << (double)A.pos.x << " " << (double)A.pos.y << " " << (double)A.vel.x << " " << (double)A.vel.y << "\n";
    cout << fixed << setprecision(4) << (double)B.pos.x << " " << (double)B.pos.y << " " << (double)B.vel.x << " " << (double)B.vel.y << endl;

    return 0;
}
