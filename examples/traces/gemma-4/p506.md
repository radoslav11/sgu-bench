```cpp
#include <iostream>
#include <string>
#include <vector>

using namespace std;

/**
 * Problem: Subsequences Of Substrings
 * We are given a text T and a message M. We need to find the number of ways 
 * (i, k) to remove i letters from the beginning and k letters from the end
 * of T such that the remaining substring T[i...n-1-k] contains M as a subsequence.
 * 
 * Let n = |T|. The remaining substring is T[i...j] where j = n-1-k.
 * A way (i, j) is valid if 0 <= i <= j < n and M is a subsequence of T[i...j].
 * For a fixed start index i, there is a minimum ending index j_min(i) 
 * such that M is a subsequence of T[i...j_min(i)]. 
 * If such a j_min(i) exists, then for all j in [j_min(i), n-1], 
 * the substring T[i...j] will also contain M as a subsequence.
 * 
 * The number of such j's is (n - 1) - j_min(i) + 1 = n - j_min(i).
 * Total number of ways is the sum over all i of (n - j_min(i)).
 * 
 * Constraints: |T| <= 10^5, |M| <= 100.
 * Complexity: O(n * 26) for precomputing next character occurrences and 
 * O(n * m) for calculating the total ways.
 * Total time complexity is roughly O(n * (m + 26)), which is ~1.26 * 10^7 operations.
 * This should fit within the 0.5s time limit.
 */

const int MAXN = 100005;
const int SIGMA = 26;
int next_occ[MAXN][SIGMA];

int main() {
    // Speed up standard I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string text, message;
    if (!(cin >> text >> message)) return 0;

    int n = text.length();
    int m = message.length();

    // Precompute next_occ[i][c]: the smallest index j >= i such that text[j] == c
    for (int c = 0; c < SIGMA; ++c) {
        next_occ[n][c] = n;
    }
    for (int i = n - 1; i >= 0; --i) {
        for (int c = 0; c < SIGMA; ++c) {
            next_occ[i][c] = next_occ[i + 1][c];
        }
        next_occ[i][text[i] - 'a'] = i;
    }

    long long total_ways = 0;
    for (int i = 0; i < n; ++i) {
        int current_pos = i;
        int last_pos = -1;
        bool possible = true;

        // For a fixed start i, greedily find the first possible indices 
        // for characters in M.
        for (int k = 0; k < m; ++k) {
            int char_idx = message[k] - 'a';
            
            // Ensure current_pos doesn't go out of bounds
            if (current_pos >= n) {
                possible = false;
                break;
            }

            int found_pos = next_occ[current_pos][char_idx];
            if (found_pos == n) {
                possible = false;
                break;
            }

            last_pos = found_pos;
            current_pos = found_pos + 1;
        }

        // If M is a subsequence of T starting from index i, 
        // any substring T[i...j] where j >= last_pos is valid.
        if (possible) {
            total_ways += (long long)(n - last_pos);
        }
    }

    cout << total_ways << endl;

    return 0;
}
```
