#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>

using namespace std;

/**
 * Problem Analysis:
 * The goal is to maximize sqrt(sum_{i in L} Ai^2), which is equivalent to maximizing sum_{i in L} Ai^2,
 * where L is the set of sons who marry a girl they like.
 * This is a maximum weight matching problem where weights are assigned to the nodes (the sons) 
 * of one side of the bipartite graph.
 * Such problems can be solved using a greedy approach because the sets of nodes in a bipartite 
 * graph that can be covered by a matching form a transversal matroid.
 * In a matroid, the greedy algorithm (sorting nodes by weight and adding each if it maintains 
 * the independent property) correctly finds the maximum weight basis.
 * 
 * Algorithm:
 * 1. Sort sons in descending order of their love value Ai.
 * 2. For each son, try to include him in the current matching using an augmenting path (DFS).
 * 3. If an augmenting path is found, update the matching.
 * 4. Output the final matching of girls to sons.
 * 
 * Complexity:
 * Sorting takes O(N log N).
 * The greedy matching process performs at most N DFS calls, and each DFS takes O(E), where E is the 
 * number of edges. Total complexity is O(N * E), which is at most O(N^3).
 * For N = 400, N^3 = 64,000,000. This should pass within the 0.25s time limit with careful implementation.
 */

const int MAXN = 405;

struct Son {
    int id;
    int love;
};

// Sort sons by love value in descending order.
bool compareSons(const Son& a, const Son& b) {
    if (a.love != b.love) {
        return a.love > b.love;
    }
    // Deterministic tie-breaker for stability.
    return a.id < b.id;
}

int n;
int love_vals[MAXN];
int adj[MAXN][MAXN];
int deg[MAXN];
int matchG[MAXN]; // matchG[girl] stores the son ID matched to the girl
int matchS[MAXN]; // matchS[son] stores the girl ID matched to the son
int visited[MAXN]; // used in DFS to avoid cycles and repeated visits
int timer = 0; // used to efficiently reset the visited array

/**
 * Standard DFS for finding an augmenting path in a bipartite graph.
 * @param u The current son's ID.
 * @return true if an augmenting path is found, false otherwise.
 */
bool dfs(int u) {
    for (int i = 0; i < deg[u]; ++i) {
        int v = adj[u][i];
        if (visited[v] != timer) {
            visited[v] = timer;
            // If girl v is unmatched or the son matched with her can be reassigned.
            if (matchG[v] == 0 || dfs(matchG[v])) {
                matchG[v] = u;
                matchS[u] = v;
                return true;
            }
        }
    }
    return false;
}

int main() {
    // Speed up I/O operations.
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> n)) return 0;

    for (int i = 1; i <= n; ++i) {
        cin >> love_vals[i];
    }

    for (int i = 1; i <= n; ++i) {
        int k;
        cin >> k;
        deg[i] = k;
        for (int j = 0; j < k; ++j) {
            cin >> adj[i][j];
        }
    }

    // Prepare the list of sons for sorting.
    Son sortedSons[MAXN];
    for (int i = 1; i <= n; ++i) {
        sortedSons[i - 1] = {i, love_vals[i]};
    }

    // Sort sons descending by their love values.
    sort(sortedSons, sortedSons + n, compareSons);

    // Initialize matchings and visited array.
    memset(matchG, 0, sizeof(matchG));
    memset(matchS, 0, sizeof(matchS));
    memset(visited, 0, sizeof(visited));

    // Greedy algorithm for transversal matroid.
    for (int i = 0; i < n; ++i) {
        int u = sortedSons[i].id;
        timer++;
        dfs(u);
    }

    // Output the girl each son is matched with (0 if not matched).
    for (int i = 1; i <= n; ++i) {
        cout << matchS[i] << (i == n ? "" : " ");
    }
    cout << endl;

    return 0;
}
