#include <iostream>
#include <vector>

/**
 * Problem: Find a "beautiful graph" with the maximum number of edges for a given n.
 * A "beautiful graph" is defined as an undirected graph with no loops and 
 * no two simple cycles sharing exactly one edge.
 * 
 * Analysis:
 * A 2-connected beautiful graph is either a triangle (K3) or a complete bipartite
 * graph K(2, m) for m >= 2.
 * 
 * For K(2, m):
 * - V = m + 2
 * - E = 2 * m
 * - For n vertices, if n >= 4, we can use K(2, n-2) to get E = 2 * (n-2) = 2n - 4.
 * 
 * For K3:
 * - V = 3
 * - E = 3
 * 
 * Let's check small cases of n:
 * - n = 1: V = 1, E = 0 (no edges)
 * - n = 2: V = 2, E = 1 (single edge)
 * - n = 3: V = 3, E = 3 (triangle K3)
 * - n = 4: V = 4, E = 4 (K(2, 2), which is a 4-cycle)
 * - n = 5: V = 5, E = 6 (K(2, 3))
 * - n = 6: V = 6, E = 8 (K(2, 4))
 * 
 * For n >= 4, the maximum number of edges is 2n - 4 using K(2, n-2).
 * For n = 3, the maximum is 3 (K3).
 * For n = 2, the maximum is 1 (edge).
 * For n = 1, the maximum is 0 (no edges).
 */

int main() {
    // Speed up standard I/O
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int n;
    if (!(std::cin >> n)) return 0;

    if (n == 1) {
        std::cout << "1 0" << std::endl;
    } else if (n == 2) {
        std::cout << "2 1" << std::endl;
        std::cout << "1 2" << std::endl;
    } else if (n == 3) {
        // A cycle of 3 vertices is beautiful and has the most edges for n=3.
        std::cout << "3 3" << std::endl;
        std::cout << "1 2" << std::endl;
        std::cout << "2 3" << std::endl;
        std::cout << "3 1" << std::endl;
    } else {
        // For n >= 4, K(2, n-2) is a beautiful graph with 2n-4 edges.
        // Let vertex 1 and vertex 2 be the two vertices in the first partition of the bipartite graph.
        // The remaining n-2 vertices (3 to n) are in the second partition.
        // Each vertex in the second partition is connected to both vertex 1 and vertex 2.
        int V = n;
        int E = 2 * (n - 2);
        std::cout << V << " " << E << "\n";
        for (int i = 3; i <= n; ++i) {
            std::cout << "1 " << i << "\n";
            std::cout << "2 " << i << "\n";
        }
    }

    return 0;
}
