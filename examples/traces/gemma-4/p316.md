#include <iostream>
#include <vector>

/**
 * Problem Analysis:
 * - N tanks, each starts with 100 HP and 0 score.
 * - M shots are recorded in the log.
 * - For each shot:
 *   - Target's HP decreases by 8.
 *   - If the target's HP was > 0 before the shot, the shooter gets 3 points.
 * - Only "active" tanks (HP > 0) can shoot.
 * - At the end, tanks with HP > 0 receive bonus score: current HP / 2.
 * - Final HP and score must be printed for each tank.
 * 
 * Constraints:
 * - 2 <= N <= 6
 * - 1 <= M <= 4000
 * - Time limit: 0.25s (needs to be efficient)
 * - Memory limit: 64MB
 * 
 * Complexity:
 * - Time: O(M), which is at most 4000 operations.
 * - Space: O(N), which is at most 6.
 * 
 * Logic:
 * - We use two arrays to store the hit points and scores of each tank.
 * - We iterate through each shot, updating the scores and HP accordingly.
 * - Finally, we calculate the bonus for tanks with positive HP and output the results.
 */

using namespace std;

int main() {
    // Faster I/O for the tight time limit
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M;
    if (!(cin >> N >> M)) {
        return 0;
    }

    // Use N + 1 to make indexing 1-based as per problem description
    // All tanks start with 100 hit points and 0 score.
    vector<int> hp(N + 1, 100);
    vector<int> score(N + 1, 0);

    for (int i = 0; i < M; ++i) {
        int shooter, target;
        if (!(cin >> shooter >> target)) break;

        // Rule: If target's HP before the shot was > 0, the shooter gets 3 points.
        if (hp[target] > 0) {
            score[shooter] += 3;
        }
        
        // Rule: Each shot always reduces the target tank's HP by 8.
        hp[target] -= 8;
    }

    // Final calculations and output
    for (int i = 1; i <= N; ++i) {
        // Rule: At the end of the match each tank with positive hit points 
        // additionally receives the number of score points equal to half of its hit points.
        // Since output expects integer scores, we use integer division.
        if (hp[i] > 0) {
            score[i] += hp[i] / 2;
        }
        
        // Output HP and score for each tank
        cout << hp[i] << " " << score[i] << "\n";
    }

    return 0;
}
