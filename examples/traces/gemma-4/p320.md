```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <queue>

using namespace std;

/**
 * Problem Analysis:
 * A square is dangerous if:
 * 1. It is part of a "big leader" (a connected component of same-numbered squares with size >= K).
 * 2. It is "trapped" by a big leader (all paths from the square to the boundary pass through the big leader).
 * 
 * Approach:
 * 1. Identify all connected components (leaders) and their sizes using BFS.
 * 2. Build a dual graph where each node is a component, and an edge exists between 
 *    two components if they are adjacent in the grid.
 * 3. Add a special "Boundary" node that is connected to all components on the edge of the grid.
 * 4. A component is "trapped" if it is separated from the Boundary node by a big leader 
 *    acting as an articulation point in the dual graph.
 * 5. Use Tarjan's articulation point algorithm to find these separated components.
 * 6. Count all squares that belong to a big leader or a trapped component.
 */

const int MAXN = 505;
const int MAX_SQUARES = 250005;
const int MAX_EDGES = 2000005;

int N, M, K;
int grid[MAXN][MAXN];
int comp_id[MAXN][MAXN];
int comp_size[MAX_SQUARES];
bool is_big[MAX_SQUARES];
int comp_count = 0;

int head[MAX_SQUARES], to_edge[MAX_EDGES], nxt_edge[MAX_EDGES], edge_cnt = 0;

void add_edge(int u, int v) {
    to_edge[edge_cnt] = v;
    nxt_edge[edge_cnt] = head[u];
    head[u] = edge_cnt++;
}

int tin[MAX_SQUARES], tout[MAX_SQUARES], low[MAX_SQUARES], timer = 0;
int mark[MAX_SQUARES];

void dfs_articulation(int u, int p, int p_node) {
    tin[u] = low[u] = ++timer;
    for (int i = head[u]; i != -1; i = nxt_edge[i]) {
        int v = to_edge[i];
        if (v == p) continue;
        if (tin[v]) {
            low[u] = min(low[u], tin[v]);
        } else {
            dfs_articulation(v, u, p_node);
            low[u] = min(low[u], low[v]);
            if (low[v] >= tin[u]) {
                // u separates the subtree of v from the root (p_node)
                if (u != p_node && is_big[u]) {
                    mark[tin[v]]++;
                    mark[tout[v] + 1]--;
                }
            }
        }
    }
    tout[u] = timer;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> N >> M >> K)) return 0;

    for (int i = 0; i < N; ++i) {
        string row;
        cin >> row;
        for (int j = 0; j < M; ++j) {
            grid[i][j] = row[j] - '0';
            comp_id[i][j] = -1;
        }
    }

    // 1. Identify components
    int dr[] = {-1, 1, 0, 0};
    int dc[] = {0, 0, -1, 1};
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < M; ++j) {
            if (comp_id[i][j] == -1) {
                int current_comp = comp_count++;
                queue<pair<int, int>> q;
                q.push({i, j});
                comp_id[i][j] = current_comp;
                int size = 0;
                int leader_val = grid[i][j];

                while (!q.empty()) {
                    pair<int, int> curr = q.front();
                    q.pop();
                    size++;
                    for (int d = 0; d < 4; ++d) {
                        int ni = curr.first + dr[d];
                        int nj = curr.second + dc[d];
                        if (ni >= 0 && ni < N && nj >= 0 && nj < M &&
                            comp_id[ni][nj] == -1 && grid[ni][nj] == leader_val) {
                            comp_id[ni][nj] = current_comp;
                            q.push({ni, nj});
                        }
                    }
                }
                comp_size[current_comp] = size;
                if (size >= K) is_big[current_comp] = true;
            }
        }
    }

    // 2. Build dual graph
    for (int i = 0; i < comp_count + 1; ++i) head[i] = -1;
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < M; ++j) {
            int u = comp_id[i][j];
            // Check neighbors for edges in dual graph
            if (i + 1 < N) {
                int v = comp_id[i + 1][j];
                if (u != v) {
                    add_edge(u, v);
                    add_edge(v, u);
                }
            }
            if (j + 1 < M) {
                int v = comp_id[i][j + 1];
                if (u != v) {
                    add_edge(u, v);
                    add_edge(v, u);
                }
            }
            // Boundary connection
            if (i == 0 || i == N - 1 || j == 0 || j == M - 1) {
                add_edge(u, comp_count);
                add_edge(comp_count, u);
            }
        }
    }

    // 3. Find trapped components using articulation points
    int boundary_node = comp_count;
    dfs_articulation(boundary_node, -1, boundary_node);

    // Prefix sum to determine which components are trapped
    for (int i = 1; i <= timer; ++i) {
        mark[i] += mark[i - 1];
    }

    // 4. Count dangerous squares
    int dangerous_count = 0;
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < M; ++j) {
            int cid = comp_id[i][j];
            if (is_big[cid] || (mark[tin[cid]] > 0)) {
                dangerous_count++;
            }
        }
    }

    cout << dangerous_count << endl;

    return 0;
}
```
