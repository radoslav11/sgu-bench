            Node* res = new Node{'1', 0, {}};

            consume(v);

        }

            consume('~');
            Node* child = parse_negation();
            Node* res = new Node{'~', 0, {child}};
            return res;
        }
        return parse_atom();
    }

    Node* parse_conjunction() {
        Node* node = parse_negation();
        while (peek() == '&') {
            consume('&');
            Node* right = parse_negation();
            node = new Node{'&', 0, {node, right}};
        }
        return node;
    }

    Node* parse_disjunction() {
        Node* node = parse_conjunction();
        while (peek() == '|') {
            consume('|');
            Node* right = parse_conjunction();
            node = new Node{'|', 0, {node, right}};
        }
        return node;
    }

    Node* parse_implication() {
        Node* node = parse_disjunction();
        if (peek() == '=' && peek_two() == "=>") {
            consume("=>");
            Node* right = parse_implication();
            node = new Node{'>', 0, {node, right}};
        }
        return node;
    }

    Node* parse_equivalence() {
        vector<Node*> es;
        es.push_back(parse_implication());
        while (true) {
            if (peek() == '=' && (peek_two() == "==" || peek_two() == "=")) {
                if (peek_two() == "==") consume("==");
                else consume("=");
                es.push_back(parse_implication());
            } else break;
        }
        if (es.size() == 1) return es[0];
        Node* res = new Node{'E', 0, es};
        return res;
    }

    Node* parse() {
        return parse_equivalence();
    }
};
