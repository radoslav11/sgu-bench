#include <iostream>
#include <vector>
#include <bitset>

using namespace std;

/**
 * Problem Analysis:
 * Find pairs (A, B) of prime numbers such that A <= B, A + B is prime, and A + B <= N.
 * 
 * Let A and B be prime numbers.
 * If both A and B are odd primes, A + B will be an even number greater than 2, 
 * which means A + B cannot be prime.
 * If both A and B are even primes, A = 2 and B = 2, but A + B = 4, which is not prime.
 * Therefore, one of the primes must be even and the other must be odd.
 * Since the only even prime number is 2, and we are given A <= B, it must be that A = 2.
 * 
 * The problem then reduces to finding all odd prime numbers B such that 2 + B is also prime,
 * with the constraint that 2 + B <= N (or B <= N - 2).
 * 
 * Constraints:
 * N <= 10^6.
 * Time limit: 0.25 sec (very tight).
 * Memory limit: 4096 KB (about 4 MB).
 * 
 * Implementation details:
 * 1. Use a sieve of Eratosthenes to find all primes up to N.
 * 2. Use std::bitset to store primality to efficiently stay within the 4 MB memory limit.
 *    std::bitset<1000001> will use approximately 125 KB.
 * 3. Iterate through all candidate B's from 3 up to N-2 and check if B and B+2 are both prime.
 * 4. Print the count and the pairs.
 */

// bitset stores 1 bit per element, so 10^6 bits = 125,000 bytes (approx 0.125 MB).
bitset<1000001> is_prime;

/**
 * Standard Sieve of Eratosthenes to find all primes up to n.
 */
void sieve(int n) {
    is_prime.set();
    is_prime[0] = is_prime[1] = 0;
    for (int p = 2; p * p <= n; p++) {
        if (is_prime[p]) {
            for (int i = p * p; i <= n; i += p)
                is_prime[i] = 0;
        }
    }
}

int main() {
    // Synchronizing with stdio is disabled for faster I/O, which is critical for 0.25s limit.
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    if (!(cin >> N)) return 0;

    // The smallest prime sum for two primes A <= B is 2 + 3 = 5.
    if (N < 5) {
        cout << 0 << "\n";
        return 0;
    }

    // Find primes up to N.
    sieve(N);

    // Find all odd primes B such that B+2 is prime and B+2 <= N.
    // Since we found A must be 2, we store pairs as {2, B}.
    vector<pair<int, int>> results;
    // B must be at least 3 to be an odd prime. 
    // B+2 <= N implies B <= N-2.
    for (int b = 3; b <= N - 2; ++b) {
        if (is_prime[b] && is_prime[b + 2]) {
            results.push_back({2, b});
        }
    }

    // Output the number of pairs found.
    cout << results.size() << "\n";
    // Output each pair.
    for (const auto& p : results) {
        cout << p.first << " " << p.second << "\n";
    }

    return 0;
}
