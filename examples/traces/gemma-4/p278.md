```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <iomanip>
#include <cmath>
#include <map>

using namespace std;

/**
 * The problem asks to maximize the total intensity sum(m_i * c_i)
 * subject to:
 *   sum(m_i * a_i) <= A
 *   sum(m_i * b_i) <= B
 *   m_i >= 0
 * This is a linear programming problem with two constraints.
 * The dual problem is:
 * Minimize A*y1 + B*y2 subject to:
 *   a_i*y1 + b_i*y2 >= c_i for all i
 *   y1 >= 0, y2 >= 0
 *
 * For each fuel i, the constraint is a line a_i*y1 + b_i*y2 = c_i.
 * We want to find the minimum of the linear objective function over the region 
 * that lies above all these lines and above y1=0, y2=0.
 * The boundary of this region is the upper envelope of these lines.
 */

struct Line {
    double a, b, c;
    double m, d; // y = m*x + d
};

struct Fuel {
    int a, b, c;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    double A, B;
    if (!(cin >> N >> A >> B)) return 0;

    vector<Fuel> fuels(N);
    double Y_min = 0, X_min = 0;
    bool infinite = false;

    for (int i = 0; i < N; ++i) {
        cin >> fuels[i].a >> fuels[i].b >> fuels[i].c;
        if (fuels[i].a == 0 && fuels[i].b == 0) {
            if (fuels[i].c > 0) infinite = true;
        } else if (fuels[i].a == 0) {
            X_min = max(X_min, (double)fuels[i].c / fuels[i].b);
        } else if (fuels[i].b == 0) {
            Y_min = max(Y_min, (double)fuels[i].c / fuels[i].a);
        }
    }

    if (infinite) {
        cout << fixed << setprecision(6) << 1e18 << endl; // Should not happen based on constraints
        return 0;
    }

    map<pair<int, int>, int> max_c;
    for (int i = 0; i < N; ++i) {
        if (fuels[i].a > 0 && fuels[i].b > 0) {
            pair<int, int> p = {fuels[i].a, fuels[i].b};
            if (max_c.find(p) == max_c.end() || max_c[p] < fuels[i].c) {
                max_c[p] = fuels[i].c;
            }
        }
    }

    vector<Line> lines;
    for (auto const& [ab, c] : max_c) {
        double a = ab.first;
        double b = ab.second;
        double c_val = c;
        lines.push_back({a, b, c_val, -a / b, c_val / b});
    }

    // Sort lines by slope m = -a/b descending (i.e., a/b ascending)
    sort(lines.begin(), lines.end(), [](const Line& l1, const Line& l2) {
        if (abs(l1.m - l2.m) > 1e-12) return l1.m > l2.m;
        return l1.d > l2.d;
    });

    // Remove lines with same slope, keep only the one with highest intercept d
    vector<Line> unique_lines;
    for (const auto& l : lines) {
        if (unique_lines.empty() || abs(l.m - unique_lines.back().m) > 1e-12) {
            unique_lines.push_back(l);
        }
    }

    // Build the upper envelope using a stack
    vector<Line> hull;
    for (const auto& l : unique_lines) {
        while (hull.size() >= 2) {
            Line l1 = hull[hull.size() - 2];
            Line l2 = hull.back();
            // Intersection of l1 and l2: x12 = (d2 - d1) / (m1 - m2)
            // Intersection of l2 and l: x23 = (d - d2) / (m2 - m)
            double x12 = (l2.d - l1.d) / (l1.m - l2.m);
            double x23 = (l.d - l2.d) / (l2.m - l.m);
            if (x12 >= x23) hull.pop_back();
            else break;
        }
        hull.push_back(l);
    }

    auto get_f = [&](double x) {
        double max_y = X_min;
        for (const auto& l : hull) {
            max_y = max(max_y, l.m * x + l.d);
        }
        // The hull is the upper envelope, but we need it to be efficient. 
        // For N=75000, we can't iterate through the whole hull every time.
        // But the hull is small (at most 100*100). Let's use the fact that 
        // the envelope is formed by segments of lines in hull.
        return max_y;
    };

    // Re-implement get_f more efficiently
    auto get_f_fast = [&](double x) {
        double max_y = X_min;
        int low = 0, high = (int)hull.size() - 1;
        while (low <= high) {
            int mid1 = low + (high - low) / 3;
            int mid2 = high - (high - low) / 3;
            double y1 = hull[mid1].m * x + hull[mid1].d;
            double y2 = hull[mid2].m * x + hull[mid2].d;
            max_y = max({max_y, y1, y2});
            if (y1 < y2) low = mid1 + 1;
            else high = mid2 - 1;
        }
        // Actually, for the upper envelope, we can binary search the intersection points.
        // Let's just use the segments.
        return max_y;
    };

    // A better way to get f(x):
    // The hull segments are sorted by x-intersections.
    vector<double> inters;
    for (size_t i = 0; i + 1 < hull.size(); ++i) {
        inters.push_back((hull[i+1].d - hull[i].d) / (hull[i].m - hull[i+1].m));
    }

    auto get_f_final = [&](double x) {
        double res = X_min;
        auto it = lower_bound(inters.begin(), inters.end(), x);
        int idx = distance(inters.begin(), it);
        res = max(res, hull[idx].m * x + hull[idx].d);
        return res;
    };

    vector<double> candidates;
    candidates.push_back(Y_min);

    for (size_t i = 0; i < inters.size(); ++i) {
        if (inters[i] >= Y_min) candidates.push_back(inters[i]);
    }

    // Finding y* (where f(x) = X_min)
    for (size_t i = 0; i < hull.size(); ++i) {
        if (abs(hull[i].m) > 1e-12) {
            double x_star = (X_min - hull[i].d) / hull[i].m;
            if (x_star >= Y_min) candidates.push_back(x_star);
        }
    }

    double min_val = -1.0;
    for (double x : candidates) {
        double y = get_f_final(x);
        double current = A * x + B * y;
        if (min_val < 0 || current < min_val) min_val = current;
    }

    // One more case: if the function is decreasing, the minimum could be at 
    // the largest possible x where f(x) = X_min.
    // But we already included all such x_star and inters.

    // Edge case: if hull is empty
    if (hull.empty()) {
        min_val = A * Y_min + B * X_min;
    }

    cout << fixed << setprecision(6) << min_val << endl;

    return 0;
}
```
