```cpp
#include <iostream>
#include <vector>
#include <string>

using namespace std;

/**
 * Problem: Kakuro
 * A Kakuro puzzle on an n x m grid.
 * Rules:
 * - Each white cell gets a single digit (1-9).
 * - Each digit in a run must be unique.
 * - The sum of digits in each run must match the given clue.
 * n, m <= 6.
 * 
 * Strategy:
 * Backtracking with pruning.
 * Given the small constraints (up to 6x6), a standard backtracking search should be fast.
 */

int n, m;
string grid_str[6][6];
bool is_white[6][6];
int h_clue_val[6][6], v_clue_val[6][6];
int cell_value[6][6];
int white_cell_to_h_run[6][6];
int white_cell_to_v_run[6][6];

struct WhiteCell {
    int r, c;
    int hRunIdx;
    int vRunIdx;
};

struct Run {
    int targetSum;
    int currentSum;
    int numCells;
    int cellsFilled;
    int usedDigitsMask;

    Run() : targetSum(0), currentSum(0), numCells(0), cellsFilled(0), usedDigitsMask(0) {}
};

vector<WhiteCell> white_cells;
vector<Run> h_runs, v_runs;

// Backtracking function to fill digits into white cells
bool solve(int idx) {
    if (idx == (int)white_cells.size()) {
        return true;
    }

    WhiteCell &wc = white_cells[idx];
    Run &hr = h_runs[wc.hRunIdx];
    Run &vr = v_runs[wc.vRunIdx];

    for (int d = 1; d <= 9; ++d) {
        // Rule: digit must be unique within each run
        if (!(hr.usedDigitsMask & (1 << d)) && !(vr.usedDigitsMask & (1 << d))) {
            int new_h_sum = hr.currentSum + d;
            int new_v_sum = vr.currentSum + d;

            // Rule: sum must not exceed the clue
            if (new_h_sum <= hr.targetSum && new_v_sum <= vr.targetSum) {
                int rem_h = hr.numCells - hr.cellsFilled - 1;
                int rem_v = vr.numCells - vr.cellsFilled - 1;

                // If it's the last cell in a run, the sum must exactly match the clue
                if (rem_h == 0 && new_h_sum != hr.targetSum) continue;
                if (rem_v == 0 && new_v_sum != vr.targetSum) continue;
                
                // Pruning: check if it's possible to reach targetSum with remaining cells
                if (rem_h > 0) {
                    if (new_h_sum + rem_h * 1 > hr.targetSum || new_h_sum + rem_h * 9 < hr.targetSum) continue;
                }
                if (rem_v > 0) {
                    if (new_v_sum + rem_v * 1 > vr.targetSum || new_v_sum + rem_v * 9 < vr.targetSum) continue;
                }

                // Apply state changes
                hr.currentSum = new_h_sum;
                hr.cellsFilled++;
                hr.usedDigitsMask |= (1 << d);
                vr.currentSum = new_v_sum;
                vr.cellsFilled++;
                vr.usedDigitsMask |= (1 << d);
                cell_value[wc.r][wc.c] = d;

                if (solve(idx + 1)) return true;

                // Backtrack (revert state changes)
                cell_value[wc.r][wc.c] = 0;
                vr.usedDigitsMask &= ~(1 << d);
                vr.cellsFilled--;
                vr.currentSum -= d;
                hr.usedDigitsMask &= ~(1 << d);
                hr.cellsFilled--;
                hr.currentSum -= d;
            }
        }
    }
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> n >> m)) return 0;

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            cin >> grid_str[i][j];
            if (grid_str[i][j] == ".....") {
                is_white[i][j] = true;
            } else {
                is_white[i][j] = false;
                // Parse AA\BB where AA is vertical clue and BB is horizontal clue
                string AA = grid_str[i][j].substr(0, 2);
                string BB = grid_str[i][j].substr(3, 2);
                v_clue_val[i][j] = (AA == "XX" ? 0 : stoi(AA));
                h_clue_val[i][j] = (BB == "XX" ? 0 : stoi(BB));
            }
        }
    }

    // Identify and parse all horizontal runs
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            if (is_white[i][j]) {
                int start_j = j;
                while (j < m && is_white[i][j]) j++;
                int end_j = j - 1;

                Run run;
                run.targetSum = h_clue_val[i][start_j - 1];
                run.numCells = end_j - start_j + 1;
                h_runs.push_back(run);
                int hIdx = (int)h_runs.size() - 1;

                for (int k = start_j; k <= end_j; ++k) {
                    white_cell_to_h_run[i][k] = hIdx;
                }
                j--; // Re-adjust loop index j
            }
        }
    }

    // Identify and parse all vertical runs
    for (int j = 0; j < m; ++j) {
        for (int i = 0; i < n; ++i) {
            if (is_white[i][j]) {
                int start_i = i;
                while (i < n && is_white[i][j]) i++;
                int end_i = i - 1;

                Run run;
                run.targetSum = v_clue_val[start_i - 1][j];
                run.numCells = end_i - start_i + 1;
                v_runs.push_back(run);
                int vIdx = (int)v_runs.size() - 1;

                for (int k = start_i; k <= end_i; ++k) {
                    white_cell_to_v_run[k][j] = vIdx;
                }
                i--; // Re-adjust loop index i
            }
        }
    }

    // Collect all white cells to process in backtracking
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            if (is_white[i][j]) {
                white_cells.push_back({i, j, white_cell_to_h_run[i][j], white_cell_to_v_run[i][j]});
            }
        }
    }

    if (solve(0)) {
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < m; ++j) {
                if (is_white[i][j]) {
                    cout << cell_value[i][j];
                } else {
                    cout << "_";
                }
                if (j < m - 1) cout << " ";
            }
            cout << "\n";
        }
    }

    return 0;
}
```
