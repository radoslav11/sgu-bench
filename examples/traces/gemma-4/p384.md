```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

/**
 * Problem: 384. Country
 * The key property of the graph is that every two distinct cities have exactly one common neighbor.
 * This is known as the Friendship Theorem: such a graph must consist of several triangles
 * sharing a common center vertex 'c'.
 * 
 * For any vertex v != c, its neighbors in the initial graph are 'c' and its "partner" p(v).
 * After some edges are deleted, the distance between any two cities x and y can be computed:
 * - If x == y, distance is 0.
 * - If x == c or y == c, we calculate dist(x, c) or dist(y, c).
 *   dist(v, c) is 1 if the edge (v, c) exists, 
 *   or 2 if the edges (v, p(v)) and (p(v), c) both exist. Otherwise, INF.
 * - If x and y are in the same triangle (i.e., y = p(x)), the distance is:
 *   1 if the edge (x, y) exists,
 *   2 if the edges (x, c) and (c, y) both exist. Otherwise, INF.
 * - Otherwise (x and y are in different triangles), the distance is dist(x, c) + dist(y, c).
 * 
 * Complexity: O(N + M + Q), which is well within the 0.75s limit.
 */

const int INF = 1000000000;

struct Edge {
    int u, v;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<Edge> edges(m + 1);
    vector<int> degree(n + 1, 0);
    vector<vector<int>> adj(n + 1);

    for (int i = 1; i <= m; ++i) {
        cin >> edges[i].u >> edges[i].v;
        degree[edges[i].u]++;
        degree[edges[i].v]++;
        adj[edges[i].u].push_back(edges[i].v);
        adj[edges[i].v].push_back(edges[i].u);
    }

    // Find the center vertex 'c'. In a friendship graph, c has degree n-1.
    // For n=3, any vertex can be c.
    int c = 1;
    for (int i = 2; i <= n; ++i) {
        if (degree[i] > degree[c]) {
            c = i;
        }
    }

    vector<int> partner(n + 1, 0);
    for (int i = 1; i <= n; ++i) {
        if (i != c) {
            for (int neighbor : adj[i]) {
                if (neighbor != c) {
                    partner[i] = neighbor;
                    break;
                }
            }
        }
    }

    vector<bool> exists(m + 1, true);
    vector<int> road_idx_c_v(n + 1, 0);
    vector<int> road_idx_v_pv(n + 1, 0);

    for (int i = 1; i <= m; ++i) {
        int u = edges[i].u;
        int v = edges[i].v;
        if (u == c) {
            road_idx_c_v[v] = i;
        } else if (v == c) {
            road_idx_c_v[u] = i;
        } else {
            road_idx_v_pv[u] = i;
            road_idx_v_pv[v] = i;
        }
    }

    auto get_dist_to_c = [&](int v) -> int {
        if (v == c) return 0;
        if (road_idx_c_v[v] != 0 && exists[road_idx_c_v[v]]) return 1;
        int pv = partner[v];
        if (pv != 0 && road_idx_v_pv[v] != 0 && exists[road_idx_v_pv[v]] &&
            road_idx_c_v[pv] != 0 && exists[road_idx_c_v[pv]]) return 2;
        return INF;
    };

    auto get_dist_same_triangle = [&](int x, int y) -> int {
        if (road_idx_v_pv[x] != 0 && exists[road_idx_v_pv[x]]) return 1;
        if (road_idx_c_v[x] != 0 && exists[road_idx_c_v[x]] &&
            road_idx_c_v[y] != 0 && exists[road_idx_c_v[y]]) return 2;
        return INF;
    };

    string type;
    while (cin >> type) {
        if (type == "DELETE") {
            int x;
            cin >> x;
            if (x >= 1 && x <= m) exists[x] = false;
        } else if (type == "LENGTH") {
            int x, y;
            cin >> x >> y;
            if (x == y) {
                cout << 0 << "\n";
                continue;
            }
            int ans = INF;
            if (x == c || y == c) {
                ans = get_dist_to_c(x == c ? y : x);
            } else if (partner[x] == y || partner[y] == x) {
                ans = get_dist_same_triangle(x, y);
            } else {
                int d1 = get_dist_to_c(x);
                int d2 = get_dist_to_c(y);
                if (d1 < INF && d2 < INF) ans = d1 + d2;
            }
            if (ans >= INF) cout << -1 << "\n";
            else cout << ans << "\n";
        }
    }

    return 0;
}
```
