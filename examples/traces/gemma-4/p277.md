#include <iostream>
#include <vector>
#include <set>
#include <algorithm>

using namespace std;

/**
 * Problem: Maintain a convex hull incrementally and compute its doubled area.
 * Approach: Maintain the upper and lower hulls separately using two std::set containers.
 * Use cross products for area and orientation checks to avoid floating-point issues.
 */

struct Point {
    long long x, y;
    bool operator<(const Point& other) const {
        if (x != other.x) return x < other.x;
        return y < other.y;
    }
};

// Standard cross product of two vectors (OA and OB)
long long cp(Point a, Point b) {
    return a.x * b.y - a.y * b.x;
}

// Cross product of three points (ABC)
long long cp(Point a, Point b, Point c) {
    return (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
}

struct Hull {
    set<Point> hull;
    long long s = 0;
    bool is_upper;

    Hull(bool upper) : is_upper(upper) {}

    void add(Point p) {
        // Find point with same x if any
        auto it = hull.lower_bound({p.x, -2000000000000000000LL});
        if (it != hull.end() && it->x == p.x) {
            if (is_upper) {
                if (p.y <= it->y) return;
            } else {
                if (p.y >= it->y) return;
            }
            
            // Remove the point with same x before adding new one
            Point old_p = *it;
            auto it_prev = (it == hull.begin()) ? hull.end() : prev(it);
            auto it_next = next(it);
            
            if (it_prev != hull.end() && it_next != hull.end()) {
                if (is_upper) s -= cp(*it_prev, old_p) + cp(old_p, *it_next) - cp(*it_prev, *it_next);
                else s -= cp(*it_next, old_p) + cp(old_p, *it_prev) - cp(*it_next, *it_prev);
            } else if (it_prev != hull.end()) {
                if (is_upper) s -= cp(*it_prev, old_p);
                else s -= cp(old_p, *it_prev);
            } else if (it_next != hull.end()) {
                if (is_upper) s -= cp(old_p, *it_next);
                else s -= cp(*it_next, old_p);
            }
            hull.erase(it);
            it = hull.lower_bound({p.x, -2000000000000000000LL});
        }

        // Check if new point p is already inside/on the boundary
        if (it != hull.end() && it != hull.begin()) {
            auto it_prev = prev(it);
            if (is_upper) {
                if (cp(*it_prev, p, *it) >= 0) return;
            } else {
                if (cp(*it_prev, p, *it) <= 0) return;
            }
        }

        // Add p to the hull and update current sum of cross products
        hull.insert(p);
        it = hull.find(p);
        if (it == hull.begin()) {
            auto it_next = next(it);
            if (it_next != hull.end()) {
                if (is_upper) s += cp(p, *it_next);
                else s += cp(*it_next, p);
            }
        } else if (next(it) == hull.end()) {
            auto it_prev = prev(it);
            if (it_prev != hull.end()) {
                if (is_upper) s += cp(*it_prev, p);
                else s += cp(p, *it_prev);
            }
        } else {
            auto it_prev = prev(it);
            auto it_next = next(it);
            if (is_upper) s += cp(*it_prev, p) + cp(p, *it_next) - cp(*it_prev, *it_next);
            else s += cp(*it_next, p) + cp(p, *it_prev) - cp(*it_next, *it_prev);
        }

        // Remove redundant points to the right
        auto it_next = next(it);
        while (it_next != hull.end() && next(it_next) != hull.end()) {
            auto it_next_next = next(it_next);
            if (is_upper) {
                if (cp(p, *it_next, *it_next_next) >= 0) {
                    s -= cp(p, *it_next) + cp(*it_next, *it_next_next) - cp(p, *it_next_next);
                    hull.erase(it_next);
                    it_next = next(it);
                } else break;
            } else {
                if (cp(p, *it_next, *it_next_next) <= 0) {
                    s -= cp(*it_next_next, *it_next) + cp(*it_next, p) - cp(*it_next_next, p);
                    hull.erase(it_next);
                    it_next = next(it);
                } else break;
            }
        }

        // Remove redundant points to the left
        if (it != hull.begin()) {
            auto it_prev = prev(it);
            while (it_prev != hull.begin()) {
                auto it_prev_prev = prev(it_prev);
                if (is_upper) {
                    if (cp(*it_prev_prev, *it_prev, p) >= 0) {
                        s -= cp(*it_prev_prev, *it_prev) + cp(*it_prev, p) - cp(*it_prev_prev, p);
                        hull.erase(it_prev);
                        it_prev = prev(it);
                    } else break;
                } else {
                    if (cp(*it_prev_prev, *it_prev, p) <= 0) {
                        s -= cp(p, *it_prev) + cp(*it_prev, *it_prev_prev) - cp(p, *it_prev_prev);
                        hull.erase(it_prev);
                        it_prev = prev(it);
                    } else break;
                }
            }
        }
    }
};

int main() {
    // Fastest input/output
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    Point p1, p2, p3;
    if (!(cin >> p1.x >> p1.y >> p2.x >> p2.y >> p3.x >> p3.y)) return 0;
    
    int N;
    if (!(cin >> N)) return 0;

    // Initialize upper and lower hulls with the first 3 cities
    Hull upper(true), lower(false);
    upper.add(p1); upper.add(p2); upper.add(p3);
    lower.add(p1); lower.add(p2); lower.add(p3);

    for (int i = 0; i < N; ++i) {
        Point p;
        cin >> p.x >> p.y;
        upper.add(p);
        lower.add(p);
        // Doubled area is |S_upper + S_lower| where S_upper = sum(cp(Pi, Pi+1)) and S_lower = sum(cp(Qi+1, Qi))
        long long area2 = upper.s + lower.s;
        if (area2 < 0) area2 = -area2;
        cout << area2 << "\n";
    }

    return 0;
}
