#include <iostream>

using namespace std;

/**
 * The number of trailing zeros in N! is determined by the number of times
 * the factor 10 occurs in the prime factorization of N!.
 * Since 10 = 2 * 5, and there are always more factors of 2 than factors of 5 
 * in N!, the number of trailing zeros is equal to the number of factors of 5 
 * in the prime factorization of N!.
 * 
 * Legendre's Formula gives the number of factors of a prime p in N! as:
 * Z(N) = floor(N/p) + floor(N/p^2) + floor(N/p^3) + ...
 * 
 * Here, Z(N) = floor(N/5) + floor(N/25) + floor(N/125) + ...
 * Since Z(N) is a non-decreasing function of N, we can find the smallest N
 * such that Z(N) = Q using binary search.
 */

typedef long long ll;

/**
 * Calculates the number of trailing zeros in N! using Legendre's Formula.
 * @param n The number N for which we are calculating the number of zeros in N!.
 * @return The number of factors of 5 in N!.
 */
ll countZeros(ll n) {
    ll count = 0;
    while (n > 0) {
        count += n / 5;
        n /= 5;
    }
    return count;
}

int main() {
    // Faster I/O for competitive programming
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    ll Q;
    if (!(cin >> Q)) return 0;

    // Binary search range.
    // Since Z(N) is roughly N/4, for Q = 10^8, N is around 4 * 10^8.
    // Using 10^9 as an upper bound covers all possible Q up to 10^8.
    ll low = 0, high = 1000000000LL;
    ll ans = -1;

    while (low <= high) {
        ll mid = low + (high - low) / 2;
        if (countZeros(mid) >= Q) {
            // If Z(mid) is at least Q, this mid could be our answer.
            // We continue searching to find the *minimal* such N.
            ans = mid;
            high = mid - 1;
        } else {
            low = mid + 1;
        }
    }

    // After finding the smallest N such that Z(N) >= Q, 
    // we must verify if Z(N) is exactly equal to Q.
    // If it's not, then no such N exists (this happens when Z(N) jumps).
    if (ans != -1 && countZeros(ans) == Q) {
        cout << ans << endl;
    } else {
        cout << "No solution" << endl;
    }

    return 0;
}
