#include <iostream>
#include <vector>
#include <string>
#include <cstring>
#include <algorithm>

using namespace std;

/**
 * Problem: Almost Lucky Numbers
 * Definition: 
 *   - Lucky number: 2N digits and the sum of the first N digits equals the sum of the last N digits.
 *   - Almost lucky number: Can be made lucky by changing exactly one digit (no leading zeros).
 * Goal: Count almost lucky numbers in [A, B].
 * Range: 0 <= A <= B <= 10^9.
 * 
 * A number is almost lucky if it has an even number of digits (2, 4, 6, 8, or 10)
 * and there exists exactly one digit such that changing it makes it lucky.
 * 
 * For a 2N-digit number, let S1 be the sum of the first N digits and S2 be the sum of the last N.
 * Let delta = S1 - S2. 
 * A number is almost lucky if delta != 0 AND there exists an i such that 
 * changing digit d_i to d'_i makes S1' = S2'.
 * 
 * For i <= N: d'_i = d_i - delta. We need 0 <= d_i - delta <= 9 and (if i=0, d_i - delta != 0).
 * For i > N: d'_i = d_i + delta. We need 0 <= d_i + delta <= 9.
 * 
 * We can solve this using digit DP by iterating over all possible non-zero delta values.
 * Total almost lucky = sum_{delta != 0} (count(S1 - S2 = delta) - count(S1 - S2 = delta AND no digit can be changed to satisfy it)).
 */

long long dp[20][200][2][2];
int target_delta_val;
int N_val;
vector<int> current_digits;

long long solve(int idx, int delta, bool is_less, bool is_no_good) {
    if (idx == 2 * N_val) {
        return (delta == target_delta_val) ? 1 : 0;
    }
    // delta offset by 100 to handle negative values
    if (dp[idx][delta + 100][is_less][is_no_good] != -1) {
        return dp[idx][delta + 100][is_less][is_no_good];
    }

    long long res = 0;
    int limit = is_less ? 9 : current_digits[idx];
    for (int d = 0; d <= limit; ++d) {
        if (idx == 0 && d == 0) continue; // Leading zero not allowed

        bool is_good = false;
        if (idx < N_val) {
            // For i <= N, d'_i = d_i - delta. Condition: 0 <= d_i - delta <= 9
            // and if i=0, d'_i != 0.
            if (d - target_delta_val >= 0 && d - target_delta_val <= 9) {
                if (idx > 0 || d - target_delta_val != 0) {
                    is_good = true;
                }
            }
        } else {
            // For i > N, d'_i = d_i + delta. Condition: 0 <= d_i + delta <= 9
            if (d + target_delta_val >= 0 && d + target_delta_val <= 9) {
                is_good = true;
            }
        }

        if (is_no_good && is_good) continue;

        int next_delta = delta;
        if (idx < N_val) next_delta += d;
        else next_delta -= d;

        res += solve(idx + 1, next_delta, is_less || (d < limit), is_no_good);
    }
    return dp[idx][delta + 100][is_less][is_no_good] = res;
}

long long count_2N(int N, int X) {
    if (X < 0) return 0;
    long long min_val = 1;
    for (int i = 0; i < 2 * N - 1; ++i) min_val *= 10;
    if (X < min_val) return 0;

    long long max_val = min_val * 10 - 1;
    int limit = (X > max_val) ? (int)max_val : X;

    string s = to_string(limit);
    current_digits.clear();
    for (char c : s) current_digits.push_back(c - '0');

    long long total_almost_lucky = 0;
    for (int delta = -(9 * N); delta <= 9 * N; ++delta) {
        if (delta == 0) continue;
        target_delta_val = delta;
        N_val = N;

        // Part 1: Count numbers with S1 - S2 = target_delta
        memset(dp, -1, sizeof(dp));
        long long c1 = solve(0, 0, false, false);

        // Part 2: Count numbers with S1 - S2 = target_delta AND no digit is "good"
        memset(dp, -1, sizeof(dp));
        long long c2 = solve(0, 0, false, true);

        total_almost_lucky += (c1 - c2);
    }
    return total_almost_lucky;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long A, B;
    if (!(cin >> A >> B)) return 0;

    long long ans = 0;
    // 10^9 is 10-digit and can be almost lucky
    if (A <= 1000000000LL && 1000000000LL <= B) {
        ans++;
        B = 999999999LL;
    }

    // Check numbers with 2, 4, 6, 8 digits
    for (int N = 1; N <= 4; ++N) {
        ans += count_2N(N, (int)B);
        ans -= count_2N(N, (int)A - 1);
    }

    // If B was 10^9, we need to check if A-1 is within bounds.
    // Our approach handles [A, B] as count(B) - count(A-1) correctly for 2, 4, 6, 8 digits.
    // However, if A was 10^9, A-1 is 999,999,999.
    // But count_2N handles all 2N digits correctly.

    // Correcting logic for counting:
    // The range [A, B] should be handled by:
    // result = count_in_range(A, B)
    // For each 2N, result += count_2N(N, B) - count_2N(N, A-1).

    // Let's rebuild logic to be more robust.
    ans = 0;
    if (A <= 1000000000LL && 1000000000LL <= B) {
        ans++;
    }

    for (int N = 1; N <= 4; ++N) {
        long long up = B;
        long long low = A;

        // For 2N digits, the numbers are in [10^(2N-1), 10^(2N)-1]
        long long min_val = 1;
        for (int i = 0; i < 2 * N - 1; ++i) min_val *= 10;
        long long max_val = min_val * 10 - 1;

        // Range intersection: [max(low, min_val), min(up, max_val)]
        long long start = max(low, min_val);
        long long end = min(up, max_val);

        if (start <= end) {
            ans += count_2N(N, (int)end) - count_2N(N, (int)start - 1);
        }
    }

    cout << ans << endl;

    return 0;
}
