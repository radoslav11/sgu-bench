```cpp
#include <iostream>
#include <vector>

using namespace std;

/**
 * Problem Analysis:
 * The problem asks for the number of non-isomorphic Abelian groups of order n.
 * According to the fundamental theorem of finite abelian groups, every finite 
 * abelian group is isomorphic to a direct sum of cyclic groups of prime power orders.
 * 
 * If the prime factorization of n is n = p1^a1 * p2^a2 * ... * pk^ak, then the
 * number of non-isomorphic abelian groups of order n is given by the product of the
 * number of ways to partition each exponent ai. 
 * That is: Number of Abelian groups = P(a1) * P(a2) * ... * P(ak), 
 * where P(a) is the number of partitions of the integer a.
 * 
 * Constraints:
 * n <= 10^12. 
 * The largest exponent possible is for n = 2^a <= 10^12, which gives a <= 39.
 * A trial division up to 10^6 is efficient enough for the 0.25s time limit.
 */

typedef long long ll;

// P[i] will store the number of partitions of the integer i.
// Since n <= 10^12, the maximum exponent i is at most 40.
ll P[45];

/**
 * Precomputes the partition function P(i) up to 44 using dynamic programming.
 * dp[i][j] = number of ways to partition i using only integers up to j.
 */
void precompute_partitions() {
    int max_a = 44;
    vector<vector<ll>> dp(max_a + 1, vector<ll>(max_a + 1, 0));
    
    // Base case: There is one way to partition 0 (the empty sum).
    for (int j = 0; j <= max_a; ++j) {
        dp[0][j] = 1;
    }
    
    // Fill the DP table
    for (int i = 1; i <= max_a; ++i) {
        for (int j = 1; j <= max_a; ++j) {
            // Case 1: Don't use j in the partition.
            dp[i][j] = dp[i][j - 1];
            // Case 2: Use at least one j in the partition.
            if (i >= j) {
                dp[i][j] += dp[i - j][j];
            }
        }
    }
    
    // The number of ways to partition i is dp[i][i].
    for (int i = 0; i <= max_a; ++i) {
        P[i] = dp[i][i];
    }
}

int main() {
    // Faster I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    precompute_partitions();

    ll n;
    // Input n can be up to 10^12, which requires long long.
    if (!(cin >> n)) {
        return 0;
    }

    // For n = 1, the trivial group is the only one.
    if (n == 1) {
        cout << 1 << endl;
        return 0;
    }

    ll total_groups = 1;
    ll temp_n = n;

    /**
     * Prime Factorization of n:
     * We iterate from i = 2 up to sqrt(temp_n).
     * For each factor i, we count its exponent in the prime factorization of n.
     */
    for (ll i = 2; i * i <= temp_n; ++i) {
        if (temp_n % i == 0) {
            int count = 0;
            while (temp_n % i == 0) {
                count++;
                temp_n /= i;
            }
            // For every exponent 'count' of a prime factor, multiply total_groups by P(count).
            if (count < 45) {
                total_groups *= P[count];
            }
        }
    }

    /**
     * If after trial division temp_n is still > 1, the remaining temp_n is a prime factor 
     * with exponent 1.
     */
    if (temp_n > 1) {
        total_groups *= P[1];
    }

    // The result is the product of the partition numbers of all exponents.
    cout << total_groups << endl;

    return 0;
}
```
