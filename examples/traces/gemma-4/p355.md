#include <cstdio>
#include <algorithm>

using namespace std;

/**
 * Problem Analysis:
 * The problem asks to color numbers from 1 to N such that if A is divisible by B,
 * A and B have different colors. We want to minimize the number of colors.
 * 
 * This is equivalent to finding the longest chain of divisors in the set {1, ..., N}.
 * A chain is a sequence of numbers x_1, x_2, ..., x_k where x_i | x_{i+1}.
 * The length of the longest chain ending at x is 1 + (number of prime factors of x, 
 * counting multiplicities). Let omega(x) be the number of prime factors of x.
 * Then the number of colors needed is max_{1 <= x <= N} (omega(x) + 1).
 * 
 * We can assign color C(x) = omega(x) + 1.
 * If B divides A and A != B, then omega(A) > omega(B), hence C(A) != C(B).
 * 
 * Time Complexity: O(N) using a linear sieve to compute omega(x) for all x <= N.
 * Space Complexity: O(N) to store omega(x) and the list of primes.
 */

const int MAXN = 1000005;
int omega[MAXN];
int primes[MAXN];
int prime_count = 0;

/**
 * Sieve of Eratosthenes-like linear sieve to compute the number of prime 
 * factors (omega) for all numbers from 1 to n.
 */
void sieve(int n) {
    for (int i = 2; i <= n; ++i) {
        if (omega[i] == 0) {
            // i is prime
            omega[i] = 1;
            primes[prime_count++] = i;
        }
        for (int j = 0; j < prime_count; ++j) {
            long long next = 1LL * i * primes[j];
            if (next > n) break;
            
            // next = i * primes[j]. Since primes[j] is a prime factor,
            // omega(next) = omega(i) + 1.
            omega[(int)next] = omega[i] + 1;
            
            // If primes[j] divides i, we stop to ensure the linear time property.
            if (i % primes[j] == 0) break;
        }
    }
}

int main() {
    int n;
    // Reading input
    if (scanf("%d", &n) != 1) return 0;
    if (n < 1) return 0;

    // Initialize omega[1] = 0
    omega[1] = 0;
    
    // Compute omega(x) for all x in [2, n]
    if (n >= 2) {
        sieve(n);
    }

    // Find the maximum value of omega(x)
    int max_omega = 0;
    for (int i = 1; i <= n; ++i) {
        if (omega[i] > max_omega) {
            max_omega = omega[i];
        }
    }

    // The total number of colors needed is max_omega + 1
    printf("%d\n", max_omega + 1);
    
    // Print the color of each number from 1 to N
    for (int i = 1; i <= n; ++i) {
        printf("%d%c", omega[i] + 1, (i == n ? '\n' : ' '));
    }

    return 0;
}
