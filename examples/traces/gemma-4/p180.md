#include <cstdio>

/**
 * Problem: Find the number of inversions in an array of N integers.
 * N <= 65537, Ai <= 10^9.
 * Time limit: 0.25s (tight).
 * Memory limit: 4096 KB (tight).
 * 
 * Solution: Use the Merge Sort algorithm to count inversions in O(N log N) time.
 * This approach is memory efficient and generally fast in practice.
 * N = 65537 requires at most ~2.15e9 inversions, so the result should be a long long.
 */

const int MAXN = 65537;
int A[MAXN];
int temp[MAXN];

/**
 * mergeSort: Recursively sorts the array and counts inversions.
 * @param left Starting index of the current subarray.
 * @param right Ending index of the current subarray.
 * @return The number of inversions in the subarray.
 */
long long mergeSort(int left, int right) {
    if (left >= right) {
        return 0;
    }

    int mid = left + (right - left) / 2;
    long long inv_count = 0;

    // Recursively sort left and right halves and sum their inversion counts.
    inv_count += mergeSort(left, mid);
    inv_count += mergeSort(mid + 1, right);

    // Merge the two sorted halves.
    int i = left;      // Index for the left half
    int j = mid + 1;   // Index for the right half
    int k = left;      // Index for the temporary array

    while (i <= mid && j <= right) {
        if (A[i] <= A[j]) {
            temp[k++] = A[i++];
        } else {
            // If A[i] > A[j], then all elements from A[i] to A[mid] are greater than A[j].
            temp[k++] = A[j++];
            inv_count += (mid - i + 1);
        }
    }

    // Copy remaining elements from the left half, if any.
    while (i <= mid) {
        temp[k++] = A[i++];
    }

    // Copy remaining elements from the right half, if any.
    while (j <= right) {
        temp[k++] = A[j++];
    }

    // Copy the merged elements back into the original array.
    for (int p = left; p <= right; p++) {
        A[p] = temp[p];
    }

    return inv_count;
}

int main() {
    int N;
    // Fast I/O using scanf.
    if (scanf("%d", &N) != 1) {
        return 0;
    }

    // Read the N integers into array A.
    for (int i = 0; i < N; i++) {
        if (scanf("%d", &A[i]) != 1) {
            break;
        }
    }

    // Perform merge sort and output the total inversion count.
    // Using %lld to print long long.
    printf("%lld\n", mergeSort(0, N - 1));

    return 0;
}
