#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Problem Analysis:
 * The problem asks for two lists of cities:
 * A: Cities that can be part of a longest tour (where a tour is an increasing sequence of x and y).
 * B: Cities that must be part of every longest tour.
 * 
 * This is equivalent to finding the longest increasing subsequence in 2D.
 * For a city i to be in A: L_prefix(i) + L_suffix(i) - 1 == L_max.
 * For a city i to be in B: It must be the only city at some position k in a longest path.
 * 
 * Sorting Strategy:
 * To handle the 2D LIS efficiently:
 * Sort cities by x ascending, and for the same x, sort by y descending.
 * This ensures that in a 1D LIS of y coordinates, no two cities with the same x can be picked.
 * 
 * Time Complexity: O(N log N)
 * Space Complexity: O(N)
 */

struct City {
    int x, y, id;
};

struct FenwickTree {
    int size;
    vector<int> tree;
    FenwickTree(int n) : size(n), tree(n + 1, 0) {}
    
    void update(int i, int val) {
        for (; i <= size; i += i & -i) {
            tree[i] = max(tree[i], val);
        }
    }
    
    int query(int i) {
        int res = 0;
        for (; i > 0; i -= i & -i) {
            res = max(res, tree[i]);
        }
        return res;
    }
};

int main() {
    // Speed up I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    vector<City> cities(n);
    vector<int> ys;
    for (int i = 0; i < n; ++i) {
        cin >> cities[i].x >> cities[i].y;
        cities[i].id = i + 1;
        ys.push_back(cities[i].y);
    }

    // Sort cities by x ascending, y descending
    sort(cities.begin(), cities.end(), [](const City& a, const City& b) {
        if (a.x != b.x) return a.x < b.x;
        return a.y > b.y;
    });

    // Coordinate compression for y
    sort(ys.begin(), ys.end());
    ys.erase(unique(ys.begin(), ys.end()), ys.end());
    auto get_y_idx = [&](int y) {
        return lower_bound(ys.begin(), ys.end(), y) - ys.begin() + 1;
    };

    int M = ys.size();
    vector<int> f(n), g(n);

    // Calculate f[i]: Length of LIS ending at city i (in sorted order)
    FenwickTree ft1(M);
    for (int i = 0; i < n; ++i) {
        int y_idx = get_y_idx(cities[i].y);
        f[i] = 1 + ft1.query(y_idx - 1);
        ft1.update(y_idx, f[i]);
    }

    // Calculate g[i]: Length of LIS starting at city i (in sorted order)
    // We use a Fenwick tree on reversed y-indices to query suffix maximums
    FenwickTree ft2(M);
    for (int i = n - 1; i >= 0; --i) {
        int y_idx = get_y_idx(cities[i].y);
        // query(M - y_idx) gives max in original y range [y_idx + 1, M]
        g[i] = 1 + ft2.query(M - y_idx);
        ft2.update(M - y_idx + 1, g[i]);
    }

    // Determine L_max
    int L = 0;
    for (int i = 0; i < n; ++i) {
        if (f[i] > L) L = f[i];
    }

    // Identify cities in list A and count occurrences at each position in longest path
    vector<int> A;
    vector<int> count_S(L + 1, 0);
    for (int i = 0; i < n; ++i) {
        if (f[i] + g[i] - 1 == L) {
            A.push_back(cities[i].id);
            count_S[f[i]]++;
        }
    }
    sort(A.begin(), A.end());

    // Identify cities in list B
    vector<int> B;
    for (int i = 0; i < n; ++i) {
        if (f[i] + g[i] - 1 == L && count_S[f[i]] == 1) {
            B.push_back(cities[i].id);
        }
    }
    sort(B.begin(), B.end());

    // Output result
    cout << A.size();
    for (int id : A) cout << " " << id;
    cout << "\n";

    cout << B.size();
    for (int id : B) cout << " " << id;
    cout << "\n";

    return 0;
}
