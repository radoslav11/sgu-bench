#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Problem: Find the centroid(s) of a tree.
 * Centroid definition: A vertex whose removal results in connected components 
 * such that the size of the largest component is minimized.
 * 
 * Constraints:
 * N <= 16,000.
 * Time limit: 0.25s (very tight).
 * Memory limit: 4096 KB (very tight).
 * 
 * Strategy:
 * 1. Use an array-based adjacency list to minimize memory usage and speed up access.
 * 2. Use an iterative BFS-based approach to find a topological order for traversal 
 *    to avoid potential stack overflow in deep trees (line graphs) within the 4MB memory limit.
 * 3. Calculate subtree sizes and the largest component size for each vertex.
 * 4. Find the minimum of these values and identify all vertices achieving it.
 */

const int MAXN = 16005;

// Adjacency list using arrays for efficiency
int head[MAXN];
int to_edge[2 * MAXN];
int nxt_edge[2 * MAXN];
int edge_count = 0;

void add_edge(int u, int v) {
    edge_count++;
    to_edge[edge_count] = v;
    nxt_edge[edge_count] = head[u];
    head[u] = edge_count;
}

int sz[MAXN];
int max_comp[MAXN];
int q[MAXN];
int parent[MAXN];
int N;

int main() {
    // Fast I/O
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> N)) return 0;

    // Case for a single vertex
    if (N == 1) {
        cout << "0 1\n";
        cout << "1\n";
        return 0;
    }

    // Reading the N-1 edges
    for (int i = 0; i < N - 1; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        add_edge(u, v);
        add_edge(v, u);
    }

    // Perform BFS to find a traversal order and parents
    // This helps in computing subtree sizes in a post-order (bottom-up) fashion
    int head_q = 0, tail_q = 0;
    q[tail_q++] = 1; // Root the tree at vertex 1
    parent[1] = 0;

    while (head_q < tail_q) {
        int u = q[head_q++];
        for (int i = head[u]; i != 0; i = nxt_edge[i]) {
            int v = to_edge[i];
            if (v != parent[u]) {
                parent[v] = u;
                q[tail_q++] = v;
            }
        }
    }

    // Compute subtree sizes and max component size for each vertex in reverse BFS order
    for (int i = N - 1; i >= 0; --i) {
        int u = q[i];
        sz[u] = 1;
        max_comp[u] = 0;
        for (int j = head[u]; j != 0; j = nxt_edge[j]) {
            int v = to_edge[j];
            if (v != parent[u]) {
                sz[u] += sz[v];
                if (sz[v] > max_comp[u]) {
                    max_comp[u] = sz[v];
                }
            }
        }
        // The component "above" the vertex u
        int comp_above = N - sz[u];
        if (comp_above > max_comp[u]) {
            max_comp[u] = comp_above;
        }
    }

    // Find the minimum associated value
    int min_val = N + 1;
    for (int i = 1; i <= N; ++i) {
        if (max_comp[i] < min_val) {
            min_val = max_comp[i];
        }
    }

    // Identify and collect all vertices that are centroids
    vector<int> centroids;
    for (int i = 1; i <= N; ++i) {
        if (max_comp[i] == min_val) {
            centroids.push_back(i);
        }
    }

    // Output results
    cout << min_val << " " << (int)centroids.size() << "\n";
    for (int i = 0; i < (int)centroids.size(); ++i) {
        cout << centroids[i] << (i == (int)centroids.size() - 1 ? "" : " ");
    }
    cout << "\n";

    return 0;
}
