#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

const int MAX = 705;
const int LOG = 10;

int H_odd[MAX][MAX], V_odd[MAX][MAX];
int H_even[MAX][MAX], V_even[MAX][MAX];
int st_H_odd[LOG][MAX][MAX], st_V_odd[LOG][MAX][MAX];
int st_H_even[LOG][MAX][MAX], st_V_even[LOG][MAX][MAX];
int logs[MAX];

vector<int> manacher_odd(const string& s) {
    int n = s.size();
    vector<int> d1(n);
    for (int i = 0, l = 0, r = -1; i < n; i++) {
        int k = (i > r) ? 1 : min(d1[l + r - i], r - i + 1);
        while (0 <= i - k && i + k < n && s[i - k] == s[i + k]) {
            k++;
        }
        d1[i] = k--;
        if (i + k > r) {
            l = i - k;
            r = i + k;
        }
    }
    return d1;
}

vector<int> manacher_even(const string& s) {
    int n = s.size();
    vector<int> d2(n);
    for (int i = 0, l = 0, r = -1; i < n; i++) {
        int k = (i > r) ? 0 : min(d2[l + r - i + 1], r - i + 1);
        while (0 <= i - k - 1 && i + k < n && s[i - k - 1] == s[i + k]) {
            k++;
        }
        d2[i] = k--;
        if (i + k > r) {
            l = i - k - 1;
            r = i + k;
        }
    }
    return d2;
}

int query_H_odd(int j, int L, int R) {
    if (L > R) return 1000;
    int k = logs[R - L + 1];
    return min(st_H_odd[k][j][L], st_H_odd[k][j][R - (1 << k) + 1]);
}

int query_V_odd(int i, int L, int R) {
    if (L > R) return 1000;
    int k = logs[R - L + 1];
    return min(st_V_odd[k][i][L], st_V_odd[k][i][R - (1 << k) + 1]);
}

int query_H_even(int j, int L, int R) {
    if (L > R) return 1000;
    int k = logs[R - L + 1];
    return min(st_H_even[k][j][L], st_H_even[k][j][R - (1 << k) + 1]);
}

int query_V_even(int i, int L, int R) {
    if (L > R) return 1000;
    int k = logs[R - L + 1];
    return min(st_V_even[k][i][L], st_V_even[k][i][R - (1 << k) + 1]);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int h, w;
    if (!(cin >> h >> w)) return 0;
    vector<string> grid(h);
    for (int i = 0; i < h; ++i) cin >> grid[i];

    for (int i = 0; i < h; ++i) {
        vector<int> d1 = manacher_odd(grid[i]);
        vector<int> d2 = manacher_even(grid[i]);
        for (int j = 0; j < w; ++j) {
            H_odd[i][j] = d1[j];
            H_even[i][j] = d2[j];
        }
    }

    for (int j = 0; j < w; ++j) {
        string col = "";
        for (int i = 0; i < h; ++i) col += grid[i][j];
        vector<int> d1 = manacher_odd(col);
        vector<int> d2 = manacher_even(col);
        for (int i = 0; i < h; ++i) {
            V_odd[i][j] = d1[i];
            V_even[i][j] = d2[i];
        }
    }

    for (int i = 2; i < MAX; ++i) logs[i] = logs[i / 2] + 1;

    for (int j = 0; j < w; ++j) {
        for (int i = 0; i < h; ++i) st_H_odd[0][j][i] = H_odd[i][j];
        for (int k = 1; k < LOG; ++k)
            for (int i = 0; i + (1 << k) <= h; ++i)
                st_H_odd[k][j][i] = min(st_H_odd[k - 1][j][i], st_H_odd[k - 1][j][i + (1 << (k - 1))]);

        for (int i = 0; i < h; ++i) st_H_even[0][j][i] = H_even[i][j]; // Error in logic here, fixed below
    }
    // Redoing building correctly for even cases (note: indexing j+1 in queries)
    for (int j = 0; j < w; ++j) {
        for (int i = 0; i < h; ++i) st_H_even[0][j][i] = H_even[i][j];
        for (int k = 1; k < LOG; ++k)
            for (int i = 0; i + (1 << k) <= h; ++i)
                st_H_even[k][j][i] = min(st_H_even[k - 1][j][i], st_H_even[k - 1][j][i + (1 << (k - 1))]);
    }

    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) st_V_odd[0][i][j] = V_odd[i][j];
        for (int k = 1; k < LOG; ++k)
            for (int j = 0; j + (1 << k) <= w; ++j)
                st_V_odd[k][i][j] = min(st_V_odd[k - 1][i][j], st_V_odd[k - 1][i][j + (1 << (k - 1))]);
    }

    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) st_V_even[0][i][j] = V_even[i][j];
        for (int k = 1; k < LOG; ++k)
            for (int j = 0; j + (1 << k) <= w; ++j)
                st_V_even[k][i][j] = min(st_V_even[k - 1][i][j], st_V_even[k - 1][i][j + (1 << (k - 1))]);
    }

    int max_n = 0, r1 = 1, c1 = 1, r2 = 1, c2 = 1;

    for (int i = 0; i < h; ++i) {
        for (int j = 0; j < w; ++j) {
            int k_lim = min({H_odd[i][j], V_odd[i][j], i + 1, h - i, j + 1, w - j});
            if (2 * k_lim - 1 <= max_n) continue;
            int low = max(1, (max_n / 2) + 1), high = k_lim, best_k = -1;
            while (low <= high) {
                int k = low + (high - low) / 2;
                if (query_H_odd(j, i - k + 1, i + k - 1) >= k && query_V_odd(i, j - k + 1, j + k - 1) >= k) {
                    best_k = k; low = k + 1;
                } else high = k - 1;
            }
            if (best_k != -1 && 2 * best_k - 1 > max_n) {
                max_n = 2 * best_k - 1;
                r1 = i - best_k + 2; c1 = j - best_k + 2; r2 = i + best_k; c2 = j + best_k;
            }
        }
    }

    for (int i = 0; i < h - 1; ++i) {
        for (int j = 0; j < w - 1; ++j) {
            int k_lim = min({i + 1, h - i - 1, j + 1, w - j - 1});
            if (2 * k_lim <= max_n) continue;
            int low = max(1, (max_n / 2) + 1), high = k_lim, best_k = -1;
            while (low <= high) {
                int k = low + (high - low) / 2;
                if (query_H_even(j + 1, i - k + 1, i + k) >= k && query_V_even(i + 1, j - k + 1, j + k) >= k) {
                    best_k = k; low = k + 1;
                } else high = k - 1;
            }
            if (best_k != -1 && 2 * best_k > max_n) {
                max_n = 2 * best_k;
                r1 = i - best_k + 2; c1 = j - best_k + 2; r2 = i + best_k + 1; c2 = j + best_k + 1;
            }
        }
    }

    cout << r1 << " " << c1 << " " << r2 << " " << c2 << endl;

    return 0;
}
