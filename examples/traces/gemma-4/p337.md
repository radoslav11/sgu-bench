#include <iostream>
#include <string>
#include <vector>
#include <string_view>
#include <algorithm>

using namespace std;

/**
 * Problem: 337. Keven
 * The task is to find the K-even substring of maximal length in a cyclic string.
 * A string is K-even if the first half differs from the second half in at most K positions.
 * If multiple such substrings of maximal length exist, we choose the lexicographically smallest.
 * 
 * Constraints:
 * - K up to 2000.
 * - String length N up to 2000 (odd).
 * - Time limit: 0.25s.
 * 
 * Approach:
 * - Since N is small (2000), the maximum even length L is N-1.
 * - We iterate through possible half-lengths M = L/2 from (N-1)/2 down to 1.
 * - For each M, we can find all starting positions i such that the substring 
 *   of length 2M starting at i is K-even using a sliding window of differences.
 * - To handle lexicographical order efficiently, we use a Suffix Array to compare 
 *   substrings of the same length. A Suffix Array gives the rank of each suffix, 
 *   and for substrings of equal length, the one with the smaller suffix rank 
 *   is lexicographically smaller.
 * - The complexity is O(N^2) to find the maximum L and O(N log^2 N) for Suffix Array.
 *   Total time complexity is O(N^2).
 */

int main() {
    // Fast I/O
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int K;
    if (!(cin >> K)) return 0;
    string S;
    if (!(cin >> S)) return 0;

    int N = (int)S.length();
    // To handle cyclic behavior, extend the string by doubling it.
    string S_ext = S + S;
    int N_ext = (int)S_ext.length();

    // Build Suffix Array for the extended string S_ext to find the rank of each position.
    // Using O(N log^2 N) construction.
    vector<int> sa(N_ext), rank(N_ext);
    for (int i = 0; i < N_ext; ++i) {
        sa[i] = i;
        rank[i] = (unsigned char)S_ext[i];
    }

    for (int len = 1; len < N_ext; len <<= 1) {
        auto cmp = [&](int i, int j) {
            if (rank[i] != rank[j]) return rank[i] < rank[j];
            int ri = (i + len < N_ext) ? rank[i + len] : -1;
            int rj = (j + len < N_ext) ? rank[j + len] : -1;
            return ri < rj;
        };
        sort(sa.begin(), sa.end(), cmp);
        vector<int> new_rank(N_ext);
        new_rank[sa[0]] = 0;
        for (int i = 1; i < N_ext; ++i) {
            // Determine if sa[i] is different from sa[i-1] in the current sorted order.
            bool different = cmp(sa[i - 1], sa[i]);
            new_rank[sa[i]] = new_rank[sa[i - 1]] + (different ? 1 : 0);
        }
        rank = new_rank;
    }

    int maxL = 0;
    string best_S_res = "";

    // Iterate through half-lengths M (from (N-1)/2 down to 1) to find maximal even length L = 2M.
    for (int M = (N - 1) / 2; M >= 1; --M) {
        int L = 2 * M;
        // C[i] is 1 if S[i] != S[(i+M)%N], else 0.
        // A substring of length 2M starting at i is K-even if sum of C[j] for j in [i, i+M-1] is <= K.
        vector<int> C(N);
        for (int i = 0; i < N; ++i) {
            C[i] = (S[i] != S[(i + M) % N] ? 1 : 0);
        }

        // Initial window sum of size M.
        int current_diffs = 0;
        for (int j = 0; j < M; ++j) {
            current_diffs += C[j];
        }

        int best_idx_for_this_M = -1;
        int best_rank_for_this_M = N_ext + 1;

        // Check all starting positions i from 0 to N-1 using a sliding window.
        for (int i = 0; i < N; ++i) {
            if (current_diffs <= K) {
                // If current_diffs <= K, i is a valid starting position.
                // Pick the one with the smallest rank in the Suffix Array.
                if (rank[i] < best_rank_for_this_M) {
                    best_rank_for_this_M = rank[i];
                    best_idx_for_this_M = i;
                }
            }
            // Slide the window: subtract C[i] and add C[(i + M) % N].
            current_diffs = current_diffs - C[i] + C[(i + M) % N];
        }

        // If we found any valid starting position for this M, we have our maximum L.
        if (best_idx_for_this_M != -1) {
            maxL = L;
            best_S_res = S_ext.substr(best_idx_for_this_M, L);
            break;
        }
    }

    // Output the result.
    if (maxL > 0) {
        cout << best_S_res << endl;
    } else {
        cout << endl;
    }

    return 0;
}
