```cpp
#include <iostream>
#include <vector>

using namespace std;

/**
 * Problem: Goggle
 * The task is to implement a search engine that can find sequences containing
 * certain numbers and not containing others. Numbers to be excluded are 
 * provided as negative integers.
 * 
 * Input:
 * - n: number of sequences in the database
 * - m: number of queries
 * - n lines: each describing a sequence, starting with its length k, then k integers
 * - m lines: each describing a query, starting with its length l, then l integers
 * 
 * Constraints:
 * - n, m <= 10
 * - k, l <= 10
 * - Sequence elements are between 1 and 100
 * - Query elements are between -100 and 100, non-zero
 * - Time limit: 0.25s
 * 
 * Complexity:
 * - O(m * n * l * k) where each loop is at most 10.
 * - Total operations roughly 10^4, which easily fits within 0.25s.
 */

struct Sequence {
    vector<int> elements;
};

int main() {
    // Speed up standard input and output
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    // Read the number of database sequences and the number of queries
    if (!(cin >> n >> m)) return 0;

    // Store the database sequences
    vector<Sequence> database(n);
    for (int i = 0; i < n; ++i) {
        int k;
        if (!(cin >> k)) break;
        database[i].elements.resize(k);
        for (int j = 0; j < k; ++j) {
            cin >> database[i].elements[j];
        }
    }

    // Process each query
    for (int i = 0; i < m; ++i) {
        int l;
        if (!(cin >> l)) break;
        vector<int> query(l);
        for (int j = 0; j < l; ++j) {
            cin >> query[j];
        }

        // To keep track of indices of matching sequences
        vector<int> matched_indices;
        for (int seq_idx = 0; seq_idx < n; ++seq_idx) {
            bool possible = true;
            for (int q_val : query) {
                if (q_val > 0) {
                    // If query element is positive, the sequence must contain it
                    bool found = false;
                    for (int element : database[seq_idx].elements) {
                        if (element == q_val) {
                            found = true;
                            break;
                        }
                    }
                    if (!found) {
                        possible = false;
                        break;
                    }
                } else {
                    // If query element is negative, the sequence must NOT contain its absolute value
                    int target = -q_val;
                    bool found = false;
                    for (int element : database[seq_idx].elements) {
                        if (element == target) {
                            found = true;
                            break;
                        }
                    }
                    if (found) {
                        possible = false;
                        break;
                    }
                }
            }
            // If the sequence satisfies all query conditions, add its index to results
            if (possible) {
                matched_indices.push_back(seq_idx);
            }
        }

        // Output the results for the current query
        cout << matched_indices.size() << "\n";
        for (int idx : matched_indices) {
            // Print sequence length followed by the elements themselves
            cout << database[idx].elements.size();
            for (int val : database[idx].elements) {
                cout << " " << val;
            }
            cout << "\n";
        }
    }

    return 0;
}
```
