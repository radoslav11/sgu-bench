```cpp
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

/**
 * Problem: Synchrograph
 * 
 * A vertex v in a synchrograph is alive if, after any valid sequence of fires, 
 * it is potentially alive. A vertex is potentially alive if there is a sequence 
 * of fires such that the vertex itself can be fired. 
 * 
 * In a synchrograph, firing a vertex v:
 * - Decreases each incoming arc (u, v) by 1.
 * - Increases each outgoing arc (v, w) by 1.
 * 
 * For a vertex v to be potentially alive, its incoming arc marks must all be >= 1.
 * If we fire a vertex v, its incoming arc marks decrease. For v to be "alive" 
 * (remains potentially alive regardless of any valid sequence of fires), 
 * its incoming arcs must be replenished.
 * The only way to replenish an incoming arc (u, v) is by firing vertex u.
 * 
 * Thus, a vertex v is alive if:
 * 1. It has no incoming arcs (it can be fired as many times as needed, and 
 *    firing it only increases its outgoing marks).
 * 2. All its incoming neighbors u (where (u, v) is an arc) are themselves alive, 
 *    which ensures that any depletion of the arc (u, v) due to firing v can 
 *    be compensated for by firing u infinitely many times.
 * 
 * If a vertex has incoming arcs but one of its predecessors is not alive, 
 * that predecessor can only be fired a finite number of times, meaning the 
 * incoming arc to v can only be replenished a finite number of times.
 * This would eventually make the arc (u, v) zero, potentially making v 
 * no longer able to be fired.
 * 
 * Therefore, the algorithm is to find all vertices v that satisfy:
 * (In(v) == 0) OR (forall u in In(v), u is alive).
 * This can be computed using a Kahn-like approach (topological sort variation).
 */

int main() {
    // Optimization for fast I/O
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    // Using 1-based indexing for vertices as per the problem
    vector<vector<int>> adj(N + 1);
    vector<int> in_degree(N + 1, 0);

    for (int i = 0; i < M; ++i) {
        int u, v;
        long long w; // mark value, though not used in the logic
        cin >> u >> v >> w;
        adj[u].push_back(v);
        in_degree[v]++;
    }

    vector<bool> alive(N + 1, false);
    // We need to track how many predecessors of each vertex are currently marked as 'alive'.
    // A vertex becomes 'alive' when all its incoming neighbors are 'alive'.
    vector<int> alive_predecessors_count(N + 1, 0);
    
    // in_degree[i] currently represents the total number of incoming edges for vertex i.
    // We'll use a copy for the algorithm to keep track of remaining predecessors.
    vector<int> remaining_in_degree = in_degree;
    queue<int> q;

    // Initially, all vertices with in-degree 0 are alive.
    for (int i = 1; i <= N; ++i) {
        if (in_degree[i] == 0) {
            alive[i] = true;
            q.push(i);
        }
    }

    // Propagate the 'alive' status through the graph.
    while (!q.empty()) {
        int u = q.front();
        q.pop();

        for (int v : adj[u]) {
            if (!alive[v]) {
                // One more predecessor of v is now confirmed to be alive.
                alive_predecessors_count[v]++;
                // If all predecessors of v are now marked as alive, v is also alive.
                if (alive_predecessors_count[v] == in_degree[v]) {
                    alive[v] = true;
                    q.push(v);
                }
            }
        }
    }

    // Output the results: 1 if alive, 0 otherwise.
    for (int i = 1; i <= N; ++i) {
        cout << (alive[i] ? 1 : 0) << (i == N ? "" : "\n");
    }
    cout << endl;

    return 0;
}
```
