#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <map>
#include <set>
#include <cctype>

using namespace std;

struct Symbol {
    string val;
    bool is_terminal;
    bool operator<(const Symbol& other) const {
        if (val != other.val) return val < other.val;
        return is_terminal < other.is_terminal;
    }
    bool operator==(const Symbol& other) const {
        return val == other.val && is_terminal == other.is_terminal;
    }
};

struct Segment {
    bool is_call;
    string name;
    char if_char;
    vector<Segment> then_segments;
    vector<pair<char, vector<Segment>>> branches;
    bool ends_in_error;
};

vector<string> tokenize(string s) {
    vector<string> tokens;
    for (int i = 0; i < s.length(); ) {
        if (isspace(s[i])) { i++; continue; }
        if (s[i] == '\'') {
            string t = "'";
            t += s[i + 1];
            t += "'";
            tokens.push_back(t);
            i += 3;
        } else if (isalpha(s[i])) {
            string t;
            while (i < s.length() && isalpha(s[i])) t += s[i++];
            tokens.push_back(t);
        } else if (isdigit(s[i])) {
            string t;
            while (i < s.length() && isdigit(s[i])) t += s[i++];
            tokens.push_back(t);
        } else {
            string t;
            t += s[i++];
            tokens.push_back(t);
        }
    }
    return tokens;
}

vector<Segment> parse_segments(const vector<string>& tokens, int& index);

Segment parse_if_branch(const vector<string>& tokens, int& index) {
    // Expect: If Peek='c' Then Begin Skip;
    index++; // If
    index++; // Peek
    index++; // =
    char c = tokens[index++][1];
    index++; // 'c'
    index++; // Then
    index++; // Begin
    index++; // Skip
    index++; // ;
    vector<Segment> then_segs = parse_segments(tokens, index);
    index++; // End
    return {false, "", c, then_segs, {}, false};
}

Segment parse_if(const vector<string>& tokens, int& index) {
    Segment first = parse_if_branch(tokens, index);
    vector<pair<char, vector<Segment>>> branches;
    branches.push_back({first.if_char, first.then_segments});
    bool ends_in_error = false;

    while (index < tokens.size() && tokens[index] == "Else") {
        index++; // Else
        if (index < tokens.size() && tokens[index] == "If") {
            Segment next_branch = parse_if_branch(tokens, index);
            branches.push_back({next_branch.if_char, next_branch.then_segments});
        } else if (index < tokens.size() && tokens[index] == "Error") {
            index++; // Error
            index++; // ;
            ends_in_error = true;
            break;
        }
    }
    return {false, "", '\0', {}, branches, ends_in_error};
}

vector<Segment> parse_segments(const vector<string>& tokens, int& index) {
    vector<Segment> segments;
    while (index < tokens.size() && tokens[index] != "End" && tokens[index] != "Else") {
        if (tokens[index] == "If") {
            segments.push_back(parse_if(tokens, index));
        } else {
            segments.push_back({true, tokens[index], '\0', {}, {}, false});
            index++;
        }
    }
    return segments;
}

void parse_proc(const vector<string>& tokens, int& index, map<string, vector<Segment>>& procs) {
    index++; // Procedure/Function
    string name = tokens[index++];
    if (index < tokens.size() && tokens[index] == ":") {
        index += 3; // :Char;
    }
    if (index < tokens.size() && tokens[index] == "Forward") {
        index += 2; // Forward;
        return;
    }
    index++; // Begin
    procs[name] = parse_segments(tokens, index);
    index++; // End
    index++; // ;
}

set<vector<Symbol>> find_paths(const vector<Segment>& segments, const map<string, vector<Segment>>& procs) {
    if (segments.empty()) return {{}};
    
    set<vector<Symbol>> results;
    const Segment& seg = segments[0];
    
    if (seg.is_call) {
        set<vector<Symbol>> rest = find_paths(vector<Segment>(segments.begin() + 1, segments.end()), procs);
        for (auto path : rest) {
            if (seg.name != "Skip") {
                string n = seg.name;
                for (char &c : n) c = tolower(c);
                path.insert(path.begin(), {n, false});
            }
            results.insert(path);
        }
    } else {
        // If-else structure
        for (auto& branch : seg.branches) {
            set<vector<Symbol>> then_paths = find_paths(branch.second, procs);
            set<vector<Symbol>> after_paths = find_paths(vector<Segment>(segments.begin() + 1, segments.end()), procs);
            for (auto tp : then_paths) {
                for (auto ap : after_paths) {
                    vector<Symbol> combined = {{string(1, branch.first), true}};
                    combined.insert(combined.end(), tp.begin(), tp.end());
                    combined.insert(combined.end(), ap.begin(), ap.end());
                    results.insert(combined);
                }
            }
        }
        if (!seg.ends_in_error) {
            set<vector<Symbol>> after_paths = find_paths(vector<Segment>(segments.begin() + 1, segments.end()), procs);
            for (auto ap : after_paths) {
                results.insert(ap);
            }
        }
    }
    return results;
}

string format_path(const vector<Symbol>& path) {
    string res = "";
    for (int i = 0; i < path.size(); ++i) {
        if (path[i].is_terminal) {
            int j = i;
            string combined = "";
            while (j < path.size() && path[j].is_terminal) {
                combined += path[j].val[0];
                j++;
            }
            res += "'" + combined + "'";
            i = j - 1;
        } else {
            res += "<" + path[i].val + ">";
        }
    }
    return res;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);
    string line, input;
    while (getline(cin, line)) input += line + "\n";
    vector<string> tokens = tokenize(input);
    map<string, vector<Segment>> procs;
    int idx = 0;
    while (idx < tokens.size()) {
        if (tokens[idx] == "Program" || tokens[idx] == "Var") {
            // Skip declarations
            if (tokens[idx] == "Program") idx += 2;
            else {
                idx++;
                while (idx < tokens.size() && tokens[idx] != "Procedure" && tokens[idx] != "Function" && tokens[idx] != "Begin") idx++;
            }
        } else if (tokens[idx] == "Procedure" || tokens[idx] == "Function") {
            parse_proc(tokens, idx, procs);
        } else {
            idx++;
        }
    }

    vector<string> proc_names;
    for (auto const& [name, segs] : procs) proc_names.push_back(name);

    vector<string> output_lines;
    for (const string& name : proc_names) {
        set<vector<Symbol>> paths = find_paths(procs[name], procs);
        set<string> formatted_paths;
        for (auto path : paths) {
            formatted_paths.insert(format_path(path));
        }
        string line_out = "<" + name + ">::=";
        bool first = true;
        for (const string& p : formatted_paths) {
            if (!first) line_out += "|";
            line_out += p;
            first = false;
        }
        output_lines.push_back(line_out);
    }
    
    sort(output_lines.begin(), output_lines.end());
    for (const string& l : output_lines) cout << l << "\n";

    return 0;
}
