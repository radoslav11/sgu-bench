```cpp
#include <iostream>
#include <vector>
#include <stack>

using namespace std;

/**
 * Problem: 216. Royal Federation
 * 
 * The goal is to partition N cities (forming a tree) into several provinces.
 * Each province must have between B and 3B cities.
 * Each province has a headquarters (capital) such that the path from any city in 
 * the province to the capital consists of only cities within the same province, 
 * except potentially the capital itself.
 * 
 * Algorithm:
 * A DFS-based approach with a stack to keep track of cities not yet assigned to a province.
 * 1. When visiting a node u, record the current stack size.
 * 2. Recursively visit each child v of u.
 * 3. After returning from v, if the number of new cities added to the stack (from the 
 *    subtree of v) is at least B, group these cities into a new province with u 
 *    as the capital.
 * 4. After visiting all children, push u onto the stack.
 * 5. Any remaining cities in the stack after the main DFS are added to the last 
 *    province created. 
 * 
 * Correctness:
 * - Each newly formed province has size between B and 2B-1.
 * - The remaining nodes in the stack (at most B nodes) are added to the last province,
 *   making its size at most (2B-1) + B = 3B-1.
 * - The path property is maintained because the cities added to a province under 
 *   node u are from the subtrees of u's children, and the path to u in the tree 
 *   only passes through u's descendants in those subtrees.
 */

const int MAXN = 10005;
vector<int> adj[MAXN];
int province_assignment[MAXN];
vector<int> capitals;
stack<int> stk;
int province_count = 0;
int N, B;

void dfs(int u, int p) {
    int size_before = stk.size();
    for (int v : adj[u]) {
        if (v == p) continue;
        dfs(v, u);
        // If the nodes added to the stack by child v's subtree is >= B, form a province.
        if ((int)stk.size() - size_before >= B) {
            province_count++;
            capitals.push_back(u);
            while ((int)stk.size() > size_before) {
                province_assignment[stk.top()] = province_count;
                stk.pop();
            }
        }
    }
    stk.push(u);
}

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> N >> B)) return 0;

    for (int i = 0; i < N - 1; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    // Special case: if B > N, no province can be formed, but B <= N is guaranteed.
    if (N < B) {
        cout << 0 << endl;
        return 0;
    }

    dfs(1, 0);

    // If no provinces were formed during DFS, all nodes must form one province.
    if (province_count == 0) {
        province_count = 1;
        capitals.push_back(1);
    }

    // Any remaining cities in the stack are added to the last created province.
    while (!stk.empty()) {
        province_assignment[stk.top()] = province_count;
        stk.pop();
    }

    // Output the result
    cout << province_count << "\n";
    for (int i = 1; i <= N; ++i) {
        cout << province_assignment[i] << (i == N ? "" : " ");
    }
    cout << "\n";
    for (int i = 0; i < (int)capitals.size(); ++i) {
        cout << capitals[i] << (i == (int)capitals.size() - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
```
