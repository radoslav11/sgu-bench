#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <map>

using namespace std;

/**
 * Problem: 242. Student's Morning
 * Task: Select exactly K universities such that each is visited by at least two students.
 * Constraint: Each student is in at most one university and prefers it.
 * 
 * Model as Min-Cost Max-Flow:
 * - Source S, Sink T.
 * - S -> University U_i: 
 *     - Edge 1: capacity 2, cost -1
 *     - Edge 2: capacity N, cost 0
 * - U_i -> Student S_j: capacity 1, cost 0 (if student j prefers university i)
 * - Student S_j -> T: capacity 1, cost 0
 * 
 * To find K universities, we look for a flow of exactly 2K with min cost <= -2K.
 * If cost is <= -2K, it means we found at least K universities with at least 2 students each.
 * We can output exactly K of them.
 */

const int INF = 1e9;

struct Edge {
    int to, cap, flow, cost, rev;
};

vector<vector<Edge>> adj;
vector<int> dist;
vector<int> parent_edge;
vector<int> parent_node;

void add_edge(int u, int v, int cap, int cost) {
    adj[u].push_back({v, cap, 0, cost, (int)adj[v].size()});
    adj[v].push_back({u, 0, 0, -cost, (int)adj[u].size() - 1});
}

bool spfa(int s, int t, int &flow, int &cost, int target_flow) {
    dist.assign(adj.size(), INF);
    parent_node.assign(adj.size(), -1);
    parent_edge.assign(adj.size(), -1);
    vector<bool> in_queue(adj.size(), false);
    queue<int> q;

    dist[s] = 0;
    q.push(s);
    in_queue[s] = true;

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        in_queue[u] = false;

        for (int i = 0; i < (int)adj[u].size(); ++i) {
            Edge &e = adj[u][i];
            if (e.cap - e.flow > 0 && dist[e.to] > dist[u] + e.cost) {
                dist[e.to] = dist[u] + e.cost;
                parent_node[e.to] = u;
                parent_edge[e.to] = i;
                if (!in_queue[e.to]) {
                    q.push(e.to);
                    in_queue[e.to] = true;
                }
            }
        }
    }

    if (dist[t] == INF) return false;

    int push = target_flow - flow;
    int curr = t;
    while (curr != s) {
        int prev = parent_node[curr];
        int idx = parent_edge[curr];
        push = min(push, adj[prev][idx].cap - adj[prev][idx].flow);
        curr = prev;
    }

    flow += push;
    cost += push * dist[t];
    curr = t;
    while (curr != s) {
        int prev = parent_node[curr];
        int idx = parent_edge[curr];
        adj[prev][idx].flow += push;
        int rev_idx = adj[prev][idx].rev;
        adj[curr][rev_idx].flow -= push;
        curr = prev;
    }

    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, K;
    if (!(cin >> N >> K)) return 0;

    vector<vector<int>> student_prefs(N);
    map<int, int> uni_count;
    for (int i = 0; i < N; ++i) {
        int num_pref;
        cin >> num_pref;
        for (int j = 0; j < num_pref; ++j) {
            int uni;
            cin >> uni;
            student_prefs[i].push_back(uni);
            uni_count[uni]++;
        }
    }

    if (K == 0) {
        cout << "YES" << endl;
        return 0;
    }

    if (K > N / 2) {
        cout << "NO" << endl;
        return 0;
    }

    vector<int> useful_unis;
    for (auto const& [uni, count] : uni_count) {
        if (count >= 2) {
            useful_unis.push_back(uni);
        }
    }

    if ((int)useful_unis.size() < K) {
        cout << "NO" << endl;
        return 0;
    }

    int M = useful_unis.size();
    map<int, int> uni_to_idx;
    for (int i = 0; i < M; ++i) {
        uni_to_idx[useful_unis[i]] = i;
    }

    int S = 0, T = M + N + 1;
    adj.resize(T + 1);

    for (int i = 0; i < M; ++i) {
        add_edge(S, i + 1, 2, -1);
        add_edge(S, i + 1, N, 0);
    }

    for (int i = 0; i < N; ++i) {
        for (int uni : student_prefs[i]) {
            if (uni_to_idx.count(uni)) {
                add_edge(uni_to_idx[uni] + 1, M + i + 1, 1, 0);
            }
        }
        add_edge(M + i + 1, T, 1, 0);
    }

    int flow = 0, cost = 0;
    while (flow < 2 * K && spfa(S, T, flow, cost, 2 * K));

    if (flow == 2 * K && cost <= -2 * K) {
        cout << "YES" << endl;
        int found_k = 0;
        for (int i = 0; i < M && found_k < K; ++i) {
            // Check if the -1 cost edge has a flow of 2
            bool has_two = false;
            for (auto &e : adj[S]) {
                if (e.to == i + 1 && e.cost == -1 && e.flow == 2) {
                    has_two = true;
                    break;
                }
            }
            if (has_two) {
                found_k++;
                vector<int> students;
                for (auto &e : adj[i + 1]) {
                    if (e.to > M && e.to <= M + N && e.flow == 1) {
                        students.push_back(e.to - M);
                    }
                }
                cout << students.size();
                for (int s_id : students) cout << " " << s_id;
                cout << "\n";
            }
        }
    } else {
        cout << "NO" << endl;
    }

    return 0;
}
