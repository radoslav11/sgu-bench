#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Problem Analysis:
 * The goal is to simulate a magic trick where an N x N grid of pictures is presented.
 * We start at picture 1. In each turn, the audience moves the finger K_i times (K_i >= N)
 * to adjacent available pictures. Then, the magician removes some available pictures.
 * Finally, only one picture should remain. All K_i must be unique.
 * 
 * Strategy:
 * 1. Generate a snake-like path through the N x N grid.
 *    The snake path ensures that every subsequent picture in the path is physically adjacent to the previous one.
 *    This path visits all N^2 pictures.
 * 2. Use the snake path to manage moves and removals.
 *    In each turn, if we are at position P_curr and want to move K steps, we move to P_curr+K in the snake path.
 *    The pictures between P_curr and P_curr+K (inclusive of P_curr but exclusive of P_curr+K) 
 *    are removed to ensure we progress through the path.
 * 3. Maintain the current position (index in the snake path) and the number of removed pictures.
 *    As long as the total removed pictures is less than N^2 - 1, we continue to pick a unique K_i.
 *    If a K_i allows us to reach the end of the path (the last picture), we use it to remove all remaining pictures.
 */

int main() {
    // Optimize input/output for speed
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    if (!(cin >> N)) return 0;

    // Generate the snake path through the grid
    vector<int> path;
    path.push_back(0); // Dummy element to make it 1-indexed
    for (int r = 0; r < N; ++r) {
        if (r % 2 == 0) {
            for (int c = 0; c < N; ++c) {
                path.push_back(r * N + c + 1);
            }
        } else {
            for (int c = N - 1; c >= 0; --c) {
                path.push_back(r * N + c + 1);
            }
        }
    }

    int N2 = N * N;
    vector<bool> is_available(N2 + 1, true);
    int curr_idx = 1;
    int removed_total = 0;

    // Available K values: N <= K < 300
    vector<int> K_list;
    for (int k = N; k < 300; ++k) {
        K_list.push_back(k);
    }

    int k_idx = 0;
    while (removed_total < N2 - 1 && k_idx < K_list.size()) {
        int K = K_list[k_idx++];
        int new_idx = curr_idx + K;

        // If moving K steps goes beyond the total number of pictures, we cannot use this K
        if (new_idx > N2) continue;

        vector<int> to_remove;
        // Check if this turn will be the final turn to leave only one picture
        if (removed_total + K >= N2 - 1) {
            // This is the last turn. We must reach the end or enough to leave only one.
            // To ensure only one picture (at new_idx) remains, remove all other available pictures.
            // But we must actually be able to perform the K moves. 
            // If new_idx is exactly N2, we can reach the last picture.
            if (new_idx == N2) {
                for (int i = 1; i <= N2; ++i) {
                    if (is_available[path[i]] && i != new_idx) {
                        to_remove.push_back(path[i]);
                    }
                }
            } else {
                // If K is too large or something, we can't make it the "last" turn properly
                // unless it's exactly the end. We skip if new_idx is not N2.
                continue;
            }
        } else {
            // Standard turn: remove pictures along the snake path from current position to just before the new position.
            for (int i = curr_idx; i < new_idx; ++i) {
                if (is_available[path[i]]) {
                    to_remove.push_back(path[i]);
                }
            }
        }

        // If we found pictures to remove, output the turn
        if (!to_remove.empty()) {
            cout << K;
            for (int pic : to_remove) {
                cout << " " << pic;
                is_available[pic] = false;
                removed_total++;
            }
            cout << "\n";
            curr_idx = new_idx;
        }

        // If we finished the trick, stop.
        if (removed_total == N2 - 1) break;
    }

    return 0;
}
