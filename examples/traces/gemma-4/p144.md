#include <iostream>
#include <iomanip>

/**
 * Problem: 144. Meeting
 * 
 * Analysis:
 * Two people arrive at a meeting point at times t1 and t2, respectively.
 * Both t1 and t2 are independently and uniformly distributed in the interval [X, Y].
 * The time duration of the interval is T = Y - X hours.
 * The meeting condition is |t1 - t2| <= Z minutes, where Z is in minutes.
 * 
 * Let's convert everything to hours:
 * T = Y - X (in hours)
 * Z_hours = Z / 60.0 (in hours)
 * 
 * The problem asks for the probability P(|t1 - t2| <= Z_hours) for t1, t2 in [X, Y].
 * Since t1 and t2 are uniformly distributed in [X, Y], we can simplify the interval to [0, T].
 * The total area of the sample space is T * T.
 * The area where they do NOT meet is the region where |t1 - t2| > Z_hours.
 * This region consists of two right triangles:
 * 1) t1 - t2 > Z_hours  => t1 > t2 + Z_hours
 * 2) t2 - t1 > Z_hours  => t2 > t1 + Z_hours
 * 
 * Each triangle has side lengths (T - Z_hours).
 * Area of one triangle = 0.5 * (T - Z_hours)^2.
 * Area of two triangles = (T - Z_hours)^2.
 * 
 * Total meeting area = T^2 - (T - Z_hours)^2.
 * Probability P = (T^2 - (T - Z_hours)^2) / T^2.
 * 
 * Let k = Z_hours / T.
 * Since Z <= 60 * (Y - X), we know Z_hours <= T, so 0 < k <= 1.
 * P = 1 - ((T - Z_hours) / T)^2
 * P = 1 - (1 - k)^2
 * P = 1 - (1 - 2k + k^2)
 * P = 2k - k^2.
 * 
 * Sample Input: 11 12 20.0
 * T = 12 - 11 = 1.0
 * Z_hours = 20.0 / 60.0 = 1/3.0
 * k = (1/3.0) / 1.0 = 1/3.0
 * P = 2 * (1/3.0) - (1/3.0)^2 = 2/3 - 1/9 = 5/9 ≈ 0.5555556.
 * 
 * The output format requires 7 decimal digits.
 */

int main() {
    // Using fast I/O for speed, although not strictly necessary for this O(1) problem.
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int X, Y;
    double Z;

    // Read input: X and Y are integers, Z is a real number.
    if (!(std::cin >> X >> Y >> Z)) {
        return 0;
    }

    // T is the total duration in hours.
    double T = static_cast<double>(Y) - static_cast<double>(X);
    
    // Z_hours is the max waiting time in hours.
    double Z_hours = Z / 60.0;

    // k is the ratio of waiting time to the total interval time.
    // Since Z <= 60 * (Y-X), k will be in the range (0, 1].
    double k = Z_hours / T;

    // To handle potential floating-point errors where k might slightly exceed 1.0.
    if (k > 1.0) k = 1.0;

    // Calculate probability P = 2k - k^2.
    double probability = 2.0 * k - k * k;

    // Output the probability with 7 decimal digits.
    // std::fixed and std::setprecision(7) ensure proper rounding and formatting.
    std::cout << std::fixed << std::setprecision(7) << probability << std::endl;

    return 0;
}
