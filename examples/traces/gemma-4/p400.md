int count_to_cover(long long R, vector<int>& capacities) {
    if (R <= 0) return 0;
    sort(capacities.rbegin(), capacities.rend());
    int count = 0;
    for (int cap : capacities) {
        if (R <= 0) break;
        if (cap > 0) {
            count++;
            R -= cap;
        }
    }
    return (R > 0) ? (count + 1) : count; // This part is wrong; if R > 0 after all caps, it's not possible.
    // However, for MaxRank, it's always possible to distribute balloons.
}

// A more accurate count_to_cover
int count_to_cover_fixed(long long R, vector<int>& capacities) {
    if (R <= 0) return 0;
    sort(capacities.rbegin(), capacities.rend());
    int count = 0;
    for (int cap : capacities) {
        if (R <= 0) break;
        count++;
        R -= cap;
    }
    return (R > 0) ? 1000000 : count; 
}
