#include <iostream>
#include <vector>
#include <algorithm>
#include <map>

using namespace std;

/**
 * Problem Summary:
 * We have n dumbbells with mass mi and cost ci.
 * We need to find the maximum number of sets (t) such that:
 * 1. Each set contains exactly k dumbbells.
 * 2. Each set has dumbbells of k distinct masses.
 * 3. For each pair of sets, the masses of dumbbells must coincide.
 * Primary goal: Maximize t.
 * Secondary goal: Maximize the total cost of the t * k dumbbells.
 * 
 * Constraints:
 * n, k <= 4000.
 * mi, ci <= 4000.
 * 
 * Approach:
 * 1. Group the costs of dumbbells by their mass.
 * 2. For each mass, sort its costs in descending order to easily pick the most expensive ones.
 * 3. Precompute prefix sums for each mass to allow O(1) calculation of the sum of the top t costs.
 * 4. Iterate t from 1 up to n/k.
 * 5. For a fixed t, identify all masses that have at least t dumbbells.
 * 6. If the number of such masses is at least k, then t sets are possible.
 * 7. For each such mass, calculate the sum of its t largest costs.
 * 8. Pick the k largest such sums to maximize the total cost for the current t.
 * 9. The largest t that satisfies the condition is our maximum number of sets.
 * 10. The secondary goal (maximize cost) is solved by picking the k largest sums for that t.
 * 
 * Complexity:
 * Time Complexity: O(n * M), where M is the number of unique masses. M <= n.
 * Space Complexity: O(n) to store the dumbbell information.
 */

int main() {
    // Faster I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, k;
    if (!(cin >> n >> k)) return 0;

    // Using a map to group costs by mass. 
    // Since mass can go up to 4000, a vector of vectors would also work.
    map<int, vector<int>> mass_map;
    for (int i = 0; i < n; ++i) {
        int m, c;
        cin >> m >> c;
        mass_map[m].push_back(c);
    }

    // For each mass, sort costs descending and compute prefix sums.
    vector<vector<long long>> mass_prefix_sums;
    for (auto& entry : mass_map) {
        vector<int>& costs = entry.second;
        sort(costs.begin(), costs.end(), greater<int>());
        
        vector<long long> p_sum(costs.size() + 1, 0);
        for (int i = 0; i < (int)costs.size(); ++i) {
            p_sum[i + 1] = p_sum[i] + costs[i];
        }
        mass_prefix_sums.push_back(p_sum);
    }

    int max_t = 0;
    long long max_s = 0;

    // Iterate through all possible number of sets t.
    for (int t = 1; t <= n / k; ++t) {
        vector<long long> current_sums;
        for (const auto& p_sum : mass_prefix_sums) {
            // A mass is viable for t sets if it has at least t dumbbells.
            if ((int)p_sum.size() - 1 >= t) {
                current_sums.push_back(p_sum[t]);
            }
        }

        // If we have at least k viable masses, t sets can be formed.
        if ((int)current_sums.size() >= k) {
            max_t = t;
            // To maximize the total cost for this t, pick the k largest sums.
            // Use nth_element to get the top k elements in O(M) time.
            nth_element(current_sums.begin(), current_sums.begin() + k - 1, current_sums.end(), greater<long long>());
            
            long long current_total_cost = 0;
            for (int i = 0; i < k; ++i) {
                current_total_cost += current_sums[i];
            }
            max_s = current_total_cost;
        } else {
            // If we cannot form t sets, we cannot form any more than t sets.
            break;
        }
    }

    // Output the results. If no sets were possible, max_t and max_s will be 0.
    cout << max_t << " " << max_s << endl;

    return 0;
}
