#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <random>
#include <ctime>

using namespace std;

struct Neighbor {
    int mask;
    int action_type; // 0: +i, 1: -i, 2: ++i, 3: --i
    int idx;        // 0-indexed i
};

int N, K;
int target_count = 0;
bool visited[1 << 20];
int degree[1 << 20];
Neighbor neighbors_pool[1 << 20][64];
int neighbors_count[1 << 20];
int path_masks[1 << 20];
int path_action_type[1 << 20];
int path_action_idx[1 << 20];
int current_path_len = 0;

void get_neighbors(int u, int n, int k, int* count, Neighbor* pool) {
    *count = 0;
    int pop = __builtin_popcount(u);
    for (int i = 0; i < n; ++i) {
        if ((u >> i) & 1) {
            if (pop > 1) pool[*count++] = {u ^ (1 << i), 1, i};
            if (i > 0 && !((u >> (i - 1)) & 1)) pool[*count++] = {(u ^ (1 << i)) | (1 << (i - 1)), 3, i};
            if (i < n - 1 && !((u >> (i + 1)) & 1)) pool[*count++] = {(u ^ (1 << i)) | (1 << (i + 1)), 2, i};
        } else {
            if (pop < k) pool[*count++] = {u | (1 << i), 0, i};
        }
    }
}

bool dfs(int u) {
    visited[u] = true;
    path_masks[current_path_len++] = u;
    if (current_path_len == target_count) return true;

    int count = neighbors_count[u];
    static pair<int, int> sorted_neighbors[64];
    for (int i = 0; i < count; ++i) {
        int v = neighbors_pool[u][i].mask;
        sorted_neighbors[i] = {visited[v] ? 1000 : degree[v], i};
    }
    sort(sorted_neighbors, sorted_neighbors + count);

    for (int i = 0; i < count; ++i) {
        int idx = sorted_neighbors[i].second;
        int v = neighbors_pool[u][idx].mask;
        if (!visited[v]) {
            path_action_type[current_path_len - 1] = neighbors_pool[u][idx].action_type;
            path_action_idx[current_path_len - 1] = neighbors_pool[u][idx].idx;

            for (int j = 0; j < neighbors_count[v]; ++j) {
                degree[neighbors_pool[v][j].mask]--;
            }

            if (dfs(v)) return true;

            for (int j = 0; j < neighbors_count[v]; ++j) {
                degree[neighbors_pool[v][j].mask]++;
            }
        }
    }

    visited[u] = false;
    current_path_len--;
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> N >> K)) return 0;

    for (int i = 1; i < (1 << N); ++i) {
        int pop = __builtin_popcount(i);
        if (pop <= K) target_count++;
    }

    for (int i = 1; i < (1 << N); ++i) {
        int pop = __builtin_popcount(i);
        if (pop <= K) {
            get_neighbors(i, N, K, &neighbors_count[i], neighbors_pool[i]);
            for (int j = 0; j < neighbors_count[i]; ++j) {
                int v = neighbors_pool[i][j].mask;
                if (v > 0 && __builtin_popcount(v) <= K) {
                    degree[i]++;
                }
            }
        }
    }

    // Try starting from single-element sets
    for (int start_node = 1; start_node < (1 << N); ++start_node) {
        if (__builtin_popcount(start_node) == 1) {
            // Reset everything for each potential start node
            for (int i = 0; i < (1 << N); ++i) {
                visited[i] = false;
                current_path_len = 0;
            }
            
            // Re-calculate degree for the specific current state
            for (int i = 1; i < (1 << N); ++i) {
                if (__builtin_popcount(i) <= K) {
                    int d = 0;
                    for (int j = 0; j < neighbors_count[i]; ++j) {
                        int v = neighbors_pool[i][j].mask;
                        if (v > 0 && __builtin_popcount(v) <= K) d++;
                    }
                    degree[i] = d;
                }
            }

            // To use the path_action_type/idx efficiently, we need to capture the first action too.
            // The DFS will fill path_action_type[current_path_len-1] which is the action from S_i to S_{i+1}.
            // We need the action from 0 to S_1 separately.
            
            // A small hack to get the first action from 0 to S_1
            int first_mask = start_node;
            int first_idx = __builtin_ctz(first_mask);
            
            // Before DFS, we need the action 0 -> S_1
            // We'll handle this by manually putting it in before the loop
            
            // However, we must also correctly handle degree decrementing for the DFS
            // The degree[v] should be the number of UNVISITED neighbors of v.
            // Initial degree is calculated above.
            
            // Before calling DFS, we must manually mark start_node as visited
            // and decrement degrees of its neighbors.
            visited[start_node] = true;
            current_path_len = 1;
            path_masks[0] = start_node;
            for (int j = 0; j < neighbors_count[start_node]; ++j) {
                int v = neighbors_pool[start_node][j].mask;
                if (v > 0 && __builtin_popcount(v) <= K) degree[v]--;
            }

            if (dfs(start_node)) {
                // Success! Now construct the output.
                string res = "";
                res += "+"; res += to_string(first_idx + 1);
                for (int i = 0; i < current_path_len - 1; ++i) {
                    int type = path_action_type[i];
                    int idx = path_action_idx[i];
                    if (type == 0) res += "+" + to_string(idx + 1);
                    else if (type == 1) res += "-" + to_string(idx + 1);
                    else if (type == 2) res += "++" + to_string(idx + 1);
                    else if (type == 3) res += "--" + to_string(idx + 1);
                }
                int last_mask = path_masks[current_path_len - 1];
                int last_idx = __builtin_ctz(last_mask);
                res += "-" + to_string(last_idx + 1);
                cout << res << endl;
                return 0;
            }
        }
    }

    cout << 0 << endl;
    return 0;
}
