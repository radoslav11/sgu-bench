```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>

using namespace std;

/**
 * Problem: Mr. Beetle II
 * The beetle travels in a straight line from (x1, y1) to (x2, y2).
 * We need to find the coordinates of the bottom-left corner of the n-th cell
 * whose interior is passed through by the line segment.
 * 
 * Since x1, y1, x2, y2 are integers, any line with x1=x2 or y1=y2 will be 
 * a grid line (border), and thus no cell interior will be passed.
 * For non-zero dx and dy, the line passes through a new cell whenever 
 * it crosses an integer x or y coordinate.
 * 
 * The crossing times t in (0, 1) are (k - x1) / dx and (m - y1) / dy.
 * We can find these t values, sort them, and the n-th cell is the one 
 * in the n-th interval formed by these t's.
 * 
 * Memory limit is very tight (4 MB), so we cannot store all possible t values 
 * if they reach 4 * 10^6. However, n is only 10^5, so we only need 
 * the first n unique t's.
 */

int main() {
    // Optimize standard I/O
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    long long x1, y1, x2, y2, n;
    if (!(cin >> x1 >> y1 >> x2 >> y2 >> n)) return 0;

    long long dx = x2 - x1;
    long long dy = y2 - y1;

    // If the line segment lies on a grid line, no cell interiors are passed.
    if (dx == 0 || dy == 0) {
        cout << "no solution" << endl;
        return 0;
    }

    // Pre-calculate parameters for generating t_x and t_y
    long long k_start, k_end, k_step;
    if (x2 > x1) {
        k_start = x1 + 1;
        k_end = x2 - 1;
        k_step = 1;
    } else {
        k_start = x1 - 1;
        k_end = x2 + 1;
        k_step = -1;
    }
    long long k_total = (x2 > x1) ? (x2 - x1 - 1) : (x1 - x2 - 1);
    if (k_total < 0) k_total = 0;

    long long m_start, m_end, m_step;
    if (y2 > y1) {
        m_start = y1 + 1;
        m_end = y2 - 1;
        m_step = 1;
    } else {
        m_start = y1 - 1;
        m_end = y2 + 1;
        m_step = -1;
    }
    long long m_total = (y2 > y1) ? (y2 - y1 - 1) : (y1 - y2 - 1);
    if (m_total < 0) m_total = 0;

    // We need up to n + 1 unique t's to find the n-th interval.
    vector<long double> unique_t;
    unique_t.push_back(0.0);

    long long ki = 0, mi = 0;
    // Two-pointer approach to merge the two sorted lists of t's
    while (ki < k_total || mi < m_total) {
        long double tx_val = 2e18, ty_val = 2e18;
        if (ki < k_total) {
            long long k = k_start + ki * k_step;
            tx_val = (long double)(k - x1) / dx;
        }
        if (mi < m_total) {
            long long m = m_start + mi * m_step;
            ty_val = (long double)(m - y1) / dy;
        }

        long double current_t;
        // Use a small epsilon for floating point comparisons
        if (abs(tx_val - ty_val) < 1e-15) {
            current_t = tx_val;
            ki++;
            mi++;
        } else if (tx_val < ty_val) {
            current_t = tx_val;
            ki++;
        } else {
            current_t = ty_val;
            mi++;
        }

        // Only collect t values in (0, 1)
        if (current_t > 1e-16 && current_t < 1.0 - 1e-16) {
            // Since they are sorted, check for duplicate (though tx/ty are distinct)
            if (unique_t.size() == 1 || current_t > unique_t.back() + 1e-16) {
                unique_t.push_back(current_t);
            }
        }
        // Only need up to n + 1 to define n intervals
        if (unique_t.size() > (size_t)n + 1) break;
    }
    unique_t.push_back(1.0);

    // The number of intervals is unique_t.size() - 1.
    if (n >= (long long)unique_t.size()) {
        cout << "no solution" << endl;
    } else {
        // The n-th interval is (unique_t[n-1], unique_t[n])
        long double t_mid = (unique_t[n - 1] + unique_t[n]) / 2.0;
        long double xm = (long double)x1 + t_mid * dx;
        long double ym = (long double)y1 + t_mid * dy;

        // Use floor with a small epsilon to avoid precision errors
        // with x_mid or y_mid being very close to an integer.
        // Since the line doesn't pass through corners/borders, x_mid and y_mid 
        // will not be integers.
        long long res_x = (long long)floor(xm + 1e-14);
        long long res_y = (long long)floor(ym + 1e-14);

        cout << res_x << " " << res_y << endl;
    }

    return 0;
}
```
