#include <iostream>
#include <string>
#include <algorithm>
#include <vector>

using namespace std;

/**
 * Problem Analysis:
 * The condition that each girl must own the ball during the game is equivalent to 
 * saying that K and N are coprime, i.e., gcd(K, N) = 1.
 * We are asked to find the largest K such that 1 <= K <= N/2 and gcd(K, N) = 1.
 * 
 * Case 1: N is odd.
 * The largest possible K is N/2 (integer division). 
 * Let K = (N-1)/2. Since N is odd, gcd((N-1)/2, N) = gcd((N-1)/2, N - 2*((N-1)/2)) = gcd((N-1)/2, 1) = 1.
 * So for odd N, the answer is K = floor(N/2).
 * 
 * Case 2: N is even.
 * Let M = N/2. Since N is even, any K such that gcd(K, N) = 1 must be odd.
 * If M is even:
 *   The largest odd number K <= M is M-1.
 *   Since gcd(M-1, M) = 1 and M-1 is odd, gcd(M-1, 2M) = gcd(M-1, M) = 1.
 *   So the answer is K = M-1.
 * If M is odd:
 *   The largest odd number K <= M is M (if M is coprime to N) or M-2 (if not).
 *   Since N = 2M, gcd(M, N) = M. If M > 1, gcd(M, N) != 1.
 *   If M = 1, then N = 2, but N >= 3.
 *   If M > 1 and M is odd, then K = M-2 is odd and gcd(M-2, 2M) = gcd(M-2, M) = gcd(M-2, 2) = 1.
 *   So for odd M > 1, the answer is K = M-2.
 * 
 * Time complexity: O(number of digits), which is O(2000) for N up to 10^2000.
 * This is well within the 0.25s time limit.
 */

// Performs BigInt division by 2 (string representation)
string divideBy2(string n) {
    string res = "";
    int remainder = 0;
    for (char c : n) {
        int digit = remainder * 10 + (c - '0');
        res += (char)((digit / 2) + '0');
        remainder = digit % 2;
    }
    size_t first_nonzero = res.find_first_not_of('0');
    if (string::npos == first_nonzero) {
        return "0";
    }
    return res.substr(first_nonzero);
}

// Performs BigInt subtraction: n - x (where x is a small integer)
string subtract(string n, int x) {
    int len = n.length();
    int carry = x;
    for (int i = len - 1; i >= 0 && carry > 0; i--) {
        int digit = (n[i] - '0') - carry;
        if (digit < 0) {
            digit += 10;
            carry = 1;
        } else {
            carry = 0;
        }
        n[i] = (char)(digit + '0');
    }
    size_t first_nonzero = n.find_first_not_of('0');
    if (string::npos == first_nonzero) {
        return "0";
    }
    return n.substr(first_nonzero);
}

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string n;
    if (!(cin >> n)) return 0;

    int last_digit = n.back() - '0';
    if (last_digit % 2 != 0) {
        // N is odd: answer is floor(N/2)
        cout << divideBy2(n) << endl;
    } else {
        // N is even: let M = N/2
        string m = divideBy2(n);
        int last_digit_m = m.back() - '0';
        if (last_digit_m % 2 != 0) {
            // M is odd: answer is M-2 (since M is not coprime to 2M unless M=1)
            // N >= 3, so for even N, N >= 4, so M >= 2. 
            // If M is odd, M >= 3. Thus M-2 >= 1.
            if (m == "1") {
                // M=1 would imply N=2, but N >= 3
                cout << "1" << endl;
            } else {
                cout << subtract(m, 2) << endl;
            }
        } else {
            // M is even: answer is M-1
            cout << subtract(m, 1) << endl;
        }
    }

    return 0;
}
