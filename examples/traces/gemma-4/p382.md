// A simple BigInt implementation to handle very large numerators and denominators.
// For n = 50, the denominator can have about 1225 digits.
struct BigInt {
    vector<int> digits; // digits are stored in reverse order (0-indexed)
    static const int BASE = 10;

    BigInt(long long v = 0) {
        if (v == 0) digits.push_back(0);
        while (v > 0) {
            digits.push_back(v % BASE);
            v /= BASE;
        }
    }

    void multiply(int v) {
        if (v == 0) {
            digits.clear();
            digits.push_back(0);
            return;
        }
        int carry = 0;
        for (size_t i = 0; i < digits.size() || carry; ++i) {
            if (i == digits.size()) digits.push_back(0);
            long long cur = carry + digits[i] * 1LL * v;
            digits[i] = cur % BASE;
            carry = cur / BASE;
        }
        while (digits.size() > 1 && digits.back() == 0) digits.pop_back();
    }

    void multiply(const BigInt& other) {
        if (this->isZero() || other.isZero()) {
            digits.clear();
            digits.push_back(0);
            return;
        }
        vector<int> res(digits.size() + other.digits.size(), 0);
        for (size_t i = 0; i < digits.size(); ++i) {
            int carry = 0;
            for (size_t j = 0; j < other.digits.size() || carry; ++j) {
                long long cur = res[i + j] +
                                digits[i] * 1LL * (j < other.digits.size() ? other.digits[j] : 0) + carry;
                res[i + j] = cur % BASE;
                carry = cur / BASE;
            }
        }
        while (res.size() > 1 && res.back() == 0) res.pop_back();
        digits = res;
    }

    void add(const BigInt& other) {
        int carry = 0;
        for (size_t i = 0; i < max(digits.size(), other.digits.size()) || carry; ++i) {
            if (i == digits.size()) digits.push_back(0);
            int cur = digits[i] + (i < other.digits.size() ? other.digits[i] : 0) + carry;
            digits[i] = cur % BASE;
            carry = cur / BASE;
        }
    }

    void shiftLeft(int n) {
        if (isZero()) return;
        if (digits.size() == 1 && digits[0] == 0) return;
        if (n <= 0) return;
        vector<int> new_digits(n, 0);
        new_digits.insert(new_digits.end(), digits.begin(), digits.end());
        digits = new_digits;
    }

    bool isZero() const {
        return digits.empty() || (digits.size() == 1 && digits[0] == 0);
    }

    bool operator<(const BigInt& other) const {
        if (digits.size() != other.digits.size()) return digits.size() < other.digits.size();
        for (int i = (int)digits.size() - 1; i >= 0; --i) {
            if (digits[i] != other.digits[i]) return digits[i] < other.digits[i];
        }
        return false;
    }

    bool operator==(const BigInt& other) const {
        return digits == other.digits;
    }

    string toString() const {
        if (isZero()) return "0";
        string s = "";
        for (int i = (int)digits.size() - 1; i >= 0; --i) s += (char)(digits[i] + '0');
        return s;
    }
};
