#include <iostream>
#include <string>
#include <unordered_map>
#include <queue>
#include <vector>
#include <algorithm>

using namespace std;

struct Info {
    int cost;
    string parent;
    string added;
};

struct State {
    string s;
    int cost;
    bool operator>(const State& other) const {
        return cost > other.cost;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    if (!(cin >> N)) return 0;
    string S1, S2;
    cin >> S1 >> S2;

    if (S1 == S2) {
        cout << S1 << "\n";
        return 0;
    }

    unordered_map<string, Info> dist1, dist2;
    dist1.reserve(100000);
    dist2.reserve(100000);

    priority_queue<State, vector<State>, greater<State>> pq1, pq2;

    dist1[S1] = {0, "", ""};
    pq1.push({S1, 0});

    dist2[S2] = {0, "", ""};
    pq2.push({S2, 0});

    int best_cost = 1e9;
    string best_mid = "";

    auto get_next = [&](const string& P) {
        vector<pair<string, pair<int, string>>> res;
        res.reserve(N);
        for (int k = 0; k < N; ++k) {
            string next_P = P.substr(k + 1) + P[k] + P.substr(0, k);
            string added = (k == 0) ? string(1, P[0]) : string(1, P[k]) + P.substr(0, k);
            res.push_back({next_P, {k + 1, added}});
        }
        return res;
    };

    auto get_prev = [&](const string& Q) {
        vector<pair<string, pair<int, string>>> res;
        res.reserve(N);
        for (int j = 0; j < N; ++j) {
            string prev_P = Q.substr(j + 1) + Q[j] + Q.substr(0, j);
            int k = N - 1 - j;
            string added = (k == 0) ? string(1, prev_P[0]) : string(1, prev_P[k]) + prev_P.substr(0, k);
            res.push_back({prev_P, {N - j, added}});
        }
        return res;
    };

    while (!pq1.empty() && !pq2.empty()) {
        if (pq1.top().cost < pq2.top().cost) {
            State cur = pq1.top();
            pq1.pop();
            if (cur.cost > dist1[cur.s].cost) continue;

            if (dist2.count(cur.s)) {
                int total = cur.cost + dist2[cur.s].cost;
                if (total < best_cost) {
                    best_cost = total;
                    best_mid = cur.s;
                }
            }

            auto next_states = get_next(cur.s);
            for (auto& edge : next_states) {
                string nxt = edge.first;
                int weight = edge.second.first;
                if (!dist1.count(nxt) || dist1[nxt].cost > cur.cost + weight) {
                    dist1[nxt] = {cur.cost + weight, cur.s, edge.second.second};
                    pq1.push({nxt, cur.cost + weight});
                }
            }
        } else {
            State cur = pq2.top();
            pq2.pop();
            if (cur.cost > dist2[cur.s].cost) continue;

            if (dist1.count(cur.s)) {
                int total = cur.cost + dist1[cur.s].cost;
                if (total < best_cost) {
                    best_cost = total;
                    best_mid = cur.s;
                }
            }

            auto prev_states = get_prev(cur.s);
            for (auto& edge : prev_states) {
                string prv = edge.first;
                int weight = edge.second.first;
                if (!dist2.count(prv) || dist2[prv].cost > cur.cost + weight) {
                    dist2[prv] = {cur.cost + weight, cur.s, edge.second.second};
                    pq2.push({prv, cur.cost + weight});
                }
            }
        }

        if (!pq1.empty() && !pq2.empty()) {
            if (pq1.top().cost + pq2.top().cost >= best_cost) {
                break;
            }
        }
    }

    string result = S1;
    vector<string> adds_fwd;
    string curr = best_mid;
    while (curr != S1) {
        adds_fwd.push_back(dist1[curr].added);
        curr = dist1[curr].parent;
    }
    reverse(adds_fwd.begin(), adds_fwd.end());
    for (const string& a : adds_fwd) {
        result += a;
    }

    curr = best_mid;
    while (curr != S2) {
        result += dist2[curr].added;
        curr = dist2[curr].parent;
    }

    cout << result << "\n";
    return 0;
}
