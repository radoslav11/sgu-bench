```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <tuple>
#include <algorithm>
#include <functional>

using namespace std;

const long long INF = 1e18;

struct Edge {
    int to;
    long long t;
    long long p;
    int idx;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    vector<vector<Edge>> adj(n + 1);
    for (int i = 1; i <= n - 1; ++i) {
        int u, v;
        long long t, p;
        cin >> u >> v >> t >> p;
        adj[u].push_back({v, t, p, i});
        adj[v].push_back({u, t, p, i});
    }

    auto bfs = [&](int start) {
        vector<long long> dist(n + 1, -1);
        vector<int> parent(n + 1, 0);
        queue<int> q;
        q.push(start);
        dist[start] = 0;
        vector<int> order;
        order.push_back(start);

        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (auto& edge : adj[u]) {
                int v = edge.to;
                if (dist[v] == -1) {
                    dist[v] = dist[u] + edge.t;
                    parent[v] = u;
                    q.push(v);
                    order.push_back(v);
                }
            }
        }
        return make_tuple(dist, parent, order);
    };

    auto [d1, p1, ord1] = bfs(1);
    int U = 1;
    long long max_d1 = -1;
    for (int i = 1; i <= n; ++i) {
        if (d1[i] > max_d1) {
            max_d1 = d1[i];
            U = i;
        }
    }

    auto [dU, pU, ordU] = bfs(U);
    long long D = -1;
    int V = U;
    for (int i = 1; i <= n; ++i) {
        if (dU[i] > D) {
            D = dU[i];
            V = i;
        }
    }

    vector<int> R_nodes;
    for (int i = 1; i <= n; ++i) {
        if (dU[i] == D) {
            R_nodes.push_back(i);
        }
    }

    auto [dV, pV, ordV] = bfs(V);
    vector<int> L_nodes;
    for (int i = 1; i <= n; ++i) {
        if (dV[i] == D) {
            L_nodes.push_back(i);
        }
    }

    vector<int> cntL_U(n + 1, 0), cntR_U(n + 1, 0);
    for (int x : L_nodes) cntL_U[x] = 1;
    for (int x : R_nodes) cntR_U[x] = 1;

    for (int i = (int)ordU.size() - 1; i >= 0; --i) {
        int x = ordU[i];
        if (pU[x] != 0) {
            cntL_U[pU[x]] += cntL_U[x];
            cntR_U[pU[x]] += cntR_U[x];
        }
    }

    int A = U;
    long long max_dA = -1;
    for (int x = 1; x <= n; ++x) {
        if (cntR_U[x] == (int)R_nodes.size() && dU[x] > max_dA) {
            max_dA = dU[x];
            A = x;
        }
    }

    vector<int> cntL_V(n + 1, 0);
    for (int x : L_nodes) cntL_V[x] = 1;
    for (int i = (int)ordV.size() - 1; i >= 0; --i) {
        int x = ordV[i];
        if (pV[x] != 0) {
            cntL_V[pV[x]] += cntL_V[x];
        }
    }

    int B = V;
    long long max_dB = -1;
    for (int x = 1; x <= n; ++x) {
        if (cntL_V[x] == (int)L_nodes.size() && dV[x] > max_dB) {
            max_dB = dV[x];
            B = x;
        }
    }

    int c_L = B;
    int c_R = A;

    vector<bool> is_L(n + 1, false);
    for (int x : L_nodes) is_L[x] = true;

    vector<bool> is_R(n + 1, false);
    for (int x : R_nodes) is_R[x] = true;

    long long best_ans = INF;
    vector<int> best_edges;

    auto solve = [&](int root, const vector<bool>& is_target) {
        vector<long long> dp_cost(n + 1, 0);
        vector<bool> pick_edge(n + 1, false);
        vector<bool> has_target(n + 1, false);

        function<pair<long long, bool>(int, int, long long)> dfs = [&](int u, int p, long long cost_to_parent) {
            has_target[u] = is_target[u];
            long long sum_cost = 0;
            for (auto& edge : adj[u]) {
                int v = edge.to;
                if (v == p) continue;
                auto res = dfs(v, u, edge.p);
                if (res.second) {
                    has_target[u] = true;
                    sum_cost += res.first;
                }
            }
            if (!has_target[u]) {
                return make_pair(0LL, false);
            }
            if (is_target[u] && cost_to_parent != INF) {
                sum_cost = INF;
            }
            long long best = min(cost_to_parent, sum_cost);
            pick_edge[u] = (cost_to_parent <= sum_cost);
            return make_pair(best, true);
        };

        auto res = dfs(root, 0, INF);
        if (res.first < best_ans) {
            best_ans = res.first;
            best_edges.clear();
            function<void(int, int, int, bool)> collect = [&](int u, int p, int edge_idx, bool should_pick) {
                if (should_pick) {
                    if (edge_idx != -1) best_edges.push_back(edge_idx);
                    return;
                }
                if (!has_target[u]) return;
                for (auto& edge : adj[u]) {
                    int v = edge.to;
                    if (v == p) continue;
                    collect(v, u, edge.idx, pick_edge[v]);
                }
            };
            collect(root, 0, -1, pick_edge[root]);
        }
    };

    solve(c_R, is_L);
    solve(c_L, is_R);

    cout << best_ans << "\n";
    cout << best_edges.size() << "\n";
    for (int i = 0; i < (int)best_edges.size(); ++i) {
        cout << best_edges[i] << (i + 1 == (int)best_edges.size() ? "" : " ");
    }
    cout << "\n";

    return 0;
}
```
