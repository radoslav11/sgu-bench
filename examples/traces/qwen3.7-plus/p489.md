#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    long long m;
    if (!(cin >> n >> m)) return 0;

    if (n == 1) {
        cout << 1 % m << "\n";
        return 0;
    }

    vector<long long> dp(n + 1, 0);
    vector<long long> next_dp(n + 1, 0);
    dp[1] = 1;

    for (int i = 2; i <= n; ++i) {
        if (i % 2 == 0) {
            long long sum = 0;
            for (int j = 1; j <= i; ++j) {
                next_dp[j] = sum;
                sum += dp[j];
                sum = (sum >= m ? sum - m : sum);
            }
        } else {
            long long sum = 0;
            for (int j = i; j >= 1; --j) {
                next_dp[j] = sum;
                sum += dp[j - 1];
                sum = (sum >= m ? sum - m : sum);
            }
        }
        for (int j = 1; j <= i; ++j) {
            dp[j] = next_dp[j];
        }
    }

    long long total = 0;
    for (int j = 1; j <= n; ++j) {
        total += dp[j];
        total = (total >= m ? total - m : total);
    }

    long long ans = (total * 2) % m;
    cout << ans << "\n";
    
    return 0;
}
