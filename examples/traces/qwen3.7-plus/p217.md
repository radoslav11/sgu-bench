#include <iostream>
#include <cmath>
#include <algorithm>
#include <iomanip>

using namespace std;

int main() {
    // Optimize standard I/O operations for performance
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    double r1, r2;
    if (!(cin >> r1 >> r2)) return 0;
    
    // Let a be the smaller radius and b be the larger radius
    double a = min(r1, r2);
    double b = max(r1, r2);
    
    // Number of intervals for Simpson's rule
    // 2,000,000 intervals provide extreme precision (error ~ 1e-13) 
    // and execute well within the 0.5s time limit.
    long long n = 2000000;
    const double PI = acos(-1.0);
    double h = (PI / 2.0) / n;
    
    // Integrand at theta = 0 is a^2 * b, at theta = pi/2 is 0
    double sum = a * a * b;
    double sum_odd = 0.0;
    double sum_even = 0.0;
    
    double a2 = a * a;
    double b2 = b * b;
    
    for (long long i = 1; i < n; ++i) {
        double theta = i * h;
        double s = sin(theta);
        double c = cos(theta);
        
        // Using max(0.0, ...) prevents potential NaN from sqrt of tiny negative numbers 
        // due to floating-point precision issues when a == b and theta is close to pi/2.
        double val = a2 * c * c * sqrt(max(0.0, b2 - a2 * s * s));
        
        if (i & 1) {
            sum_odd += val;
        } else {
            sum_even += val;
        }
    }
    
    // Simpson's 1/3 rule formula
    double ans = 8.0 * (h / 3.0) * (sum + 4.0 * sum_odd + 2.0 * sum_even);
    
    // Print with sufficient precision to be safely within the 10^-4 accuracy requirement
    cout << fixed << setprecision(6) << ans << "\n";
    
    return 0;
}
