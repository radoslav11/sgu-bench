#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <queue>
#include <map>

using namespace std;

struct Point {
    long long x, y;
    bool operator==(const Point& o) const { return x == o.x && y == o.y; }
    bool operator<(const Point& o) const {
        if (x != o.x) return x < o.x;
        return y < o.y;
    }
};

struct Edge {
    int u, v;
    long long w;
    int id;
};

struct OutEdge {
    int v;
    int edge_idx; // 0 to 2N-1
    long long dx, dy;
};

int half_plane(long long dx, long long dy) {
    return dy > 0 || (dy == 0 && dx > 0) ? 0 : 1;
}

bool compare_angles(const OutEdge& a, const OutEdge& b) {
    int ha = half_plane(a.dx, a.dy);
    int hb = half_plane(b.dx, b.dy);
    if (ha != hb) return ha < hb;
    return a.dx * b.dy - a.dy * b.dx > 0;
}

struct Dinic {
    struct FlowEdge {
        int to;
        long long cap;
        long long flow;
        int rev;
    };
    vector<vector<FlowEdge>> adj;
    vector<int> level;
    vector<int> ptr;

    Dinic(int n) {
        adj.assign(n, vector<FlowEdge>());
        level.assign(n, 0);
        ptr.assign(n, 0);
    }

    void add_edge(int from, int to, long long cap) {
        adj[from].push_back({to, cap, 0, (int)adj[to].size()});
        adj[to].push_back({from, cap, 0, (int)adj[from].size() - 1});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        level[s] = 0;
        queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& edge : adj[v]) {
                if (edge.cap - edge.flow > 0 && level[edge.to] == -1) {
                    level[edge.to] = level[v] + 1;
                    q.push(edge.to);
                }
            }
        }
        return level[t] != -1;
    }

    long long dfs(int v, int t, long long pushed) {
        if (pushed == 0) return 0;
        if (v == t) return pushed;
        for (int& cid = ptr[v]; cid < (int)adj[v].size(); ++cid) {
            auto& edge = adj[v][cid];
            int tr = edge.to;
            if (level[v] + 1 != level[tr] || edge.cap - edge.flow == 0) continue;
            long long push = dfs(tr, t, min(pushed, edge.cap - edge.flow));
            if (push == 0) continue;
            edge.flow += push;
            adj[tr][edge.rev].flow -= push;
            return push;
        }
        return 0;
    }

    long long max_flow(int s, int t) {
        long long flow = 0;
        while (bfs(s, t)) {
            fill(ptr.begin(), ptr.end(), 0);
            while (long long pushed = dfs(s, t, 1e18)) {
                flow += pushed;
            }
        }
        return flow;
    }
};

bool point_in_polygon(Point P, const vector<Point>& poly) {
    bool inside = false;
    int n = poly.size();
    for (int i = 0, j = n - 1; i < n; j = i++) {
        Point p1 = poly[i];
        Point p2 = poly[j];
        if (((p1.y > P.y) != (p2.y > P.y)) &&
            (P.x < (double)(p2.x - p1.x) * (P.y - p1.y) / (p2.y - p1.y) + p1.x)) {
            inside = !inside;
        }
    }
    return inside;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    if (!(cin >> N)) return 0;

    vector<Edge> edges(N);
    map<Point, int> pt_to_id;
    vector<Point> pts;

    auto get_id = [&](Point p) {
        if (pt_to_id.find(p) == pt_to_id.end()) {
            pt_to_id[p] = pts.size();
            pts.push_back(p);
        }
        return pt_to_id[p];
    };

    for (int i = 0; i < N; ++i) {
        long long x1, y1, x2, y2, w;
        cin >> x1 >> y1 >> x2 >> y2 >> w;
        Point p1 = {x1, y1};
        Point p2 = {x2, y2};
        int u = get_id(p1);
        int v = get_id(p2);
        edges[i] = {u, v, w, i};
    }

    long long Ax, Ay, Bx, By;
    cin >> Ax >> Ay >> Bx >> By;
    Point A = {Ax, Ay};
    Point B = {Bx, By};

    int V = pts.size();
    vector<vector<OutEdge>> adj(V);

    for (int i = 0; i < N; ++i) {
        int u = edges[i].u;
        int v = edges[i].v;
        long long dx = pts[v].x - pts[u].x;
        long long dy = pts[v].y - pts[u].y;
        adj[u].push_back({v, 2 * i, dx, dy});
        adj[v].push_back({u, 2 * i + 1, -dx, -dy});
    }

    for (int i = 0; i < V; ++i) {
        sort(adj[i].begin(), adj[i].end(), compare_angles);
    }

    vector<int> edge_to_face(2 * N, -1);
    vector<vector<int>> faces_edges;

    for (int i = 0; i < 2 * N; ++i) {
        if (edge_to_face[i] != -1) continue;
        int face_id = faces_edges.size();
        vector<int> current_face;
        int curr = i;
        while (true) {
            edge_to_face[curr] = face_id;
            current_face.push_back(curr);
            int u_idx = curr / 2;
            int curr_v = (curr % 2 == 0) ? edges[u_idx].v : edges[u_idx].u;
            int curr_u = (curr % 2 == 0) ? edges[u_idx].u : edges[u_idx].v;

            const auto& list = adj[curr_v];
            int next_idx = -1;
            for (int j = 0; j < (int)list.size(); ++j) {
                if (list[j].v == curr_u) {
                    next_idx = j;
                    break;
                }
            }
            
            int next_j = (next_idx - 1 + (int)list.size()) % list.size();
            curr = list[next_j].edge_idx;

            if (curr == i) break;
        }
        faces_edges.push_back(current_face);
    }

    int face_A = -1, face_B = -1;
    int F = faces_edges.size();

    for (int i = 0; i < F; ++i) {
        vector<Point> poly;
        for (int e_idx : faces_edges[i]) {
            int u_idx = e_idx / 2;
            int u = (e_idx % 2 == 0) ? edges[u_idx].u : edges[u_idx].v;
            poly.push_back(pts[u]);
        }
        if (point_in_polygon(A, poly)) face_A = i;
        if (point_in_polygon(B, poly)) face_B = i;
    }

    Dinic dinic(F);
    struct DualEdge {
        int u, v;
        long long w;
        int orig_id;
    };
    vector<DualEdge> dual_edges;

    for (int i = 0; i < N; ++i) {
        int f1 = edge_to_face[2 * i];
        int f2 = edge_to_face[2 * i + 1];
        if (f1 != f2) {
            dinic.add_edge(f1, f2, edges[i].w);
            dinic.add_edge(f2, f1, edges[i].w);
            dual_edges.push_back({f1, f2, edges[i].w, edges[i].id});
        }
    }

    if (face_A == -1 || face_B == -1) {
        return 0;
    }

    dinic.max_flow(face_A, face_B);

    vector<bool> vis(F, false);
    queue<int> q;
    q.push(face_A);
    vis[face_A] = true;
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (auto& edge : dinic.adj[u]) {
            if (edge.cap - edge.flow > 0 && !vis[edge.to]) {
                vis[edge.to] = true;
                q.push(edge.to);
            }
        }
    }

    vector<int> C;
    long long total_weight = 0;
    for (const auto& de : dual_edges) {
        if (vis[de.u] != vis[de.v]) {
            C.push_back(de.orig_id);
            total_weight += de.w;
        }
    }

    int num_vertices = 0;
    for (int i = 0; i < V; ++i) num_vertices++;

    vector<vector<pair<int, int>>> adj_cycle(num_vertices);
    for (int k : C) {
        int u = edges[k].u;
        int v = edges[k].v;
        adj_cycle[u].push_back({v, k});
        adj_cycle[v].push_back({u, k});
    }

    vector<int> result;
    if (!C.empty()) {
        int start_k = C[0];
        int start_u = edges[start_k].u;
        int start_v = edges[start_k].v;

        int curr_u = start_u;
        int curr_v = start_v;
        int prev_edge = start_k;

        result.push_back(start_k + 1);

        while (true) {
            int next_u = -1, next_edge = -1;
            for (auto& edge : adj_cycle[curr_v]) {
                if (edge.second != prev_edge) {
                    next_u = edge.first;
                    next_edge = edge.second;
                    break;
                }
            }
            if (next_u == start_u) {
                break;
            }
            result.push_back(next_edge + 1);
            prev_edge = next_edge;
            curr_u = curr_v;
            curr_v = next_u;
        }
    }

    cout << total_weight << "\n";
    cout << result.size() << "\n";
    for (int i = 0; i < (int)result.size(); ++i) {
        cout << result[i] << (i + 1 == (int)result.size() ? "" : " ");
    }
    cout << "\n";

    return 0;
}
