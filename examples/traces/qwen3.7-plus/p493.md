#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Building {
    long long L, R, H;
};

const int INF = 1e9;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int T;
    if (!(cin >> T)) return 0;

    while (T--) {
        int N;
        cin >> N;
        vector<Building> b(N + 1);
        for (int i = 1; i <= N; i++) {
            cin >> b[i].L >> b[i].R >> b[i].H;
        }

        sort(b.begin() + 1, b.end(), [](const Building& a, const Building& b) {
            return a.L < b.L;
        });

        vector<int> max_L(N + 1, 0);
        for (int i = 1; i <= N; i++) {
            max_L[i] = i;
            for (int k = i + 1; k <= N; k++) {
                bool ok = true;
                for (int p = i + 1; p < k; p++) {
                    long long den = b[k].L - b[i].R;
                    long long num1 = b[i].H * (b[k].L - b[p].L);
                    if (num1 < b[p].H * den) { ok = false; break; }
                    long long num2 = b[i].H * (b[k].L - b[p].R);
                    if (num2 < b[p].H * den) { ok = false; break; }
                }
                if (ok) max_L[i] = k;
                else break;
            }
        }

        vector<int> min_R(N + 1, 0);
        for (int i = 1; i <= N; i++) {
            min_R[i] = i;
            for (int k = i - 1; k >= 1; k--) {
                bool ok = true;
                for (int p = k + 1; p < i; p++) {
                    long long den = b[i].L - b[k].R;
                    long long num1 = b[i].H * (b[p].L - b[k].R);
                    if (num1 < b[p].H * den) { ok = false; break; }
                    long long num2 = b[i].H * (b[p].R - b[k].R);
                    if (num2 < b[p].H * den) { ok = false; break; }
                }
                if (ok) min_R[i] = k;
                else break;
            }
        }

        vector<vector<int>> dp(N + 2, vector<int>(N + 2, INF));
        dp[1][1] = 0;

        for (int i = 1; i <= N; i++) {
            vector<vector<int>> new_dp(N + 2, vector<int>(N + 2, INF));
            for (int j = 1; j <= i; j++) {
                for (int m = 1; m <= N; m++) {
                    if (dp[j][m] == INF) continue;

                    int val = dp[j][m];

                    if (min_R[i] <= j) {
                        if (val + 1 < new_dp[i + 1][m]) {
                            new_dp[i + 1][m] = val + 1;
                        }
                    }

                    if (m >= i) {
                        int nm = max(m, max_L[i]);
                        int nj = (j == i) ? i + 1 : j;
                        if (val + 1 < new_dp[nj][nm]) {
                            new_dp[nj][nm] = val + 1;
                        }

                        if (min_R[i] <= j) {
                            if (val + 2 < new_dp[i + 1][nm]) {
                                new_dp[i + 1][nm] = val + 2;
                            }
                        }

                        int nj2 = min(j, i);
                        if (val < new_dp[nj2][m]) {
                            new_dp[nj2][m] = val;
                        }
                    }
                }
            }
            dp = move(new_dp);
        }

        int ans = INF;
        for (int m = 1; m <= N; m++) {
            ans = min(ans, dp[N + 1][m]);
        }
        cout << ans << "\n";
    }
    return 0;
}
