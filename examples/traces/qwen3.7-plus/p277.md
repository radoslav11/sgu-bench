#include <iostream>
#include <vector>
#include <algorithm>
#include <set>

using namespace std;

struct Point {
    long long x, y;
    int id;
    bool operator<(const Point& o) const {
        if (x != o.x) return x < o.x;
        if (y != o.y) return y < o.y;
        return id < o.id;
    }
    bool operator==(const Point& o) const {
        return x == o.x && y == o.y;
    }
};

Point operator-(const Point& a, const Point& b) {
    return {a.x - b.x, a.y - b.y, 0};
}

long long cross(const Point& a, const Point& b) {
    return a.x * b.y - a.y * b.x;
}

long long cross(const Point& a, const Point& b, const Point& c) {
    return cross(b - a, c - a);
}

// Check if point c is strictly to the right of the directed line a -> b
bool is_right(const Point& a, const Point& b, const Point& c) {
    return cross(b - a, c - a) < 0;
}

// Check if point c is strictly to the left of the directed line a -> b
bool is_left(const Point& a, const Point& b, const Point& c) {
    return cross(b - a, c - a) > 0;
}

// Check if point p is inside or on the boundary of the convex hull
bool is_inside(const Point& p, const vector<Point>& hull) {
    int n = hull.size();
    if (n < 3) return false;
    
    long long c0 = cross(hull[0], hull[1], p);
    if (c0 < 0) return false;
    if (c0 == 0) {
        if (hull[0].x <= p.x && p.x <= hull[1].x && hull[0].y <= p.y && p.y <= hull[1].y) return true;
        return false;
    }
    
    for (int i = 1; i < n; i++) {
        int j = (i + 1 == n) ? 0 : i + 1;
        long long c = cross(hull[i], hull[j], p);
        if (c < 0) return false;
        if (c == 0) {
            if (min(hull[i].x, hull[j].x) <= p.x && p.x <= max(hull[i].x, hull[j].x) &&
                min(hull[i].y, hull[j].y) <= p.y && p.y <= max(hull[i].y, hull[j].y)) {
                return true;
            }
            return false;
        }
    }
    return true;
}

        vector<Point> upper, lower;
        for (const auto& p : hull) {
            while (upper.size() >= 2 && !is_right(upper[upper.size() - 2], upper.back(), p)) {
                upper.pop_back();
            }
            upper.push_back(p);
            
            while (lower.size() >= 2 && !is_left(lower[lower.size() - 2], lower.back(), p)) {
                lower.pop_back();
            }
            lower.push_back(p);
        }
        
        hull.clear();
        for (size_t i = 0; i < upper.size(); i++) hull.push_back(upper[i]);
        for (int i = (int)lower.size() - 2; i > 0; i--) hull.push_back(lower[i]);
    }
    
    auto calc_area = [&]() {
        long long area2 = 0;
        int sz = hull.size();
        for (int i = 0; i < sz; i++) {
            int j = (i + 1 == sz) ? 0 : i + 1;
            area2 += cross(hull[i], hull[j]);
        }
        return abs(area2);
    };
    
    long long current_area = calc_area();
    
    for (int k = 0; k < n; k++) {
        Point p;
        cin >> p.x >> p.y;
        p.id = k + 3;
        
        if (!is_inside(p, hull)) {

                int j = (i + 1 == sz) ? 0 : i + 1;
                if (is_left(hull[i], hull[j], p) || (cross(hull[i], hull[j], p) == 0 && 
                    min(hull[i].x, hull[j].x) <= p.x && p.x <= max(hull[i].x, hull[j].x) &&
                    min(hull[i].y, hull[j].y) <= p.y && p.y <= max(hull[i].y, hull[j].y))) {
                    // keep edge
                } else {
                    new_hull.push_back(hull[i]);
                }
            }
            
            sort(new_hull.begin(), new_hull.end());
            new_hull.push_back(p);
            sort(new_hull.begin(), new_hull.end());
            new_hull.erase(unique(new_hull.begin(), new_hull.end()), new_hull.end());
            
            vector<Point> upper, lower;

                    upper.pop_back();
                }
                upper.push_back(pt);

                    lower.pop_back();

        }

    }
