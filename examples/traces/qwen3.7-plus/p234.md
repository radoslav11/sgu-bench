```cpp
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

const int MAX_M = 205;
const int MAX_N = 205;
const int MAXV = 40005;
const int MAXE = 160005;

int grid[MAX_M][MAX_N];
int u_id[MAX_M][MAX_N];
int v_id[MAX_M][MAX_N];
int u_count = 0, v_count = 0;

int head[MAXV];
int to_edge[MAXE];
int nxt[MAXE];
int edge_cnt = 0;

void add_edge(int u, int v) {
    to_edge[edge_cnt] = v;
    nxt[edge_cnt] = head[u];
    head[u] = edge_cnt++;
}

int match_u[MAXV];
int match_v[MAXV];
int dist[MAXV];
int ptr[MAXV];
const int INF = 1e9;

bool bfs() {
    queue<int> q;
    for (int u = 0; u < u_count; ++u) {
        if (match_u[u] == -1) {
            dist[u] = 0;
            q.push(u);
        } else {
            dist[u] = INF;
        }
    }
    bool found = false;
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int e = head[u]; e != -1; e = nxt[e]) {
            int v = to_edge[e];
            if (match_v[v] == -1) {
                found = true;
            } else if (dist[match_v[v]] == INF) {
                dist[match_v[v]] = dist[u] + 1;
                q.push(match_v[v]);
            }
        }
    }
    return found;
}

bool dfs(int u) {
    for (int &e = ptr[u]; e != -1; e = nxt[e]) {
        int v = to_edge[e];
        if (match_v[v] == -1 || (dist[match_v[v]] == dist[u] + 1 && dfs(match_v[v]))) {
            match_u[u] = v;
            match_v[v] = u;
            return true;
        }
    }
    return false;
}

int hopcroft_karp() {
    for (int i = 0; i < u_count; ++i) match_u[i] = -1;
    for (int i = 0; i < v_count; ++i) match_v[i] = -1;
    int matching = 0;
    while (bfs()) {
        for (int u = 0; u < u_count; ++u) {
            ptr[u] = head[u];
            if (match_u[u] == -1) {
                if (dfs(u)) {
                    matching++;
                }
            }
        }
    }
    return matching;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int M, N;
    if (!(cin >> M >> N)) return 0;

    for (int i = 0; i < MAXV; ++i) {
        head[i] = -1;
    }

    for (int r = 0; r < M; ++r) {
        for (int c = 0; c < N; ++c) {
            cin >> grid[r][c];
            if (grid[r][c] == 1) {
                if ((r + c) % 2 == 0) {
                    u_id[r][c] = u_count++;
                } else {
                    v_id[r][c] = v_count++;
                }
            }
        }
    }

    int dr[] = {-1, 1, 0, 0};
    int dc[] = {0, 0, -1, 1};

    for (int r = 0; r < M; ++r) {
        for (int c = 0; c < N; ++c) {
            if (grid[r][c] == 1 && (r + c) % 2 == 0) {
                int u = u_id[r][c];
                for (int i = 0; i < 4; ++i) {
                    int nr = r + dr[i];
                    int nc = c + dc[i];
                    if (nr >= 0 && nr < M && nc >= 0 && nc < N && grid[nr][nc] == 1) {
                        int v = v_id[nr][nc];
                        add_edge(u, v);
                    }
                }
            }
        }
    }

    hopcroft_karp();

    vector<bool> in_su(u_count, false);
    vector<bool> in_sv(v_count, false);
    queue<int> q;

    for (int u = 0; u < u_count; ++u) {
        if (match_u[u] == -1) {
            in_su[u] = true;
            q.push(u);
        }
    }

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        
        for (int e = head[u]; e != -1; e = nxt[e]) {
            int v = to_edge[e];
            if (match_u[u] != v) { // unmatched edge
                if (!in_sv[v]) {
                    in_sv[v] = true;
                    int next_u = match_v[v];
                    if (next_u != -1 && !in_su[next_u]) {
                        in_su[next_u] = true;
                        q.push(next_u);
                    }
                }
            }
        }
    }

    int mis_count = 0;
    for (int u = 0; u < u_count; ++u) {
        if (in_su[u]) mis_count++;
    }
    for (int v = 0; v < v_count; ++v) {
        if (!in_sv[v]) mis_count++;
    }

    cout << mis_count << "\n";

    for (int r = 0; r < M; ++r) {
        for (int c = 0; c < N; ++c) {
            if (grid[r][c] == 0) {
                cout << '#';
            } else {
                if ((r + c) % 2 == 0) {
                    if (in_su[u_id[r][c]]) {
                        cout << 'G';
                    } else {
                        cout << '.';
                    }
                } else {
                    if (!in_sv[v_id[r][c]]) {
                        cout << 'G';
                    } else {
                        cout << '.';
                    }
                }
            }
        }
        cout << "\n";
    }

    return 0;
}
```
