```cpp
#include <iostream>
#include <vector>

using namespace std;

int N, B;
vector<vector<int>> adj;
vector<int> capitals;
vector<int> province_of;
vector<vector<int>> provinces;

int dfs(int u, int p) {
    int sz = 1;
    for (int v : adj[u]) {
        if (v == p) continue;
        sz += dfs(v, u);
    }
    if (sz >= B) {
        capitals.push_back(u);
        sz = 0;
    }
    return sz;
}

void assign(int u, int p, int prov_id) {
    province_of[u] = prov_id;
    provinces[prov_id].push_back(u);
    for (int v : adj[u]) {
        if (v == p) continue;
        if (province_of[v] == 0) {
            assign(v, u, prov_id);
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    if (!(cin >> N >> B)) return 0;
    
    adj.resize(N + 1);
    for (int i = 0; i < N - 1; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    
    if (N < B) {
        cout << 0 << "\n";
        return 0;
    }
    
    province_of.assign(N + 1, 0);
    dfs(1, 0);
    
    provinces.resize(capitals.size() + 1);
    for (size_t i = 0; i < capitals.size(); i++) {
        assign(capitals[i], 0, i + 1);
    }
    
    vector<int> remainder;
    for (int i = 1; i <= N; i++) {
        if (province_of[i] == 0) {
            remainder.push_back(i);
        }
    }
    
    if (!remainder.empty()) {
        int last_prov = capitals.size();
        for (int node : remainder) {
            province_of[node] = last_prov;
            provinces[last_prov].push_back(node);
        }
    }
    
    bool possible = true;
    for (size_t i = 1; i < provinces.size(); i++) {
        if (provinces[i].size() < (size_t)B || provinces[i].size() > (size_t)3 * B) {
            possible = false;
            break;
        }
    }
    
    if (!possible) {
        cout << 0 << "\n";
        return 0;
    }
    
    cout << capitals.size() << "\n";
    for (int i = 1; i <= N; i++) {
        cout << province_of[i] << (i == N ? "" : " ");
    }
    cout << "\n";
    for (size_t i = 0; i < capitals.size(); i++) {
        cout << capitals[i] << (i == capitals.size() - 1 ? "" : " ");
    }
    cout << "\n";
    
    return 0;
}
```
