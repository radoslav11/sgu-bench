#include <iostream>
#include <iomanip>
#include <algorithm>

struct BigInt {
    uint32_t digits[8];
    int size;

    BigInt(uint64_t v = 0) {
        size = 0;
        for (int i = 0; i < 8; ++i) digits[i] = 0;
        if (v == 0) {
            digits[0] = 0;
            size = 1;
        } else {
            while (v > 0) {
                digits[size++] = v % 1000000000;
                v /= 1000000000;
            }
        }
    }

    BigInt operator+(const BigInt& o) const {
        BigInt res;
        res.size = 0;
        uint64_t carry = 0;
        int n = std::max(size, o.size);
        for (int i = 0; i < n || carry; ++i) {
            uint64_t sum = carry;
            if (i < size) sum += digits[i];
            if (i < o.size) sum += o.digits[i];
            res.digits[res.size++] = sum % 1000000000;
            carry = sum / 1000000000;
        }
        if (res.size == 0) {
            res.digits[0] = 0;
            res.size = 1;
        }
        return res;
    }

    BigInt operator*(const BigInt& o) const {
        if (size == 1 && digits[0] == 0) return BigInt(0);
        if (o.size == 1 && o.digits[0] == 0) return BigInt(0);
        BigInt res;
        res.size = 0;
        for (int i = 0; i < 8; ++i) res.digits[i] = 0;
        
        for (int i = 0; i < size; ++i) {
            uint64_t carry = 0;
            for (int j = 0; j < o.size; ++j) {
                uint64_t cur = res.digits[i + j] + (uint64_t)digits[i] * o.digits[j] + carry;
                res.digits[i + j] = cur % 1000000000;
                carry = cur / 1000000000;
            }
            int idx = i + o.size;
            while (carry > 0 && idx < 8) {
                uint64_t cur = res.digits[idx] + carry;
                res.digits[idx] = cur % 1000000000;
                carry = cur / 1000000000;
                idx++;
            }
        }
        res.size = 8;
        while (res.size > 1 && res.digits[res.size - 1] == 0) {
            res.size--;
        }
        return res;
    }

    void print() const {
        if (size == 1 && digits[0] == 0) {
            std::cout << 0 << "\n";
            return;
        }
        std::cout << digits[size - 1];
        for (int i = size - 2; i >= 0; --i) {
            std::cout << std::setw(9) << std::setfill('0') << digits[i];
        }
        std::cout << "\n";
    }
    
    bool is_zero() const {
        return size == 1 && digits[0] == 0;
    }
};

BigInt C[50][50];
BigInt fact[50];
BigInt DP_AB[45][45][45];

void precompute() {
    for (int i = 0; i <= 40; ++i) {
        C[i][0] = BigInt(1);
        for (int j = 1; j <= i; ++j) {
            C[i][j] = C[i-1][j-1] + C[i-1][j];
        }
    }
    fact[0] = BigInt(1);
    for (int i = 1; i <= 40; ++i) {
        fact[i] = fact[i-1] * BigInt(i);
    }
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    precompute();

    long long N_in, M_in, W_in, H_in, K_in;
    if (!(std::cin >> N_in >> M_in >> W_in >> H_in >> K_in)) {
        return 0;
    }

    int N = (int)N_in;
    int M = (int)M_in;
    long long W = W_in;
    long long H = H_in;
    long long K = K_in;

    long long b_val = std::max(0LL, std::min((long long)N - 1, H + M - 1) - std::max(0LL, H) + 1);
    int b = (int)b_val;
    int a = N - b;
    int c = M - b;

    long long y_val = std::max(0LL, std::min((long long)N - 1, W + M - 1) - std::max(0LL, W) + 1);
    int y = (int)y_val;
    int x = N - y;
    int z = M - y;

    for (int uA = 0; uA <= x; ++uA) {
        for (int vA = 0; vA <= y; ++vA) {
            if (uA + vA > a) continue;
            BigInt waysA = C[a][uA + vA] * C[x][uA] * C[y][vA] * fact[uA + vA];
            if (waysA.is_zero()) continue;

            for (int uB = 0; uB <= x - uA; ++uB) {
                for (int vB = 0; vB <= y - vA; ++vB) {
                    for (int wB = 0; wB <= z; ++wB) {
                        if (uB + vB + wB > b) continue;
                        BigInt waysB = C[b][uB + vB + wB] * C[x - uA][uB] * C[y - vA][vB] * C[z][wB] * fact[uB + vB + wB];
                        if (waysB.is_zero()) continue;

                        int U = uA + uB;
                        int V = vA + vB;
                        int W_state = wB;
                        
                        DP_AB[U][V][W_state] = DP_AB[U][V][W_state] + (waysA * waysB);
                    }
                }
            }
        }
    }

    BigInt total_ways = BigInt(0);
    for (int u = 0; u <= x; ++u) {
        for (int v = 0; v <= y; ++v) {
            for (int w = 0; w <= z; ++w) {
                if (DP_AB[u][v][w].is_zero()) continue;
                
                for (int vC = 0; vC <= y - v; ++vC) {
                    for (int wC = 0; wC <= z - w; ++wC) {
                        if (vC + wC > c) continue;
                        if (u + v + w + vC + wC != K) continue;
                        
                        BigInt waysC = C[c][vC + wC] * C[y - v][vC] * C[z - w][wC] * fact[vC + wC];
                        total_ways = total_ways + (DP_AB[u][v][w] * waysC);
                    }
                }
            }
        }
    }

    total_ways.print();

    return 0;
}
