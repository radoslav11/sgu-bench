#include <iostream>
#include <string>
#include <map>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    if (!(cin >> n)) return 0;
    
    int val[4] = {0, 0, 0, 0};
    map<char, int> ptr;
    ptr['a'] = 0;
    ptr['b'] = 1;
    ptr['c'] = 2;
    ptr['d'] = 3;

    for (int i = 0; i < n; ++i) {
        string s;
        cin >> s;
        if (s.substr(0, 8) == "writeln(") {
            char p = s[8];
            cout << val[ptr[p]] << "\n";
        } else {
            size_t pos = s.find(":=");
            if (pos == string::npos) continue;
            string left = s.substr(0, pos);
            string right = s.substr(pos + 2);
            if (!right.empty() && right.back() == ';') {
                right.pop_back();
            }

            if (left.length() == 2 && left[1] == '^') {
                char p1 = left[0];
                if (right.length() == 2 && right[1] == '^') {
                    char p2 = right[0];
                    val[ptr[p1]] = val[ptr[p2]];
                } else {
                    int c = stoi(right);
                    val[ptr[p1]] = c;
                }
            } else {
                char p1 = left[0];
                char p2 = right[0];
                ptr[p1] = ptr[p2];
            }
        }
    }
    return 0;
}
