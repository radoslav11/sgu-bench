#include <iostream>

using namespace std;

int main() {
    // Optimize standard I/O operations for competitive programming
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long long N, H;
    if (!(cin >> N >> H)) return 0;

    long long c = H + 1;

    // Function to calculate the pumpkin number that ends up at coordinate (x, H)
    auto get_index = [&](long long x) {
        if (x == 0) {
            return c * c;
        } else if (x < 0) {
            long long k = -x;
            return (k + c - 1) * (k + c - 1) + c;
        } else {
            long long k = x;
            return (k + c) * (k + c) - k;
        }
    };

    // Binary search for the leftmost valid x (most negative)
    long long low = 0, high = 2000000000LL;
    long long best_left = 0;
    while (low <= high) {
        long long mid = low + (high - low) / 2;
        if (get_index(-mid) <= N) {
            best_left = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    long long L = -best_left;

    // Binary search for the rightmost valid x (most positive)
    low = 0; 
    high = 2000000000LL;
    long long best_right = 0;
    while (low <= high) {
        long long mid = low + (high - low) / 2;
        if (get_index(mid) <= N) {
            best_right = mid;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    long long R = best_right;

    // Output the pumpkin numbers from leftmost to rightmost
    for (long long x = L; x <= R; ++x) {
        cout << get_index(x) << (x == R ? "" : " ");
    }
    cout << "\n";

    return 0;
}
