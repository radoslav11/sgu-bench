#include <iostream>
#include <iomanip>
#include <cmath>

using namespace std;

double normalize(double v) {
    if (v > -1e-9 && v < 1e-9) return 0.0;
    return v;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N, n1, n2;
    if (!(cin >> N >> n1 >> n2)) return 0;
    
    double x1, y1, x2, y2;
    cin >> x1 >> y1 >> x2 >> y2;

    int k = (n2 - n1 + N) % N;
    double PI = acos(-1.0);
    double theta = k * 2.0 * PI / N;

    double vx = x2 - x1;
    double vy = y2 - y1;

    double cos_theta = cos(theta);
    double sin_theta = sin(theta);

    double denom = 2.0 - 2.0 * cos_theta;
    
    double ux = ((1.0 - cos_theta) * vx + sin_theta * vy) / denom;
    double uy = (-sin_theta * vx + (1.0 - cos_theta) * vy) / denom;

    double cx = x1 + ux;
    double cy = y1 + uy;

    cout << fixed << setprecision(6);
    for (int i = 1; i <= N; ++i) {
        int k_i = (i - n1 + N) % N;
        double alpha = k_i * 2.0 * PI / N;
        double cos_alpha = cos(alpha);
        double sin_alpha = sin(alpha);

        double dx = x1 - cx;
        double dy = y1 - cy;

        double rx = dx * cos_alpha + dy * sin_alpha;
        double ry = -dx * sin_alpha + dy * cos_alpha;

        cout << normalize(cx + rx) << " " << normalize(cy + ry) << "\n";
    }

    return 0;
}
