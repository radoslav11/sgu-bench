```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Participant {
    int id;
    long long W, R, P;
    long long current_time;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, K;
    if (!(cin >> N >> K)) return 0;

    vector<Participant> parts(N);
    for (int i = 0; i < N; ++i) {
        parts[i].id = i + 1;
        cin >> parts[i].W >> parts[i].R >> parts[i].P;
        parts[i].current_time = 20000000LL - parts[i].W;
    }

    vector<long long> A(K);
    for (int i = 0; i < K; ++i) {
        cin >> A[i];
    }

    vector<long long> B(N);
    for (int i = 0; i < N; ++i) {
        cin >> B[i];
    }

    sort(B.rbegin(), B.rend());

    for (int j = 0; j < K; ++j) {
        sort(parts.begin(), parts.end(), [](const Participant& a, const Participant& b) {
            if (a.current_time != b.current_time) {
                return a.current_time < b.current_time;
            }
            return a.P > b.P;
        });

        for (int k = 0; k < N; ++k) {
            long long b_val = B[k];
            long long a_val = A[j];
            long long S = 3 * a_val * a_val + 5 * a_val * b_val + 2 * b_val * b_val;
            long long time_taken = 20000000LL - parts[k].R - S;
            parts[k].current_time += time_taken;
        }
    }

    sort(parts.begin(), parts.end(), [](const Participant& a, const Participant& b) {
        if (a.current_time != b.current_time) {
            return a.current_time < b.current_time;
        }
        return a.P > b.P;
    });

    for (int i = 0; i < N; ++i) {
        cout << parts[i].id << (i == N - 1 ? "" : " ");
    }
    cout << "\n";

    return 0;
}
```
