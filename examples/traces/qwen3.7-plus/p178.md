#include <iostream>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    unsigned long long n;
    if (cin >> n) {
        unsigned long long k = 0;
        while (true) {
            unsigned long long max_n = (k + 1) * (1ULL << (k + 1)) - 1;
            if (max_n >= n) {
                cout << k << "\n";
                break;
            }
            k++;
        }
    }
    
    return 0;
}
