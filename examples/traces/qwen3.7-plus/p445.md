#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
#include <algorithm>

using namespace std;

struct Point {
    double x, y;
};

double dist(Point a, Point b) {
    return hypot(a.x - b.x, a.y - b.y);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    while (cin >> n) {
        double vw, vc;
        cin >> vw >> vc;

        vector<Point> pts(n);
        for (int i = 0; i < n; i++) {
            cin >> pts[i].x >> pts[i].y;
        }

        double total_time = 0.0;

        for (int i = 0; i < n - 1; i++) {
            Point p1 = pts[i];
            Point p2 = pts[i+1];

            if (p1.y <= 0 && p2.y <= 0) {
                total_time += dist(p1, p2) / vw;
            } else if (p1.y > 0 && p2.y > 0) {
                double tunnel_time = (p2.x - p1.x) / vc;
                double walk_time = dist(p1, p2) / vw;
                total_time += min(tunnel_time, walk_time);
            } else {
                double x_mid = p1.x + (p2.x - p1.x) * (-p1.y) / (p2.y - p1.y);
                Point pmid = {x_mid, 0.0};

                if (p1.y > 0) {
                    double tunnel_time = (pmid.x - p1.x) / vc;
                    double walk_time = dist(p1, pmid) / vw;
                    total_time += min(tunnel_time, walk_time);
                    
                    total_time += dist(pmid, p2) / vw;
                } else {
                    total_time += dist(p1, pmid) / vw;
                    
                    double tunnel_time = (p2.x - pmid.x) / vc;
                    double walk_time = dist(pmid, p2) / vw;
                    total_time += min(tunnel_time, walk_time);
                }
            }
        }

        cout << fixed << setprecision(9) << total_time << "\n";
    }

    return 0;
}
