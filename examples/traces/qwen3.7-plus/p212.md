#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Edge {
    int to;
    long long cap;
    long long flow;
    int id;
};

int N, M, L;
int source_node, sink_node;
vector<int> level_arr;
vector<Edge> edges;
vector<vector<int>> adj;
vector<int> ptr;

long long dfs(int u, long long pushed) {
    if (u == sink_node || pushed == 0) return pushed;
    for (int& cid = ptr[u]; cid < (int)adj[u].size(); ++cid) {
        int id = adj[u][cid];
        int v = edges[id].to;
        long long tr = edges[id].cap - edges[id].flow;
        if (tr == 0) continue;
        long long tr_pushed = dfs(v, min(pushed, tr));
        if (tr_pushed == 0) continue;
        edges[id].flow += tr_pushed;
        return tr_pushed;
    }
    return 0;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> N >> M >> L)) return 0;

    level_arr.resize(N + 1);
    source_node = -1;
    sink_node = -1;
    for (int i = 1; i <= N; ++i) {
        cin >> level_arr[i];
        if (level_arr[i] == 1) source_node = i;
        if (level_arr[i] == L) sink_node = i;
    }

    edges.resize(M);
    adj.resize(N + 1);

    for (int i = 0; i < M; ++i) {
        int u, v;
        long long c;
        cin >> u >> v >> c;
        edges[i] = {v, c, 0, i};
        adj[u].push_back(i);
    }

    ptr.assign(N + 1, 0);
    const long long INF = 2000000000000000000LL;

    while (dfs(source_node, INF) > 0) {
        // Continue finding augmenting paths until no more forward paths exist
    }

    for (int i = 0; i < M; ++i) {
        cout << edges[i].flow << "\n";
    }

    return 0;
}
