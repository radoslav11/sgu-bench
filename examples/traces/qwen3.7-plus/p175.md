#include <iostream>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    long long n, q;
    if (cin >> n >> q) {
        long long ans = 0;
        while (n > 1) {
            long long k = n / 2;
            if (q <= k) {
                ans += (n - k);
                long long next_q = k - q + 1;
                n = k;
                q = next_q;
            } else {
                long long next_q = n - q + 1;
                n = n - k;
                q = next_q;
            }
        }
        ans += 1;
        cout << ans << "\n";
    }
    return 0;
}
