#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const long long INF = 1e18;

struct Edge {
    int u, v;
    long long w;
};

int N, M;
int head_edge[200005], to[400005], nxt[400005];
long long weight[400005];
int edge_cnt = 0;

void add_edge(int u, int v, long long w) {
    to[edge_cnt] = v;
    weight[edge_cnt] = w;
    nxt[edge_cnt] = head_edge[u];
    head_edge[u] = edge_cnt++;
}

vector<Edge> non_tree_edges;

int in[200005], out[200005], timer_dfs;
int depth[200005], sz[200005], heavy[200005], head[200005];
int parent_node[200005][18];
long long d_dist[200005];

void dfs1(int u, int p, long long dist, int d) {
    in[u] = ++timer_dfs;
    parent_node[u][0] = p;
    depth[u] = d;
    d_dist[u] = dist;
    sz[u] = 1;
    heavy[u] = 0;
    for (int i = 1; i < 18; ++i) {
        parent_node[u][i] = parent_node[parent_node[u][i - 1]][i - 1];
    }
    for (int e = head_edge[u]; e != -1; e = nxt[e]) {
        int v = to[e];
        long long w = weight[e];
        if (v != p) {
            dfs1(v, u, dist + w, d + 1);
            sz[u] += sz[v];
            if (sz[v] > sz[heavy[u]]) {
                heavy[u] = v;
            }
        }
    }
    out[u] = timer_dfs;
}

void dfs2(int u, int p, int h) {
    head[u] = h;
    if (heavy[u]) {
        dfs2(heavy[u], u, h);
    }
    for (int e = head_edge[u]; e != -1; e = nxt[e]) {
        int v = to[e];
        if (v != p && v != heavy[u]) {
            dfs2(v, u, v);
        }
    }
}

int get_kth_ancestor(int u, int k) {
    for (int i = 0; i < 18; ++i) {
        if ((k >> i) & 1) {
            u = parent_node[u][i];
        }
    }
    return u;
}

bool is_ancestor(int u, int v) {
    return in[u] <= in[v] && out[v] <= out[u];
}

struct SegTree {
    int n;
    long long* tree;
    SegTree(int n) : n(n) {
        tree = new long long[4 * n + 1];
        for (int i = 0; i <= 4 * n; ++i) tree[i] = INF;
    }
    ~SegTree() {
        delete[] tree;
    }
    
    void update(int node, int start, int end, int l, int r, long long val) {
        if (r < start || end < l) return;
        if (l <= start && end <= r) {
            tree[node] = min(tree[node], val);
            return;
        }
        int mid = start + (end - start) / 2;
        update(2 * node, start, mid, l, r, val);
        update(2 * node + 1, mid + 1, end, l, r, val);
    }
    
    long long query(int node, int start, int end, int idx) {
        if (start == end) return tree[node];
        int mid = start + (end - start) / 2;
        long long res = tree[node];
        if (idx <= mid) res = min(res, query(2 * node, start, mid, idx));
        else res = min(res, query(2 * node + 1, mid + 1, end, idx));
        return res;
    }
};

SegTree* seg;

void update_path(int u, int v, long long val) {
    while (head[u] != head[v]) {
        seg->update(1, 1, N, in[head[u]], in[u], val);
        u = parent_node[head[u]][0];
    }
    seg->update(1, 1, N, in[v], in[u], val);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    if (!(cin >> N >> M)) return 0;
    
    for (int i = 1; i <= N; ++i) {
        head_edge[i] = -1;
    }
    
    for (int i = 0; i < M; ++i) {
        int u, v;
        long long w;
        int t;
        cin >> u >> v >> w >> t;
        if (t == 1) {
            add_edge(u, v, w);
            add_edge(v, u, w);
        } else {
            non_tree_edges.push_back({u, v, w});
        }
    }
    
    parent_node[1][0] = 1;
    dfs1(1, 1, 0, 0);
    dfs2(1, 1, 1);
    
    seg = new SegTree(N);
    
    for (auto& edge : non_tree_edges) {
        int u = edge.u;
        int v = edge.v;
        long long w = edge.w;
        long long W = d_dist[u] + d_dist[v] + w;
        
        if (is_ancestor(u, v)) {
            int diff = depth[v] - depth[u] - 1;
            int c = get_kth_ancestor(v, diff);
            update_path(v, c, W);
        } else if (is_ancestor(v, u)) {
            int diff = depth[u] - depth[v] - 1;
            int c = get_kth_ancestor(u, diff);
            update_path(u, c, W);
        } else {
            seg->update(1, 1, N, in[u], out[u], W);
            seg->update(1, 1, N, in[v], out[v], W);
        }
    }
    
    for (int i = 2; i <= N; ++i) {
        long long min_W = seg->query(1, 1, N, in[i]);
        if (min_W >= INF) {
            cout << -1 << (i == N ? "" : " ");
        } else {
            cout << min_W - d_dist[i] << (i == N ? "" : " ");
        }
    }
    cout << "\n";
    
    delete seg;
    return 0;
}
