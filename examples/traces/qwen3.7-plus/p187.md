#include <iostream>

using namespace std;

struct Node {
    int l, r;
    int sz;
    int val;
    unsigned pr_and_rev;
};

const int MAXN = 130005;
Node tr[MAXN];
int node_cnt = 0;
int root = 0;

inline bool get_rev(int u) {
    return tr[u].pr_and_rev & 1;
}

inline void set_rev(int u, bool v) {
    if (v) tr[u].pr_and_rev |= 1;
    else tr[u].pr_and_rev &= ~1u;
}

inline unsigned get_pr(int u) {
    return tr[u].pr_and_rev >> 1;
}

inline void set_pr(int u, unsigned v) {
    tr[u].pr_and_rev = (v << 1) | (tr[u].pr_and_rev & 1);
}

inline int add_node(int value, unsigned pr) {
    int u = ++node_cnt;
    tr[u].l = tr[u].r = 0;
    tr[u].sz = 1;
    tr[u].val = value;
    tr[u].pr_and_rev = 0;
    set_pr(u, pr);
    set_rev(u, false);
    return u;
}

inline void push_down(int u) {
    if (get_rev(u)) {
        int tmp = tr[u].l;
        tr[u].l = tr[u].r;
        tr[u].r = tmp;
        if (tr[u].l) set_rev(tr[u].l, !get_rev(tr[u].l));
        if (tr[u].r) set_rev(tr[u].r, !get_rev(tr[u].r));
        set_rev(u, false);
    }
}

inline void update(int u) {
    tr[u].sz = 1;
    if (tr[u].l) tr[u].sz += tr[tr[u].l].sz;
    if (tr[u].r) tr[u].sz += tr[tr[u].r].sz;
}

void split(int u, int k, int &l, int &r) {
    if (!u) {
        l = r = 0;
        return;
    }
    push_down(u);
    int left_sz = tr[u].l ? tr[tr[u].l].sz : 0;
    if (left_sz >= k) {
        split(tr[u].l, k, l, r);
        tr[u].l = r;
        r = u;
    } else {
        split(tr[u].r, k - left_sz - 1, l, r);
        tr[u].r = l;
        l = u;
    }
    update(u);
}

int merge(int l, int r) {
    if (!l || !r) return l ? l : r;
    if (get_pr(l) > get_pr(r)) {
        push_down(l);
        tr[l].r = merge(tr[l].r, r);
        update(l);
        return l;
    } else {
        push_down(r);
        tr[r].l = merge(l, tr[r].l);
        update(r);
        return r;
    }
}

void reverse_range(int P, int Q) {
    int t1, t2, t3;
    split(root, Q, t1, t3);
    split(t1, P - 1, t1, t2);
    set_rev(t2, !get_rev(t2));
    root = merge(t1, merge(t2, t3));
}

unsigned rng_state = 123456789;
inline unsigned next_rand() {
    rng_state ^= rng_state << 13;
    rng_state ^= rng_state >> 17;
    rng_state ^= rng_state << 5;
    return rng_state & 0x7FFFFFFF; // Ensure 31 bits max
}

void build(int l, int r, int &u) {
    if (l > r) return;
    int mid = l + (r - l) / 2;
    u = add_node(mid, next_rand());
    build(l, mid - 1, tr[u].l);
    build(mid + 1, r, tr[u].r);
    update(u);
}

bool first = true;
void print(int u) {
    if (!u) return;
    push_down(u);
    print(tr[u].l);
    if (!first) cout << " ";
    cout << tr[u].val;
    first = false;
    print(tr[u].r);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n, m;
    if (!(cin >> n >> m)) return 0;
    
    build(1, n, root);
    
    for (int i = 0; i < m; i++) {
        int p, q;
        cin >> p >> q;
        reverse_range(p, q);
    }
    
    print(root);
    cout << "\n";
    
    return 0;
}
