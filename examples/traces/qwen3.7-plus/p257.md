#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

struct Crystal {
    int id;
    int score;
    char p, o, s;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long long P, O, S;
    if (!(cin >> P >> O >> S)) return 0;

    int N;
    if (!(cin >> N)) return 0;

    vector<Crystal> crystals(N);
    for (int i = 0; i < N; ++i) {
        string str;
        cin >> str;
        crystals[i].id = i;
        crystals[i].p = str[0];
        crystals[i].o = str[1];
        crystals[i].s = str[2];
        crystals[i].score = (str[0] == 'B' ? 1 : 0) + 
                            (str[1] == 'B' ? 1 : 0) + 
                            (str[2] == 'B' ? 1 : 0);
    }

    sort(crystals.begin(), crystals.end(), [](const Crystal& a, const Crystal& b) {
        if (a.score != b.score) {
            return a.score > b.score;
        }
        return a.id < b.id;
    });

    long long rem_P = P, rem_O = O, rem_S = S;
    string result(N, ' ');

    for (int i = 0; i < N; ++i) {
        int val_P = (crystals[i].p == 'B') ? 2 : 1;
        int val_O = (crystals[i].o == 'B') ? 2 : 1;
        int val_S = (crystals[i].s == 'B') ? 2 : 1;

        bool need_P = (rem_P > 0);
        bool need_O = (rem_O > 0);
        bool need_S = (rem_S > 0);

        if (!need_P && !need_O && !need_S) {
            result[crystals[i].id] = 'P';
            continue;
        }

        int max_val = 0;
        if (need_P) max_val = max(max_val, val_P);
        if (need_O) max_val = max(max_val, val_O);
        if (need_S) max_val = max(max_val, val_S);

        long long best_rem = -1;
        char best_creditor = ' ';

        if (need_P && val_P == max_val) {
            if (rem_P > best_rem) {
                best_rem = rem_P;
                best_creditor = 'P';
            }
        }
        if (need_O && val_O == max_val) {
            if (rem_O > best_rem) {
                best_rem = rem_O;
                best_creditor = 'O';
            }
        }
        if (need_S && val_S == max_val) {
            if (rem_S > best_rem) {
                best_rem = rem_S;
                best_creditor = 'S';
            }
        }

        result[crystals[i].id] = best_creditor;
        if (best_creditor == 'P') rem_P -= val_P;
        else if (best_creditor == 'O') rem_O -= val_O;
        else if (best_creditor == 'S') rem_S -= val_S;
    }

    if (rem_P <= 0 && rem_O <= 0 && rem_S <= 0) {
        cout << result << "\n";
    } else {
        cout << "no solution\n";
    }

    return 0;
}
