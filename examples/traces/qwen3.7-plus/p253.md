#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Point {
    long long x, y;
};

long long dir(Point a, Point b, Point c) {
    return (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M, K;
    if (!(cin >> N >> M >> K)) return 0;

    vector<Point> V(N);
    for (int i = 0; i < N; ++i) {
        cin >> V[i].x >> V[i].y;
    }

    int count = 0;
    for (int i = 0; i < M; ++i) {
        Point P;
        cin >> P.x >> P.y;

        if (dir(V[0], V[1], P) < 0) continue;
        if (dir(V[0], V[N-1], P) > 0) continue;

        int l = 1, r = N - 1;
        int ans = 1;
        while (l <= r) {
            int mid = l + (r - l) / 2;
            if (dir(V[0], V[mid], P) >= 0) {
                ans = mid;
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }

        int next = (ans == N - 1) ? 0 : ans + 1;
        long long d = dir(V[ans], V[next], P);
        if (d > 0) {
            count++;
        } else if (d == 0) {
            if (P.x >= min(V[ans].x, V[next].x) && P.x <= max(V[ans].x, V[next].x) &&
                P.y >= min(V[ans].y, V[next].y) && P.y <= max(V[ans].y, V[next].y)) {
                count++;
            }
        }
    }

    if (count >= K) {
        cout << "YES\n";
    } else {
        cout << "NO\n";
    }

    return 0;
}
