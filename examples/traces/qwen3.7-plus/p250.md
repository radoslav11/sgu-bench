#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct BestBlock {
    int area = 0;
    int r_start = -1;
    int r_end = -1;
    int w = 0;
};

BestBlock up_best[185][185];
BestBlock down_best[185][185];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    vector<vector<int>> grid(N, vector<int>(M));
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < M; ++j) {
            cin >> grid[i][j];
        }
    }

    int best_area = -1;
    int best_x = -1;
    int best_r1 = -1, best_r2 = -1, best_r3 = -1, best_r4 = -1;
    int best_w1 = -1, best_w_mid = -1, best_w3 = -1;

    for (int x = 0; x < M; ++x) {
        vector<int> A(N, 0);
        for (int i = 0; i < N; ++i) {
            int count = 0;
            for (int j = x; j < M; ++j) {
                if (grid[i][j] == 0) {
                    count++;
                } else {
                    break;
                }
            }
            A[i] = count;
        }

        for (int i = 0; i <= N; ++i) {
            for (int w = 0; w <= M + 1; ++w) {
                up_best[i][w] = {0, -1, -1, 0};
                down_best[i][w] = {0, -1, -1, 0};
            }
        }

        for (int r2 = 1; r2 <= N; ++r2) {
            int w_curr = A[r2 - 1];
            for (int r1 = r2 - 1; r1 >= 0; --r1) {
                w_curr = min(w_curr, A[r1]);
                if (w_curr == 0) break;
                int area = (r2 - r1) * w_curr;
                if (area > up_best[r2][w_curr].area) {
                    up_best[r2][w_curr] = {area, r1, r2, w_curr};
                }
            }
            for (int w = M; w >= 1; --w) {
                if (up_best[r2][w + 1].area > up_best[r2][w].area) {
                    up_best[r2][w] = up_best[r2][w + 1];
                }
            }
        }

        for (int r3 = 0; r3 < N; ++r3) {
            int w_curr = A[r3];
            for (int r4 = r3 + 1; r4 <= N; ++r4) {
                w_curr = min(w_curr, A[r4 - 1]);
                if (w_curr == 0) break;
                int area = (r4 - r3) * w_curr;
                if (area > down_best[r3][w_curr].area) {
                    down_best[r3][w_curr] = {area, r3, r4, w_curr};
                }
            }
            for (int w = M; w >= 1; --w) {
                if (down_best[r3][w + 1].area > down_best[r3][w].area) {
                    down_best[r3][w] = down_best[r3][w + 1];
                }
            }
        }

        auto process_segment = [&](int L, int R) {
            int current_max_U = -1e9;
            int best_r2_for_U = -1;
            for (int r3 = L + 1; r3 <= R + 1; ++r3) {
                int r2_candidate = r3 - 1;
                if (up_best[r2_candidate][w_curr + 1].area > 0) {
                    int val_U = up_best[r2_candidate][w_curr + 1].area - r2_candidate * w_curr;
                    if (val_U > current_max_U) {
                        current_max_U = val_U;
                        best_r2_for_U = r2_candidate;
                    }
                }

                if (down_best[r3][w_curr + 1].area > 0 && current_max_U > -1e9) {
                    int val_D = down_best[r3][w_curr + 1].area + r3 * w_curr;
                    int total = val_D + current_max_U;
                    if (total > best_area) {
                        best_area = total;
                        best_x = x;
                        best_r2 = best_r2_for_U;
                        best_r3 = r3;
                        best_w_mid = w_curr;

                        best_r1 = up_best[best_r2][w_curr + 1].r_start;
                        best_w1 = up_best[best_r2][w_curr + 1].w;

                        best_w3 = down_best[best_r3][w_curr + 1].w;
                        best_r4 = down_best[best_r3][w_curr + 1].r_end;
                    }
                }
            }
        };

        for (int w = 1; w <= M; ++w) {
            int L = -1;
            for (int i = 0; i < N; ++i) {
                if (A[i] >= w) {
                    if (L == -1) L = i;
                } else {
                    if (L != -1) {
                        int w_curr = w;
                        process_segment(L, i - 1);
                        L = -1;
                    }
                }
            }
            if (L != -1) {
                int w_curr = w;
                process_segment(L, N - 1);
            }
        }
    }

    if (best_area == -1) {
        cout << -1 << "\n";
    } else {
        cout << best_area << "\n";
        vector<vector<int>> ans = grid;
        for (int i = best_r1; i < best_r2; ++i) {
            for (int j = best_x; j < best_x + best_w1; ++j) {
                ans[i][j] = 8;
            }
        }
        for (int i = best_r2; i < best_r3; ++i) {
            for (int j = best_x; j < best_x + best_w_mid; ++j) {
                ans[i][j] = 8;
            }
        }
        for (int i = best_r3; i < best_r4; ++i) {
            for (int j = best_x; j < best_x + best_w3; ++j) {
                ans[i][j] = 8;
            }
        }

        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < M; ++j) {
                cout << ans[i][j] << (j == M - 1 ? "" : " ");
            }
            cout << "\n";
        }
    }

    return 0;
}
