#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

using namespace std;

#pragma GCC optimize("O3,unroll-loops")
#pragma GCC target("avx2,bmi,bmi2,lzcnt,popcnt")

struct Point {
    double x, y;
};

struct Line {
    double A, B, C;
};

inline double min_dist(double x, double y, int n, const vector<Line>& orig) {
    double res = 1e18;
    for (int i = 0; i < n; i++) {
        double d = orig[i].A * x + orig[i].B * y + orig[i].C;
        if (d < res) res = d;
    }
    return res;
}

double get_max_R_for_x(double x, int n, const vector<Line>& orig) {
    double min_y = -1e18, max_y = 1e18;
    for (int i = 0; i < n; i++) {
        if (abs(orig[i].B) > 1e-9) {
            double y_bound = (-orig[i].C - orig[i].A * x) / orig[i].B;
            if (orig[i].B > 0) {
                if (y_bound < max_y) max_y = y_bound;
            } else {
                if (y_bound > min_y) min_y = y_bound;
            }
        }
    }
    
    if (min_y > max_y) {
        double mid_y = (min_y + max_y) / 2.0;
        return min_dist(x, mid_y, n, orig);
    }
    
    double low = min_y, high = max_y;
    for (int iter = 0; iter < 50; iter++) {
        double m1 = low + (high - low) / 3.0;
        double m2 = high - (high - low) / 3.0;
        double f1 = min_dist(x, m1, n, orig);
        double f2 = min_dist(x, m2, n, orig);
        if (f1 < f2) {
            low = m1;
        } else {
            high = m2;
        }
    }
    return min_dist(x, low, n, orig);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    if (!(cin >> n)) return 0;
    
    vector<Point> v(n);
    double min_x = 1e18, max_x = -1e18;
    for (int i = 0; i < n; i++) {
        cin >> v[i].x >> v[i].y;
        if (v[i].x < min_x) min_x = v[i].x;
        if (v[i].x > max_x) max_x = v[i].x;
    }
    
    vector<Line> orig(n);
    for (int i = 0; i < n; i++) {
        int j = (i + 1) % n;
        double dx = v[j].x - v[i].x;
        double dy = v[j].y - v[i].y;
        double len = hypot(dx, dy);
        if (len < 1e-9) continue;
        orig[i].A = -dy / len;
        orig[i].B = dx / len;
        orig[i].C = -(orig[i].A * v[i].x + orig[i].B * v[i].y);
    }
    
    double low_x = min_x, high_x = max_x;
    for (int iter = 0; iter < 60; iter++) {
        double m1 = low_x + (high_x - low_x) / 3.0;
        double m2 = high_x - (high_x - low_x) / 3.0;
        double f1 = get_max_R_for_x(m1, n, orig);
        double f2 = get_max_R_for_x(m2, n, orig);
        if (f1 < f2) {
            low_x = m1;
        } else {
            high_x = m2;
        }
    }
    
    double ans = get_max_R_for_x(low_x, n, orig);
    if (ans < 0) ans = 0;
    
    cout << fixed;
    cout.precision(6);
    cout << ans << "\n";
    
    return 0;
}
