#include <iostream>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;
    
    int count = 0;
    for (int i = 0; i < n; ++i) {
        long long x;
        cin >> x;
        
        long long res = 1;
        long long base = x % k;
        int exp = m;
        
        while (exp > 0) {
            if (exp % 2 == 1) {
                res = (res * base) % k;
            }
            base = (base * base) % k;
            exp /= 2;
        }
        
        if (res == 0) {
            count++;
        }
    }
    
    cout << count << "\n";
    
    return 0;
}
