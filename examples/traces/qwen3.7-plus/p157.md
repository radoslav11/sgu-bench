```cpp
#include <iostream>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    if (cin >> n) {
        long long catalan = 1;
        for (int i = 1; i <= n; ++i) {
            catalan = catalan * (4 * i - 2) / (i + 1);
        }
        cout << catalan << "\n";
    }
    
    return 0;
}
```
