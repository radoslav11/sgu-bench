```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int F, V;
    if (!(cin >> F >> V)) return 0;

    vector<vector<int>> A(F, vector<int>(V));
    for (int i = 0; i < F; ++i) {
        for (int j = 0; j < V; ++j) {
            cin >> A[i][j];
        }
    }

    const int INF = 1e9;
    vector<vector<int>> dp(F, vector<int>(V, -INF));
    vector<vector<int>> parent(F, vector<int>(V, -1));

    for (int j = 0; j < V; ++j) {
        dp[0][j] = A[0][j];
    }

    for (int i = 1; i < F; ++i) {
        for (int j = i; j < V; ++j) {
            int best_val = -INF;
            int best_k = -1;
            for (int k = i - 1; k < j; ++k) {
                if (dp[i - 1][k] > best_val) {
                    best_val = dp[i - 1][k];
                    best_k = k;
                }
            }
            if (best_k != -1) {
                dp[i][j] = best_val + A[i][j];
                parent[i][j] = best_k;
            }
        }
    }

    int max_val = -INF;
    int best_j = -1;
    for (int j = F - 1; j < V; ++j) {
        if (dp[F - 1][j] > max_val) {
            max_val = dp[F - 1][j];
            best_j = j;
        }
    }

    vector<int> ans(F);
    int curr = best_j;
    for (int i = F - 1; i >= 0; --i) {
        ans[i] = curr + 1;
        curr = parent[i][curr];
    }

    cout << max_val << "\n";
    for (int i = 0; i < F; ++i) {
        cout << ans[i] << (i == F - 1 ? "" : " ");
    }
    cout << "\n";

    return 0;
}
```
