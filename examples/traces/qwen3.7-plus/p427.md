#include <iostream>
#include <vector>
#include <string>
#include <map>
#include <algorithm>

using namespace std;

struct Point {
    long long x, y, z;
    bool operator<(const Point& o) const {
        if (x != o.x) return x < o.x;
        if (y != o.y) return y < o.y;
        return z < o.z;
    }
    bool operator==(const Point& o) const {
        return x == o.x && y == o.y && z == o.z;
    }
};

long long parse_coord(const string& s) {
    long long res = 0;
    bool neg = false;
    int i = 0;
    if (s[i] == '-') {
        neg = true;
        i++;
    }
    while (i < (int)s.length() && s[i] != '.') {
        res = res * 10 + (s[i] - '0');
        i++;
    }
    if (i < (int)s.length() && s[i] == '.') {
        i++;
    }
    while (i < (int)s.length()) {
        res = res * 10 + (s[i] - '0');
        i++;
    }
    return neg ? -res : res;
}

struct PointStr {
    string sx, sy, sz;
    Point p;
};

int N;
vector<vector<int>> adj;
vector<int> path;
vector<bool> visited;
vector<PointStr> unique_vertices;
map<Point, int> p_to_id;

bool check_valid(int depth) {
    int start_node = path[0];
    int start_deg = 0;
    for (int w : adj[start_node]) {
        if (!visited[w]) start_deg++;
    }
    if (start_deg == 0) return false;
    
    for (int v = 0; v < N; v++) {
        if (!visited[v] && v != start_node) {
            int deg = 0;
            for (int w : adj[v]) {
                if (!visited[w] || w == start_node) deg++;
            }
            if (deg < 2) return false;
        }
    }
    return true;
}

bool check_articulation(int depth) {
    int start_node = path[0];
    for (int v = 0; v < N; v++) {
        if (!visited[v] && v != start_node) {
            vector<bool> temp_visited(N, false);
            temp_visited[v] = true;
            
            int components = 0;
            for (int i = 0; i < N; i++) {
                if ((i == start_node || !visited[i]) && !temp_visited[i]) {
                    components++;
                    if (components > 2) return false;
                    
                    vector<int> q;
                    q.push_back(i);
                    temp_visited[i] = true;
                    int head = 0;
                    while (head < (int)q.size()) {
                        int curr = q[head++];
                        for (int w : adj[curr]) {
                            if ((w == start_node || !visited[w]) && !temp_visited[w]) {
                                temp_visited[w] = true;
                                q.push_back(w);
                            }
                        }
                    }
                }
            }
        }
    }
    return true;
}

bool dfs(int u, int depth) {
    path[depth] = u;
    visited[u] = true;
    
    if (depth == N - 1) {
        if (binary_search(adj[u].begin(), adj[u].end(), path[0])) {
            return true;
        }
        visited[u] = false;
        return false;
    }
    
    if (!check_valid(depth) || !check_articulation(depth)) {
        visited[u] = false;
        return false;
    }
    
    vector<pair<int, int>> next_nodes;
    for (int v : adj[u]) {
        if (!visited[v]) {
            int unvisited_neighbors = 0;
            for (int w : adj[v]) {
                if (!visited[w]) unvisited_neighbors++;
            }
            next_nodes.push_back({unvisited_neighbors, v});
        }
    }
    
    if (depth == N - 2) {
        vector<pair<int, int>> filtered_next;
        for (auto& p : next_nodes) {
            if (binary_search(adj[p.second].begin(), adj[p.second].end(), path[0])) {
                filtered_next.push_back(p);
            }
        }
        next_nodes = filtered_next;
    }
    
    sort(next_nodes.begin(), next_nodes.end());
    
    for (auto& p : next_nodes) {
        if (dfs(p.second, depth + 1)) return true;
    }
    
    visited[u] = false;
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int f;
    if (!(cin >> f)) return 0;
    
    int num_vertices = 0;
    for (int i = 0; i < f; i++) {
        int k;
        cin >> k;
        vector<int> facet_vertices(k);
        for (int j = 0; j < k; j++) {
            string sx, sy, sz;
            cin >> sx >> sy >> sz;
            Point p = {parse_coord(sx), parse_coord(sy), parse_coord(sz)};
            if (p_to_id.find(p) == p_to_id.end()) {
                p_to_id[p] = num_vertices;
                unique_vertices.push_back({sx, sy, sz, p});
                adj.push_back(vector<int>());
                num_vertices++;
            }
            facet_vertices[j] = p_to_id[p];
        }
        for (int j = 0; j < k; j++) {
            int u = facet_vertices[j];
            int v = facet_vertices[(j + 1) % k];
            if (u != v) {
                adj[u].push_back(v);
                adj[v].push_back(u);
            }
        }
    }
    
    N = num_vertices;
    for (int i = 0; i < N; i++) {
        sort(adj[i].begin(), adj[i].end());
        adj[i].erase(unique(adj[i].begin(), adj[i].end()), adj[i].end());
    }
    
    path.resize(N);
    visited.assign(N, false);
    
    vector<pair<int, int>> start_candidates;
    for (int i = 0; i < N; i++) {
        start_candidates.push_back({(int)adj[i].size(), i});
    }
    sort(start_candidates.begin(), start_candidates.end());
    
    bool found = false;
    for (auto& cand : start_candidates) {
        int start_v = cand.second;
        fill(visited.begin(), visited.end(), false);
        if (dfs(start_v, 0)) {
            found = true;
            break;
        }
    }
    
    if (found) {
        cout << "Yes\n" << N << "\n";
        for (int i = 0; i < N; i++) {
            int id = path[i];
            cout << unique_vertices[id].sx << " " 
                 << unique_vertices[id].sy << " " 
                 << unique_vertices[id].sz << "\n";
        }
    } else {
        cout << "No\n";
    }
    
    return 0;
}
