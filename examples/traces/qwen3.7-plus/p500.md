#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <deque>

using namespace std;

const long double PI = acos(-1.0L);

struct PointLL {
    long long x, y;
    bool operator<(const PointLL& o) const {
        if (x != o.x) return x < o.x;
        return y < o.y;
    }
};

long long cross_ll(const PointLL& a, const PointLL& b, const PointLL& c) {
    return (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x);
}

vector<PointLL> convex_hull_ll(vector<PointLL> pts) {
    sort(pts.begin(), pts.end());
    vector<PointLL> h;
    for (auto& p : pts) {
        while (h.size() >= 2 && cross_ll(h[h.size()-2], h.back(), p) <= 0) {
            h.pop_back();
        }
        h.push_back(p);
    }
    int t = h.size() + 1;
    for (int i = (int)pts.size() - 2; i >= 0; --i) {
        PointLL p = pts[i];
        while (h.size() >= t && cross_ll(h[h.size()-2], h.back(), p) <= 0) {
            h.pop_back();
        }
        h.push_back(p);
    }
    if (h.size() > 1) h.pop_back();
    return h;
}

struct Point {
    long double x, y;
    Point operator+(const Point& o) const { return {x + o.x, y + o.y}; }
    Point operator-(const Point& o) const { return {x - o.x, y - o.y}; }
    Point operator*(long double k) const { return {x * k, y * k}; }
};

long double cross(const Point& a, const Point& b) { return a.x * b.y - a.y * b.x; }
long double dot(const Point& a, const Point& b) { return a.x * b.x + a.y * b.y; }

struct Line {
    Point p, d;
    long double angle;
    Line() : p({0,0}), d({0,0}), angle(0) {}
    Line(Point p, Point d) : p(p), d(d) {
        angle = atan2(d.y, d.x);
    }
    bool operator<(const Line& o) const {
        if (abs(angle - o.angle) > 1e-12) return angle < o.angle;
        return cross(d, o.p - p) > 0;
    }
};

Point intersect(const Line& a, const Line& b) {
    return a.p + a.d * (cross(b.p - a.p, b.d) / cross(a.d, b.d));
}

bool check(const Line& a, const Line& b, const Line& c) {
    Point p = intersect(a, b);
    return cross(c.d, p - c.p) >= -1e-9;
}

vector<Point> half_plane_intersection(vector<Line> lines) {
    sort(lines.begin(), lines.end());
    deque<Line> q;
    for (auto& line : lines) {
        if (!q.empty() && abs(cross(q.back().d, line.d)) < 1e-12) continue;
        while (q.size() >= 2 && !check(q[q.size()-2], q.back(), line)) {
            q.pop_back();
        }
        q.push_back(line);
    }
    while (q.size() >= 2 && !check(q[q.size()-2], q.back(), q[0])) {
        q.pop_back();
    }
    vector<Point> poly;
    for (size_t i = 0; i < q.size(); ++i) {
        poly.push_back(intersect(q[i], q[(i + 1) % q.size()]));
    }
    return poly;
}

long double area_segment(long double d, long double r) {
    if (d >= r) return 0.0;
    if (d <= -r) return PI * r * r;
    return r * r * acos(d / r) - d * sqrt(max(0.0L, r * r - d * d));
}

long double solve_region(const vector<PointLL>& ch_J, const vector<PointLL>& ch_S, bool region1) {
    bool valid = true;
    vector<Line> lines;
    auto add_hp = [&](long double a, long double b, long double c) {
        if (abs(a) < 1e-12 && abs(b) < 1e-12) {
            if (c < 0) valid = false;
            return;
        }
        Point p, d;
        d = {-b, a};
        if (abs(a) >= abs(b)) p = {c / a, 0.0};
        else p = {0.0, c / b};
        lines.push_back(Line(p, d));
    };

    if (region1) {
        for (auto& v : ch_J) add_hp(v.x, v.y, -1.0);
        for (auto& u : ch_S) add_hp(-u.x, -u.y, 1.0);
    } else {
        for (auto& v : ch_J) add_hp(-v.x, -v.y, 1.0);
        for (auto& u : ch_S) add_hp(u.x, u.y, -1.0);
    }

    if (!valid) return 1e18;

    long double M = 1e15;
    add_hp(1, 0, M);
    add_hp(-1, 0, M);
    add_hp(0, 1, M);
    add_hp(0, -1, M);

    vector<Point> poly = half_plane_intersection(lines);
    if (poly.size() < 3) return 1e18;

    long double min_dist_sq = 1e30;
    for (size_t i = 0; i < poly.size(); ++i) {
        Point p1 = poly[i];
        Point p2 = poly[(i + 1) % poly.size()];
        long double d1 = p1.x * p1.x + p1.y * p1.y;
        if (d1 < min_dist_sq) min_dist_sq = d1;

        Point seg = p2 - p1;
        long double len_sq = seg.x * seg.x + seg.y * seg.y;
        if (len_sq > 1e-12) {
            long double t = -dot(p1, seg) / len_sq;
            if (t >= 0 && t <= 1) {
                Point proj = p1 + seg * t;
                long double d_proj = proj.x * proj.x + proj.y * proj.y;
                if (d_proj < min_dist_sq) min_dist_sq = d_proj;
            }
        }
    }
    return sqrt(min_dist_sq);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long double r;
    if (!(cin >> r)) return 0;

    int n;
    cin >> n;
    vector<PointLL> J_ll(n);
    for (int i = 0; i < n; ++i) {
        cin >> J_ll[i].x >> J_ll[i].y;
    }

    int m;
    cin >> m;
    vector<PointLL> S_ll(m);
    for (int i = 0; i < m; ++i) {
        cin >> S_ll[i].x >> S_ll[i].y;
    }

    vector<PointLL> ch_J = convex_hull_ll(J_ll);
    vector<PointLL> ch_S = convex_hull_ll(S_ll);

    long double D1 = solve_region(ch_J, ch_S, true);
    long double d1 = (D1 > 1e17) ? 0.0 : (1.0 / D1);
    long double min_area = area_segment(d1, r);

    long double D2 = solve_region(ch_J, ch_S, false);
    long double d2 = (D2 > 1e17) ? 0.0 : (1.0 / D2);
    long double max_area = PI * r * r - area_segment(d2, r);

    cout.precision(15);
    cout << fixed << min_area << " " << max_area << "\n";

    return 0;
}
