#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int MAX = 10005;
bool is_prime[MAX + 1];
vector<int> primes;
vector<int> super_primes;

void sieve() {
    fill(is_prime, is_prime + MAX + 1, true);
    is_prime[0] = is_prime[1] = false;
    for (int p = 2; p * p <= MAX; p++) {
        if (is_prime[p]) {
            for (int i = p * p; i <= MAX; i += p)
                is_prime[i] = false;
        }
    }
    for (int p = 2; p <= MAX; p++) {
        if (is_prime[p]) {
            primes.push_back(p);
        }
    }
    for (size_t i = 0; i < primes.size(); i++) {
        if (is_prime[i + 1]) {
            super_primes.push_back(primes[i]);
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    if (cin >> n) {
        sieve();
        
        const int INF = 1000000000;
        vector<int> dp(n + 1, INF);
        dp[0] = 0;
        
        for (int i = 1; i <= n; i++) {
            for (int sp : super_primes) {
                if (i >= sp && dp[i - sp] + 1 < dp[i]) {
                    dp[i] = dp[i - sp] + 1;
                }
            }
        }
        
        if (dp[n] == INF) {
            cout << 0 << "\n";
        } else {
            vector<int> ans;
            int cur = n;
            while (cur > 0) {
                for (int i = (int)super_primes.size() - 1; i >= 0; i--) {
                    int sp = super_primes[i];
                    if (cur >= sp && dp[cur - sp] == dp[cur] - 1) {
                        ans.push_back(sp);
                        cur -= sp;
                        break;
                    }
                }
            }
            
            sort(ans.begin(), ans.end(), greater<int>());
            
            cout << dp[n] << "\n";
            for (int i = 0; i < (int)ans.size(); i++) {
                cout << ans[i] << (i + 1 == (int)ans.size() ? "" : " ");
            }
            cout << "\n";
        }
    }
    return 0;
}
