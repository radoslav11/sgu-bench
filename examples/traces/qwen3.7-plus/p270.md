#include <iostream>
#include <vector>

using namespace std;

int n, m;
vector<pair<int, int>> edges;

int cnt(int u, int v) {
    int c = 0;
    for (auto& e : edges) {
        if ((e.first == u && e.second == v) || (e.first == v && e.second == u)) c++;
    }
    return c;
}

bool connected_without(int target, int skip_u, int skip_v) {
    vector<vector<int>> adj(n + 1);
    for (auto& e : edges) {
        int u = e.first, v = e.second;
        if ((u == skip_u && v == skip_v) || (u == skip_v && v == skip_u)) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    vector<bool> vis(n + 1, false);
    vector<int> q;
    q.push_back(1);
    vis[1] = true;
    int head = 0;
    while (head < (int)q.size()) {
        int curr = q[head++];
        if (curr == target) return true;
        for (int nxt : adj[curr]) {
            if (!vis[nxt]) {
                vis[nxt] = true;
                q.push_back(nxt);
            }
        }
    }
    return false;
}

bool has_cycle_ge3(int u) {
    vector<bool> is_nei(n + 1, false);
    for (auto& e : edges) {
        if (e.first == u) is_nei[e.second] = true;
        else if (e.second == u) is_nei[e.first] = true;
    }
    vector<int> nei_list;
    for (int i = 1; i <= n; ++i) {
        if (is_nei[i]) nei_list.push_back(i);
    }
    
    if (nei_list.size() < 2) return false;
    
    vector<vector<int>> adj(n + 1);
    for (auto& e : edges) {
        if (e.first == u || e.second == u) continue;
        adj[e.first].push_back(e.second);
        adj[e.second].push_back(e.first);
    }
    
    vector<bool> vis(n + 1, false);
    for (int start : nei_list) {
        if (vis[start]) continue;
        vector<int> comp;
        vector<int> q;
        q.push_back(start);
        vis[start] = true;
        int head = 0;
        while (head < (int)q.size()) {
            int curr = q[head++];
            comp.push_back(curr);
            for (int nxt : adj[curr]) {
                if (!vis[nxt]) {
                    vis[nxt] = true;
                    q.push_back(nxt);
                }
            }
        }
        int count_in_comp = 0;
        for (int node : comp) {
            if (is_nei[node]) count_in_comp++;
        }
        if (count_in_comp >= 2) return true;
    }
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    if (!(cin >> n >> m)) return 0;
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        edges.push_back({u, v});
    }
    
    vector<int> ans;
    for (int v = 1; v <= n; ++v) {
        bool reachable = false;
        if (v != 1) {
            bool c1 = connected_without(v, 1, v);
            bool c2 = (cnt(1, v) >= 1 && has_cycle_ge3(1));
            bool c3 = false;
            for (int x = 1; x <= n; ++x) {
                if (x != 1 && x != v && cnt(1, x) >= 2) c3 = true;
            }
            bool c4 = (cnt(1, v) >= 1 && has_cycle_ge3(v));
            bool c5 = false;
            for (int y = 1; y <= n; ++y) {
                if (y != 1 && y != v && cnt(v, y) >= 2) c5 = true;
            }
            bool c6 = (cnt(1, v) >= 3);
            
            if (c1 || c2 || c3 || c4 || c5 || c6) {
                reachable = true;
            } else {
                if (cnt(1, v) % 2 == 1) reachable = true;
            }
        } else {
            bool c1 = has_cycle_ge3(1);
            bool c2 = false;
            for (int x = 1; x <= n; ++x) {
                if (x != 1 && cnt(1, x) >= 2) c2 = true;
            }
            if (c1 || c2) {
                reachable = true;
            } else {
                vector<bool> vis(n + 1, false);
                vector<int> q;
                q.push_back(1);
                vis[1] = true;
                int head = 0;
                while (head < (int)q.size()) {
                    int curr = q[head++];
                    for (auto& e : edges) {
                        int nxt = (e.first == curr) ? e.second : (e.second == curr ? e.first : 0);
                        if (nxt != 0 && !vis[nxt]) {
                            vis[nxt] = true;
                            q.push_back(nxt);
                        }
                    }
                }
                int comp_size = 0;
                int other_node = -1;
                for (int i = 1; i <= n; ++i) {
                    if (vis[i]) {
                        comp_size++;
                        if (i != 1) other_node = i;
                    }
                }
                if (comp_size == 1) {
                    reachable = true;
                } else if (comp_size == 2 && cnt(1, other_node) % 2 == 0) {
                    reachable = true;
                }
            }
        }
        if (reachable) ans.push_back(v);
    }
    
    for (int i = 0; i < (int)ans.size(); ++i) {
        cout << ans[i] << (i + 1 == (int)ans.size() ? "" : " ");
    }
    cout << "\n";
    return 0;
}
