#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long long N, u_int, t_int;
    if (!(cin >> N >> u_int >> t_int)) return 0;
    
    double u = u_int;
    double t = t_int;

    vector<double> w(N), v(N);
    long long total_w = 0;
    double max_v = 0;
    for (int i = 0; i < N; ++i) {
        long long w_int, v_int;
        cin >> w_int >> v_int;
        w[i] = w_int;
        v[i] = v_int;
        total_w += w_int;
        if (v_int > max_v) max_v = v_int;
    }

    if (t_int * u_int < total_w) {
        cout << -1 << "\n";
        return 0;
    }

    vector<double> T(N);
    if (t_int * u_int == total_w) {
        for (int i = 0; i < N; ++i) {
            T[i] = w[i] / u;
        }
    } else {
        double low = max_v + u;
        double high = 1e12;
        double best_c = high;

        for (int iter = 0; iter < 200; ++iter) {
            double mid = low + (high - low) / 2.0;
            double sum_t = 0;
            for (int i = 0; i < N; ++i) {
                double D = mid - v[i];
                double diff = D * D - u * u;
                if (diff < 0) diff = 0;
                double denom = u * sqrt(diff);
                if (denom > 1e-12) {
                    sum_t += w[i] / denom;
                } else {
                    sum_t = 1e18;
                    break;
                }
            }
            if (sum_t > t) {
                low = mid;
            } else {
                high = mid;
                best_c = mid;
            }
        }

        double sum_t_actual = 0;
        for (int i = 0; i < N; ++i) {
            double D = best_c - v[i];
            double diff = D * D - u * u;
            if (diff < 0) diff = 0;
            double denom = u * sqrt(diff);
            T[i] = w[i] / denom;
            sum_t_actual += T[i];
        }

        for (int i = 0; i < N; ++i) {
            T[i] = T[i] * (t / sum_t_actual);
        }
    }

    double Y = 0;
    double W = 0;
    for (int i = 0; i < N; ++i) {
        W += w[i];
        double ux = w[i] / T[i];
        double uy = sqrt(max(0.0, u * u - ux * ux));
        Y += T[i] * (v[i] + uy);
    }

    double dist = sqrt(W * W + Y * Y);

    cout << fixed << setprecision(10) << dist << "\n";
    for (int i = 0; i < N; ++i) {
        cout << fixed << setprecision(10) << T[i] << (i == N - 1 ? "" : " ");
    }
    cout << "\n";

    return 0;
}
