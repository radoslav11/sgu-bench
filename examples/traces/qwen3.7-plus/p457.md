#include <iostream>
#include <vector>
#include <queue>
#include <cstring>
#include <functional>

using namespace std;

struct Edge {
    int to;
    long long cap;
    long long flow;
    int rev;
    int id;
};

vector<vector<Edge>> adj;
vector<long long> level;
vector<int> ptr;

struct OrigEdge {
    int u, v;
    long long L, C;
    long long flow;
};
vector<OrigEdge> orig_edges;
int edge_idx[105][105];
long long demand[105];

void add_orig(int u, int v, long long L, long long C) {
    edge_idx[u][v] = orig_edges.size();
    int idx = adj[u].size();
    int rev = adj[v].size();
    adj[u].push_back({v, C - L, 0, rev, (int)orig_edges.size()});
    adj[v].push_back({u, 0, 0, idx, -1});
    orig_edges.push_back({u, v, L, C, L});
    demand[u] -= L;
    demand[v] += L;
}

long long* BA_rev_cap_ptr = nullptr;
void add_BA(int u, int v) {
    int idx = adj[u].size();
    int rev = adj[v].size();
    adj[u].push_back({v, 1e18, 0, rev, (int)orig_edges.size()});
    adj[v].push_back({u, 0, 0, idx, -1});
    orig_edges.push_back({u, v, 0, 1e18, 0});
    BA_rev_cap_ptr = &adj[v][rev].cap;
}

bool bfs(int s, int t) {
    fill(level.begin(), level.end(), -1);
    level[s] = 0;
    queue<int> q;
    q.push(s);
    while (!q.empty()) {
        int v = q.front(); q.pop();
        for (auto& edge : adj[v]) {
            if (edge.cap - edge.flow > 0 && level[edge.to] == -1) {
                level[edge.to] = level[v] + 1;
                q.push(edge.to);
            }
        }
    }
    return level[t] != -1;
}

long long dfs_dinic(int v, int t, long long pushed) {
    if (pushed == 0) return 0;
    if (v == t) return pushed;
    for (int& cid = ptr[v]; cid < adj[v].size(); ++cid) {
        auto& edge = adj[v][cid];
        int tr = edge.to;
        if (level[v] + 1 != level[tr] || edge.cap - edge.flow == 0) continue;
        long long push = dfs_dinic(tr, t, min(pushed, edge.cap - edge.flow));
        if (push == 0) continue;
        edge.flow += push;
        adj[tr][edge.rev].flow -= push;
        return push;
    }
    return 0;
}

long long dinic(int s, int t) {
    long long flow = 0;
    while (bfs(s, t)) {
        fill(ptr.begin(), ptr.end(), 0);
        while (long long pushed = dfs_dinic(s, t, 1e18)) {
            flow += pushed;
        }
    }
    return flow;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n, m, A, B;
    if (!(cin >> n >> m >> A >> B)) return 0;
    
    adj.assign(n + 2, vector<Edge>());
    level.resize(n + 2);
    ptr.resize(n + 2);
    memset(edge_idx, -1, sizeof(edge_idx));
    memset(demand, 0, sizeof(demand));
    
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        int t;
        cin >> u >> v >> w >> t;
        if (t == 1) {
            add_orig(u, v, w, w);
        } else {
            add_orig(u, v, 0, w);
        }
    }
    
    add_BA(B, A);
    
    int S = 0, T = n + 1;
    long long total_demand = 0;
    for (int i = 1; i <= n; ++i) {
        if (demand[i] > 0) {
            int idx = adj[S].size();
            int rev = adj[i].size();
            adj[S].push_back({i, demand[i], 0, rev, -1});
            adj[i].push_back({S, 0, 0, idx, -1});
            total_demand += demand[i];
        } else if (demand[i] < 0) {
            int idx = adj[i].size();
            int rev = adj[T].size();
            adj[i].push_back({T, -demand[i], 0, rev, -1});
            adj[T].push_back({i, 0, 0, idx, -1});
        }
    }
    
    long long flow_st = dinic(S, T);
    if (flow_st < total_demand) {
        cout << 0 << "\n";
        return 0;
    }
    
    if (BA_rev_cap_ptr) {
        *BA_rev_cap_ptr = 0;
    }
    
    dinic(A, B);
    
    for (int i = 0; i < (int)orig_edges.size(); ++i) {
        for (auto& edge : adj[orig_edges[i].u]) {
            if (edge.id == i) {
                orig_edges[i].flow += edge.flow;
                break;
            }
        }
    }
    
    long long P = orig_edges.back().flow;
    if (P == 0) {
        cout << 0 << "\n";
        return 0;
    }
    
    cout << P << "\n";
    
    vector<vector<pair<int, int>>> orig_adj(n + 1);
    for (int i = 0; i < (int)orig_edges.size(); ++i) {
        orig_adj[orig_edges[i].u].push_back({orig_edges[i].v, i});
    }
    
    for (long long i = 0; i < P; ++i) {
        vector<int> path;
        vector<bool> in_path(n + 1, false);
        
        function<bool(int)> dfs = [&](int u) {
            if (u == B) {
                path.push_back(u);
                return true;
            }
            in_path[u] = true;
            path.push_back(u);
            for (auto& p : orig_adj[u]) {
                int v = p.first;
                int idx = p.second;
                if (orig_edges[idx].flow > 0 && !in_path[v]) {
                    if (dfs(v)) return true;
                }
            }
            path.pop_back();
            in_path[u] = false;
            return false;
        };
        
        if (!dfs(A)) break;
        
        for (size_t j = 0; j < path.size() - 1; ++j) {
            int u = path[j];
            int v = path[j+1];
            int idx = edge_idx[u][v];
            orig_edges[idx].flow--;
        }
        orig_edges.back().flow--;
        
        for (size_t j = 0; j < path.size(); ++j) {
            cout << path[j] << (j + 1 == path.size() ? "" : " ");
        }
        cout << "\n";
    }
    
    return 0;
}
