#include <iostream>
#include <vector>
#include <iomanip>
#include <algorithm>

using namespace std;

struct BigInt {
    vector<long long> digits;
    BigInt(long long v = 0) {
        if (v == 0) digits.push_back(0);
        while (v > 0) {
            digits.push_back(v % 1000000000);
            v /= 1000000000;
        }
    }
    void add(const BigInt& o) {
        long long carry = 0;
        size_t n = max(digits.size(), o.digits.size());
        for (size_t i = 0; i < n || carry; ++i) {
            if (i == digits.size()) digits.push_back(0);
            long long sum = digits[i] + carry + (i < o.digits.size() ? o.digits[i] : 0);
            digits[i] = sum % 1000000000;
            carry = sum / 1000000000;
        }
    }
    void mul(int v) {
        if (v == 0) {
            digits = {0};
            return;
        }
        long long carry = 0;
        for (size_t i = 0; i < digits.size() || carry; ++i) {
            if (i == digits.size()) digits.push_back(0);
            long long prod = digits[i] * v + carry;
            digits[i] = prod % 1000000000;
            carry = prod / 1000000000;
        }
        while (digits.size() > 1 && digits.back() == 0) digits.pop_back();
    }
    int mod(int v) const {
        long long rem = 0;
        for (int i = (int)digits.size() - 1; i >= 0; --i) {
            rem = (rem * 1000000000 + digits[i]) % v;
        }
        return (int)rem;
    }
    void div(int v) {
        if (v == 1) return;
        long long carry = 0;
        for (int i = (int)digits.size() - 1; i >= 0; --i) {
            long long cur = carry * 1000000000 + digits[i];
            digits[i] = cur / v;
            carry = cur % v;
        }
        while (digits.size() > 1 && digits.back() == 0) digits.pop_back();
    }
    void print() const {
        if (digits.empty() || (digits.size() == 1 && digits[0] == 0)) {
            cout << 0;
            return;
        }
        cout << digits.back();
        for (int i = (int)digits.size() - 2; i >= 0; --i) {
            cout << setfill('0') << setw(9) << digits[i];
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int K, N;
    if (!(cin >> K >> N)) return 0;

    int m = N - K;
    if (m == 1) {
        cout << 0 << "\n";
        return 0;
    }

    BigInt D;
    if (m == 0) {
        D = BigInt(1);
    } else {
        BigInt D0(1);
        BigInt D1(0);
        for (int i = 2; i <= m; ++i) {
            BigInt next_D = D0;
            next_D.add(D1);
            next_D.mul(i - 1);
            D0 = D1;
            D1 = next_D;
        }
        D = D1;
    }

    vector<int> primes;
    vector<bool> is_prime(105, true);
    is_prime[0] = is_prime[1] = false;
    for (int p = 2; p <= 100; ++p) {
        if (is_prime[p]) {
            primes.push_back(p);
            for (int i = p * p; i <= 100; i += p) {
                is_prime[i] = false;
            }
        }
    }

    BigInt den(1);
    for (int p : primes) {
        int count = 0;
        int temp = K;
        while (temp > 0) {
            count += temp / p;
            temp /= p;
        }
        temp = m;
        while (temp > 0) {
            count += temp / p;
            temp /= p;
        }
        
        while (count > 0 && D.mod(p) == 0) {
            D.div(p);
            count--;
        }
        
        for (int i = 0; i < count; ++i) {
            den.mul(p);
        }
    }

    D.print();
    cout << "/";
    den.print();
    cout << "\n";

    return 0;
}
