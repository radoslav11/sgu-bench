#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct City {
    int x, y, id;
    int compressed_y;
    int dp_pre, dp_suf;
};

struct BIT {
    int n;
    vector<int> tree;
    BIT(int n) : n(n), tree(n + 1, 0) {}
    void update(int i, int delta) {
        for (; i <= n; i += i & -i) {
            tree[i] = max(tree[i], delta);
        }
    }
    int query(int i) {
        int res = 0;
        for (; i > 0; i -= i & -i) {
            res = max(res, tree[i]);
        }
        return res;
    }
    void clear() {
        fill(tree.begin(), tree.end(), 0);
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    vector<City> cities(n);
    vector<int> ys(n);
    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
        cities[i].id = i + 1;
        ys[i] = cities[i].y;
    }

    sort(ys.begin(), ys.end());
    ys.erase(unique(ys.begin(), ys.end()), ys.end());
    int M = ys.size();

    for (int i = 0; i < n; i++) {
        cities[i].compressed_y = lower_bound(ys.begin(), ys.end(), cities[i].y) - ys.begin() + 1;
    }

    sort(cities.begin(), cities.end(), [](const City& a, const City& b) {
        if (a.x != b.x) return a.x < b.x;
        return a.y < b.y;
    });

    BIT bit_pre(M);
    int i = 0;
    while (i < n) {
        int j = i;
        while (j < n && cities[j].x == cities[i].x) {
            cities[j].dp_pre = 1 + bit_pre.query(cities[j].compressed_y - 1);
            j++;
        }
        for (int k = i; k < j; k++) {
            bit_pre.update(cities[k].compressed_y, cities[k].dp_pre);
        }
        i = j;
    }

    sort(cities.begin(), cities.end(), [](const City& a, const City& b) {
        if (a.x != b.x) return a.x > b.x;
        return a.y < b.y;
    });

    BIT bit_suf(M);
    i = 0;
    while (i < n) {
        int j = i;
        while (j < n && cities[j].x == cities[i].x) {
            cities[j].dp_suf = 1 + bit_suf.query(M - cities[j].compressed_y);
            j++;
        }
        for (int k = i; k < j; k++) {
            bit_suf.update(M - cities[k].compressed_y + 1, cities[k].dp_suf);
        }
        i = j;
    }

    int max_L = 0;
    for (int k = 0; k < n; k++) {
        max_L = max(max_L, cities[k].dp_pre);
    }

    vector<int> list_a;
    vector<int> counts(n + 1, 0);

    for (int k = 0; k < n; k++) {
        if (cities[k].dp_pre + cities[k].dp_suf - 1 == max_L) {
            list_a.push_back(cities[k].id);
            counts[cities[k].dp_pre]++;
        }
    }

    vector<int> list_b;
    for (int k = 0; k < n; k++) {
        if (cities[k].dp_pre + cities[k].dp_suf - 1 == max_L) {
            if (counts[cities[k].dp_pre] == 1) {
                list_b.push_back(cities[k].id);
            }
        }
    }

    sort(list_a.begin(), list_a.end());
    sort(list_b.begin(), list_b.end());

    cout << list_a.size();
    for (int id : list_a) cout << " " << id;
    cout << "\n";

    cout << list_b.size();
    for (int id : list_b) cout << " " << id;
    cout << "\n";

    return 0;
}
