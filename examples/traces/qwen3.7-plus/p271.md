#include <iostream>
#include <vector>
#include <deque>
#include <string>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M, K;
    if (!(cin >> N >> M >> K)) return 0;

    deque<string> top_k;
    vector<string> initial_bottom;
    vector<string> fallen_bottom;
    bool rev = false;

    for (int i = 0; i < N; ++i) {
        string S;
        cin >> S;
        if (K == 0) {
            initial_bottom.push_back(S);
        } else if (i < K) {
            top_k.push_back(S);
        } else {
            initial_bottom.push_back(S);
        }
    }

    for (int i = 0; i < M; ++i) {
        string op;
        cin >> op;
        if (op[0] == 'A') {
            string S = op.substr(4, op.length() - 5);
            if (K == 0) {
                fallen_bottom.push_back(S);
            } else if (top_k.size() < (size_t)K) {
                if (rev) {
                    top_k.push_back(S);
                } else {
                    top_k.push_front(S);
                }
            } else {
                if (rev) {
                    string out_book = top_k.front();
                    top_k.pop_front();
                    top_k.push_back(S);
                    fallen_bottom.push_back(out_book);
                } else {
                    string out_book = top_k.back();
                    top_k.pop_back();
                    top_k.push_front(S);
                    fallen_bottom.push_back(out_book);
                }
            }
        } else if (op == "ROTATE") {
            if (K > 0) {
                rev = !rev;
            }
        }
    }

    if (rev) {
        for (auto it = top_k.rbegin(); it != top_k.rend(); ++it) {
            cout << *it << "\n";
        }
    } else {
        for (const auto& S : top_k) {
            cout << S << "\n";
        }
    }

    for (auto it = fallen_bottom.rbegin(); it != fallen_bottom.rend(); ++it) {
        cout << *it << "\n";
    }

    for (const auto& S : initial_bottom) {
        cout << S << "\n";
    }

    return 0;
}
