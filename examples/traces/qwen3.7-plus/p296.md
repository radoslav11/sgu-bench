```cpp
#include <iostream>
#include <string>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    string n;
    int k;
    if (cin >> n >> k) {
        size_t keep = n.length() - k;
        string res = "";
        int rem = k;
        
        for (char c : n) {
            while (rem > 0 && !res.empty() && res.back() < c) {
                res.pop_back();
                rem--;
            }
            if (res.length() < keep) {
                res.push_back(c);
            } else {
                rem--;
            }
        }
        
        cout << res << "\n";
    }
    
    return 0;
}
```
