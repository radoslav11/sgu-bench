```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <random>
#include <algorithm>
#include <cstdio>

using namespace std;

int n, m;
bool adj_mat[10][10];
double best_x[10], best_y[10];
double global_best_cost = 1e18;

mt19937_64 rng(1337);

struct PairInfo {
    int i, j;
};
vector<PairInfo> edge_pairs;
vector<PairInfo> non_edge_pairs;

inline double get_cost(const double* x, const double* y) {
    double c = 0;
    for (const auto& p : edge_pairs) {
        double dx = x[p.i] - x[p.j];
        double dy = y[p.i] - y[p.j];
        double d2 = dx * dx + dy * dy;
        double diff = d2 - 1.0;
        c += diff * diff * 2500.0;
    }
    for (const auto& p : non_edge_pairs) {
        double dx = x[p.i] - x[p.j];
        double dy = y[p.i] - y[p.j];
        double d2 = dx * dx + dy * dy;
        if (d2 < 0.0001) {
            double d = sqrt(d2);
            c += 10000.0 * (0.01 - d) * (0.01 - d);
        } else if (d2 > 0.9801 && d2 < 1.0201) {
            double d = sqrt(d2);
            c += 10000.0 * (d - 1.0) * (d - 1.0);
        }
    }
    return c;
}

void solve() {
    if (m == 0) {
        cout << "Yes\n";
        for (int i = 0; i < n; ++i) {
            printf("%.10f %.10f\n", (double)(i * 2), 0.0);
        }
        return;
    }

    int u = -1, v = -1;
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            if (adj_mat[i][j]) {
                u = i; v = j;
                break;
            }
        }
        if (u != -1) break;
    }

    vector<int> comp(n, -1);
    int comp_id = 0;
    int u_comp = -1;
    
    for (int i = 0; i < n; ++i) {
        if (comp[i] == -1) {
            vector<int> q;
            q.push_back(i);
            comp[i] = comp_id;
            int head = 0;
            while (head < (int)q.size()) {
                int curr = q[head++];
                for (int j = 0; j < n; ++j) {
                    if (adj_mat[curr][j] && comp[j] == -1) {
                        comp[j] = comp_id;
                        q.push_back(j);
                    }
                }
            }
            if (comp[u] == comp_id) {
                u_comp = comp_id;
            }
            comp_id++;
        }
    }

    vector<int> f_x, f_y;
    for (int i = 0; i < n; ++i) {
        if (i != u && i != v) {
            f_x.push_back(i);
            f_y.push_back(i);
        }
    }
    int K = f_x.size() + f_y.size();

    uniform_real_distribution<double> dist_unif(0.0, 1.0);
    
    int restarts = 150;
    int steps_per_restart = 200000;

    for (int r = 0; r < restarts; ++r) {
        double x[10], y[10];
        for (int i = 0; i < n; ++i) {
            x[i] = 0.0; y[i] = 0.0;
        }
        x[u] = 0.0; y[u] = 0.0;
        x[v] = 1.0; y[v] = 0.0;
        
        for (int i = 0; i < n; ++i) {
            if (i == u || i == v) continue;
            if (comp[i] == u_comp) {
                x[i] = (dist_unif(rng) * 2.0 - 1.0) * 3.0;
                y[i] = (dist_unif(rng) * 2.0 - 1.0) * 3.0;
            } else {
                x[i] = 10.0 * comp[i] + (dist_unif(rng) * 2.0 - 1.0) * 2.0;
                y[i] = (dist_unif(rng) * 2.0 - 1.0) * 2.0;
            }
        }

        double current_cost = get_cost(x, y);
        double best_cost_r = current_cost;
        double best_x_r[10], best_y_r[10];
        for (int i = 0; i < n; ++i) { best_x_r[i] = x[i]; best_y_r[i] = y[i]; }

        double T = 1.0;
        for (int step = 0; step < steps_per_restart; ++step) {
            T = 1.0 - (double)step / steps_per_restart;
            T = T * T;
            if (T < 1e-9) T = 1e-9;

            int idx = rng() % K;
            int vi;
            bool is_x;
            if (idx < (int)f_x.size()) {
                vi = f_x[idx];
                is_x = true;
            } else {
                vi = f_y[idx - f_x.size()];
                is_x = false;
            }

            double old_val = is_x ? x[vi] : y[vi];
            double delta = (dist_unif(rng) * 2.0 - 1.0) * T * 0.5;
            if (is_x) x[vi] += delta;
            else y[vi] += delta;

            double new_cost = get_cost(x, y);
            if (new_cost < best_cost_r) {
                best_cost_r = new_cost;
                for (int i = 0; i < n; ++i) { best_x_r[i] = x[i]; best_y_r[i] = y[i]; }
            }

            if (new_cost < current_cost || exp((current_cost - new_cost) / T) > dist_unif(rng)) {
                current_cost = new_cost;
            } else {
                if (is_x) x[vi] = old_val;
                else y[vi] = old_val;
            }
        }
        
        if (best_cost_r < global_best_cost) {
            global_best_cost = best_cost_r;
            for (int i = 0; i < n; ++i) { best_x[i] = best_x_r[i]; best_y[i] = best_y_r[i]; }
        }
    }

    double x[10], y[10];
    for (int i = 0; i < n; ++i) { x[i] = best_x[i]; y[i] = best_y[i]; }
    double current_cost = get_cost(x, y);
    bool improved = true;
    int polish_iter = 0;
    while (improved && polish_iter < 100) {
        improved = false;
        for (int idx = 0; idx < K; ++idx) {
            int vi;
            bool is_x;
            if (idx < (int)f_x.size()) {
                vi = f_x[idx];
                is_x = true;
            } else {
                vi = f_y[idx - f_x.size()];
                is_x = false;
            }
            
            double best_dir = 0;
            double min_c = current_cost;
            double steps[] = {-1e-4, -1e-5, -1e-6, -1e-7, -1e-8, 0.0, 1e-8, 1e-7, 1e-6, 1e-5, 1e-4};
            for (double dir : steps) {
                double old_val = is_x ? x[vi] : y[vi];
                if (is_x) x[vi] = old_val + dir;
                else y[vi] = old_val + dir;
                
                double c = get_cost(x, y);
                if (c < min_c) {
                    min_c = c;
                    best_dir = dir;
                }
                
                if (is_x) x[vi] = old_val;
                else y[vi] = old_val;
            }
            
            if (best_dir != 0) {
                if (is_x) x[vi] += best_dir;
                else y[vi] += best_dir;
                current_cost = min_c;
                improved = true;
            }
        }
        polish_iter++;
    }
    
    if (current_cost < global_best_cost) {
        global_best_cost = current_cost;
        for (int i = 0; i < n; ++i) { best_x[i] = x[i]; best_y[i] = y[i]; }
    }

    if (global_best_cost < 1e-10) {
        cout << "Yes\n";
        for (int i = 0; i < n; ++i) {
            printf("%.10f %.10f\n", best_x[i], best_y[i]);
        }
    } else {
        cout << "No\n";
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    if (!(cin >> n >> m)) return 0;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            adj_mat[i][j] = false;
        }
    }
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj_mat[u][v] = true;
        adj_mat[v][u] = true;
    }
    
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            if (adj_mat[i][j]) {
                edge_pairs.push_back({i, j});
            } else {
                non_edge_pairs.push_back({i, j});
            }
        }
    }
    
    solve();
    return 0;
}
```
