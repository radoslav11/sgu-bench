#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

long long ceil_div(long long a, long long b) {
    if (a >= 0) return (a + b - 1) / b;
    else return a / b;
}

long long solve_case(long long x, long long y, int d2, int d_start, long long b) {
    int r = (d2 - d_start + 1) % 6;
    if (r < 0) r += 6;
    
    vector<long long> c(6, 0);
    for (int i = 0; i < r; ++i) {
        c[(d_start + i) % 6] = 1;
    }
    
    long long q_min = 0;
    
    auto update_q = [&](long long num) {
        long long q_req = ceil_div(num, 2 * b - 2);
        if (q_req > q_min) q_min = q_req;
    };
    
    update_q(c[0] + c[1] - c[3]*b - c[4]*b - x - y);
    update_q(c[0] + c[5] - c[3]*b - c[2]*b - x);
    update_q(c[3] + c[4] - c[1]*b - c[0]*b + x + y);
    update_q(c[4] + c[5] - c[1]*b - c[2]*b + y);
    update_q(c[2] + c[3] - c[5]*b - c[0]*b + x);
    update_q(c[1] + c[2] - c[5]*b - c[4]*b - y);
    
    long long best_cost = -1;
    
    for (long long q = q_min; q <= q_min + 2; ++q) {
        vector<long long> curr_m(6);
        for (int i = 0; i < 6; ++i) {
            curr_m[i] = q + c[i];
        }
        
        long long cL1 = curr_m[0] - x - curr_m[3]*b;
        long long cU1 = curr_m[0]*b - x - curr_m[3];
        long long cL2 = y + curr_m[4] - curr_m[1]*b;
        long long cU2 = y - curr_m[1] + curr_m[4]*b;
        long long cL3 = curr_m[2] - curr_m[5]*b;
        long long cU3 = curr_m[2]*b - curr_m[5];
        
        long long ct_min = max({cL1, cL2, cL3});
        long long ct_max = min({cU1, cU2, cU3});
        
        if (ct_min > ct_max) continue;
        
        vector<long long> cands = {ct_min, ct_max, curr_m[0] - curr_m[3] - x, curr_m[4] - curr_m[1] + y, curr_m[2] - curr_m[5]};
        for (long long t : cands) {
            if (t >= ct_min && t <= ct_max) {
                long long T1 = max(t + 2*curr_m[3] + x, -t + 2*curr_m[0] - x);
                long long T2 = max(-t + 2*curr_m[4] + y, t + 2*curr_m[1] - y);
                long long T3 = max(t + 2*curr_m[5], -t + 2*curr_m[2]);
                long long cost = T1 + T2 + T3;
                if (best_cost == -1 || cost < best_cost) {
                    best_cost = cost;
                }
            }
        }
    }
    
    return best_cost;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    long long x, y;
    int d2, d1, b;
    if (cin >> x >> y >> d2 >> d1 >> b) {
        if (x == 0 && y == 0 && d1 == d2) {
            cout << 0 << "\n";
        } else {
            long long ans1 = solve_case(x, y, d2, d1, b);
            long long ans2 = solve_case(x, y, d2, (d1 + 1) % 6, b);
            
            long long ans = -1;
            if (ans1 != -1) ans = ans1;
            if (ans2 != -1) {
                if (ans == -1 || ans2 < ans) ans = ans2;
            }
            cout << ans << "\n";
        }
    }
    return 0;
}
