#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int dp[11][91][11][11][11][11][2];
int vis[11][91][11][11][11][11][2];
int query_id = 0;

long long dfs(int k, int D, int m1, int M1, int m2, int M2, int is_less, int N, const string& S) {
    if (k == 2 * N) {
        if (D == 45) return 0; // D == 0 means the number is already lucky, not almost lucky
        int actual_D = D - 45;
        int actual_M1 = M1 - 1;
        int actual_M2 = M2 - 1;
        
        int Min1 = m1 - 9;
        int Max1 = actual_M1;
        int Min2 = -actual_M2;
        int Max2 = 9 - m2;
        
        if (actual_D != 0 && ((Min1 <= actual_D && actual_D <= Max1) || (Min2 <= actual_D && actual_D <= Max2))) {
            return 1;
        }
        return 0;
    }
    
    if (vis[k][D][m1][M1][m2][M2][is_less] == query_id) {
        return dp[k][D][m1][M1][m2][M2][is_less];
    }
    
    long long res = 0;
    int limit = is_less ? 9 : (S[k] - '0');
    
    for (int d = 0; d <= limit; ++d) {
        if (k == 0 && d == 0) continue; // No leading zeroes allowed
        
        int nD = D;
        int nm1 = m1, nM1 = M1, nm2 = m2, nM2 = M2;
        
        if (k < N) {
            nD = D + d;
            nm1 = min(m1, d);
            nM1 = max(M1, d - (k == 0 ? 1 : 0) + 1);
        } else {
            nD = D - d;
            nm2 = min(m2, d);
            nM2 = max(M2, d + 1);
        }
        
        int n_is_less = is_less || (d < limit);
        res += dfs(k + 1, nD, nm1, nM1, nm2, nM2, n_is_less, N, S);
    }
    
    vis[k][D][m1][M1][m2][M2][is_less] = query_id;
    return dp[k][D][m1][M1][m2][M2][is_less] = res;
}

long long count_almost_lucky(string S) {
    if (S == "0") return 0;
    long long ans = 0;
    int len = S.length();
    for (int L = 2; L <= 10; L += 2) {
        if (L < len) {
            query_id++;
            ans += dfs(0, 45, 10, 0, 10, 0, 1, L / 2, S);
        } else if (L == len) {
            query_id++;
            ans += dfs(0, 45, 10, 0, 10, 0, 0, L / 2, S);
        }
    }
    return ans;
}

string decrement(string s) {
    if (s == "0") return "-1";
    int i = s.length() - 1;
    while (i >= 0 && s[i] == '0') {
        s[i] = '9';
        i--;
    }
    if (i >= 0) {
        s[i]--;
    }
    if (s[0] == '0' && s.length() > 1) {
        s = s.substr(1);
    }
    return s;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    string A, B;
    if (cin >> A >> B) {
        if (A == "0") {
            cout << count_almost_lucky(B) << "\n";
        } else {
            cout << count_almost_lucky(B) - count_almost_lucky(decrement(A)) << "\n";
        }
    }
    return 0;
}
