#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int val;
    while (cin >> val) {
        vector<int> grid(16);
        grid[0] = val;
        for (int i = 1; i < 16; ++i) {
            cin >> grid[i];
        }
        
        int inv = 0;
        int blank_row_from_bottom = 0;
        
        for (int i = 0; i < 16; ++i) {
            if (grid[i] == 0) {
                int row_from_top = i / 4;
                blank_row_from_bottom = 4 - row_from_top;
            }
            for (int j = i + 1; j < 16; ++j) {
                if (grid[i] != 0 && grid[j] != 0) {
                    if (grid[i] > grid[j]) {
                        inv++;
                    }
                }
            }
        }
        
        if ((inv % 2) != (blank_row_from_bottom % 2)) {
            cout << "YES\n";
        } else {
            cout << "NO\n";
        }
    }
    
    return 0;
}
