#include <iostream>
#include <vector>

using namespace std;

int K;
long long N;
int A[32];
int solution[32];
long long count_solutions = 0;
bool found = false;

vector<int> group_cells[28];
vector<int> cell_groups[32];
uint32_t group_mask[28];

void add_group(int g, const vector<int>& cells) {
    for (int c : cells) {
        cell_groups[c].push_back(g);
        group_cells[g].push_back(c);
    }
}

void init() {
    add_group(0, {1, 2});
    add_group(1, {3, 4, 5, 6, 7});
    add_group(2, {8, 9, 10, 11, 12, 13});
    add_group(3, {14, 15, 16, 17, 18});
    add_group(4, {19, 20, 21, 22, 23, 24});
    add_group(5, {25, 26, 27, 28, 29});
    add_group(6, {30, 31});

    add_group(7, {3, 8});
    add_group(8, {1, 4, 9, 14, 19});
    add_group(9, {2, 5, 10, 15, 20, 25});
    add_group(10, {6, 11, 16, 21, 26});
    add_group(11, {7, 12, 17, 22, 27, 30});
    add_group(12, {13, 18, 23, 28, 31});
    add_group(13, {24, 29});

    add_group(14, {7, 13});
    add_group(15, {2, 6, 12, 18, 24});
    add_group(16, {1, 5, 11, 17, 23, 29});
    add_group(17, {4, 10, 16, 22, 28});
    add_group(18, {3, 9, 15, 21, 27, 31});
    add_group(19, {8, 14, 20, 26, 30});
    add_group(20, {19, 25});

    add_group(21, {1, 2, 4, 5, 6, 10, 11});
    add_group(22, {3, 4, 8, 9, 10, 14, 15});
    add_group(23, {6, 7, 11, 12, 13, 17, 18});
    add_group(24, {10, 11, 15, 16, 17, 21, 22});
    add_group(25, {14, 15, 19, 20, 21, 25, 26});
    add_group(26, {17, 18, 22, 23, 24, 28, 29});
    add_group(27, {21, 22, 26, 27, 28, 30, 31});
}

void dfs(int u) {
    if (found) return;
    if (u == 32) {
        count_solutions++;
        if (count_solutions == N) {
            found = true;
            for (int i = 1; i <= 31; i++) {
                solution[i] = A[i];
            }
        }
        return;
    }

    uint32_t avail = (uint32_t)((1ULL << (K + 1)) - 2);
    for (int g : cell_groups[u]) {
        avail &= ~group_mask[g];
    }

    if (A[u] != 0) {
        int v = A[u];
        uint32_t bit = (1 << v);
        if (avail & bit) {
            for (int g : cell_groups[u]) group_mask[g] |= bit;
            dfs(u + 1);
            for (int g : cell_groups[u]) group_mask[g] &= ~bit;
        }
        return;
    }

    while (avail) {
        uint32_t bit = avail & -avail;
        avail ^= bit;
        
        for (int g : cell_groups[u]) group_mask[g] |= bit;
        dfs(u + 1);
        for (int g : cell_groups[u]) group_mask[g] &= ~bit;
        
        if (found) return;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    if (!(cin >> K >> N)) return 0;
    for (int i = 1; i <= 31; i++) {
        cin >> A[i];
    }
    
    init();
    
    bool initial_ok = true;
    for (int i = 1; i <= 31; i++) {
        if (A[i] != 0) {
            uint32_t bit = (1 << A[i]);
            for (int g : cell_groups[i]) {
                if (group_mask[g] & bit) {
                    initial_ok = false;
                    break;
                }
            }
            if (!initial_ok) break;
            for (int g : cell_groups[i]) {
                group_mask[g] |= bit;
            }
        }
    }

    if (!initial_ok) {
        cout << "No way\n";
        return 0;
    }

    dfs(1);

    if (found) {
        cout << "Found\n";
        for (int i = 1; i <= 31; i++) {
            cout << solution[i] << (i == 31 ? "" : " ");
        }
        cout << "\n";
    } else {
        cout << "No way\n";
    }

    return 0;
}
