```cpp
#include <iostream>
#include <vector>
#include <string>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    string s;
    if (!getline(cin, s)) return 0;
    
    if (!s.empty() && s.back() == '\r') {
        s.pop_back();
    }

    vector<string> lines;
    lines.push_back("");
    int row = 0;
    int col = 0;

    for (char c : s) {
        if (c == 'Q') {
            break;
        } else if (c >= 'a' && c <= 'z' || c == ' ') {
            while (lines[row].length() < (size_t)col) {
                lines[row] += ' ';
            }
            lines[row].insert(lines[row].begin() + col, c);
            col++;
        } else if (c == 'L') {
            if (col > 0) {
                col--;
            }
        } else if (c == 'R') {
            col++;
        } else if (c == 'U') {
            if (row > 0) {
                row--;
            }
        } else if (c == 'D') {
            if (row + 1 < (int)lines.size()) {
                row++;
            }
        } else if (c == 'N') {
            string right = (col <= (int)lines[row].length()) ? lines[row].substr(col) : "";
            if (col <= (int)lines[row].length()) {
                lines[row] = lines[row].substr(0, col);
            }
            lines.insert(lines.begin() + row + 1, right);
            row++;
            col = 0;
        } else if (c == 'E') {
            col = lines[row].length();
        } else if (c == 'H') {
            col = 0;
        } else if (c == 'X') {
            if (col > (int)lines[row].length()) {
                while (lines[row].length() < (size_t)col) {
                    lines[row] += ' ';
                }
                if (row + 1 < (int)lines.size()) {
                    lines[row] += lines[row + 1];
                    lines.erase(lines.begin() + row + 1);
                }
            } else if (col < (int)lines[row].length()) {
                lines[row].erase(col, 1);
            }
        } else if (c == 'B') {
            if (col > (int)lines[row].length()) {
                col--;
            } else if (col == 0) {
                if (row > 0) {
                    int prev_len = lines[row - 1].length();
                    lines[row - 1] += lines[row];
                    lines.erase(lines.begin() + row);
                    row--;
                    col = prev_len;
                }
            } else {
                lines[row].erase(col - 1, 1);
                col--;
            }
        }
    }

    for (size_t i = 0; i < lines.size(); ++i) {
        if (i > 0) {
            cout << "\n";
        }
        cout << lines[i];
    }
    cout << "\n";

    return 0;
}
```
