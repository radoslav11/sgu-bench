#include <iostream>

using namespace std;

bool is_prime(long long n) {
    if (n < 2) return false;
    if (n == 2) return true;
    if (n % 2 == 0) return false;
    for (long long i = 3; i * i <= n; i += 2) {
        if (n % i == 0) return false;
    }
    return true;
}

bool is_nearly_prime(long long n) {
    if (n < 4) return false;
    for (long long i = 2; i * i <= n; ++i) {
        if (n % i == 0) {
            return is_prime(n / i);
        }
    }
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int N;
    if (cin >> N) {
        while (N--) {
            long long x;
            cin >> x;
            if (is_nearly_prime(x)) {
                cout << "Yes\n";
            } else {
                cout << "No\n";
            }
        }
    }
    return 0;
}
