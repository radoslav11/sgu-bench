#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int INF = 1e9;

int N;
int parent_node[500005];

int head[500005];
int to_edge[500005];
int nxt_edge[500005];
int edge_cnt = 0;

void add_edge(int u, int v) {
    to_edge[edge_cnt] = v;
    nxt_edge[edge_cnt] = head[u];
    head[u] = edge_cnt++;
}

int S[500005];
int max_gain[500005];
int best_child[500005];
int dp[500005][2];
bool matched_with_parent[500005];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> N)) return 0;

    for (int i = 1; i <= N; ++i) {
        head[i] = -1;
        max_gain[i] = -INF;
        best_child[i] = -1;
        S[i] = 0;
        dp[i][0] = 0;
        dp[i][1] = -INF;
    }

    for (int i = 2; i <= N; ++i) {
        cin >> parent_node[i];
        add_edge(parent_node[i], i);
    }

    for (int v = N; v >= 2; --v) {
        dp[v][0] = S[v];
        if (max_gain[v] != -INF) {
            dp[v][1] = S[v] + max_gain[v];
        } else {
            dp[v][1] = -INF;
        }

        int p = parent_node[v];
        int max_v = max(dp[v][0], dp[v][1]);
        S[p] += max_v;
        
        int gain = 1 + dp[v][0] - max_v;
        if (gain > max_gain[p]) {
            max_gain[p] = gain;
            best_child[p] = v;
        }
    }
    
    dp[1][0] = S[1];
    if (max_gain[1] != -INF) {
        dp[1][1] = S[1] + max_gain[1];
    } else {
        dp[1][1] = -INF;
    }

    vector<int> receivers;
    for (int u = 1; u <= N; ++u) {
        if (matched_with_parent[u]) {
            for (int e = head[u]; e != -1; e = nxt_edge[e]) {
                int v = to_edge[e];
                matched_with_parent[v] = false;
            }
        } else {
            if (dp[u][1] > dp[u][0]) {
                int v = best_child[u];
                receivers.push_back(v);
                for (int e = head[u]; e != -1; e = nxt_edge[e]) {
                    int w = to_edge[e];
                    if (w == v) {
                        matched_with_parent[w] = true;
                    } else {
                        matched_with_parent[w] = false;
                    }
                }
            } else {
                for (int e = head[u]; e != -1; e = nxt_edge[e]) {
                    int v = to_edge[e];
                    matched_with_parent[v] = false;
                }
            }
        }
    }

    sort(receivers.begin(), receivers.end());

    long long total_money = (long long)receivers.size() * 1000LL;
    cout << total_money << "\n";
    for (size_t i = 0; i < receivers.size(); ++i) {
        cout << receivers[i] << (i + 1 == receivers.size() ? "" : " ");
    }
    cout << "\n";

    return 0;
}
