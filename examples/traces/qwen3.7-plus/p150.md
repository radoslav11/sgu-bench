#include <iostream>

using namespace std;

long long get_abs(long long x) {
    return x > 0 ? x : -x;
}

long long gcd(long long a, long long b) {
    while (b) {
        long long temp = a % b;
        a = b;
        b = temp;
    }
    return a;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long long x1, y1, x2, y2, n;
    if (!(cin >> x1 >> y1 >> x2 >> y2 >> n)) return 0;

    long long dx = x2 - x1;
    long long dy = y2 - y1;

    if (dx == 0 || dy == 0) {
        cout << "no solution\n";
        return 0;
    }

    long long abs_dx = get_abs(dx);
    long long abs_dy = get_abs(dy);

    long long max_cells = abs_dx + abs_dy - gcd(abs_dx, abs_dy);

    if (n > max_cells) {
        cout << "no solution\n";
        return 0;
    }

    long long sx = (dx > 0) ? 1 : -1;
    long long sy = (dy > 0) ? 1 : -1;

    long long X = x1 + (sx > 0 ? 0 : -1);
    long long Y = y1 + (sy > 0 ? 0 : -1);

    long long kx = 0;
    long long ky = 0;

    for (long long i = 0; i < n - 1; ++i) {
        long long Vx = (kx + 1) * abs_dy;
        long long Vy = (ky + 1) * abs_dx;

        if (Vx < Vy) {
            X += sx;
            kx++;
        } else if (Vx > Vy) {
            Y += sy;
            ky++;
        } else {
            X += sx;
            Y += sy;
            kx++;
            ky++;
        }
    }

    cout << X << " " << Y << "\n";

    return 0;
}
