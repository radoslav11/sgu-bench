#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>

using namespace std;

struct Event {
    int x;
    int type; // 0 for RIGHT, 1 for LEFT
    int id;
    bool operator<(const Event& other) const {
        if (x != other.x) return x < other.x;
        return type < other.type;
    }
};

struct Rect {
    int xmin, xmax, ymin, ymax;
    long long area;
};

int N;
long long W, H;
vector<Rect> rects;
vector<Event> events;
vector<int> Y;

const int MAX_NODES = 500005;
priority_queue<pair<int, int>> active_pq[MAX_NODES];
priority_queue<pair<int, int>> deleted_pq[MAX_NODES];

void update(int node, int l, int r, int ql, int qr, int y_min, int id, bool add) {
    if (ql <= l && r <= qr) {
        if (add) {
            active_pq[node].push({y_min, id});
        } else {
            deleted_pq[node].push({y_min, id});
        }
        return;
    }
    int mid = l + (r - l) / 2;
    if (ql <= mid) update(2 * node, l, mid, ql, qr, y_min, id, add);
    if (qr > mid) update(2 * node + 1, mid + 1, r, ql, qr, y_min, id, add);
}

pair<int, int> query(int node, int l, int r, int p) {
    pair<int, int> res = {-1, -1};
    
    while (!active_pq[node].empty() && !deleted_pq[node].empty() && 
           active_pq[node].top() == deleted_pq[node].top()) {
        active_pq[node].pop();
        deleted_pq[node].pop();
    }
    
    if (!active_pq[node].empty()) {
        res = active_pq[node].top();
    }
    
    if (l == r) return res;
    
    int mid = l + (r - l) / 2;
    if (p <= mid) {
        pair<int, int> left_res = query(2 * node, l, mid, p);
        if (left_res.first > res.first) res = left_res;
    } else {
        pair<int, int> right_res = query(2 * node + 1, mid + 1, r, p);
        if (right_res.first > res.first) res = right_res;
    }
    return res;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> N >> W >> H)) return 0;

    rects.resize(N + 1);
    rects[0] = {0, (int)W, 0, (int)H, (long long)W * H};
    Y.push_back(0);
    Y.push_back(H);

    for (int i = 1; i <= N; ++i) {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        int xmin = min(x1, x2);
        int xmax = max(x1, x2);
        int ymin = min(y1, y2);
        int ymax = max(y1, y2);
        rects[i] = {xmin, xmax, ymin, ymax, (long long)(xmax - xmin) * (ymax - ymin)};
        Y.push_back(ymin);
        Y.push_back(ymax);
    }

    sort(Y.begin(), Y.end());
    Y.erase(unique(Y.begin(), Y.end()), Y.end());
    int M = Y.size();

    for (int i = 0; i <= N; ++i) {
        events.push_back({rects[i].xmin, 1, i});
        events.push_back({rects[i].xmax, 0, i});
    }

    sort(events.begin(), events.end());

    vector<int> parent(N + 1, -1);

    for (const auto& ev : events) {
        if (ev.type == 1) { // LEFT
            if (ev.id != 0) {
                int L = lower_bound(Y.begin(), Y.end(), rects[ev.id].ymin) - Y.begin();
                int R = lower_bound(Y.begin(), Y.end(), rects[ev.id].ymax) - Y.begin();
                int p_idx = L + 1;
                pair<int, int> p_data = query(1, 1, M - 1, p_idx);
                parent[ev.id] = p_data.second;
            }
            int L = lower_bound(Y.begin(), Y.end(), rects[ev.id].ymin) - Y.begin();
            int R = lower_bound(Y.begin(), Y.end(), rects[ev.id].ymax) - Y.begin();
            update(1, 1, M - 1, L + 1, R, rects[ev.id].ymin, ev.id, true);
        } else { // RIGHT
            int L = lower_bound(Y.begin(), Y.end(), rects[ev.id].ymin) - Y.begin();
            int R = lower_bound(Y.begin(), Y.end(), rects[ev.id].ymax) - Y.begin();
            update(1, 1, M - 1, L + 1, R, rects[ev.id].ymin, ev.id, false);
        }
    }

    vector<long long> facet_areas(N + 1, 0);
    for (int i = 0; i <= N; ++i) {
        facet_areas[i] = rects[i].area;
    }

    for (int i = 1; i <= N; ++i) {
        facet_areas[parent[i]] -= rects[i].area;
    }

    sort(facet_areas.begin(), facet_areas.end());

    for (int i = 0; i <= N; ++i) {
        cout << facet_areas[i] << (i == N ? "" : " ");
    }
    cout << "\n";

    return 0;
}
