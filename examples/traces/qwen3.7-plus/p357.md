```cpp
#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int b[10];
    int up, down, dash;

    if (!(cin >> b[1] >> b[2] >> b[3] >> up)) return 0;
    cin >> b[4] >> b[5] >> b[6] >> down;
    cin >> b[7] >> b[8] >> b[9];
    cin >> dash >> b[0];

    int X, Y;
    cin >> X >> Y;

    vector<int> dist(100, 1e9);
    dist[X] = 0;
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    pq.push({0, X});

    for (int v = 0; v <= 9; ++v) {
        if (b[v]) {
            if (dist[v] > 1) {
                dist[v] = 1;
                pq.push({1, v});
            }
        }
    }

    for (int v = 10; v <= 99; ++v) {
        if (dash && b[v / 10] && b[v % 10]) {
            if (dist[v] > 3) {
                dist[v] = 3;
                pq.push({3, v});
            }
        }
    }

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dist[u]) continue;

        if (up) {
            int nxt = (u + 1) % 100;
            if (dist[nxt] > d + 1) {
                dist[nxt] = d + 1;
                pq.push({dist[nxt], nxt});
            }
        }
        if (down) {
            int nxt = (u + 99) % 100;
            if (dist[nxt] > d + 1) {
                dist[nxt] = d + 1;
                pq.push({dist[nxt], nxt});
            }
        }
    }

    if (dist[Y] == 1e9) {
        cout << -1 << "\n";
    } else {
        cout << dist[Y] << "\n";
    }

    return 0;
}
```
