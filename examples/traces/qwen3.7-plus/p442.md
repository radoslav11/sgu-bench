using namespace std;

struct BigInt {
    vector<int> d;
    BigInt() { d.push_back(0); }
    BigInt(long long v) {
        if (v == 0) d.push_back(0);
        else {
            while (v > 0) {
                d.push_back(v % 10);
                v /= 10;
            }
        }
    }
};

void add(BigInt& a, const BigInt& b) {
    int carry = 0;
    int n = max(a.d.size(), b.d.size());
    for (int i = 0; i < n || carry; ++i) {
        if (i == (int)a.d.size()) a.d.push_back(0);
        int sum = a.d[i] + carry + (i < (int)b.d.size() ? b.d[i] : 0);
        a.d[i] = sum % 10;
        carry = sum / 10;
    }
}

BigInt mul(const BigInt& a, long long v) {
    if (v == 0 || (a.d.size() == 1 && a.d[0] == 0)) return BigInt(0LL);
    BigInt res;
    res.d.clear();
    long long carry = 0;
    for (int i = 0; i < (int)a.d.size() || carry; ++i) {
        if (i == (int)res.d.size()) res.d.push_back(0);
        long long cur = (i < (int)a.d.size() ? a.d[i] : 0) * v + carry;
        res.d[i] = cur % 10;
        carry = cur / 10;
    }
    while (res.d.size() > 1 && res.d.back() == 0) res.d.pop_back();
    return res;
}

string to_string(const BigInt& a) {
    if (a.d.empty()) return "0";
    string res;
    for (int i = (int)a.d.size() - 1; i >= 0; --i) {
        res += (char)('0' + a.d[i]);
    }
    return res;
}
