#include <iostream>
#include <vector>

using namespace std;

int n;
vector<vector<int>> adj;
vector<vector<bool>> edge_used;
vector<vector<int>> ans;
vector<bool> visited;

void dfs(int u, int p, int c) {
    visited[u] = true;
    int next_c = 3 - c;
    for (int v : adj[u]) {
        if (edge_used[u][v]) continue;
        edge_used[u][v] = true;
        edge_used[v][u] = true;
        ans[u][v] = next_c;
        ans[v][u] = next_c;
        
        next_c = 3 - next_c;
        
        if (!visited[v]) {
            dfs(v, u, ans[u][v]);
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> n)) return 0;
    
    adj.resize(n + 1);
    edge_used.assign(n + 1, vector<bool>(n + 1, false));
    ans.assign(n + 1, vector<int>(n + 1, 0));
    visited.assign(n + 1, false);
    
    for (int i = 1; i <= n; ++i) {
        int v;
        while (cin >> v && v != 0) {
            adj[i].push_back(v);
        }
    }
    
    for (int i = 1; i <= n; ++i) {
        if (!visited[i] && !adj[i].empty()) {
            vector<int> comp;
            vector<bool> comp_vis(n + 1, false);
            vector<int> q;
            q.push_back(i);
            comp_vis[i] = true;
            
            int head = 0;
            while (head < (int)q.size()) {
                int u = q[head++];
                comp.push_back(u);
                for (int v : adj[u]) {
                    if (!comp_vis[v]) {
                        comp_vis[v] = true;
                        q.push_back(v);
                    }
                }
            }
            
            bool all_deg_2 = true;
            int start_node = -1;
            int deg1_node = -1;
            
            for (int u : comp) {
                if (adj[u].size() != 2) {
                    all_deg_2 = false;
                }
                if (adj[u].size() >= 3) {
                    start_node = u;
                }
                if (adj[u].size() == 1) {
                    deg1_node = u;
                }
            }
            
            if (all_deg_2) {
                if (comp.size() % 2 != 0) {
                    cout << "No solution\n";
                    return 0;
                } else {
                    start_node = comp[0];
                }
            } else {
                if (start_node == -1) {
                    start_node = deg1_node;
                }
            }
            
            if (start_node != -1) {
                dfs(start_node, -1, 2);
            }
        }
    }
    
    for (int i = 1; i <= n; ++i) {
        bool first = true;
        for (int v : adj[i]) {
            if (!first) cout << " ";
            cout << ans[i][v];
            first = false;
        }
        cout << " 0\n";
    }
    
    return 0;
}
