#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>
#include <random>
#include <chrono>

using namespace std;

int n;

long long get_swap_delta(const vector<int>& arr, int i, int j) {
    if (i == j) return 0;
    
    vector<pair<int, int>> edges;
    auto add_edge = [&](int u, int v) {
        if (u > v) swap(u, v);
        edges.push_back({u, v});
    };
    
    auto get_neighbors = [&](int x) {
        vector<int> res;
        res.push_back((x + 2) % n);
        res.push_back((x - 2 + n) % n);
        if (x % 2 == 0) res.push_back(x + 1);
        else res.push_back(x - 1);
        return res;
    };
    
    for (int v : get_neighbors(i)) add_edge(i, v);
    for (int v : get_neighbors(j)) add_edge(j, v);
    
    sort(edges.begin(), edges.end());
    edges.erase(unique(edges.begin(), edges.end()), edges.end());
    
    long long old_risk = 0;
    long long new_risk = 0;
    
    for (auto& e : edges) {
        long long diff_old = arr[e.first] - arr[e.second];
        old_risk += diff_old * diff_old;
        
        long long val_u = arr[e.first];
        long long val_v = arr[e.second];
        if (e.first == i) val_u = arr[j];
        if (e.first == j) val_u = arr[i];
        if (e.second == i) val_v = arr[j];
        if (e.second == j) val_v = arr[i];
        
        long long diff_new = val_u - val_v;
        new_risk += diff_new * diff_new;
    }
    
    return new_risk - old_risk;
}

long long calc_total_risk(const vector<int>& arr) {
    long long r = 0;
    for (int i = 0; i < n; ++i) {
        long long d1 = arr[i] - arr[(i + 2) % n];
        r += d1 * d1;
        if (i % 2 == 0) {
            long long d2 = arr[i] - arr[i + 1];
            r += d2 * d2;
        }
    }
    return r;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    if (!(cin >> n)) return 0;
    
    mt19937 rng(chrono::steady_clock::now().time_since_epoch().count());
    uniform_int_distribution<int> dist(0, n - 1);
    uniform_int_distribution<int> dist2(0, 1);
    
    long long best_risk = -1;
    vector<int> best_a;
    
    int num_restarts = 200;
    
    for (int restart = 0; restart < num_restarts; ++restart) {
        vector<int> cur_a(n);
        
        if (restart % 2 == 0) {
            iota(cur_a.begin(), cur_a.end(), 1);
            shuffle(cur_a.begin(), cur_a.end(), rng);
        } else {
            vector<int> O(n / 2);
            for (int i = 0; i < n / 2; ++i) {
                if (dist2(rng) == 0) {
                    O[i] = i + 1;
                } else {
                    O[i] = n - i;
                }
            }
            shuffle(O.begin(), O.end(), rng);
            vector<int> sorted_O = O;
            sort(sorted_O.begin(), sorted_O.end());
            vector<int> arranged_O(n / 2);
            int left = 0, right = n / 2 - 1;
            for (int i = 0; i < n / 2; ++i) {
                if (i % 2 == 0) arranged_O[i] = sorted_O[left++];
                else arranged_O[i] = sorted_O[right--];
            }
            for (int i = 0; i < n / 2; ++i) {
                cur_a[2 * i] = arranged_O[i];
                cur_a[2 * i + 1] = n + 1 - arranged_O[i];
            }
        }
        
        long long cur_risk = calc_total_risk(cur_a);
        int stagnation = 0;
        
        while (stagnation < 5000) {
            int i = dist(rng);
            int j = dist(rng);
            if (i == j) continue;
            
            long long delta = get_swap_delta(cur_a, i, j);
            if (delta > 0) {
                swap(cur_a[i], cur_a[j]);
                cur_risk += delta;
                stagnation = 0;
            } else {
                stagnation++;
            }
        }
        
        if (cur_risk > best_risk) {
            best_risk = cur_risk;
            best_a = cur_a;
        }
    }
    
    for (int i = 0; i < n; ++i) {
        cout << best_a[i] << (i == n - 1 ? "" : " ");
    }
    cout << "\n";
    
    return 0;
}
