#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

bool cmp(const string& a, const string& b) {
    return a + b < b + a;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    if (cin >> n) {
        vector<string> s(n);
        for (int i = 0; i < n; ++i) {
            cin >> s[i];
        }
        
        sort(s.begin(), s.end(), cmp);
        
        for (int i = 0; i < n; ++i) {
            cout << s[i];
        }
        cout << "\n";
    }
    
    return 0;
}
