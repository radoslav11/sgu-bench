#include <iostream>
#include <vector>
#include <array>
#include <iomanip>
#include <algorithm>

using namespace std;

struct BigInt {
    vector<int> digits; // base 10^9

    BigInt() {
        digits.push_back(0);
    }

    BigInt(long long v) {
        if (v == 0) {
            digits.push_back(0);
        } else {
            while (v > 0) {
                digits.push_back(v % 1000000000);
                v /= 1000000000;
            }
        }
    }

    void multiply(long long v) {
        if (v == 0) {
            digits = {0};
            return;
        }
        if (digits.size() == 1 && digits[0] == 0) return;
        long long carry = 0;
        for (size_t i = 0; i < digits.size(); ++i) {
            long long cur = (long long)digits[i] * v + carry;
            digits[i] = cur % 1000000000;
            carry = cur / 1000000000;
        }
        while (carry > 0) {
            digits.push_back(carry % 1000000000);
            carry /= 1000000000;
        }
    }

    void add(const BigInt& other) {
        long long carry = 0;
        size_t n = max(digits.size(), other.digits.size());
        digits.resize(n, 0);
        for (size_t i = 0; i < n; ++i) {
            long long cur = digits[i] + carry;
            if (i < other.digits.size()) cur += other.digits[i];
            digits[i] = cur % 1000000000;
            carry = cur / 1000000000;
        }
        if (carry > 0) digits.push_back(carry);
    }

    void subtract(const BigInt& other) {
        long long borrow = 0;
        for (size_t i = 0; i < digits.size(); ++i) {
            long long cur = digits[i] - borrow;
            if (i < other.digits.size()) cur -= other.digits[i];
            if (cur < 0) {
                digits[i] = cur + 1000000000;
                borrow = 1;
            } else {
                digits[i] = cur;
                borrow = 0;
            }
        }
        while (digits.size() > 1 && digits.back() == 0) {
            digits.pop_back();
        }
    }

    void print() const {
        if (digits.empty()) {
            cout << 0 << "\n";
            return;
        }
        cout << digits.back();
        for (int i = (int)digits.size() - 2; i >= 0; --i) {
            cout << setfill('0') << setw(9) << digits[i];
        }
        cout << "\n";
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, K;
    if (!(cin >> N >> K)) return 0;

    vector<array<int, 3>> triple(K);
    for (int i = 0; i < K; ++i) {
        cin >> triple[i][0] >> triple[i][1] >> triple[i][2];
    }

    vector<BigInt> W(N + 1);
    W[0] = BigInt(1);
    if (N >= 1) {
        W[1] = BigInt(1);
    }
    for (int i = 2; i <= N; ++i) {
        W[i] = W[i - 1];
        long long mult = (long long)(3 * i - 1) * (3 * i - 2) / 2;
        W[i].multiply(mult);
    }

    vector<int> C_arr(K + 1, 0);
    vector<int> seen(3 * N + 1, 0);
    int current_version = 0;

    int num_masks = 1 << K;
    for (int mask = 0; mask < num_masks; ++mask) {
        current_version++;
        bool ok = true;
        int s = 0;
        for (int i = 0; i < K; ++i) {
            if ((mask >> i) & 1) {
                s++;
                for (int v : triple[i]) {
                    if (seen[v] == current_version) {
                        ok = false;
                        break;
                    }
                    seen[v] = current_version;
                }
                if (!ok) break;
            }
        }
        if (ok) {
            C_arr[s]++;
        }
    }

    BigInt sum_pos(0);
    BigInt sum_neg(0);

    for (int s = 0; s <= K; ++s) {
        if (C_arr[s] == 0) continue;
        if (N - s < 0) continue;
        BigInt term = W[N - s];
        term.multiply(C_arr[s]);
        if (s % 2 == 0) {
            sum_pos.add(term);
        } else {
            sum_neg.add(term);
        }
    }

    sum_pos.subtract(sum_neg);
    sum_pos.print();

    return 0;
}
