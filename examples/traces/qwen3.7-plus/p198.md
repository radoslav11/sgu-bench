#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <queue>

using namespace std;

struct Point {
    double x, y;
};

struct Circle {
    double x, y, r, r_eps_sq;
};

void get_intersections(const Circle& c1, const Circle& c2, vector<Point>& out) {
    double dx = c2.x - c1.x;
    double dy = c2.y - c1.y;
    double d2 = dx * dx + dy * dy;
    double d = sqrt(d2);
    
    if (d > c1.r + c2.r + 1e-6) return;
    if (d < abs(c1.r - c2.r) - 1e-6) return;
    if (d < 1e-9) return;
    
    double a = (c1.r * c1.r - c2.r * c2.r + d2) / (2.0 * d);
    double h2 = c1.r * c1.r - a * a;
    if (h2 < 0) h2 = 0;
    double h = sqrt(h2);
    
    double xm = c1.x + a * dx / d;
    double ym = c1.y + a * dy / d;
    
    out.push_back({xm + h * (-dy) / d, ym + h * dx / d});
    if (h > 1e-6) {
        out.push_back({xm - h * (-dy) / d, ym - h * dx / d});
    }
}

bool valid_segment(const Point& p1, const Point& p2, const vector<Circle>& circles) {
    double dx = p2.x - p1.x;
    double dy = p2.y - p1.y;
    double l2 = dx * dx + dy * dy;
    if (l2 < 1e-12) return true;
    
    double min_x = min(p1.x, p2.x) - 1e-6;
    double max_x = max(p1.x, p2.x) + 1e-6;
    double min_y = min(p1.y, p2.y) - 1e-6;
    double max_y = max(p1.y, p2.y) + 1e-6;
    
    for (const auto& c : circles) {
        if (c.x + c.r < min_x || c.x - c.r > max_x || c.y + c.r < min_y || c.y - c.r > max_y) continue;
        
        double t = (c.x - p1.x) * dx + (c.y - p1.y) * dy;
        if (t < 0) {
            if ((p1.x - c.x) * (p1.x - c.x) + (p1.y - c.y) * (p1.y - c.y) < c.r_eps_sq) return false;
        } else if (t > l2) {
            if ((p2.x - c.x) * (p2.x - c.x) + (p2.y - c.y) * (p2.y - c.y) < c.r_eps_sq) return false;
        } else {
            double tx = t / l2;
            double px = p1.x + tx * dx;
            double py = p1.y + tx * dy;
            if ((px - c.x) * (px - c.x) + (py - c.y) * (py - c.y) < c.r_eps_sq) return false;
        }
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N;
    if (!(cin >> N)) return 0;
    
    vector<Circle> circles(N);
    for (int i = 0; i < N; ++i) {
        cin >> circles[i].x >> circles[i].y >> circles[i].r;
    }
    
    Point P;
    double ship_r;
    cin >> P.x >> P.y >> ship_r;
    
    for (int i = 0; i < N; ++i) {
        circles[i].r += ship_r;
        double safe_r = max(0.0, circles[i].r - 1e-6);
        circles[i].r_eps_sq = safe_r * safe_r;
    }
    
    for (int i = 0; i < N; ++i) {
        double dist_p = hypot(P.x - circles[i].x, P.y - circles[i].y);
        if (dist_p < circles[i].r - 1e-6) {
            cout << "NO\n";
            return 0;
        }
    }
    
    vector<Point> V;
    
    for (int i = 0; i < N; ++i) {
        for (int j = i + 1; j < N; ++j) {
            vector<Point> inters;
            get_intersections(circles[i], circles[j], inters);
            for (const auto& p : inters) {
                bool valid = true;
                for (int k = 0; k < N; ++k) {
                    if ((p.x - circles[k].x) * (p.x - circles[k].x) + (p.y - circles[k].y) * (p.y - circles[k].y) < circles[k].r_eps_sq) {
                        valid = false;
                        break;
                    }
                }
                if (valid) {
                    V.push_back(p);
                }
            }
        }
    }
    
    for (int i = 0; i < N; ++i) {
        for (int k = 0; k < 18; ++k) {
            double angle = k * (2.0 * acos(-1.0) / 18.0);
            Point p = {circles[i].x + circles[i].r * cos(angle), circles[i].y + circles[i].r * sin(angle)};
            bool valid = true;
            for (int j = 0; j < N; ++j) {
                if ((p.x - circles[j].x) * (p.x - circles[j].x) + (p.y - circles[j].y) * (p.y - circles[j].y) < circles[j].r_eps_sq) {
                    valid = false;
                    break;
                }
            }
            if (valid) {
                V.push_back(p);
            }
        }
    }
    
    if (V.size() > 0) {
        sort(V.begin(), V.end(), [](const Point& a, const Point& b) {
            if (abs(a.x - b.x) > 1e-5) return a.x < b.x;
            return a.y < b.y;
        });
        
        vector<Point> unique_V;
        for (const auto& p : V) {
            if (unique_V.empty() || hypot(p.x - unique_V.back().x, p.y - unique_V.back().y) > 1e-5) {
                unique_V.push_back(p);
            }
        }
        V = move(unique_V);
    }
    
    V.push_back(P);
    int P_idx = V.size() - 2;
    
    Point Q = {200000.0, 200000.0};
    V.push_back(Q);
    int Q_idx = V.size() - 1;
    
    vector<vector<int>> adj(V.size());
    
    const double PI = acos(-1.0);
    for (int i = 0; i < N; ++i) {
        vector<pair<double, int>> angles;
        for (size_t k = 0; k < V.size(); ++k) {
            double dist = hypot(V[k].x - circles[i].x, V[k].y - circles[i].y);
            if (abs(dist - circles[i].r) < 1e-4) {
                angles.push_back({atan2(V[k].y - circles[i].y, V[k].x - circles[i].x), (int)k});
            }
        }
        if (angles.size() < 2) continue;
        
        sort(angles.begin(), angles.end());
        
        for (size_t j = 0; j < angles.size(); ++j) {
            int u = angles[j].second;
            int v = angles[(j + 1) % angles.size()].second;
            double ang_u = angles[j].first;
            double ang_v = angles[(j + 1) % angles.size()].first;
            
            double diff = ang_v - ang_u;
            if (diff < 0) diff += 2.0 * PI;
            double mid_ang = ang_u + diff / 2.0;
            if (mid_ang >= 2.0 * PI) mid_ang -= 2.0 * PI;
            
            Point mid_pt = {circles[i].x + circles[i].r * cos(mid_ang), circles[i].y + circles[i].r * sin(mid_ang)};
            
            bool valid_arc = true;
            for (int k = 0; k < N; ++k) {
                if (k == i) continue;
                if ((mid_pt.x - circles[k].x) * (mid_pt.x - circles[k].x) + (mid_pt.y - circles[k].y) * (mid_pt.y - circles[k].y) < circles[k].r_eps_sq) {
                    valid_arc = false;
                    break;
                }
            }
            if (valid_arc) {
                adj[u].push_back(v);
                adj[v].push_back(u);
            }
        }
    }
    
    for (size_t i = 0; i < V.size(); ++i) {
        for (size_t j = i + 1; j < V.size(); ++j) {
            if (valid_segment(V[i], V[j], circles)) {
                adj[i].push_back(j);
                adj[j].push_back(i);
            }
        }
    }
    
    vector<bool> visited(V.size(), false);
    queue<int> q;
    q.push(P_idx);
    visited[P_idx] = true;
    
    while (!q.empty()) {
        int curr = q.front();
        q.pop();
        
        if (curr == Q_idx) {
            cout << "YES\n";
            return 0;
        }
        
        for (int nxt : adj[curr]) {
            if (!visited[nxt]) {
                visited[nxt] = true;
                q.push(nxt);
            }
        }
    }
    
    cout << "NO\n";
    return 0;
}
