#include <iostream>
#include <vector>

using namespace std;

const int INF = 1e9;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, k;
    if (!(cin >> n >> k)) return 0;

    long long m_ll = 1LL * n * k / 2;
    int m = (int)m_ll;

    vector<int> head(n + 1, -1);
    vector<int> to_edge(2 * m);
    vector<int> nxt_edge(2 * m);
    int edge_cnt = 0;

    auto add_edge = [&](int u, int v) {
        to_edge[edge_cnt] = v;
        nxt_edge[edge_cnt] = head[u];
        head[u] = edge_cnt++;
    };

    for (int i = 1; i <= m; ++i) {
        int u, v;
        cin >> u >> v;
        add_edge(u, v);
        add_edge(v, u);
    }

    vector<char> used_edge(m + 1, 0);

    struct StackFrame {
        int u;
        int edge_id;
    };
    vector<StackFrame> stk;
    stk.reserve(n + 1);

    vector<int> match_head(n + 1, -1);
    vector<int> match_to(m);
    vector<int> match_id(m);
    vector<int> match_nxt(m);
    int match_edge_cnt = 0;

    auto add_match_edge = [&](int u, int v, int id) {
        match_to[match_edge_cnt] = v;
        match_id[match_edge_cnt] = id;
        match_nxt[match_edge_cnt] = match_head[u];
        match_head[u] = match_edge_cnt++;
    };

    for (int i = 1; i <= n; ++i) {
        if (head[i] == -1) continue;
        stk.push_back({i, -1});
        while (!stk.empty()) {
            int u = stk.back().u;
            bool found = false;
            while (head[u] != -1) {
                int e_idx = head[u];
                head[u] = nxt_edge[e_idx];
                int v = to_edge[e_idx];
                int id = (e_idx / 2) + 1;
                if (!used_edge[id]) {
                    used_edge[id] = 1;
                    stk.push_back({v, id});
                    found = true;
                    break;
                }
            }
            if (!found) {
                int edge_id = stk.back().edge_id;
                stk.pop_back();
                if (edge_id != -1) {
                    int prev_u = stk.back().u;
                    add_match_edge(prev_u, u, edge_id);
                }
            }
        }
    }

    vector<int> matchL(n + 1, 0);
    vector<int> matchR(n + 1, 0);
    vector<int> matchL_id(n + 1, 0);
    vector<int> dist_match(n + 1, 0);

    auto bfs = [&]() {
        vector<int> q(n + 1);
        int head_q = 0, tail_q = 0;
        for (int i = 1; i <= n; ++i) {
            if (matchL[i] == 0) {
                dist_match[i] = 0;
                q[tail_q++] = i;
            } else {
                dist_match[i] = INF;
            }
        }
        dist_match[0] = INF;
        
        while (head_q < tail_q) {
            int u = q[head_q++];
            if (dist_match[u] < dist_match[0]) {
                for (int e = match_head[u]; e != -1; e = match_nxt[e]) {
                    int v = match_to[e];
                    if (dist_match[matchR[v]] == INF) {
                        dist_match[matchR[v]] = dist_match[u] + 1;
                        q[tail_q++] = matchR[v];
                    }
                }
            }
        }
        return dist_match[0] != INF;
    };

    auto dfs_hk = [&](auto& self, int u) -> bool {
        if (u != 0) {
            for (int e = match_head[u]; e != -1; e = match_nxt[e]) {
                int v = match_to[e];
                int id = match_id[e];
                if (dist_match[matchR[v]] == dist_match[u] + 1) {
                    if (self(self, matchR[v])) {
                        matchR[v] = u;
                        matchL[u] = v;
                        matchL_id[u] = id;
                        return true;
                    }
                }
            }
            dist_match[u] = INF;
            return false;
        }
        return true;
    };

    int matches = 0;
    while (bfs()) {
        for (int i = 1; i <= n; ++i) {
            if (matchL[i] == 0 && dfs_hk(dfs_hk, i)) {
                matches++;
            }
        }
    }

    if (matches != n) {
        cout << "NO\n";
    } else {
        cout << "YES\n";
        for (int i = 1; i <= n; ++i) {
            cout << matchL_id[i] << "\n";
        }
    }

    return 0;
}
