#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <random>

using namespace std;

long long power(long long base, long long exp, long long mod) {
    long long res = 1;
    base %= mod;
    while (exp > 0) {
        if (exp % 2 == 1) res = (res * base) % mod;
        base = (base * base) % mod;
        exp /= 2;
    }
    return res;
}

int modInverse(int a, int m) {
    int m0 = m, t, q;
    int x0 = 0, x1 = 1;
    if (m == 1) return 0;
    while (a > 1) {
        q = a / m;
        t = m;
        m = a % m;
        a = t;
        t = x0;
        x0 = x1 - q * x0;
        x1 = t;
    }
    if (x1 < 0) x1 += m0;
    return x1;
}

int get_primitive_root(int p) {
    if (p == 2) return 1;
    int phi = p - 1;
    vector<int> factors;
    int temp = phi;
    for (int i = 2; i * i <= temp; ++i) {
        if (temp % i == 0) {
            factors.push_back(i);
            while (temp % i == 0) temp /= i;
        }
    }
    if (temp > 1) factors.push_back(temp);
    
    for (int g = 2; g < p; ++g) {
        bool ok = true;
        for (int q : factors) {
            if (power(g, phi / q, p) == 1) {
                ok = false;
                break;
            }
        }
        if (ok) return g;
    }
    return -1;
}

int gcd(int a, int b) {
    while (b) {
        a %= b;
        swap(a, b);
    }
    return a;
}

vector<int> val_to_x(1000005, -1);
vector<int> used_A;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    mt19937 rng(1337);
    
    int t;
    if (!(cin >> t)) return 0;
    while (t--) {
        long long n, p;
        cin >> n >> p;
        
        if (p == 2) {
            cout << -1 << "\n";
            continue;
        }
        
        int k = gcd((int)n, (int)(p - 1));
        int V = (p - 1) / k;
        
        if (V <= 5000) {
            int n_prime = n / k;
            int inv_n = modInverse(n_prime, V);
            int g = get_primitive_root((int)p);
            
            vector<int> pow_g(V + 1);
            pow_g[0] = 1;
            for (int i = 1; i <= V; ++i) {
                pow_g[i] = (1LL * pow_g[i - 1] * g) % p;
            }
            
            used_A.clear();
            vector<int> S;
            for (int j = 0; j < V; ++j) {
                long long exp = (1LL * j * k) % (p - 1);
                int A_j = (int)power(g, exp, p);
                int X_j = (1LL * j * inv_n) % V;
                int x_j = pow_g[X_j];
                
                if (val_to_x[A_j] == -1) {
                    val_to_x[A_j] = x_j;
                    used_A.push_back(A_j);
                    S.push_back(A_j);
                }
            }
            
            bool found = false;
            int ans_x = -1, ans_y = -1, ans_z = -1;
            
            for (int A : S) {
                for (int B : S) {
                    int C = A + B;
                    if (C >= (int)p) C -= p;
                    if (C != 0 && val_to_x[C] != -1) {
                        ans_x = val_to_x[A];
                        ans_y = val_to_x[B];
                        ans_z = val_to_x[C];
                        found = true;
                        break;
                    }
                }
                if (found) break;
            }
            
            for (int A : used_A) {
                val_to_x[A] = -1;
            }
            
            if (found) {
                cout << ans_x << " " << ans_y << " " << ans_z << "\n";
            } else {
                cout << -1 << "\n";
            }
        } else {
            int trials = 0;
            int max_trials = 10000;
            bool found = false;
            
            while (trials < max_trials) {
                int x = uniform_int_distribution<int>(1, (int)p - 1)(rng);
                int y = uniform_int_distribution<int>(1, (int)p - 1)(rng);
                long long xn = power(x, n, p);
                long long yn = power(y, n, p);
                long long u = (xn + yn) % p;
                
                if (u == 0) {
                    trials++;
                    continue;
                }
                
                if (power(u, (p - 1) / k, p) == 1) {
                    int g = get_primitive_root((int)p);
                    long long h = power(g, n, p);
                    long long h_inv = power(h, p - 2, p);
                    
                    int M = sqrt(V) + 1;
                    vector<pair<int, int>> table;
                    table.reserve(M);
                    long long cur = u;
                    for (int b = 0; b < M; ++b) {
                        table.push_back({(int)cur, b});
                        cur = (cur * h_inv) % p;
                    }
                    sort(table.begin(), table.end());
                    
                    long long hm = power(h, M, p);
                    long long current = 1;
                    int ans_X = -1;
                    for (int a = 0; a <= M; ++a) {
                        auto it = lower_bound(table.begin(), table.end(), make_pair((int)current, -1));
                        if (it != table.end() && it->first == current) {
                            ans_X = a * M + it->second;
                            break;
                        }
                        current = (current * hm) % p;
                    }
                    
                    if (ans_X != -1) {
                        int z = (int)power(g, ans_X, p);
                        cout << x << " " << y << " " << z << "\n";
                        found = true;
                        break;
                    }
                }
                trials++;
            }
            
            if (!found) {
                cout << -1 << "\n";
            }
        }
    }
    return 0;
}
