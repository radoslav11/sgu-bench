```cpp
#include <iostream>
#include <vector>
#include <string>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M, K;
    if (!(cin >> N >> M >> K)) return 0;

    vector<string> symbols(K, string(N * M, ' '));
    for (int i = 0; i < K; ++i) {
        for (int r = 0; r < N; ++r) {
            string row;
            cin >> row;
            for (int c = 0; c < M; ++c) {
                symbols[i][r * M + c] = row[c];
            }
        }
    }

    int P = K * (K - 1) / 2;
    int num_states = 1 << P;
    
    // dp[state] will store the minimum number of pixels needed to achieve the state
    vector<vector<int>> dp(2, vector<int>(num_states, 1000000000));
    vector<vector<int>> parent_pixel(2, vector<int>(num_states, -1));
    vector<vector<int>> parent_mask(2, vector<int>(num_states, -1));

    dp[0][0] = 0;

    vector<pair<int, int>> pairs;
    for (int i = 0; i < K; ++i) {
        for (int j = i + 1; j < K; ++j) {
            pairs.push_back({i, j});
        }
    }

    for (int i = 0; i < N * M; ++i) {
        int cur = i % 2;
        int nxt = (i + 1) % 2;
        
        dp[nxt] = dp[cur];
        parent_pixel[nxt] = parent_pixel[cur];
        parent_mask[nxt] = parent_mask[cur];

        int m = 0;
        for (int p = 0; p < P; ++p) {
            int u = pairs[p].first;
            int v = pairs[p].second;
            if (symbols[u][i] != symbols[v][i]) {
                m |= (1 << p);
            }
        }

        if (m == 0) continue;

        for (int mask = 0; mask < num_states; ++mask) {
            if (dp[cur][mask] != 1000000000) {
                int nmask = mask | m;
                if (dp[cur][mask] + 1 < dp[nxt][nmask]) {
                    dp[nxt][nmask] = dp[cur][mask] + 1;
                    parent_pixel[nxt][nmask] = i;
                    parent_mask[nxt][nmask] = mask;
                }
            }
        }
    }

    int final_mask = (1 << P) - 1;
    int final_state = (N * M) % 2;
    int ans = dp[final_state][final_mask];
    
    vector<int> chosen_pixels;
    int curr_state = final_state;
    int curr_mask = final_mask;
    
    while (curr_mask > 0) {
        int p = parent_pixel[curr_state][curr_mask];
        if (p == -1) break; // Safety break
        chosen_pixels.push_back(p);
        curr_mask = parent_mask[curr_state][curr_mask];
        curr_state = p % 2;
    }

    cout << ans << "\n";
    vector<string> result(N, string(M, '0'));
    for (int p : chosen_pixels) {
        result[p / M][p % M] = '1';
    }

    for (int r = 0; r < N; ++r) {
        cout << result[r] << "\n";
    }

    return 0;
}
```
