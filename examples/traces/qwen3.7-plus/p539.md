```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    vector<pair<int, int>> a(n);
    for (int i = 0; i < n; ++i) {
        cin >> a[i].first;
        a[i].second = i;
    }

    vector<pair<int, int>> sorted_a = a;
    stable_sort(sorted_a.begin(), sorted_a.end());

    vector<int> target(n);
    for (int i = 0; i < n; ++i) {
        target[sorted_a[i].second] = i;
    }

    vector<bool> visited(n, false);
    vector<vector<int>> cycles;

    for (int i = 0; i < n; ++i) {
        if (!visited[i]) {
            vector<int> cycle;
            int curr = i;
            while (!visited[curr]) {
                visited[curr] = true;
                cycle.push_back(curr);
                curr = target[curr];
            }
            cycles.push_back(cycle);
        }
    }

    vector<pair<int, int>> T1, T2;
    for (const auto& c : cycles) {
        int L = c.size();
        if (L == 1) continue;
        if (L == 2) {
            T1.push_back({c[0], c[1]});
        } else {
            if (L % 2 != 0) {
                for (int k = 1; k <= (L - 1) / 2; ++k) {
                    T1.push_back({c[k], c[L - k]});
                }
                T2.push_back({c[0], c[1]});
                for (int k = 2; k <= (L - 1) / 2; ++k) {
                    T2.push_back({c[k], c[L - k + 1]});
                }
            } else {
                for (int k = 1; k <= L / 2 - 1; ++k) {
                    T1.push_back({c[k], c[L - k]});
                }
                T2.push_back({c[0], c[1]});
                for (int k = 2; k <= L / 2; ++k) {
                    T2.push_back({c[k], c[L - k + 1]});
                }
            }
        }
    }

    if (T1.empty() && T2.empty()) {
        cout << 0 << "\n";
    } else if (T2.empty()) {
        cout << 1 << "\n";
        cout << T1.size();
        for (auto& p : T1) {
            cout << " " << p.first + 1 << " " << p.second + 1;
        }
        cout << "\n";
    } else {
        cout << 2 << "\n";
        cout << T1.size();
        for (auto& p : T1) {
            cout << " " << p.first + 1 << " " << p.second + 1;
        }
        cout << "\n";
        cout << T2.size();
        for (auto& p : T2) {
            cout << " " << p.first + 1 << " " << p.second + 1;
        }
        cout << "\n";
    }

    return 0;
}
```
