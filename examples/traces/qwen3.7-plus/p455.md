#include <iostream>

using namespace std;

long long f(long long x, long long A, long long B, long long C) {
    return (long long)(((__int128_t)A * x + B) % C);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long long A, B, C;
    if (!(cin >> A >> B >> C)) return 0;

    long long x0 = 1;
    
    // Brent's cycle-finding algorithm
    long long power = 1;
    long long lambda = 1;
    long long tortoise = x0;
    long long hare = f(x0, A, B, C);
    long long steps = 1;
    
    while (tortoise != hare) {
        if (power == lambda) {
            tortoise = hare;
            power *= 2;
            lambda = 0;
        }
        hare = f(hare, A, B, C);
        lambda++;
        steps++;
        
        // If we exceed a safe bound, the answer is definitely > 2,000,000
        if (steps > 5000000) {
            cout << -1 << "\n";
            return 0;
        }
    }
    
    // Find mu
    tortoise = x0;
    hare = x0;
    for (long long i = 0; i < lambda; ++i) {
        hare = f(hare, A, B, C);
    }
    
    long long mu = 0;
    while (tortoise != hare) {
        tortoise = f(tortoise, A, B, C);
        hare = f(hare, A, B, C);
        mu++;
    }
    
    long long ans = mu + lambda;
    if (ans <= 2000000) {
        cout << ans << "\n";
    } else {
        cout << -1 << "\n";
    }
    
    return 0;
}
