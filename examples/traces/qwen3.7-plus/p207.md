#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Robber {
    int id;
    long long rem;
    bool operator<(const Robber& other) const {
        if (rem != other.rem) {
            return rem > other.rem; // Descending order of remainder
        }
        return id < other.id;       // Ascending order of id for stability
    }
};

int main() {
    // Optimize standard I/O operations for competitive programming
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    long long m, y;
    if (!(cin >> n >> m >> y)) return 0;

    vector<long long> x(n);
    vector<long long> k(n);
    vector<Robber> robbers(n);

    long long sum_k = 0;
    for (int i = 0; i < n; ++i) {
        cin >> x[i];
        // Initial fair share without fractional part
        k[i] = (x[i] * m) / y;
        sum_k += k[i];
        
        robbers[i].id = i;
        robbers[i].rem = (x[i] * m) % y;
    }

    // Number of coins left to distribute
    int rem_coins = m - sum_k;

    // Sort robbers by remainder descending to minimize the additional unfairness
    sort(robbers.begin(), robbers.end());

    // Distribute the remaining coins to those with the largest remainders
    for (int i = 0; i < rem_coins; ++i) {
        k[robbers[i].id]++;
    }

    // Output the result
    for (int i = 0; i < n; ++i) {
        cout << k[i] << (i == n - 1 ? "" : " ");
    }
    cout << "\n";

    return 0;
}
