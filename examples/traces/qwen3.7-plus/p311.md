#include <iostream>
#include <string>

using namespace std;

const int MAX_C = 1000000;
const int MAX_NODES = 2097155; // 2 * 1048576

struct Node {
    long long count;
    long long sum;
} tree[MAX_NODES];

bool lazy_tree[MAX_NODES];

void push_up(int node) {
    tree[node].count = tree[2 * node].count + tree[2 * node + 1].count;
    tree[node].sum = tree[2 * node].sum + tree[2 * node + 1].sum;
}

void push_down(int node) {
    if (lazy_tree[node]) {
        lazy_tree[2 * node] = true;
        tree[2 * node].count = 0;
        tree[2 * node].sum = 0;
        
        lazy_tree[2 * node + 1] = true;
        tree[2 * node + 1].count = 0;
        tree[2 * node + 1].sum = 0;
        
        lazy_tree[node] = false;
    }
}

void add(int node, int L, int R, int price, long long count, long long price_val) {
    if (lazy_tree[node]) {
        if (L != R) {
            push_down(node);
        } else {
            lazy_tree[node] = false;
        }
    }
    
    if (L == R) {
        tree[node].count += count;
        tree[node].sum += count * price_val;
        return;
    }
    
    int mid = L + (R - L) / 2;
    if (price <= mid) {
        add(2 * node, L, mid, price, count, price_val);
    } else {
        add(2 * node + 1, mid + 1, R, price, count, price_val);
    }
    push_up(node);
}

long long get_sum(int node, int L, int R, long long n) {
    if (n == 0) return 0;
    if (tree[node].count <= n) {
        return tree[node].sum;
    }
    if (L == R) {
        return n * (long long)L;
    }
    push_down(node);
    int mid = L + (R - L) / 2;
    if (tree[2 * node].count >= n) {
        return get_sum(2 * node, L, mid, n);
    } else {
        return tree[2 * node].sum + get_sum(2 * node + 1, mid + 1, R, n - tree[2 * node].count);
    }
}

void remove_first(int node, int L, int R, long long n) {
    if (n == 0) return;
    if (tree[node].count <= n) {
        tree[node].count = 0;
        tree[node].sum = 0;
        lazy_tree[node] = true;
        return;
    }
    if (L == R) {
        tree[node].count -= n;
        tree[node].sum -= n * (long long)L;
        return;
    }
    push_down(node);
    int mid = L + (R - L) / 2;
    if (tree[2 * node].count >= n) {
        remove_first(2 * node, L, mid, n);
    } else {
        remove_first(2 * node, L, mid, tree[2 * node].count);
        remove_first(2 * node + 1, mid + 1, R, n - tree[2 * node].count);
    }
    push_up(node);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    string type;
    while (cin >> type) {
        if (type == "ARRIVE") {
            long long n;
            int c;
            cin >> n >> c;
            add(1, 1, MAX_C, c, n, c);
        } else if (type == "BUY") {
            long long n, t;
            cin >> n >> t;
            if (tree[1].count < n) {
                cout << "UNHAPPY\n";
            } else {
                long long cost = get_sum(1, 1, MAX_C, n);
                if (cost <= t) {
                    cout << "HAPPY\n";
                    remove_first(1, 1, MAX_C, n);
                } else {
                    cout << "UNHAPPY\n";
                }
            }
        }
    }
    
    return 0;
}
