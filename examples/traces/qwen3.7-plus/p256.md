```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int M, N;
    if (!(cin >> M >> N)) return 0;

    vector<int> A(N), B(N);
    int valid_N = 0;
    for (int i = 0; i < N; ++i) {
        int a, b;
        cin >> a >> b;
        if (a > 0) {
            A[valid_N] = a;
            B[valid_N] = b;
            valid_N++;
        }
    }
    N = valid_N;

    if (M == 0) {
        cout << 0 << "\n";
        return 0;
    }
    
    if (N == 0) {
        cout << -1 << "\n";
        return 0;
    }

    vector<int> pow5(N + 1, 1);
    for (int i = 0; i < N; ++i) {
        pow5[i + 1] = pow5[i] * 5;
    }
    int max_state = pow5[N];

    vector<short> dp(max_state, -1);
    vector<unsigned short> last_updated(max_state, 0);
    
    vector<int> current_states;
    current_states.push_back(0);
    dp[0] = 0;
    last_updated[0] = 1;

    for (int t = 1; t <= 1000; ++t) {
        vector<int> next_states;
        next_states.reserve(current_states.size() * (N + 1));

        for (int s : current_states) {
            short bal = dp[s];
            
            int ns = 0;
            int temp = s;
            for(int j = 0; j < N; ++j) {
                int r = temp % 5;
                if (r > 0) ns += (r - 1) * pow5[j];
                temp /= 5;
            }
            
            if (dp[ns] < bal) {
                if (dp[ns] == -1 || last_updated[ns] != t) {
                    next_states.push_back(ns);
                    last_updated[ns] = t;
                }
                dp[ns] = bal;
            }
            
            temp = s;
            for(int j = 0; j < N; ++j) {
                int r = temp % 5;
                if (r == 0) {
                    int pick_ns = ns + B[j] * pow5[j];
                    short new_bal = bal + A[j];
                    if (new_bal >= M) {
                        cout << t << "\n";
                        return 0;
                    }
                    if (dp[pick_ns] < new_bal) {
                        if (dp[pick_ns] == -1 || last_updated[pick_ns] != t) {
                            next_states.push_back(pick_ns);
                            last_updated[pick_ns] = t;
                        }
                        dp[pick_ns] = new_bal;
                    }
                }
                temp /= 5;
            }
        }
        
        current_states = move(next_states);
        
        if (current_states.empty()) {
            break;
        }
    }
    
    cout << -1 << "\n";
    return 0;
}
```
