#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

const int INF = 1e9;

struct Edge {
    int to;
    int cap;
    int flow;
    int cost;
    int rev;
    int type;
};

vector<vector<Edge>> adj;
vector<int> pot;
vector<int> dist;
vector<int> parent_edge;
vector<int> parent_node;

void add_edge(int from, int to, int cap, int cost, int type) {
    adj[from].push_back({to, cap, 0, cost, (int)adj[to].size(), type});
    adj[to].push_back({from, 0, 0, -cost, (int)adj[from].size() - 1, 0});
}

bool dijkstra(int s, int t) {
    dist.assign(adj.size(), INF);
    parent_edge.assign(adj.size(), -1);
    parent_node.assign(adj.size(), -1);
    
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    
    dist[s] = 0;
    pq.push({0, s});
    
    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();
        
        if (d > dist[u]) continue;
        
        for (int i = 0; i < (int)adj[u].size(); ++i) {
            Edge& e = adj[u][i];
            if (e.cap - e.flow > 0) {
                int reduced_cost = e.cost + pot[u] - pot[e.to];
                if (dist[e.to] > dist[u] + reduced_cost) {
                    dist[e.to] = dist[u] + reduced_cost;
                    parent_node[e.to] = u;
                    parent_edge[e.to] = i;
                    pq.push({dist[e.to], e.to});
                }
            }
        }
    }
    
    if (dist[t] == INF) return false;
    
    for (size_t i = 0; i < adj.size(); ++i) {
        if (dist[i] != INF) {
            pot[i] += dist[i];
        }
    }
    
    return true;
}

int min_cost_max_flow(int s, int t, int target_flow) {
    int flow = 0;
    int cost = 0;
    
    pot.assign(adj.size(), INF);
    pot[s] = 0;
    queue<int> q;
    vector<bool> in_queue(adj.size(), false);
    q.push(s);
    in_queue[s] = true;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        in_queue[u] = false;
        for (auto& e : adj[u]) {
            if (e.cap - e.flow > 0 && pot[e.to] > pot[u] + e.cost) {
                pot[e.to] = pot[u] + e.cost;
                if (!in_queue[e.to]) {
                    q.push(e.to);
                    in_queue[e.to] = true;
                }
            }
        }
    }
    
    while (flow < target_flow && dijkstra(s, t)) {
        int push = target_flow - flow;
        int curr = t;
        while (curr != s) {
            int p = parent_node[curr];
            int idx = parent_edge[curr];
            push = min(push, adj[p][idx].cap - adj[p][idx].flow);
            curr = p;
        }
        
        flow += push;
        
        int actual_cost = 0;
        curr = t;
        while (curr != s) {
            int p = parent_node[curr];
            int idx = parent_edge[curr];
            actual_cost += adj[p][idx].cost;
            curr = p;
        }
        cost += push * actual_cost;
        
        curr = t;
        while (curr != s) {
            int p = parent_node[curr];
            int idx = parent_edge[curr];
            int rev_idx = adj[p][idx].rev;
            adj[p][idx].flow += push;
            adj[curr][rev_idx].flow -= push;
            curr = p;
        }
    }
    return cost;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n, m1, m2;
    if (!(cin >> n >> m1 >> m2)) return 0;
    
    vector<pair<int, int>> juan_edges;
    vector<pair<int, int>> rosa_edges;
    vector<vector<int>> adj_undir(n + 1);
    
    for (int i = 0; i < m1; ++i) {
        int u, v;
        cin >> u >> v;
        juan_edges.push_back({u, v});
        adj_undir[u].push_back(v);
        adj_undir[v].push_back(u);
    }
    
    for (int i = 0; i < m2; ++i) {
        int u, v;
        cin >> u >> v;
        rosa_edges.push_back({u, v});
        adj_undir[u].push_back(v);
        adj_undir[v].push_back(u);
    }
    
    vector<int> color(n + 1, -1);
    for (int i = 1; i <= n; ++i) {
        if (color[i] == -1) {
            queue<int> q;
            q.push(i);
            color[i] = 0;
            while (!q.empty()) {
                int u = q.front();
                q.pop();
                for (int v : adj_undir[u]) {
                    if (color[v] == -1) {
                        color[v] = color[u] ^ 1;
                        q.push(v);
                    }
                }
            }
        }
    }
    
    int S = 0;
    int T = 2 * n + 1;
    adj.assign(T + 1, vector<Edge>());
    
    for (int i = 1; i <= n; ++i) {
        add_edge(S, i, 1, 0, 0);
        add_edge(i + n, T, 1, 0, 0);
        add_edge(i, i + n, 1, 0, 0); // self loop
    }
    
    for (auto& edge : juan_edges) {
        int u = edge.first;
        int v = edge.second;
        if (color[u] == 0 && color[v] == 1) {
            add_edge(u, v + n, 1, -1, 1);
        } else if (color[u] == 1 && color[v] == 0) {
            add_edge(v, u + n, 1, -1, 1);
        }
    }
    
    for (auto& edge : rosa_edges) {
        int u = edge.first;
        int v = edge.second;
        if (color[u] == 1 && color[v] == 0) {
            add_edge(u, v + n, 1, -1, 2);
        } else if (color[u] == 0 && color[v] == 1) {
            add_edge(v, u + n, 1, -1, 2);
        }
    }
    
    int cost = min_cost_max_flow(S, T, n);
    int k = -cost / 2;
    
    vector<pair<int, int>> juan_matches;
    vector<pair<int, int>> rosa_matches;
    
    for (int u = 1; u <= n; ++u) {
        for (auto& e : adj[u]) {
            if (e.flow == 1 && e.type != 0) {
                int v = e.to - n;
                if (e.type == 1) {
                    juan_matches.push_back({u, v});
                } else if (e.type == 2) {
                    rosa_matches.push_back({u, v});
                }
            }
        }
    }
    
    sort(juan_matches.begin(), juan_matches.end());
    sort(rosa_matches.begin(), rosa_matches.end());
    
    cout << k << "\n";
    for (auto& p : juan_matches) {
        cout << p.first << " " << p.second << "\n";
    }
    for (auto& p : rosa_matches) {
        cout << p.first << " " << p.second << "\n";
    }
    
    return 0;
}
