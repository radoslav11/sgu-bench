#include <iostream>
#include <vector>
#include <iomanip>
#include <cmath>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int T, N;
    if (!(cin >> T >> N)) return 0;

    vector<double> qty(N);
    for (int i = 0; i < N; ++i) {
        cin >> qty[i];
    }

    vector<vector<double>> price(T + 1, vector<double>(N));
    for (int t = 0; t <= T; ++t) {
        for (int i = 0; i < N; ++i) {
            cin >> price[t][i];
        }
    }

    vector<double> V(N);
    double VP = 0.0;
    for (int i = 0; i < N; ++i) {
        V[i] = qty[i] * price[0][i];
        VP += V[i];
    }

    vector<double> w(N);
    for (int i = 0; i < N; ++i) {
        w[i] = V[i] / VP;
    }

    vector<vector<double>> r(T, vector<double>(N));
    for (int t = 1; t <= T; ++t) {
        for (int i = 0; i < N; ++i) {
            r[t - 1][i] = (price[t - 1][i] - price[t][i]) / price[t][i];
        }
    }

    vector<double> m(N, 0.0);
    for (int i = 0; i < N; ++i) {
        double sum = 0.0;
        for (int t = 0; t < T; ++t) {
            sum += r[t][i];
        }
        m[i] = sum / T;
    }

    vector<vector<double>> cov(N, vector<double>(N, 0.0));
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            double sum = 0.0;
            for (int t = 0; t < T; ++t) {
                sum += (r[t][i] - m[i]) * (r[t][j] - m[j]);
            }
            cov[i][j] = sum / T;
        }
    }

    double mP = 0.0;
    for (int i = 0; i < N; ++i) {
        mP += w[i] * m[i];
    }

    double sP_sq = 0.0;
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            sP_sq += w[i] * w[j] * cov[i][j];
        }
    }
    
    double sP = sqrt(max(0.0, sP_sq));

    double VaR = -VP * (mP - 1.644854 * sP);

    cout << fixed << setprecision(2) << VaR << "\n";

    return 0;
}
