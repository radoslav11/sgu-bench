#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

void print(__int128 n) {
    if (n == 0) {
        cout << 0 << "\n";
        return;
    }
    string s;
    while (n > 0) {
        s += (char)('0' + (n % 10));
        n /= 10;
    }
    reverse(s.begin(), s.end());
    cout << s << "\n";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;

    }

    __int128 base = n + 1;
    int exp = n - 1;

        if (exp % 2 == 1) {
            ans *= base;

}
