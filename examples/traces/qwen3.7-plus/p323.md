#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Edge {
    int u, v, p, c, id;
    bool operator<(const Edge& other) const {
        return p < other.p;
    }
};

struct DSU {
    int parent_node[2005];
    int rank_arr[2005];
    int comp;
    
    void init(int n) {
        for (int i = 1; i <= n; ++i) {
            parent_node[i] = i;
            rank_arr[i] = 0;
        }
        comp = n;
    }
    
    int find(int i) {
        int root = i;
        while (root != parent_node[root]) {
            root = parent_node[root];
        }
        int curr = i;
        while (curr != root) {
            int nxt = parent_node[curr];
            parent_node[curr] = root;
            curr = nxt;
        }
        return root;
    }
    
    bool unite(int i, int j) {
        int root_i = find(i);
        int root_j = find(j);
        if (root_i != root_j) {
            if (rank_arr[root_i] < rank_arr[root_j]) {
                parent_node[root_i] = root_j;
            } else if (rank_arr[root_i] > rank_arr[root_j]) {
                parent_node[root_j] = root_i;
            } else {
                parent_node[root_j] = root_i;
                rank_arr[root_i]++;
            }
            comp--;
            return true;
        }
        return false;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N, M, K;
    if (!(cin >> N >> M >> K)) return 0;
    
    vector<Edge> edges(K);
    for (int i = 0; i < K; ++i) {
        cin >> edges[i].u >> edges[i].v >> edges[i].c >> edges[i].p;
        edges[i].id = i + 1;
    }
    
    sort(edges.begin(), edges.end());
    
    long long best_cost = 3e18;
    int best_R = 1;
    vector<int> best_ops;
    
    DSU dsu;
    vector<int> current_ops;
    current_ops.reserve(N);
    
    for (int R = 1; R <= M; ++R) {
        dsu.init(N);
        long long current_cost = 0;
        current_ops.clear();
        bool possible = true;
        
        for (int k = 0; k < K; ++k) {
            if (dsu.unite(edges[k].u, edges[k].v)) {
                if (edges[k].c != R) {
                    current_cost += edges[k].p;
                    if (current_cost >= best_cost) {
                        possible = false;
                        break;
                    }
                    current_ops.push_back(edges[k].id);
                }
                if (dsu.comp == 1) {
                    break;
                }
            }
        }
        
        if (possible && dsu.comp == 1) {
            best_cost = current_cost;
            best_R = R;
            best_ops = current_ops;
        }
    }
    
    cout << best_cost << " " << best_R << " " << best_ops.size() << "\n";
    for (int id : best_ops) {
        cout << id << "\n";
    }
    
    return 0;
}
