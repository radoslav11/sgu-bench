#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

const int INF = 1e9;

struct Edge {
    int to;
    int cap;
    int flow;
    int rev;
};

struct Dinic {
    int n;
    vector<vector<Edge>> adj;
    vector<int> level;
    vector<int> ptr;

    Dinic(int n) : n(n) {
        adj.resize(n + 1);
        level.resize(n + 1);
        ptr.resize(n + 1);
    }

    void add_edge(int from, int to, int cap) {
        adj[from].push_back({to, cap, 0, (int)adj[to].size()});
        adj[to].push_back({from, 0, 0, (int)adj[from].size() - 1});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        level[s] = 0;
        queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& edge : adj[v]) {
                if (edge.cap - edge.flow > 0 && level[edge.to] == -1) {
                    level[edge.to] = level[v] + 1;
                    q.push(edge.to);
                }
            }
        }
        return level[t] != -1;
    }

    int dfs(int v, int t, int pushed) {
        if (pushed == 0) return 0;
        if (v == t) return pushed;
        for (int& cid = ptr[v]; cid < (int)adj[v].size(); ++cid) {
            auto& edge = adj[v][cid];
            int tr = edge.to;
            if (level[v] + 1 != level[tr] || edge.cap - edge.flow == 0) continue;
            int push = dfs(tr, t, min(pushed, edge.cap - edge.flow));
            if (push == 0) continue;
            edge.flow += push;
            adj[tr][edge.rev].flow -= push;
            return push;
        }
        return 0;
    }

    int max_flow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(ptr.begin(), ptr.end(), 0);
            while (int pushed = dfs(s, t, INF)) {
                flow += pushed;
            }
        }
        return flow;
    }
};

struct TreeEdge {
    int to;
    int id;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    vector<int> u(M + 1), v(M + 1), c(M + 1);
    vector<vector<TreeEdge>> adj(N + 1);

    for (int i = 1; i <= M; ++i) {
        cin >> u[i] >> v[i] >> c[i];
        if (i <= N - 1) {
            adj[u[i]].push_back({v[i], i});
            adj[v[i]].push_back({u[i], i});
        }
    }

    vector<pair<int, int>> constraints;
    for (int i = N; i <= M; ++i) {
        vector<int> parent_node(N + 1, 0);
        vector<int> parent_edge(N + 1, 0);
        queue<int> q;
        vector<bool> vis(N + 1, false);
        
        q.push(u[i]);
        vis[u[i]] = true;
        
        while (!q.empty()) {
            int curr = q.front();
            q.pop();
            if (curr == v[i]) break;
            
            for (auto& edge : adj[curr]) {
                int nxt = edge.to;
                if (!vis[nxt]) {
                    vis[nxt] = true;
                    parent_node[nxt] = curr;
                    parent_edge[nxt] = edge.id;
                    q.push(nxt);
                }
            }
        }
        
        int curr = v[i];
        while (curr != u[i]) {
            constraints.push_back({parent_edge[curr], i});
            curr = parent_node[curr];
        }
    }

    vector<int> unique_c;
    for (int i = 1; i <= M; ++i) {
        unique_c.push_back(c[i]);
    }
    sort(unique_c.begin(), unique_c.end());
    unique_c.erase(unique(unique_c.begin(), unique_c.end()), unique_c.end());

    vector<int> V_test;
    V_test.push_back(1);
    for (int uc : unique_c) {
        V_test.push_back(uc + 1);
    }
    sort(V_test.begin(), V_test.end());
    V_test.erase(unique(V_test.begin(), V_test.end()), V_test.end());

    vector<int> d(M + 1, 0);
    int S = 0;
    int T_node = M + 1;

    vector<bool> vis_dinic(M + 2, false);
    queue<int> q_dinic;

    for (size_t k = 0; k + 1 < V_test.size(); ++k) {
        int v = V_test[k];
        int v_next = V_test[k + 1];
        int len = v_next - v;

        Dinic dinic(M + 2);

        for (int i = 1; i <= M; ++i) {
            if (i <= N - 1) {
                if (v <= c[i]) {
                    dinic.add_edge(S, i, 1);
                } else {
                    dinic.add_edge(i, T_node, 1);
                }
            } else {
                if (v > c[i]) {
                    dinic.add_edge(i, T_node, 1);
                }
            }
        }

        for (auto& cons : constraints) {
            dinic.add_edge(cons.first, cons.second, INF);
        }

        dinic.max_flow(S, T_node);

        fill(vis_dinic.begin(), vis_dinic.end(), false);
        while (!q_dinic.empty()) q_dinic.pop();
        
        q_dinic.push(S);
        vis_dinic[S] = true;
        while (!q_dinic.empty()) {
            int curr = q_dinic.front();
            q_dinic.pop();
            for (auto& edge : dinic.adj[curr]) {
                if (edge.cap - edge.flow > 0 && !vis_dinic[edge.to]) {
                    vis_dinic[edge.to] = true;
                    q_dinic.push(edge.to);
                }
            }
        }

        for (int i = 1; i <= M; ++i) {
            if (vis_dinic[i]) {
                d[i] += len;
            }
        }
    }

    for (int i = 1; i <= M; ++i) {
        cout << d[i] << "\n";
    }

    return 0;
}
