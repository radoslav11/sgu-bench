#include <iostream>

using namespace std;

int main() {
    // Optimize standard I/O operations for competitive programming
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n, m;
    if (cin >> n >> m) {
        long long total_sum = 0;
        for (int i = 0; i < m; ++i) {
            long long s;
            cin >> s;
            total_sum += s;
        }
        
        // The total amount paid by all friends is total_sum.
        // To minimize the maximum difference between any two friends, 
        // the amounts paid must be as evenly distributed as possible.
        // This means everyone pays either floor(total_sum / n) or 
        // floor(total_sum / n) + 1 rubles.
        // The number of friends who pay the extra 1 ruble is exactly 
        // the remainder of the total sum divided by n.
        cout << total_sum % n << "\n";
    }
    
    return 0;
}
