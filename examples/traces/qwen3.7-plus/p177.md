#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int MAXN = 1005;

int parent_node[MAXN][MAXN];
bool is_black[MAXN][MAXN];

int find(int r, int c) {
    int root = c;
    while (parent_node[r][root] != root) {
        root = parent_node[r][root];
    }
    int curr = c;
    while (curr != root) {
        int nxt = parent_node[r][curr];
        parent_node[r][curr] = root;
        curr = nxt;
    }
    return root;
}

struct Operation {
    int x1, y1, x2, y2;
    char c;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    vector<Operation> ops(M);
    for (int i = 0; i < M; ++i) {
        cin >> ops[i].x1 >> ops[i].y1 >> ops[i].x2 >> ops[i].y2 >> ops[i].c;
    }

    for (int i = 1; i <= N; ++i) {
        for (int j = 1; j <= N + 2; ++j) {
            parent_node[i][j] = j;
        }
    }

    for (int i = M - 1; i >= 0; --i) {
        int r1 = min(ops[i].x1, ops[i].x2);
        int r2 = max(ops[i].x1, ops[i].x2);
        int c1 = min(ops[i].y1, ops[i].y2);
        int c2 = max(ops[i].y1, ops[i].y2);
        char color = ops[i].c;

        for (int r = r1; r <= r2; ++r) {
            int c = find(r, c1);
            while (c <= c2) {
                if (color == 'b') {
                    is_black[r][c] = true;
                }
                parent_node[r][c] = find(r, c + 1);
                c = parent_node[r][c];
            }
        }
    }

    int white_count = 0;
    for (int r = 1; r <= N; ++r) {
        for (int c = 1; c <= N; ++c) {
            if (!is_black[r][c]) {
                white_count++;
            }
        }
    }

    cout << white_count << "\n";

    return 0;
}
