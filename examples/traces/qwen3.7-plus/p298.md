#include <iostream>
#include <vector>
#include <deque>
#include <algorithm>

using namespace std;

struct Edge {
    int to;
    int weight;
};

bool spfa(int start, const vector<vector<Edge>>& adj, vector<int>& dist) {
    int n = adj.size() - 1;
    dist.assign(n + 1, 1e9);
    dist[start] = 0;
    deque<int> q;
    q.push_back(start);
    vector<bool> in_q(n + 1, false);
    in_q[start] = true;
    vector<int> cnt(n + 1, 0);
    
    while (!q.empty()) {
        int u = q.front();
        q.pop_front();
        in_q[u] = false;
        
        for (const auto& edge : adj[u]) {
            int v = edge.to;
            int w = edge.weight;
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                if (!in_q[v]) {
                    if (!q.empty() && dist[v] < dist[q.front()]) {
                        q.push_front(v);
                    } else {
                        q.push_back(v);
                    }
                    in_q[v] = true;
                    cnt[v]++;
                    if (cnt[v] > n + 1) {
                        return false;
                    }
                }
            }
        }
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N, M;
    if (!(cin >> N >> M)) return 0;
    
    vector<vector<Edge>> adj(N + 1);
    vector<vector<Edge>> adj2(N + 1);
    vector<vector<Edge>> reach_adj(N + 1);
    
    for (int i = 0; i < M; ++i) {
        int X, Y, C;
        cin >> X >> Y >> C;
        // A_X >= A_Y + C  =>  A_Y - A_X <= -C
        adj[X].push_back({Y, -C});
        
        // For minA, we use negated logic: edge Y -> X with weight -C
        adj2[Y].push_back({X, -C});
        
        // For reachability and longest path from 1
        reach_adj[Y].push_back({X, C});
    }
    
    for (int i = 1; i <= N; ++i) {
        adj[0].push_back({i, 10000});
        adj[i].push_back({0, 10000});
        
        adj2[0].push_back({i, 10000});
        adj2[i].push_back({0, 10000});
    }
    
    vector<int> max_dist;
    if (!spfa(0, adj, max_dist)) {
        cout << -1 << "\n";
        return 0;
    }
    
    vector<int> min_dist;
    if (!spfa(0, adj2, min_dist)) {
        cout << -1 << "\n";
        return 0;
    }
    
    vector<int> minA(N + 1);
    for (int i = 1; i <= N; ++i) {
        minA[i] = -min_dist[i];
    }
    
    vector<int> d(N + 1, -1e9);
    d[1] = 0;
    deque<int> q;
    q.push_back(1);
    vector<bool> in_q(N + 1, false);
    in_q[1] = true;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop_front();
        in_q[u] = false;
        for (const auto& edge : reach_adj[u]) {
            int v = edge.to;
            int w = edge.weight;
            if (d[u] + w > d[v]) {
                d[v] = d[u] + w;
                if (!in_q[v]) {
                    if (!q.empty() && d[v] > d[q.front()]) {
                        q.push_front(v);
                    } else {
                        q.push_back(v);
                    }
                    in_q[v] = true;
                }
            }
        }
    }
    
    int min_shift = -10000;
    int max_shift = 10000;
    
    for (int i = 1; i <= N; ++i) {
        if (d[i] > -1e8) { // reachable from 1
            min_shift = max(min_shift, -10000 - d[i]);
            max_shift = min(max_shift, 10000 - d[i]);
        }
    }
    
    for (int u = 1; u <= N; ++u) {
        if (d[u] > -1e8) continue;
        for (const auto& edge : reach_adj[u]) {
            int v = edge.to;
            int w = edge.weight;
            if (d[v] > -1e8) {
                min_shift = max(min_shift, minA[u] + w - d[v]);
            }
        }
    }
    
    int shift = max_shift;
    if (min_shift > max_shift) {
        shift = min_shift; 
    }
    
    for (int i = 1; i <= N; ++i) {
        if (d[i] > -1e8) {
            cout << d[i] + shift << (i == N ? "" : " ");
        } else {
            cout << minA[i] << (i == N ? "" : " ");
        }
    }
    cout << "\n";
    
    return 0;
}
