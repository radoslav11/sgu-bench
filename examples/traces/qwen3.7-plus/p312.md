#include <iostream>
#include <vector>
#include <cmath>
#include <iomanip>
#include <string>

using namespace std;

struct Point {
    double x, y;
};

double get_signed_area(const vector<Point>& P) {
    double area = 0;
    int n = P.size();
    for(int i = 0; i < n; ++i) {
        area += P[i].x * P[(i+1)%n].y - P[(i+1)%n].x * P[i].y;
    }
    return area / 2.0;
}

bool point_in_triangle(Point p, Point a, Point b, Point c) {
    double a1 = fabs((b.x - a.x) * (p.y - a.y) - (b.y - a.y) * (p.x - a.x));
    double a2 = fabs((c.x - b.x) * (p.y - b.y) - (c.y - b.y) * (p.x - b.x));
    double a3 = fabs((a.x - c.x) * (p.y - c.y) - (a.y - c.y) * (p.x - c.x));
    double total = fabs((b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x));
    return fabs(a1 + a2 + a3 - total) < 1e-6;
}

bool is_inside(Point p, const vector<Point>& V) {
    if (V.size() == 3) {
        return point_in_triangle(p, V[0], V[1], V[2]);
    } else {
        return point_in_triangle(p, V[0], V[1], V[2]) || point_in_triangle(p, V[0], V[2], V[3]);
    }
}

struct Solver {
    int N;
    vector<Point> V;
    vector<double> K, C, a, b, R;
    double total_signed;
    int sign_orient;

    Solver(int n) : N(n), V(n), K(n), C(n), a(n), b(n), R(n) {}

    void solve() {
        for(int i = 0; i < N; ++i) {
            cin >> V[i].x >> V[i].y;
        }
        string ratio_str;
        cin >> ratio_str;
        
        int idx = 0;
        string current = "";
        for(char c : ratio_str) {
            if (c == ':') {
                K[idx++] = stod(current);
                current = "";
            } else {
                current += c;
            }
        }
        K[idx] = stod(current);
        
        double total_K = 0;
        for(double k : K) total_K += k;
        
        total_signed = get_signed_area(V);
        sign_orient = (total_signed > 0) ? 1 : -1;
        double total_area = fabs(total_signed);
        
        for(int i = 0; i < N; ++i) {
            C[i] = total_area * K[i] / total_K;
            int next = (i + 1) % N;
            a[i] = V[i].y - V[next].y;
            b[i] = V[next].x - V[i].x;
            double c_i = V[i].x * V[next].y - V[i].y * V[next].x;
            double S_i = C[i] * sign_orient;
            R[i] = 2 * S_i - c_i;
        }
        
        if (N == 3) {
            Point P;
            if (!intersect_lines(0, 1, P)) {
                cout << -1 << "\n";
                return;
            }
            if (fabs(a[2]*P.x + b[2]*P.y - R[2]) > 1e-4) {
                cout << -1 << "\n";
                return;
            }
            if (!is_inside(P, V)) {
                cout << -1 << "\n";
                return;
            }
            vector<Point> p0 = {V[0], V[1], P};
            vector<Point> p1 = {V[1], V[2], P};
            vector<Point> p2 = {V[2], V[0], P};
            
            if (!check_areas({p0, p1, p2})) {
                cout << -1 << "\n";
                return;
            }
            print_result({p0, p1, p2});
            return;
        }
        
        vector<vector<Point>> result;
        if (try_single_point(result)) {
            print_result(result);
            return;
        }
        
        for (int q = 0; q < 4; ++q) {
            if (try_option(q, result)) {
                print_result(result);
                return;
            }
        }
        
        cout << -1 << "\n";
    }
    
    bool intersect_lines(int i, int j, Point& P) {
        double det = a[i]*b[j] - a[j]*b[i];
        if (fabs(det) < 1e-9) return false;
        P.x = (R[i]*b[j] - R[j]*b[i]) / det;
        P.y = (a[i]*R[j] - a[j]*R[i]) / det;
        return true;
    }
    
    bool check_areas(const vector<vector<Point>>& parts) {
        for(int i = 0; i < N; ++i) {
            double area = fabs(get_signed_area(parts[i]));
            if (area < 1e-7 || fabs(area - C[i]) > 1e-4) return false;
            double sa = get_signed_area(parts[i]);
            if (sign_orient == 1 && sa < 1e-7) return false;
            if (sign_orient == -1 && sa > -1e-7) return false;
        }
        return true;
    }
    
    bool try_single_point(vector<vector<Point>>& result) {
        Point P;
        if (!intersect_lines(0, 1, P)) return false;
        if (fabs(a[2]*P.x + b[2]*P.y - R[2]) > 1e-4) return false;
        if (fabs(a[3]*P.x + b[3]*P.y - R[3]) > 1e-4) return false;
        if (!is_inside(P, V)) return false;
        
        vector<vector<Point>> parts;
        for(int i = 0; i < N; ++i) {
            parts.push_back({V[i], V[(i+1)%N], P});
        }
        if (!check_areas(parts)) return false;
        
        result = parts;
        return true;
    }
    
    bool try_option(int q, vector<vector<Point>>& result) {
        int iA1 = (q - 1 + 4) % 4;
        int iA2 = (q - 2 + 4) % 4;
        int iB = (q + 1) % 4;
        
        Point A;
        if (!intersect_lines(iA1, iA2, A)) return false;
        
        Point B0;
        if (fabs(b[iB]) > fabs(a[iB])) {
            B0.x = 0; B0.y = R[iB] / b[iB];
        } else {
            B0.y = 0; B0.x = R[iB] / a[iB];
        }
        Point uB = {-b[iB], a[iB]};
        
        auto cross_orig = [](Point p, Point q) { return p.x * q.y - p.y * q.x; };
        
        int vq = q;
        int vq1 = (q + 1) % 4;
        
        double K_val = 0.5 * ( cross_orig(V[vq], V[vq1]) + cross_orig(V[vq1], B0) + cross_orig(B0, A) + cross_orig(A, V[vq]) );
        double D = 0.5 * ( cross_orig(V[vq1], uB) + cross_orig(uB, A) );
        
        double target = C[q] * sign_orient;
        double s = (fabs(D) < 1e-9) ? 0 : (target - K_val) / D;
        
        Point B = {B0.x + s * uB.x, B0.y + s * uB.y};
        
        vector<vector<Point>> parts(4);
        for(int i = 0; i < 4; ++i) {
            if (i == q) {
                parts[i] = {V[i], V[(i+1)%4], B, A};
            } else {
                if (i == iA1 || i == iA2) {
                    parts[i] = {V[i], V[(i+1)%4], A};
                } else {
                    parts[i] = {V[i], V[(i+1)%4], B};
                }
            }
        }
        
        if (!check_areas(parts)) return false;
        if (!is_inside(A, V)) return false;
        if (!is_inside(B, V)) return false;
        
        result = parts;
        return true;
    }
    
    void print_result(const vector<vector<Point>>& res) {
        cout << fixed << setprecision(10);
        for (const auto& p : res) {
            cout << p.size();
            for (const auto& pt : p) {
                cout << " " << pt.x << " " << pt.y;
            }
            cout << "\n";
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int N;
    if (cin >> N) {
        Solver solver(N);
        solver.solve();
    }
    return 0;
}
