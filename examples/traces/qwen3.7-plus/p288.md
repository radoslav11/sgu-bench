#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N;
    if (!(cin >> N)) return 0;
    
    if (N == 1) {
        cout << 0 << "\n0\n";
        return 0;
    }
    
    int Neven = (N % 2 == 0) ? N : N + 1;
    int T = Neven - 1;
    
    cout << T << "\n";
    
    vector<int> A(N * N, 0);
    
    for (int r = 1; r <= T; ++r) {
        for (int i = 0; i < Neven - 1; ++i) {
            int j = (Neven - 1 - i + r) % (Neven - 1);
            if (i != j) {
                if (i < N && j < N) {
                    A[i * N + j] = r;
                    A[j * N + i] = r;
                }
            } else {
                if (Neven - 1 < N) {
                    A[i * N + (Neven - 1)] = r;
                    A[(Neven - 1) * N + i] = r;
                }
            }
        }
    }
    
    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N - 1; ++j) {
            cout << A[i * N + j] << " ";
        }
        cout << A[i * N + N - 1] << "\n";
    }
    
    return 0;
}
