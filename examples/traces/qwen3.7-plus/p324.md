#include <iostream>
#include <string>
#include <vector>
#include <cctype>

using namespace std;

void solve() {
    int N;
    if (!(cin >> N)) return;
    string dummy;
    getline(cin, dummy);
    for (int k = 0; k < N; k++) {
        string s;
        getline(cin, s);
        if (!s.empty() && s.back() == '\r') {
            s.pop_back();
        }
        if (!s.empty() && s.front() == '"' && s.back() == '"') {
            s = s.substr(1, s.length() - 2);
        }

        vector<string> tokens;
        int i = 0;
        while (i < (int)s.length()) {
            if (isspace((unsigned char)s[i])) {
                i++;
                continue;
            }
            if (isalpha((unsigned char)s[i])) {
                int j = i;
                while (j < (int)s.length() && isalpha((unsigned char)s[j])) {
                    j++;
                }
                tokens.push_back(s.substr(i, j - i));
                i = j;
            } else if (s[i] == '-') {
                int j = i;
                int dash_count = 0;
                while (j < (int)s.length()) {
                    if (s[j] == '-') {
                        dash_count++;
                        j++;
                    } else if (isspace((unsigned char)s[j])) {
                        j++;
                    } else {
                        break;
                    }
                }
                tokens.push_back(string(dash_count, '-'));
                i = j;
            } else {
                i++;
            }
        }

        vector<string> res;
        bool error = false;
        for (size_t idx = 0; idx < tokens.size(); idx++) {
            if (tokens[idx][0] == '-') {
                int L = tokens[idx].length();
                if (L == 1) {
                    bool prev_is_word = (!res.empty() && res.back()[0] != '-');
                    bool next_is_word = (idx + 1 < tokens.size() && tokens[idx+1][0] != '-');
                    if (prev_is_word && next_is_word) {
                        res.back() = res.back() + "-" + tokens[idx+1];
                        idx++; // skip the next word as it is merged
                    } else {
                        error = true;
                        break;
                    }
                } else {
                    int rem = L;
                    while (rem > 0) {
                        if (rem == 2) {
                            res.push_back("--");
                            rem -= 2;
                        } else if (rem == 3) {
                            res.push_back("---");
                            rem -= 3;
                        } else {
                            res.push_back("--");
                            rem -= 2;
                        }
                    }
                }
            } else {
                res.push_back(tokens[idx]);
            }
        }

        if (error) {
            cout << "error\n";
        } else {
            string result = "";
            for (size_t idx = 0; idx < res.size(); idx++) {
                if (idx > 0) {
                    result += " ";
                }
                result += res[idx];
            }
            if (!res.empty() && res.front()[0] == '-') {
                result = " " + result;
            }
            if (!res.empty() && res.back()[0] == '-') {
                result += " ";
            }
            cout << "\"" << result << "\"\n";
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    solve();
    return 0;
}
