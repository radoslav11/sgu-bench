#include <iostream>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int k;
    if (cin >> k) {
        while (k--) {
            int n;
            cin >> n;
            bool is_zero = true;
            long long current_product = 1;
            long long total_sum = 0;
            for (int i = 0; i < n; ++i) {
                long long a;
                cin >> a;
                if (i == 0 && a > 0) {
                    is_zero = false;
                }
                current_product = (current_product * (a % 9)) % 9;
                total_sum = (total_sum + current_product) % 9;
            }
            if (is_zero) {
                cout << 0 << "\n";
            } else {
                if (total_sum == 0) {
                    cout << 9 << "\n";
                } else {
                    cout << total_sum << "\n";
                }
            }
        }
    }
    return 0;
}
