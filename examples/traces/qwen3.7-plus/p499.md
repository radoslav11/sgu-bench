```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    if (!(cin >> n)) return 0;
    
    vector<int> a(n);
    int max_val = 0;
    for (int i = 0; i < n; i++) {
        cin >> a[i];
        if (a[i] > max_val) {
            max_val = a[i];
        }
    }
    
    vector<int> cnt(max_val + 1, 0);
    for (int i = 0; i < n; i++) {
        cnt[a[i]]++;
    }
    
    for (int d = max_val; d >= 1; d--) {
        int count = 0;
        for (int j = d; j <= max_val; j += d) {
            if (cnt[j]) {
                count++;
                if (count >= 2) {
                    cout << d << "\n";
                    return 0;
                }
            }
        }
    }
    
    return 0;
}
```
