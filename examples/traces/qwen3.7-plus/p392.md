#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

struct QueryPart {
    string s;
    long long count;
};

vector<QueryPart> parse_query(const string& in) {
    vector<QueryPart> res;
    int n = in.length();
    int i = 0;
    while (i < n) {
        if (in[i] == '(') {
            int j = i + 1;
            while (j < n && in[j] != ')') j++;
            string s = in.substr(i + 1, j - i - 1);
            i = j + 1;
            long long k = 0;
            while (i < n && isdigit(in[i])) {
                k = k * 10 + (in[i] - '0');
                i++;
            }
            res.push_back({s, k});
        } else {
            int j = i;
            while (j < n && in[j] != '(' && !isdigit(in[j])) {
                j++;
            }
            string s = in.substr(i, j - i);
            if (!s.empty()) {
                res.push_back({s, 1});
            }
            i = j;
        }
    }
    return res;
}

int R, C;
vector<string> grid_dir;
vector<string> grid_char;

int pre_len[905];
string pre_str_arr[905];
int cyc_len[905];
string cyc_str_arr[905];

char get_char(int u, long long idx) {
    if (idx < pre_len[u]) {
        return pre_str_arr[u][idx];
    }
    if (cyc_len[u] == 0) {
        return '\0';
    }
    long long rem = idx - pre_len[u];
    return cyc_str_arr[u][rem % cyc_len[u]];
}

bool check(int u, const vector<QueryPart>& parts) {
    if (parts.empty()) return true;
    if (get_char(u, 0) != parts[0].s[0]) return false;

    long long current_idx = 0;
    for (const auto& part : parts) {
        const string& s = part.s;
        long long count = part.count;
        long long L = (long long)s.length() * count;
        long long s_len = s.length();
        
        if (cyc_len[u] == 0) {
            if (current_idx + L > pre_len[u]) return false;
            for (long long k = 0; k < L; k++) {
                if (s[k % s_len] != get_char(u, current_idx + k)) {
                    return false;
                }
            }
        } else {
            long long K_max = max(0LL, pre_len[u] - current_idx) + cyc_len[u] + s_len;
            long long M = min(L, K_max);
            
            bool match = true;
            for (long long k = 0; k < M; k++) {
                if (s[k % s_len] != get_char(u, current_idx + k)) {
                    match = false;
                    break;
                }
            }
            if (!match) return false;
        }
        current_idx += L;
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> R >> C)) return 0;

    grid_dir.resize(R);
    for (int i = 0; i < R; i++) {
        cin >> grid_dir[i];
    }

    grid_char.resize(R);
    for (int i = 0; i < R; i++) {
        cin >> grid_char[i];
    }

    for (int r = 0; r < R; r++) {
        for (int c = 0; c < C; c++) {
            int u = r * C + c;
            int curr_r = r, curr_c = c;
            vector<pair<int, int>> path;
            vector<int> pos_in_path(R * C, -1);
            
            while (curr_r >= 0 && curr_r < R && curr_c >= 0 && curr_c < C) {
                int curr_u = curr_r * C + curr_c;
                if (pos_in_path[curr_u] != -1) {
                    int cycle_start = pos_in_path[curr_u];
                    pre_len[u] = cycle_start;
                    for (int i = 0; i < cycle_start; i++) {
                        pre_str_arr[u] += grid_char[path[i].first][path[i].second];
                    }
                    cyc_len[u] = path.size() - cycle_start;
                    for (int i = cycle_start; i < (int)path.size(); i++) {
                        cyc_str_arr[u] += grid_char[path[i].first][path[i].second];
                    }
                    break;
                }
                pos_in_path[curr_u] = path.size();
                path.push_back({curr_r, curr_c});
                
                char dir = grid_dir[curr_r][curr_c];
                if (dir == '<') curr_c--;
                else if (dir == '>') curr_c++;
                else if (dir == '^') curr_r--;
                else if (dir == 'v') curr_r++;
            }
            
            if (curr_r < 0 || curr_r >= R || curr_c < 0 || curr_c >= C) {
                pre_len[u] = path.size();
                for (int i = 0; i < (int)path.size(); i++) {
                    pre_str_arr[u] += grid_char[path[i].first][path[i].second];
                }
                cyc_len[u] = 0;
            }
        }
    }

    int Q;
    if (!(cin >> Q)) return 0;
    for (int i = 0; i < Q; i++) {
        string q;
        cin >> q;
        vector<QueryPart> parts = parse_query(q);
        bool found = false;
        for (int r = 0; r < R; r++) {
            for (int c = 0; c < C; c++) {
                int u = r * C + c;
                if (check(u, parts)) {
                    cout << "YES (" << (r + 1) << "," << (c + 1) << ")\n";
                    found = true;
                    break;
                }
            }
            if (found) break;
        }
        if (!found) {
            cout << "NO\n";
        }
    }

    return 0;
}
