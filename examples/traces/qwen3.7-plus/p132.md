```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

int pow3_arr[8] = {1, 3, 9, 27, 81, 243, 729, 2187};
int dp[2][2187];

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int M, N;
    if (!(cin >> M >> N)) return 0;

    vector<int> cake(M, 0);
    for (int i = 0; i < M; ++i) {
        string s;
        cin >> s;
        for (int j = 0; j < N; ++j) {
            if (s[j] == '*') {
                cake[i] |= (1 << j);
            }
        }
    }

    int max_state = pow3_arr[N];
    for (int i = 0; i < max_state; ++i) {
        dp[0][i] = 1e9;
    }
    
    // Initial state: all 1s, meaning no constraints from "above" and cells can be 0.
    int initial_state = (pow3_arr[N] - 1) / 2;
    dp[0][initial_state] = 0;

    for (int r = 0; r < M; ++r) {
        int curr = r % 2;
        int next_r = (r + 1) % 2;
        for (int i = 0; i < max_state; ++i) {
            dp[next_r][i] = 1e9;
        }

        int cake_curr = cake[r];
        int cake_next = (r + 1 < M) ? cake[r + 1] : 0;

        for (int state = 0; state < max_state; ++state) {
            if (dp[curr][state] == 1e9) continue;

            auto dfs = [&](auto& self, int c, int next_state, int cost, bool prev_was_zero) -> void {
                if (c == N) {
                    if (cost < dp[next_r][next_state]) {
                        dp[next_r][next_state] = cost;
                    }
                    return;
                }

                int s = (state / pow3_arr[c]) % 3;
                bool is_candle = (cake_curr >> c) & 1;

                if (is_candle) {
                    if (s == 2) return; // Should be unreachable due to checks when placing V
                    self(self, c + 1, next_state + pow3_arr[c], cost, false);
                } else {
                    if (s == 2) {
                        // Already covered by a vertical domino from above
                        self(self, c + 1, next_state + pow3_arr[c], cost, false);
                    } else {
                        // Option 1: Place Vertical domino (covers current and next row)
                        if (r + 1 < M) {
                            bool is_candle_next = (cake_next >> c) & 1;
                            if (!is_candle_next) {
                                self(self, c + 1, next_state + 2 * pow3_arr[c], cost + 1, false);
                            }
                        }

                        // Option 2: Place Horizontal domino (covers current and next column)
                        if (c + 1 < N) {
                            int s_next = (state / pow3_arr[c + 1]) % 3;
                            bool is_candle_next_col = (cake_curr >> (c + 1)) & 1;
                            if (!is_candle_next_col && s_next != 2) {
                                self(self, c + 2, next_state + pow3_arr[c] + pow3_arr[c + 1], cost + 1, false);
                            }
                        }

                        // Option 3: Leave empty (0)
                        // Valid only if not forced to be covered (s == 1) and previous was not 0
                        if (s == 1 && !prev_was_zero) {
                            self(self, c + 1, next_state, cost, true);
                        }
                    }
                }
            };

            dfs(dfs, 0, 0, 0, false);
        }
    }

    int ans = 1e9;
    int final_r = M % 2;
    for (int i = 0; i < max_state; ++i) {
        if (dp[final_r][i] < ans) {
            ans = dp[final_r][i];
        }
    }

    cout << ans << "\n";

    return 0;
}
```
