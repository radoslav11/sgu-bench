#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <iomanip>

using namespace std;

const double PI = acos(-1.0);
const double EPS = 1e-7;
const double ANG_EPS = 1e-6;

struct Point {
    double x, y;
};

double dist(Point a, Point b) {
    return hypot(a.x - b.x, a.y - b.y);
}

struct Circle {
    Point c;
    double r;
    Point A, B, M;
};

double get_angle(Point p, Point c) {
    return atan2(p.y - c.y, p.x - c.x);
}

double ccw_dist(Point c, Point A, Point B) {
    double a1 = get_angle(A, c);
    double a2 = get_angle(B, c);
    double res = a2 - a1;
    while (res < 0) res += 2 * PI;
    while (res >= 2 * PI) res -= 2 * PI;
    return res;
}

bool is_on_arc(Point c, Point A, Point B, Point M, Point P) {
    if (dist(A, B) < 1e-5) {
        return true; 
    }
    double len = ccw_dist(c, A, B);
    double dist_m = ccw_dist(c, A, M);
    double dist_p = ccw_dist(c, A, P);
    
    bool m_ccw = (dist_m <= len + ANG_EPS);
    if (m_ccw) {
        return dist_p <= len + ANG_EPS || dist_p >= 2 * PI - ANG_EPS;
    } else {
        return dist_p >= len - ANG_EPS || dist_p <= ANG_EPS;
    }
}

bool is_strictly_inside(Point c, Point A, Point B, Point M, Point P) {
    if (dist(A, B) < 1e-5) {
        return dist(P, A) > 1e-5;
    }
    if (dist(P, A) < 1e-5 || dist(P, B) < 1e-5) return false;
    
    double len = ccw_dist(c, A, B);
    double dist_m = ccw_dist(c, A, M);
    double dist_p = ccw_dist(c, A, P);
    
    bool m_ccw = (dist_m <= len + ANG_EPS);
    if (m_ccw) {
        return dist_p > ANG_EPS && dist_p < len - ANG_EPS;
    } else {
        return (dist_p > len + ANG_EPS) && (dist_p < 2 * PI - ANG_EPS);
    }
}

Circle make_circle(Point p1, Point p2, Point p3) {
    Circle res;
    res.A = p1;
    res.B = p2;
    res.M = p3;
    
    double dx1 = p2.x - p1.x;
    double dy1 = p2.y - p1.y;
    double dx2 = p3.x - p2.x;
    double dy2 = p3.y - p2.y;
    
    double px1 = -dy1;
    double py1 = dx1;
    double px2 = -dy2;
    double py2 = dx2;
    
    double mx1 = (p1.x + p2.x) / 2.0;
    double my1 = (p1.y + p2.y) / 2.0;
    double mx2 = (p2.x + p3.x) / 2.0;
    double my2 = (p2.y + p3.y) / 2.0;
    
    double dx = (p3.x - p1.x) / 2.0;
    double dy = (p3.y - p1.y) / 2.0;
    
    double D = -px1 * py2 + px2 * py1;
    
    if (abs(D) < 1e-9) {
        res.c = {1e9, 1e9};
        res.r = 1e9;
        return res;
    }
    
    double t1 = (dx * (-py2) - (-px2) * dy) / D;
    
    res.c.x = mx1 + t1 * px1;
    res.c.y = my1 + t1 * py1;
    res.r = dist(res.c, p1);
    
    return res;
}

vector<Point> intersect_circles(Point c1, double r1, Point c2, double r2) {
    vector<Point> res;
    double d = dist(c1, c2);
    if (d > r1 + r2 + 1e-7 || d < abs(r1 - r2) - 1e-7 || d < 1e-7) {
        return res;
    }
    double a = (r1 * r1 - r2 * r2 + d * d) / (2 * d);
    double h2 = r1 * r1 - a * a;
    if (h2 < 0) h2 = 0;
    double h = sqrt(h2);
    
    double x3 = c1.x + a * (c2.x - c1.x) / d;
    double y3 = c1.y + a * (c2.y - c1.y) / d;
    
    Point p1 = {x3 + h * (c2.y - c1.y) / d, y3 - h * (c2.x - c1.x) / d};
    Point p2 = {x3 - h * (c2.y - c1.y) / d, y3 + h * (c2.x - c1.x) / d};
    
    res.push_back(p1);
    if (dist(p1, p2) > 1e-7) {
        res.push_back(p2);
    }
    return res;
}

bool check_infinite_overlap(Circle c1, Circle c2) {
    if (dist(c1.c, c2.c) > 1e-5 || abs(c1.r - c2.r) > 1e-5) {
        return false;
    }
    if (dist(c1.A, c1.B) < 1e-5) {
        if (dist(c2.A, c2.B) >= 1e-5) return true;
        if (dist(c2.M, c2.A) >= 1e-5) return true;
        return false;
    }
    if (dist(c2.A, c2.B) < 1e-5) {
        if (dist(c1.A, c1.B) >= 1e-5) return true;
        if (dist(c1.M, c1.A) >= 1e-5) return true;
        return false;
    }
    
    if (is_strictly_inside(c1.c, c1.A, c1.B, c1.M, c2.A)) return true;
    if (is_strictly_inside(c1.c, c1.A, c1.B, c1.M, c2.B)) return true;
    if (is_strictly_inside(c1.c, c1.A, c1.B, c1.M, c2.M)) return true;
    
    if (is_strictly_inside(c2.c, c2.A, c2.B, c2.M, c1.A)) return true;
    if (is_strictly_inside(c2.c, c2.A, c2.B, c2.M, c1.B)) return true;
    if (is_strictly_inside(c2.c, c2.A, c2.B, c2.M, c1.M)) return true;
    
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int N;
    if (!(cin >> N)) return 0;
    
    vector<Circle> arcs(N);
    for (int i = 0; i < N; ++i) {
        Point p1, p2, p3;
        cin >> p1.x >> p1.y >> p2.x >> p2.y >> p3.x >> p3.y;
        arcs[i] = make_circle(p1, p2, p3);
    }
    
    vector<Point> intersections;
    for (int i = 0; i < N; ++i) {
        for (int j = i + 1; j < N; ++j) {
            if (check_infinite_overlap(arcs[i], arcs[j])) {
                cout << "Infinity\n";
                return 0;
            }
            
            vector<Point> pts = intersect_circles(arcs[i].c, arcs[i].r, arcs[j].c, arcs[j].r);
            for (Point p : pts) {
                if (is_on_arc(arcs[i].c, arcs[i].A, arcs[i].B, arcs[i].M, p) &&
                    is_on_arc(arcs[j].c, arcs[j].A, arcs[j].B, arcs[j].M, p)) {
                    intersections.push_back(p);
                }
            }
        }
    }
    
    sort(intersections.begin(), intersections.end(), [](const Point& a, const Point& b) {
        if (abs(a.x - b.x) > 1e-6) return a.x < b.x;
        return a.y < b.y;
    });
    
    vector<Point> unique_intersections;
    for (Point p : intersections) {
        if (unique_intersections.empty() || dist(unique_intersections.back(), p) >= 1e-4) {
            unique_intersections.push_back(p);
        }
    }
    
    cout << unique_intersections.size() << "\n";
    cout << fixed << setprecision(3);
    for (Point p : unique_intersections) {
        cout << p.x << " " << p.y << "\n";
    }
    
    return 0;
}
