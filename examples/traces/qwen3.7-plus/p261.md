#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

using namespace std;

long long power(long long base, long long exp, long long mod) {
    long long res = 1;
    base %= mod;
    while (exp > 0) {
        if (exp % 2 == 1) res = (res * base) % mod;
        base = (base * base) % mod;
        exp /= 2;
    }
    return res;
}

long long extgcd(long long a, long long b, long long &x, long long &y) {
    if (b == 0) {
        x = 1;
        y = 0;
        return a;
    }
    long long x1, y1;
    long long d = extgcd(b, a % b, x1, y1);
    x = y1;
    y = x1 - y1 * (a / b);
    return d;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    long long P, K, A;
    if (!(cin >> P >> K >> A)) return 0;

    if (A == 0) {
        cout << 1 << "\n" << 0 << "\n";
        return 0;
    }

    if (P == 2) {
        cout << 1 << "\n" << A << "\n";
        return 0;
    }

    long long P_minus_1 = P - 1;
    vector<long long> factors;
    long long temp = P_minus_1;
    for (long long i = 2; i * i <= temp; ++i) {
        if (temp % i == 0) {
            factors.push_back(i);
            while (temp % i == 0) {
                temp /= i;
            }
        }
    }
    if (temp > 1) {
        factors.push_back(temp);
    }

    long long g = 2;
    while (true) {
        bool ok = true;
        for (long long q : factors) {
            if (power(g, P_minus_1 / q, P) == 1) {
                ok = false;
                break;
            }
        }
        if (ok) break;
        g++;
    }

    long long S = sqrt(P) + 1;
    vector<pair<long long, int>> baby;
    long long cur = A;
    for (int j = 0; j < S; ++j) {
        baby.push_back({cur, j});
        cur = (cur * g) % P;
    }
    sort(baby.begin(), baby.end());
    
    vector<pair<long long, int>> unique_baby;
    for (size_t i = 0; i < baby.size(); ++i) {
        if (i == 0 || baby[i].first != baby[i-1].first) {
            unique_baby.push_back(baby[i]);
        }
    }
    baby = unique_baby;

    long long cur_g_S = power(g, S, P);
    long long cur_i = 1;
    long long m = -1;
    for (int i = 0; i <= S; ++i) {
        auto it = lower_bound(baby.begin(), baby.end(), make_pair(cur_i, -1));
        if (it != baby.end() && it->first == cur_i) {
            m = (1LL * i * S - it->second) % P_minus_1;
            if (m < 0) m += P_minus_1;
            break;
        }
        cur_i = (cur_i * cur_g_S) % P;
    }

    long long x, y;
    long long D = extgcd(K, P_minus_1, x, y);
    if (m % D != 0) {
        cout << 0 << "\n";
        return 0;
    }

    long long mod = P_minus_1 / D;
    long long m_prime = m / D;
    long long x_mod = (x % mod + mod) % mod;
    long long y0 = (x_mod * (m_prime % mod)) % mod;

    vector<long long> roots;
    long long step = mod;
    for (long long i = 0; i < D; ++i) {
        long long current_y = (y0 + i * step) % P_minus_1;
        long long current_x = power(g, current_y, P);
        roots.push_back(current_x);
    }

    sort(roots.begin(), roots.end());

    cout << roots.size() << "\n";
    for (size_t i = 0; i < roots.size(); ++i) {
        cout << roots[i] << (i + 1 == roots.size() ? "" : " ");
    }
    cout << "\n";

    return 0;
}
