#include <iostream>
#include <iomanip>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int x, y;
    double z;
    if (cin >> x >> y >> z) {
        double t_min = 60.0 * (y - x);
        double ratio = (t_min - z) / t_min;
        if (ratio < 0.0) {
            ratio = 0.0;
        }
        double prob = 1.0 - ratio * ratio;
        cout << fixed << setprecision(7) << prob << "\n";
    }
    
    return 0;
}
