#include <iostream>

using namespace std;

uint64_t n;
int X_arr[65];
int L;
uint64_t best_ans = 0;
bool found = false;

bool can_extend(int k) {
    for (int i = 1; i < k; ++i) {
        if (X_arr[i] == 1) {
            int cmp_len = k - i;
            bool less = false;
            for (int j = 0; j < cmp_len; ++j) {
                if (X_arr[i + j] > X_arr[j]) {
                    break;
                } else if (X_arr[i + j] < X_arr[j]) {
                    less = true;
                    break;
                }
            }
            if (less) return false;
        }
    }
    return true;
}

bool is_canonical() {
    for (int i = 1; i < L; ++i) {
        if (X_arr[i] == 1) {
            int cmp_len = L - i;
            bool less = false;
            for (int j = 0; j < cmp_len; ++j) {
                if (X_arr[i + j] > X_arr[j]) {
                    break;
                } else if (X_arr[i + j] < X_arr[j]) {
                    less = true;
                    break;
                }
            }
            if (less) return false;
        }
    }
    return true;
}

void dfs(int k, bool less) {
    if (found) return;
    if (k == L) {
        if (is_canonical()) {
            uint64_t val = 0;
            for (int i = 0; i < L; ++i) {
                val = (val << 1) | X_arr[i];
            }
            best_ans = val;
            found = true;
        }
        return;
    }
    
    int max_bit = less ? 1 : ((n >> (L - 1 - k)) & 1);
    for (int bit = max_bit; bit >= 0; --bit) {
        X_arr[k] = bit;
        if (can_extend(k + 1)) {
            dfs(k + 1, less || (bit < max_bit));
            if (found) return;
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    if (cin >> n) {
        L = 64 - __builtin_clzll(n);
        dfs(0, false);
        cout << best_ans << "\n";
    }
    return 0;
}
