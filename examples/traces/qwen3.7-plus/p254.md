#pragma GCC optimize("O3")
#include <iostream>

using namespace std;

const int MAXN = 2000005;
int bit_arr[MAXN];

int find_kth(int k, int n) {
    int idx = 0;
    for (int i = 21; i >= 0; --i) {
        int next_idx = idx + (1 << i);
        if (next_idx <= n && bit_arr[next_idx] < k) {
            idx = next_idx;
            k -= bit_arr[idx];
        }
    }
    return idx + 1;
}

void add(int i, int delta, int n) {
    for (; i <= n; i += i & -i) {
        bit_arr[i] += delta;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, q;
    if (!(cin >> n >> q)) return 0;

    for (int i = 1; i <= n; ++i) {
        bit_arr[i] = i & -i;
    }

    int rem = n;
    int curr_idx = 1;
    int dir = 1;
    int last_erased = 0;

    while (rem > 0) {
        int steps = (q - 1) % rem;
        int target_idx;
        if (dir == 1) {
            target_idx = (curr_idx - 1 + steps) % rem + 1;
        } else {
            target_idx = (curr_idx - 1 - steps + rem) % rem + 1;
        }

        int val = find_kth(target_idx, n);
        last_erased = val;

        add(val, -1, n);
        rem--;

        if (rem == 0) break;

        if (target_idx == rem + 1) {
            curr_idx = 1;
        } else {
            curr_idx = target_idx;
        }

        int curr_val = find_kth(curr_idx, n);
        if (curr_val % 2 == 1) {
            dir = 1;
        } else {
            dir = -1;
        }
    }

    cout << last_erased << "\n";

    return 0;
}
