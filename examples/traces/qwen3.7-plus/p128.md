```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstring>

using namespace std;

struct Point {
    int x, y, id;
};

struct Event {
    int x;
    int type; // -1: remove, 0: query, 1: add
    int y;
    int y2;
    
    bool operator<(const Event& other) const {
        if (x != other.x) return x < other.x;
        return type < other.type;
    }
};

int bit[20005];

void add_bit(int idx, int val) {
    for (; idx <= 20002; idx += idx & -idx) {
        bit[idx] += val;
    }
}

int query_bit(int idx) {
    int sum = 0;
    for (; idx > 0; idx -= idx & -idx) {
        sum += bit[idx];
    }
    return sum;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    if (!(cin >> N)) return 0;

    vector<Point> pts(N);
    for (int i = 0; i < N; i++) {
        cin >> pts[i].x >> pts[i].y;
        pts[i].id = i;
    }

    vector<Point> by_x = pts;
    sort(by_x.begin(), by_x.end(), [](const Point& a, const Point& b) {
        if (a.x != b.x) return a.x < b.x;
        return a.y < b.y;
    });

    vector<Point> by_y = pts;
    sort(by_y.begin(), by_y.end(), [](const Point& a, const Point& b) {
        if (a.y != b.y) return a.y < b.y;
        return a.x < b.x;
    });

    int adj[10000][2];
    int deg[10000] = {0};

    for (int i = 0; i < N; ) {
        int j = i;
        while (j < N && by_x[j].x == by_x[i].x) j++;
        if ((j - i) % 2 != 0) {
            cout << 0 << "\n";
            return 0;
        }
        for (int k = i; k < j; k += 2) {
            int u = by_x[k].id;
            int v = by_x[k+1].id;
            int y1 = by_x[k].y;
            int y2 = by_x[k+1].y;
            if (y1 == y2) {
                cout << 0 << "\n";
                return 0;
            }
            adj[u][deg[u]++] = v;
            adj[v][deg[v]++] = u;
        }
        i = j;
    }

    for (int i = 0; i < N; ) {
        int j = i;
        while (j < N && by_y[j].y == by_y[i].y) j++;
        if ((j - i) % 2 != 0) {
            cout << 0 << "\n";
            return 0;
        }
        for (int k = i; k < j; k += 2) {
            int u = by_y[k].id;
            int v = by_y[k+1].id;
            int x1 = by_y[k].x;
            int x2 = by_y[k+1].x;
            if (x1 == x2) {
                cout << 0 << "\n";
                return 0;
            }
            adj[u][deg[u]++] = v;
            adj[v][deg[v]++] = u;
        }
        i = j;
    }

    long long total_length = 0;
    for (int i = 0; i < N; ) {
        int j = i;
        while (j < N && by_x[j].x == by_x[i].x) j++;
        for (int k = i; k < j; k += 2) {
            total_length += abs(by_x[k].y - by_x[k+1].y);
        }
        i = j;
    }
    for (int i = 0; i < N; ) {
        int j = i;
        while (j < N && by_y[j].y == by_y[i].y) j++;
        for (int k = i; k < j; k += 2) {
            total_length += abs(by_y[k].x - by_y[k+1].x);
        }
        i = j;
    }

    int curr = 0;
    int prev = -1;
    int count = 0;
    bool visited[10000] = {false};

    while (!visited[curr]) {
        visited[curr] = true;
        count++;
        int next = (adj[curr][0] == prev) ? adj[curr][1] : adj[curr][0];
        prev = curr;
        curr = next;
    }

    if (count != N) {
        cout << 0 << "\n";
        return 0;
    }

    vector<Event> events;
    for (int i = 0; i < N; ) {
        int j = i;
        while (j < N && by_y[j].y == by_y[i].y) j++;
        for (int k = i; k < j; k += 2) {
            int x1 = by_y[k].x;
            int x2 = by_y[k+1].x;
            int y = by_y[k].y;
            if (x1 > x2) swap(x1, x2);
            events.push_back({x1, 1, y, 0});
            events.push_back({x2, -1, y, 0});
        }
        i = j;
    }

    for (int i = 0; i < N; ) {
        int j = i;
        while (j < N && by_x[j].x == by_x[i].x) j++;
        for (int k = i; k < j; k += 2) {
            int y1 = by_x[k].y;
            int y2 = by_x[k+1].y;
            int x = by_x[k].x;
            if (y1 > y2) swap(y1, y2);
            events.push_back({x, 0, y1, y2});
        }
        i = j;
    }

    sort(events.begin(), events.end());

    memset(bit, 0, sizeof(bit));

    for (const auto& ev : events) {
        if (ev.type == 1) {
            add_bit(ev.y + 10001, 1);
        } else if (ev.type == -1) {
            add_bit(ev.y + 10001, -1);
        } else if (ev.type == 0) {
            int y1_bit = ev.y + 10001;
            int y2_bit = ev.y2 + 10001;
            int cnt = query_bit(y2_bit - 1) - query_bit(y1_bit);
            if (cnt > 0) {
                cout << 0 << "\n";
                return 0;
            }
        }
    }

    cout << total_length << "\n";
    return 0;
}
```
