```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    if (!(cin >> n)) return 0;

    int m;
    cin >> m;

    vector<int> A(n, 0);

    for (int k = 0; k < m; ++k) {
        int i;
        cin >> i;

        vector<pair<int, int>> changes;

        if (A[i] == 0) {
            A[i] = 1;
            changes.push_back({i, 1});
        } else if (A[i] == 1) {
            bool left_is_2 = (i > 0 && A[i - 1] == 2);
            bool right_is_2 = (i < n - 1 && A[i + 1] == 2);

            if (left_is_2 || right_is_2) {
                A[i] = 0;
                changes.push_back({i, 0});

                A[i + 1]++;
                if (A[i + 1] == 3) {
                    A[i + 1] = 1;
                    changes.push_back({i + 1, 1});
                    if (i + 2 < n) {
                        A[i + 2]++;
                        changes.push_back({i + 2, A[i + 2]});
                    }
                } else {
                    changes.push_back({i + 1, A[i + 1]});
                }
            } else {
                A[i] = 2;
                changes.push_back({i, 2});
            }
        } else if (A[i] == 2) {
            A[i] = 1;
            changes.push_back({i, 1});

            A[i + 1]++;
            changes.push_back({i + 1, A[i + 1]});
        }

        cout << changes.size();
        for (auto p : changes) {
            cout << " " << p.first << " " << p.second;
        }
        cout << "\n";
    }

    return 0;
}
```
