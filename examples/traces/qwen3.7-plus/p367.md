```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    long long T;
    if (!(cin >> n >> T)) return 0;

    vector<long long> t(n + 1);
    for (int i = 1; i <= n; ++i) {
        cin >> t[i];
    }

    int m;
    cin >> m;

    vector<vector<int>> adj(n + 1);
    vector<int> in_degree(n + 1, 0);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        in_degree[v]++;
    }

    queue<int> q;
    for (int i = 1; i <= n; ++i) {
        if (in_degree[i] == 0) {
            q.push(i);
        }
    }

    vector<int> valid_nodes;
    vector<int> topo_order(n + 1, 0);
    int order_idx = 0;

    while (!q.empty()) {
        int u = q.front();
        q.pop();
        valid_nodes.push_back(u);
        topo_order[u] = order_idx++;

        for (int v : adj[u]) {
            in_degree[v]--;
            if (in_degree[v] == 0) {
                q.push(v);
            }
        }
    }

    sort(valid_nodes.begin(), valid_nodes.end(), [&](int u, int v) {
        if (t[u] != t[v]) {
            return t[u] < t[v];
        }
        return topo_order[u] < topo_order[v];
    });

    long long current_time = 0;
    long long penalty = 0;
    vector<int> solved;

    for (int u : valid_nodes) {
        if (current_time + t[u] <= T) {
            current_time += t[u];
            penalty += current_time;
            solved.push_back(u);
        } else {
            break;
        }
    }

    cout << solved.size() << " " << penalty << "\n";
    for (size_t i = 0; i < solved.size(); ++i) {
        cout << solved[i] << (i + 1 == solved.size() ? "" : " ");
    }
    cout << "\n";

    return 0;
}
```
