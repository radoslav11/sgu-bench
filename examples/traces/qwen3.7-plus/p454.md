#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

struct Run {
    int target;
    int id;
    vector<pair<int, int>> cells;
};

struct RunState {
    int target;
    int current_sum;
    int used_mask;
    int assigned_cells;
    int total_cells;
};

struct CellState {
    int r, c;
    int h_run, v_run;
    int val;
};

int n, m;
vector<vector<string>> grid;
vector<CellState> cells;
vector<RunState> h_run_states;
vector<RunState> v_run_states;
vector<int> unassigned;
int ans[10][10];
int cell_h_run[10][10];
int cell_v_run[10][10];

int get_min_sum(int cells_left, int used_mask) {
    if (cells_left == 0) return 0;
    int sum = 0;
    int count = 0;
    for (int d = 1; d <= 9; ++d) {
        if (!(used_mask & (1 << d))) {
            sum += d;
            count++;
            if (count == cells_left) break;
        }
    }
    return sum;
}

int get_max_sum(int cells_left, int used_mask) {
    if (cells_left == 0) return 0;
    int sum = 0;
    int count = 0;
    for (int d = 9; d >= 1; --d) {
        if (!(used_mask & (1 << d))) {
            sum += d;
            count++;
            if (count == cells_left) break;
        }
    }
    return sum;
}

void assign_cell(int cell_idx, int d) {
    CellState& cell = cells[cell_idx];
    cell.val = d;
    ans[cell.r][cell.c] = d;
    
    RunState& hr = h_run_states[cell.h_run];
    hr.current_sum += d;
    hr.used_mask |= (1 << d);
    hr.assigned_cells++;
    
    RunState& vr = v_run_states[cell.v_run];
    vr.current_sum += d;
    vr.used_mask |= (1 << d);
    vr.assigned_cells++;
}

void unassign_cell(int cell_idx, int d) {
    CellState& cell = cells[cell_idx];
    cell.val = 0;
    ans[cell.r][cell.c] = 0;
    
    RunState& hr = h_run_states[cell.h_run];
    hr.current_sum -= d;
    hr.used_mask &= ~(1 << d);
    hr.assigned_cells--;
    
    RunState& vr = v_run_states[cell.v_run];
    vr.current_sum -= d;
    vr.used_mask &= ~(1 << d);
    vr.assigned_cells--;
}

bool dfs() {
    if (unassigned.empty()) return true;
    
    int best_cell = -1;
    int min_choices = 10;
    int best_i = -1;
    
    for (int i = 0; i < (int)unassigned.size(); ++i) {
        int idx = unassigned[i];
        int choices = 0;
        CellState& cell = cells[idx];
        RunState& hr = h_run_states[cell.h_run];
        RunState& vr = v_run_states[cell.v_run];
        
        int rem_h = hr.total_cells - hr.assigned_cells;
        int rem_v = vr.total_cells - vr.assigned_cells;
        int rem_sum_h = hr.target - hr.current_sum;
        int rem_sum_v = vr.target - vr.current_sum;
        
        for (int d = 1; d <= 9; ++d) {
            if ((hr.used_mask & (1 << d)) || (vr.used_mask & (1 << d))) continue;
            if (d > rem_sum_h || d > rem_sum_v) continue;
            
            int new_mask_h = hr.used_mask | (1 << d);
            int new_mask_v = vr.used_mask | (1 << d);
            
            int min_h = get_min_sum(rem_h - 1, new_mask_h);
            int max_h = get_max_sum(rem_h - 1, new_mask_h);
            if (rem_sum_h - d < min_h || rem_sum_h - d > max_h) continue;
            
            int min_v = get_min_sum(rem_v - 1, new_mask_v);
            int max_v = get_max_sum(rem_v - 1, new_mask_v);
            if (rem_sum_v - d < min_v || rem_sum_v - d > max_v) continue;
            
            choices++;
        }
        if (choices < min_choices) {
            min_choices = choices;
            best_cell = idx;
            best_i = i;
            if (min_choices == 0) break;
            if (min_choices == 1) break;
        }
    }
    
    if (min_choices == 0) return false;
    
    CellState& cell = cells[best_cell];
    RunState& hr = h_run_states[cell.h_run];
    RunState& vr = v_run_states[cell.v_run];
    
    int rem_h = hr.total_cells - hr.assigned_cells;
    int rem_v = vr.total_cells - vr.assigned_cells;
    int rem_sum_h = hr.target - hr.current_sum;
    int rem_sum_v = vr.target - vr.current_sum;
    
    vector<int> valid_choices;
    for (int d = 1; d <= 9; ++d) {
        if ((hr.used_mask & (1 << d)) || (vr.used_mask & (1 << d))) continue;
        if (d > rem_sum_h || d > rem_sum_v) continue;
        
        int new_mask_h = hr.used_mask | (1 << d);
        int new_mask_v = vr.used_mask | (1 << d);
        
        int min_h = get_min_sum(rem_h - 1, new_mask_h);
        int max_h = get_max_sum(rem_h - 1, new_mask_h);
        if (rem_sum_h - d < min_h || rem_sum_h - d > max_h) continue;
        
        int min_v = get_min_sum(rem_v - 1, new_mask_v);
        int max_v = get_max_sum(rem_v - 1, new_mask_v);
        if (rem_sum_v - d < min_v || rem_sum_v - d > max_v) continue;
        
        valid_choices.push_back(d);
    }
    
    swap(unassigned[best_i], unassigned.back());
    unassigned.pop_back();
    
    for (int d : valid_choices) {
        assign_cell(best_cell, d);
        if (dfs()) return true;
        unassign_cell(best_cell, d);
    }
    
    unassigned.push_back(best_cell);
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    if (!(cin >> n >> m)) return 0;
    
    grid.assign(n, vector<string>(m));
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            cin >> grid[i][j];
        }
    }
    
    vector<Run> h_runs, v_runs;
    
    for (int r = 0; r < n; ++r) {
        for (int c = 0; c < m; ++c) {
            if (grid[r][c] == ".....") {
                int c_end = c;
                while (c_end + 1 < m && grid[r][c_end + 1] == ".....") {
                    c_end++;
                }
                Run run;
                string clue_str = grid[r][c - 1].substr(3, 2);
                run.target = stoi(clue_str);
                run.id = h_runs.size();
                for (int k = c; k <= c_end; ++k) {
                    run.cells.push_back({r, k});
                    cell_h_run[r][k] = run.id;
                }
                h_runs.push_back(run);
                c = c_end;
            }
        }
    }
    
    for (int c = 0; c < m; ++c) {
        for (int r = 0; r < n; ++r) {
            if (grid[r][c] == ".....") {
                int r_end = r;
                while (r_end + 1 < n && grid[r_end + 1][c] == ".....") {
                    r_end++;
                }
                Run run;
                string clue_str = grid[r - 1][c].substr(0, 2);
                run.target = stoi(clue_str);
                run.id = v_runs.size();
                for (int k = r; k <= r_end; ++k) {
                    run.cells.push_back({k, c});
                    cell_v_run[k][c] = run.id;
                }
                v_runs.push_back(run);
                r = r_end;
            }
        }
    }
    
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < m; ++j) {
            if (grid[i][j] == ".....") {
                CellState cell;
                cell.r = i;
                cell.c = j;
                cell.h_run = cell_h_run[i][j];
                cell.v_run = cell_v_run[i][j];
                cell.val = 0;
                cells.push_back(cell);
                unassigned.push_back(cells.size() - 1);
            }
        }
    }
    
    for (size_t i = 0; i < h_runs.size(); ++i) {
        RunState rs;
        rs.target = h_runs[i].target;
        rs.current_sum = 0;
        rs.used_mask = 0;
        rs.assigned_cells = 0;
        rs.total_cells = h_runs[i].cells.size();
        h_run_states.push_back(rs);
    }
    
    for (size_t i = 0; i < v_runs.size(); ++i) {
        RunState rs;
        rs.target = v_runs[i].target;
        rs.current_sum = 0;
        rs.used_mask = 0;
        rs.assigned_cells = 0;
        rs.total_cells = v_runs[i].cells.size();
        v_run_states.push_back(rs);
    }
    
    dfs();
    
    for (int r = 0; r < n; ++r) {
        for (int c = 0; c < m; ++c) {
            if (grid[r][c] == ".....") {
                cout << ans[r][c];
            } else {
                cout << "_";
            }
            if (c + 1 < m) cout << " ";
        }
        cout << "\n";
    }
    
    return 0;
}
