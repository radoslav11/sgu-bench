#include <iostream>

using namespace std;

long long gcd(long long a, long long b) {
    while (b) {
        a %= b;
        swap(a, b);
    }
    return a;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    int case_num = 1;
    while (cin >> n) {
        if (n == 1) {
            cout << "Case #" << case_num++ << ": -1/1\n";
            continue;
        }
        
        long long gamma;
        if (n % 2 == 1) {
            gamma = (n + 1) / 2;
        } else {
            gamma = n / 2 + (n % 4 == 2 ? 1 : 0);
        }
        
        long long num = 2 - gamma;
        long long den = gamma;
        long long g = gcd(num < 0 ? -num : num, den);
        
        num /= g;
        den /= g;
        
        cout << "Case #" << case_num++ << ": " << num << "/" << den << "\n";
    }
    
    return 0;
}
