#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

vector<pair<long long, long long>> get_intervals(const vector<long long>& p, long long t) {
    vector<pair<long long, long long>> res;
    if (p.empty()) return res;
    long long start = p[0];
    for (size_t i = 0; i < p.size(); ++i) {
        if (i == p.size() - 1 || p[i+1] - p[i] > 2 * t) {
            res.push_back({start - t, p[i] + t});
            if (i < p.size() - 1) {
                start = p[i+1];
            }
        }
    }
    return res;
}

bool is_valid(const vector<pair<long long, long long>>& I1, const vector<pair<long long, long long>>& I2, long long d) {
    size_t i = 0, j = 0;
    size_t n1 = I1.size(), n2 = I2.size();
    while (i < n1 && j < n2) {
        long long L = I1[i].first > I2[j].first ? I1[i].first : I2[j].first;
        long long R = I1[i].second < I2[j].second ? I1[i].second : I2[j].second;
        if (L <= R) {
            if (R - L > d) return false;
        }
        if (I1[i].second < I2[j].second) {
            i++;
        } else {
            j++;
        }
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    long long d;
    if (!(cin >> n >> m >> d)) return 0;

    vector<long long> p(n);
    for (int i = 0; i < n; ++i) {
        cin >> p[i];
    }

    vector<long long> q(m);
    for (int i = 0; i < m; ++i) {
        cin >> q[i];
    }

    vector<long long> C1;
    C1.push_back(1);
    for (int i = 0; i < n - 1; ++i) {
        long long diff = p[i+1] - p[i];
        C1.push_back((diff + 1) / 2);
    }
    sort(C1.begin(), C1.end());
    C1.erase(unique(C1.begin(), C1.end()), C1.end());

    vector<long long> C2;
    C2.push_back(1);
    for (int i = 0; i < m - 1; ++i) {
        long long diff = q[i+1] - q[i];
        C2.push_back((diff + 1) / 2);
    }
    sort(C2.begin(), C2.end());
    C2.erase(unique(C2.begin(), C2.end()), C2.end());

    int K1 = C1.size();
    int K2 = C2.size();

    int best_sum = 1e9;
    long long best_t1 = -1, best_t2 = -1;

    int k2 = K2 - 1;
    for (int k1 = 0; k1 < K1; ++k1) {
        vector<pair<long long, long long>> I1 = get_intervals(p, C1[k1]);
        int c1 = I1.size();
        
        while (k2 >= 0) {
            vector<pair<long long, long long>> I2 = get_intervals(q, C2[k2]);
            if (is_valid(I1, I2, d)) {
                int current_sum = c1 + (int)I2.size();
                if (current_sum < best_sum) {
                    best_sum = current_sum;
                    best_t1 = C1[k1];
                    best_t2 = C2[k2];
                }
                break;
            }
            k2--;
        }
    }

    if (best_sum == 1e9) {
        cout << "No solution\n";
    } else {
        cout << best_t1 << " " << best_t2 << "\n";
    }

    return 0;
}
