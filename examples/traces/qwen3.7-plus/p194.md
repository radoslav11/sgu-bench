```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

struct Edge {
    int to;
    long long cap;
    long long flow;
    int rev;
    int id;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    while (cin >> n >> m) {
        vector<long long> D(n + 1, 0);
        vector<long long> L(m + 1, 0);
        vector<vector<Edge>> adj(n + 2);

        auto add_edge = [&](int u, int v, long long cap, int id) {
            adj[u].push_back({v, cap, 0, (int)adj[v].size(), id});
            adj[v].push_back({u, 0, 0, (int)adj[u].size() - 1, 0});
        };

        int S = 0, T = n + 1;
        long long sum_D = 0;
        for (int i = 1; i <= m; ++i) {
            int u, v;
            long long l, c;
            cin >> u >> v >> l >> c;
            L[i] = l;
            D[u] -= l;
            D[v] += l;
            add_edge(u, v, c - l, i);
        }

        for (int i = 1; i <= n; ++i) {
            if (D[i] > 0) {
                add_edge(S, i, D[i], 0);
                sum_D += D[i];
            } else if (D[i] < 0) {
                add_edge(i, T, -D[i], 0);
            }
        }

        vector<int> level(n + 2);
        vector<int> ptr(n + 2);

        auto bfs = [&]() {
            fill(level.begin(), level.end(), -1);
            level[S] = 0;
            queue<int> q;
            q.push(S);
            while (!q.empty()) {
                int v = q.front();
                q.pop();
                for (auto& edge : adj[v]) {
                    if (edge.cap - edge.flow > 0 && level[edge.to] == -1) {
                        level[edge.to] = level[v] + 1;
                        q.push(edge.to);
                    }
                }
            }
            return level[T] != -1;
        };

        auto dfs = [&](auto& self, int v, long long pushed) -> long long {
            if (pushed == 0) return 0;
            if (v == T) return pushed;
            for (int& cid = ptr[v]; cid < (int)adj[v].size(); ++cid) {
                auto& edge = adj[v][cid];
                int tr = edge.to;
                if (level[v] + 1 != level[tr] || edge.cap - edge.flow == 0) continue;
                long long push = self(self, tr, min(pushed, edge.cap - edge.flow));
                if (push == 0) continue;
                edge.flow += push;
                adj[tr][edge.rev].flow -= push;
                return push;
            }
            return 0;
        };

        long long total_flow = 0;
        while (bfs()) {
            fill(ptr.begin(), ptr.end(), 0);
            while (long long pushed = dfs(dfs, S, 2e18)) {
                total_flow += pushed;
            }
        }

        if (total_flow == sum_D) {
            cout << "YES\n";
            vector<long long> ans(m + 1, 0);
            for (int u = 1; u <= n; ++u) {
                for (auto& edge : adj[u]) {
                    if (edge.id > 0) {
                        ans[edge.id] = L[edge.id] + edge.flow;
                    }
                }
            }
            for (int i = 1; i <= m; ++i) {
                cout << ans[i] << "\n";
            }
        } else {
            cout << "NO\n";
        }
    }

    return 0;
}
```
