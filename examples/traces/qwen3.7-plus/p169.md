#include <iostream>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    long long K;
    if (cin >> K) {
        if (K == 1) {
            cout << 8 << "\n";
        } else {
            int count = 1; // d = 1 is always valid
            if ((K - 1) % 3 == 0) {
                count += 2; // d = 2 and d = 5 are valid
            }
            if ((K - 1) % 6 == 0) {
                count += 1; // d = 6 is valid
            }
            cout << count << "\n";
        }
    }
    
    return 0;
}
