#include <iostream>
#include <string>
#include <cctype>

using namespace std;

string s;
int pos = 0;

void process(int end_pos, bool is_group) {
    while (pos < end_pos) {
        char c = s[pos];
        if (isspace(c)) {
            pos++;
            continue;
        }
        if (c == '}' && is_group) {
            pos++;
            break;
        }
        if (c == '{' && !is_group) {
            pos++;
            process(end_pos, true);
            continue;
        }
        if (c == '^' || c == '_') {
            if (c == '^') cout << "<sup>";
            else cout << "<sub>";
            pos++;
            while (pos < end_pos && isspace(s[pos])) pos++;
            if (pos < end_pos && s[pos] == '{') {
                pos++;
                process(end_pos, true);
            } else if (pos < end_pos) {
                char single = s[pos];
                pos++;
                if (isalpha(single)) {
                    cout << "<i>" << single << "</i>";
                } else {
                    if (single == '+' || single == '-' || single == '*' || single == '/') {
                        cout << "&nbsp;" << single << "&nbsp;";
                    } else {
                        cout << single;
                    }
                }
            }
            if (c == '^') cout << "</sup>";
            else cout << "</sub>";
        } else if (isalpha(c)) {
            cout << "<i>";
            while (pos < end_pos && isalpha(s[pos])) {
                cout << s[pos];
                pos++;
            }
            cout << "</i>";
        } else if (c == '+' || c == '-' || c == '*' || c == '/') {
            cout << "&nbsp;" << c << "&nbsp;";
            pos++;
        } else {
            cout << c;
            pos++;
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    string line;
    while (getline(cin, line)) {
        size_t first = line.find('$');
        size_t last = line.rfind('$');
        if (first != string::npos && last != string::npos && first < last) {
            s = line.substr(first + 1, last - first - 1);
            pos = 0;
            process(s.length(), false);
            cout << "\n";
        }
    }
    return 0;
}
