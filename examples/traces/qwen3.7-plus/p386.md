#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

using namespace std;

long long power(long long base, long long exp, long long mod) {
    long long res = 1;
    base %= mod;
    while (exp > 0) {
        if (exp % 2 == 1) res = (long long)((__int128)res * base % mod);
        base = (long long)((__int128)base * base % mod);
        exp /= 2;
    }
    return res;
}

long long modInverse(long long n, long long mod) {
    return power(n, mod - 2, mod);
}

long long extGcd(long long a, long long b, long long &x, long long &y) {
    if (b == 0) { x = 1; y = 0; return a; }
    long long x1, y1;
    long long d = extGcd(b, a % b, x1, y1);
    x = y1;
    y = x1 - y1 * (a / b);
    return d;
}

long long get_mod_det(int n, const vector<vector<long long>>& A, long long mod) {
    vector<vector<long long>> M = A;
    long long det = 1;
    int sign = 1;
    for (int k = 0; k < n; ++k) {
        int pivot = k;
        for (int i = k + 1; i < n; ++i) {
            if (M[i][k] != 0) {
                pivot = i;
                break;
            }
        }
        if (M[pivot][k] == 0) return 0;
        if (pivot != k) {
            swap(M[k], M[pivot]);
            sign = -sign;
        }
        long long inv = modInverse(M[k][k], mod);
        det = (long long)((__int128)det * M[k][k] % mod);
        for (int i = k + 1; i < n; ++i) {
            long long factor = (long long)((__int128)M[i][k] * inv % mod);
            for (int j = k; j < n; ++j) {
                M[i][j] = (M[i][j] - (long long)((__int128)factor * M[k][j] % mod)) % mod;
                if (M[i][j] < 0) M[i][j] += mod;
            }
        }
    }
    if (sign == -1) {
        det = (mod - det) % mod;
    }
    return det;
}

long double get_ld_det(int n, const vector<vector<long long>>& A) {
    vector<vector<long double>> M(n, vector<long double>(n));
    for(int i = 0; i < n; ++i)
        for(int j = 0; j < n; ++j)
            M[i][j] = A[i][j];
            
    long double det = 1.0;
    int sign = 1;
    for (int k = 0; k < n; ++k) {
        int pivot = k;
        for (int i = k + 1; i < n; ++i) {
            if (abs(M[i][k]) > abs(M[pivot][k])) {
                pivot = i;
            }
        }
        if (abs(M[pivot][k]) < 1e-9) {
            return 0.0;
        }
        if (pivot != k) {
            swap(M[k], M[pivot]);
            sign = -sign;
        }
        det *= M[k][k];
        for (int i = k + 1; i < n; ++i) {
            long double factor = M[i][k] / M[k][k];
            for (int j = k; j < n; ++j) {
                M[i][j] -= factor * M[k][j];
            }
        }
    }
    return det * sign;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int n;
    long long p;
    if (!(cin >> n >> p)) return 0;
    
    vector<vector<long long>> A(n, vector<long long>(n));
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cin >> A[i][j];
        }
    }
    
    long double ld_det = get_ld_det(n, A);
    long long q1 = 1000000007;
    long long q2 = 1000000009;
    long long d1 = get_mod_det(n, A, q1);
    long long d2 = get_mod_det(n, A, q2);
    
    long long x, y;
    extGcd(q1, q2, x, y);
    __int128 m12 = (__int128)q1 * q2;
    __int128 r12 = ((__int128)d1 * y % m12 * q2 + (__int128)d2 * x % m12 * q1) % m12;
    if (r12 < 0) r12 += m12;
    if (r12 > m12 / 2) r12 -= m12;
    
    int final_sign = 1;
    if (ld_det > 1e15) {
        final_sign = 1;
    } else if (ld_det < -1e15) {
        final_sign = -1;
    } else {
        if (r12 > 0) final_sign = 1;
        else if (r12 < 0) final_sign = -1;
        else final_sign = 1;
    }
    
    long long ans = get_mod_det(n, A, p);
    if (final_sign == -1) {
        ans = (p - ans) % p;
    }
    
    long long pow2 = 1;
    for (int i = 0; i < n; ++i) {
        pow2 = (long long)((__int128)pow2 * 2 % p);
    }
    
    ans = (long long)((__int128)ans * pow2 % p);
    cout << ans << "\n";
    
    return 0;
}
