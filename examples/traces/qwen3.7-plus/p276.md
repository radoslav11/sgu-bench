```cpp
#include <iostream>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int s, p;
    if (cin >> s >> p) {
        int diff = p - s;
        if (diff <= 0) {
            cout << 0 << "\n";
        } else if (diff < 300) {
            cout << 1 << "\n";
        } else if (diff < 900) {
            cout << 2 << "\n";
        } else if (diff < 1800) {
            cout << 3 << "\n";
        } else {
            cout << 4 << "\n";
        }
    }
    
    return 0;
}
```
