#include <iostream>
#include <vector>
#include <string>
#include <bitset>

using namespace std;

struct Matrix {
    int size;
    vector<bitset<64>> rows;
    Matrix(int n) : size(n), rows(n) {
        for(int i = 0; i < n; ++i) rows[i].reset();
    }
    Matrix operator*(const Matrix& other) const {
        Matrix res(size);
        for(int i = 0; i < size; ++i) {
            for(int k = 0; k < size; ++k) {
                if(rows[i].test(k)) {
                    res.rows[i] ^= other.rows[k];
                }
            }
        }
        return res;
    }
};

Matrix power(Matrix base, int exp) {
    Matrix res(base.size);
    for (int i = 0; i < base.size; ++i) res.rows[i].set(i);
    while (exp > 0) {
        if (exp % 2 == 1) res = res * base;
        base = base * base;
        exp /= 2;
    }
    return res;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M, K, L;
    if (!(cin >> N >> M >> K >> L)) return 0;

    vector<pair<int, int>> ops(M);
    for (int i = 0; i < M; ++i) {
        cin >> ops[i].first >> ops[i].second;
    }

    vector<string> c(L), c_prime(L);
    for (int i = 0; i < L; ++i) {
        cin >> c[i] >> c_prime[i];
    }

    string state;
    cin >> state;

    vector<int> A(K - 1, 0);
    vector<bitset<64>> aug(L);
    for (int s = 0; s < L; ++s) {
        for (int i = 0; i < K - 1; ++i) {
            if (c[s][i + 1] == '1') {
                aug[s].set(i);
            }
        }
        if ((c[s][0] - '0') ^ (c_prime[s][K - 1] - '0')) {
            aug[s].set(K - 1);
        }
    }

    int row = 0;
    for (int col = 0; col < K - 1 && row < L; ++col) {
        int pivot = row;
        while (pivot < L && !aug[pivot].test(col)) {
            pivot++;
        }
        if (pivot == L) continue;
        swap(aug[row], aug[pivot]);
        for (int i = 0; i < L; ++i) {
            if (i != row && aug[i].test(col)) {
                aug[i] ^= aug[row];
            }
        }
        row++;
    }

    for (int i = 0; i < K - 1; ++i) {
        if (aug[i].test(K - 1)) {
            A[i] = 1;
        }
    }

    Matrix NInv(K);
    for (int i = 0; i < K - 1; ++i) {
        NInv.rows[0].set(i, A[i]);
    }
    NInv.rows[0].set(K - 1, 1);
    for (int i = 1; i < K; ++i) {
        NInv.rows[i].set(i - 1, 1);
    }

    for (int i = M - 1; i >= 0; --i) {
        int start = ops[i].first - 1;
        int D = ops[i].second;
        Matrix T = power(NInv, D);

        vector<int> C(K);
        for (int j = 0; j < K; ++j) {
            C[j] = state[start + j] - '0';
        }

        vector<int> C_new(K, 0);
        for (int j = 0; j < K; ++j) {
            int val = 0;
            for (int k = 0; k < K; ++k) {
                if (T.rows[j].test(k)) {
                    val ^= C[k];
                }
            }
            C_new[j] = val;
        }

        for (int j = 0; j < K; ++j) {
            state[start + j] = C_new[j] + '0';
        }
    }

    cout << state << "\n";

    return 0;
}
