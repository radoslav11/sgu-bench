```cpp
#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, f;
    if (!(cin >> n >> f)) return 0;
    
    vector<int> e(n);
    vector<bool> highlighted(105, false);
    for (int i = 0; i < n; ++i) {
        cin >> e[i];
        highlighted[e[i]] = true;
    }

    int current_floor = f;
    vector<int> ans;

    while (true) {
        int min_idx = n;
        int a = -1;
        for (int i = 0; i < n; ++i) {
            if (highlighted[e[i]]) {
                min_idx = i;
                a = e[i];
                break;
            }
        }
        if (min_idx == n) break;

        int dir = (a > current_floor) ? 1 : -1;
        for (int curr = current_floor + dir; curr != a + dir; curr += dir) {
            if (highlighted[curr]) {
                ans.push_back(curr);
                highlighted[curr] = false;
            }
        }
        current_floor = a;
    }

    for (int i = 0; i < (int)ans.size(); ++i) {
        cout << ans[i] << (i + 1 == (int)ans.size() ? "" : " ");
    }
    cout << "\n";

    return 0;
}
```
