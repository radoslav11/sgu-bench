#include <iostream>
#include <cmath>
#include <iomanip>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    double x0, y0, r;
    if (cin >> x0 >> y0 >> r) {
        double pi = acos(-1.0);
        double ans = pi * r * r * log(x0 * x0 + y0 * y0);
        cout << fixed << setprecision(15) << ans << "\n";
    }
    
    return 0;
}
