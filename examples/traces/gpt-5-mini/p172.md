#include <bits/stdc++.h>
using namespace std;
int main(){
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int N, M;
    if(!(cin >> N >> M)) return 0;
    vector<vector<int>> adj(N+1);
    for(int i=0;i<M;i++){
        int u,v; cin >> u >> v;
        if(u>=1 && u<=N && v>=1 && v<=N){
            adj[u].push_back(v);
            adj[v].push_back(u);
        }
    }
    vector<int> color(N+1, -1);
    for(int s=1;s<=N;s++){
        if(color[s] != -1) continue;
        // BFS
        queue<int> q;
        color[s]=0;
        q.push(s);
        while(!q.empty()){
            int u=q.front(); q.pop();
            for(int v: adj[u]){
                if(color[v]==-1){
                    color[v]=color[u]^1;
                    q.push(v);
                } else {
                    if(color[v]==color[u]){
                        cout << "no\n";
                        return 0;
                    }
                }
            }
        }
    }
    vector<int> day1;
    for(int i=1;i<=N;i++){
        if(color[i]==0) day1.push_back(i);
    }
    cout << "yes\n";
    cout << day1.size() << "\n";
    for(size_t i=0;i<day1.size();i++){
        if(i) cout << " ";
        cout << day1[i];
    }
    if(day1.size()) cout << "\n";
    return 0;
}
